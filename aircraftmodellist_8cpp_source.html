350  CAircraftModelList CAircraftModelList::findClosestColorDistance(const CRgbColor &fuselage, const CRgbColor &tail) const
351  {
352  double distance = 2.0;
353  CAircraftModelList models;
354  for (const CAircraftModel &m : (*this))
355  {
356  const CLivery &l = m.getLivery();
357  if (!l.hasColorTail() || !l.hasColorFuselage()) { continue; }
358  const double d = l.getColorDistance(fuselage, tail);
359  if (qFuzzyCompare(d, distance))
360  {
361  models.push_back(m);
362  }
363  else if (distance > d)
364  {
365  models.clear();
366  models.push_back(m);
367  distance = d;
368  }
369  }
370  return models;
371  }
