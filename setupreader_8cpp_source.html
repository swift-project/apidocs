73  bool CSetupReader::parseCmdLineArguments()
74  {
75  // copy vars at beginning to simplify a threadsafe version in the future
76  const QString cmdLineBootstrapUrl = this->getCmdLineBootstrapUrl();
77  BootstrapMode bootstrapMode = stringToEnum(sApp->getParserValue(m_cmdBootstrapMode));
78  const bool ignoreCmdBootstrapUrl = m_ignoreCmdBootstrapUrl;
79  const bool checkCmdBootstrapUrl = m_checkCmdBootstrapUrl;
80  const QString bootstrapUrlFileValue = CGlobalSetup::buildBootstrapFileUrl(cmdLineBootstrapUrl);
81  QString localSetupFileValue;
82  const QUrl url(bootstrapUrlFileValue);
83  const QString urlString(url.toString());
84  bool ok = false;
85 
86  if (urlString.isEmpty() && bootstrapMode == Explicit)
87  {
88  bootstrapMode = Implicit; // no URL, we use implicit mode
89  }
90 
91  do
92  {
93  // check on local file
94  if (url.isLocalFile())
95  {
96  localSetupFileValue = url.toLocalFile();
97  const QFile f(localSetupFileValue);
98  if (!f.exists())
99  {
100  sApp->cmdLineErrorMessage(QStringLiteral("File '%1' does not exist)").arg(localSetupFileValue));
101  break;
102  }
103  }
104 
105  // check on explicit URL
106  if (bootstrapMode == Explicit)
107  {
108  if (!url.isLocalFile())
109  {
110  bool retry = false;
111 
112  // "retry" possible in some cases
113  do
114  {
115  if (ignoreCmdBootstrapUrl || !checkCmdBootstrapUrl || CNetworkUtils::canConnect(url, CNetworkUtils::getLongTimeoutMs())) // cppcheck-suppress knownConditionTrueFalse
116  {
117  ok = true;
118  break;
119  }
120  retry = sApp->cmdLineErrorMessage(QStringLiteral("URL '%1' not reachable").arg(urlString), "", true);
121  }
122  while (retry);
123  }
124  }
125  }
126  while (false);
127 
128  m_localSetupFileValue = localSetupFileValue;
129  m_bootstrapUrlFileValue = bootstrapUrlFileValue;
130  m_bootstrapMode = bootstrapMode;
131  return ok;
132  }

158  CStatusMessageList CSetupReader::readLocalBootstrapFile(const QString &fileName)
159  {
160  if (fileName.isEmpty()) { return CStatusMessage(this).error(u"No file name for local bootstrap file"); }
161  if (!sApp || sApp->isShuttingDown()) { return CStatusMessage(this).error(u"No sApp, shutting down?"); }
162  QString fn;
163  const QFile file(fileName);
164  if (!file.exists())
165  {
166  // relative name?
167  const QString dir(sApp->getCmdSwiftPrivateSharedDir());
168  if (dir.isEmpty()) { return CStatusMessage(this).error(u"Empty shared directory '%1' for bootstrap file") << dir; }
169 
170  // no version for local files, as those come with the current code
171  fn = CFileUtils::appendFilePaths(dir, "bootstrap/" + CSwiftDirectories::bootstrapFileName());
172  }
173  else
174  {
175  fn = fileName;
176  }
177 
178  const QString content(CFileUtils::readFileToString(fn));
179  if (content.isEmpty()) { return CStatusMessage(this).error(u"File '%1' not existing or empty") << fn; }
180 
181  try
182  {
183  CGlobalSetup s;
184  s.convertFromJson(content);
185  s.markAsLoadedFromFile(true);
186  const CStatusMessage setMsg = m_setup.set(s);
187  const CStatusMessage setInfo = CStatusMessage(this).info(u"Setup cache updated from local file '%1'") << fn;
188  return setMsg.isSuccess() ? setInfo : setMsg;
189  }
190  catch (const CJsonException &ex)
191  {
192  return CStatusMessage::fromJsonException(ex, this, QStringLiteral("Parsing local setup file '%1'").arg(fn));
193  }
194  }

236  bool CSetupReader::prefillCacheWithLocalResourceBootstrapFile()
237  {
238  if (m_shutdown) { return false; }
239  m_setup.synchronize(); // make sure it is loaded
240  const CGlobalSetup cachedSetup = m_setup.get();
241  const bool cacheAvailable = cachedSetup.wasLoaded();
242  if (cacheAvailable)
243  {
244  CLogMessage(this).info(u"Setup cache prefill (bootstrap already cached, no prefill needed");
245  return false;
246  }
247  const QString fn = CSwiftDirectories::bootstrapResourceFilePath();
248  const CStatusMessageList msgs = this->readLocalBootstrapFile(fn);
249  CLogMessage::preformatted(msgs);
250  return true;
251  }

302  CStatusMessageList CSetupReader::manageSetupAvailability(bool webRead, bool localRead)
303  {
304  Q_ASSERT_X(!(webRead && localRead), Q_FUNC_INFO, "Local and web read together seems to be wrong");
305  CStatusMessageList msgs;
306 
307  bool available = false;
308  if (webRead || localRead)
309  {
310  available = true;
311  }
312  else
313  {
314  const bool cacheAvailable = m_setup.get().wasLoaded(); // loaded from web or file
315  available = cacheAvailable && m_bootstrapMode != Explicit;
316  }
317 
318  if (available && !webRead && !localRead)
319  {
320  msgs.push_back(CStatusMessage(this, CStatusMessage::SeverityInfo, u"Setup available, but not updated this time"));
321  }
322  else if (!available)
323  {
324  msgs.push_back(CStatusMessage(this, CStatusMessage::SeverityError, u"Setup not available"));
325  if (m_bootstrapMode == Explicit)
326  {
327  msgs.push_back(CStatusMessage(this, CStatusMessage::SeverityError, u"Mode is 'explicit', likely URL '" % m_bootstrapUrlFileValue % u"' is not reachable"));
328  }
329  }
330  m_setupAvailable = available;
331  emit this->setupHandlingCompleted(available);
332  return msgs;
333  }
