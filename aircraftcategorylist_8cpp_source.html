171  CAircraftCategoryList CAircraftCategoryList::findInParallelBranch(const CAircraftCategory &category) const
172  {
173  if (category.isNull() || this->isEmpty()) { return {}; }
174  if (category.isFirstLevel()) { return {}; }
175  const bool isL2 = category.getDepth() == 2;
176  const QList<int> loopLevels = isL2 ? this->getFirstLevels() : this->getSecondLevels();
177  if (loopLevels.isEmpty()) { return {}; }
178 
179  QList<int> level = category.getLevel();
180  CAircraftCategoryList result;
181  for (int l = loopLevels.front(); loopLevels.back() >= l; ++l)
182  {
183  level[isL2 ? 0 : 1] = l;
184  if (category.isLevel(level)) { continue; } // ignore category itself
185  const CAircraftCategory cat = this->findByFullLevel(level);
186  if (!cat.isNull()) { result.push_back(cat); }
187  }
188  return result;
189  }
