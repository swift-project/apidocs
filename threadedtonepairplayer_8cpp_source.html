105  QByteArray CThreadedTonePairPlayer::getAudioByTonePairs(const QList<CTonePair> &tonePairs)
106  {
107  Q_ASSERT(!tonePairs.isEmpty());
108  QByteArray finalBufferData;
109 
110  for (const auto &tonePair : std::as_const(tonePairs))
111  {
112  if (m_tonePairCache.contains(tonePair))
113  {
114  QByteArray bufferData;
115  bufferData = m_tonePairCache.value(tonePair);
116  finalBufferData.append(bufferData);
117  }
118  else
119  {
120  QByteArray bufferData;
121  bufferData = generateAudioFromTonePairs(tonePair);
122  m_tonePairCache.insert(tonePair, bufferData);
123  finalBufferData.append(bufferData);
124  }
125  }

129  QByteArray CThreadedTonePairPlayer::generateAudioFromTonePairs(const CTonePair &tonePair)
130  {
131  const int bytesPerSample = m_audioFormat.sampleSize() / 8;
132  const int bytesForAllChannels = m_audioFormat.channelCount() * bytesPerSample;
133 
134  QByteArray bufferData;
135  qint64 bytesPerTonePair = m_audioFormat.sampleRate() * bytesForAllChannels * tonePair.getDurationMs() / 1000;
136  bufferData.resize(static_cast<int>(bytesPerTonePair));
137  unsigned char *bufferPointer = reinterpret_cast<unsigned char *>(bufferData.data());
138 
139  qint64 last0AmplitudeSample = bytesPerTonePair; // last sample when amplitude was 0
140  int sampleIndexPerTonePair = 0;
141  while (bytesPerTonePair)
142  {
143  // http://hyperphysics.phy-astr.gsu.edu/hbase/audio/sumdif.html
144  // http://math.stackexchange.com/questions/164369/how-do-you-calculate-the-frequency-perceived-by-humans-of-two-sinusoidal-waves-a
145  const double pseudoTime = static_cast<double>(sampleIndexPerTonePair % this->m_audioFormat.sampleRate()) / this->m_audioFormat.sampleRate();
146  double amplitude = 0.0; // amplitude -1 -> +1 , 0 is silence
147  if (tonePair.getFirstFrequencyHz() > 10)
148  {
149  // the combination of two frequencies actually would have 2*amplitude,
150  // but I have to normalize with amplitude -1 -> +1
151  amplitude = tonePair.getSecondFrequencyHz() == 0 ?
152  qSin(2 * M_PI * tonePair.getFirstFrequencyHz() * pseudoTime) :
153  qSin(M_PI * (tonePair.getFirstFrequencyHz() + tonePair.getSecondFrequencyHz()) * pseudoTime) *
154  qCos(M_PI * (tonePair.getFirstFrequencyHz() - tonePair.getSecondFrequencyHz()) * pseudoTime);
155  }
156 
157  // avoid overflow
158  Q_ASSERT(amplitude <= 1.0 && amplitude >= -1.0);
159  if (amplitude < -1.0) { amplitude = -1.0; }
160  else if (amplitude > 1.0) { amplitude = 1.0; }
161  else if (qAbs(amplitude) < 1.0 / 65535)
162  {
163  amplitude = 0;
164  last0AmplitudeSample = bytesPerTonePair;
165  }
166 
167  // generate this for all channels, usually 1 channel
168  for (int i = 0; i < this->m_audioFormat.channelCount(); ++i)
169  {
170  this->writeAmplitudeToBuffer(amplitude, bufferPointer);
171  bufferPointer += bytesPerSample;
172  bytesPerTonePair -= bytesPerSample;
173  }
174  ++sampleIndexPerTonePair;
175  }
176 
177  // fixes the range from the last 0 pass through
178  if (last0AmplitudeSample > 0)
179  {
180  bufferPointer -= last0AmplitudeSample;
181  while (last0AmplitudeSample)
182  {
183  const double amplitude = 0.0; // amplitude -1 -> +1 , 0 is silence
184 
185  // generate this for all channels, usually 1 channel
186  for (int i = 0; i < this->m_audioFormat.channelCount(); ++i)
187  {
188  this->writeAmplitudeToBuffer(amplitude, bufferPointer);
189  bufferPointer += bytesPerSample;
190  last0AmplitudeSample -= bytesPerSample;
191  }
192  }
193  }
194  return bufferData;
195  }

197  void CThreadedTonePairPlayer::writeAmplitudeToBuffer(double amplitude, unsigned char *bufferPointer)
198  {
199  Q_ASSERT(this->m_audioFormat.sampleSize() == 16);

203  const qint16 value = static_cast<qint16>(amplitude * 32767);
