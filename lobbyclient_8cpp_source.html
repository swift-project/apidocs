86  bool CLobbyClient::canLobbyConnect()
87  {
88  if (!m_dpLobbyClient) { return false; }
89  GUID appGuid = CFs9Sdk::guid();
90  DWORD dwSize = 0;
91  DWORD dwItems = 0;
92  const HRESULT hr = m_dpLobbyClient->EnumLocalPrograms(&appGuid, nullptr, &dwSize, &dwItems, 0);
93  if (hr == DPNERR_BUFFERTOOSMALL)
94  {
95  QScopedArrayPointer<BYTE> memPtr(new BYTE[dwSize]);
96  DPL_APPLICATION_INFO *appInfo = reinterpret_cast<DPL_APPLICATION_INFO *>(memPtr.data());
97 
98  m_dpLobbyClient->EnumLocalPrograms(&appGuid, memPtr.data(), &dwSize, &dwItems, 0);
99 
100  if (dwItems > 0)
101  {
102  CLogMessage(this).debug() << "Found lobby application:" << QString::fromWCharArray(appInfo->pwszApplicationName);
103  return true;
104  }
105  else
106  {
107  return false;
108  }
109  }
110  else
111  {
112  return false;
113  }

144  freeConnectSettings(dnConnectInfo.pdplConnectionSettings);

148  HRESULT CLobbyClient::allocAndInitConnectSettings(const QString &address, GUID *pAppGuid, DPL_CONNECTION_SETTINGS **ppdplConnectSettings)
149  {
150  HRESULT hr;
151 
152  IDirectPlay8Address *pHostAddress = nullptr;
153  IDirectPlay8Address *pDeviceAddress = nullptr;
154 
155  QScopedPointer<GUID> pSPGuid(new GUID);
156  memcpy(pSPGuid.data(), &CLSID_DP8SP_TCPIP, sizeof(GUID));
157 
158  // Create a host address
159  if (FAILED(hr = CoCreateInstance(CLSID_DirectPlay8Address, nullptr, CLSCTX_INPROC_SERVER,
160  IID_IDirectPlay8Address, reinterpret_cast<void **>(&pHostAddress))))
161  {
162  return logDirectPlayError(hr);
163  }
164 
165  // Set the SP to pHostAddress
166  if (FAILED(hr = pHostAddress->SetSP(pSPGuid.data())))
167  {
168  return logDirectPlayError(hr);
169  }
170 
171  // Create a device address to specify which device we are using
172  if (FAILED(hr = CoCreateInstance(CLSID_DirectPlay8Address, NULL, CLSCTX_INPROC_SERVER,
173  IID_IDirectPlay8Address, reinterpret_cast<void **>(&pDeviceAddress))))
174  {
175  return logDirectPlayError(hr);
176  }
177 
178  // Set the SP to pDeviceAddress
179  if (FAILED(hr = pDeviceAddress->SetSP(&CLSID_DP8SP_TCPIP)))
180  {
181  return logDirectPlayError(hr);
182  }
183 
184  if (FAILED(hr = pHostAddress->BuildFromURLA(address.toLocal8Bit().data())))
185  {
186  return logDirectPlayError(hr);
187  }
188 
189  // Setup the DPL_CONNECTION_SETTINGS
190  DPL_CONNECTION_SETTINGS *pSettings = new DPL_CONNECTION_SETTINGS;
191 
192  // Allocate space for device address pointers
193  // We cannot use QScopedArrayPointer, because memory needs to be valid
194  // for the lifetime of DPL_CONNECTION_SETTINGS.
195  IDirectPlay8Address **apDevAddress = new IDirectPlay8Address *[1];
196 
197  // Set the device addresses
198  apDevAddress[0] = pDeviceAddress;
199 
200  QString session = sApp->swiftVersionString();
201  QScopedArrayPointer<wchar_t> wstrSessionName(new wchar_t[session.size() + 1]);
202  session.toWCharArray(wstrSessionName.data());
203  wstrSessionName[session.size()] = 0;
204 
205  // Fill in the connection settings
206  ZeroMemory(pSettings, sizeof(DPL_CONNECTION_SETTINGS));
207  pSettings->dwSize = sizeof(DPL_CONNECTION_SETTINGS);
208  pSettings->dpnAppDesc.dwSize = sizeof(DPN_APPLICATION_DESC);
209  pSettings->dwFlags = 0;
210  pSettings->dpnAppDesc.guidApplication = *pAppGuid;
211  pSettings->dpnAppDesc.guidInstance = GUID_NULL;
212  pSettings->dpnAppDesc.dwFlags = DPNSESSION_NODPNSVR;
213  pSettings->pdp8HostAddress = pHostAddress;
214  pSettings->ppdp8DeviceAddresses = apDevAddress;
215  pSettings->cNumDeviceAddresses = 1;
216  pSettings->dpnAppDesc.pwszSessionName = new WCHAR[wcslen(wstrSessionName.data()) + 1];
217  wcscpy(pSettings->dpnAppDesc.pwszSessionName, wstrSessionName.data());
218 
219  // FIXME: Use players callsign
220  QString playerName("Player");
221  WCHAR wstrPlayerName[m_maxSizePlayerName];
222  playerName.toWCharArray(wstrPlayerName);
223  wstrPlayerName[playerName.size()] = 0;
224  pSettings->pwszPlayerName = new WCHAR[wcslen(wstrPlayerName) + 1];
225  wcscpy(pSettings->pwszPlayerName, wstrPlayerName);
226 
227  *ppdplConnectSettings = pSettings;
228 
229  return S_OK;
230  }

232  void CLobbyClient::freeConnectSettings(DPL_CONNECTION_SETTINGS *pSettings)
233  {
234  if (!pSettings) return;
235 
236  SafeDeleteArray(pSettings->pwszPlayerName);
237  SafeDeleteArray(pSettings->dpnAppDesc.pwszSessionName);
238  SafeDeleteArray(pSettings->dpnAppDesc.pwszPassword);
239 
240  // The following two lines came from DX9 SDK samples, but they don't make sense.
241  // Deleteing a void* pointer is considered unsafe. If you really had to pass
242  // anything non-void here, make sure the original object is cleaned up properly.
243  // SafeDeleteArray(pSettings->dpnAppDesc.pvReservedData);
244  // SafeDeleteArray(pSettings->dpnAppDesc.pvApplicationReservedData);
245  SafeRelease(pSettings->pdp8HostAddress);
246  SafeRelease(pSettings->ppdp8DeviceAddresses[0]);
247  SafeDeleteArray(pSettings->ppdp8DeviceAddresses);
248  SafeDelete(pSettings);

256  HRESULT CLobbyClient::directPlayLobbyMessageHandler(DWORD messageId, void *msgBuffer)
257  {
258  switch (messageId)
259  {
260  case DPL_MSGID_DISCONNECT:
261  {
262  PDPL_MESSAGE_DISCONNECT pDisconnectMsg;
263  pDisconnectMsg = (PDPL_MESSAGE_DISCONNECT)msgBuffer;
264  Q_UNUSED(pDisconnectMsg)
265 
266  emit disconnected();
267 
268  // We should free any data associated with the
269  // app here, but there is none.
270  break;
271  }
272 
273  case DPL_MSGID_RECEIVE:
274  {
275  PDPL_MESSAGE_RECEIVE pReceiveMsg;
276  pReceiveMsg = (PDPL_MESSAGE_RECEIVE)msgBuffer;
277  Q_UNUSED(pReceiveMsg)
278 
279  // The lobby app sent us data. This sample doesn't
280  // expected data from the app, but it is useful
281  // for more complex clients.
282  break;
283  }
284 
285  case DPL_MSGID_SESSION_STATUS:
286  {
287  PDPL_MESSAGE_SESSION_STATUS pStatusMsg;
288  pStatusMsg = (PDPL_MESSAGE_SESSION_STATUS)msgBuffer;
289 
290  QString message;
291  message.append(QString("%1: ").arg(pStatusMsg->hSender, 0, 16));
292  switch (pStatusMsg->dwStatus)
293  {
294  case DPLSESSION_CONNECTED:
295  message.append("Session connected");
296  break;
297  case DPLSESSION_COULDNOTCONNECT:
298  message.append("Session could not connect");
299  break;
300  case DPLSESSION_DISCONNECTED:
301  message.append("Session disconnected");
302  break;
303  case DPLSESSION_TERMINATED:
304  message.append("Session terminated");
305  break;
306  case DPLSESSION_HOSTMIGRATED:
307  message.append("Host migrated");
308  break;
309  case DPLSESSION_HOSTMIGRATEDHERE:
310  message.append("Host migrated to this client");
311  break;
312  default:
313  message.append(QStringLiteral("Unknown PDPL_MESSAGE_SESSION_STATUS: %1").arg(pStatusMsg->dwStatus));
314  break;
315  }
316 
317  CLogMessage(this).info(message);
318  break;
319  }
320 
321  case DPL_MSGID_CONNECTION_SETTINGS:
322  {
323  PDPL_MESSAGE_CONNECTION_SETTINGS pConnectionStatusMsg;
324  pConnectionStatusMsg = (PDPL_MESSAGE_CONNECTION_SETTINGS)msgBuffer;
325  Q_UNUSED(pConnectionStatusMsg)
326 
327  // The app has changed the connection settings.
328  // This simple client doesn't handle this, but more complex clients may
329  // want to.
330  break;
331  }
332  }
333  return S_OK;
334  }
