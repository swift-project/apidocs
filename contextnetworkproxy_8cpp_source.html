41  void CContextNetworkProxy::relaySignals(const QString &serviceName, QDBusConnection &connection)
42  {
43  bool s = connection.connect(serviceName, IContextNetwork::ObjectPath(), IContextNetwork::InterfaceName(),
44  "connectionStatusChanged", this, SIGNAL(connectionStatusChanged(BlackMisc::Network::CConnectionStatus, BlackMisc::Network::CConnectionStatus)));
45  Q_ASSERT(s);
46  s = connection.connect(serviceName, IContextNetwork::ObjectPath(), IContextNetwork::InterfaceName(),
47  "changedAtcStationsBooked", this, SIGNAL(changedAtcStationsBooked()));
48  Q_ASSERT(s);
49  s = connection.connect(serviceName, IContextNetwork::ObjectPath(), IContextNetwork::InterfaceName(),
50  "changedAtcStationsOnline", this, SIGNAL(changedAtcStationsOnline()));
51  Q_ASSERT(s);
52  s = connection.connect(serviceName, IContextNetwork::ObjectPath(), IContextNetwork::InterfaceName(),
53  "changedAtcStationsBookedDigest", this, SIGNAL(changedAtcStationsBookedDigest()));
54  Q_ASSERT(s);
55  s = connection.connect(serviceName, IContextNetwork::ObjectPath(), IContextNetwork::InterfaceName(),
56  "changedAtcStationsOnlineDigest", this, SIGNAL(changedAtcStationsOnlineDigest()));
57  Q_ASSERT(s);
58  s = connection.connect(serviceName, IContextNetwork::ObjectPath(), IContextNetwork::InterfaceName(),
59  "changedAircraftInRange", this, SIGNAL(changedAircraftInRange()));
60  Q_ASSERT(s);
61  s = connection.connect(serviceName, IContextNetwork::ObjectPath(), IContextNetwork::InterfaceName(),
62  "changedAtcStationOnlineConnectionStatus", this, SIGNAL(changedAtcStationOnlineConnectionStatus(BlackMisc::Aviation::CAtcStation, bool)));
63  Q_ASSERT(s);
64  s = connection.connect(serviceName, IContextNetwork::ObjectPath(), IContextNetwork::InterfaceName(),
65  "changedAircraftInRangeDigest", this, SIGNAL(changedAircraftInRangeDigest()));
66  Q_ASSERT(s);
67  s = connection.connect(serviceName, IContextNetwork::ObjectPath(), IContextNetwork::InterfaceName(),
68  "kicked", this, SIGNAL(kicked(QString)));
69  Q_ASSERT(s);
70  s = connection.connect(serviceName, IContextNetwork::ObjectPath(), IContextNetwork::InterfaceName(),
71  "textMessagesReceived", this, SIGNAL(textMessagesReceived(BlackMisc::Network::CTextMessageList)));
72  Q_ASSERT(s);
73  s = connection.connect(serviceName, IContextNetwork::ObjectPath(), IContextNetwork::InterfaceName(),
74  "textMessageSent", this, SIGNAL(textMessageSent(BlackMisc::Network::CTextMessage)));
75  Q_ASSERT(s);
76  s = connection.connect(serviceName, IContextNetwork::ObjectPath(), IContextNetwork::InterfaceName(),
77  "changedRemoteAircraftEnabled", this, SIGNAL(changedRemoteAircraftEnabled(BlackMisc::Simulation::CSimulatedAircraft)));
78  Q_ASSERT(s);
79  s = connection.connect(serviceName, IContextNetwork::ObjectPath(), IContextNetwork::InterfaceName(),
80  "changedFastPositionUpdates", this, SIGNAL(changedFastPositionUpdates(BlackMisc::Simulation::CSimulatedAircraft)));
81  Q_ASSERT(s);
82  s = connection.connect(serviceName, IContextNetwork::ObjectPath(), IContextNetwork::InterfaceName(),
83  "changedGndFlagCapability", this, SIGNAL(changedGndFlagCapability(BlackMisc::Simulation::CSimulatedAircraft)));
84  Q_ASSERT(s);
85  s = connection.connect(serviceName, IContextNetwork::ObjectPath(), IContextNetwork::InterfaceName(),
86  "addedAircraft", this, SIGNAL(addedAircraft(BlackMisc::Simulation::CSimulatedAircraft)));
87  Q_ASSERT(s);
88  s = connection.connect(serviceName, IContextNetwork::ObjectPath(), IContextNetwork::InterfaceName(),
89  "removedAircraft", this, SIGNAL(removedAircraft(BlackMisc::Aviation::CCallsign)));
90  Q_ASSERT(s);
91  s = connection.connect(serviceName, IContextNetwork::ObjectPath(), IContextNetwork::InterfaceName(),
92  "connectedServerChanged", this, SIGNAL(connectedServerChanged(BlackMisc::Network::CServer)));
93  Q_ASSERT(s);
94  Q_UNUSED(s);
95  this->relayBaseClassSignals(serviceName, connection, IContextNetwork::ObjectPath(), IContextNetwork::InterfaceName());
96  }
