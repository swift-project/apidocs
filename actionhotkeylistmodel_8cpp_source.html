31  QVariant CActionHotkeyListModel::data(const QModelIndex &index, int role) const
32  {
33  if (!index.isValid()) { return QVariant(); }
34  if (index.row() >= m_actionHotkeys.size() || index.row() < 0) { return QVariant(); }
35 
36  if (role == Qt::DisplayRole)
37  {
38  const int col = index.column();
39  if (col == 0)
40  {
41  const CIdentifier identifier = m_actionHotkeys[index.row()].getApplicableMachine();
42  return identifier.getMachineName();
43  }
44  if (col == 1)
45  {
46  const CHotkeyCombination combination = m_actionHotkeys[index.row()].getCombination();
47  return combination.asStringWithDeviceNames();
48  // return combination.toQString();
49  }
50  if (col == 2)
51  {
52  return m_actionHotkeys[index.row()].getAction();
53  }
54  }
55  else if (role == ActionHotkeyRole)
56  {
57  const auto hotkey = m_actionHotkeys[index.row()];
58  return QVariant::fromValue(hotkey);
59  }
60  return {};
61  }
