80  void CMainKeypadAreaComponent::onMainInfoAreaChanged(int currentTabIndex, const QList<int> &dockedIndexes, const QList<int> &floatingIndexes)
81  {
82  this->unsetInfoAreaButtons();

97  if (pb) { pb->setChecked(true); }
98  }
99 
100  this->updateConnectionStatus();
101 
102  Q_UNUSED(dockedIndexes)
103  }

110  void CMainKeypadAreaComponent::buttonSelected()
111  {
112  if (!sGui || sGui->isShuttingDown()) { return; }
113  QPushButton *senderButton = static_cast<QPushButton *>(QObject::sender());
114  Q_ASSERT_X(senderButton, Q_FUNC_INFO, "No sender button");
115  if (!senderButton) { return; }
116  const CMainInfoAreaComponent::InfoArea infoArea = buttonToMainInfoArea(senderButton);
117  if (infoArea != CMainInfoAreaComponent::InfoAreaNone)
118  {
119  Q_ASSERT(senderButton->isCheckable());
120  emit this->selectedMainInfoAreaDockWidget(infoArea);
121  senderButton->setChecked(true); // re-check if got unchecked, we use checked buttons like normal buttons
122  return;
123  }
124  else if (senderButton == ui->pb_CockpitIdent && sGui->getIContextOwnAircraft())
125  {
126  emit this->identPressed();
127  }
128  else if (senderButton == ui->pb_Opacity050)
129  {
130  emit this->changedOpacity(50);
131  }
132  else if (senderButton == ui->pb_Opacity100)
133  {
134  emit this->changedOpacity(100);
135  }
136  else if (senderButton == ui->pb_SoundMaxVolume && sGui->getIContextAudio())
137  {
138  sGui->getCContextAudioBase()->setMasterOutputVolume(100);
139  }
140  else if (senderButton == ui->pb_SoundMute && sGui->getIContextAudio())
141  {
142  const bool mute = sGui->getCContextAudioBase()->isMuted();
143  sGui->getCContextAudioBase()->setMute(!mute);
144  }
145  else if (senderButton == ui->pb_Connect)
146  {
147  emit this->connectPressed();
148  this->updateConnectionStatus();
149  }
150  else if (senderButton == ui->pb_Audio)
151  {
152  emit this->audioPressed();
153  }
154  }
