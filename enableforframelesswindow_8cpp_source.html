54  void CEnableForFramelessWindow::setFrameless(bool frameless)
55  {
56  const bool isFrameless = this->isFrameless();
57  if (isFrameless == frameless) { return; }
58 
59  QWidget *w = this->getWidget();
60  if (!w) { return; }
61  const QRect oldFrameGeometry = w->frameGeometry();
62  const QRect oldGeometry = w->geometry();
63 
64  WindowMode nonFrameLessMode = m_originalWindowMode; // Tool/Normal Window
65  if (nonFrameLessMode == WindowFrameless) { nonFrameLessMode = WindowNormal; }
66  this->setMode(frameless ? WindowFrameless : nonFrameLessMode);
67 
68  if (frameless)
69  {
70  // from framed to frameless
71  w->setGeometry(oldFrameGeometry);
72  m_windowFrameSizeW = oldFrameGeometry.width() - oldGeometry.width();
73  m_windowFrameSizeH = oldFrameGeometry.height() - oldGeometry.height();
74  }
75  else
76  {
77  if (m_windowFrameSizeW >= 0 && m_windowFrameSizeH >= 0)
78  {
79  QRect newGeometry = oldGeometry;
80  // newGeometry.setWidth(oldGeometry.width() - m_windowFrameSizeW);
81  // newGeometry.setHeight(oldGeometry.height() - m_windowFrameSizeH);
82  newGeometry.setX(oldGeometry.x() + m_windowFrameSizeW);
83  newGeometry.setY(oldGeometry.y() + m_windowFrameSizeH);
84  w->setGeometry(newGeometry);
85  }
86  }
87  }

112  const QString ws(s.trimmed().toLower());
113  if (ws.isEmpty()) { return WindowNormal; }

141  {
142  Q_ASSERT_X(m_widget, "CEnableForFramelessWindow::setWindowAttributes", "Missing widget representing window");
143  Q_ASSERT_X(!m_framelessPropertyName.isEmpty(), "CEnableForFramelessWindow::setWindowAttributes", "Missing property name");
144 
145  const bool frameless = (mode == WindowFrameless);
146 
147  // http://stackoverflow.com/questions/18316710/frameless-and-transparent-window-qt5
148  // https://bugreports.qt.io/browse/QTBUG-52206
149  // UpdateLayeredWindowIndirect failed for ptDst
150  if (m_isMainApplicationWindow && CGuiUtility::isTopLevelWindow(m_widget))
151  {
152  m_widget->setAttribute(Qt::WA_NativeWindow);
153 
154  // causing a BLACK background
155  m_widget->setAttribute(Qt::WA_NoSystemBackground, frameless);
156  m_widget->setAttribute(Qt::WA_TranslucentBackground, frameless); // causing QTBUG-52206
157  }
158 
159  // Qt::WA_PaintOnScreen leads to a warning
160  // setMask(QRegion(10, 10, 10, 10) would work, but requires "complex" calcs for rounded corners
162  this->setDynamicProperties(frameless);

206  bool CEnableForFramelessWindow::handleMouseMoveEvent(QMouseEvent *event)
207  {
208  Q_ASSERT(m_widget);
209  if (m_windowMode == WindowFrameless && event->buttons() == Qt::LeftButton && !m_framelessDragPosition.isNull())
210  {
211  const QSize s = m_widget->size();
212  const bool changedSize = (m_moveSize != s);
213 
214  // avoid "jumping around" if window was resized
215  // resizing in frameless is subject of QSizeGrip in the status bar
216  if (changedSize)
217  {
218  m_moveSize = s;
219  return false;
220  }
221 
222  m_widget->move(event->globalPos() - m_framelessDragPosition);
223  event->accept();
224  return true;
225  }
226  return false;
227  }

229  bool CEnableForFramelessWindow::handleChangeEvent(QEvent *event)
230  {
231  if (event->type() != QEvent::WindowStateChange) { return false; }
232  if (m_windowMode != WindowTool) { return false; }
233  if (!m_widget) { return false; }
234 
235  // make sure a tool window is changed to Normal window so it is show in taskbar
236  // here we are already in transition state, so isMinimized means will be minimize right now
237  // this check here is needed if minimized is called from somewhere else than ps_showMinimized
238 
239  QPointer<QWidget> widgetSelf(m_widget); // almost as good as myself
240  if (m_widget->isMinimized())
241  {
242  // still tool, force normal window
243  // decouple, otherwise we end up in infinite loop as it triggers a new changeEvent
244 
245  QTimer::singleShot(0, m_widget, [=] {
246  if (!widgetSelf) { return; }
247  this->showMinimizedModeChecked();
248  });
249  }
250  else
251  {
252  // not tool, force tool window
253  // decouple, otherwise we end up in infinite loop as it triggers a new changeEvent
254  QTimer::singleShot(0, m_widget, [=] {
255  if (!widgetSelf) { return; }
256  this->showNormalModeChecked();
257  });
258  }
259  event->accept();
260  return true;
261  }

326  {

331  case WindowTool:
