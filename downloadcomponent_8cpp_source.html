95  {

103  QTimer::singleShot(delayMs, this, [=] {

105  this->triggerDownloadingOfFiles();

171  bool CDownloadComponent::triggerDownloadingOfFile(const CRemoteFile &remoteFile)
172  {
173  if (!sGui || !sGui->hasWebDataServices() || sGui->isShuttingDown()) { return false; }
174  if (!this->existsDownloadDir())
175  {
176  const CStatusMessage msg = CStatusMessage(this, CLogCategories::validation()).error(u"Invalid download directory");
177  this->showOverlayMessage(msg, CDownloadComponent::OverlayMsgTimeoutMs);
178  return false;
179  }
180 
181  const CUrl download = remoteFile.getSmartUrl();
182  if (download.isEmpty())
183  {
184  const CStatusMessage msg = CStatusMessage(this, CLogCategories::validation()).error(u"No download URL for file name '%1'") << remoteFile.getBaseNameAndSize();
185  this->showOverlayMessage(msg, CDownloadComponent::OverlayMsgTimeoutMs);
186  return false;
187  }
188 
189  this->showStartedFileMessage(remoteFile);
190  m_fileInProgress = remoteFile;
191  const QString saveAsFile = CFileUtils::appendFilePaths(ui->le_DownloadDir->text(), remoteFile.getBaseName());
192  const QFileInfo fiSaveAs(saveAsFile);
193  if (fiSaveAs.exists())
194  {
195  const QString msg = QStringLiteral("File '%1' already exists locally.\n\nDo you want to reload the file?").arg(fiSaveAs.absoluteFilePath());
196  QMessageBox::StandardButton reply = QMessageBox::question(this, "File exists", msg, QMessageBox::Yes | QMessageBox::No);
197  if (reply != QMessageBox::Yes)
198  {
199  const QPointer<CDownloadComponent> myself(this);
200  QTimer::singleShot(10, this, [=] {
201  if (!myself || !sGui || sGui->isShuttingDown()) { return; }
202  this->downloadedFile(CStatusMessage(this).info(u"File was already downloaded"));
203  });
204  return true;
205  }
206  }
207 
208  QNetworkReply *reply = sGui->downloadFromNetwork(download, saveAsFile, { this, &CDownloadComponent::downloadedFile });
209  bool success = false;
210  if (reply)
211  {
212  // this->showLoading(10 * 1000);
213  CLogMessage(this).info(u"Triggered downloading of file from '%1'") << download.getHost();
214  connect(reply, &QNetworkReply::downloadProgress, this, &CDownloadComponent::downloadProgress, Qt::QueuedConnection);
215  m_reply = reply;
216  success = true;
217  }
218  else
219  {
220  const CStatusMessage msg = CStatusMessage(this, CLogCategories::validation()).error(u"Starting download for '%1' failed") << download.getFullUrl();
221  this->showOverlayMessage(msg, CDownloadComponent::OverlayMsgTimeoutMs);
222  }
223  return success;
224  }

226  void CDownloadComponent::downloadedFile(const CStatusMessage &status)
227  {
228  // reset in progress
229  const CRemoteFile justDownloaded(m_fileInProgress);
230  m_fileInProgress = CRemoteFile();
231  m_reply = nullptr;
232  this->showCompletedFileMessage(justDownloaded);
233  this->hideLoading();
234 
235  if (sGui && sGui->isShuttingDown()) { return; }
236  if (status.isWarningOrAbove())
237  {
238  this->showOverlayMessage(status, CDownloadComponent::OverlayMsgTimeoutMs);
239  this->clear();
240  return;
241  }
242 
243  const bool t = this->triggerDownloadingOfNextFile();
244  if (!t) { this->lastFileDownloaded(); }
245  }

259  void CDownloadComponent::startDownloadedExecutable()
260  {
261  if (!ui->cb_StartAfterDownload->isChecked()) { return; }
262  if (!this->haveAllDownloadsCompleted()) { return; }
263  const CRemoteFileList executables = m_remoteFiles.findExecutableFiles();
264  if (executables.isEmpty()) { return; }
265 
266  // try to start
267  const QDir dir(ui->le_DownloadDir->text());
268  if (!dir.exists()) { return; }
269 
270  QString msg;
271  if (CBuildConfig::isRunningOnMacOSPlatform())
272  {
273  msg = "To install close swift, "
274  "mount the disk image '%1' and run the installer inside "
275  "to proceed with the update.";
276  }
277  else
278  {
279  msg = ui->cb_Shutdown->isChecked() ?
280  QString("Start '%1' and close swift?") :
281  QString("Start '%1'?");
282  }
283 
284  for (const CRemoteFile &rf : executables)
285  {
286  const QString executable = CFileUtils::appendFilePaths(dir.absolutePath(), rf.getBaseName());
287  QFile executableFile(executable);
288  if (!executableFile.exists()) { continue; }
289 
290  QMessageBox::StandardButton reply = QMessageBox::question(this, "Start?", msg.arg(rf.getName()), QMessageBox::Yes | QMessageBox::No);
291  if (reply != QMessageBox::Yes) { return; }
292 
293  const CPlatform p = CArtifact::artifactNameToPlatform(rf.getName());
294  if (!CPlatform::canRunOnCurrentPlatform(p))
295  {
296  // cannot run on this OS, just show the directory where the download resides
297  // do not close
298  ui->pb_OpenDownloadDir->click();
299  return;
300  }
301 
302  if (CBuildConfig::isRunningOnLinuxPlatform() && !executableFile.permissions().testFlag(QFile::ExeOwner))
303  {
304  executableFile.setPermissions(QFile::ReadOwner | QFile::WriteOwner | QFile::ExeOwner | QFile::ReadGroup | QFile::ExeGroup | QFile::ReadOther | QFile::ExeOther);
305  }
306 
307  const bool shutdown = ui->cb_Shutdown->isChecked();
308  const bool started = QProcess::startDetached(executable, {}, dir.absolutePath());
309  if (started && shutdown && sGui)
310  {
311  QTimer::singleShot(250, sGui, [] {
312  if (!sGui) { return; }
313  CGuiApplication::exit();
314  });
315  break;
316  }
317  } // files
318  }
