216  CMenuActions CViewBaseNonTemplate::initMenuActions(CViewBaseNonTemplate::MenuFlag menu)
217  {
218  if (m_menuFlagActions.contains(menu)) { return m_menuFlagActions.value(menu); }
219 
220  CMenuActions ma;
221  switch (menu)
222  {
223  case MenuRefresh:
224  {
225  static const QMetaMethod requestSignal = QMetaMethod::fromSignal(&CViewBaseNonTemplate::requestUpdate);
226  if (!this->isSignalConnected(requestSignal)) break;
227  ma.addAction(CIcons::refresh16(), "Update", CMenuAction::pathViewUpdates(), { this, &CViewBaseNonTemplate::triggerReload });
228  break;
229  }
230  case MenuBackend:
231  {
232  static const QMetaMethod requestSignal = QMetaMethod::fromSignal(&CViewBaseNonTemplate::requestNewBackendData);
233  if (!this->isSignalConnected(requestSignal)) break;
234  ma.addAction(CIcons::refresh16(), "Reload from backend", CMenuAction::pathViewUpdates(), { this, &CViewBaseNonTemplate::triggerReloadFromBackend });
235  break;
236  }
237  case MenuDisplayAutomatically:
238  {
239  QAction *a = ma.addAction(CIcons::appMappings16(), "Automatically display (when loaded)", CMenuAction::pathViewUpdates(), { this, &CViewBaseNonTemplate::toggleAutoDisplay });
240  a->setCheckable(true);
241  a->setChecked(this->displayAutomatically());
242  break;
243  }
244  case MenuRemoveSelectedRows:
245  {
246  ma.addAction(CIcons::delete16(), "Remove selected rows", CMenuAction::pathViewAddRemove(), { this, &CViewBaseNonTemplate::removeSelectedRowsChecked }, CShortcut::keyDelete());
247  break;
248  }
249  case MenuClear:
250  {
251  ma.addAction(CIcons::delete16(), "Clear", CMenuAction::pathViewAddRemove(), { this, &CViewBaseNonTemplate::clear });
252  break;
253  }
254  case MenuFilter:
255  {
256  if (m_filterWidget)
257  {
258  const bool dialog = qobject_cast<QDialog *>(m_filterWidget);
259  if (dialog) ma.addAction(CIcons::filter16(), "Show filter " + CShortcut::toParenthesisString(CShortcut::keyDisplayFilter()), CMenuAction::pathViewFilter(), { this, &CViewBaseNonTemplate::displayFilterDialog }, CShortcut::keyDisplayFilter());
260  ma.addAction(CIcons::filter16(), "Remove Filter", CMenuAction::pathViewFilter(), { this, &CViewBaseNonTemplate::removeFilter });
261  }
262  break;
263  }
264  case MenuMaterializeFilter:
265  {
266  ma.addAction(CIcons::tableRelationship16(), "Materialize filtered data", CMenuAction::pathViewFilter(), { this, &CViewBaseNonTemplate::materializeFilter });
267  break;
268  }
269  case MenuLoad:
270  {
271  ma.addAction(CIcons::disk16(), "Load from file ", CMenuAction::pathViewLoadSave(), { this, &CViewBaseNonTemplate::loadJsonAction });
272  break;
273  }
274  case MenuSave:
275  {
276  ma.addAction(CIcons::disk16(), "Save data in file " + CShortcut::toParenthesisString(CShortcut::keySaveViews()), CMenuAction::pathViewLoadSave(), { this, &CViewBaseNonTemplate::saveJsonAction }, CShortcut::keySaveViews());

279  ma.addAction(CIcons::disk16(), "Save selected data in file", CMenuAction::pathViewLoadSave(), { this, &CViewBaseNonTemplate::saveSelectedJsonAction });

289  }

294  break;

304  }
