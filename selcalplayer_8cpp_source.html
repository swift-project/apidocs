31  CTime CSelcalPlayer::play(int volume, const CSelcal &selcal)
32  {
33  CTime duration = CTime::null();
34  if (selcal.isValid())
35  {
36  const QList<CFrequency> frequencies = selcal.getFrequencies();
37  Q_ASSERT(frequencies.size() == 4);
38  const CTime oneSec(1000.0, CTimeUnit::ms());
39  const CTonePair t1(frequencies.at(0), frequencies.at(1), oneSec);
40  const CTonePair t2({}, {}, oneSec / 5.0);
41  const CTonePair t3(frequencies.at(2), frequencies.at(3), oneSec);
42  QList<CTonePair> tonePairs;
43  tonePairs << t1 << t2 << t3;
44  m_threadedPlayer->play(volume, tonePairs);
45  duration = oneSec * 2.5;
46  }
47  return duration;
