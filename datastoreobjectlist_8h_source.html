198  int replaceOrAddObjectsByKey(const CONTAINER &container)
199  {
200  if (container.isEmpty()) { return 0; }

202  {
203  this->container() = container;
204  return this->container().size();
205  }
206  CONTAINER newValues(this->container());
207  const QSet<KEYTYPE> keys(container.toDbKeySet());
208  newValues.removeObjectsWithKeys(keys);
209  int removeSize = newValues.size(); // size after removing data
210  newValues.push_back(container);
211  this->container() = newValues;
212  return newValues.size() - removeSize;
