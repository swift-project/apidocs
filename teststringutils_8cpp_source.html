70  {
71  const QStringList dts(
72  { "2018-01-01 11:11:11",
73  "2012-05-09 03:04:05.777",
74  "2012-05-09 00:00:00.000",
75  "2015-12-31 03:04:05",

77  "1975-01-01 14:13:17",
78  "1982-05-09 03:01:05.123",
79  "2000-05-02 00:04:00.000",
80  "2002-12-31 03:34:33",
81  "1992-11-01 21:59:29.999" });
82 
83  const int size = QString("yyyyMMddHHmmss").size();
84  for (const QString &dt : dts)
85  {
86  const QString c = removeDateTimeSeparators(dt);
87  const QDateTime dt1 = parseDateTimeStringOptimized(c);
88  const QDateTime dt2 = (c.length() == size) ?
89  fromStringUtc(c, "yyyyMMddHHmmss") :
90  fromStringUtc(c, "yyyyMMddHHmmsszzz");
91  QDateTime dt3 = (c.length() == size) ?
92  QDateTime::fromString(c, "yyyyMMddHHmmss") :
93  QDateTime::fromString(c, "yyyyMMddHHmmsszzz");
94  dt3.setOffsetFromUtc(0);
95 
96  const qint64 ms1 = dt1.toMSecsSinceEpoch();
97  const qint64 ms2 = dt2.toMSecsSinceEpoch();
98  const qint64 delta = ms1 - ms2;
99 
100  QVERIFY2(dt1 == dt2, "Expect same results of QDateTime");
101  QVERIFY2(dt1 == dt3, "Expect same results of QDateTime");
102  QVERIFY2(delta == 0, "Expect same results timestamp");
103  }

107  QElapsedTimer time;
108  time.start();

110  {
111  for (const QString &dt : dts)

115  parseDateTimeStringOptimized(c);
