165  int CStatusMessageList::keepLatest(int estimtatedNumber)
166  {
167  const int oldSize = this->size();
168  if (estimtatedNumber >= oldSize) { return 0; }
169  if (estimtatedNumber < 1)
170  {
171  this->clear();
172  return oldSize;
173  }
174 
175  CStatusMessageList copy(*this);
176  copy.sortLatestFirst();
177  const QDateTime ts = copy[estimtatedNumber - 1].getUtcTimestamp();
178  copy = *this; // keep order
179  copy.removeBefore(ts);
180  *this = copy;
181  return oldSize - this->size();
182  }

211  CStatusMessage CStatusMessageList::toSingleMessage() const
212  {
213  if (this->isEmpty()) { return CStatusMessage(); }
214  if (this->size() == 1) { return this->front(); }
215  QStringList newMsgs;
216  CStatusMessage::StatusSeverity s = CStatusMessage::SeverityDebug;
217  CLogCategoryList cats;
218  for (const CStatusMessage &msg : *this)
219  {
220  if (msg.isEmpty()) { continue; }
221  newMsgs.append(msg.getMessage());
222  CStatusMessage::StatusSeverity ms = msg.getSeverity();
223  if (s < ms) { s = ms; }
224  cats.push_back(msg.getCategories());
225  }
226  const CStatusMessage newMsg(cats, s, newMsgs.join(", "));
227  return newMsg;
228  }

230  QString CStatusMessageList::toHtml(const CPropertyIndexList &indexes) const
231  {
232  if (indexes.isEmpty() || this->isEmpty()) { return {}; }
233  QString html;
234  int line = 1;
235  const bool withLineNumbers = indexes.contains(CPropertyIndexRef::GlobalIndexLineNumber);
236  CPropertyIndexList usedIndexes(indexes);
237  if (withLineNumbers) { usedIndexes.remove(CPropertyIndexRef::GlobalIndexLineNumber); }
238  for (const CStatusMessage &statusMessage : *this)
239  {
240  QString rowHtml;
241  if (withLineNumbers)
242  {
243  rowHtml = u"<td>" % QString::number(line++) % u"</td>";
244  }
245 
246  for (const CPropertyIndex &index : usedIndexes)
247  {
248  rowHtml += u"<td>" %
249  statusMessage.propertyByIndex(index).toString().toHtmlEscaped().replace('\n', "<br>") %
250  u"</td>";
251  }
252 
253  const QString severityClass = statusMessage.getSeverityAsString();
254  html += QStringLiteral("<tr class=\"%1\">%2</tr>").arg(severityClass, rowHtml);
255  }
256  return u"<table>" % html % u"</table>";
257  }
