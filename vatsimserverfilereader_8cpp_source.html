55  void CVatsimServerFileReader::read()
56  {
57  this->threadAssertCheck();
58  if (!this->doWorkCheck()) { return; }
59  if (!this->isInternetAccessible("No network/internet access, cannot read VATSIM server file")) { return; }
60  if (this->isNotVATSIMEcosystem()) { return; }
61 
62  Q_ASSERT_X(sApp, Q_FUNC_INFO, "Missing application");
63  const QUrl url = sApp->getVatsimServerFileUrl();
64  if (url.isEmpty()) { return; }
65  this->getFromNetworkAndLog(url, { this, &CVatsimServerFileReader::parseVatsimFile });
66  }

68  void CVatsimServerFileReader::parseVatsimFile(QNetworkReply *nwReplyPtr)
69  {
70  // wrap pointer, make sure any exit cleans up reply
71  // required to use delete later as object is created in a different thread
72  QScopedPointer<QNetworkReply, QScopedPointerDeleteLater> nwReply(nwReplyPtr);
73  this->threadAssertCheck();
74  if (this->isNotVATSIMEcosystem()) { return; }
75 
76  // Worker thread, make sure to write only synced here!
77  if (!this->doWorkCheck())
78  {
79  CLogMessage(this).info(u"Terminated VATSIM file parsing process");
80  return; // stop, terminate straight away, ending thread
81  }
82 
83  this->logNetworkReplyReceived(nwReplyPtr);
84 
85  const QUrl url = nwReply->url();
86  const QString urlString = url.toString();
87 
88  if (nwReply->error() == QNetworkReply::NoError)
89  {
90  const QString dataFileData = nwReply->readAll();
91  nwReply->close(); // close asap
92 
93  if (dataFileData.isEmpty()) { return; }
94  if (!this->didContentChange(dataFileData)) // Quick check by hash
95  {
96  CLogMessage(this).info(u"VATSIM file '%1' has same content, skipped") << urlString;
97  return;
98  }
99  auto jsonDoc = QJsonDocument::fromJson(dataFileData.toUtf8());
100  if (jsonDoc.isEmpty()) { return; }
101 
102  // build on local vars for thread safety
103  CServerList fsdServers;
104 
105  for (QJsonValueRef server : jsonDoc.array())
106  {
107  if (!this->doWorkCheck())
108  {
109  CLogMessage(this).info(u"Terminated VATSIM file parsing process");
110  return;
111  }
112  fsdServers.push_back(parseServer(server.toObject()));
113  if (!fsdServers.back().hasName()) { fsdServers.pop_back(); }
114  }
115 
116  // Setup for VATSIM servers and sorting for comparison
117  fsdServers.sortBy(&CServer::getName, &CServer::getDescription);
118 
119  // update cache itself is thread safe
120  CVatsimSetup vs(m_lastGoodSetup.get());
121  const bool changedSetup = vs.setServers(fsdServers, {});
122  if (changedSetup)
123  {
124  vs.setCurrentUtcTime();
125  m_lastGoodSetup.set(vs);
126  }
127 
128  // data read finished
129  emit this->dataFileRead(dataFileData.size() / 1000);
130  emit this->dataRead(CEntityFlags::VatsimDataFile, CEntityFlags::ReadFinished, dataFileData.size() / 1000, url);
131  }
132  else
133  {
134  // network error
135  CLogMessage(this).warning(u"Reading VATSIM data file failed '%1' '%2'") << nwReply->errorString() << urlString;
136  nwReply->abort();
137  emit this->dataRead(CEntityFlags::VatsimDataFile, CEntityFlags::ReadFailed, 0, url);
138  }
139  }
