113  void CFlightPlanRemarks::parseFlightPlanRemarks(bool force)
114  {
115  // examples: VFPS = VATSIM Flightplan Prefile System
116  // 1) RT/KESTREL OPR/MYTRAVEL REG/G-DAJC SEL/FP-ES PER/C NAV/RNP10
117  // 2) OPR/UAL CALLSIGN/UNITED
118  // 3) /v/FPL-VIR9-IS-A346/DEP/S-EGLL/ARR/KJFK/REG/G-VGAS/TCAS RVR/200 OPR/VIRGIN AIRLINES
119 
120  if (!force && m_isParsed) { return; }
121  m_isParsed = true;
122  if (m_remarks.isEmpty()) { return; }
123  const QString remarks = m_remarks.toUpper();
124  const QString callsign = CCallsign::unifyCallsign(this->getRemark(remarks, "REG/")); // registration is a callsign
125  if (CCallsign::isValidAircraftCallsign(callsign)) { m_registration = CCallsign(callsign, CCallsign::Aircraft); }
126  m_voiceCapabilities = m_voiceCapabilities.isUnknown() ? CVoiceCapabilities(m_remarks) : m_voiceCapabilities;
127  m_flightOperator = this->getRemark(remarks, "OPR/"); // operator, e.g. British airways, sometimes people use ICAO code here
128  m_selcalCode = CSelcal(this->getRemark(remarks, "SEL/"));
129  m_radioTelephony = getRemark(remarks, "CALLSIGN/"); // used similar to radio telephony
130  if (m_radioTelephony.isEmpty()) { m_radioTelephony = getRemark(remarks, "RT/"); }
131  if (!m_flightOperator.isEmpty() && CAirlineIcaoCode::isValidAirlineDesignator(m_flightOperator))
132  {
133  // if people use ICAO code as flight operator swap with airline ICAO
134  m_airlineIcao = CAirlineIcaoCode(m_flightOperator);
135  m_flightOperator.clear();
136  }
137  }

139  void CFlightPlanRemarks::setIcaoEquipmentCodes(const QString &eq)
140  {
141  // https://en.wikipedia.org/wiki/Equipment_codes
142  const QString r = eq.toUpper().trimmed();
143  QStringList remarks;
144  if (r.isEmpty()) { return; }
145  if (r.contains('W'))
146  {
147  if (r.contains('G')) { remarks << "/L"; }
148  else if (r.contains('D') || r.contains('L') || r.contains('O')) { remarks << "/Z"; }
149  else if (r.contains('A')) { remarks << "/W"; }
150  }
151  else
152  {
153  if (r.contains('G')) { remarks << "/G"; }
154  else if (r.contains('D') || r.contains('L') || r.contains('O')) { remarks << "/I"; }
155  }
156 
157  if (r.contains('G')) { remarks << "NAV/GPSRNAV"; }
158  else if (r.contains('D') || r.contains('L') || r.contains('O')) { remarks << "NAV/VORNDB"; }
159 
160  m_remarks = remarks.join(" ");
161  this->parseFlightPlanRemarks(true);
162  }

164  QString CFlightPlanRemarks::getRemark(const QString &remarks, const QString &marker)
165  {
166  const int maxIndex = remarks.size() - 1;
167  int f = remarks.indexOf(marker);
168  if (f < 0) { return {}; }
169  f += marker.length();
170  if (maxIndex <= f) { return {}; }
171 
172  // the remarks are poorly formatted:
173  // 1) sometimes the values are enclosed in "/", like "/REG/D-AMBZ/"
174  // 2) sometimes the values are containing space, like "/OPR/DELTA AIRLINES"
175  // 3) in many cases the end delimiter is a new marker or the EOL (otherwise 1 applies)
176 
177  thread_local const QRegularExpression nextMarker("\\s+\\w*/|$");
178  int to1 = remarks.indexOf(nextMarker, f + 1); // for case 2,3
179  if (to1 < 0) { to1 = maxIndex + 1; }
180  int to2 = remarks.indexOf('/', f + 1); // for case 1
181  if (to2 < 0) { to2 = maxIndex + 1; } // no more end markers, ends after last character
182  const int to = qMin(to1, to2);
183  const QString cut = remarks.mid(f, to - f).simplified();
184  return cut;
185  }

187  QString CFlightPlanRemarks::replaceRemark(const QString &remarks, const QString &marker, const QString &newRemark)
188  {
189  QString r(remarks);
190  const int maxIndex = remarks.size() - 1;
191  int f = remarks.indexOf(marker);
192  if (f >= 0)
193  {
194  f += marker.length();
195  if (maxIndex <= f) { return remarks; }
196  thread_local const QRegularExpression nextMarker("\\s+\\w*/|$");
197  int to1 = remarks.indexOf(nextMarker, f + 1); // for case 2,3
198  if (to1 < 0) { to1 = maxIndex + 1; }
199  int to2 = remarks.indexOf('/', f + 1); // for case 1
200  if (to2 < 0) { to2 = maxIndex + 1; } // no more end markers, ends after last character
201  const int to = qMin(to1, to2);
202  r.remove(f, to - f);
203  }
204  return r.isEmpty() ? newRemark : r % u" " % newRemark;
205  }

496  const QDomNodeList destList = doc.elementsByTagName("destination");

501  fp.setDestinationAirportIcao(icao);

503  const QDomNodeList altList = doc.elementsByTagName("alternate");
504  if (!altList.isEmpty())
505  {
506  const QDomNode alternate = altList.at(0);
507  const QString icao = alternate.firstChildElement("icao_code").text();
508  fp.setAlternateAirportIcao(icao);
509  }
510  const QDomNodeList generalList = doc.elementsByTagName("general");
511  if (!generalList.isEmpty())
512  {
513  bool ok;
514  const QDomNode general = generalList.at(0);
515  QString route = general.firstChildElement("route").text();
516  fp.setRoute(route.remove("DCT").simplified().trimmed());
517  const QString airline = general.firstChildElement("icao_airline").text();
518  const QString flightNumber = general.firstChildElement("flight_number").text();
519  fp.setCallsign(CCallsign(airline + flightNumber, CCallsign::Aircraft));
520  const QString cruiseAlt = general.firstChildElement("initial_altitude").text();
521  const int cruiseAltFt = cruiseAlt.toInt(&ok);
522  if (ok)
523  {
524  CAltitude ca(cruiseAltFt, CAltitude::MeanSeaLevel, CLengthUnit::ft());
525  if (cruiseAlt.endsWith("00") && cruiseAltFt > 5000)
526  {
527  ca.toFlightLevel();
528  }
529  fp.setCruiseAltitude(ca);
530  if (cruiseAltFt >= 10000) { fp.setFlightRule(CFlightPlan::IFR); } // good guess
531  else { fp.setFlightRule(CFlightPlan::VFR); }
532  }
533  else
534  {
535  fp.setCruiseAltitudeString(cruiseAlt);
536  }

738  QStringList CFlightPlan::splitEquipmentCode(const QString &equipmentCodeAndAircraft)
739  {
740  static const QStringList empty({ "", "", "" });
741  if (empty.isEmpty()) { return empty; }
742  QStringList firstSplit = equipmentCodeAndAircraft.split('-');
743  if (firstSplit.size() >= 2)
744  {
745  // format like B789/H-SDE1E2E3FGHIJ2J3J4J5M1RWXY/LB1D1
746  QString equipment = firstSplit[1];
747  QStringList split = firstSplit[0].split('/');
748  if (split.size() >= 3)
749  {
750  return { split[2], split[1], equipment.isEmpty() ? split[0] : equipment }; // "F/B789/H"
751  }
752  else if (split.size() >= 2)
753  {
754  if (split[0].size() <= 1) // "H/B789"
755  {
756  return { split[0], split[1], equipment };
757  }
758  else // "B789/H"
759  {
760  return { split[1], split[0], equipment };
761  }
762  }
763  else // "B789"
764  {
765  return { {}, split[0], equipment };
766  }
767  }
768  QStringList split = equipmentCodeAndAircraft.split('/');
769  if (split.length() >= 3)
770  {
771  if (split[1].size() == 1 && CAircraftIcaoCode::isValidDesignator(split[0]))
772  {
773  using std::swap;
774  swap(split[0], split[1]); // "A359/H/L"
775  }
776  return split; // "H/B738/F"
777  }
778  if (split.length() == 2)
779  {
780  if (split[0].length() == 1)
781  {
782  // we assume prefix + ICAO
783  // e.g. "H/B748"
784  split.push_back("");
785  return split;
786  }
787  else
788  {
789  // we assume ICAO + suffix
790  // e.g. "B748/F"
791  split.push_front("");
792  return split;
793  }
794  }
795 
796  // one part only
797  if (split[0].length() > 1 && CAircraftIcaoCode::isValidDesignator(split[0]))
798  {
799  QStringList sl(empty);
800  sl[1] = split[0]; // only ICAO
801  return sl;
802  }
803  if (split[0].length() != 1) { return empty; } // something invalid
804 
805  // one part, one char only. hard to tell
806  QStringList sl(empty);
807  if (faaEquipmentCodes().contains(split[0]))
808  {
809  sl[2] = split[0]; // return as equipment code
810  return sl;
811  }
812  sl[0] = split[0];
813  return sl;
814  }
