61  void CVoiceCapabilities::setFromFlightPlanRemarks(const QString &flightPlanRemarks)
62  {
63  // http://www.vatpac.org/cms/index.php?option=com_content&view=article&id=608&Itemid=213
64 
65  // this function is very frequently called, so I put the most likely scenarios on top
66  const QString r = flightPlanRemarks.toUpper().trimmed();
67  if (flightPlanRemarks.isEmpty())
68  {
69  this->setCapabilities(Unknown);
70  return;
71  }
72  if (r.contains("/V/"))
73  {
74  this->setCapabilities(Voice);
75  return;
76  }
77  if (r.contains("/T/"))
78  {
79  this->setCapabilities(TextOnly);
80  return;
81  }
82  if (r.contains("/R/"))
83  {
84  this->setCapabilities(VoiceReceivingOnly);
85  return;
86  }
87  if (r.contains("/VOICE/"))
88  {
89  this->setCapabilities(Voice);
90  return;
91  }
92  if (r.contains("/TEXT/"))
93  {
94  this->setCapabilities(TextOnly);
95  return;
96  }
