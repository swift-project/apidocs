45  void CDatabaseReader::readInBackgroundThread(CEntityFlags::Entity entities, const QDateTime &newerThan)
46  {
47  if (!this->doWorkCheck()) { return; }
48 
49  // we accept cached data
50  Q_ASSERT_X(!entities.testFlag(CEntityFlags::DbInfoObjectEntity), Q_FUNC_INFO, "Read info objects directly");
51  const bool hasDbInfoObjects = this->hasDbInfoObjects(); // no info objects is not necessarily an error, but indicates a) either data not available (DB down) or b) only caches are used
52  // const bool hasSharedInfoObjects = this->hasSharedInfoObjects();
53  CEntityFlags::Entity allEntities = entities;
54  CEntityFlags::Entity cachedEntities = CEntityFlags::NoEntity;
55  CEntityFlags::Entity dbEntities = CEntityFlags::NoEntity;
56  CEntityFlags::Entity sharedEntities = CEntityFlags::NoEntity;
57  CEntityFlags::Entity currentEntity = CEntityFlags::iterateDbEntities(allEntities); // CEntityFlags::InfoObjectEntity will be ignored
58  while (currentEntity)
59  {
60  const CDatabaseReaderConfig config(this->getConfigForEntity(currentEntity));
61  const QString currentEntityName = CEntityFlags::flagToString(currentEntity);
62 
63  // retrieval mode
64  const CDbFlags::DataRetrievalMode rm = config.getRetrievalMode(); // DB reading, cache, shared
65  Q_ASSERT_X(!rm.testFlag(CDbFlags::Unspecified), Q_FUNC_INFO, "Missing retrieval mode");
66  const QString rmString = CDbFlags::flagToString(rm);
67  const CDbFlags::DataRetrievalModeFlag rmDbOrSharedFlag = CDbFlags::modeToModeFlag(rm & CDbFlags::DbReadingOrShared);
68  const QString rmDbOrSharedFlagString = CDbFlags::flagToString(rmDbOrSharedFlag);
69  const bool rmDbReadingOrShared = (rmDbOrSharedFlag == CDbFlags::DbReading || rmDbOrSharedFlag == CDbFlags::Shared);
70  const int currentEntityCount = this->getCacheCount(currentEntity);
71 
72  if (rm.testFlag(CDbFlags::Ignore) || rm.testFlag(CDbFlags::Canceled))
73  {
74  // do not load
75  }
76  else if (rm.testFlag(CDbFlags::Cached))
77  {
78  //
79  // !!! info object comparisons only for: cache + shared or cache + DB data
80  //
81  if (hasDbInfoObjects && rmDbReadingOrShared)
82  {
83  // check mode here for consistency
84  Q_ASSERT_X(!getBaseUrl(rmDbOrSharedFlag).isEmpty(), Q_FUNC_INFO, "Wrong retrieval mode");
85 
86  CUrl oldUrlInfo;
87  CUrl newUrlInfo;
88  const bool changedUrl = this->hasChangedUrl(currentEntity, oldUrlInfo, newUrlInfo);
89  const QDateTime cacheTs(this->getCacheTimestamp(currentEntity));
90  const QDateTime latestEntityTs(this->getLatestEntityTimestampFromDbInfoObjects(currentEntity));
91  const qint64 cacheTimestamp = cacheTs.isValid() ? cacheTs.toMSecsSinceEpoch() : -1;
92  const qint64 latestEntityTimestamp = latestEntityTs.isValid() ? latestEntityTs.toMSecsSinceEpoch() : -1;
93  Q_ASSERT_X(latestEntityTimestamp >= 0, Q_FUNC_INFO, "Missing timestamp");
94  if (!changedUrl && cacheTimestamp >= latestEntityTimestamp && cacheTimestamp >= 0 && latestEntityTimestamp >= 0)
95  {
96  this->admitCaches(currentEntity);
97  cachedEntities |= currentEntity; // read from cache
98  CLogMessage(this).info(u"Using cache for '%1' (%2, %3)") << currentEntityName << cacheTs.toString() << cacheTimestamp;
99  }
100  else
101  {
102  Q_ASSERT_X(rmDbReadingOrShared, Q_FUNC_INFO, "Wrong retrieval mode");
103  if (rmDbOrSharedFlag == CDbFlags::DbReading) { dbEntities |= currentEntity; }
104  else if (rmDbOrSharedFlag == CDbFlags::Shared) { sharedEntities |= currentEntity; }
105 
106  if (changedUrl)
107  {
108  CLogMessage(this).info(u"Data location for '%1' changed ('%2'->'%3'), will override cache for reading '%4'")
109  << currentEntityName << oldUrlInfo.toQString()
110  << newUrlInfo.toQString() << rmDbOrSharedFlagString;
111  }
112  else
113  {
114  CLogMessage(this).info(u"Cache for '%1' outdated, latest entity (%2, %3), reading '%4'") << currentEntityName << latestEntityTs.toString() << latestEntityTimestamp << rmDbOrSharedFlagString;
115  }
116  }
117  }
118  else
119  {
120  if (!rmDbReadingOrShared) { CLogMessage(this).info(u"No DB or shared reading for '%1', read mode is: '%2'") << currentEntityName << rmString; }
121  if (!hasDbInfoObjects) { CLogMessage(this).info(u"No DB info objects for '%1', read mode is: '%2'") << currentEntityName << rmString; }
122  if (currentEntityCount > 0)
123  {
124  CLogMessage(this).info(u"Cache for '%1' already read, %2 entries") << currentEntityName << currentEntityCount;
125  }
126  else
127  {
128  // no info objects, server down or no shared/db read mode
129  this->admitCaches(currentEntity);
130  if (!rmDbReadingOrShared)
131  {
132  // intentionally we do not want to read from DB/shared
133  CLogMessage(this).info(u"Triggered reading cache for '%1', read mode: %2") << currentEntityName << rmString;
134  }
135  else
136  {
137  // we want to read from DB/shared, but have no info object
138  CLogMessage(this).info(u"No info object for '%1', triggered reading cache, read mode: %2") << currentEntityName << rmString;
139  }
140  }
141  cachedEntities |= currentEntity; // read from cache
142  }
143  }
144  else
145  {
146  // cache ignored
147  Q_ASSERT_X(rmDbReadingOrShared, Q_FUNC_INFO, "Wrong retrieval mode");
148  if (rmDbOrSharedFlag == CDbFlags::DbReading) { dbEntities |= currentEntity; }
149  else if (rmDbOrSharedFlag == CDbFlags::Shared) { sharedEntities |= currentEntity; }
150  }
151  currentEntity = CEntityFlags::iterateDbEntities(allEntities);
152  }
153 
154  // signals for the cached entities
155  if (cachedEntities != CEntityFlags::NoEntity)
156  {
157  this->emitReadSignalPerSingleCachedEntity(cachedEntities, true);
158  }
159 
160  // Real read from DB
161  if (dbEntities != CEntityFlags::NoEntity)
162  {
163  CLogMessage(this).info(u"Start reading DB entities: %1") << CEntityFlags::flagToString(dbEntities);
164  this->startReadFromBackendInBackgroundThread(dbEntities, CDbFlags::DbReading, newerThan);
165  }
166 
167  // Real read from shared
168  if (sharedEntities != CEntityFlags::NoEntity)
169  {
170  CLogMessage(this).info(u"Start reading shared entities: %1") << CEntityFlags::flagToString(sharedEntities);
171  this->startReadFromBackendInBackgroundThread(dbEntities, CDbFlags::Shared, newerThan);
172  }
173  }

181  CEntityFlags::Entity CDatabaseReader::triggerLoadingDirectlyFromSharedFiles(CEntityFlags::Entity entities, bool checkCacheTsUpfront)
182  {
183  if (entities == CEntityFlags::NoEntity) { return CEntityFlags::NoEntity; }
184  if (checkCacheTsUpfront)
185  {
186  CEntityFlags::Entity newerHeaderEntities = this->getEntitesWithNewerSharedInfoObject(entities);
187  if (newerHeaderEntities != entities)
188  {
189  const CEntityFlags::Entity validInCacheEntities = (entities ^ newerHeaderEntities) & entities;
190  CLogMessage(this).info(u"Reduced '%1' to '%2' before triggering load of shared files (still in cache)") << CEntityFlags::flagToString(entities) << CEntityFlags::flagToString(newerHeaderEntities);
191 
192  // In case we have difference entities we treat them as they were loaded, they result from still being in the cache
193  // Using timer to first finish this function, then the resulting signal
194  if (validInCacheEntities != CEntityFlags::NoEntity)
195  {
196  QPointer<CDatabaseReader> myself(this);
197  QTimer::singleShot(0, this, [=] {
198  if (!myself) { return; }
199  if (!sApp || sApp->isShuttingDown()) { return; }
200  emit this->dataRead(validInCacheEntities, CEntityFlags::ReadFinished, 0, {});
201  });
202  }
203  if (newerHeaderEntities == CEntityFlags::NoEntity) { return CEntityFlags::NoEntity; }
204  entities = newerHeaderEntities;
205  }
206  }
207  this->startReadFromBackendInBackgroundThread(entities, CDbFlags::Shared);
208  return entities;
209  }

211  void CDatabaseReader::startReadFromBackendInBackgroundThread(CEntityFlags::Entity entities, CDbFlags::DataRetrievalModeFlag mode, const QDateTime &newerThan)
212  {
213  Q_ASSERT_X(mode == CDbFlags::DbReading || mode == CDbFlags::Shared, Q_FUNC_INFO, "Wrong mode");
214 
215  // ps_read is implemented in the derived classes
216  if (entities == CEntityFlags::NoEntity) { return; }
217  if (!this->isInternetAccessible(QStringLiteral("No network/internet access, will not read %1").arg(CEntityFlags::flagToString(entities)))) { return; }
218 
220  QPointer<CDatabaseReader> myself(this);
221  QTimer::singleShot(0, this, [=] {
222  if (!sApp || sApp->isShuttingDown() || !myself) { return; }
223  this->read(entities, mode, newerThan);
224  });
225  }

228  {
229  Q_ASSERT_X(nwReply, Q_FUNC_INFO, "missing reply");
230  JsonDatastoreResponse datastoreResponse;

240  }
241  else

243  CDatabaseReader::stringToDatastoreResponse(dataFileData, datastoreResponse);
244  }
245  }
246  return datastoreResponse;

257  bool CDatabaseReader::setHeaderInfoPart(CDatabaseReader::HeaderResponse &headerResponse, QNetworkReply *nwReply) const
258  {
259  Q_ASSERT_X(nwReply, Q_FUNC_INFO, "Missing reply");
260  this->threadAssertCheck();
261  if (!this->doWorkCheck())
262  {
263  nwReply->abort();
264  headerResponse.setMessage(CStatusMessage(this, CStatusMessage::SeverityError, u"Terminated data parsing process"));
265  return false; // stop, terminate straight away, ending thread
266  }
267 
268  headerResponse.setValues(nwReply);
269  if (nwReply->error() == QNetworkReply::NoError)
270  {
271  // do not close because of obtaining data
272  return true;
273  }
274  else
275  {
276  // no valid response
277  const QString error(nwReply->errorString());
278  const QString url(nwReply->url().toString());
279  nwReply->abort();
280  headerResponse.setMessage(CStatusMessage(this, CStatusMessage::SeverityError,
281  u"Reading data failed: '" % error % u"' " % url));
282  return false;
283  }
284  }

383  int CDatabaseReader::requestHeadersOfSharedFiles(CEntityFlags::Entity entities)
384  {
385  if (!this->isInternetAccessible(QStringLiteral("No network/internet access, will not read shared file headers for %1").arg(CEntityFlags::flagToString(entities)))) { return false; }
386 
387  CEntityFlags::Entity allEntities = entities & CEntityFlags::AllDbEntitiesNoInfoObjects;
388  CEntityFlags::Entity currentEntity = CEntityFlags::iterateDbEntities(allEntities);
389  const CUrl urlSharedDbdata = CDatabaseReader::getWorkingSharedDbdataDirectoryUrl();
390  if (urlSharedDbdata.isEmpty())
391  {
392  CLogMessage(this).warning(u"No working shared URL, cannot request headers");
393  return 0;
394  }
395 
396  int c = 0;
397  while (currentEntity != CEntityFlags::NoEntity)
398  {
399  const QString fileName = CDbInfo::entityToSharedName(currentEntity);
400  Q_ASSERT_X(!fileName.isEmpty(), Q_FUNC_INFO, "No file name for entity");
401  CUrl url = urlSharedDbdata;
402  url.appendPath(fileName);
403 
404  const QString entityString = CEntityFlags::flagToString(currentEntity);
405  CLogMessage(this).info(u"Triggered read of header for shared file of '%1'") << entityString;
406  const QNetworkReply *reply = sApp->headerFromNetwork(url, { this, &CDatabaseReader::receivedSharedFileHeader });
407  if (reply) { c++; }
408  currentEntity = CEntityFlags::iterateDbEntities(allEntities);
409  }
410  return c > 0;
411  }

472  CEntityFlags::Entity CDatabaseReader::emitReadSignalPerSingleCachedEntity(CEntityFlags::Entity cachedEntities, bool onlyIfHasData)
473  {
474  if (cachedEntities == CEntityFlags::NoEntity) { return CEntityFlags::NoEntity; }
475  CEntityFlags::Entity emitted = CEntityFlags::NoEntity;
476  CEntityFlags::Entity cachedEntitiesToEmit = cachedEntities;
477  CEntityFlags::Entity currentCachedEntity = CEntityFlags::iterateDbEntities(cachedEntitiesToEmit);
478  while (currentCachedEntity)
479  {
480  const int c = this->getCacheCount(currentCachedEntity);
481  if (!onlyIfHasData || c > 0)
482  {
483  emit this->dataRead(currentCachedEntity, CEntityFlags::ReadFinished, c, {});
484  emitted |= currentCachedEntity;
485  }
486  currentCachedEntity = CEntityFlags::iterateDbEntities(cachedEntitiesToEmit);
487  }
488  return emitted;
489  }

561  bool CDatabaseReader::hasReceivedOkReply(QString &message) const
562  {
563  QReadLocker rl(&m_statusLock);

746  void CDatabaseReader::stringToDatastoreResponse(const QString &jsonContent, JsonDatastoreResponse &datastoreResponse)
747  {
748  const int status = datastoreResponse.getHttpStatusCode();
749  if (jsonContent.isEmpty())
750  {
751  static const QString errorMsg = "Empty JSON string, status: %1, URL: '%2', load time: %3";
752  datastoreResponse.setMessage(CStatusMessage(static_cast<CDatabaseReader *>(nullptr),
753  CStatusMessage::SeverityError,
754  errorMsg.arg(status).arg(datastoreResponse.getUrlString(), datastoreResponse.getLoadTimeStringWithStartedHint())));
755  return;
756  }
757 
758  const QJsonDocument jsonResponse = CDatabaseUtils::databaseJsonToQJsonDocument(jsonContent);
759  if (jsonResponse.isEmpty())
760  {
761  if (CNetworkUtils::looksLikePhpErrorMessage(jsonContent))
762  {
763  static const QString errorMsg = "Looks like PHP errror, status %1, URL: '%2', msg: %3";
764  const QString phpErrorMessage = CNetworkUtils::removeHtmlPartsFromPhpErrorMessage(jsonContent);
765  datastoreResponse.setMessage(CStatusMessage(static_cast<CDatabaseReader *>(nullptr),
766  CStatusMessage::SeverityError,
767  errorMsg.arg(status).arg(datastoreResponse.getUrlString(), phpErrorMessage)));
768  }
769  else
770  {
771  static const QString errorMsg = "Empty JSON document, URL: '%1', load time: %2";
772  datastoreResponse.setMessage(CStatusMessage(static_cast<CDatabaseReader *>(nullptr),
773  CStatusMessage::SeverityError,
774  errorMsg.arg(datastoreResponse.getUrlString(), datastoreResponse.getLoadTimeStringWithStartedHint())));
775  }
776  return;
777  }
778 
779  if (jsonResponse.isArray())
780  {
781  // directly an array, no further info
782  datastoreResponse.setJsonArray(jsonResponse.array());
783  datastoreResponse.setLastModifiedTimestamp(QDateTime::currentDateTimeUtc());
784  }
785  else
786  {
787  const QJsonObject responseObject(jsonResponse.object());
788  datastoreResponse.setJsonArray(responseObject["data"].toArray());
789  const QString ts(responseObject["latest"].toString());
790  datastoreResponse.setLastModifiedTimestamp(ts.isEmpty() ? QDateTime::currentDateTimeUtc() : CDatastoreUtility::parseTimestamp(ts));
791  datastoreResponse.setRestricted(responseObject["restricted"].toBool());
792  }
793  }
