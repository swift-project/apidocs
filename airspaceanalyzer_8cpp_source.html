168  void CAirspaceAnalyzer::watchdogCheckTimeouts()
169  {
170  // this is a trick to not remove everything while debugging
171  const qint64 currentTimeMsEpoch = QDateTime::currentMSecsSinceEpoch();
172  const qint64 callDiffMs = currentTimeMsEpoch - m_lastWatchdogCallMsSinceEpoch;
173  const qint64 callThresholdMs = static_cast<qint64>(m_updateTimer.interval() * 1.5);
174  m_lastWatchdogCallMsSinceEpoch = currentTimeMsEpoch;
175  if (callDiffMs > callThresholdMs)
176  {
177  // allow some time to normalize before checking again
178  m_doNotRunAgainBefore = currentTimeMsEpoch + 2 * callThresholdMs;
179  return;
180  }
181  if (m_doNotRunAgainBefore > currentTimeMsEpoch) { return; }
182  m_doNotRunAgainBefore = -1;
183 
184  // checks
185  const qint64 aircraftTimeoutMs = m_timeoutAircraft.valueInteger(CTimeUnit::ms());
186  const qint64 atcTimeoutMs = m_timeoutAtc.valueInteger(CTimeUnit::ms());
187  const qint64 timeoutAircraftEpochMs = currentTimeMsEpoch - aircraftTimeoutMs;
188  const qint64 timeoutAtcEpochMs = currentTimeMsEpoch - atcTimeoutMs;
189  const bool enabled = m_enabledWatchdog;
190 
191  const QList<CCallsign> callsignsAircraft = m_aircraftCallsignTimestamps.keys();
192  for (const CCallsign &callsign : callsignsAircraft) // clazy:exclude=container-anti-pattern,range-loop
193  {
194  if (!enabled) { m_aircraftCallsignTimestamps[callsign] = timeoutAircraftEpochMs + 1000; } // fake value so it can be re-enabled
195  const qint64 tsv = m_aircraftCallsignTimestamps.value(callsign);
196  if (tsv > timeoutAircraftEpochMs) { continue; }
197  CLogMessage(this).debug() << QStringLiteral("Aircraft '%1' timed out after %2ms").arg(callsign.toQString()).arg(currentTimeMsEpoch - tsv);
198  m_aircraftCallsignTimestamps.remove(callsign);
199  emit this->timeoutAircraft(callsign);
200  }
201 
202  const QList<CCallsign> callsignsAtc = m_atcCallsignTimestamps.keys();
203  for (const CCallsign &callsign : callsignsAtc) // clazy:exclude=container-anti-pattern,range-loop
204  {
205  if (!enabled) { m_aircraftCallsignTimestamps[callsign] = timeoutAtcEpochMs + 1000; } // fake value so it can be re-enabled
206  const qint64 tsv = m_atcCallsignTimestamps.value(callsign);
207  if (tsv > timeoutAtcEpochMs) { continue; }
208  CLogMessage(this).debug() << QStringLiteral("ATC '%1' timed out after %2ms").arg(callsign.toQString()).arg(currentTimeMsEpoch - tsv);
209  m_atcCallsignTimestamps.remove(callsign);
210  emit this->timeoutAtc(callsign);
211  }
212  }

214  void CAirspaceAnalyzer::analyzeAirspace()
215  {
216  Q_ASSERT_X(!CThreadUtils::thisIsMainThread(), Q_FUNC_INFO, "Expect to run in background thread");
217  Q_ASSERT_X(thread() != qApp->thread(), Q_FUNC_INFO, "Expect to run in background thread affinity");
218 
219  bool restricted, enabled;
220  int maxAircraft;
221  CLength maxRenderedDistance;
222  {
223  QReadLocker l(&m_lockRestrictions);
224  restricted = m_simulatorRenderedAircraftRestricted;
225  enabled = m_simulatorRenderingEnabled;
226  maxAircraft = m_simulatorMaxRenderedAircraft;
227  maxRenderedDistance = m_simulatorMaxRenderedDistance;
228  }
229 
230  // remark for simulation snapshot is used when there are restrictions
231  // nevertheless we calculate all the time as the snapshot could be used in other scenarios
232 
233  CSimulatedAircraftList aircraftInRange(this->getAircraftInRange()); // thread safe copy from provider
234  CAirspaceAircraftSnapshot snapshot(
235  aircraftInRange,
236  restricted, enabled,
237  maxAircraft, maxRenderedDistance);
238 
239  // lock block
240  {
241  QWriteLocker l(&m_lockSnapshot);
242  bool wasValid = m_latestAircraftSnapshot.isValidSnapshot();
243  if (wasValid)
244  {
245  snapshot.setRestrictionChanged(m_latestAircraftSnapshot);
246  }
247  m_latestAircraftSnapshot = snapshot;
248  if (!wasValid) { return; } // ignore the 1st snapshot
249  }
250 
251  emit this->airspaceAircraftSnapshot(snapshot);
252  }
253 } // ns
