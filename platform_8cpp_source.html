128  CPlatform::Platform CPlatform::stringToPlatform(const QString &str)
129  {
130  const QString s(str.trimmed().toLower());
131  if (s.contains("win"))
132  {
133  if (s.contains("32")) return Win32;
134  return Win64;
135  }
136  if (s.contains("linux")) return Linux;
137  if (s.contains("mac") || s.contains("osx")) return MacOS;
138 
139  // special ones
140  if (s.contains("debian")) return Linux;
141  if (s.contains("darwin")) return MacOS;
142  if (s.contains("all")) return All;
143  if (s.contains("independent")) return Independent;
144 
145  return UnknownOs;
146  }
