38  bool CFsCommonUtil::adjustFileDirectory(CAircraftModel &model, const QString &simObjectsDirectory)
39  {
40  if (model.hasExistingCorrespondingFile()) { return true; }
41  if (simObjectsDirectory.isEmpty()) { return false; }
42  if (!model.hasFileName()) { return false; } // we can do nothing here
43 
44  const QString simObjectsDirectoryFix = CFileUtils::fixWindowsUncPath(simObjectsDirectory);
45  const QDir dir(simObjectsDirectoryFix);
46  if (!dir.exists()) { return false; }
47 
48  const QString lastSegment = u'/' % CFileUtils::lastPathSegment(simObjectsDirectoryFix) % u'/';
49  const int index = model.getFileName().lastIndexOf(lastSegment);
50  if (index < 0) { return false; }
51  const QString relPart = model.getFileName().mid(index + lastSegment.length());
52  if (relPart.isEmpty()) { return false; }
53  const QString newFile = CFileUtils::appendFilePathsAndFixUnc(simObjectsDirectory, relPart);
54  const QFileInfo nf(newFile);
55  if (!nf.exists()) { return false; }
56 
57  model.setFileName(newFile);
58  return true;
59  }

87  if (!td.exists())

94  targetDir = CFileUtils::appendFilePathsAndFixUnc(targetDir, lastSegment);
95  const bool hasDir = td.mkpath(targetDir);
96  if (!hasDir)
97  {
98  messages.push_back(CStatusMessage(static_cast<CFsCommonUtil *>(nullptr), CStatusMessage::SeverityError, u"Cannot create target directory '%1'") << targetDir);
99  return -1;
100  }
101 
102  const int copied = CDirectoryUtils::copyDirectoryRecursively(CSwiftDirectories::shareTerrainProbeDirectory(), targetDir, true);
103  messages.push_back(CStatusMessage(static_cast<CFsCommonUtil *>(nullptr), CStatusMessage::SeverityInfo, u"Copied %1 files from '%2' to '%3'") << copied << CSwiftDirectories::shareTerrainProbeDirectory() << targetDir);
104  return copied;
105  }

107  CStatusMessageList CFsCommonUtil::validateAircraftConfigFiles(const CAircraftModelList &models, CAircraftModelList &validModels, CAircraftModelList &invalidModels, bool ignoreEmptyFileNames, int stopAtFailedFiles, std::atomic_bool &wasStopped)
108  {
109  CStatusMessage m;
110  CAircraftModelList sorted(models);
111  sorted.sortByFileName();
112  wasStopped = false;
113  CStatusMessageList msgs = sorted.validateFiles(validModels, invalidModels, ignoreEmptyFileNames, stopAtFailedFiles, wasStopped, QString(), true);
114  if (wasStopped || validModels.isEmpty()) { return msgs; }
115 
116  const CAircraftModelList nonFsModels = validModels.findNonFsFamilyModels();
117  if (!nonFsModels.isEmpty())
118  {
119  for (const CAircraftModel &model : nonFsModels)
120  {
121  m = CStatusMessage(static_cast<CFsCommonUtil *>(nullptr), CStatusMessage::SeverityError, QStringLiteral("Removed '%1' non FS family model").arg(model.getModelStringAndDbKey()), true);
122  msgs.push_back(m);
123  if (wasStopped) { break; } // allow to break from "outside"
124  }
125 
126  const int d = validModels.removeIfNotFsFamily();
127  m = CStatusMessage(static_cast<CFsCommonUtil *>(nullptr), CStatusMessage::SeverityError, QStringLiteral("Removed %1 non FS family models").arg(d), true);
128  msgs.push_back(m);
129  }
130 
131  // all those files should work
132  int removedCfgEntries = 0;
133  const QSet<QString> fileNames = validModels.getAllFileNames();
134  for (const QString &fileName : fileNames)
135  {
136  bool ok = false;
137  if (wasStopped) { break; } // allow to break from "outside"
138  const CAircraftCfgEntriesList entries = CAircraftCfgParser::performParsingOfSingleFile(fileName, ok, msgs);
139  const QSet<QString> removeModelStrings = entries.getTitleSetUpperCase();
140  const CAircraftModelList removedModels = validModels.removeIfFileButNotInSet(fileName, removeModelStrings);
141  for (const CAircraftModel &removedModel : removedModels)
142  {
143  removedCfgEntries++;
144  m = CStatusMessage(static_cast<CFsCommonUtil *>(nullptr), CStatusMessage::SeverityError, QStringLiteral("'%1' removed because no longer in '%2'").arg(removedModel.getModelStringAndDbKey(), removedModel.getFileName()), true);
145  msgs.push_back(m);
146  CAircraftModelList::addAsValidOrInvalidModel(removedModel, false, validModels, invalidModels);
147  }
148  }
149 
150  if (removedCfgEntries < 1)
151  {
152  m = CStatusMessage(static_cast<CFsCommonUtil *>(nullptr), CStatusMessage::SeverityInfo, QStringLiteral("Not removed any models, all OK!"), true);
153  msgs.push_back(m);
154  }
155 
156  if (!validModels.isEmpty())
157  {
158  m = CStatusMessage(static_cast<CFsCommonUtil *>(nullptr), CStatusMessage::SeverityInfo, QStringLiteral("cfg validation, valid models: %1").arg(validModels.size()), true);
159  msgs.push_back(m);
160  }
161  if (!invalidModels.isEmpty())
162  {
163  m = CStatusMessage(static_cast<CFsCommonUtil *>(nullptr), CStatusMessage::SeverityWarning, QStringLiteral("cfg validation, invalid models: %1").arg(invalidModels.size()), true);
164  msgs.push_back(m);
165  }
166 
167  // finished
168  return msgs;
169  }

185  CStatusMessageList CFsCommonUtil::validateSimObjectsPath(
186  const QSet<QString> &simObjectDirs, const CAircraftModelList &models,
187  CAircraftModelList &validModels, CAircraftModelList &invalidModels,
188  bool ignoreEmptyFileNames, int stopAtFailedFiles, std::atomic_bool &stopped)
189  {
190  CStatusMessageList msgs;
191  if (simObjectDirs.isEmpty())
192  {
193  msgs.push_back(CStatusMessage(static_cast<CFsCommonUtil *>(nullptr)).validationInfo(u"No SimObject directories from cfg files, skipping validation"));
194  return msgs;
195  }
196 
197  CAircraftModelList sortedModels(models);
198  sortedModels.sortByFileName();
199  if (sortedModels.isEmpty())
200  {
201  msgs.push_back(CStatusMessage(static_cast<CFsCommonUtil *>(nullptr)).validationInfo(u"No models to validate"));
202  return msgs;
203  }
204 
205  // info
206  const QString simObjDirs = joinStringSet(simObjectDirs, ", ");
207  msgs.push_back(CStatusMessage(static_cast<CFsCommonUtil *>(nullptr)).validationInfo(u"Validating %1 models against %2 SimObjects path(s): '%3'") << models.size() << simObjectDirs.size() << simObjDirs);
208 
209  // validate
210  int failed = 0;
211  for (const CAircraftModel &model : models)
212  {
213  if (!model.hasFileName())
214  {
215  if (ignoreEmptyFileNames) { continue; }
216  msgs.push_back(CStatusMessage(static_cast<CFsCommonUtil *>(nullptr)).validationWarning(u"No file name for model '%1'") << model.getModelString());
217  CAircraftModelList::addAsValidOrInvalidModel(model, false, validModels, invalidModels);
218  continue;
219  }
220 
221  bool ok = false;
222  for (const QString &path : simObjectDirs)
223  {
224  if (!model.isInPath(path, CFileUtils::osFileNameCaseSensitivity())) { continue; }
225  ok = true;
226  break;
227  }
228  CAircraftModelList::addAsValidOrInvalidModel(model, ok, validModels, invalidModels);
229  if (!ok)
230  {
231  msgs.push_back(CStatusMessage(static_cast<CFsCommonUtil *>(nullptr)).validationWarning(u"Model '%1' '%2' in none of the %3 SimObjects path(s)") << model.getModelString() << model.getFileName() << simObjectDirs.size());
232  failed++;
233  }
234 
235  if (stopAtFailedFiles > 0 && failed >= stopAtFailedFiles)
236  {
237  stopped = true;
238  msgs.push_back(CStatusMessage(static_cast<CFsCommonUtil *>(nullptr)).validationWarning(u"Stopping after %1 failed models") << failed);
239  break;
240  }
241  } // models
242 
243  return msgs;
244  }
