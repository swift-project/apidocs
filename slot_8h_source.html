26  QMetaObject::Connection connectOnce(T *sender, F signal, U *receiver, G &&slot, Qt::ConnectionType type = Qt::AutoConnection)
27  {
28  std::promise<QMetaObject::Connection> promise;
29  auto called = std::make_shared<std::atomic_flag>();
30  called->clear();
31  auto wrapper = [receiver, called, connection = promise.get_future().share(), slot = std::forward<G>(slot)](auto &&...args) {
32  if (called->test_and_set()) { return; }
33  QObject::disconnect(connection.get());
34  Private::invokeSlot(slot, receiver, std::forward<decltype(args)>(args)...);
35  };
36  auto connection = QObject::connect(sender, signal, receiver, std::move(wrapper), type);
37  promise.set_value(connection);
38  return connection;
39  }
