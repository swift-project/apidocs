145  void CNetworkDetailsComponent::onOverrideCredentialsToPilot()
146  {
147  CServer server;
148  const QObject *s = QObject::sender();
149  if (s == ui->pb_OverrideCredentialsOtherServers)
150  {
151  server = this->getCurrentOtherServer();
152  }
153  else if (s == ui->pb_OverrideCredentialsVatsim)
154  {
155  // the VATSIM server selected has no valid user credentials
156  server = m_networkSetup.getLastVatsimServer();
157  }
158  else { return; }
159  emit this->overridePilot(server.getUser());
160  }

177  void CNetworkDetailsComponent::onWebServiceDataRead(CEntityFlags::Entity entity, CEntityFlags::ReadState state, int number, const QUrl &url)
178  {
179  if (!CEntityFlags::isFinishedReadState(state)) { return; }
180  Q_UNUSED(number)
181  Q_UNUSED(url)
182 
183  if (entity == CEntityFlags::VatsimDataFile)
184  {
185  CServerList vatsimFsdServers = sGui->getIContextNetwork()->getVatsimFsdServers();
186  if (vatsimFsdServers.isEmpty()) { return; }
187  vatsimFsdServers.sortBy(&CServer::getName);
188  const CServer currentServer = m_networkSetup.getLastVatsimServer();
189  ui->comp_VatsimServers->setServers(vatsimFsdServers, true);
190  ui->comp_VatsimServers->preSelect(currentServer.getName());
191  }
192  }
