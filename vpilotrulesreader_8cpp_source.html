135  bool CVPilotRulesReader::read(bool convertToModels)
136  {
137  int loadedFiles = 0;
138  QStringList filesWithProblems;
139  CVPilotModelRuleSet rules;
140  const QStringList fileList(getFiles());
141  for (const QString &fn : fileList)
142  {
143  if (m_shutdown) { return false; }
144  loadedFiles++;
145  bool s = this->loadFile(fn, rules);
146  if (!s) { filesWithProblems.append(fn); }
147  }
148 
149  {
150  QWriteLocker l(&m_lockData);
151  this->m_loadedFiles = loadedFiles;
152  this->m_fileListWithProblems = filesWithProblems;
153  this->m_rules = rules;
154  if (m_shutdown) { return false; }
155  }
156 
157  const bool success = loadedFiles > 0;
158  if (convertToModels)
159  {
160  const CAircraftModelList vPilotModels(rules.toAircraftModels()); // long lasting operation
161  this->ps_setCache(vPilotModels);
162  }
163 
164  emit readFinished(success);
165  return success;
166  }

214  bool CVPilotRulesReader::loadFile(const QString &fileName, CVPilotModelRuleSet &ruleSet)
215  {
216  QFile f(fileName);
217  if (!f.exists()) { return false; }
218  if (!f.open(QFile::ReadOnly | QFile::Text)) { return false; }
219  QByteArray fc = f.readAll();
220  if (fc.isEmpty()) { return false; }
221  QDomDocument doc;
222  if (!doc.setContent(fc)) { return false; }
223  QDomNodeList rules = doc.elementsByTagName("ModelMatchRule");
224  if (rules.isEmpty()) { return false; }
225 
226  QDomNodeList mmRuleSet = doc.elementsByTagName("ModelMatchRuleSet");
227  if (mmRuleSet.size() < 1) { return true; }
228 
229  const QDomNamedNodeMap attributes = mmRuleSet.at(0).attributes();
230  QString folder = attributes.namedItem("Folder").nodeValue().trimmed();
231  if (folder.isEmpty())
232  {
233  folder = QFileInfo(fileName).fileName().replace(".vmr", "");
234  }
235 
236  // "2/1/2014 12:00:00 AM", "5/26/2014 2:00:00 PM"
237  const QString updated = attributes.namedItem("UpdatedOn").nodeValue();
238  QDateTime qt = fromStringUtc(updated, "M/d/yyyy h:mm:ss AP");
239  qint64 updatedTimestamp = qt.toMSecsSinceEpoch();
240 
241  int rulesSize = rules.size();
242  for (int i = 0; i < rulesSize; i++)
243  {
244  const QDomNamedNodeMap ruleAttributes = rules.at(i).attributes();
245  const QString typeCode = ruleAttributes.namedItem("TypeCode").nodeValue();
246  const QString modelName = ruleAttributes.namedItem("ModelName").nodeValue();
247  // remark, callsign prefix is airline ICAO code
248  const QString callsignPrefix = ruleAttributes.namedItem("CallsignPrefix").nodeValue();
249  if (modelName.isEmpty()) { continue; }
250 
251  // split if we have multiple models
252  if (modelName.contains("//"))
253  {
254  // multiple models
255  const QStringList models = modelName.split("//");
256  for (const QString &model : models)
257  {
258  if (model.isEmpty()) { continue; }
259  CVPilotModelRule rule(model, folder, typeCode, callsignPrefix, updatedTimestamp);
260  ruleSet.push_back(rule);
261  }
262  }
263  else
264  {
265  // single model
266  CVPilotModelRule rule(modelName, folder, typeCode, callsignPrefix, updatedTimestamp);
267  ruleSet.push_back(rule);
268  }
269  }
270  return true;
271  }
