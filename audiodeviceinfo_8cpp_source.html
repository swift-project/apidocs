103  int CAudioDeviceInfo::comparePropertyByIndex(CPropertyIndexRef index, const CAudioDeviceInfo &compareValue) const
104  {
105  if (index.isMyself()) { return m_deviceName.compare(compareValue.m_deviceName, Qt::CaseInsensitive); }
106  const ColumnIndex i = index.frontCasted<ColumnIndex>();
107  switch (i)
108  {
109  case IndexDeviceTypeAsString:
110  case IndexDeviceType: return Compare::compare(m_type, compareValue.m_type);
111  case IndexName: return m_deviceName.compare(compareValue.m_deviceName, Qt::CaseInsensitive);
112  case IndexIdentifier: return m_identifier.comparePropertyByIndex(index.copyFrontRemoved(), compareValue.getIdentifier());
113  default: break;
114  }
115  BLACK_VERIFY_X(false, Q_FUNC_INFO, qUtf8Printable("No comparison for index " + index.toQString()));
