75  void CDbLiverySelectorComponent::setLivery(const QString &code)
76  {
77  if (!sGui || sGui->isShuttingDown() || !sGui->hasWebDataServices()) { return; }
78  const int dbKey = CDatastoreUtility::extractIntegerKey(code);
79  CLivery livery;
80 
81  if (dbKey >= 0)
82  {
83  livery = sGui->getWebDataServices()->getLiveryForDbKey(dbKey);
84  }
85 
86  if (!livery.hasValidDbKey())
87  {
88  const QString liveryCode = this->stripExtraInfo(code.toUpper().trimmed());
89  if (m_currentLivery.matchesCombinedCode(liveryCode)) { return; }
90  livery = sGui->getWebDataServices()->getLiveries().findByCombinedCode(liveryCode);
91  }
92 
93  if (livery.hasCompleteData())
94  {
95  this->setLivery(livery);
96  }
97  else
98  {
99  ui->lbl_Description->setText("");
100  ui->le_Livery->setText(code);
101  }
102  }

176  void CDbLiverySelectorComponent::dropEvent(QDropEvent *event)
177  {
178  if (!event || !acceptDrop(event->mimeData())) { return; }
179  const CVariant valueVariant(toCVariant(event->mimeData()));
180  if (valueVariant.isValid())
181  {
182  if (valueVariant.canConvert<CLivery>())
183  {
184  const CLivery livery(valueVariant.value<CLivery>());
185  if (!livery.hasValidDbKey()) { return; }
186  this->setLivery(livery);
187  }
188  else if (valueVariant.canConvert<CLiveryList>())
189  {
190  const CLiveryList liveries(valueVariant.value<CLiveryList>());
191  if (liveries.isEmpty()) { return; }
192  this->setLivery(liveries.front());
193  }
194  }
195  }

197  void CDbLiverySelectorComponent::onLiveriesRead(CEntityFlags::Entity entity, CEntityFlags::ReadState readState, int count, const QUrl &url)
198  {
199  Q_UNUSED(url)
200 
201  if (!sGui || sGui->isShuttingDown() || !sGui->hasWebDataServices()) { return; }
202  if (entity.testFlag(CEntityFlags::LiveryEntity) && CEntityFlags::isFinishedReadState(readState))
203  {
204  if (count > 0)
205  {
206  const QStringList codes(sGui->getWebDataServices()->getLiveries().getCombinedCodesPlusInfoAndId(true));
207  QCompleter *c = new QCompleter(codes, this);
208  c->setCaseSensitivity(Qt::CaseInsensitive);
209  c->setCompletionMode(QCompleter::PopupCompletion);
210  c->setMaxVisibleItems(10);
211  connect(c, qOverload<const QString &>(&QCompleter::activated), this, &CDbLiverySelectorComponent::onCompleterActivated);
212 
213  ui->le_Livery->setCompleter(c);
214  m_completerLiveries.reset(c); // deletes any old completer
215  }
216  else
217  {
218  m_completerLiveries.reset(nullptr);
219  }
220  }

235  QString CDbLiverySelectorComponent::stripExtraInfo(const QString &liveryCode) const
236  {
237  if (liveryCode.isEmpty()) { return {}; }
238  const QString l(liveryCode.trimmed().toUpper());
239  int is = l.indexOf(' ');
240  int ib = l.indexOf('(');
241  int i = qMin(is, ib);
242  if (i < 0) { return l; }
243  return l.left(i);
244  }
