173  void CContextOwnAircraft::evaluateUpdateHistory()
174  {
175  if (!m_history) { return; }
176  if (!this->getIContextSimulator()) { return; }
177  if (!this->getIContextSimulator()->isSimulatorSimulating()) { return; }
178 
179  QReadLocker rl(&m_lockAircraft);
180  const CAircraftSituationList situations = m_situationHistory; // latest first
181  rl.unlock();
182 
183  constexpr int minElements = qRound(MaxHistoryElements * 0.75);
184  if (m_situationHistory.size() < minElements) { return; }
185 
186  // using copy to minimize lock time
187  // 500km/h => 1sec: 0.1388 km
188  // we check if there are situation for own aircraft outside the max.distance
189  static const CLength maxDistance(25, CLengthUnit::km());
190  const CAircraftSituation latest = situations.front();
191  const bool jumpDetected = situations.containsObjectOutsideRange(situations.front(), maxDistance);
192 
193  if (jumpDetected)
194  {
195  const CAircraftSituationList ownDistances = situations.findFarthest(1, latest);
196  const CLength distance = ownDistances.front().calculateGreatCircleDistance(latest);
197  {
198  QWriteLocker wl(&m_lockAircraft);
199  m_situationHistory.clear();
200  }
201  emit this->movedAircraft(distance);
202  }
203  else
204  {
205  const bool to = situations.isTakingOff(true);
206  if (to)
207  {
208  emit this->isTakingOff();
209  }
210  else
211  {
212  const bool td = situations.isTouchingDown(true);
213  if (td) { emit this->isTouchingDown(); }
214  }
215  }
216  }

245  void CContextOwnAircraft::evaluateComStations(bool atcChanged)
246  {
247  if (!sApp || sApp->isShuttingDown() || !sApp->getCContextAudioBase() || !sApp->getIContextNetwork()) { return; }
248 
249  CComSystem com1;
250  CComSystem com2;
251  CComSystem lastCom1;
252  CComSystem lastCom2;
253  {
254  QReadLocker l(&m_lockAircraft);
255  com1 = m_ownAircraft.getCom1System();
256  com2 = m_ownAircraft.getCom2System();
257  lastCom1 = m_lastEvaluatedCom1;
258  lastCom2 = m_lastEvaluatedCom2;
259  }
260 
261  const bool changedCom1Freq = (lastCom1.getFrequencyActive() != com1.getFrequencyActive());
262  const bool changedCom2Freq = (lastCom2.getFrequencyActive() != com2.getFrequencyActive());
263 
264  if (!atcChanged && !changedCom1Freq && !changedCom2Freq) { return; }
265 
266  const CAtcStationList atcs = sApp->getIContextNetwork()->getAtcStationsOnline(true).findInRange();
267  const bool atcCom1 = atcs.hasComUnitTunedInChannelSpacing(com1);
268  const bool atcCom2 = atcs.hasComUnitTunedInChannelSpacing(com2);
269 
270  // remember if I was tuned in, abusing the flag
271  com1.setReceiveEnabled(atcCom1);
272  com2.setReceiveEnabled(atcCom2);
273  QWriteLocker l(&m_lockAircraft);
274  m_lastEvaluatedCom1 = com1;
275  m_lastEvaluatedCom2 = com1;
276  }

278  bool CContextOwnAircraft::updateOwnSituation(const CAircraftSituation &situation)
279  {
280  QWriteLocker l(&m_lockAircraft);
281  // there is intentionally no equal check
282  m_ownAircraft.setSituation(situation);
283 
284  if (m_situationHistory.isEmpty() || qAbs(situation.getTimeDifferenceMs(m_situationHistory.front())) > MinHistoryDeltaMs)
285  {
286  m_situationHistory.push_frontKeepLatestAdjustedFirst(situation, true);
287  if (m_situationHistory.size() > MaxHistoryElements) { m_situationHistory.pop_back(); }

498  bool CContextOwnAircraft::parseCommandLine(const QString &commandLine, const CIdentifier &originator)
499  {
500  Q_UNUSED(originator)
501  if (commandLine.isEmpty()) { return false; }
502  CSimpleCommandParser parser(
503  { ".x", ".xpdr", // transponder
504  ".com1", ".com2", // com1, com2 frequencies
505  ".c1", ".c2", // com1, com2 frequencies
506  ".selcal" });
507  parser.parse(commandLine);
508  if (!parser.isKnownCommand()) { return false; }
509 
510  CSimulatedAircraft myAircraft(this->getOwnAircraft());
511  if (parser.matchesCommand(".x", ".xpdr") && parser.countParts() > 1)
512  {
513  CTransponder transponder = myAircraft.getTransponder();
514  int xprCode = parser.toInt(1);
515  if (CTransponder::isValidTransponderCode(xprCode))
516  {
517  transponder.setTransponderCode(xprCode);
518  // todo RW: replace originator
519  this->updateCockpit(myAircraft.getCom1System(), myAircraft.getCom2System(), transponder, CIdentifier("commandline"));
520  return true;
521  }
522  else
523  {
524  CTransponder::TransponderMode mode = CTransponder::modeFromString(parser.part(1));
525  transponder.setTransponderMode(mode);
526  // todo RW: replace originator
527  this->updateCockpit(myAircraft.getCom1System(), myAircraft.getCom2System(), transponder, CIdentifier("commandline"));
528  return true;
529  }
530  }
531  else if (parser.commandStartsWith("com") || parser.commandStartsWith("c"))
532  {
533  CFrequency frequency(parser.toDouble(1), CFrequencyUnit::MHz());
534  if (CComSystem::isValidComFrequency(frequency))
535  {
536  CComSystem com1 = myAircraft.getCom1System();
537  CComSystem com2 = myAircraft.getCom2System();
538  if (parser.commandEndsWith("1"))
539  {
540  com1.setFrequencyActive(frequency);
541  }
542  else if (parser.commandEndsWith("2"))
543  {
544  com2.setFrequencyActive(frequency);
545  }
546  else
547  {
548  return false;
549  }
550  this->updateCockpit(com1, com2, myAircraft.getTransponder(), identifier());
551  return true;
552  }
553  }
554  else if (parser.matchesCommand(".selcal"))
555  {
556  if (CSelcal::isValidCode(parser.part(1)))
557  {
558  this->updateSelcal(parser.part(1), this->identifier());
559  return true;
560  }
561  }
562  return false;
563  }
