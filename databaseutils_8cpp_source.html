56  CAircraftModel CDatabaseUtils::consolidateModelWithDbData(const CAircraftModel &model, const CAircraftModel &dbModel, bool force, bool *modified)
57  {
58  Q_ASSERT_X(sApp, Q_FUNC_INFO, "Missing application object");
59  Q_ASSERT_X(sApp->hasWebDataServices(), Q_FUNC_INFO, "No web services");
60 
61  if (modified) { *modified = false; }
62  if (!model.hasModelString()) { return model; }
63  if (!force && model.hasValidDbKey()) { return model; }
64  const int distributorOrder = model.getDistributorOrder(); // later restore that order
65 
66  if (dbModel.isLoadedFromDb())
67  {
68  // take the db model as original
69  if (modified) { *modified = true; }
70  CAircraftModel dbModelModified(dbModel);
71  dbModelModified.updateMissingParts(model);
72  dbModelModified.setDistributorOrder(distributorOrder);
73  dbModelModified.setSimulator(dbModel.getSimulator()); // DB simulator settings have priority
74  return dbModelModified;
75  }
76 
77  // we try our best to update by DB data here
78  // since we have no(!) DB model, we update each of it subobjects
79  CAircraftModel consolidatedModel(model); // copy over
80  if (!consolidatedModel.getLivery().hasValidDbKey())
81  {
82  const CLivery dbLivery(sApp->getWebDataServices()->smartLiverySelector(consolidatedModel.getLivery()));
83  if (dbLivery.hasValidDbKey())
84  {
85  if (modified) { *modified = true; }
86  consolidatedModel.setLivery(dbLivery);
87  }
88  }
89  if (!consolidatedModel.getAircraftIcaoCode().hasValidDbKey() && consolidatedModel.hasAircraftDesignator())
90  {
91  // try to find DB aircraft ICAO here
92  const CAircraftIcaoCode dbIcao(sApp->getWebDataServices()->smartAircraftIcaoSelector(consolidatedModel.getAircraftIcaoCode()));
93  if (dbIcao.hasValidDbKey())
94  {
95  if (modified) { *modified = true; }
96  consolidatedModel.setAircraftIcaoCode(dbIcao);
97  }
98  }
99 
100  const CDistributor dbDistributor(sApp->getWebDataServices()->getDistributors().smartDistributorSelector(model.getDistributor(), model));
101  if (dbDistributor.isLoadedFromDb())
102  {
103  if (modified) { *modified = true; }
104  consolidatedModel.setDistributor(dbDistributor);
105  }
106  consolidatedModel.updateLocalFileNames(model);
107  consolidatedModel.setDistributorOrder(distributorOrder);
108  return consolidatedModel;
109  }

155  CAircraftModelList CDatabaseUtils::consolidateModelsWithSimulatorModelsAllowsGuiRefresh(const CAircraftModelList &models, const CAircraftModelList &simulatorModels, QStringList &removedModelStrings, bool processEvents)
156  {
157  if (models.isEmpty() || simulatorModels.isEmpty()) { return models; }
158 
159  QElapsedTimer timer;
160  timer.start();
161  const QSet<QString> allOwnModelsModelStrings = simulatorModels.getModelStringSet();
162  CAircraftModelList consolidatedModels;
163  removedModelStrings.clear();
164 
165  int c = 0;
166  for (const CAircraftModel &model : models)
167  {
168  c++;
169  if (processEvents && c % 125 == 0)
170  {
171  if (!sApp || sApp->isShuttingDown()) { return models; }
172  sApp->processEventsFor(25);
173  }
174 
175  const QString ms(model.getModelString());
176  if (ms.isEmpty()) { continue; }
177  if (!allOwnModelsModelStrings.contains(ms))
178  {
179  removedModelStrings.push_back(ms);
180  continue;
181  }
182  CAircraftModel consolidated = simulatorModels.findFirstByModelStringAliasOrDefault(ms);
183  if (consolidated.hasModelString())
184  {
185  CDatabaseUtils::consolidateModelWithDbData(consolidated, true);
186  consolidatedModels.push_back(consolidated);
187  }
188  else
189  {
190  consolidatedModels.push_back(model);
191  }
192  }
193  CLogMessage(static_cast<CDatabaseUtils *>(nullptr)).info(u"Consolidated %1 vs. %2 in %3 ms") << models.size() << simulatorModels.size() << timer.elapsed() << "ms";
194  return consolidatedModels;
195  }

338  CAircraftModelList CDatabaseUtils::updateSimulatorForFsFamily(const CAircraftModelList &ownModels, CStatusMessageList *updateInfo, int maxToStash, IProgressIndicator *progressIndicator, bool processEvents)
339  {
340  if (!sApp || !sApp->getWebDataServices() || sApp->isShuttingDown()) { return CAircraftModelList(); }
341  const CAircraftModelList dbFsFamilyModels(sApp->getWebDataServices()->getModels().findFsFamilyModels());
342  CAircraftModelList stashModels;
343  if (dbFsFamilyModels.isEmpty() || ownModels.isEmpty()) { return stashModels; }
344  const QSet<QString> dbKeys = dbFsFamilyModels.getModelStringSet();
345  const int maxModelsCount = (maxToStash >= 0) ? maxToStash : ownModels.size();
346  if (maxModelsCount < 1) { return stashModels; }
347 
348  int c = 0; // counter
349  for (const CAircraftModel &ownModel : ownModels)
350  {
351  c++;
352 
353  // process events
354  if (processEvents && c % 500 == 0)
355  {
356  if (progressIndicator)
357  {
358  const int percentage = c * 100 / maxModelsCount;
359  progressIndicator->updateProgressIndicatorAndProcessEvents(percentage);
360  }
361  else
362  {
363  sApp->processEventsFor(10);
364  }
365  }
366 
367  // values to be skipped
368  if (maxToStash >= 0 && stashModels.size() > maxToStash) { break; }
369  if (!dbKeys.contains(ownModel.getModelString())) { continue; }
370  if (ownModel.matchesSimulatorFlag(CSimulatorInfo::XPLANE)) { continue; }
371 
372  // in DB
373  CAircraftModel dbModel = dbFsFamilyModels.findFirstByModelStringAliasOrDefault(ownModel.getModelString());
374  if (!dbModel.isLoadedFromDb()) { continue; }
375 
376  // update simulator and add
377  CSimulatorInfo simulator(dbModel.getSimulator());
378  const CSimulatorInfo dbModelSimulator(simulator);
379  const CSimulatorInfo delta = simulator.add(ownModel.getSimulator());
380  if (delta.getSimulator() == CSimulatorInfo::None) { continue; }
381 
382  dbModel.setSimulator(simulator);
383  stashModels.push_back(dbModel); // changed DB model
384  if (updateInfo)
385  {
386  const CStatusMessage m = CStatusMessage(static_cast<CDatabaseUtils *>(nullptr)).info(u"%1 -> %2 (%3) for '%4'")
387  << dbModelSimulator.toQString(true) << simulator.toQString(true)
388  << delta.toQString(true) << dbModel.getModelStringAndDbKey();
389  updateInfo->push_back(m);
390  }
391  }
392  progressIndicator->updateProgressIndicatorAndProcessEvents(100);
393  return stashModels;
394  }

396  QJsonDocument CDatabaseUtils::databaseJsonToQJsonDocument(const QString &content)
397  {
398  static const QString compressed("swift:");
399  if (content.isEmpty()) { return QJsonDocument(); }
400  QByteArray byteData;
401  if (Json::looksLikeJson(content))
402  {
403  // uncompressed
404  byteData = content.toUtf8();
405  }
406  else if (content.startsWith(compressed) && content.length() > compressed.length() + 3)
407  {
408  do
409  {
410  // "swift:1234:base64encoded
411  const int cl = compressed.length();
412  const int contentIndex = content.indexOf(':', cl);
413  if (contentIndex < cl) break; // should not happen, malformed
414  const QString ls = content.mid(cl, contentIndex - cl); // content length
415  bool ok;
416  const qint32 size = ls.toInt(&ok);
417  if (!ok) break; // malformed size
418  if (size < 1) break;
419 
420  QByteArray ba;
421  ba.append(content.mid(contentIndex));
422  ba = QByteArray::fromBase64(ba);
423  ba.insert(0, CCompressUtils::lengthHeader(size)); // adding 4 bytes length header
424  byteData = qUncompress(ba);
425  }
426  while (false);
427  }
428 
429  if (byteData.isEmpty()) { return QJsonDocument(); }
430  return QJsonDocument::fromJson(byteData);
431  }
