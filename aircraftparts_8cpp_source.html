119  void CAircraftParts::setPropertyByIndex(CPropertyIndexRef index, const QVariant &variant)
120  {
121  if (index.isMyself())
122  {
123  (*this) = variant.value<CAircraftParts>();
124  return;
125  }
126  if (ITimestampWithOffsetBased::canHandleIndex(index))
127  {
128  ITimestampWithOffsetBased::setPropertyByIndex(index, variant);
129  return;
130  }
131 
132  const ColumnIndex i = index.frontCasted<ColumnIndex>();
133  switch (i)
134  {
135  case IndexEngines: m_engines = variant.value<decltype(m_engines)>(); break;
136  case IndexFlapsPercentage: m_flapsPercentage = variant.toInt(); break;
137  case IndexGearDown: m_gearDown = variant.toBool(); break;
138  case IndexLights: m_lights.setPropertyByIndex(index.copyFrontRemoved(), variant); break;
139  case IndexSpoilersOut: m_spoilersOut = variant.toBool(); break;
140  case IndexIsOnGround: m_isOnGround = variant.toBool(); break;
141  default: CValueObject::setPropertyByIndex(index, variant); break;
142  }

145  int CAircraftParts::comparePropertyByIndex(CPropertyIndexRef index, const CAircraftParts &compareValue) const
146  {
147  if (index.isMyself()) { return ITimestampWithOffsetBased::comparePropertyByIndex(CPropertyIndex(), compareValue); }
148  if (ITimestampWithOffsetBased::canHandleIndex(index)) { return ITimestampWithOffsetBased::comparePropertyByIndex(index, compareValue); }
149 
150  const ColumnIndex i = index.frontCasted<ColumnIndex>();
151  switch (i)
152  {
153  case IndexEngines: return Compare::compare(this->getEnginesCount(), compareValue.getEnginesCount());
154  case IndexFlapsPercentage: return Compare::compare(m_flapsPercentage, compareValue.getFlapsPercent());
155  case IndexGearDown: return Compare::compare(m_gearDown, compareValue.isGearDown());
156  case IndexSpoilersOut: return Compare::compare(m_spoilersOut, compareValue.isSpoilersOut());
157  case IndexIsOnGround: return Compare::compare(m_isOnGround, compareValue.isOnGround());
158  case IndexLights: return m_lights.comparePropertyByIndex(index.copyFrontRemoved(), compareValue.getLights());
159  default: break;
160  }
161  Q_ASSERT_X(false, Q_FUNC_INFO, "No comparison");
162  return 0;
163  }
