141  CAirlineIcaoCode CAirlineIcaoCodeList::smartAirlineIcaoSelector(const CAirlineIcaoCode &icaoPattern, const CCallsign &callsign) const
142  {
143  if (icaoPattern.hasValidDbKey()) { return icaoPattern; }
144  const CAirlineIcaoCode patternUsed = icaoPattern.thisOrCallsignCode(callsign);
145 
146  // search by parts
147  CAirlineIcaoCodeList codesFound;
148  if (patternUsed.hasValidDesignator())
149  {
150  if (patternUsed.isVirtualAirline())
151  {
152  // we can tell for sure we search an VA
153  codesFound = this->findByVDesignator(patternUsed.getVDesignator());
154  }
155  else
156  {
157  // we do not know if we are looking for an VA
158  codesFound = this->findByDesignator(patternUsed.getDesignator());
159  }
160  }
161  else
162  {
163  codesFound = this->findByIataCode(patternUsed.getIataCode());
164  }
165 
166  if (codesFound.size() == 1) { return codesFound.front(); }
167  if (codesFound.isEmpty())
168  {
169  // nothing found so far
170  codesFound = this->findByNamesOrTelephonyDesignator(patternUsed.getName());
171  codesFound = codesFound.ifPossibleReduceByTelephonyDesignator(patternUsed.getTelephonyDesignator());
172  codesFound = codesFound.ifPossibleReduceByCountry(patternUsed.getCountryIso());
173  }
174  else
175  {
176  // further reduce
177  bool reduced = false;
178  codesFound = codesFound.ifPossibleReduceNameTelephonyCountry(callsign, patternUsed.getName(), patternUsed.getTelephonyDesignator(), patternUsed.getCountryIso(), reduced, QString(), nullptr);
179  }
180 
181  return codesFound.frontOrDefault();
182  }

184  CAirlineIcaoCodeList CAirlineIcaoCodeList::ifPossibleReduceNameTelephonyCountry(const CCallsign &cs, const QString &airlineName, const QString &telephony, const QString &countryIso, bool &reduced, const QString &loginfo, CStatusMessageList *log) const
185  {
186  reduced = false;
187  if (this->isEmpty())
188  {
189  if (log) { CCallsign::addLogDetailsToList(log, cs, loginfo % u" Empty input list, cannot reduce", getLogCategories()); }
190  return *this;
191  }
192 
193  if (telephony.isEmpty() && airlineName.isEmpty() && countryIso.isEmpty())
194  {
195  if (log) { CCallsign::addLogDetailsToList(log, cs, loginfo % u" No name/telephony/country, cannot reduce " % QString::number(this->size()) % u" entries", getLogCategories()); }
196  return *this;
197  }
198 
199  CAirlineIcaoCodeList step1Data = airlineName.isEmpty() ? *this : this->findByNamesOrTelephonyDesignator(airlineName);
200  if (step1Data.isEmpty() || step1Data.size() == this->size())
201  {
202  if (log) { CCallsign::addLogDetailsToList(log, cs, loginfo % QStringLiteral(" cannot reduce by '%1'").arg(airlineName), getLogCategories()); }
203  step1Data = *this;
204  }
205  else
206  {
207  reduced = true;
208  if (log) { CCallsign::addLogDetailsToList(log, cs, loginfo % QStringLiteral(" reduced by '%1'").arg(airlineName), getLogCategories()); }
209  }
210  if (step1Data.size() == 1) { return step1Data; }
211 
212  CAirlineIcaoCodeList step2Data = telephony.isEmpty() ? step1Data : step1Data.findByNamesOrTelephonyDesignator(telephony);
213  if (step2Data.isEmpty() || step2Data.size() == this->size())
214  {
215  if (log) { CCallsign::addLogDetailsToList(log, cs, loginfo % QStringLiteral(" cannot reduce by name '%1'").arg(telephony), getLogCategories()); }
216  step2Data = step1Data;
217  }
218  else
219  {
220  reduced = true;
221  if (log) { CCallsign::addLogDetailsToList(log, cs, loginfo % QStringLiteral(" reduced by telephony '%1'").arg(telephony), getLogCategories()); }
222  }
223  if (step2Data.size() == 1) { return step2Data; }
224 
225  CAirlineIcaoCodeList step3Data = countryIso.isEmpty() ? step2Data : step2Data.findByCountryIsoCode(countryIso);
226  if (step3Data.isEmpty() || step3Data.size() == this->size())
227  {
228  if (log) { CCallsign::addLogDetailsToList(log, cs, loginfo % QStringLiteral(" cannot reduce by country '%1'").arg(countryIso), getLogCategories()); }
229  step3Data = step2Data;
230  }
231  else
232  {
233  reduced = true;
234  if (log) { CCallsign::addLogDetailsToList(log, cs, loginfo % QStringLiteral(" reduced by '%1'").arg(countryIso), getLogCategories()); }
235  }
236  return step3Data;
237  }

257  CAirlineIcaoCode CAirlineIcaoCodeList::findBestMatchByCallsign(const CCallsign &callsign) const
258  {
259  if (this->isEmpty() || callsign.isEmpty()) { return CAirlineIcaoCode(); }
260  const QString airline = callsign.getAirlinePrefix().toUpper();
261  if (airline.isEmpty()) { return CAirlineIcaoCode(); }

263  this->findFirstByOrDefault(&CAirlineIcaoCode::getDesignator, airline) :

268  CAirlineIcaoCodeList CAirlineIcaoCodeList::fromDatabaseJson(const QJsonArray &array, bool ignoreIncomplete, CAirlineIcaoCodeList *inconsistent)
269  {
270  CAirlineIcaoCodeList codes;
271  for (const QJsonValue &value : array)
272  {
273  const CAirlineIcaoCode icao(CAirlineIcaoCode::fromDatabaseJson(value.toObject()));
274  const bool incomplete = !icao.hasCompleteData();
275  if (incomplete)
276  {
277  if (ignoreIncomplete) { continue; }
278  if (inconsistent)
279  {
280  inconsistent->push_back(icao);
281  continue;
282  }
283  }
284  codes.push_back(icao);
285  }
286  return codes;
