120  int CAircraftIcaoCode::calculateScore(const CAircraftIcaoCode &otherCode, CStatusMessageList *log) const
121  {
122  if (this->isDbEqual(otherCode))
123  {
124  addLogDetailsToList(log, *this, QString("Equal DB code: 100"));
125  return 100;
126  }
127 
128  int score = 0;
129  if (this->hasValidDesignator() && this->getDesignator() == otherCode.getDesignator())
130  {
131  // 0..65
132  score += 50; // same designator
133  if (log)
134  {
135  addLogDetailsToList(log, *this, QStringLiteral("Same designator: %1").arg(score));
136  }
137 
138  int scoreOld = score;
139  if (this->getRank() == 0) { score += 15; }
140  else if (this->getRank() == 1) { score += 12; }
141  else if (this->getRank() < 10) { score += (10 - this->getRank()); }
142  if (score > scoreOld)
143  {
144  addLogDetailsToList(log, *this, QStringLiteral("Added rank: %1").arg(score));
145  }
146  }
147  else
148  {
149  if (this->hasFamily() && this->getFamily() == otherCode.getFamily())
150  {
151  score += 40;
152  addLogDetailsToList(log, *this, QStringLiteral("Added family: %1").arg(score));
153  }
154  else if (this->hasValidCombinedType() && otherCode.getCombinedType() == this->getCombinedType())
155  {
156  score += 30;
157  addLogDetailsToList(log, *this, QStringLiteral("Added combined code: %1").arg(score));
158  }
159  else if (this->hasValidCombinedType())
160  {
161  // totally 15
162 
163  // engine count
164  const int eMy = this->getEnginesCount();
165  const int eOther = otherCode.getEnginesCount();
166 
167  if (eMy == eOther && eMy >= 0)
168  {
169  score += 4;
170  }
171  else if (eMy > 0 && eOther > 0)
172  {
173  const int eDiff = qAbs(eMy - eOther);
174  if (eDiff == 1) { score += 2; }
175  else if (eDiff == 2) { score += 1; }
176  }
177 
178  // engine type
179  const QString tMy = this->getEngineType();
180  const QString tOther = this->getEngineType();
181 
182  if (tMy == tOther)
183  {
184  score += 4;
185  }
186  else if (!tMy.isEmpty() && !tOther.isEmpty())
187  {
188  if (isEPTEngineType(tMy[0]) && isEPTEngineType(tOther[0])) { score += 2; }
189  }
190 
191  // aircraft type
192  if (this->getAircraftType() == otherCode.getAircraftType())
193  {
194  score += 7;
195  }
196  else if (this->isVtol() && otherCode.isVtol())
197  {
198  score += 4;
199  }
200  addLogDetailsToList(log, *this, QStringLiteral("Added combined code parts: %1").arg(score));
201  }
202  }
203 
204  // 0..65 so far
205  if (this->hasManufacturer() && otherCode.hasManufacturer())
206  {
207  if (this->matchesManufacturer(otherCode.getManufacturer()))
208  {
209  score += 10;
210  addLogDetailsToList(log, *this, QStringLiteral("Matches manufacturer '%1': %2").arg(this->getManufacturer()).arg(score));
211  }
212  else if (this->getManufacturer().contains(otherCode.getManufacturer(), Qt::CaseInsensitive))
213  {
214  addLogDetailsToList(log, *this, QStringLiteral("Contains manufacturer '%1': %2").arg(this->getManufacturer()).arg(score));
215  score += 5;
216  }
217  }
218 
219  // 0..75 so far
220  if (this->hasCategory() && otherCode.hasCategory() && this->getCategory() == otherCode.getCategory())
221  {
222  score += 8;
223  addLogDetailsToList(log, *this, QStringLiteral("Matches military flag '%1': %2").arg(boolToYesNo(this->isMilitary())).arg(score));
224  }
225  else if (this->isMilitary() == otherCode.isMilitary())
226  {
227  score += 8;
228  addLogDetailsToList(log, *this, QStringLiteral("Matches military flag '%1': %2").arg(boolToYesNo(this->isMilitary())).arg(score));
229  }
230  // 0..85
231  return score;
232  }

234  void CAircraftIcaoCode::guessModelParameters(CLength &guessedCGOut, CSpeed &guessedVRotateOut) const
235  {
236  // we do not override values
237  if (!guessedCGOut.isNull() && !guessedVRotateOut.isNull()) { return; }
238 
239  // init to defaults
240  CLength guessedCG = CLength(1.5, CLengthUnit::m());
241  CSpeed guessedVRotate = CSpeed(70, CSpeedUnit::km_h());
242 
243  const int engines = this->getEnginesCount();
244  const QChar engineType = this->getEngineTypeChar().toUpper();
245  do
246  {
247  if (engines == 1)
248  {
249  if (engineType == 'T')
250  {
251  guessedCG = CLength(2.0, CLengthUnit::m());
252  break;
253  }
254  }
255  else if (engines == 2)
256  {
257  guessedCG = CLength(2.0, CLengthUnit::m());
258  guessedVRotate = CSpeed(100, CSpeedUnit::kts());
259  if (engineType == 'T')
260  {
261  guessedCG = CLength(2.0, CLengthUnit::m());
262  break;
263  }
264  if (engineType == 'J')
265  {
266  // a B737 has VR 105-160kts
267  guessedVRotate = CSpeed(120, CSpeedUnit::kts());
268  guessedCG = CLength(2.5, CLengthUnit::m());
269  break;
270  }
271  }
272  else if (engines > 2)
273  {
274  guessedCG = CLength(4.0, CLengthUnit::m());
275  guessedVRotate = CSpeed(70, CSpeedUnit::kts());
276  if (engineType == 'J')
277  {
278  // A typical B747 has VR around 160kts
279  guessedCG = CLength(6.0, CLengthUnit::m());
280  guessedVRotate = CSpeed(140, CSpeedUnit::kts());
281  break;
282  }
283  }
284 
285  if (engineType == 'J')
286  {
287  // MIL Jets a bit faster
288  if (this->isMilitary()) { guessedVRotate *= 1.20; }
289  else if (this->matchesDesignator("CONC")) { guessedVRotate = CSpeed(199, CSpeedUnit::kts()); }
290  }
291 
292  // VTOL
293  if (this->isVtol()) { guessedVRotate = CSpeed(0, CSpeedUnit::kts()); }
294  }
295  while (false);
296 
297  if (guessedCGOut.isNull()) { guessedCGOut = guessedCG; }
298  if (guessedVRotateOut.isNull()) { guessedVRotateOut = guessedVRotate; }
299  }

423  bool CAircraftIcaoCode::matchesCombinedType(const QString &combinedType) const
424  {
425  const QString cc(combinedType.toUpper().trimmed().replace(' ', '*').replace('-', '*'));
426  if (combinedType.length() != 3) { return false; }
427  if (cc == this->getCombinedType()) { return true; }
428 
429  const bool wildcard = cc.contains('*');
430  if (!wildcard) { return false; }
431  const QChar at = cc.at(0);

433  const QChar et = cc.at(2);
434  if (at != '*')
435  {

442  }
443  if (et == '*') { return true; }
444  const QString cet = this->getEngineType();
445  return cet.length() == 1 && cet.at(0) == et;

584  bool CAircraftIcaoCode::matchesDesignator(const QString &designator, int fuzzyMatch, int *result) const
585  {
586  Q_ASSERT_X(fuzzyMatch >= -1 && fuzzyMatch <= 100, Q_FUNC_INFO, "fuzzyMatch range 0..100 or -1");
587  if (designator.isEmpty()) { return false; }
588  const QString d = designator.trimmed().toUpper();
589  if (fuzzyMatch >= 0)
590  {
591  const int r = fuzzyShortStringComparision(this->getDesignator(), d) >= fuzzyMatch;
592  if (result) { *result = r; }
593  return (r >= fuzzyMatch);
594  }
595  else
596  {
