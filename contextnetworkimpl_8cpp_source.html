53  CContextNetwork::CContextNetwork(CCoreFacadeConfig::ContextMode mode, CCoreFacade *runtime) : IContextNetwork(mode, runtime)
54  {
56  Q_ASSERT(this->getRuntime());
57  Q_ASSERT(this->getIContextOwnAircraft());
58  Q_ASSERT(this->getIContextOwnAircraft()->isUsingImplementingObject());
59  CContextNetwork::registerHelp();
60 
61  // 1. Init by "network driver"
62  m_fsdClient = new CFSDClient(
63  this, // client provider
64  this->getRuntime()->getCContextOwnAircraft(), // own aircraft provider
65  this, // remote aircraft provider
66  this); // thread owner
67  m_fsdClient->start(); // FSD thread
68  connect(m_fsdClient, &CFSDClient::connectionStatusChanged, this, &CContextNetwork::onFsdConnectionStatusChanged, Qt::QueuedConnection);
69  connect(m_fsdClient, &CFSDClient::killRequestReceived, this, &CContextNetwork::kicked, Qt::QueuedConnection);
70  connect(m_fsdClient, &CFSDClient::textMessagesReceived, this, &CContextNetwork::onTextMessagesReceived, Qt::QueuedConnection);
71  connect(m_fsdClient, &CFSDClient::textMessageSent, this, &CContextNetwork::onTextMessageSent, Qt::QueuedConnection);
72  connect(m_fsdClient, &CFSDClient::severeNetworkError, this, &CContextNetwork::severeNetworkError, Qt::QueuedConnection);
73 
74  // 2. Update timer for data (network data such as frequency)
75  // we use 2 timers so we can query at different times (not too many queirs at once)
76  m_requestAircraftDataTimer = new QTimer(this);
77  connect(m_requestAircraftDataTimer, &QTimer::timeout, this, &CContextNetwork::requestAircraftDataUpdates);
78  m_requestAircraftDataTimer->start(30 * 1000);
79  m_requestAircraftDataTimer->setObjectName("CContextNetwork::m_requestAircraftDataTimer");
80 
81  m_requestAtisTimer = new QTimer(this);
82  connect(m_requestAtisTimer, &QTimer::timeout, this, &CContextNetwork::requestAtisUpdates);
83  m_requestAtisTimer->start(13 * 1000); // should not be called at the same time as above
84  m_requestAtisTimer->setObjectName("CContextNetwork::m_requestAtisTimer");
85 
86  // 3. send staggered model matching signals, to avoid to many matchings at the same time
87  m_staggeredMatchingTimer = new QTimer(this);
88  connect(m_staggeredMatchingTimer, &QTimer::timeout, this, &CContextNetwork::emitReadyForMatching);
89  m_staggeredMatchingTimer->start(200);
90  m_staggeredMatchingTimer->setObjectName("CContextNetwork::m_staggeredMatchingTimer");
91 
92  // 4. Airspace contents
93  Q_ASSERT_X(this->getRuntime()->getCContextOwnAircraft(), Q_FUNC_INFO, "this and own aircraft context must be local");
94  Q_ASSERT_X(this->getRuntime()->getCContextSimulator(), Q_FUNC_INFO, "this and own simulator context must be local");
95  m_airspace = new CAirspaceMonitor(
96  this->getRuntime()->getCContextOwnAircraft(),
97  this->getRuntime()->getCContextSimulator(),
98  m_fsdClient, this);
99  m_fsdClient->setClientProvider(m_airspace);
100  connect(m_airspace, &CAirspaceMonitor::changedAtcStationOnlineConnectionStatus, this, &CContextNetwork::changedAtcStationOnlineConnectionStatus, Qt::QueuedConnection);
101  connect(m_airspace, &CAirspaceMonitor::changedAtcStationsOnline, this, &CContextNetwork::changedAtcStationsOnline, Qt::QueuedConnection);
102  connect(m_airspace, &CAirspaceMonitor::changedAtcStationsBooked, this, &CContextNetwork::changedAtcStationsBooked, Qt::QueuedConnection);
103  connect(m_airspace, &CAirspaceMonitor::changedAircraftInRange, this, &CContextNetwork::changedAircraftInRange, Qt::QueuedConnection);
104  connect(m_airspace, &CAirspaceMonitor::removedAircraft, this, &IContextNetwork::removedAircraft, Qt::QueuedConnection); // DBus
105  connect(m_airspace, &CAirspaceMonitor::readyForModelMatching, this, &CContextNetwork::onReadyForModelMatching); // intentionally NOT QueuedConnection
106  connect(m_airspace, &CAirspaceMonitor::addedAircraft, this, &CContextNetwork::addedAircraft, Qt::QueuedConnection);
107  connect(m_airspace, &CAirspaceMonitor::changedAtisReceived, this, &CContextNetwork::onChangedAtisReceived, Qt::QueuedConnection);
108  }

211  void CContextNetwork::gracefulShutdown()
212  {
213  this->disconnect(); // all signals
214  if (this->isConnected()) { this->disconnectFromNetwork(); }
215  if (m_fsdClient)
216  {
217  m_fsdClient->gracefulShutdown();
218  m_fsdClient->setClientProvider(nullptr);
219  m_fsdClient->deleteLater();

221  }
222 
223  if (m_airspace)
224  {
225  m_airspace->gracefulShutdown();
226  m_airspace->deleteLater();
227  m_airspace = nullptr;
228  }

231  CStatusMessage CContextNetwork::connectToNetwork(const CServer &server, const QString &extraLiveryString, bool sendLivery, const QString &extraModelString, bool sendModelString, const CCallsign &partnerCallsign, CLoginMode mode)
232  {
233  if (!this->canUseFsd()) { return { CStatusMessage({ CLogCategories::validation() }, CStatusMessage::SeverityInfo, u"Invalid FSD state (shutdown)") }; }
234  if (this->isDebugEnabled()) { CLogMessage(this, CLogCategories::contextSlot()).debug() << Q_FUNC_INFO; }
235 
236  QString msg;
237  if (!server.getUser().hasCredentials()) { return CStatusMessage({ CLogCategories::validation() }, CStatusMessage::SeverityError, u"Invalid user credentials"); }
238  if (!this->ownAircraft().getAircraftIcaoCode().hasDesignator()) { return CStatusMessage({ CLogCategories::validation() }, CStatusMessage::SeverityError, u"Invalid ICAO data for own aircraft"); }
239  if (!CNetworkUtils::canConnect(server, msg, 5000)) { return CStatusMessage(CStatusMessage::SeverityError, msg); }
240  if (m_fsdClient->isConnected()) { return CStatusMessage({ CLogCategories::validation() }, CStatusMessage::SeverityError, u"Already connected"); }
241  if (this->isPendingConnection()) { return CStatusMessage({ CLogCategories::validation() }, CStatusMessage::SeverityError, u"Pending connection, please wait"); }
242 
243  this->getIContextOwnAircraft()->updateOwnAircraftPilot(server.getUser());
244  const CSimulatedAircraft ownAircraft(this->ownAircraft());
245  m_fsdClient->setPartnerCallsign(isValidPartnerCallsign(ownAircraft.getCallsign(), partnerCallsign) ? partnerCallsign : CCallsign());
246 
247  // Fall back to observer mode, if no simulator is available or not simulating
248  if (!CBuildConfig::isLocalDeveloperDebugBuild() && !this->getIContextSimulator()->isSimulatorSimulating())
249  {
250  CLogMessage(this).info(u"No simulator connected or connected simulator not simulating. Falling back to observer mode");
251  mode.setLoginMode(CLoginMode::Observer);
252  }
253 
254  const CSimulatorInfo sim = this->getIContextSimulator() ? this->getIContextSimulator()->getSimulatorPluginInfo().getSimulatorInfo() : CSimulatorInfo();
255  const QString l = extraLiveryString.isEmpty() ? ownAircraft.getModel().getSwiftLiveryString(sim) : extraLiveryString;
256  const QString m = extraModelString.isEmpty() ? ownAircraft.getModelString() : extraModelString;
257 
258  // FG fix, do not send livery and model ids for FlightGear
259  // https://discordapp.com/channels/539048679160676382/567091362030419981/698124094482415616
260  if (sim.isFG() && extraModelString.isEmpty())
261  {
262  sendModelString = false;
263  }
264 
265  m_currentMode = mode;
266  m_fsdClient->setLoginMode(mode);
267  m_fsdClient->setCallsign(ownAircraft.getCallsign());
268 
269  // set this BEFORE model string as FG has different handling
270  m_fsdClient->setSimType(sim);
271 
272  m_fsdClient->setIcaoCodes(ownAircraft);
273  m_fsdClient->setLiveryAndModelString(l, sendLivery, m, sendModelString);
274  m_fsdClient->setClientName(sApp->swiftVersionChar());
275  m_fsdClient->setVersion(CBuildConfig::getVersion().majorVersion(), CBuildConfig::getVersion().minorVersion());
276 
277  int clientId = 0;
278  QString clientKey;
279  if (!getCmdLineClientIdAndKey(clientId, clientKey))
280  {
281  clientId = CBuildConfig::vatsimClientId();
282  clientKey = CBuildConfig::vatsimPrivateKey();
283  }
284 
285  m_fsdClient->setClientIdAndKey(static_cast<quint16>(clientId), clientKey.toLocal8Bit());
286  m_fsdClient->setClientCapabilities(Capabilities::AircraftInfo | Capabilities::FastPos | Capabilities::VisPos | Capabilities::AtcInfo | Capabilities::AircraftConfig);
287  m_fsdClient->setServer(server);
288 
289  m_fsdClient->setPilotRating(PilotRating::Student);
290  m_fsdClient->setAtcRating(AtcRating::Observer);
291 
292  m_fsdClient->connectToServer();
293  return CStatusMessage({ CLogCategories::validation() }, CStatusMessage::SeverityInfo, u"Connection pending " % server.getAddress() % u' ' % QString::number(server.getPort()));
294  }

337  bool CContextNetwork::parseCommandLine(const QString &commandLine, const CIdentifier &originator)
338  {
339  Q_UNUSED(originator;)
340 
341  if (!this->canUseAirspaceMonitor()) { return false; }
342  if (!this->canUseFsd()) { return false; }
343  if (commandLine.isEmpty()) { return false; }
344 
345  static const QStringList cmds({ ".msg", ".m", ".chat", ".altos", ".altoffset", ".addtimeos", ".addtimeoffset", ".wallop", ".watchdog", ".reinit", ".reinitialize", ".enable", ".disable", ".ignore", ".unignore", ".fsd" });
346  CSimpleCommandParser parser(cmds);
347  parser.parse(commandLine);
348  if (!parser.isKnownCommand()) { return false; }
349  if (parser.matchesCommand(".msg", ".m", ".chat"))
350  {
351  if (!this->getIContextNetwork() || !this->getIContextNetwork()->isConnected())
352  {
353  CLogMessage(this).validationError(u"Network needs to be connected");
354  return false;
355  }
356  else if (!this->getIContextOwnAircraft())
357  {
358  CLogMessage(this).validationError(u"No own aircraft data, no text message can be sent");
359  return false;
360  }
361  if (parser.countParts() < 3)
362  {
363  CLogMessage(this).validationError(u"Incorrect message");
364  return false;
365  }
366 
367  // set receiver
368  const CSimulatedAircraft ownAircraft(this->getIContextOwnAircraft()->getOwnAircraft());
369  const QString receiver = parser.part(1).trimmed().toLower(); // receiver
370  CCallsign ownCallsign = ownAircraft.getCallsign();
371  if (m_fsdClient)
372  {
373  // override with the preset callsign, as the own callsign can be different for partner callsign scenarios
374  // copilot scenarios
375  const CCallsign presetCallsign = m_fsdClient->getPresetCallsign();
376  if (!presetCallsign.isEmpty()) { ownCallsign = presetCallsign; }
377  }
378 
379  if (ownCallsign.isEmpty())
380  {
381  CLogMessage(this).validationError(u"No own callsign");
382  return false;
383  }
384 
385  CTextMessage tm;
386  tm.setSenderCallsign(ownCallsign);
387 
388  // based on the CPZ bug https://discordapp.com/channels/539048679160676382/539486309882789888/576765888401768449
389  // no longer use starts/ends with
390  if (receiver == QStringView(u"c1") || receiver == QStringView(u"com1") || receiver == QStringView(u"comm1"))
391  {
392  tm.setFrequency(ownAircraft.getCom1System().getFrequencyActive());
393  }
394  else if (receiver == QStringView(u"c2") || receiver == QStringView(u"com2") || receiver == QStringView(u"comm2"))
395  {
396  tm.setFrequency(ownAircraft.getCom2System().getFrequencyActive());
397  }
398  else if (receiver == "u" || receiver == QStringView(u"uni") || receiver == QStringView(u"unicom"))
399  {
400  tm.setFrequency(CPhysicalQuantitiesConstants::FrequencyUnicom());
401  }
402  else
403  {
404  const CFrequency radioFrequency = CComSystem::parseComFrequency(receiver, CPqString::SeparatorBestGuess);
405  if (!radioFrequency.isNull())
406  {
407  if (CComSystem::isValidCivilAviationFrequency(radioFrequency))
408  {
409  tm.setFrequency(radioFrequency);
410  }
411  else
412  {
413  CLogMessage(this).validationError(u"Wrong COM frequency for text message");
414  return false;
415  }
416  }
417  else
418  {
419  const CCallsign toCallsign(receiver);
420  tm.setRecipientCallsign(toCallsign);
421  }
422  }
423 
424  const QString msg(parser.partAndRemainingStringAfter(2));
425  tm.setMessage(msg);
426  if (tm.isEmpty())
427  {
428  CLogMessage(this).validationError(u"No text message body");
429  return false;
430  }
431  CTextMessageList tml(tm);
432  this->sendTextMessages(tml);
433  return true;
434  }
435  else if (parser.matchesCommand(".altos", ".altoffset"))
436  {
437  if (!m_airspace) { return false; }
438  if (parser.countParts() < 2) { return false; }
439 
440  const QString csPart(parser.part(1));
441  CCallsign cs;
442  if (csPart.contains('?'))
443  {
444  cs = IRemoteAircraftProvider::testAltitudeOffsetCallsign(); // wildcard
445  }
446  else
447  {
448  cs = CCallsign(csPart);
449  if (!m_airspace->isAircraftInRange(cs))
450  {
451  CLogMessage(this).validationError(u"Altitude offset unknown callsign");
452  return false;
453  }
454  }
455 
456  CLength os(CLength::null());
457  if (parser.hasPart(2))
458  {
459  os.parseFromString(parser.part(2), CPqString::SeparatorBestGuess);
460  }
461 
462  const bool added = this->testAddAltitudeOffset(cs, os);
463  if (added) { CLogMessage(this).info(u"Added altitude offset %1 for %2") << os.valueRoundedWithUnit(2) << cs.asString(); }
464  else { CLogMessage(this).info(u"Removed altitude offset %1") << cs.asString(); }
465 
466  return true;
467  }
468  else if (parser.matchesCommand(".addtimeos", ".addtimeoffset"))
469  {
470  if (!m_airspace) { return false; }
471  if (parser.countParts() < 2) { return false; }
472 
473  CTime os(CTime::null());
474  if (parser.hasPart(2))
475  {
476  os.parseFromString(parser.part(2), CPqString::SeparatorBestGuess);
477  }
478 
479  if (!os.isNull() && os.isPositiveWithEpsilonConsidered())
480  {
481  const qint64 ost = os.valueInteger(CTimeUnit::ms());
482  CLogMessage(this).info(u"Added add offset time %1ms") << ost;
483  }
484  else
485  {
486  CLogMessage(this).info(u"Reset add. time offset");
487  }
488  }
489  else if (parser.matchesCommand(".watchdog"))
490  {
491  if (!m_airspace) { return false; }
492  if (parser.countParts() < 2) { return false; }
493 
494  const bool watchdog = parser.toBool(1, true);
495  m_airspace->enableAnalyzer(watchdog);
496  CLogMessage(this).info(u"Enabled watchdog: %1") << boolToYesNo(watchdog);
497  }
498  else if (parser.matchesCommand(".reinit", ".reinitialize"))
499  {
500  if (!m_airspace) { return false; }
501  const int count = m_airspace->reInitializeAllAircraft();
502  if (count > 0)
503  {
504  CLogMessage(this).info(u"Re-init %1 aircraft") << count;
505  }
506  }
507  else if (parser.matchesCommand(".wallop"))
508  {
509  if (parser.countParts() < 2) { return false; }
510  if (!m_fsdClient) { return false; }
511  if (!this->isConnected())
512  {
513  CLogMessage(this).validationError(u"Network needs to be connected");
514  return false;
515  }
516  const QString wallopMsg = parser.partAndRemainingStringAfter(1);
517  if (wallopMsg.isEmpty())
518  {
519  CLogMessage(this).validationError(u"No wallop message body");
520  return false;
521  }
522  m_fsdClient->sendTextMessage(TextMessageGroups::AllSups, wallopMsg);
523  return true;
524  }
525  else if (parser.matchesCommand(".enable", ".unignore"))
526  {
527  if (parser.countParts() < 2) { return false; }
528  if (!m_fsdClient) { return false; }
529  if (!this->isConnected()) { return false; }
530  const CCallsign cs(parser.part(1));
531  if (cs.isValid()) { this->updateAircraftEnabled(cs, true); }
532  }
533  else if (parser.matchesCommand(".disable", ".ignore"))
534  {
535  if (parser.countParts() < 2) { return false; }
536  if (!m_fsdClient) { return false; }
537  if (!this->isConnected()) { return false; }
538  const CCallsign cs(parser.part(1));
539  if (cs.isValid()) { this->updateAircraftEnabled(cs, false); }
540  }
541  else if (m_airspace && parser.matchesCommand(".fsd"))
542  {
543  return m_airspace->parseCommandLine(commandLine, originator);
544  }
545 
546  return false;

762  void CContextNetwork::onTextMessagesReceived(const CTextMessageList &messages)
763  {
764  if (messages.isEmpty()) { return; }
765 
766  const CCallsign partnerCallsign = this->getPartnerCallsign();
767  const CCallsign ownCallsign = this->ownAircraft().getCallsign();
768 
769  CTextMessageList textMessages = messages.withRelayedToPrivateMessages();
770  CTextMessageList partnerMessages;
771 
772  if (!partnerCallsign.isEmpty())
773  {
774  partnerMessages = textMessages.findBySender(partnerCallsign);
775  const CTextMessageList relayedSentMessages = partnerMessages.findByNotForRecipient(ownCallsign).markedAsSent();
776  partnerMessages = partnerMessages.findByRecipient(ownCallsign); // really send to me as PM and not a forwared one
777 
778  // avoid infinite rountrips
779  textMessages = textMessages.withRemovedPrivateMessagesFromCallsign(partnerCallsign);
780 
781  // fake those as sent by myself
782  for (const CTextMessage &rsm : relayedSentMessages)
783  {
784  emit this->textMessageSent(rsm);
785  }
786  }
787 
788  // 1) relayed messaged "now look like PMs"
789  // 2) all messaged of partner are EXCLUDED
790  if (!textMessages.isEmpty()) { emit this->textMessagesReceived(textMessages); }
791  if (!partnerMessages.isEmpty()) { emit this->textMessagesReceived(partnerMessages); }
792 
793  if (textMessages.containsPrivateMessages())
794  {
795  const CTextMessageList supMessages(messages.getSupervisorMessages());
796  for (const CTextMessage &m : supMessages)
797  {
798  emit this->supervisorTextMessageReceived(m);
799  }
800 
801  // part to send to partner, "forward/relay" to partner
802  if (!partnerCallsign.isEmpty())
803  {
804  // IMPORTANT: partner messages already received
805  CTextMessageList relayedMessages;
806  const CTextMessageList privateMessages = messages.getPrivateMessages();
807  for (const CTextMessage &m : privateMessages)
808  {
809  this->createRelayMessageToPartnerCallsign(m, partnerCallsign, relayedMessages);
810  }
811 
812  if (!relayedMessages.isEmpty())
813  {
814  QPointer<CContextNetwork> myself(this);
815  QTimer::singleShot(10, this, [=] {
816  if (myself) { myself->sendTextMessages(relayedMessages); }
817  });
818  }
819  } // relay to partner
820  }
821  }

823  void CContextNetwork::onTextMessageSent(const CTextMessage &message)
824  {
825  if (message.isEmpty()) { return; }
826  if (message.isRelayedMessage()) { return; }
827 
828  if (message.isPrivateMessage())
829  {
830  // forward messages which are NO real PMs tp the partner
831  const CCallsign partnerCallsign = this->getPartnerCallsign();
832  if (!partnerCallsign.isEmpty() && message.getRecipientCallsign() != partnerCallsign)
833  {
834  QPointer<CContextNetwork> myself(this);
835  CTextMessageList relayedMessages;
836  this->createRelayMessageToPartnerCallsign(message, partnerCallsign, relayedMessages);
837  if (!relayedMessages.isEmpty())
838  {
839  QTimer::singleShot(10, this, [=] {
840  if (myself) { myself->sendTextMessages(relayedMessages); }
841  });
842  }
843  }
844  }
845 
846  emit this->textMessageSent(message);
847  }
