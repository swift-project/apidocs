122  void CSettingsSimulatorBasicsComponent::save()
123  {
124  const CSimulatorInfo simulator(ui->comp_SimulatorSelector->getValue());
125  CSimulatorSettings s = this->getSettings(simulator).getGenericSettings();
126  const QString simulatorDir(ui->le_SimulatorDirectory->text().trimmed());
127  const QStringList modelDirs(this->parseModelDirectories());
128  const QStringList excludeDirs(this->parseDirectories(ui->pte_ExcludeDirectories->toPlainText()));
129 
130  const QStringList relativeDirs = CFileUtils::makeDirectoriesRelative(excludeDirs, this->getFileBrowserModelDirectory(), m_fileCaseSensitivity);
131  s.setSimulatorDirectory(simulatorDir);
132  s.setModelDirectories(modelDirs);
133  s.setModelExcludeDirectories(relativeDirs);
134  const CStatusMessageList msgs = m_settings.setAndValidateSettings(s, simulator);
135  if (msgs.isSuccess())
136  {
137  const CStatusMessage m = m_settings.setAndSaveSettings(s, simulator);
138  if (!m.isEmpty()) { CLogMessage::preformatted(m); }
139  if (m.isSuccess())
140  {
141  this->showOverlayHTMLMessage("Saved settings", 5000);
142  }
143  else
144  {
145  this->showOverlayMessage(m);
146  }
147  m_unsavedChanges = m_unsavedChanges && !m.isSuccess(); // reset if success, but only if there were changes
148 
149  // display as it was saved
150  this->displaySettings(simulator);
151  }
152  else
153  {
154  this->showOverlayMessagesOrHTMLMessage(msgs);
155  }
156  }

158  void CSettingsSimulatorBasicsComponent::copyDefaults()
159  {
160  const CSimulatorInfo simulator(ui->comp_SimulatorSelector->getValue());
161  const bool anyValues = this->hasAnyValues();
162  if (anyValues)

164  QMessageBox::StandardButton reply = QMessageBox::question(this, "Override", "Override existing values?", QMessageBox::Yes | QMessageBox::No);

174  m_unsavedChanges = true;
175  }

177  const QStringList md(m_settings.defaultModelDirectories(simulator));
178  if (!md.isEmpty())
179  {
180  this->displayModelDirectories(md);
181  m_unsavedChanges = true;
182  }
183 
184  const QStringList excludes(ss.defaultModelExcludeDirectoryPatterns(simulator));
185  if (!excludes.isEmpty())
186  {
187  this->displayExcludeDirectoryPatterns(excludes);
188  m_unsavedChanges = true;

192  void CSettingsSimulatorBasicsComponent::adjustModelDirectory()
193  {
194  const CSimulatorInfo simulator(ui->comp_SimulatorSelector->getValue());
195  const QString simDir = this->getFileBrowserSimulatorDirectory();
196  CSpecializedSimulatorSettings s = m_settings.getSpecializedSettings(simulator);
197  s.setSimulatorDirectory(simDir);
198 
199  // There is not really a fixed place in the X-Plane install directory where models are put.
200  // We just treat the whole X-Plane directory as model directory and search for models in all subdirectories recursively.
201 
202  const QStringList parsedDirectories = this->parseModelDirectories();
203  const QStringList newDirs = parsedDirectories.size() > 1 ? parsedDirectories : this->removeDirectories(s.getModelDirectoriesFromSimulatorDirectoryOrDefault(), parsedDirectories);
204  this->displayModelDirectories(newDirs);
205  }

248  QStringList CSettingsSimulatorBasicsComponent::parseDirectories(const QString &rawString) const
249  {
250  const QString raw = rawString.trimmed();
251  if (raw.isEmpty()) { return QStringList(); }
252  QStringList dirs;
253  thread_local const QRegularExpression regExp("\n|\r\n|\r");
254  const QStringList rawLines = raw.split(regExp);
255  for (const QString &l : rawLines)
256  {
257  const QString normalized = CFileUtils::normalizeFilePathToQtStandard(l);
258  if (normalized.isEmpty()) { continue; }
259  dirs.push_back(normalized);
260  }
261  dirs = CFileUtils::removeSubDirectories(dirs);
262  return dirs;
263  }

265  QStringList CSettingsSimulatorBasicsComponent::addDirectory(const QString &directory, const QStringList &existingDirs)
266  {
267  const QString d(CFileUtils::normalizeFilePathToQtStandard(directory));
268  QStringList dirs(existingDirs);
269  if (d.isEmpty()) { return existingDirs; }
270  if (!dirs.contains(d, m_fileCaseSensitivity))
271  {
272  dirs.push_back(d);
273  }
274  dirs.removeDuplicates();
275  dirs.sort(m_fileCaseSensitivity);
276  return dirs;
277  }

285  QStringList CSettingsSimulatorBasicsComponent::removeDirectories(const QStringList &removeDirectories, const QStringList &existingDirs)
286  {
287  if (existingDirs.isEmpty() || removeDirectories.isEmpty()) { return existingDirs; }
288  const QStringList rDirs = CFileUtils::fixWindowsUncPaths(removeDirectories);
289  const QStringList eDirs = CFileUtils::fixWindowsUncPaths(existingDirs);
290  QStringList dirs;
291  for (const QString &dir : eDirs)
292  {
293  if (rDirs.contains(dir, m_fileCaseSensitivity)) { continue; }
294  dirs.push_back(dir);
295  }
296  dirs.removeDuplicates();
297  dirs.sort(m_fileCaseSensitivity);
298  return dirs;
299  }
