27  void CSettingsSimulatorMessagesComponent::save()
28  {
29  CSimulatorMessagesSettings settings;
30  settings.setRelayGloballyEnabled(ui->cb_Messages->isChecked());
31  if (ui->rb_NoTechnicalMessages->isChecked())
32  {
33  settings.disableTechnicalMessages();
34  }
35  else if (ui->rb_ErrorsOnly->isChecked())
36  {
37  settings.setTechnicalLogSeverity(CStatusMessage::SeverityError);
38  }
39  else if (ui->rb_ErrorsAndWarnings->isChecked())
40  {
41  settings.setTechnicalLogSeverity(CStatusMessage::SeverityWarning);
42  }
43  else if (ui->rb_ErrorWarningsInfo->isChecked())
44  {
45  settings.setTechnicalLogSeverity(CStatusMessage::SeverityInfo);
46  }
47 
48  settings.setRelayGloballyEnabled(ui->cb_Messages->isChecked());
49  CSimulatorMessagesSettings::TextMessageType mt = CSimulatorMessagesSettings::NoTextMessages;
50 
51  if (ui->cb_PrivateMessages->isChecked()) { mt |= CSimulatorMessagesSettings::TextMessagePrivate; }
52  if (ui->cb_SupervisorMessages->isChecked()) { mt |= CSimulatorMessagesSettings::TextMessageSupervisor; }
53  if (ui->cb_Com1->isChecked()) { mt |= CSimulatorMessagesSettings::TextMessagesCom1; }
54  if (ui->cb_Com2->isChecked()) { mt |= CSimulatorMessagesSettings::TextMessagesCom2; };
55  settings.setRelayTextMessages(mt);
56 
57  // save
58  this->m_settings.setAndSave(settings);
59  }

61  void CSettingsSimulatorMessagesComponent::load()
62  {
63  const CSimulatorMessagesSettings settings(this->m_settings.get());
64  ui->cb_Messages->setChecked(settings.isRelayGloballyEnabled());
65  if (settings.isRelayInfoMessages())
66  {
67  ui->rb_ErrorWarningsInfo->setChecked(true);
68  }
69  else if (settings.isRelayWarningMessages())
70  {
71  ui->rb_ErrorsAndWarnings->setChecked(true);
72  }
73  else if (settings.isRelayErrorsMessages())
74  {
75  ui->rb_ErrorsOnly->setChecked(true);
76  }
77  else if (!settings.isRelayTechnicalMessages())
78  {
79  ui->rb_NoTechnicalMessages->setChecked(true);
80  }
81 
82  ui->cb_PrivateMessages->setChecked(settings.isRelayPrivateTextMessages());
83  ui->cb_SupervisorMessages->setChecked(settings.isRelaySupervisorTextMessages());
84  ui->cb_Com1->setChecked(settings.isRelayCom1TextMessages());
85  ui->cb_Com2->setChecked(settings.isRelayCom2TextMessages());
