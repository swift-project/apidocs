128  qint64 CUrlLogList::getAverageResponseTime() const
129  {
130  qint64 sum = 0;
131  int c = 0;
132  for (const CUrlLog &rl : *this)
133  {
134  if (rl.isPending()) { continue; }
135  sum += rl.getResponseTimeMs();
136  c++;
137  }

139  return sum / c;
