94  QVariant CAirport::propertyByIndex(BlackMisc::CPropertyIndexRef index) const
95  {

97  const ColumnIndex i = index.frontCasted<ColumnIndex>();
98  switch (i)
99  {
100  case IndexIcao: return m_icao.propertyByIndex(index.copyFrontRemoved());
101  case IndexLocation: return QVariant(m_location);
102  case IndexDescriptiveName: return QVariant(m_descriptiveName);
103  case IndexPosition: return m_position.propertyByIndex(index.copyFrontRemoved());
104  case IndexElevation: return this->getElevation().propertyByIndex(index.copyFrontRemoved());
105  case IndexOperating: return QVariant::fromValue(this->isOperating());
106  default:
107  return (ICoordinateWithRelativePosition::canHandleIndex(index)) ?
108  ICoordinateWithRelativePosition::propertyByIndex(index) :
109  CValueObject::propertyByIndex(index);

137  break;
138  default:

151  int CAirport::comparePropertyByIndex(CPropertyIndexRef index, const CAirport &compareValue) const
152  {
153  if (index.isMyself()) { return m_icao.comparePropertyByIndex(index.copyFrontRemoved(), compareValue.getIcao()); }
154  const ColumnIndex i = index.frontCasted<ColumnIndex>();
155  switch (i)
156  {
157  case IndexIcao: return m_icao.comparePropertyByIndex(index.copyFrontRemoved(), compareValue.getIcao());
158  case IndexLocation: return m_location.compare(compareValue.getLocation(), Qt::CaseInsensitive);
159  case IndexDescriptiveName: return m_descriptiveName.compare(compareValue.getDescriptiveName(), Qt::CaseInsensitive);
160  default:
161  if (ICoordinateWithRelativePosition::canHandleIndex(index))
162  {
163  return ICoordinateWithRelativePosition::comparePropertyByIndex(index, compareValue);
164  }
165  break;
166  }
167  Q_ASSERT_X(false, Q_FUNC_INFO, "Compare failed");
168  return 0;
