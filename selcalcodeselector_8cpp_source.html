59  void CSelcalCodeSelector::setSelcalCode(const QString &selcal)
60  {
61  const QString s = selcal.isEmpty() ? " " : selcal.toUpper().trimmed();
62  if (s.length() != 4) { return; } // still incomplete code
63  if (this->getSelcalCode() == s) { return; } // avoid unintended signals
64  const QString s1 = s.left(2);
65  const QString s2 = s.right(2);
66  if (BlackMisc::Aviation::CSelcal::codePairs().contains(s1))
67  {
68  ui->cb_SelcalPairs1->setCurrentText(s1);
69  }
70  if (BlackMisc::Aviation::CSelcal::codePairs().contains(s2))
71  {
72  ui->cb_SelcalPairs2->setCurrentText(s2);
73  }
74  }
