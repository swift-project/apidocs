27  QStringList InterimPilotDataUpdate::toTokens() const
28  {
29  std::uint32_t pbh;
30  packPBH(m_pitch, m_bank, m_heading, m_onGround, pbh);
31 
32  auto tokens = QStringList {};
33  tokens.push_back(m_sender);
34  tokens.push_back(m_receiver);
35 
36  // SquawkBox used "I", but since we are using a different message format,
37  // we also need to use a different subtype.
38  // VI = vatlib interim
39  tokens.push_back("VI");
40  tokens.push_back(QString::number(m_latitude, 'f', 5));
41  tokens.push_back(QString::number(m_longitude, 'f', 5));
42  tokens.push_back(QString::number(m_altitudeTrue));
43  tokens.push_back(QString::number(m_groundSpeed));
44  tokens.push_back(QString::number(pbh));
45  return tokens;
46  }

48  InterimPilotDataUpdate InterimPilotDataUpdate::fromTokens(const QStringList &tokens)
49  {

56  double pitch = 0.0;
