34  void CAudioAdvancedDistributedComponent::toggleAudioStartStop()
35  {

37  const bool started = sGui->getCContextAudioBase()->isAudioStarted();
38  if (started)
39  {
40  sGui->getCContextAudioBase()->afvClient()->disconnectFromAndStop();

45  if (sGui->getIContextNetwork()->isConnected())
46  {
47  const bool connected = sGui->getCContextAudioBase()->connectAudioWithNetworkCredentials();
48  Q_UNUSED(connected)

50  // one reason for not connecting is NOT using the VATSIM ecosystem
51  }
52  }
53 
54  this->setButtons(2000);
55  }
