79  CHotkeyCombination CHotkeyCombination::getDeltaComparedTo(const CHotkeyCombination &other) const
80  {
81  CHotkeyCombination combination(*this);
82  const CKeyboardKeyList otherKeys = other.getKeyboardKeys();
83  for (const CKeyboardKey &key : otherKeys)
84  {
85  if (containsKeyboardKey(key)) { combination.removeKeyboardKey(key); }
86  }
87 
88  const CJoystickButtonList otherButtons = other.getJoystickButtons();
89  for (const CJoystickButton &button : otherButtons)
90  {
91  if (containsJoystickButton(button)) { combination.removeJoystickButton(button); }
92  }
93  return combination;
94  }
