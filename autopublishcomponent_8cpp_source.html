63  void CAutoPublishComponent::sendToDb()
64  {
65  if (!sGui || sGui->isShuttingDown()) { return; }
66  if (m_data.isEmpty())
67  {
68  this->showOverlayHTMLMessage("No data!", 5000);
69  return;
70  }
71 
72  if (!sGui->hasWebDataServices())
73  {
74  this->showOverlayHTMLMessage("No publishing web service!", 5000);
75  return;
76  }
77 
78  const CAircraftModelList dbModels = sGui->getWebDataServices()->getModels();
79  CStatusMessageList msgs = m_data.analyzeAgainstDBData(dbModels);
80  if (!msgs.hasErrorMessages())
81  {
82  const CStatusMessageList publishMsgs = sGui->getWebDataServices()->asyncAutoPublish(m_data);
83  msgs.push_back(publishMsgs);
84  }
85  this->showOverlayMessages(msgs);
86  }

104  void CAutoPublishComponent::onAutoPublished(bool success, const QString &url, const CStatusMessageList &msgs)
105  {
106  Q_UNUSED(url)
107  Q_UNUSED(success)
108 
109  if (success)
110  {
111  QPointer<CAutoPublishComponent> myself(this);
112  this->showOverlayMessagesWithConfirmation(msgs, true, "Clean up auto publish files?", [=] {
113  if (!myself) { return; }
114  const int timeoutMs = 5000;
115  myself->deleteAllFiles();
116  myself->showOverlayHTMLMessage("Cleaned auto publish files after uploading them to DB", timeoutMs);
117 
118  QTimer::singleShot(timeoutMs * 1.2, this, [=] {
119  if (!myself) { return; }
120  myself->closeParentDialog();
121  });
122  });
123  m_lastAutoPublish.set(QDateTime::currentMSecsSinceEpoch());
124  }
125  else
126  {
127  this->showOverlayMessages(msgs, true);
128  }
