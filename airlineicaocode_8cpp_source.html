359  // here we have 2 possible codes

361  {
362 
363  if (callsignIcao.getDesignator().endsWith(m_designator))
364  {
365  // callsign ICAO is virtual airline of myself, this is more accurate
366  return callsignIcao;

408  int CAirlineIcaoCode::calculateScore(const CAirlineIcaoCode &otherCode, CStatusMessageList *log) const
409  {
410  if (this->isDbEqual(otherCode))
411  {
412  addLogDetailsToList(log, *this, QStringLiteral("DB equal score: 100"));
413  return 100;
414  }
415  const bool bothFromDb = this->isLoadedFromDb() && otherCode.isLoadedFromDb();
416  int score = 0;
417  if (otherCode.hasValidDesignator() && this->getDesignator() == otherCode.getDesignator())
418  {
419  score += 60;
420  addLogDetailsToList(log, *this, QStringLiteral("Same designator: %1").arg(score));
421  }
422 
423  // only for DB values we check VA
424  if (bothFromDb && this->isVirtualAirline() == otherCode.isVirtualAirline())
425  {
426  score += 20;
427  addLogDetailsToList(log, *this, QStringLiteral("VA equality: %1").arg(score));
428  }
429 
430  // consider the various names
431  if (this->hasName() && this->getName() == otherCode.getName())
432  {
433  score += 20;
434  addLogDetailsToList(log, *this, QStringLiteral("Same name '%1': %2").arg(this->getName()).arg(score));
435  }
436  else if (this->hasTelephonyDesignator() && this->getTelephonyDesignator() == otherCode.getTelephonyDesignator())
437  {
438  score += 15;
439  addLogDetailsToList(log, *this, QStringLiteral("Same telephony '%1': %2").arg(this->getTelephonyDesignator()).arg(score));
440  }
441  else if (this->hasSimplifiedName() && this->getSimplifiedName() == otherCode.getSimplifiedName())
442  {
443  score += 10;
444  addLogDetailsToList(log, *this, QStringLiteral("Same simplified name '%1': %2").arg(this->getSimplifiedName()).arg(score));
445  }
446  return score;
447  }

467 
468  QString designator(json.value(prefix % u"designator").toString());

470  {

472  }

474  const QString iata(json.value(prefix % u"iata").toString());
475  const QString telephony(json.value(prefix % u"callsign").toString());
476  const QString name(json.value(prefix % u"name").toString());

481  const int groupId(json.value(prefix % u"groupid").toInt(-1));

483  const bool operating = CDatastoreUtility::dbBoolStringToBool(json.value(prefix % u"operating").toString());
484  const bool military = CDatastoreUtility::dbBoolStringToBool(json.value(prefix % u"military").toString());

500  QSet<int> iconIdsImpl()
501  {
502  QDir dir(CSwiftDirectories::imagesAirlinesDirectory());
503  Q_ASSERT_X(dir.exists(), Q_FUNC_INFO, "image directory missing");
504 
505  CSetBuilder<int> ids;
506  dir.setFilter(QDir::Files | QDir::NoSymLinks);
507  dir.setSorting(QDir::Name);
508  for (const QFileInfo &fileInfo : dir.entryInfoList())
509  {
510  const QString fn(fileInfo.fileName());
511  bool ok = fn.size() > 5;
512  if (!ok) { continue; }
513  BLACK_VERIFY_X(ok, Q_FUNC_INFO, "wrong file name");
514  const int id = fn.leftRef(5).toInt(&ok);
515  BLACK_VERIFY_X(ok, Q_FUNC_INFO, "wrong id format");
516  if (!ok) { continue; }
517  ids.insert(id);
518  }
519  return ids;
520  }
