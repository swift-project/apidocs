113  void CLoginOverviewComponent::toggleNetworkConnection()
114  {
115  if (!sGui || sGui->isShuttingDown()) { return; }
116  if (!sGui->getIContextNetwork() || !sGui->getIContextAudio()) { return; }
117 
118  const bool isConnected = sGui && sGui->getIContextNetwork()->isConnected();
119  m_networkSetup.setAutoLogoff(ui->cb_AutoLogoff->isChecked());
120 
121  CStatusMessage msg;
122  if (!isConnected)
123  {
124  // void
125  }
126  else
127  {
128  // disconnect from network
129  sGui->setExtraWindowTitle("");
130  msg = sGui->getIContextNetwork()->disconnectFromNetwork();
131  }
132 
133  // log message and trigger events
134  msg.addCategories(this);
135  CLogMessage::preformatted(msg);
136  if (msg.isSuccess())
137  {
138  emit loginOrLogoffSuccessful();
139  }
140  else
141  {
142  this->cancel();
143  }
144  }
