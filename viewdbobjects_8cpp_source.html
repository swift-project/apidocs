62  template <class T>
63  int CViewWithDbObjects<T>::selectDbKeys(const QSet<KeyType> &keys)
64  {
65  if (keys.isEmpty()) { return 0; }
66  this->clearSelection();
67  int r = -1;
68  QSet<int> rows;
69  for (const ObjectType &obj : CViewBase<ModelClass>::containerOrFilteredContainer())
70  {
71  r++;
72  if (!obj.hasValidDbKey()) { continue; }
73  if (keys.contains(obj.getDbKey()))
74  {
75  rows.insert(r);
76  }
77  }
78  return this->selectRows(rows);
79  }

91  {
92  if (keys.isEmpty()) { return 0; }
93  if (this->isEmpty()) { return 0; }
94 
95  ContainerType newObjects(this->container());
96  int delta = newObjects.removeObjectsWithKeys(keys);
97  if (delta > 0)
98  {
99  this->updateContainerMaybeAsync(newObjects);
100  }
101  return delta;

136  void COrderableViewWithDbObjects<T>::customMenu(CMenuActions &menuActions)
137  {
138  if (this->m_menus.testFlag(CViewBaseNonTemplate::MenuOrderable) && this->hasSelection())

143  {
144  // predefine menus

146 
147  if (!m_menuActions[0])
148  {
149  m_frame = new QFrame(this);
150  QHBoxLayout *layout = new QHBoxLayout(m_frame);
151  layout->setMargin(2);
152  m_frame->setLayout(layout);
153  m_leOrder = new QLineEdit(m_frame);
154  QLabel *icon = new QLabel(m_frame);
155  icon->setPixmap(menu.getPixmap());
156  layout->addWidget(icon);
157  QLabel *label = new QLabel(m_frame);
158  label->setText("Order:");
159  layout->addWidget(label);
160  layout->addWidget(m_leOrder);
161  m_validator = new QIntValidator(0, maxOrder, this);
162  m_leOrder->setValidator(m_validator);
163  QWidgetAction *orderAction = new QWidgetAction(this);
164  orderAction->setDefaultWidget(m_frame);
165  QObject::connect(m_leOrder, &QLineEdit::returnPressed, this, &COrderableViewWithDbObjects<T>::orderToLineEdit);
166  m_menuActions[0] = orderAction;
167  }
168  }
169 
170  m_validator->setRange(0, maxOrder);
171  m_leOrder->setPlaceholderText("New order 0-" + QString::number(maxOrder));
172 
173  menuActions.addAction(m_menuActions[0], CMenuAction::pathViewOrder());
174  m_menuActions[1] = menuActions.addAction(CIcons::arrowMediumNorth16(), "To top", CMenuAction::pathViewOrder(), { this, &COrderableViewWithDbObjects<T>::orderToTop });
175  m_menuActions[2] = menuActions.addAction(CIcons::arrowMediumSouth16(), "To bottom", CMenuAction::pathViewOrder(), { this, &COrderableViewWithDbObjects<T>::orderToBottom });
176  m_menuActions[3] = menuActions.addAction(CIcons::arrowMediumWest16(), "Freeze current order", CMenuAction::pathViewOrder(), { this, &COrderableViewWithDbObjects<T>::freezeCurrentOrder });
177  }
178  CViewWithDbObjects<T>::customMenu(menuActions);
179  }
