655  template <class Predicate>
656  auto separate(Predicate p) const -> QMap<decltype(p(std::declval<T>())), CSequence>
657  {
658  QMap<decltype(p(std::declval<T>())), CSequence> result;
659  auto copy = *this;
660  std::stable_sort(copy.begin(), copy.end(), [p](const T &a, const T &b) { return p(a) < p(b); });
661  for (auto it = copy.cbegin(); it != copy.cend();)
662  {
663  auto mid = std::adjacent_find(it, copy.cend(), [p](const T &a, const T &b) { return p(a) != p(b); });
664  result.insert(p(*it), makeRange(it, mid));
665  it = mid;
666  }
667  return result;
668  }
