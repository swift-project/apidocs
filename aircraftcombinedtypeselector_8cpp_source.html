36  void CAircraftCombinedTypeSelector::setCombinedType(const QString &combinedCode)
37  {
38  QString engineCount, engineType, aircraftType;
39  const QString cc(combinedCode.trimmed().toUpper().left(3));
40  if (m_cc == cc) { return; }
41  m_cc = cc;
42 
43  if (cc.length() > 0) { aircraftType = cc.at(0); }
44  if (cc.length() > 1) { engineCount = cc.mid(1, 1); }
45  if (cc.length() > 2) { engineType = cc.mid(2, 1); }
46 
47  if (this->getCombinedTypeFromComboBoxes() != cc)
48  {
49  CGuiUtility::setComboBoxValueByStartingString(ui->cb_EngineCount, engineCount, "unspecified");
50  CGuiUtility::setComboBoxValueByStartingString(ui->cb_EngineType, engineType, "unspecified");
51  CGuiUtility::setComboBoxValueByStartingString(ui->cb_Type, aircraftType, "unspecified");
52  }
53 
54  if (ui->le_CombinedType->text() != cc) { ui->le_CombinedType->setText(cc); }
55  emit this->changedCombinedType(cc);
56  }
