105  QString CUrl::getFullUrl(bool withQuery) const
106  {
107  if (m_host.isEmpty()) { return {}; }
108 
109  QString qn(m_host);
110  if (!hasDefaultPort() && hasPort()) { qn = qn.append(":").append(QString::number(m_port)); }
111  if (hasPath()) { qn = qn.append("/").append(m_path).replace("//", "/"); }
112  if (hasQuery() && withQuery) { qn = qn.append("?").append(m_query); }
113  if (hasScheme()) { qn = QString(this->getScheme()).append("://").append(qn); }
114  return qn;
115  }

266  QString CUrl::stripQueryString(const QString &query)
267  {
268  QString q(query.trimmed());
269  if (q.startsWith("?") || q.startsWith("&"))
270  {
271  q = q.mid(1);
272  }
273  if (q.endsWith("?") || q.endsWith("&"))
274  {
275  q = q.left(q.size() - 1);
276  }
277  return q;
278  }
