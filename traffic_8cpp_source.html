167  void CTraffic::switchToFollowPlaneView(const std::string &callsign)
168  {
169  if (CTraffic::ownAircraftString() != callsign && !this->containsCallsign(callsign))
170  {
171  INFO_LOG("Cannot switch to follow " + callsign);

173  }
174

177  /* This is the hotkey callback. First we simulate a joystick press and
178  * release to put us in 'free view 1'. This guarantees that no panels

188  void CTraffic::followNextPlane()
189  {
190  if (m_planesByCallsign.empty() || m_followPlaneViewCallsign.empty()) { return; }
191  auto callsignIt = std::find(m_followPlaneViewSequence.begin(), m_followPlaneViewSequence.end(), m_followPlaneViewCallsign);
192 
193  // If we are not at the end, increase by one
194  if (callsignIt != m_followPlaneViewSequence.end()) { callsignIt++; }
195  // If we were already at the end or reached it now, start from the beginning
196  if (callsignIt == m_followPlaneViewSequence.end()) { callsignIt = m_followPlaneViewSequence.begin(); }
197 
198  m_followPlaneViewCallsign = *callsignIt;
199  }

297  void CTraffic::addPlane(const std::string &callsign, const std::string &modelName, const std::string &aircraftIcao, const std::string &airlineIcao, const std::string &livery)
298  {
299  auto planeIt = m_planesByCallsign.find(callsign);
300  if (planeIt != m_planesByCallsign.end()) { return; }
301 
302  XPMPPlaneID id = nullptr;
303  if (modelName.empty() || m_modelStrings.count(modelName) == 0)
304  {
305  DEBUG_LOG("Model " + modelName + " is unknown, falling back to basic xpmp2 model matching");
306  id = XPMPCreatePlane(aircraftIcao.c_str(), airlineIcao.c_str(), livery.c_str());
307  }
308  else
309  {
310  id = XPMPCreatePlaneWithModelName(m_modelStrings[modelName].c_str(), aircraftIcao.c_str(), airlineIcao.c_str(), livery.c_str());
311  }
312 
313  if (!id)
314  {
315  emitPlaneAddingFailed(callsign);
316  return;
317  }
318 
319  Plane *plane = new Plane(id, callsign, aircraftIcao, airlineIcao, livery, modelName);
320  m_planesByCallsign[callsign] = plane;
321  m_planesById[id] = plane;
322 
323  // Create view menu item
324  CMenuItem planeViewMenuItem = m_followPlaneViewSubMenu.item(callsign, [this, callsign] { switchToFollowPlaneView(callsign); });
325  m_followPlaneViewMenuItems[callsign] = planeViewMenuItem;
326  m_followPlaneViewSequence.push_back(callsign);
327 
328  emitPlaneAdded(callsign);
329  }

331  void CTraffic::removePlane(const std::string &callsign)
332  {
333  auto menuItemIt = m_followPlaneViewMenuItems.find(callsign);
334  if (menuItemIt != m_followPlaneViewMenuItems.end())
335  {
336  m_followPlaneViewSubMenu.removeItem(menuItemIt->second);
337  m_followPlaneViewMenuItems.erase(menuItemIt);
338  }
339 
340  auto planeIt = m_planesByCallsign.find(callsign);
341  if (planeIt == m_planesByCallsign.end()) { return; }
342 
343  m_followPlaneViewSequence.erase(std::remove(m_followPlaneViewSequence.begin(), m_followPlaneViewSequence.end(), callsign),
344  m_followPlaneViewSequence.end());
345 
346  Plane *plane = planeIt->second;
347  m_planesByCallsign.erase(callsign);
348  m_planesById.erase(plane->id);
349  XPMPDestroyPlane(plane->id);
350  delete plane;
351  }

375  void CTraffic::setPlanesPositions(const std::vector<std::string> &callsigns, std::vector<double> latitudesDeg, std::vector<double> longitudesDeg, std::vector<double> altitudesFt,
376  std::vector<double> pitchesDeg, std::vector<double> rollsDeg, std::vector<double> headingsDeg, const std::vector<bool> &onGrounds)
377  {
378  const bool setOnGround = onGrounds.size() == callsigns.size();
379  for (size_t i = 0; i < callsigns.size(); i++)
380  {
381  auto planeIt = m_planesByCallsign.find(callsigns.at(i));
382  if (planeIt == m_planesByCallsign.end()) { continue; }
383 
384  Plane *plane = planeIt->second;
385  if (!plane) { continue; }
386  plane->positions[2].lat = latitudesDeg.at(i);
387  plane->positions[2].lon = longitudesDeg.at(i);
388  plane->positions[2].elevation = altitudesFt.at(i);
389  plane->positions[2].pitch = static_cast<float>(pitchesDeg.at(i));
390  plane->positions[2].roll = static_cast<float>(rollsDeg.at(i));
391  plane->positions[2].heading = static_cast<float>(headingsDeg.at(i));
392  plane->positions[2].offsetScale = 1.0f;
393  plane->positions[2].clampToGround = true;
394  plane->positionTimes[2] = std::chrono::steady_clock::now();
395 
396  // save 2 positions at 1-second intervals for use in interpolation
397  if (plane->positionTimes[2] - plane->positionTimes[1] > 1s)
398  {
399  plane->positionTimes[0] = plane->positionTimes[1];
400  plane->positionTimes[1] = plane->positionTimes[2];
401  std::memcpy(&plane->positions[0], &plane->positions[1], sizeof(plane->positions[0]));
402  std::memcpy(&plane->positions[1], &plane->positions[2], sizeof(plane->positions[0]));
403  }
404 
405  if (setOnGround) { plane->isOnGround = onGrounds.at(i); }
406  }
407  }

409  void CTraffic::setPlanesSurfaces(const std::vector<std::string> &callsigns, const std::vector<double> &gears, const std::vector<double> &flaps, const std::vector<double> &spoilers,
410  const std::vector<double> &speedBrakes, const std::vector<double> &slats, const std::vector<double> &wingSweeps, const std::vector<double> &thrusts,
411  const std::vector<double> &elevators, const std::vector<double> &rudders, const std::vector<double> &ailerons,
412  const std::vector<bool> &landLights, const std::vector<bool> &taxiLights,

414  {

416

427  plane->surfaces.flapRatio = static_cast<float>(flaps.at(i));
428  plane->surfaces.spoilerRatio = static_cast<float>(spoilers.at(i));

434  plane->surfaces.yokeHeading = static_cast<float>(rudders.at(i));

436  if (bundleTaxiLandingLights)

471  void CTraffic::getRemoteAircraftData(std::vector<std::string> &callsigns, std::vector<double> &latitudesDeg, std::vector<double> &longitudesDeg,
472  std::vector<double> &elevationsM, std::vector<bool> &waterFlags, std::vector<double> &verticalOffsets) const
473  {
474  if (callsigns.empty() || m_planesByCallsign.empty()) { return; }
475 
476  const auto requestedCallsigns = callsigns;
477  callsigns.clear();
478  latitudesDeg.clear();
479  longitudesDeg.clear();
480  elevationsM.clear();
481  verticalOffsets.clear();
482  waterFlags.clear();
483 
484  for (const auto &requestedCallsign : requestedCallsigns)
485  {
486  const auto planeIt = m_planesByCallsign.find(requestedCallsign);
487  if (planeIt == m_planesByCallsign.end()) { continue; }
488 
489  const Plane *plane = planeIt->second;
490  assert(plane);
491 
492  const double latDeg = plane->positions[2].lat;
493  const double lonDeg = plane->positions[2].lon;
494  double groundElevation = 0.0;
495  bool isWater = false;
496  if (getSettings().isTerrainProbeEnabled())
497  {
498  // we expect elevation in meters
499  groundElevation = plane->terrainProbe.getElevation(latDeg, lonDeg, plane->positions[2].elevation, requestedCallsign, isWater).front();
500  if (std::isnan(groundElevation)) { groundElevation = 0.0; }
501  }
502 
503  callsigns.push_back(requestedCallsign);
504  latitudesDeg.push_back(latDeg);
505  longitudesDeg.push_back(lonDeg);
506  elevationsM.push_back(groundElevation);
507  waterFlags.push_back(isWater);
508  verticalOffsets.push_back(0); // xpmp2 adjusts the offset for us, so effectively always zero
509  }
510  }

543  DBusHandlerResult CTraffic::dbusMessageHandler(const CDBusMessage &message_)
544  {
545  CDBusMessage message(message_);
546  const std::string sender = message.getSender();
547  const dbus_uint32_t serial = message.getSerial();
548  const bool wantsReply = message.wantsReply();
549 
550  // Debug message if needed
551  // { const std::string d = "dbusMessageHandler: " + message.getMethodName(); INFO_LOG(d.c_str()); }
552 
553  if (message.getInterfaceName() == DBUS_INTERFACE_INTROSPECTABLE)
554  {
555  if (message.getMethodName() == "Introspect")
556  {
557  sendDBusReply(sender, serial, introspection_traffic);
558  }
559  }
560  else if (message.getInterfaceName() == XSWIFTBUS_TRAFFIC_INTERFACENAME)
561  {
562  if (message.getMethodName() == "acquireMultiplayerPlanes")
563  {
564  queueDBusCall([=]() {
565  std::string owner;
566  bool acquired = acquireMultiplayerPlanes(&owner);
567  CDBusMessage reply = CDBusMessage::createReply(sender, serial);
568  reply.beginArgumentWrite();
569  reply.appendArgument(acquired);
570  reply.appendArgument(owner);
571  sendDBusMessage(reply);
572  });
573  }
574  else if (message.getMethodName() == "initialize")
575  {
576  sendDBusReply(sender, serial, initialize());
577  }

843  void CTraffic::interpolatePosition(Plane *plane)
844  {
845  std::memcpy(&plane->positions[3], &plane->positions[2], sizeof(plane->positions[2]));
846 
847  const auto now = std::chrono::steady_clock::now();
848  const auto t1 = plane->positionTimes[2] - plane->positionTimes[0];
849  const auto t2 = now - plane->positionTimes[0];
850 
851  // This interpolation is only intended to smooth over
852  // small errors. Give up if the error is too large.
853  if (t1 > 3s || t2 > 3s) { return; }
854 
855  const double dLat = plane->positions[2].lat - plane->positions[0].lat;
856  const double dLon = plane->positions[2].lon - plane->positions[0].lon;
857  const double dAlt = plane->positions[2].elevation - plane->positions[0].elevation;
858 
859  plane->positions[3].lat = plane->positions[0].lat + dLat * t2 / t1;
860  plane->positions[3].lon = plane->positions[0].lon + dLon * t2 / t1;
861  plane->positions[3].elevation = plane->positions[0].elevation + dAlt * t2 / t1;
862  }

864  void CTraffic::interpolateGear(Plane *plane)
865  {
866  const auto now = std::chrono::steady_clock::now();
867  static const float epsilon = std::numeric_limits<float>::epsilon();
868  const float f = plane->surfaces.gearPosition - plane->targetGearPosition;
869  if (std::abs(f) > epsilon)
870  {
871  constexpr float gearMoveTimeMs = 5000;
872  const auto gearPositionDiffRemaining = plane->targetGearPosition - plane->surfaces.gearPosition;
873 
874  const auto diffMs = std::chrono::duration_cast<std::chrono::milliseconds>(now - plane->prevSurfacesLerpTime);
875  const auto gearPositionDiffThisFrame = (diffMs.count()) / gearMoveTimeMs;
876  plane->surfaces.gearPosition += std::copysign(gearPositionDiffThisFrame, gearPositionDiffRemaining);
877  plane->surfaces.gearPosition = std::max(0.0f, std::min(plane->surfaces.gearPosition, 1.0f));
878  }
879  plane->prevSurfacesLerpTime = now;
880  }

882  void CTraffic::Labels::draw()
883  {
884  static const double maxRangeM = 10000;
885  static const double metersPerFt = 0.3048;
886  std::array<float, 16> worldMat = m_worldMat.getAll();
887  std::array<float, 16> projMat = m_projMat.getAll();
888  double windowWidth = static_cast<double>(m_windowWidth.get());
889  double windowHeight = static_cast<double>(m_windowHeight.get());
890  XPLMCameraPosition_t camPos {};
891  XPLMReadCameraPosition(&camPos);
892 
893  for (const auto &pair : m_traffic->m_planesById)
894  {
895  char *text = const_cast<char *>(pair.second->label);
896  const XPMPPlanePosition_t &planePos = pair.second->positions[3];
897 
898  double worldPos[4] { 0, 0, 0, 1 };
899  double localPos[4] {};
900  double windowPos[4] {};
901  XPLMWorldToLocal(planePos.lat, planePos.lon, planePos.elevation * metersPerFt, &worldPos[0], &worldPos[1], &worldPos[2]);
902  if (distanceSquared(worldPos) > maxRangeM * maxRangeM) { continue; }
903  matrixMultVec(localPos, worldMat.data(), worldPos);
904  matrixMultVec(windowPos, projMat.data(), localPos);
905 
906  windowPos[3] = 1.0 / windowPos[3];
907  windowPos[0] *= windowPos[3];
908  windowPos[1] *= windowPos[3];
909  windowPos[2] *= windowPos[3];
910 
911  if (windowPos[2] < 0.0 || windowPos[2] > 1.0)
912  {
913  continue; // plane is behind camera
914  }
915  XPLMDrawString(m_color.data(),
916  static_cast<int>(std::lround(windowWidth * (windowPos[0] * 0.5 + 0.5))),
917  static_cast<int>(std::lround(windowHeight * (windowPos[1] * 0.5 + 0.5))),
918  text, nullptr, xplmFont_Basic);
919  }

922  void CTraffic::Labels::matrixMultVec(double out[4], const float m[16], const double v[4])
923  {
924  out[0] = v[0] * m[0] + v[1] * m[4] + v[2] * m[8] + v[3] * m[12];
925  out[1] = v[0] * m[1] + v[1] * m[5] + v[2] * m[9] + v[3] * m[13];
926  out[2] = v[0] * m[2] + v[1] * m[6] + v[2] * m[10] + v[3] * m[14];
927  out[3] = v[0] * m[3] + v[1] * m[7] + v[2] * m[11] + v[3] * m[15];

938  int CTraffic::orbitPlaneFunc(XPLMCameraPosition_t *cameraPosition, int isLosingControl, void *refcon)
939  {
940  constexpr bool DEBUG = false;
941  // DEBUG_LOG_C("Follow aircraft entry", DEBUG);
942  if (isLosingControl == 1)
943  {
944  // do NOT use refcon here, might be invalid
945  INFO_LOG("Loosing camera control");
946  return 0;
947  }
948 
949  auto *traffic = static_cast<CTraffic *>(refcon);
950  if (!traffic)
951  {
952  ERROR_LOG("Cannot convert CTraffic object");
953  return 0;
954  }
955 
956  // nothing we can do
957  if (!cameraPosition)
958  {
959  ERROR_LOG("No camera object");
960  traffic->m_followPlaneViewCallsign.clear();
961  return 0;
962  }
963 
964  // Ideally we would like to test against right mouse button, but X-Plane SDK does not
965  // allow that.
966  if (!traffic->m_deltaCameraPosition.isInitialized || traffic->m_isSpacePressed)
967  {
968  int x = 0, y = 0, left = 0, top = 0, right = 0, bottom = 0;
969  // First get the screen size and mouse location. We will use this to decide
970  // what part of the orbit we are in. The mouse will move us up-down and around.
971  // fixme: In a future update, change the orbit only while right mouse button is pressed.
972  XPLMGetScreenBoundsGlobal(&left, &top, &right, &bottom);
973  XPLMGetMouseLocationGlobal(&x, &y);
974  int w = right - left;
975  int h = top - bottom;
976  x -= left;
977  y -= bottom;
978  if (DEBUG) { DEBUG_LOG("Follow aircraft coordinates w,h,x,y: " + std::to_string(w) + " " + std::to_string(h) + " " + std::to_string(x) + " " + std::to_string(y)); } // cppcheck-suppress knownConditionTrueFalse
979  if (traffic->m_lastMouseX == x && traffic->m_lastMouseY == y && traffic->m_lastMouseX >= 0 && traffic->m_lastMouseY >= 0)
980  {
981  // mouse NOT moving, we lost focus or we do NOT move anymore
982  // to avoid issues we reset the space key, see https://discordapp.com/channels/539048679160676382/539925070550794240/614162134644949002
983  traffic->m_isSpacePressed = false;
984  }
985 
986  // avoid follow aircraft in too small windows
987  // int cannot be NaN
988  if (w < 100 || h < 100)
989  {
990  WARNING_LOG("Screen w/h too small " + std::to_string(w) + "/" + std::to_string(h));
991  return 0;
992  }
993 
994  // the 1.25 factor allows to turn around completely
995  traffic->m_deltaCameraPosition.headingDeg = normalizeToZero360Deg(1.25 * 360.0 * static_cast<double>(x) / static_cast<double>(w)); // range 0-360
996  double usedCameraPitchDeg = 60.0 - (60.0 * 2.0 * static_cast<double>(y) / static_cast<double>(h)); // range +-
997 
998  // make sure we can use it with tan in range +-90 degrees and the result of tan not getting too high
999  // we limit to +-85deg, tan 45deg: 1 | tan 60deg: 1.73 | tan 85deg: 11.4
1000  if (usedCameraPitchDeg >= 85.0) { usedCameraPitchDeg = 85.0; }
1001  else if (usedCameraPitchDeg <= -85.0) { usedCameraPitchDeg = -85.0; }
1002  traffic->m_deltaCameraPosition.pitchDeg = usedCameraPitchDeg;
1003 
1004  // Now calculate where the camera should be positioned to be x
1005  // meters from the plane and pointing at the plane at the pitch and
1006  // heading we wanted above.
1007  const double distanceMeterM = traffic->m_followAircraftDistanceMultiplier * static_cast<double>(std::max(10, traffic->getSettings().getFollowAircraftDistanceM()));
1008  static const double PI = std::acos(-1);
1009  traffic->m_deltaCameraPosition.dxMeters = -distanceMeterM * sin(traffic->m_deltaCameraPosition.headingDeg * PI / 180.0);
1010  traffic->m_deltaCameraPosition.dzMeters = distanceMeterM * cos(traffic->m_deltaCameraPosition.headingDeg * PI / 180.0);
1011  traffic->m_deltaCameraPosition.dyMeters = -distanceMeterM * tan(traffic->m_deltaCameraPosition.pitchDeg * PI / 180.0);
1012 
1013  traffic->m_deltaCameraPosition.isInitialized = true;
1014  }
1015 
1016  double lxMeters = 0, lyMeters = 0, lzMeters = 0; // normally init not needed, just to avoid any issues
1017  static const double kFtToMeters = 0.3048;
1018 
1019  std::string modelName = "unknown";
1020  if (traffic->m_followPlaneViewCallsign == CTraffic::ownAircraftString())
1021  {
1022  lxMeters = traffic->m_ownAircraftPositionX.get();
1023  lyMeters = traffic->m_ownAircraftPositionY.get();
1024  lzMeters = traffic->m_ownAircraftPositionZ.get();
1025  modelName = CTraffic::ownAircraftString();
1026  }
1027  else
1028  {
1029  if (traffic->m_planesByCallsign.empty())
1030  {
1031  INFO_LOG("Follow aircraft, no planes to follow");
1032  traffic->m_followPlaneViewCallsign.clear();
1033  return 0;
1034  }
1035 
1036  if (traffic->m_followPlaneViewCallsign.empty())
1037  {
1038  INFO_LOG("Follow aircraft, no callsign to follow");
1039  traffic->m_followPlaneViewCallsign.clear();
1040  traffic->m_followAircraftDistanceMultiplier = 1.0;
1041  return 0;
1042  }
1043 
1044  const auto planeIt = traffic->m_planesByCallsign.find(traffic->m_followPlaneViewCallsign);
1045  if (planeIt == traffic->m_planesByCallsign.end())
1046  {
1047  INFO_LOG("Follow aircraft, no plane found for callsign " + traffic->m_followPlaneViewCallsign);
1048  traffic->m_followPlaneViewCallsign.clear();
1049  traffic->m_followAircraftDistanceMultiplier = 1.0;
1050  return 0;
1051  }
1052 
1053  const Plane *plane = planeIt->second;
1054  if (!plane)
1055  {
1056  ERROR_LOG("Follow aircraft, no plane from iterator for callsign " + traffic->m_followPlaneViewCallsign);
1057  traffic->m_followPlaneViewCallsign.clear();
1058  traffic->m_followAircraftDistanceMultiplier = 1.0;
1059  return 0;
1060  }
1061 
1062  modelName = plane->modelName;
1063  if (!isValidPosition(plane->positions[3]))
1064  {
1065  WARNING_LOG("Invalid follow aircraft position for " + plane->callsign);
1066  WARNING_LOG("Pos: " + pos2String(plane->positions[3]));
1067  return 0;
1068  }
1069 
1070  // avoid underflow of camera into ground
1071  if (plane->isOnGround)
1072  {
1073  if (traffic->m_deltaCameraPosition.dyMeters < 10) { traffic->m_deltaCameraPosition.dyMeters = 10; }
1074  }
1075 
1076  XPLMWorldToLocal(plane->positions[3].lat, plane->positions[3].lon, plane->positions[3].elevation * kFtToMeters, &lxMeters, &lyMeters, &lzMeters);
1077  }
1078 
1079  // Fill out the camera position info.
1080  cameraPosition->x = static_cast<float>(lxMeters + traffic->m_deltaCameraPosition.dxMeters);
1081  cameraPosition->y = static_cast<float>(lyMeters + traffic->m_deltaCameraPosition.dyMeters);
1082  cameraPosition->z = static_cast<float>(lzMeters + traffic->m_deltaCameraPosition.dzMeters);
1083  cameraPosition->pitch = CTraffic::normalizeToPlusMinus180Deg(static_cast<float>(traffic->m_deltaCameraPosition.pitchDeg));
1084  cameraPosition->heading = CTraffic::normalizeToPlusMinus180Deg(static_cast<float>(traffic->m_deltaCameraPosition.headingDeg));
1085  cameraPosition->roll = 0.0;
1086  cameraPosition->zoom = 1.0;
1087 
1088  if (!isValidPosition(cameraPosition))
1089  {
1090  WARNING_LOG("Invalid camera aircraft position");
1091  WARNING_LOG("Pos: " + pos2String(cameraPosition));
1092  traffic->m_followAircraftDistanceMultiplier = 1.0;
1093  return 0;
1094  }
1095 
1096  if (DEBUG) // cppcheck-suppress knownConditionTrueFalse
1097  {
1098  DEBUG_LOG("Camera: " + pos2String(cameraPosition));
1099  DEBUG_LOG("Follow aircraft " + traffic->m_followPlaneViewCallsign + " " + modelName);
1100  }
1101  // Return 1 to indicate we want to keep controlling the camera.
1102  return 1;
1103  }

1105  int CTraffic::followAircraftKeySniffer(char character, XPLMKeyFlags flags, char virtualKey, void *refcon)
1106  {
1107  (void)character;
1108  CTraffic *traffic = static_cast<CTraffic *>(refcon);
1109  if (!traffic || traffic->m_followPlaneViewCallsign.empty()) { return 1; } // totally ignore if nothing is being followed
1110

1115  // for the camera we use the init flag instead, so it is only run once

1117  if (flags & xplm_UpFlag) { traffic->m_isSpacePressed = false; }
1118  }

1121  else if (virtualKey == XPLM_VK_ESCAPE && (flags & xplm_UpFlag)) { traffic->m_followAircraftDistanceMultiplier = 1.0; }
1122 
1123  /* Return 1 to pass the keystroke to plugin windows and X-Plane.
1124  * Returning 0 would consume the keystroke. */
1125  return 1;
