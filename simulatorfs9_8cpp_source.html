42  CAircraftSituation aircraftSituationFromFS9(const MPPositionVelocity &positionVelocity)
43  {
44  double dHigh = positionVelocity.lat_i;
45  double dLow = positionVelocity.lat_f;
46 
47  dLow = dLow / 65536.0;
48  if (dHigh > 0)
49  {
50  dHigh = dHigh + dLow;
51  }
52  else
53  {
54  dHigh = dHigh - dLow;
55  }
56 
57  CCoordinateGeodetic position;
58  position.setLatitude(CLatitude(dHigh * 90.0 / 10001750.0, CAngleUnit::deg()));
59 
60  dHigh = positionVelocity.lon_hi;
61  dLow = positionVelocity.lon_lo;
62 
63  dLow = dLow / 65536.0;
64  if (dHigh > 0)
65  {
66  dHigh = dHigh + dLow;
67  }
68  else
69  {
70  dHigh = dHigh - dLow;
71  }
72 
73  position.setLongitude(CLongitude(dHigh * 360.0 / (65536.0 * 65536.0), CAngleUnit::deg()));
74 
75  dHigh = positionVelocity.alt_i;
76  dLow = positionVelocity.alt_f;
77 
78  dLow = dLow / 65536.0;
79 
80  CAircraftSituation situation;
81  situation.setPosition(position);
82  situation.setAltitude(CAltitude(dHigh + dLow, CAltitude::MeanSeaLevel, CLengthUnit::m()));
83  const double groundSpeed = positionVelocity.ground_velocity / 65536.0;
84  situation.setGroundSpeed(CSpeed(groundSpeed, CSpeedUnit::m_s()));
85 
86  FS_PBH pbhstrct;
87  pbhstrct.pbh = positionVelocity.pbh;
88  int pitch = qRound(std::floor(pbhstrct.pitch / CFs9Sdk::pitchMultiplier()));
89  if (pitch < -90 || pitch > 89) { CLogMessage().warning(u"FS9: Pitch value out of limits: %1") << pitch; }
90  int bank = qRound(std::floor(pbhstrct.bank / CFs9Sdk::bankMultiplier()));
91 
92  // MSFS has inverted pitch and bank angles
93  // ~ and NOT -: https://dev.swift-project.org/rSWIFTPILOTCLIENT7f90652b
94  // The problem with - is that -(-512) is 512 but the maximum is 511 so this overflows back to -512 and causes a visual glitch.
95  pitch = ~pitch;
96  bank = ~bank;
97  situation.setPitch(CAngle(pitch, CAngleUnit::deg()));
98  situation.setBank(CAngle(bank, CAngleUnit::deg()));
99  situation.setHeading(CHeading(pbhstrct.hdg / CFs9Sdk::headingMultiplier(), CHeading::Magnetic, CAngleUnit::deg()));
100 
101  return situation;
102  }

222  // actually those data should be the same as ownAircraft
223  const CComSystem newCom1 = ownAircraft.getCom1System();

237  // avoid changes of cockpit back to old values due to an outdated read back value

320  void CSimulatorFs9::dispatch()
321  {
322  if (m_useFsuipc && m_fsuipc && m_fsuipc->isOpened())
323  {
324  CSimulatedAircraft fsuipcAircraft(getOwnAircraft());
325  const bool ok = m_fsuipc->read(fsuipcAircraft, true, true, true);
326  if (ok)
327  {
328  updateOwnAircraftFromSimulator(fsuipcAircraft);
329  }
330  else
331  {
332  // FSUIPC read error means almost always that FS9 closed. Shutdown the driver.
333  CLogMessage(this).debug() << "Lost connection to FSUIPC. Shutting down.";
334  disconnectFrom();
335  }
336  synchronizeTime();
337  }

372  emitSimulatorCombinedStatus();

375  switch (messageType)
376  {
377  case CFs9Sdk::MULTIPLAYER_PACKET_ID_PARAMS:
378  {
379  MPParam mpParam;
380  MultiPlayerPacketParser::readMessage(message, mpParam);
381  // For debugging:
382  // QTextStream qtstdout(stdout);
383  // QString paramString = getChangedParamsAsString(m_lastParameters, mpParam);
384  // if (! paramString.isEmpty())
385  // {
386  // qtstdout << message.mid(4 * sizeof(qint32)).toHex() << Qt::endl;
387  // qtstdout << paramString << Qt::endl;
388  // }
389  // m_lastParameters = mpParam;
390  break;
391  }
392  case CFs9Sdk::MULTIPLAYER_PACKET_ID_CHANGE_PLAYER_PLANE:
393  {
394  MPChangePlayerPlane mpChangePlayerPlane;
395  MultiPlayerPacketParser::readMessage(message, mpChangePlayerPlane);
396  reverseLookupAndUpdateOwnAircraftModel(mpChangePlayerPlane.aircraft_name);
397  break;
398  }
399  case CFs9Sdk::MULTIPLAYER_PACKET_ID_POSITION_VELOCITY:
400  {
401  break;
402  }
403  case CFs9Sdk::MPCHAT_PACKET_ID_CHAT_TEXT_SEND:
404  {
405  MPChatText mpChatText;
406  MultiPlayerPacketParser::readMessage(message, mpChatText);
407  break;
408  }
409  default: break;
410  }

413  void CSimulatorFs9::updateOwnAircraftFromSimulator(const CSimulatedAircraft &simDataOwnAircraft)
414  {
415  // When I change cockpit values in the sim (from GUI to simulator, not originating from simulator)
416  // it takes a little while before these values are set in the simulator.
417  // To avoid jitters, I wait some update cylces to stabilize the values
418  if (m_skipCockpitUpdateCycles < 1)
419  {
420  const auto dontOverwrite = [](const CFrequency &from, const CFrequency &to) {
421  // don't overwrite 8.33 kHz frequencies
422  return CComSystem::isExclusiveWithin8_33kHzChannel(from) &&
423  from.valueInteger(CFrequencyUnit::kHz()) / 10 == to.valueInteger(CFrequencyUnit::kHz()) / 10;
424  };
425  CComSystem com1 = simDataOwnAircraft.getCom1System();
426  CComSystem com2 = simDataOwnAircraft.getCom2System();
427  const CComSystem oldCom1 = getOwnComSystem(CComSystem::Com1);
428  const CComSystem oldCom2 = getOwnComSystem(CComSystem::Com2);
429  if (dontOverwrite(oldCom1.getFrequencyActive(), com1.getFrequencyActive()))
430  com1.setFrequencyActive(oldCom1.getFrequencyActive());
431  if (dontOverwrite(oldCom1.getFrequencyStandby(), com1.getFrequencyStandby()))
432  com1.setFrequencyStandby(oldCom1.getFrequencyStandby());
433  if (dontOverwrite(oldCom2.getFrequencyActive(), com2.getFrequencyActive()))
434  com2.setFrequencyActive(oldCom2.getFrequencyActive());
435  if (dontOverwrite(oldCom2.getFrequencyStandby(), com2.getFrequencyStandby()))
436  com2.setFrequencyStandby(oldCom2.getFrequencyStandby());
437 
438  this->updateCockpit(com1, com2, simDataOwnAircraft.getTransponder(), this->identifier());
439  }
440  else
441  {
442  --m_skipCockpitUpdateCycles;
443  }
444 
445  const CAircraftSituation aircraftSituation = simDataOwnAircraft.getSituation();
446  this->updateOwnSituationAndGroundElevation(aircraftSituation);
447 
448  this->updateOwnParts(simDataOwnAircraft.getParts());
449 
450  if (m_isWeatherActivated)
451  {
452  if (CWeatherScenario::isRealWeatherScenario(m_weatherScenarioSettings.get()))
453  {
454  if (m_lastWeatherPosition.isNull() ||
455  calculateGreatCircleDistance(m_lastWeatherPosition, aircraftSituation).value(CLengthUnit::mi()) > 20)
456  {
457  m_lastWeatherPosition = aircraftSituation;
458  requestWeatherGrid(aircraftSituation, this->identifier());
459  }
460  }
461  } // weather
462 
463  // slow updates
464  if (m_ownAircraftUpdateCycles % 25 == 0)
465  {
466  this->reverseLookupAndUpdateOwnAircraftModel(simDataOwnAircraft.getModelString());
467  const CLength cg = simDataOwnAircraft.getCG();
468  if (!cg.isNull()) { this->updateOwnCG(cg); }
469  } // slow updates
470 
471  m_ownAircraftUpdateCycles++;
472  }
