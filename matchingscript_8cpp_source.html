237  QString MSModelSet::findCombinedTypeWithClosestColorLivery(const QString &combinedType, const QString &rgbColor) const
238  {
239  if (combinedType.isEmpty() || rgbColor.isEmpty()) { return QString(); }
240  CAircraftModelList models = m_modelSet.findByCombinedTypeWithColorLivery(combinedType);
241  if (models.isEmpty()) { return QString(); }
242  const CRgbColor color(rgbColor);
243  models = models.findClosestFuselageColorDistance(color);
244  return models.isEmpty() ? QString() : models.front().getModelString();
245  }

247  QString MSModelSet::findClosestCombinedTypeWithClosestColorLivery(const QString &combinedType, const QString &rgbColor) const
248  {
249  QString ms = this->findCombinedTypeWithClosestColorLivery(combinedType, rgbColor);
250  if (!ms.isEmpty()) { return ms; }
251  if (combinedType.size() != 3) { return ms; }
252  QString wildCard(combinedType);
253  if (wildCard.size() != 3) { return QString(); }
254  wildCard[1] = '*';
255  return this->findCombinedTypeWithClosestColorLivery(wildCard, rgbColor);
256  }
