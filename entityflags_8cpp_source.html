208  CEntityFlags::Entity CEntityFlags::singleEntityByName(const QString &name)
209  {
210  // order here is crucial
211  if (name.contains("model", Qt::CaseInsensitive)) { return ModelEntity; }
212  if (name.contains("category", Qt::CaseInsensitive)) { return AircraftCategoryEntity; }
213  if (name.contains("aircraft", Qt::CaseInsensitive)) { return AircraftIcaoEntity; }
214  if (name.contains("airline", Qt::CaseInsensitive)) { return AirlineIcaoEntity; }
215  if (name.contains("airport", Qt::CaseInsensitive)) { return AirportEntity; }
216  if (name.contains("distributor", Qt::CaseInsensitive)) { return DistributorEntity; }
217  if (name.contains("countr", Qt::CaseInsensitive)) { return CountryEntity; } // singular/plural
218  if (name.contains("liver", Qt::CaseInsensitive)) { return LiveryEntity; } // singular/plural
219  return NoEntity;
220  }

222  CEntityFlags::Entity CEntityFlags::multipleEntitiesByNames(const QStringList &names)
223  {
224  CEntityFlags::Entity entities = NoEntity;
225  for (const QString &name : names)
226  {
227  const CEntityFlags::Entity singleEntity = CEntityFlags::singleEntityByName(name);
228  if (singleEntity == NoEntity) { continue; }
229  entities |= singleEntity;
230  }
231  return entities;

234  CEntityFlags::EntitySet CEntityFlags::asSingleEntities(Entity entities)
235  {
236  CSetBuilder<CEntityFlags::Entity> s;
