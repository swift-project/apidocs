155  void CAircraftMatcherSetup::reset(CAircraftMatcherSetup::MatchingAlgorithm algorithm)
156  {
157  m_algorithm = static_cast<int>(algorithm);
158  MatchingMode mode = ModeNone;
159  switch (algorithm)
160  {
161  case MatchingStepwiseReduce: mode = ModeDefaultReduce; break;
162  case MatchingScoreBased: mode = ModeDefaultScore; break;
163  case MatchingStepwiseReducePlusScoreBased:
164  default:
165  mode = ModeDefaultReducePlusScore;
166  break;
167  }
168  mode |= ModeDefaultSet; // add set default
169 
170  this->setMatchingMode(mode);
171  this->setPickStrategy(PickByOrder);
172  }

334  CAircraftMatcherSetup::MatchingMode CAircraftMatcherSetup::matchingMode(
335  bool revModelString, bool revLiveryIds,
336  bool byModelString, bool byIcaoDataAircraft1st, bool byIcaoDataAirline1st, bool byFamily, bool byLivery, bool byCombinedType,
337  bool byMilitary, bool byCivilian, bool byVtol,
338  bool byGliderCategory, bool byMilitaryCategory, bool bySmallAircraftCategory,
339  bool scoreIgnoreZeros, bool scorePreferColorLiveries,
340  bool excludeNoDbData, bool excludeNoExcluded,
341  bool modelVerification, bool modelVerificationWarnError,
342  bool modelSetRemoveFailedModel, bool modelFailover)
343  {
344  if (modelFailover) { modelSetRemoveFailedModel = true; } // otherwise this does not make sense
345 
346  MatchingMode mode = byModelString ? ByModelString : ModeNone;
347  if (byIcaoDataAircraft1st) { mode |= ByIcaoOrderAircraftFirst; }
348  if (byIcaoDataAirline1st) { mode |= ByIcaoOrderAirlineFirst; }
349  if (byFamily) { mode |= ByFamily; }
350  if (byLivery) { mode |= ByLivery; }
351  if (byCombinedType) { mode |= ByCombinedType; }
352  if (byMilitary) { mode |= ByMilitary; }
353  if (byCivilian) { mode |= ByCivilian; }
354  if (byVtol) { mode |= ByVtol; }
355  if (byGliderCategory) { mode |= ByCategoryGlider; }
356  if (byMilitaryCategory) { mode |= ByCategoryMilitary; }
357  if (bySmallAircraftCategory) { mode |= ByCategorySmallAircraft; }
358  if (scoreIgnoreZeros) { mode |= ScoreIgnoreZeros; }
359  if (scorePreferColorLiveries) { mode |= ScorePreferColorLiveries; }
360  if (excludeNoDbData) { mode |= ExcludeNoDbData; }
361  if (excludeNoExcluded) { mode |= ExcludeNoExcluded; }
362  if (modelSetRemoveFailedModel) { mode |= ModelSetRemoveFailedModel; }
363  if (modelVerification) { mode |= ModelVerificationAtStartup; }
364  if (modelVerificationWarnError) { mode |= ModelVerificationOnlyWarnError; }
365  if (modelFailover) { mode |= ModelFailoverIfNoModelCanBeAdded; }
366  if (revModelString) { mode |= ReverseLookupModelString; }
367  if (revLiveryIds) { mode |= ReverseLookupSwiftLiveryIds; }
368  return mode;
369  }
