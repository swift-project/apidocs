85  CTransponder::TransponderMode CTransponder::modeFromString(const QString &modeString)
86  {
87  if (modeString.startsWith("I", Qt::CaseInsensitive)) return StateIdent;
88  if (modeString.startsWith("S", Qt::CaseInsensitive)) return StateStandby;
89  if (modeString.startsWith("Mode C", Qt::CaseInsensitive)) return ModeC;
90  if (modeString.startsWith("C", Qt::CaseInsensitive)) return ModeC;
91  if (modeString.startsWith("Mode S", Qt::CaseInsensitive)) return ModeS;
92  if (modeString.contains("1", Qt::CaseInsensitive)) return ModeMil1;
93  if (modeString.contains("2", Qt::CaseInsensitive)) return ModeMil2;
94  if (modeString.contains("3", Qt::CaseInsensitive)) return ModeMil3;
95  if (modeString.contains("4", Qt::CaseInsensitive)) return ModeMil4;
96  if (modeString.contains("5", Qt::CaseInsensitive)) return ModeMil5;
97  return StateStandby;
98  }

135  bool CTransponder::isValidTransponderCode(const QString &transponderCode)
136  {
137  if (transponderCode.isEmpty() || transponderCode.length() > 4) return false;
138  bool number;
139  int tc = transponderCode.toInt(&number);
140  if (!number) return false;
141  if (tc < 0 || tc > 7777) return false;
142  thread_local const QRegularExpression rx("^[0-7]{1,4}$");
143  return rx.match(transponderCode).hasMatch();
144  }
