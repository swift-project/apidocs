87  void CAircraftComponent::update()
88  {
89  if (!sGui || sGui->isShuttingDown()) { return; }
90 
91  // count < 1 checks if view already has been updated
92  if (sGui->getIContextNetwork()->isConnected())
93  {
94  const bool visible = (this->isVisibleWidget() && this->currentWidget() == ui->tb_AircraftInRange);
95  if (this->countAircraftInView() < 1 || visible)
96  {
97  ui->tvp_AircraftInRange->updateContainerMaybeAsync(sGui->getIContextNetwork()->getAircraftInRange());
98  }
99  }
100  if (sGui->getIContextSimulator()->getSimulatorStatus() > 0)
101  {
102  const bool visible = (this->isVisibleWidget() && this->currentWidget() == ui->tb_AirportsInRange);
103  const bool counter = ((m_updateCounter % 5) == 0); // less frequent than aircraft
104  if (this->countAirportsInRangeInView() < 1 || (visible && counter))
105  {
106  ui->tvp_AirportsInRange->updateContainerMaybeAsync(sGui->getIContextSimulator()->getAirportsInRange(true));
107  }
108  }
109 
110  m_updateCounter++;
111  }

126  void CAircraftComponent::onInfoAreaTabBarChanged(int index)
127  {
128  // ignore in those cases
129  if (!this->isVisibleWidget()) return;
130  if (this->isParentDockWidgetFloating()) return;
131  if (!sGui->getIContextNetwork()->isConnected()) return;
132 
133  // here I know I am the selected widget, update, but keep GUI responsive (hence I use a timer)
134  QPointer<CAircraftComponent> myself(this);
135  QTimer::singleShot(1000, this, [=] {
136  if (!myself) { return; }
137  myself->update();
138  });
139  Q_UNUSED(index)
140  }
