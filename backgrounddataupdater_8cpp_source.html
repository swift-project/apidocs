46  void CBackgroundDataUpdater::doWork()
47  {
48  if (!this->doWorkCheck()) { return; }
49  if (m_inWork) { return; }
50  m_inWork = true;
51 
52  emit this->consolidating(true);
53  int cycle = m_cycle;
54  this->addHistory(CLogMessage(this).info(u"Background consolidation cycle %1") << cycle);
55 
56  switch (cycle)
57  {
58  case 0:
59  // normally redundant, will be read in other places as well
60  // new metadata for next comparison
61  this->addHistory(CLogMessage(this).info(u"Triggered info reads from DB"));
62  this->triggerInfoReads();
63  break;
64  case 1:
65  this->addHistory(CLogMessage(this).info(u"Synchronize DB entities"));
66  this->syncDbEntity(CEntityFlags::AirlineIcaoEntity);
67  this->syncDbEntity(CEntityFlags::LiveryEntity);
68  this->syncDbEntity(CEntityFlags::ModelEntity);
69  this->syncDbEntity(CEntityFlags::DistributorEntity);
70  this->syncDbEntity(CEntityFlags::AircraftIcaoEntity);
71  break;
72  case 2:
73  this->addHistory(CLogMessage(this).info(u"Synchronize %1") << this->modelCaches(true).getDescription());
74  this->syncModelOrModelSetCacheWithDbData(true); // set
75  break;
76  case 3:
77  this->addHistory(CLogMessage(this).info(u"Synchronize %1") << this->modelCaches(false).getDescription());
78  this->syncModelOrModelSetCacheWithDbData(false);
79  break;
80  default:
81  break;
82  }
83 
84  ++cycle %= 4;
85  m_cycle = cycle;
86  m_inWork = false;
87  emit this->consolidating(false);
88  }

97  void CBackgroundDataUpdater::syncModelOrModelSetCacheWithDbData(bool modelSetFlag, const CAircraftModelList &dbModelsConsidered)
98  {
99  if (!this->doWorkCheck()) { return; }
100  IMultiSimulatorModelCaches &modelCaches = this->modelCaches(modelSetFlag);
101  const QDateTime latestDbModelsTs = dbModelsConsidered.isEmpty() ?
102  sApp->getWebDataServices()->getCacheTimestamp(CEntityFlags::ModelEntity) :
103  dbModelsConsidered.latestTimestamp();
104  if (!latestDbModelsTs.isValid()) { return; }
105 
106  // newer DB models as cache
107  const QDateTime dbModelsLatestSync = m_syncedModelsLatestChange.value(modelCaches.getDescription());
108  if (dbModelsLatestSync.isValid() && latestDbModelsTs <= dbModelsLatestSync) { return; }
109 
110  const QString description = modelCaches.getDescription();
111  m_syncedModelsLatestChange[description] = latestDbModelsTs;
112  const CSimulatorInfo simulators = modelCaches.simulatorsWithInitializedCache(); // simulators ever used
113  if (simulators.isNoSimulator()) { return; }
114 
115  const CAircraftModelList dbModels = dbModelsConsidered.isEmpty() ?
116  sApp->getWebDataServices()->getModels() :
117  dbModelsConsidered;
118  if (dbModels.isEmpty()) { return; }
119  const QSet<CSimulatorInfo> simulatorsSet = simulators.asSingleSimulatorSet();
120  QElapsedTimer time;
121  for (const CSimulatorInfo &singleSimulator : simulatorsSet)
122  {
123  if (!this->doWorkCheck()) { return; }
124  CAircraftModelList simulatorModels = modelCaches.getSynchronizedCachedModels(singleSimulator);
125  if (simulatorModels.isEmpty()) { continue; }
126  time.restart();
127  const CAircraftModelList dbModelsForSimulator = dbModels.matchesSimulator(singleSimulator);
128  if (dbModelsForSimulator.isEmpty()) { continue; }
129 
130  // time consuming part
131  const int c = CDatabaseUtils::consolidateModelsWithDbData(dbModelsForSimulator, simulatorModels, true);
132  if (c > 0)
133  {
134  const CStatusMessage m = modelCaches.setCachedModels(simulatorModels, singleSimulator);
135  const int msElapsed = time.elapsed();
136  this->addHistory(CLogMessage(this).info(u"Consolidated %1 models (%2) for '%3' in %4ms") << c << description << singleSimulator.convertToQString() << msElapsed);
137  CLogMessage::preformatted(m);
138  this->addHistory(m);
139  }
140  else
141  {
142  this->addHistory(CLogMessage(this).info(u"Synchronized, no changes for '%1'") << singleSimulator.convertToQString());
143  }
144 
145  if (simulatorsSet.size() > 1)
146  {
147  // just give the system some time to relax, consolidation is time consuming
148  if (!this->doWorkCheck()) { return; }
149  CEventLoop eventLoop(this);
150  eventLoop.exec(1000);
151  }
152  }
153  }

155  void CBackgroundDataUpdater::syncDbEntity(CEntityFlags::Entity entity)
156  {
157  if (!this->doWorkCheck()) { return; }
158  const QDateTime latestCacheTs = sApp->getWebDataServices()->getCacheTimestamp(entity);
159  if (!latestCacheTs.isValid()) { return; }
160 
161  const QDateTime latestDbTs = sApp->getWebDataServices()->getLatestDbEntityTimestamp(entity);
162  const QString entityStr = CEntityFlags::flagToString(entity);
163  const QString latestCacheTsStr = latestCacheTs.toString(Qt::ISODate);
164 
165  if (!latestDbTs.isValid()) { return; }
166  if (latestDbTs <= latestCacheTs)
167  {
168  this->addHistory(CLogMessage(this).info(u"Background updater (%1), no auto synchronization with DB, entity '%2', DB ts: %3 cache ts: %4")
169  << CThreadUtils::currentThreadInfo() << entityStr << latestDbTs.toString(Qt::ISODate) << latestCacheTsStr);
170  return;
171  }
172 
173  this->addHistory(CLogMessage(this).info(u"Background updater (%1) triggering read of '%2' since '%3'")
174  << CThreadUtils::currentThreadInfo() << entityStr << latestCacheTsStr);
175  sApp->getWebDataServices()->triggerLoadingDirectlyFromDb(entity, latestCacheTs);
176  }

185  void CBackgroundDataUpdater::onModelsPublished(const CAircraftModelList &modelsPublished, bool directWrite)
186  {
187  if (!this->doWorkCheck()) { return; }
188  if (modelsPublished.isEmpty()) { return; }
189  if (!m_updatePublishedModels) { return; }
190  if (!directWrite) { return; } // those models are CRs and have to be released first
191  if (m_inWork) { return; } // no 2 updates at the same time
192 
193  emit this->consolidating(true);
194  this->syncModelOrModelSetCacheWithDbData(true, modelsPublished);
195  this->syncModelOrModelSetCacheWithDbData(false, modelsPublished);
196  emit this->consolidating(false);
197  }
