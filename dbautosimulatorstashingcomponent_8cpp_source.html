33  void CDbAutoSimulatorStashingComponent::accept()
34  {
35  switch (m_state)
36  {
37  case Running: return;
38  case Completed:
39  {
40  if (!m_modelsToStash.isEmpty())
41  {
42  // this removes previously stashed models
43  this->getMappingComponent()->replaceStashedModelsUnvalidated(m_modelsToStash);
44  const CStatusMessage stashedMsg(this, CStatusMessage::SeverityInfo, QStringLiteral("Stashed %1 models").arg(m_modelsToStash.size()));
45  this->addStatusMessage(stashedMsg);
46  m_modelsToStash.clear();
47  }
48  QDialog::accept();
49  break;
50  }
51  default:
52  {
53  this->tryToStash();
54  break;
55  }
56  }
57  }

126  void CDbAutoSimulatorStashingComponent::tryToStash()
127  {
128  Q_ASSERT_X(this->getMappingComponent(), Q_FUNC_INFO, "Missing mapping component");
129 
130  if (!this->currentModelView()) { return; }
131  m_state = Running;
132  int maxObjectsStashed = -1;
133  if (!ui->le_MaxModelsStashed->text().isEmpty())
134  {
135  bool ok;
136  ui->le_MaxModelsStashed->text().toInt(&ok);
137  if (!ok) { maxObjectsStashed = 100; }
138  }
139 
140  const bool selected = ui->rb_SelectedOnly->isChecked();
141  int ownModelsCount = 0;
142  CStatusMessageList info;
143  if (selected)
144  {
145  static const QString intro("Checking %1 selected models");
146  const CAircraftModelList selectedModels(this->currentModelView()->selectedObjects());
147  ownModelsCount = selectedModels.size();
148  this->addStatusMessage(CStatusMessage(this, CStatusMessage::SeverityInfo, intro.arg(ownModelsCount)));
149  m_modelsToStash = CDatabaseUtils::updateSimulatorForFsFamily(selectedModels, &info, maxObjectsStashed, this, true);
150  }
151  else
152  {
153  const CDbMappingComponent *mappincComponent = this->getMappingComponent();
154  const QSet<CSimulatorInfo> fsFamilySims(CSimulatorInfo::allFsFamilySimulators().asSingleSimulatorSet());
155  static const QString intro("Checking %1 models for %2");
156 
157  // check all own models
158  for (const CSimulatorInfo &simulator : fsFamilySims)
159  {
160  const CAircraftModelList ownModels = mappincComponent->getOwnCachedModels(simulator);
161  const QString sim = simulator.toQString();
162  ownModelsCount += ownModels.size();
163  this->addStatusMessage(CStatusMessage(this, CStatusMessage::SeverityInfo, intro.arg(ownModels.size()).arg(sim)));
164  m_modelsToStash.push_back(CDatabaseUtils::updateSimulatorForFsFamily(ownModels, &info, maxObjectsStashed, this, true));
165  }
166  }
167 
168  const QString result("Tested %1 own models, %2 models should be updated in DB");
169  this->addStatusMessages(info);
170  this->addStatusMessage(CStatusMessage(this, CStatusMessage::SeverityInfo, result.arg(ownModelsCount).arg(m_modelsToStash.size())));
171  m_state = Completed;
172  }
