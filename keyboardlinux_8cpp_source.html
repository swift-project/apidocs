158  void CKeyboardLinux::inputReadyRead(int)
159  {
160  struct input_event eventInput;
161 
162  QFile *fileInput = qobject_cast<QFile *>(sender()->parent());
163  if (!fileInput)
164  return;
165 
166  bool found = false;
167 
168  while (fileInput->read(reinterpret_cast<char *>(&eventInput), sizeof(eventInput)) == sizeof(eventInput))
169  {
170  found = true;
171  if (eventInput.type != EV_KEY)
172  continue;
173  bool isPressed = false;
174  switch (eventInput.value)
175  {
176  case 0:
177  isPressed = false;
178  break;
179  case 1:
180  isPressed = true;
181  break;
182  default:
183  continue;
184  }
185 
186  // The + 8 offset is required for XkbKeycodeToKeysym to output the correct Keysym
187  int keyCode = eventInput.code + 8;
188  keyEvent(keyCode, isPressed);
189  }
190 
191  if (!found)
192  {
193  int fd = fileInput->handle();
194  int version = 0;
195  if ((ioctl(fd, EVIOCGVERSION, &version) < 0) || (((version >> 16) & 0xFF) < 1))
196  {
197  qWarning("CKeyboardLinux: Removing dead input device %s", qPrintable(fileInput->fileName()));
198  m_keyboardDevices.remove(fileInput->fileName());
199  }
200  }
201  }

203  void CKeyboardLinux::addRawInputDevice(const QString &filePath)
204  {
205  QSharedPointer<QFile> inputFile(new QFile(filePath));
206  if (inputFile->open(QIODevice::ReadOnly))
207  {
208  int fd = inputFile->handle();
209  if (fd < 0) { return; }
210 
211  int version = 0;
212  if (ioctl(fd, EVIOCGVERSION, &version) < 0) { return; }
213 
214  char deviceName[255];
215  if (ioctl(fd, EVIOCGNAME(sizeof(deviceName)), deviceName) < 0) { return; }
216 
217  uint8_t bitmask[EV_MAX / 8 + 1];
218  memset(bitmask, 0, sizeof(bitmask));
219  if (ioctl(fd, EVIOCGBIT(0, sizeof(bitmask)), &bitmask) < 0) { return; }
220 
221  // Keyboards support EV_SYN and EV_KEY
222  // but do NOT support EV_REL and EV_ABS
223  if (!(bitmask[EV_SYN / 8] & (1 << (EV_SYN % 8))) &&
224  !(bitmask[EV_KEY / 8] & (1 << (EV_KEY % 8))) &&
225  (bitmask[EV_REL / 8] & (1 << (EV_REL % 8))) &&
226  (bitmask[EV_ABS / 8] & (1 << (EV_ABS % 8))))
227  {
228  return;
229  }
230 
231  // Is it grabbed by someone else?
232  if ((ioctl(fd, EVIOCGRAB, 1) < 0))
233  {
234  BlackMisc::CLogMessage(this).warning(u"Device exclusively grabbed by someone else (X11 using exclusive-mode evdev?)") << deviceName;
235  }
236  else
237  {
238  ioctl(fd, EVIOCGRAB, 0);
239  uint8_t keys[KEY_MAX / 8 + 1];
240  if ((ioctl(fd, EVIOCGBIT(EV_KEY, sizeof(keys)), &keys) >= 0) && (keys[KEY_SPACE / 8] & (1 << (KEY_SPACE % 8))))
241  {
242  BlackMisc::CLogMessage(this).info(u"Found keyboard: %1") << deviceName;
243 
244  fcntl(inputFile->handle(), F_SETFL, O_NONBLOCK);
245  connect(new QSocketNotifier(inputFile->handle(), QSocketNotifier::Read, inputFile.data()), &QSocketNotifier::activated, this, &CKeyboardLinux::inputReadyRead);
246 
247  m_keyboardDevices.insert(filePath, inputFile);
248  }
249  }
250  }
251  else
252  {
253  BlackMisc::CLogMessage(this).error(u"Failed to open keyboard device %1: %2") << inputFile->fileName() << inputFile->errorString();
254  }
255  }

257  void CKeyboardLinux::keyEvent(int keyCode, bool isPressed)
258  {
259  if (isMouseButton(keyCode)) { return; }
260 
261  BlackMisc::Input::CHotkeyCombination oldCombination(m_keyCombination);
262  if (isPressed)
263  {
264  auto key = convertToKey(keyCode);
265  if (key == Key_Unknown) { return; }
266 
267  m_keyCombination.addKeyboardKey(key);
268  }
269  else
270  {
271  auto key = convertToKey(keyCode);
272  if (key == Key_Unknown) { return; }
273 
274  m_keyCombination.removeKeyboardKey(key);
275  }
276 
277  if (oldCombination != m_keyCombination)
278  {
279  emit keyCombinationChanged(m_keyCombination);
280  }
281  }

283  BlackMisc::Input::KeyCode CKeyboardLinux::convertToKey(int keyCode)
284  {
285  // The keycode received from kernel does not take keyboard layouts into account.
286  // It always defaults to US keyboards. In contrast to kernel devices, X11 is aware
287  // of user keyboard layouts. The magic below translates the key code
288  // into the correct symbol via a X11 connection.
289  // Summary of translations:
290  // Kernel key code -> X11 key symbol -> swift key code
291 
292  auto keySym = XkbKeycodeToKeysym(m_display, keyCode, 0, 0);
293  return keyMapping.value(keySym, Key_Unknown);
294  }
