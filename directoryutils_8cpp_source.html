103  bool CDirectoryUtils::canPingUncMachine(const QString &machine)
104  {
105  static QMap<QString, qint64> good;
106  static QMap<QString, qint64> bad;
107 
108  if (machine.isEmpty()) { return false; }
109  const QString m = machine.toLower();
110  if (good.contains(m)) { return true; } // good ones we only test once
111  if (bad.contains(m))
112  {
113  const qint64 ts = bad.value(m);
114  const qint64 dt = QDateTime::currentSecsSinceEpoch() - ts;
115  if (dt < 1000 * 60) { return false; } // still bad
116 
117  // outdated, test again
118  }
119 
120  const bool p = canPing(m);
121  if (p)
122  {
123  good.insert(m, QDateTime::currentSecsSinceEpoch());
124  bad.remove(m);
125  }
126  else
127  {
128  bad.insert(m, QDateTime::currentSecsSinceEpoch());
129  good.remove(m);
130  }
131  return p;
132  }

213  int CDirectoryUtils::copyDirectoryRecursively(const QString &fromDir, const QString &toDir, bool replaceOnConflict)
214  {
215  QDir dir(fromDir);
216  const QStringList fromFiles = dir.entryList(QDir::Files | QDir::NoDotAndDotDot);
217  if (!mkPathIfNotExisting(toDir)) { return -1; }
218 
219  int count = 0;
220  for (const QString &copyFile : fromFiles)
221  {
222  const QString from = CFileUtils::appendFilePaths(fromDir, copyFile);
223  const QString to = CFileUtils::appendFilePaths(toDir, copyFile);
224  if (QFile::exists(to))
225  {
226  if (!replaceOnConflict) { continue; }
227  if (!QFile::remove(to)) { return -1; }
228  }
229  if (!QFile::copy(from, to)) { return -1; }
230  count++;
231  }
232 
233  const QStringList subDirs = dir.entryList(QDir::Dirs | QDir::NoDotAndDotDot);
234  for (const QString &copyDir : subDirs)
235  {
236  const QString fromSubDir = CFileUtils::appendFilePaths(fromDir, copyDir);
237  const QString toSubDir = CFileUtils::appendFilePaths(toDir, copyDir);
238  if (!mkPathIfNotExisting(toDir)) { return -1; }
239 
240  const int c = copyDirectoryRecursively(fromSubDir, toSubDir, replaceOnConflict);
241  if (c < 0) { return -1; }
242  count += c;
243  }
244  return count;
245  }

247  CDirectoryUtils::DirComparison CDirectoryUtils::compareTwoDirectories(const QString &dirSource, const QString &dirTarget, bool nestedDirs)
248  {
249  DirComparison comp;
250  const QDir d1(dirSource);
251  const QDir d2(dirTarget);
252 
253  QFileInfoList dSourceList;
254  QFileInfoList dTargetList;
255  if (d1.exists()) { dSourceList = d1.entryInfoList(QDir::Files | QDir::NoDotAndDotDot, QDir::Name); }
256  if (d2.exists()) { dTargetList = d2.entryInfoList(QDir::Files | QDir::NoDotAndDotDot, QDir::Name); }
257 
258  // only names
259  const QSet<QString> sSourceFiles = CDirectoryUtils::fileNamesToQSet(dSourceList);
260  const QSet<QString> sTargetFiles = CDirectoryUtils::fileNamesToQSet(dTargetList);
261 
262  // full paths
263  const QSet<QString> sSourceCanonicalFiles = CDirectoryUtils::canonicalFileNamesToQSet(dSourceList);
264  const QSet<QString> sTargetCanonicalFiles = CDirectoryUtils::canonicalFileNamesToQSet(dTargetList);
265 
266  QSet<QString> missingInTarget(sSourceFiles);
267  QSet<QString> missingInSource(sTargetFiles);
268  QSet<QString> sameNames(sSourceFiles);
269  missingInSource.subtract(sSourceFiles);
270  missingInTarget.subtract(sTargetFiles);
271  sameNames.intersect(sTargetFiles);
272 
273  comp.source = sSourceCanonicalFiles;
274  comp.missingInSource = CDirectoryUtils::filesToCanonicalNames(missingInSource, sSourceCanonicalFiles);
275  comp.missingInTarget = CDirectoryUtils::filesToCanonicalNames(missingInTarget, sSourceCanonicalFiles);
276  comp.sameNameInSource = CDirectoryUtils::filesToCanonicalNames(sameNames, sSourceCanonicalFiles);
277  comp.sameNameInTarget = CDirectoryUtils::filesToCanonicalNames(sameNames, sTargetCanonicalFiles);
278 
279  Q_ASSERT_X(comp.sameNameInSource.size() == comp.sameNameInTarget.size(), Q_FUNC_INFO, "Same sets require same size");
280  QSet<QString>::const_iterator targetIt = comp.sameNameInTarget.cbegin();
281  for (const QString &sourceFile : std::as_const(comp.sameNameInSource))
282  {
283  const QFileInfo source(sourceFile);
284  const QFileInfo target(*targetIt);
285  ++targetIt; // avoid cpp check warning
286  if (source.lastModified() == target.lastModified() && source.size() == target.size()) { continue; }
287 
288  if (source.lastModified() < target.lastModified())
289  {
290  comp.newerInTarget.insert(target.canonicalFilePath());
291  }
292  else
293  {
294  // source.lastModified() >= target.lastModified()
295  comp.newerInSource.insert(source.canonicalFilePath());
296  }
297  }
298 
299  if (nestedDirs)
300  {
301  const QStringList relativeSubdirs = CDirectoryUtils::getRelativeSubDirectories(dirSource);
302  if (!relativeSubdirs.isEmpty())
303  {
304  for (const QString &relativeSubdir : relativeSubdirs)
305  {
306  const QString sourceSubdir = CFileUtils::appendFilePaths(dirSource, relativeSubdir);
307  const QString targetSubdir = CFileUtils::appendFilePaths(dirTarget, relativeSubdir);
308  const DirComparison subComparison = CDirectoryUtils::compareTwoDirectories(sourceSubdir, targetSubdir, true);
309  comp.insert(subComparison);
310  }
311  }
312  }
313 
314  comp.ok = true;
315  return comp;
316  }
