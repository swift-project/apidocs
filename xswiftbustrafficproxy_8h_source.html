65  this->headingsDeg.push_back(situation.getHeading().value(BlackMisc::PhysicalQuantities::CAngleUnit::deg()));

86  void push_back(const BlackMisc::Aviation::CCallsign &callsign, const BlackMisc::Aviation::CAircraftParts &parts)
87  {
88  this->callsigns.push_back(callsign.asString());
89  this->gears.push_back(parts.isFixedGearDown() ? 1 : 0);
90  this->flaps.push_back(parts.getFlapsPercent() / 100.0);
91  this->spoilers.push_back(parts.isSpoilersOut() ? 1 : 0);
92  this->speedBrakes.push_back(parts.isSpoilersOut() ? 1 : 0);
93  this->slats.push_back(parts.getFlapsPercent() / 100.0);
94  this->wingSweeps.push_back(0.0);
95  this->thrusts.push_back(parts.isAnyEngineOn() ? 0.75 : 0);
96  this->elevators.push_back(0.0);
97  this->rudders.push_back(0.0);
98  this->ailerons.push_back(0.0);
99  this->landLights.push_back(parts.getLights().isLandingOn());
100  this->taxiLights.push_back(parts.getLights().isTaxiOn());
101  this->beaconLights.push_back(parts.getLights().isBeaconOn());
102  this->strobeLights.push_back(parts.getLights().isStrobeOn());
103  this->navLights.push_back(parts.getLights().isNavOn());
104  this->lightPatterns.push_back(0);
105  }
