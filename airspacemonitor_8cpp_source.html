163  CFlightPlan CAirspaceMonitor::loadFlightPlanFromNetwork(const CCallsign &callsign)
164  {
165  CFlightPlan plan;
166  QPointer<CAirspaceMonitor> myself(this);
167 
168  // use cache, but not for own callsign (always reload)
169  if (m_flightPlanCache.contains(callsign)) { plan = m_flightPlanCache[callsign]; }
170  const bool ownAircraft = this->getOwnCallsign() == callsign;
171  if (ownAircraft || !plan.wasSentOrLoaded() || plan.timeDiffSentOrLoadedMs() > 30 * 1000)
172  {
173  // outdated, or not in cache at all, or NOT own aircraft
174  plan = CFlightPlan(); // reset
175  m_flightPlanCache.remove(callsign); // loading FP from network
176  m_fsdClient->sendClientQueryFlightPlan(callsign);
177 
178  // with this little trick we try to make an asynchronous signal / slot based approach
179  // a synchronous return value
180  CEventLoop eventLoop(this);
181  eventLoop.stopWhen(m_fsdClient, &CFSDClient::flightPlanReceived, [=](const auto &cs, const auto &) { return cs == callsign; });
182  if (eventLoop.exec(1500))
183  {
184  if (!myself || !sApp || sApp->isShuttingDown()) { return CFlightPlan(); } // cppcheck-suppress knownConditionTrueFalse
185  if (m_flightPlanCache.contains(callsign))
186  {
187  plan = m_flightPlanCache[callsign];
188  }
189  }
190  }
191  return plan;
192  }

238  CUserList CAirspaceMonitor::getUsersForCallsigns(const CCallsignSet &callsigns) const
239  {
240  CUserList users;
241  if (callsigns.isEmpty()) { return users; }
242  CCallsignSet searchList(callsigns);
243 
244  // myself, which is not in the lists below
245  const CSimulatedAircraft myAircraft(getOwnAircraft());
246  if (!myAircraft.getCallsign().isEmpty() && searchList.contains(myAircraft.getCallsign()))
247  {
248  searchList.remove(myAircraft.getCallsign());
249  users.push_back(myAircraft.getPilot());
250  }
251 
252  // do aircraft first, this will handle most callsigns
253  for (const CSimulatedAircraft &aircraft : this->getAircraftInRange())
254  {
255  if (searchList.isEmpty()) { break; }
256  const CCallsign callsign = aircraft.getCallsign();
257  if (searchList.contains(callsign))
258  {
259  const CUser user = aircraft.getPilot();
260  users.push_back(user);
261  searchList.remove(callsign);
262  }
263  }
264 
265  for (const CAtcStation &station : m_atcStationsOnline)
266  {
267  if (searchList.isEmpty()) { break; }
268  const CCallsign callsign = station.getCallsign();
269  if (searchList.contains(callsign))
270  {
271  const CUser user = station.getController();
272  users.push_back(user);
273  searchList.remove(callsign);
274  }
275  }
276 
277  // we might have unresolved callsigns
278  // those are the ones not in range
279  for (const CCallsign &callsign : std::as_const(searchList))
280  {
281  const CUserList usersByCallsign = sApp->getWebDataServices()->getUsersForCallsign(callsign);
282  if (usersByCallsign.isEmpty())
283  {
284  const CUser user(callsign);
285  users.push_back(user);
286  }
287  else
288  {
289  users.push_back(usersByCallsign[0]);
290  }
291  }
292  return users;
293  }

409  this->setMaxRange(d);
410  }
411  }

609  void CAirspaceMonitor::sendReadyForModelMatching(const CCallsign &callsign, MatchingReadinessFlag rf)
610  {

613 
614  // set flag and init ts
615  Readiness &readiness = this->addMatchingReadinessFlag(callsign, rf);
616 
617  // skip if already sent in the last x seconds
618  const qint64 ageMs = readiness.getAgeMs();
619  if (readiness.wasMatchingSent() && readiness.getAgeMs() < MMMaxAgeThresholdMs) { return; }
620 
621  // checking for min. situations ensures the aircraft is stable, can be interpolated ...
622  const CSimulatedAircraft remoteAircraft = this->getAircraftInRangeForCallsign(callsign);

626  minSituations && (readiness.receivedAll() ||
627  // disable, because it can be model string is unknown in FsInn data
628  // (remoteAircraft.getModel().getModelType() == CAircraftModel::TypeFSInnData) || // here we know we have all data
629  (remoteAircraft.hasModelString()) // we cannot expect more info
630  );
631 
632  const ReverseLookupLogging revLogEnabled = this->whatToReverseLog();
633  if (rf != Verified && validRemoteCs && ageMs <= MMMaxAgeMs && !complete)

636  static const QString format("hh:mm:ss.zzz");

639  const QPointer<CAirspaceMonitor> myself(this);
640  QTimer::singleShot(MMCheckAgainMs, this, [=]() {

656  });

658  // end as we will call again in some time

660 
661  } // not yet complete
662 
663  // some checks for special conditions, e.g. logout -> empty list, but still signals pending
664  if (validRemoteCs)
665  {
666  static const QString readyForMatching("Ready (%1) for matching callsign '%2' with model type '%3', ICAO: '%4' '%5'");

677  const CStatusMessage m = CCallsign::logMessage(callsign, "Ignoring this aircraft, not found in range list, disconnected, or no callsign", CAirspaceMonitor::getLogCategories(), CStatusMessage::SeverityWarning);
678  this->addReverseLookupMessage(callsign, m);
679  m_readiness.remove(callsign);
