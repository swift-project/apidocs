85  void CMatchingForm::setValue(const CAircraftMatcherSetup &setup)
86  {
87  const CAircraftMatcherSetup::MatchingMode mode = setup.getMatchingMode();
88  ui->cb_ByModelString->setChecked(mode.testFlag(CAircraftMatcherSetup::ByModelString));
89  ui->cb_ByCombinedCode->setChecked(mode.testFlag(CAircraftMatcherSetup::ByCombinedType));
90 
91  ui->rb_ByIcaoDataAircraft1st->setChecked(mode.testFlag(CAircraftMatcherSetup::ByIcaoOrderAircraftFirst));
92  ui->rb_ByIcaoDataAirline1st->setChecked(mode.testFlag(CAircraftMatcherSetup::ByIcaoOrderAirlineFirst));
93 
94  const bool nag = !mode.testFlag(CAircraftMatcherSetup::ByAirlineGroupSameAsAirline) && !mode.testFlag(CAircraftMatcherSetup::ByAirlineGroupIfNoAirline);
95  if (nag)
96  {
97  ui->rb_AirlineGroupNo->setChecked(nag);
98  }
99  else
100  {
101  ui->rb_AirlineGroupAsAirline->setChecked(mode.testFlag(CAircraftMatcherSetup::ByAirlineGroupSameAsAirline));
102  ui->rb_AirlineGroupIfNoAirline->setChecked(mode.testFlag(CAircraftMatcherSetup::CAircraftMatcherSetup::ByAirlineGroupIfNoAirline));
103  }
104 
105  ui->cb_ByLivery->setChecked(mode.testFlag(CAircraftMatcherSetup::ByLivery));
106  ui->cb_ByFamily->setChecked(mode.testFlag(CAircraftMatcherSetup::ByFamily));
107  ui->cb_ByMilitary->setChecked(mode.testFlag(CAircraftMatcherSetup::ByMilitary));
108  ui->cb_ByCivilian->setChecked(mode.testFlag(CAircraftMatcherSetup::ByCivilian));
109  ui->cb_CategoryGlider->setChecked(mode.testFlag(CAircraftMatcherSetup::ByCategoryGlider));
110  ui->cb_CategoryMilitaryAircraft->setChecked(mode.testFlag(CAircraftMatcherSetup::ByCategoryMilitary));
111  ui->cb_CategorySmallAircraft->setChecked(mode.testFlag(CAircraftMatcherSetup::ByCategorySmallAircraft));
112  ui->cb_ByVtol->setChecked(mode.testFlag(CAircraftMatcherSetup::ByVtol));
113  ui->cb_ScoreIgnoreZeros->setChecked(mode.testFlag(CAircraftMatcherSetup::ScoreIgnoreZeros));
114  ui->cb_ScorePreferColorLiveries->setChecked(mode.testFlag(CAircraftMatcherSetup::ScorePreferColorLiveries));
115  ui->cb_ExclNoExcludedModels->setChecked(mode.testFlag(CAircraftMatcherSetup::ExcludeNoExcluded));
116  ui->cb_ExclNoDbData->setChecked(mode.testFlag(CAircraftMatcherSetup::ExcludeNoDbData));
117  ui->cb_ModelSetRemoveFailed->setChecked(mode.testFlag(CAircraftMatcherSetup::ModelSetRemoveFailedModel) || mode.testFlag(CAircraftMatcherSetup::ModelFailoverIfNoModelCanBeAdded));
118  ui->cb_ModelFailedFailover->setChecked(mode.testFlag(CAircraftMatcherSetup::ModelFailoverIfNoModelCanBeAdded));
119  ui->cb_ModelSetVerification->setChecked(mode.testFlag(CAircraftMatcherSetup::ModelVerificationAtStartup));
120  ui->cb_ModelSetVerificationOnlyErrorWarning->setChecked(mode.testFlag(CAircraftMatcherSetup::ModelVerificationOnlyWarnError));
121  ui->cb_ReverseUseModelString->setChecked(mode.testFlag(CAircraftMatcherSetup::ReverseLookupModelString));
122  ui->cb_ReverseUseSwiftLiveryIds->setChecked(mode.testFlag(CAircraftMatcherSetup::ReverseLookupSwiftLiveryIds));
123 
124  this->setMatchingAlgorithm(setup);
125  this->setPickStrategy(setup);
126 
127  ui->cb_MsReverseLookup->setChecked(setup.isMsReverseLookupEnabled());
128  ui->cb_MsMatching->setChecked(setup.isMsMatchingStageEnabled());
129  ui->le_MsReverseLookup->setText(setup.getMsReverseLookupFile());
130  ui->le_MsMatching->setText(setup.getMsMatchingStageFile());
131  }

223  case CAircraftMatcherSetup::PickRandom: ui->rb_PickRandom->setChecked(true); break;

231  void CMatchingForm::setMatchingAlgorithm(const CAircraftMatcherSetup &setup)
232  {
233  switch (setup.getMatchingAlgorithm())
234  {
235  case CAircraftMatcherSetup::MatchingStepwiseReduce:
236  ui->rb_Reduction->setChecked(true);
237  ui->gb_Reduction->setEnabled(true);
238  ui->gb_Scoring->setEnabled(false);
239  break;
240  case CAircraftMatcherSetup::MatchingScoreBased:
241  ui->rb_ScoreOnly->setChecked(true);
242  ui->gb_Reduction->setEnabled(false);
243  ui->gb_Scoring->setEnabled(true);
244  break;
245  case CAircraftMatcherSetup::MatchingStepwiseReducePlusScoreBased:
246  default:
247  ui->rb_ScoreAndReduction->setChecked(true);
248  ui->gb_Reduction->setEnabled(true);
249  ui->gb_Scoring->setEnabled(true);
250  break;
251  }
252  }
