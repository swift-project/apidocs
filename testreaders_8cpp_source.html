78  void CTestReaders::readIcaoData()
79  {
80  const CUrl url(sApp->getGlobalSetup().getDbIcaoReaderUrl());
81  qDebug() << "Reader URL" << url.toQString();
82  if (!this->connectServer(url))
83  {
84  QSKIP("Server not reachable.");
85  return;
86  }
87  m_icaoReader->start();
88  m_icaoReader->readInBackgroundThread(CEntityFlags::AllIcaoEntities, QDateTime());
89 
90  // expect does not work here, as I receive multiple signals for all entities and read states
91  // I have to run my "own event loop"
92  for (int i = 0; i < 60; i++)
93  {
94  CApplication::processEventsFor(500); // process events and sleep
95  if (this->m_icaoReader->getAircraftIcaoCodesCount() > 0 && this->m_icaoReader->getAirlineIcaoCodesCount() > 0)
96  {
97  break;
98  }
99  }
100 
101  const QString m1("No aircraft ICAOs " + url.getFullUrl());
102  const QString m2("No airline ICAOs " + url.getFullUrl());
103  QVERIFY2(this->m_icaoReader->getAircraftIcaoCodesCount() > 0, qUtf8Printable(m1));
104  QVERIFY2(this->m_icaoReader->getAirlineIcaoCodesCount() > 0, qUtf8Printable(m2));
105 
106  // reader set to ignore incomplete ICAO data, so sizes must match
107  const CAircraftIcaoCodeList aircraftIcaos = this->m_icaoReader->getAircraftIcaoCodes();
108  const CAirlineIcaoCodeList airlineIcaos = this->m_icaoReader->getAirlineIcaoCodes();
109  const CAircraftIcaoCodeList aircraftIcaosValid(aircraftIcaos.findByValidDesignator());
110  const CAirlineIcaoCodeList airlineIcaosValid(airlineIcaos.findByValidDesignator());
111  const CAircraftIcaoCodeList aircraftIcaosInvalid(aircraftIcaos.findByInvalidDesignator());
112  const CAirlineIcaoCodeList airlineIcaosInvalid(airlineIcaos.findByInvalidDesignator());
113  const int aircraftValidSize = aircraftIcaosValid.size();
114  const int aircraftSize = aircraftIcaos.size();
115  const int airlineValidSize = airlineIcaosValid.size();
116  const int airlineSize = airlineIcaos.size();
117 
118  if (!aircraftIcaosInvalid.isEmpty())
119  {
120  qDebug() << aircraftIcaosInvalid.toQString();
121  }
122  if (!airlineIcaosInvalid.isEmpty())
123  {
124  qDebug() << airlineIcaosInvalid.toQString();
125  }
126 
127  QVERIFY2(aircraftValidSize == aircraftSize, "All aircraft ICAOs must be valid");
128  QVERIFY2(airlineValidSize == airlineSize, "Some airline ICAOs must be valid");
129 
130  CApplication::processEventsFor(2500); // make sure events are processed
131  }
