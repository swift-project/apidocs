87  void CPlugin::startServer()
88  {
89  XPLMRegisterFlightLoopCallback(flightLoopCallback, -1, this);
90 
91  readConfig();
92 
93  m_service = std::make_unique<CService>(this);
94  m_traffic = std::make_unique<CTraffic>(this);
95  m_weather = std::make_unique<CWeather>(this);
96 
97  m_traffic->setPlaneViewMenu(m_planeViewSubMenu);
98 
99  if (m_pluginConfig.getDBusMode() == CConfig::DBusP2P)
100  {
101  m_dbusP2PServer = std::make_unique<CDBusServer>();
102 
103  // FIXME: make listen address configurable
104  std::string listenAddress = "tcp:host=" + m_pluginConfig.getDBusAddress() + ",port=" + std::to_string(m_pluginConfig.getDBusPort());
105  if (!m_dbusP2PServer->listen(listenAddress))
106  {
107  m_service->addTextMessage("XSwiftBus startup failed!", 255, 0, 0);
108  return;
109  }
110  m_dbusP2PServer->setDispatcher(&m_dbusDispatcher);
111 
112  m_dbusP2PServer->setNewConnectionFunc([this](const std::shared_ptr<CDBusConnection> &conn) {
113  m_dbusConnection = conn;
114  m_dbusConnection->setDispatcher(&m_dbusDispatcher);
115  m_service->setDBusConnection(m_dbusConnection);
116  m_service->registerDBusObjectPath(m_service->InterfaceName(), m_service->ObjectPath());
117  m_traffic->setDBusConnection(m_dbusConnection);
118  m_traffic->registerDBusObjectPath(m_traffic->InterfaceName(), m_traffic->ObjectPath());
119  m_weather->setDBusConnection(m_dbusConnection);
120  m_weather->registerDBusObjectPath(m_weather->InterfaceName(), m_weather->ObjectPath());
121  });
122  }
123  else
124  {
125  // Todo: retry if it fails
126  bool success = m_dbusConnection->connect(CDBusConnection::SessionBus);
127 
128  if (!success)
129  {
130  // Print error
131  return;
132  }
133 
134  m_dbusConnection->setDispatcher(&m_dbusDispatcher);
135  m_dbusConnection->requestName(xswiftbusServiceName());
136 
137  m_service->setDBusConnection(m_dbusConnection);
138  m_service->registerDBusObjectPath(m_service->InterfaceName(), m_service->ObjectPath());
139  m_traffic->setDBusConnection(m_dbusConnection);
140  m_traffic->registerDBusObjectPath(m_traffic->InterfaceName(), m_traffic->ObjectPath());
141  m_weather->setDBusConnection(m_dbusConnection);
142  m_weather->registerDBusObjectPath(m_weather->InterfaceName(), m_weather->ObjectPath());
143  }
144 
146  const std::string msg = "XSwiftBus " + m_service->getVersionNumber() + " started.";
147  INFO_LOG(msg);
148  m_service->addTextMessage(msg, 0, 255, 255);
149  }

177  float CPlugin::startServerDeferred(float, float, int, void *refcon)
178  {
179  auto *plugin = static_cast<CPlugin *>(refcon);
180  if (!plugin->m_isRunning)
181  {
182  plugin->startServer();
183  plugin->m_isRunning = true;
184  INFO_LOG("XSwiftBus plugin started (deferred)");
185  }
186  return 0;
187  }

189  float CPlugin::flightLoopCallback(float, float, int, void *refcon)
190  {
191  auto *plugin = static_cast<CPlugin *>(refcon);
192 
193  plugin->m_dbusDispatcher.runOnce();
194  if (plugin->m_service) { plugin->m_service->process(); }
195  if (plugin->m_weather) { plugin->m_weather->process(); }
196  if (plugin->m_traffic) { plugin->m_traffic->process(); }
197  return -1;
198  }
