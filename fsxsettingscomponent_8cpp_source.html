141  CSimulatorFsxCommon *CFsxSettingsComponent::getFsxOrP3DSimulator() const
142  {
143  if (!sGui || sGui->isShuttingDown() || !sGui->getIContextSimulator()) { return nullptr; }
144  const CSimulatorPluginInfo plugin = sGui->getIContextSimulator()->getSimulatorPluginInfo();
145  if (plugin.isEmulatedPlugin()) { return nullptr; } // cast would fail
146 
147  // ISimulator can only be obtained in local environment, not distributed UI
148  if (!sGui->hasSimulator()) { return nullptr; }
149 
150  ISimulator *simulator = sGui->getISimulator().data();
151  if (!simulator || simulator->isEmulatedDriver()) { return nullptr; }
152  if (!simulator->getSimulatorInfo().isFsxP3DFamily()) { return nullptr; }
153  if (simulator->getSimulatorInfo() != m_simulator) { return nullptr; }
154 
156  CSimulatorFsxCommon *fsx = static_cast<CSimulatorFsxCommon *>(simulator);
157  return fsx;
158  }
