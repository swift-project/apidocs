32  CAircraftModelList CAircraftModelLoaderFlightgear::parseFlyableAirplanes(const QString &rootDirectory, const QStringList &excludeDirectories)
33  {
34  Q_UNUSED(excludeDirectories);
35  if (rootDirectory.isEmpty()) { return {}; }
36  CAircraftModelList installedModels;
37 
38  QDir searchPath(rootDirectory);
39  searchPath.setNameFilters(QStringList() << "*-set.xml");
40  QDirIterator aircraftIt(searchPath, QDirIterator::Subdirectories | QDirIterator::FollowSymlinks);
41 
42  while (aircraftIt.hasNext())
43  {
44  aircraftIt.next();
45  if (CFileUtils::isExcludedDirectory(aircraftIt.fileInfo(), excludeDirectories, Qt::CaseInsensitive)) { continue; }
46  if (aircraftIt.filePath().contains("/AI/Aircraft")) { continue; }
47  CAircraftModel model;
48  QString modelName = aircraftIt.fileName();
49  modelName = modelName.remove("-set.xml");
50  model.setName(modelName);
51  model.setModelString(getModelString(aircraftIt.fileName(), false));
52  model.setModelType(CAircraftModel::TypeOwnSimulatorModel);
53  model.setSimulator(CSimulatorInfo::fg());
54  model.setFileDetailsAndTimestamp(aircraftIt.fileInfo());
55  model.setModelMode(CAircraftModel::Exclude);
56  addUniqueModel(model, installedModels);
57  }
58 
59  return installedModels;
60  }

99  {

118  void CAircraftModelLoaderFlightgear::startLoadingFromDisk(IAircraftModelLoader::LoadMode mode, const IAircraftModelLoader::ModelConsolidationCallback &modelConsolidation, const QStringList &modelDirectories)
119  {
120  const CSimulatorInfo simulator = CSimulatorInfo::fg();
121  const QStringList modelDirs = this->getInitializedModelDirectories(modelDirectories, simulator);
122  const QStringList excludedDirectoryPatterns(m_settings.getModelExcludeDirectoryPatternsOrDefault(simulator)); // copy
123 
124  if (mode.testFlag(LoadInBackground))
125  {
126  if (m_parserWorker && !m_parserWorker->isFinished()) { return; }
127  emit this->diskLoadingStarted(simulator, mode);
128 
129  m_parserWorker = CWorker::fromTask(this, "CAircraftModelLoaderFlightgear::performParsing",
130  [this, modelDirs, excludedDirectoryPatterns, modelConsolidation]() {
131  auto models = this->performParsing(modelDirs, excludedDirectoryPatterns);
132  if (modelConsolidation) { modelConsolidation(models, true); }
133  return models;
134  });
135  m_parserWorker->thenWithResult<CAircraftModelList>(this, [=](const auto &models) {
136  this->updateInstalledModels(models);
137  m_loadingMessages.freezeOrder();
138  emit this->loadingFinished(m_loadingMessages, simulator, ParsedData);
139  });
140  }
141  else if (mode.testFlag(LoadDirectly))
142  {
143  emit this->diskLoadingStarted(simulator, mode);
144  CAircraftModelList models(this->performParsing(modelDirs, excludedDirectoryPatterns));
145  this->updateInstalledModels(models);
146  }
147  }
