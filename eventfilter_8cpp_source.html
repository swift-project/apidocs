12  bool CUpperCaseEventFilter::eventFilter(QObject *object, QEvent *event)
13  {
14  if (event->type() == QEvent::KeyPress)
15  {
16  if (QKeyEvent *e = dynamic_cast<QKeyEvent *>(event))
17  {
18  // If QKeyEvent::text() returns an empty QString then let normal
19  // processing proceed as it may be a control (e.g. cursor movement)
20  // key. Otherwise convert the text to upper case and insert it at
21  // the current cursor position.
22  QPlainTextEdit *pte = qobject_cast<QPlainTextEdit *>(object);
23  if (!pte) { return false; }
24  if (pte->isReadOnly()) { return false; }
25 
26  if (e->text().length() == 1)
27  {
28  const QChar c = e->text().front();
29  if (m_illegalChars.contains(c)) { return true; }
30  if (c.isLetter())
31  {
32  const ushort unicode = c.unicode();
33  if (m_onlyAscii && unicode > 127) { return true; }
34 
35  pte->insertPlainText(e->text().toUpper());
36  // return true to prevent further processing
37  return true;
38  }
39  else
40  {
41  // all codes like backspace etc.
42  return false;
43  }
44  } // length
45  } // key event
46  }
47  return false;
48  }
