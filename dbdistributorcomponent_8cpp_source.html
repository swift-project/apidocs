84  void CDbDistributorComponent::selectStandardModels()
85  {
86  const QObject *s = QObject::sender();
87  QSet<QString> keys;
88  if (s == ui->pb_SelectAllFsFamily)
89  {
90  this->filterBySimulator(CSimulatorInfo::AllFsFamily);
91  keys = CDistributor::standardAllFsFamily();
92  }
93  else if (s == ui->pb_SelectXPlaneBB)
94  {
95  this->filterBySimulator(CSimulatorInfo::XPLANE);
96  keys = QSet<QString>({ CDistributor::xplaneBlueBell() });
97  }
98  else if (s == ui->pb_SelectXplaneXCSL)
99  {
100  this->filterBySimulator(CSimulatorInfo::XPLANE);
101  keys = QSet<QString>({ CDistributor::xplaneXcsl() });
102  }
103 
104  // deferred because filter must first work and update
105  const QPointer<CDbDistributorComponent> myself(this);
106  QTimer::singleShot(2000, this, [=] {
107  if (!myself || !sApp || sApp->isShuttingDown()) { return; }
108  ui->tvp_Distributors->selectDbKeys(keys);
109  });
110  }
