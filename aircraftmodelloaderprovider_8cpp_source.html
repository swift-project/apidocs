35  IAircraftModelLoader *CMultiAircraftModelLoaderProvider::loaderInstance(const CSimulatorInfo &simulator)
36  {
37  Q_ASSERT_X(simulator.isSingleSimulator(), Q_FUNC_INFO, "No single simulator");
38  switch (simulator.getSimulator())
39  {
40  case CSimulatorInfo::FSX:
41  {
42  if (!m_loaderFsx) { m_loaderFsx = this->initLoader(CSimulatorInfo::fsx()); }
43  return m_loaderFsx;
44  }
45  case CSimulatorInfo::P3D:
46  {
47  if (!m_loaderP3D) { m_loaderP3D = this->initLoader(CSimulatorInfo::p3d()); }
48  return m_loaderP3D;
49  }
50  case CSimulatorInfo::XPLANE:
51  {
52  if (!m_loaderXP) { m_loaderXP = this->initLoader(CSimulatorInfo::xplane()); }
53  return m_loaderXP;
54  }
55  case CSimulatorInfo::FS9:
56  {
57  if (!m_loaderFS9) { m_loaderFS9 = this->initLoader(CSimulatorInfo::fs9()); }
58  return m_loaderFS9;
59  }
60  case CSimulatorInfo::FG:
61  {
62  if (!m_loaderFG) { m_loaderFG = this->initLoader(CSimulatorInfo::fg()); }
63  return m_loaderFG;
64  }
65  default:
66  Q_ASSERT_X(false, Q_FUNC_INFO, "Wrong simulator");
67  break;
68  }
69  return nullptr;
70  }

78  IAircraftModelLoader *CMultiAircraftModelLoaderProvider::initLoader(const CSimulatorInfo &simulator)
79  {
80  // in some cases the loading progress signal was not send properly
81  // changing to Qt::QueuedConnection has solved the issues (Ref T529)
82  IAircraftModelLoader *loader = createModelLoader(simulator, this);
83  bool c = connect(loader, &IAircraftModelLoader::loadingFinished, this, &CMultiAircraftModelLoaderProvider::loadingFinished, Qt::QueuedConnection);
84  Q_ASSERT_X(c, Q_FUNC_INFO, "Connect failed");
85  c = connect(loader, &IAircraftModelLoader::diskLoadingStarted, this, &CMultiAircraftModelLoaderProvider::diskLoadingStarted, Qt::QueuedConnection);
86  Q_ASSERT_X(c, Q_FUNC_INFO, "Connect failed");
87  c = connect(loader, &IAircraftModelLoader::cacheChanged, this, &CMultiAircraftModelLoaderProvider::cacheChanged, Qt::QueuedConnection);
88  Q_ASSERT_X(c, Q_FUNC_INFO, "Connect failed");
89  c = connect(loader, &IAircraftModelLoader::loadingProgress, this, &CMultiAircraftModelLoaderProvider::loadingProgress, Qt::QueuedConnection);
90  Q_ASSERT_X(c, Q_FUNC_INFO, "Connect failed");
91  return loader;
92  }
