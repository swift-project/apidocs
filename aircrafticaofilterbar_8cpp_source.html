66  void CAircraftIcaoFilterBar::filter(const CAircraftIcaoCode &icao)
67  {
68  this->clearForm();
69  bool filter = false;
70  if (icao.hasDesignator())
71  {
72  ui->le_Designator->setText(icao.getDesignator());
73  filter = true;
74  }
75  if (icao.hasManufacturer())
76  {
77  ui->le_Manufacturer->setText(icao.getManufacturer());
78  filter = true;
79  }
80  if (icao.hasModelDescription())
81  {
82  ui->le_Description->setText("*" + icao.getModelDescription() + "*");
83  filter = true;
84  }
85  if (filter)
86  {
87  ui->filter_Buttons->clickButton(CFilterBarButtons::Filter);
88  }
89  else
90  {
91  ui->filter_Buttons->clickButton(CFilterBarButtons::RemoveFilter);

114  void CAircraftIcaoFilterBar::initCompleters()
115  {
116  if (m_hasCompleters) { return; }
117  if (!sGui || !sGui->hasWebDataServices()) { return; }
118 
119  if (sGui->getWebDataServices()->getAircraftIcaoCodesCount() < 1) { return; }
120  m_hasCompleters = true;
121  const CAircraftIcaoCodeList codes = sGui->getWebDataServices()->getAircraftIcaoCodes();
122  const QStringList designators = codes.allDesignators().values();
123  ui->le_Designator->setCompleter(new QCompleter(designators, ui->le_Designator));
124  const QStringList families = codes.allFamilies().values();
125  ui->le_Family->setCompleter(new QCompleter(families, ui->le_Family));
126  const QStringList manufacturers = codes.allManufacturers().values();
127  ui->le_Manufacturer->setCompleter(new QCompleter(manufacturers, ui->le_Manufacturer));
128  }
