105  void CInterpolationSetupComponent::reloadSetup()
106  {
107  const bool global = (this->getSetupMode() == CInterpolationSetupComponent::SetupGlobal);
108  const bool overlay = QObject::sender() == ui->pb_Reload;
109  if (!this->checkPrerequisites(!global, overlay)) { return; }
110  if (global)
111  {
112  CInterpolationAndRenderingSetupGlobal gs = sGui->getIContextSimulator()->getInterpolationAndRenderingSetupGlobal();
113  ui->form_InterpolationSetup->setValue(gs);
114  }
115  else
116  {
117  const CCallsign cs = ui->comp_CallsignCompleter->getCallsign(false);
118  if (!cs.isValid()) { return; }
119  const CInterpolationAndRenderingSetupPerCallsign setup = sGui->getIContextSimulator()->getInterpolationAndRenderingSetupPerCallsignOrDefault(cs);
120  ui->form_InterpolationSetup->setValue(setup);
121  }
122  }

124  void CInterpolationSetupComponent::saveSetup()
125  {
126  const bool global = (this->getSetupMode() == CInterpolationSetupComponent::SetupGlobal);
127  if (!this->checkPrerequisites(!global, true)) { return; }
128  CInterpolationAndRenderingSetupPerCallsign setup = ui->form_InterpolationSetup->getValue();
129  CInterpolationAndRenderingSetupGlobal gs = sGui->getIContextSimulator()->getInterpolationAndRenderingSetupGlobal();
130  if (global)
131  {
132  gs.setBaseValues(setup);
133  gs.setLogInterpolation(false); // that would globally log all values
134  sGui->getIContextSimulator()->setInterpolationAndRenderingSetupGlobal(gs);
135  CLogMessage(this).info(u"Set global setup: '%1'") << gs.toQString(true);
136 
137  const QPointer<CInterpolationSetupComponent> myself(this);
138  QTimer::singleShot(250, this, [=] {
139  if (!myself) { return; }
140  this->reloadSetup();
141  });
142  }
143  else
144  {
145  const CCallsign cs = ui->comp_CallsignCompleter->getCallsign(false);
146  if (!cs.isValid()) { return; }
147  setup.setCallsign(cs);
148  const bool ignoreGlobal = ui->cb_IgnoreGlobal->isChecked();
149  if (ignoreGlobal && setup.isEqualToGlobal(gs))
150  {
151  static const CStatusMessage m = CStatusMessage(this).validationWarning(u"Same as global setup");
152  this->showOverlayMessage(m);
153  return;
154  }
155 
156  CInterpolationSetupList setups = ui->tvp_InterpolationSetup->container();
157  const int replaced = setups.replaceOrAddObjectByCallsign(setup);
158  if (replaced < 1) { return; }
159  const bool set = this->setSetupsToContext(setups);
160  if (!set) { return; }
161 
162  const QPointer<CInterpolationSetupComponent> myself(this);
163  QTimer::singleShot(250, this, [=] {
164  if (!myself) { return; }
165  this->displaySetupsPerCallsign();
166  });
167  }
168  }

170  void CInterpolationSetupComponent::removeOrResetSetup()
171  {
172  const bool global = (this->getSetupMode() == CInterpolationSetupComponent::SetupGlobal);
173  if (!this->checkPrerequisites(!global, true)) {}
174  if (global)
175  {
176  // reset
177  CInterpolationAndRenderingSetupGlobal gs;
178  sGui->getIContextSimulator()->setInterpolationAndRenderingSetupGlobal(gs);
179  this->reloadSetup();
180  }
181  else
182  {
183  // delete/remove
184  const CCallsign cs = ui->comp_CallsignCompleter->getCallsign(false);
185  CInterpolationSetupList setups = ui->tvp_InterpolationSetup->container();
186  const int removed = setups.removeByCallsign(cs);
187  if (removed < 1) { return; } // nothing done
188  const bool set = this->setSetupsToContext(setups);
189  if (!set) { return; }
190 
191  const QPointer<CInterpolationSetupComponent> myself(this);
192  QTimer::singleShot(100, this, [=] {
193  if (!myself) { return; }
194  this->displaySetupsPerCallsign();
195  });
196  }
197  }

213  bool CInterpolationSetupComponent::checkPrerequisites(bool checkSim, bool showOverlay)
214  {
215  if (!sGui || !sGui->getIContextSimulator() || sGui->isShuttingDown())
216  {
217  if (showOverlay)
218  {
219  const CStatusMessage m = CStatusMessage(this).validationError(u"No context");
220  this->showOverlayMessage(m);
221  }
222  return false;
223  }
224  if (checkSim && !sGui->getIContextSimulator()->isSimulatorAvailable())
225  {
226  if (showOverlay)
227  {
228  const CStatusMessage m = CStatusMessage(this).validationError(u"No simulator available");
229  this->showOverlayMessage(m);
230  }
231  return false;
232  }
233  return true;
234  }

236  bool CInterpolationSetupComponent::setSetupsToContext(const CInterpolationSetupList &setups, bool force)
237  {
238  if (!sGui || sGui->isShuttingDown() || !sGui->getIContextSimulator()) { return false; }
239  if (!force && setups == m_lastSetSetups) { return false; }
240  const bool ignoreGlobal = ui->cb_IgnoreGlobal->isChecked();
241  sGui->getIContextSimulator()->setInterpolationAndRenderingSetupsPerCallsign(setups, ignoreGlobal);
242  m_lastSetSetups = setups;
243  return true;
244  }

252  void CInterpolationSetupComponent::onObjectsDeleted(const CVariant &deletedObjects)
253  {
254  if (deletedObjects.canConvert<CInterpolationSetupList>())
255  {
256  const CInterpolationSetupList deletedSetups = deletedObjects.value<CInterpolationSetupList>();
257  if (deletedSetups.isEmpty()) { return; }
258 
259  // make sure the setups are really deleted
260  // it can be they are already in the container, but there is no guarantee
261  CInterpolationSetupList setups = ui->tvp_InterpolationSetup->container();
262  setups.removeByCallsigns(deletedSetups.getCallsigns());
263  const bool set = this->setSetupsToContext(setups, true);
264  Q_UNUSED(set)
