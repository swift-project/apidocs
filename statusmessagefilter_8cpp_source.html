17  CStatusMessageList CStatusMessageFilter::filter(const CStatusMessageList &inContainer) const
18  {
19  if (!this->isValid()) { return inContainer; }
20  CStatusMessageList outContainer;
21  for (const CStatusMessage &msg : inContainer)
22  {
23  if (!(m_severity == CStatusMessage::SeverityInfo || m_severity == CStatusMessage::SeverityDebug))
24  {
25  if (!msg.isSeverityHigherOrEqual(this->m_severity)) { continue; }
26  }
27 
28  if (!this->m_msgText.isEmpty())
29  {
30  if (!this->stringMatchesFilterExpression(msg.getMessage(), this->m_msgText)) { continue; }
31  }
32 
33  if (!this->m_category.isEmpty())
34  {
35  if (!this->stringMatchesFilterExpression(CLogPattern::humanOrTechnicalCategoriesFrom(msg).join(", "), this->m_category)) { continue; }
36  }
37 
38  outContainer.push_back(msg);
39  }
40  return outContainer;
41  }

43  CVariant CStatusMessageFilter::getAsValueObject() const
44  {
45  if (!m_category.contains('*'))
46  {
47  if (CLogPattern::allHumanReadableNames().contains(m_category))
48  {
49  return CVariant::from(CLogPattern::fromHumanReadableName(m_category).withSeverityAtOrAbove(m_severity));
50  }
51  return CVariant::from(CLogPattern::exactMatch(m_category).withSeverityAtOrAbove(m_severity));
52  }
53 
54  CLogCategoryList categories = CLogCategoryList::fromQStringList(CLogCategories::allSpecialCategories());
55  categories.removeIf([this](const CLogCategory &cat) { return this->stringMatchesFilterExpression(cat.toQString(), this->m_category); });
56  CSequence<QString> humanNames = CLogPattern::allHumanReadableNames();
57  humanNames.removeIf([this](const QString &name) { return this->stringMatchesFilterExpression(name, this->m_category); });
58  auto humanCats = humanNames.transform([](const QString &name) {
59  const auto strings = CLogPattern::fromHumanReadableName(name).getCategoryStrings();
60  return strings.isEmpty() ? QString {} : *strings.begin();
61  });
62 
63  return CVariant::from(CLogPattern::anyOf(categories.join(humanCats)).withSeverityAtOrAbove(m_severity));
64  }
