19  {

33  CSimulatorInfoList CSimulatorInfoList::splitIntoSingleSimulators() const
34  {
35  if (this->isEmpty()) { return CSimulatorInfoList(); }
36  CSimulatorInfoList newList;
37  for (const CSimulatorInfo &simulator : *this)
38  {
39  if (simulator.isUnspecified() || simulator.isNoSimulator()) { continue; }
40  if (simulator.isSingleSimulator())
41  {
42  newList.push_back(simulator);
43  }
44  else
45  {
46  newList.push_back(splitIntoSingleSimulators(simulator));
47  }
48  }
49  return newList;
50  }
