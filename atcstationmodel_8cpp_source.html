47  void CSampleAtcStationModel::updateAtcStations(const QVector<CSampleAtcStation> &atcStations)
48  {
49  // Add stations which didn't exist yet
50  for (const auto &station : atcStations)
51  {
52  if (!m_atcStations.contains(station)) { addStation(station); }
53  }
54 
55  // Remove all stations which are no longer there
56  for (int i = m_atcStations.size() - 1; i >= 0; i--)
57  {
58  CSampleAtcStation &station = m_atcStations[i];
59  if (!m_atcStations.contains(station))
60  {
61  removeStationAtPosition(i);
62  }
63  }
64  }

86  QVariant CSampleAtcStationModel::data(const QModelIndex &index, int role) const
87  {
88  if (index.row() < 0 || index.row() >= m_atcStations.count())
89  return QVariant();
