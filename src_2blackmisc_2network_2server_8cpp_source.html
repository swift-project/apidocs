171  CStatusMessageList CServer::validate() const
172  {
173  static const CLogCategoryList cats(CLogCategoryList(this).withValidation());
174  CStatusMessageList msgs;
175  if (this->getName().isEmpty()) { msgs.push_back(CStatusMessage(CStatusMessage::SeverityError, u"Missing name")); }
176  if (this->getAddress().isEmpty()) { msgs.push_back(CStatusMessage(CStatusMessage::SeverityError, u"Missing address")); }
177  if (this->getDescription().isEmpty()) { msgs.push_back(CStatusMessage(CStatusMessage::SeverityWarning, u"Missing description")); }
178  if (this->getPort() < 1 || this->getPort() > 65535) { msgs.push_back(CStatusMessage(CStatusMessage::SeverityError, u"Wrong port")); }
179  msgs.push_back(this->getUser().validate());
180  msgs.push_back(this->getFsdSetup().validate());
181  msgs.push_back(this->getVoiceSetup().validate());
182  msgs.addCategories(cats);
183  msgs.sortBySeverity();
184  return msgs;
185  }

195  QVariant CServer::propertyByIndex(CPropertyIndexRef index) const
196  {
197  if (index.isMyself()) { return QVariant::fromValue(*this); }
198  if (ITimestampBased::canHandleIndex(index)) { return ITimestampBased::propertyByIndex(index); }
199 
200  const ColumnIndex i = index.frontCasted<ColumnIndex>();
201  switch (i)
202  {
203  case IndexAddress: return QVariant::fromValue(m_address);
204  case IndexDescription: return QVariant::fromValue(m_description);
205  case IndexName: return QVariant::fromValue(m_name);
206  case IndexPort: return QVariant::fromValue(m_port);
207  case IndexUser: return m_user.propertyByIndex(index.copyFrontRemoved());
208  case IndexFsdSetup: return m_fsdSetup.propertyByIndex(index.copyFrontRemoved());
209  case IndexVoiceSetup: return m_voiceSetup.propertyByIndex(index.copyFrontRemoved());
210  case IndexEcosystem: return m_ecosystem.propertyByIndex(index.copyFrontRemoved());
211  case IndexIsAcceptingConnections: return QVariant::fromValue(m_isAcceptingConnections);
212  case IndexServerType: return QVariant::fromValue(m_serverType);
213  case IndexServerTypeAsString: return QVariant::fromValue(getServerTypeAsString());
214  default: return CValueObject::propertyByIndex(index);
215  }

218  void CServer::setPropertyByIndex(CPropertyIndexRef index, const QVariant &variant)
219  {
220  if (index.isMyself())
221  {
222  (*this) = variant.value<CServer>();
223  return;
224  }
225  if (ITimestampBased::canHandleIndex(index))
226  {
227  ITimestampBased::setPropertyByIndex(index, variant);
228  return;
229  }
230 
231  const ColumnIndex i = index.frontCasted<ColumnIndex>();
232  switch (i)
233  {
234  case IndexAddress: this->setAddress(variant.value<QString>()); break;
235  case IndexPort: this->setPort(variant.value<qint32>()); break;
236  case IndexDescription: this->setDescription(variant.value<QString>()); break;
237  case IndexName: this->setName(variant.value<QString>()); break;
238  case IndexUser: m_user.setPropertyByIndex(index.copyFrontRemoved(), variant); break;
239  case IndexFsdSetup: m_fsdSetup.setPropertyByIndex(index.copyFrontRemoved(), variant); break;
240  case IndexVoiceSetup: m_voiceSetup.setPropertyByIndex(index.copyFrontRemoved(), variant); break;
241  case IndexEcosystem: m_ecosystem.setPropertyByIndex(index.copyFrontRemoved(), variant); break;
242  case IndexServerType: this->setServerType(static_cast<ServerType>(variant.toInt())); break;
243  case IndexIsAcceptingConnections: this->setIsAcceptingConnections(variant.value<bool>()); break;
244  default: CValueObject::setPropertyByIndex(index, variant); break;
245  }
246  }

248  int CServer::comparePropertyByIndex(CPropertyIndexRef index, const CServer &compareValue) const
249  {
250  if (index.isMyself()) { return this->getName().compare(compareValue.getName()); }
251  if (ITimestampBased::canHandleIndex(index)) { return ITimestampBased::comparePropertyByIndex(index, compareValue); }
252  const ColumnIndex i = index.frontCasted<ColumnIndex>();
253  switch (i)
254  {
255  case IndexAddress: return this->getAddress().compare(compareValue.getAddress(), Qt::CaseInsensitive);
256  case IndexDescription: return this->getDescription().compare(compareValue.getDescription(), Qt::CaseInsensitive);
257  case IndexFsdSetup: return m_fsdSetup.comparePropertyByIndex(index.copyFrontRemoved(), compareValue.getFsdSetup());
258  case IndexVoiceSetup: return m_voiceSetup.comparePropertyByIndex(index.copyFrontRemoved(), compareValue.getVoiceSetup());
259  case IndexName: return this->getName().compare(compareValue.getName(), Qt::CaseInsensitive);
260  case IndexPort: return Compare::compare(this->getPort(), compareValue.getPort());
261  case IndexUser: return this->getUser().comparePropertyByIndex(index.copyFrontRemoved(), compareValue.getUser());
262  case IndexEcosystem: return this->getEcosystem().comparePropertyByIndex(index.copyFrontRemoved(), compareValue.getEcosystem());
263  case IndexIsAcceptingConnections: return Compare::compare(this->isAcceptingConnections(), compareValue.isAcceptingConnections());
264  case IndexServerType:
265  case IndexServerTypeAsString:
266  return this->getServerTypeAsString().compare(compareValue.getServerTypeAsString(), Qt::CaseInsensitive);
267  default: break;
268  }
269  BLACK_VERIFY_X(false, Q_FUNC_INFO, qUtf8Printable("No comparison for index " + index.toQString()));
270  return 0;
271  }
