26  CCountry CCountryList::findBestMatchByCountryName(const QString &candidate) const
27  {
28  if (candidate.isEmpty()) { return CCountry(); }
29 
30  thread_local const QRegularExpression reg("^[a-z]+", QRegularExpression::CaseInsensitiveOption);
31  const QRegularExpressionMatch match = reg.match(candidate);
32  const QString cn(match.hasMatch() ? match.captured(0) : candidate);
33  const CCountryList countries = this->findBy([&](const CCountry &country) {
34  return country.matchesCountryName(cn);
35  });
36 
37  if (countries.isEmpty()) { return this->findFirstByAlias(cn); }
38  if (countries.size() < 2) { return countries.frontOrDefault(); }
39 
40  // find best match by further reducing
41  for (const CCountry &c : countries)
42  {
43  if (c.getName() == cn) { return c; }
44  if (c.getName().startsWith(cn, Qt::CaseInsensitive)) { return c; }
45  if (cn.startsWith(c.getName(), Qt::CaseInsensitive)) { return c; }
46  }
47  return countries.front();
48  }
