213  template <class ObjectType>
214  bool compareForModelSort(const ObjectType &a, const ObjectType &b, Qt::SortOrder order, const BlackMisc::CPropertyIndex &index, const BlackMisc::CPropertyIndexList &tieBreakers, std::true_type)
215  {
216  const int c = a.comparePropertyByIndex(index, b);
217  if (c == 0)
218  {
219  if (!tieBreakers.isEmpty())
220  {
221  return compareForModelSort<ObjectType>(a, b, order, tieBreakers.front(), tieBreakers.copyFrontRemoved(), std::true_type());
222  }
223  return false;
224  }
225  return (order == Qt::AscendingOrder) ? (c < 0) : (c > 0);
226  }
227 
229  template <typename ObjectType>
230  bool compareForModelSort(const ObjectType &a, const ObjectType &b, Qt::SortOrder order, const BlackMisc::CPropertyIndex &index, const BlackMisc::CPropertyIndexList &tieBreakers, std::false_type)
231  {
232  const BlackMisc::CVariant aQv = a.propertyByIndex(index);
233  const BlackMisc::CVariant bQv = b.propertyByIndex(index);
234  if (!tieBreakers.isEmpty() && aQv == bQv)
235  {
236  return compareForModelSort<ObjectType>(a, b, order, tieBreakers.front(), tieBreakers.copyFrontRemoved(), std::false_type());
237  }
238  return (order == Qt::AscendingOrder) ? (aQv < bQv) : (bQv < aQv);
239  }
