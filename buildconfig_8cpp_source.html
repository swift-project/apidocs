93  const QString &CBuildConfig::compiledWithInfo(bool shortVersion)
94  {
95  if (shortVersion)
96  {
97  static QString infoShort;
98  if (infoShort.isEmpty())
99  {
100  QStringList sl;
101  if (CBuildConfig::isCompiledWithBlackCore()) { sl << "BlackCore"; }
102  if (CBuildConfig::isCompiledWithBlackSound()) { sl << "BlackSound"; }
103  if (CBuildConfig::isCompiledWithBlackInput()) { sl << "BlackInput"; }
104  if (CBuildConfig::isCompiledWithGui()) { sl << "BlackGui"; }
105  if (CBuildConfig::isCompiledWithFs9Support()) { sl << "FS9"; }
106  if (CBuildConfig::isCompiledWithFsxSupport()) { sl << "FSX"; }
107  if (CBuildConfig::isCompiledWithXPlaneSupport()) { sl << "XPlane"; }
108  if (CBuildConfig::isCompiledWithP3DSupport()) { sl << "P3D"; }
109  if (CBuildConfig::isCompiledWithFGSupport()) { sl << "FG"; }
110  infoShort = sl.join(", ");
111  if (infoShort.isEmpty()) { infoShort = "<none>"; }
112  }
113  return infoShort;
114  }
115  else
116  {
117  static QString infoLong;
118  if (infoLong.isEmpty())
119  {
120  infoLong = infoLong.append("BlackCore: ").append(boolToYesNo(isCompiledWithBlackCore()));
121  infoLong = infoLong.append(" BlackInput: ").append(boolToYesNo(isCompiledWithBlackInput()));
122  infoLong = infoLong.append(" BlackSound: ").append(boolToYesNo(isCompiledWithBlackSound()));
123  infoLong = infoLong.append(" GUI: ").append(boolToYesNo(isCompiledWithGui()));
124 
125  infoLong = infoLong.append(" FS9: ").append(boolToYesNo(isCompiledWithFs9Support()));
126  infoLong = infoLong.append(" FSX: ").append(boolToYesNo(isCompiledWithFsxSupport()));
127  infoLong = infoLong.append(" P3D: ").append(boolToYesNo(isCompiledWithP3DSupport()));
128  infoLong = infoLong.append(" XPlane: ").append(boolToYesNo(isCompiledWithXPlaneSupport()));
129  infoLong = infoLong.append(" FG: ").append(boolToYesNo(isCompiledWithFGSupport()));
130  }
131  return infoLong;
132  }
133  }
