123  void CAtcStationTreeView::customMenu(const QPoint &point)
124  {
125  if (!this->stationModel()) { return; }

127 
128  QMenu *menu = new QMenu(this); // menu

134 
135  connect(com1, &QAction::triggered, this, &CAtcStationTreeView::tuneInAtcCom1);
136  connect(com2, &QAction::triggered, this, &CAtcStationTreeView::tuneInAtcCom2);
137  connect(text, &QAction::triggered, this, &CAtcStationTreeView::requestTextMessage);
138  connect(resize, &QAction::triggered, this, &CAtcStationTreeView::fullResizeToContentsImpl);
139 
140  menu->addAction(com1);
141  menu->addAction(com2);
142  menu->addAction(text);
143  menu->addSeparator();
144  menu->addAction(resize);
145 
146  menu->popup(this->viewport()->mapToGlobal(point));

149  void CAtcStationTreeView::storeState()
150  {
151  m_expanded.clear();
152  for (int row = 0; row < this->model()->rowCount(); ++row)
153  {
154  const QModelIndex i = this->model()->index(row, 0);
155  const bool expanded = this->isExpanded(i);
156  const QString suffix = this->suffixForIndex(i);
157  m_expanded.insert(suffix, expanded);
158  }
159  }
160 
161  void CAtcStationTreeView::restoreState()
162  {
163  for (int row = 0; row < this->model()->rowCount(); ++row)
164  {
165  const QModelIndex i = this->model()->index(row, 0);
166  const QString suffix = this->suffixForIndex(i); // suffix of new data
167  const bool expanded = m_expanded.value(suffix, true); // default expanded
168  this->setExpanded(i, expanded);
169  }
170  }
