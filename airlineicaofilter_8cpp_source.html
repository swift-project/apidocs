23  for (const CAirlineIcaoCode &icao : inContainer)
24  {
25  if (m_id >= 0)
26  {
27  // search only for id
28  if (icao.isLoadedFromDb() && icao.getDbKey() == m_id)

31  break;
32  }
33  continue;
34  }
35 
36  if (!m_vDesignator.isEmpty())
37  {
38  // based on T72, also find VLHA based on LHA
39  if (m_vDesignator.length() == 4)
40  {
41  if (!this->stringMatchesFilterExpression(icao.getVDesignator(), m_vDesignator)) { continue; }
42  }
43  else
44  {
45  if (!this->stringMatchesFilterExpression(icao.getDesignator(), m_vDesignator)) { continue; }
46  }
47  }
48 
49  if (!this->m_name.isEmpty())

51  if (!this->stringMatchesFilterExpression(icao.getName(), this->m_name)) { continue; }
52  }
53 
54  if (!this->m_countryIso.isEmpty())
55  {
56  if (icao.getCountryIso() != m_countryIso) { continue; }
57  }
58 
59  if (!m_real && m_va)

62  }
63
