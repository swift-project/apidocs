80  void CSimConnectSettingsComponent::checkSwiftSimConnectCfgFile()

85  ui->le_ExistsSimConnectCfg->setText(file.exists() ? fileName : "no file");
86 
87  // only works for local file (which the SimConnect file normally is)

94  {
95  ui->le_Address->setText(address);

98  {
99  ui->le_Port->setText(QString::number(port));

103  void CSimConnectSettingsComponent::testSwiftSimConnectConnection()
104  {
105  const QString address = ui->le_Address->text().trimmed();
106  const QString port = ui->le_Port->text().trimmed();
107 
108  if (address.isEmpty() || port.isEmpty())
109  {
110  QMessageBox::warning(qApp->activeWindow(), tr("Connection invalid"), tr("Address and/or port not specified!"));
111  return;
112  }
113  if (!CNetworkUtils::isValidIPv4Address(address))
114  {
115  QMessageBox::warning(qApp->activeWindow(), tr("Connection invalid"), tr("Wrong IPv4 address!"));
116  return;
117  }
118  if (!CNetworkUtils::isValidPort(port))
119  {
120  QMessageBox::warning(qApp->activeWindow(), tr("Connection invalid"), tr("Invalid port!"));
121  return;
122  }
123  const int p = port.toInt();
124  QString msg;
125  if (!CNetworkUtils::canConnect(address, p, msg))
126  {
127  QMessageBox::warning(qApp->activeWindow(), tr("Connection invalid"), msg);
128  return;
129  }
130 
131  QMessageBox::information(qApp->activeWindow(), tr("Connection successful"), tr("Connected to '%1:%2'.").arg(address, port));
132  }

134  void CSimConnectSettingsComponent::saveSimConnectCfgFile()
135  {
136  const QString address = ui->le_Address->text().trimmed();
137  const QString port = ui->le_Port->text().trimmed();
138 
139  if (address.isEmpty() || port.isEmpty())
140  {
141  QMessageBox::warning(qApp->activeWindow(), tr("Connection invalid"), tr("Address and/or port not specified!"));
142  return;
143  }
144  if (!CNetworkUtils::isValidIPv4Address(address))
145  {
146  QMessageBox::warning(qApp->activeWindow(), tr("Connection invalid"), tr("Wrong IPv4 address!"));
147  return;
148  }
149  if (!CNetworkUtils::isValidPort(port))
150  {
151  QMessageBox::warning(qApp->activeWindow(), tr("Connection invalid"), tr("Invalid port!"));
152  return;
153  }
154 
155  const int p = port.toInt();
156  QString fileName;
157 
158  if (sGui->getIContextSimulator())
159  {
160  const BlackMisc::Simulation::CSimulatorInternals internals(sGui->getIContextSimulator()->getSimulatorInternals());
161  fileName = internals.getStringValue("fsx/SimConnectCfgFilename");
162  }
163 
164  if (fileName.isEmpty())
165  {
166  fileName = CSimConnectUtilities::getSwiftLocalSimConnectCfgFilename();
167  }
168 
169  if (fileName.isEmpty())
170  {
171  QMessageBox::warning(qApp->activeWindow(), tr("Failed writing SimConnect.cfg"), tr("No file name specified!"));
172  return;
173  }
174 
175  if (sGui->getIContextApplication()->writeToFile(fileName, CSimConnectUtilities::simConnectCfg(address, p)))
176  {
177  QMessageBox::information(qApp->activeWindow(), tr("File saved"), tr("File '%1' saved.").arg(fileName));
178  this->checkSwiftSimConnectCfgFile();
179  }
180  else
181  {
182  QMessageBox::warning(qApp->activeWindow(), tr("Failed writing SimConnect.cfg"), tr("Failed writing '%1'!").arg(fileName));
183  }
184  }

186  void CSimConnectSettingsComponent::saveSimConnectIniFileDialog()
187  {
188  const QString iniFile = ui->pte_SimConnectIni->toPlainText();
189  const QString dir = CSimConnectUtilities::getSimConnectIniFileDirectory(m_simulator);
190  bool madeDir = false;
191  QDir d(dir);
192  if (!d.exists())
193  {
194  d.mkdir(dir);
195  madeDir = true;
196  }
197 
198  const QString defaultFileName = CFileUtils::appendFilePaths(dir, CSimConnectUtilities::simConnectIniFilename());
199  const QString fileName = QFileDialog::getSaveFileName(this, tr("Save SimConnect.ini"),
200  defaultFileName, tr("FSX/P3D ini files (*.ini)"));
201  if (!fileName.isEmpty())
202  {
203  const bool written = CFileUtils::writeStringToFile(iniFile, fileName);
204  if (!written && madeDir) { d.removeRecursively(); } // clean up own created dir
205  if (written) { CLogMessage(this).info(u"Written '%1'") << fileName; }
206  }
207  // always refresh info, as the dialog can also be used to delete the file
208  this->setSimConnectInfo();
209  }
