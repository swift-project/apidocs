134 void CSwiftCore::disconnectFromNetwork()
135 {
136  if (!sGui || sGui->isShuttingDown()) { return; }
137  if (!sGui->getIContextNetwork()) { return; }
138  if (!sGui->getIContextNetwork()->isConnected()) { return; }
139 
140  const QMessageBox::StandardButton reply = QMessageBox::question(this, "Disconnect", "Disconnect from network?", QMessageBox::Yes | QMessageBox::No);
141  if (reply != QMessageBox::Yes) { return; }
142  sGui->getIContextNetwork()->disconnectFromNetwork();
143 }
