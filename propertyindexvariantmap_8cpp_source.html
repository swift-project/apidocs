31  bool CPropertyIndexVariantMap::matchesVariant(const CVariant &variant) const
32  {
33  if (this->isEmpty()) { return this->isWildcard(); }
34  const auto &map = this->map();
35  for (auto it = map.begin(); it != map.end(); ++it)
36  {
37  // QVariant cannot be compared directly
38  const CVariant p = variant.propertyByIndex(it.key()); // from value object
39  const CVariant v = it.value(); // from map
40  if (p != v) return false;
41  }
42  return true;
43  }

45  QString CPropertyIndexVariantMap::convertToQString(bool i18n) const
46  {
47  if (this->isEmpty()) return QStringLiteral("{wildcard: %1}").arg(m_wildcard ? "true" : "false");
48  QString s;
49  for (const CPropertyIndex &index : makeKeysRange(m_values))
50  {
51  CVariant v = m_values.value(index);
52 
53  s.isEmpty() ?
54  s.append("{wildcard: ").append(m_wildcard ? "true" : "false").append(" ") :
55  s.append(", ");
56 
57  s.append('{').append(index.toQString(i18n)).append(": ");
58  s.append("(").append(QString::number(v.userType())).append(") ");
59  QString vs = v.toQString(i18n);
60  s.append(vs);
61  s.append('}');
62  }
63  s = s.append("}");
64  return s;
65  }
