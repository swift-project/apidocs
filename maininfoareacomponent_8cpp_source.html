108  void CMainInfoAreaComponent::selectArea(CMainInfoAreaComponent::InfoArea infoArea)
109  {
110  CInfoArea::selectArea(static_cast<int>(infoArea));
111 
112  const Qt::KeyboardModifiers km = QGuiApplication::queryKeyboardModifiers();
113  const bool shift = km.testFlag(Qt::ShiftModifier);
114  if (!shift) { return; }
115 
116  // pressing shift will go to overview
117  if (infoArea == InfoAreaSettings)
118  {
119  ui->comp_Settings->setSettingsOverviewTab();
120  }
121  else if (infoArea == InfoAreaAircraft)
122  {
123  ui->comp_Aircraft->setTab(CAircraftComponent::TabAircraftInRange);
124  }
125  else if (infoArea == InfoAreaAtc)
126  {
127  ui->comp_AtcStations->setTab(CAtcStationComponent::TabAtcOnline);
128  }
129  else if (infoArea == InfoAreaMapping)
130  {
131  ui->comp_Mappings->setTab(CMappingComponent::TabRenderedAircraft);
132  }
133  }
