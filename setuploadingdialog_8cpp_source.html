127  void CSetupLoadingDialog::displaySetupCacheInfo()
128  {
129  if (this->hasSetupReader())

131  // reset if it was temporarily ignored
132  const CSetupReader *sr = sApp->getSetupReader();
133  const QDateTime setupTs = sr->getSetupCacheTimestamp();
134  static const QDateTime zeroTime = QDateTime::fromMSecsSinceEpoch(0);
135  ui->le_SetupCache->setText(setupTs.isValid() && setupTs > zeroTime ?
136  setupTs.toString(Qt::ISODateWithMs) :
137  "No cache timestamp");
138  }
139  else
140  {
141  ui->le_SetupCache->setText("No setup reader");
142  }
143 
144  const bool hasCachedSetup = this->hasCachedSetup();
145  ui->pb_LoadFromDisk->setEnabled(!hasCachedSetup);
146  ui->pb_LoadFromDisk->setToolTip(hasCachedSetup ? "Cached setup already available" : "No cached setup");
147  ui->pb_TryToFix->setEnabled(!hasCachedSetup);
