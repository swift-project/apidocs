125  void CLoginAdvComponent::toggleNetworkConnection()
126  {
127  if (!sGui || sGui->isShuttingDown()) { return; }
128  if (!sGui->getIContextNetwork() || !sGui->getIContextAudio()) { return; }
129 
130  const bool isConnected = sGui && sGui->getIContextNetwork()->isConnected();
131  const bool vatsimLogin = ui->comp_NetworkDetails->isVatsimServerSelected();
132  m_networkSetup.setAutoLogoff(ui->cb_AutoLogoff->isChecked());
133 
134  ui->form_Pilot->setVatsimValidation(vatsimLogin);
135 
136  CServer currentServer; // used for login
137  CSimulatedAircraft ownAircraft; // used own aircraft
138  CStatusMessage msg;
139  if (!isConnected)
140  {
141  const CStatusMessageList aircraftMsgs = ui->comp_OwnAircraft->validate();
142  if (aircraftMsgs.isFailure())
143  {
144  this->showOverlayHTMLMessage(CStatusMessage(this).validationWarning(u"Invalid aircraft data, login not possible"), OverlayMessageMs);
145  return;
146  }
147 
148  const CStatusMessageList pilotMsgs = ui->form_Pilot->validate();
149  if (pilotMsgs.isFailure())
150  {
151  // this->showOverlayHTMLMessage(CStatusMessage(this).validationWarning(u"Invalid pilot data, login not possible"), OverlayMessageMs);
152  this->showOverlayMessagesOrHTMLMessage(pilotMsgs, false, OverlayMessageMs);
153  return;
154  }
155 
156  // sync values with GUI values
157  const COwnAircraftComponent::CGuiAircraftValues values = ui->comp_OwnAircraft->getAircraftValuesFromGui();
158  this->updateOwnAircraftCallsignAndPilotFromGuiValues();
159  ui->comp_OwnAircraft->updateOwnAircaftIcaoValuesFromGuiValues();
160 
161  // Login mode
162  const CLoginMode mode = ui->comp_NetworkDetails->getLoginMode();
163  switch (mode.getLoginMode())
164  {
165  case CLoginMode::Observer: CLogMessage(this).info(u"Login in observer mode"); break;
166  default: break; // INetwork::LoginNormal
167  }
168 
169  // Server
170  currentServer = this->getCurrentServer();
171  const CUser user = this->getUserFromPilotGuiValues();
172  currentServer.setUser(user);
173 
174  // FSD setup, then override
175  if (ui->comp_NetworkDetails->isFsdSetupOverrideEnabled())
176  {
177  const CFsdSetup fsd = ui->comp_NetworkDetails->getFsdSetup();
178  currentServer.setFsdSetup(fsd);
179  }
180 
181  if (ui->comp_NetworkDetails->isVoiceSetupOverrideEnabled())
182  {
183  const CVoiceSetup voice = ui->comp_NetworkDetails->getVoiceSetup();
184  currentServer.setVoiceSetup(voice);
185  }
186 
187  // update for own aircraft context
188  sGui->getIContextOwnAircraft()->updateOwnAircraftPilot(currentServer.getUser());
189 
190  // set own aircraft from all values
191  ownAircraft = sGui->getIContextOwnAircraft()->getOwnAircraft();
192 
193  // check the copilot stuff
194  CCallsign partnerCs;
195  if (ui->comp_NetworkDetails->hasPartnerCallsign())
196  {
197  partnerCs = ui->comp_NetworkDetails->getPartnerCallsign();
198  if (partnerCs == ownAircraft.getCallsign())
199  {
200  this->showOverlayHTMLMessage("Your callsign and the pilot/copilot callsign must be NOT the same", OverlayMessageMs);
201  return;
202  }
203 
204  const bool ok = (partnerCs.asString().startsWith(ownAircraft.getCallsignAsString(), Qt::CaseInsensitive) || ownAircraft.getCallsignAsString().startsWith(partnerCs.asString(), Qt::CaseInsensitive));
205  if (!ok)
206  {
207  this->showOverlayHTMLMessage("Callsign and the pilot/copilot callsign appear not to be synchronized", OverlayMessageMs);
208  return;
209  }
210  }
211 
212  // Login
213  if (sGui && sGui->getCContextAudioBase())
214  {
215  sGui->getCContextAudioBase()->setVoiceSetup(currentServer.getVoiceSetup());
216  }
217 
218  msg = sGui->getIContextNetwork()->connectToNetwork(currentServer, values.ownLiverySend, values.useLivery, values.ownAircraftModelStringSend, values.useModelString, partnerCs, mode);
219  if (msg.isSuccess())
220  {
221  Q_ASSERT_X(currentServer.isValidForLogin(), Q_FUNC_INFO, "invalid server");
222  sGui->setExtraWindowTitle(QStringLiteral("[%1]").arg(ownAircraft.getCallsignAsString()));
223  CCrashHandler::instance()->crashAndLogInfoUserName(currentServer.getUser().getRealNameAndId());
224  CCrashHandler::instance()->crashAndLogInfoFlightNetwork(currentServer.getEcosystem().toQString(true));
225  CCrashHandler::instance()->crashAndLogAppendInfo(currentServer.getServerSessionId(false));
226  m_networkSetup.setLastServer(currentServer);
227  if (vatsimLogin) { m_networkSetup.setLastVatsimServer(currentServer); }
228  }
229  else
230  {
231  sGui->setExtraWindowTitle("");
232  }
233  }
234  else
235  {
236  // disconnect from network
237  sGui->setExtraWindowTitle("");
238  msg = sGui->getIContextNetwork()->disconnectFromNetwork();
239  }
240 
241  // log message and trigger events
242  msg.addCategories(this);
243  CLogMessage::preformatted(msg);
244  if (msg.isSuccess())
245  {
246  this->setGuiLoginAsValues(ownAircraft);
247  emit this->loginOrLogoffSuccessful();
248  }
249  else
250  {
251  emit this->loginOrLogoffCancelled();
252  }
253  }

324  void CLoginAdvComponent::autoLogoffDetection()
325  {
326  if (!ui->cb_AutoLogoff->isChecked()) { return; }
327  if (!this->hasValidContexts()) { return; }
328  if (!sGui->getIContextNetwork()->isConnected()) { return; } // nothing to logoff
329 
330  const CStatusMessage m = CStatusMessage(this, CStatusMessage::SeverityInfo, u"Auto logoff in progress (could be simulator shutdown, crash, closing simulator)");
331  const int delaySecs = 30;
332  this->showOverlayHTMLMessage(m, qRound(1000 * delaySecs * 0.8));
333 
334  emit this->requestLoginPage();

337  void CLoginAdvComponent::autoLogoffFrameRate(bool fatal)
338  {
340  if (!ui->cb_AutoLogoff->isChecked()) { return; }
341  if (!this->hasValidContexts()) { return; }
342  if (!sGui->getIContextNetwork()->isConnected()) { return; }
343 
344  const auto msg = fatal ? CStatusMessage(this, CStatusMessage::SeverityError, u"Sim frame rate too low to maintain constant simulation rate. Disconnecting to avoid disrupting the network.") : CStatusMessage(this, CStatusMessage::SeverityWarning, u"Sim frame rate too low to maintain constant simulation rate. Reduce graphics quality to avoid disconnection.");
345  const int delaySecs = 30;
346  this->showOverlayHTMLMessage(msg, qRound(1000 * delaySecs * 0.8));
347 
348  emit this->requestLoginPage();

351  bool CLoginAdvComponent::updateOwnAircraftCallsignAndPilotFromGuiValues()
352  {
353  if (!this->hasValidContexts()) { return false; }
354  CSimulatedAircraft ownAircraft(sGui->getIContextOwnAircraft()->getOwnAircraft());
355  const CCallsign cs = ui->comp_OwnAircraft->getCallsignFromGui();
356  bool changedCallsign = false;
357  if (!cs.isEmpty() && ownAircraft.getCallsign() != cs)
358  {
359  sGui->getIContextOwnAircraft()->updateOwnCallsign(cs);
360  ownAircraft.setCallsign(cs); // also update
361  changedCallsign = true;
362  }
363  CUser pilot = ownAircraft.getPilot();
364  const CUser uiUser = ui->form_Pilot->getUser();
365  pilot.setRealName(uiUser.getRealName());
366  pilot.setHomeBase(uiUser.getHomeBase());
367  pilot.setId(uiUser.getId());
368  pilot.setCallsign(cs);
369  bool changedPilot = false;
370  if (ownAircraft.getPilot() != pilot)
371  {
372  // it can be that the callsign was changed and this results in unchanged here
373  changedPilot = sGui->getIContextOwnAircraft()->updateOwnAircraftPilot(pilot);
374  }
375  return changedCallsign || changedPilot;
376  }
