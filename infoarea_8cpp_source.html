575  void CInfoArea::tabifyAllWidgets()
576  {
577  if (!sGui || sGui->isShuttingDown()) { return; }
578  this->setTabPosition(Qt::LeftDockWidgetArea, QTabWidget::East);
579  const bool init = m_tabBar ? false : true;
580 
581  for (int i = 0; i < m_dockWidgetInfoAreas.size(); i++)
582  {
583  CDockWidgetInfoArea *first = i > 0 ? m_dockWidgetInfoAreas.at(i - 1) : nullptr;
584  CDockWidgetInfoArea *after = m_dockWidgetInfoAreas.at(i);
585  Q_ASSERT(after);
586 
587  // trick, init widget as floating
588  // this completely initializes the tab bar and all docked widgets
589  if (init)
590  {
591  // float
592  const QPoint offset(i * 10, i * 10);
593  // after->setVisible(false);
594  // after->setFloating(true);
595  after->setOffsetWhenFloating(offset, after->isFrameless());
596  const QSize floatingSize = this->getPreferredSizeWhenFloating(i);
597  after->setPreferredSizeWhenFloating(floatingSize);
598  after->initialFloating();
599 
600  // dock again
601  // after->setFloating(false);
602  // after->setVisible(true);
603 
604  // reset floating flag, we want new resizing and position for first real floating
605  after->resetWasAlreadyFloating();
606  }
607  else
608  {
609  after->setFloating(false);
610  }
611 
612  // we can not tabify first
613  if (!first) { continue; }
614 
615  this->tabifyDockWidget(first, after);
616  }
617 
618  // as now tabified, set tab
619  if (init)
620  {
621  m_tabBar = this->findChild<QTabBar *>();
622 
623  // if we have > 1 docked widgets, we have a tab bar
624  if (m_tabBar && sGui)
