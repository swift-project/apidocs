98  void CDbLoginComponent::onLoginClicked()
99  {
100  const QString un(ui->le_Username->text().trimmed());
101  const QString pw(ui->le_Password->text().trimmed());
102  const CStatusMessageList msgs = m_loginService.login(un, pw);
103 
104  if (msgs.hasWarningOrErrorMessages())
105  {
106  CLogMessage::preformatted(msgs);
107  displayOverlayMessages(msgs);
108  return;
109  }
110  else if (!msgs.isEmpty())
111  {
112  CLogMessage::preformatted(msgs);
113  }
114  this->showLoading(5000);

140  void CDbLoginComponent::setUserInfo(const CAuthenticatedUser &user)
141  {
142  if (!sGui || sGui->isShuttingDown()) { return; }
143  if (user.isAuthenticated())
144  {
145  CLogMessage(this).info(u"User authenticated: %1") << user.toQString();
146  this->setModeLogin(false);
147  ui->le_Name->setText(user.getRealNameAndId());
148  ui->te_Roles->setPlainText(user.getRolesAsString());
149  if (user.canDirectlyWriteModels())
150  {
151  ui->le_Info->setText("You can directly update models");
152  }
153  else
154  {
155  ui->le_Info->setText("You can create model change requests");
156  }
157 
158  // crashpad info
159  CCrashHandler::instance()->crashAndLogInfoUserName(user.getRealNameAndId());
160  CCrashHandler::instance()->crashAndLogAppendInfo(QStringLiteral("Login as user %1 %2").arg(user.getRealNameAndId(), user.getRolesAsString()));
161  }
162  else
163  {
164  ui->le_Name->clear();
165  ui->te_Roles->clear();
166  this->setModeLogin(true);
167  }
