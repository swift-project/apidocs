98  bool CAircraftModelStringCompleter::setSimulator(const CSimulatorInfo &simulator)
99  {
100  if (simulator.isSingleSimulator()) { return false; }
101  if (this->getSimulator() == simulator) { return false; }
102  m_currentSimulator = simulator;
103  QPointer<CAircraftModelStringCompleter> myself(this);
104  QTimer::singleShot(100, this, [=] {
105  if (!myself) { return; }
106  this->setCompleter(true);
107  });
108  return true;
109  }

121  void CAircraftModelStringCompleter::setCompleter(bool simChanged)
122  {
123  if (!sGui || sGui->isShuttingDown()) { return; }
124  QStringList modelStrings;
125  CompleterSourceFlag dataSource = None;
126  QString simInfo = m_currentSimulator.toQString();
127  if (ui->rb_Db->isChecked())
128  {
129  if (!simChanged && m_currentDataSource == DB) { return; }
130  modelStrings = sGui->getWebDataServices()->getModelCompleterStrings();
131  dataSource = DB;
132  simInfo = QStringLiteral("DB models");
133  }
134  else if (ui->rb_ModelSet->isChecked() && sGui && sGui->getIContextSimulator())
135  {
136  if (!simChanged && m_currentDataSource == ModelSet) { return; }
137  modelStrings = sGui->getIContextSimulator()->getModelSetCompleterStrings(true);
138  dataSource = ModelSet;
139  }
140  else if (ui->rb_OwnModels->isChecked())
141  {
142  if (!simChanged && m_currentDataSource == OwnModels) { return; }
143  modelStrings = m_modelCaches.getCachedModels(this->getSimulator()).toCompleterStrings();
144  dataSource = OwnModels;
145  }
146 
147  m_currentDataSource = dataSource;
148  ui->le_modelString->setCompleter(new QCompleter(modelStrings, this));
149  ui->le_modelString->setPlaceholderText(QStringLiteral("model strings (%1/%2)").arg(modelStrings.size()).arg(simInfo));
150  }
