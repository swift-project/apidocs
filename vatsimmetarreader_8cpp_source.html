72  void CVatsimMetarReader::read()
73  {
74  this->threadAssertCheck();
75  if (!this->doWorkCheck()) { return; }
76  if (!this->isInternetAccessible("No network/internet access, cannot read METARs")) { return; }
77  if (this->isNotVATSIMEcosystem()) { return; }
78 
79  CFailoverUrlList urls(sApp->getVatsimMetarUrls());
80  const CUrl url(urls.obtainNextWorkingUrl(true));
81  if (url.isEmpty()) { return; }
82  Q_ASSERT_X(sApp, Q_FUNC_INFO, "No Application");
83  this->getFromNetworkAndLog(url.withAppendedQuery("id=all"), { this, &CVatsimMetarReader::decodeMetars });
84  }

86  void CVatsimMetarReader::decodeMetars(QNetworkReply *nwReplyPtr)
87  {
88  // wrap pointer, make sure any exit cleans up reply
89  // required to use delete later as object is created in a different thread
90  QScopedPointer<QNetworkReply, QScopedPointerDeleteLater> nwReply(nwReplyPtr);
91 
92  // Worker thread, make sure to write thread safe!
93  this->threadAssertCheck();
94  if (this->isNotVATSIMEcosystem()) { return; }
95 
96  if (!this->doWorkCheck())
97  {
98  CLogMessage(this).info(u"Terminated METAR decoding process"); // for users
99  return; // stop, terminate straight away, ending thread
100  }
101 
102  this->logNetworkReplyReceived(nwReplyPtr);
103  const QUrl url = nwReply->url();
104  const QString metarUrl = url.toString();
105 
106  if (nwReply->error() == QNetworkReply::NoError)
107  {
108  QString metarData = nwReply->readAll();
109  nwReply->close(); // close asap
110 
111  if (metarData.isEmpty()) // Quick check by hash
112  {
113  CLogMessage(this).warning(u"No METAR data from '%1', skipped") << metarUrl;
114  return;
115  }
116  if (!this->didContentChange(metarData)) // Quick check by hash
117  {
118  CLogMessage(this).info(u"METAR file from '%1' has same content, skipped") << metarUrl;
119  return;
120  }
121 
122  CMetarList metars;
123  int invalidLines = 0;
124  QTextStream lineReader(&metarData);
125  while (!lineReader.atEnd())
126  {
127  if (!this->doWorkCheck()) { return; }
128  const QString line = lineReader.readLine();
129  // some check for obvious errors
130  if (line.contains("<html")) { continue; }
131  const CMetar metar = m_metarDecoder.decode(line);
132  if (metar != CMetar())
133  {
134  metars.push_back(metar);
135  }
136  else
137  {
138  invalidLines++;
139  }
140  }
141 
142  CLogMessage(this).info(u"METARs: %1 Metars (invalid %2) from '%3'") << metars.size() << invalidLines << metarUrl;
143  {
144  QWriteLocker l(&m_lock);
145  m_metars = metars;
146  }
147 
148  emit metarsRead(metars);
149  emit dataRead(CEntityFlags::MetarEntity, CEntityFlags::ReadFinished, metars.size(), url);
150  }
151  else
152  {
153  // network error
154  CLogMessage(this).warning(u"Reading METARs failed '%1' for '%2'") << nwReply->errorString() << metarUrl;
155  nwReply->abort();
156  emit this->dataRead(CEntityFlags::MetarEntity, CEntityFlags::ReadFailed, 0, url);
157  }
158  } // method
