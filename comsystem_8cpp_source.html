119  int CComSystem::round8_33kHzChannel(int fKHz)
120  {
121  if (!isValid8_33kHzChannel(fKHz))
122  {
123  const int diff = static_cast<int>(fKHz) % 5;
124  int lower = fKHz - diff;
125  if (!isValid8_33kHzChannel(lower)) { lower -= 5; }
126  Q_ASSERT_X(isValid8_33kHzChannel(lower), Q_FUNC_INFO, "Lower frequency not valid");
127 
128  int upper = fKHz + (5 - diff);
129  if (!isValid8_33kHzChannel(upper)) { upper += 5; }
130  Q_ASSERT_X(isValid8_33kHzChannel(upper), Q_FUNC_INFO, "Upper frequency not valid");
131 
132  const int lowerDiff = abs(fKHz - lower);
133  const int upperDiff = abs(fKHz - upper);
134 
135  fKHz = lowerDiff < upperDiff ? lower : upper;
136  fKHz = std::clamp(fKHz, 118000, 136990);
137  }
138  return fKHz;
139  }

141  void CComSystem::roundToChannelSpacing(CFrequency &frequency, ChannelSpacing channelSpacing)
142  {
143  if (frequency.isNull()) { return; }
144  const double channelSpacingKHz = CComSystem::channelSpacingToFrequencyKHz(channelSpacing);
145  const double fKHz = frequency.valueRounded(CFrequencyUnit::kHz(), 0);
146 
147  if (channelSpacing == ChannelSpacing8_33KHz)
148  {
149  const int freqKHz = round8_33kHzChannel(fKHz);
150  frequency.switchUnit(CFrequencyUnit::kHz());
151  frequency.setCurrentUnitValue(freqKHz);
152  }
153  else
154  {
155  const int dDown = static_cast<int>(fKHz / channelSpacingKHz);
156  const double fDownKHz = dDown * channelSpacingKHz;
157  const double fUpKHz = (dDown + 1) * channelSpacingKHz;
158  const bool down = qAbs(fKHz - fDownKHz) < qAbs(fUpKHz - fKHz); // which is the closest value
159  const double fMHz(CMathUtils::round((down ? fDownKHz : fUpKHz) / 1000.0, 3));
160  frequency.switchUnit(CFrequencyUnit::MHz());
161  frequency.setCurrentUnitValue(fMHz);
162  }
163  }

178  bool CComSystem::isSameFrequency(const CFrequency &freq1, const CFrequency &freq2)
179  {
180  using namespace BlackMisc::PhysicalQuantities::Literals;
181  if (freq1.isNull() || freq2.isNull()) { return false; }
182 
183  // Normalize .x20 => .x25 and .70 => .x75
184  auto normalize = [](CFrequency freq) {
185  const int freq_end = static_cast<int>(freq.value(CFrequencyUnit::kHz())) % 100;
186  if (freq_end == 20 || freq_end == 70)
187  {
188  freq += 5_kHz;
189  }
190  return freq;
191  };
192 
193  CFrequency normalized_freq1 = normalize(freq1);
194  CFrequency normalized_freq2 = normalize(freq2);
195 
196  if (normalized_freq1 == normalized_freq2) { return true; } // shortcut for many of such comparisons
197 
198  // Avoid precision errors in Hz range
199  return normalized_freq1.valueInteger(CFrequencyUnit::kHz()) == normalized_freq2.valueInteger(CFrequencyUnit::kHz());
200  }

202  CFrequency CComSystem::parseComFrequency(const QString &input, CPqString::SeparatorMode sep)
203  {
204  if (input.isEmpty()) { return CFrequency::null(); }
205  CFrequency comFreq;
206  if (isDigitsOnlyString(input))
207  {
208  const double f = input.toDouble();
209  comFreq = CFrequency(f, f > 999 ? CFrequencyUnit::kHz() : CFrequencyUnit::MHz());
210  }
211  else
212  {
213  comFreq.parseFromString(input, sep);
214  if (comFreq.isNull())
215  {
216  bool ok;
217  const double f = CPqString::parseNumber(input, ok, sep);
218  if (ok)
219  {
220  comFreq = CFrequency(f, f > 999 ? CFrequencyUnit::kHz() : CFrequencyUnit::MHz());
221  }
222  else
223  {
224  comFreq = CFrequency::null();
225  }
226  }
227  }
228 
229  if (comFreq.isNull()) { return CFrequency::null(); }
230  roundToChannelSpacing(comFreq, ChannelSpacing8_33KHz);
231  return isValidComFrequency(comFreq) ? comFreq : CFrequency::null();
232  }
