145  void CDbDistributorSelectorComponent::dropEvent(QDropEvent *event)
146  {
147  if (!event || !acceptDrop(event->mimeData())) { return; }
148  CVariant valueVariant(toCVariant(event->mimeData()));
149  if (valueVariant.isValid())
150  {
151  if (valueVariant.canConvert<CDistributor>())
152  {
153  CDistributor distributor(valueVariant.value<CDistributor>());
154  if (!distributor.hasValidDbKey()) { return; }
155  this->setDistributor(distributor);
156  }
157  else if (valueVariant.canConvert<CDistributorList>())
158  {
159  CDistributorList distributors(valueVariant.value<CDistributorList>());
160  if (distributors.isEmpty()) { return; }
161  this->setDistributor(distributors.front());
162  }
163  }
164  }

166  void CDbDistributorSelectorComponent::onDistributorsRead(CEntityFlags::Entity entity, CEntityFlags::ReadState readState, int count)
167  {
168  if (!sGui || sGui->isShuttingDown() || !sGui->hasWebDataServices()) { return; }
169  if (entity.testFlag(CEntityFlags::DistributorEntity) && CEntityFlags::isFinishedReadState(readState))
170  {
171  if (count > 0)
172  {
173  const QStringList keysAndAliases(sGui->getWebDataServices()->getDistributors().getDbKeysAndAliases(true));
174  QCompleter *c = new QCompleter(keysAndAliases, this);
175  c->setCaseSensitivity(Qt::CaseInsensitive);
176  c->setCompletionMode(QCompleter::PopupCompletion);
177  c->setMaxVisibleItems(10);
178  connect(c, qOverload<const QString &>(&QCompleter::activated), this, &CDbDistributorSelectorComponent::onCompleterActivated);
179 
180  ui->le_Distributor->setCompleter(c);
181  m_completerDistributors.reset(c); // deletes any old completer
182  this->setReadOnly(false);
183  }
184  else
185  {
186  m_completerDistributors.reset(nullptr);
187  this->setReadOnly(true);
188  }
189  }
190  }
