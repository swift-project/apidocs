36  CAircraftModelList CAircraftModelFilter::filter(const CAircraftModelList &inContainer) const
37  {
38  if (!this->isEnabled()) { return inContainer; }
39  CAircraftModelList outContainer;
40  for (const CAircraftModel &model : inContainer)
41  {
42  if (m_id >= 0)
43  {
44  // search only for id
45  if (model.isLoadedFromDb() && model.getDbKey() == m_id)
46  {
47  outContainer.push_back(model);
48  break;
49  }
50  continue;
51  }
52 
53  if (!m_simulatorInfo.isAllSimulators())
54  {
55  if (!m_simulatorInfo.matchesAny(model.getSimulator())) { continue; }
56  }
57 
58  if (!m_modelKey.isEmpty())
59  {
60  if (!this->stringMatchesFilterExpression(model.getModelString(), m_modelKey)) { continue; }
61  }
62 
63  if (m_military != Qt::PartiallyChecked)
64  {
65  if (m_military == Qt::Checked)
66  {
67  // military only
68  if (!model.isMilitary()) { continue; }
69  }
70  else if (m_military == Qt::Unchecked)
71  {
72  // civilian only
73  if (model.isMilitary()) { continue; }
74  }
75  }
76 
77  if (m_colorLiveries != Qt::PartiallyChecked)
78  {
79  if (m_colorLiveries == Qt::Checked)
80  {
81  // only color liveries
82  if (!model.getLivery().isColorLivery()) { continue; }
83  }
84  else if (m_colorLiveries == Qt::Unchecked)
85  {
86  // Only airline liveries
87  if (model.getLivery().isColorLivery()) { continue; }
88  }
89  }
90 
91  if (!m_description.isEmpty())
92  {
93  if (!this->stringMatchesFilterExpression(model.getDescription(), m_description)) { continue; }
94  }
95 
96  if (m_modelMode != CAircraftModel::All && m_modelMode != CAircraftModel::Undefined)
97  {
98  if (!model.matchesMode(m_modelMode)) { continue; }
99  }
100 
101  if (m_dbKeyFilter != BlackMisc::Db::All && m_dbKeyFilter != BlackMisc::Db::Undefined)
102  {
103  if (!model.matchesDbKeyState(m_dbKeyFilter)) { continue; }
104  }
105 
106  if (!m_fileName.isEmpty())
107  {
108  if (!this->stringMatchesFilterExpression(model.getFileName(), m_fileName)) { continue; }
109  }
110 
111  if (!m_aircraftIcao.isEmpty())
112  {
113  if (!this->stringMatchesFilterExpression(model.getAircraftIcaoCodeDesignator(), m_aircraftIcao)) { continue; }
114  }
115 
116  if (!m_aircraftManufacturer.isEmpty())
117  {
118  if (!this->stringMatchesFilterExpression(model.getAircraftIcaoCode().getManufacturer(), m_aircraftManufacturer)) { continue; }
119  }
120 
121  if (!m_airlineIcao.isEmpty())
122  {
123  if (!this->stringMatchesFilterExpression(model.getAirlineIcaoCodeDesignator(), m_airlineIcao)) { continue; }
124  }
125 
126  if (!m_airlineName.isEmpty())
127  {
128  if (!this->stringMatchesFilterExpression(model.getAirlineIcaoCode().getName(), m_airlineName)) { continue; }
129  }
130 
131  if (!m_liveryCode.isEmpty())
132  {
133  if (!this->stringMatchesFilterExpression(model.getLivery().getCombinedCode(), m_liveryCode)) { continue; }
134  }
135 
136  if (m_distributor.hasValidDbKey())
137  {
138  if (!model.getDistributor().matchesKeyOrAlias(m_distributor)) { continue; }
139  }
140 
141  if (!m_combinedType.isEmpty())
142  {
143  if (!model.getAircraftIcaoCode().matchesCombinedType(m_combinedType)) { continue; }
144  }
145 
146  outContainer.push_back(model);
147  }
148  return outContainer;
149  }

151  bool CAircraftModelFilter::valid() const
152  {
153  const bool allEmpty =
154  m_id < 0 &&
155  m_modelKey.isEmpty() && m_description.isEmpty() &&
156  m_aircraftManufacturer.isEmpty() && m_aircraftIcao.isEmpty() &&
157  m_airlineIcao.isEmpty() && m_airlineName.isEmpty() &&
158  m_liveryCode.isEmpty() && m_fileName.isEmpty() && m_combinedType.isEmpty();
159  if (!allEmpty) { return true; }
160  const bool noSim = m_simulatorInfo.isNoSimulator() || m_simulatorInfo.isAllSimulators();
161  const bool noModelMode = (m_modelMode == CAircraftModel::Undefined || m_modelMode == CAircraftModel::All);
162  const bool noDbState = (m_dbKeyFilter == BlackMisc::Db::Undefined || m_dbKeyFilter == BlackMisc::Db::All);
163  const bool noKey = !m_distributor.hasValidDbKey();
164  const bool noColorRestriction = (m_colorLiveries == Qt::PartiallyChecked);
165  const bool noMilitary = (m_military == Qt::PartiallyChecked);
166  return !(noSim && noModelMode && noDbState && noKey && noMilitary && noColorRestriction);
167  }
