320

324  if (str == "CAPS") return ClientQueryType::Capabilities;
325  if (str == "C?") return ClientQueryType::Com1Freq;
326  if (str == "RN") return ClientQueryType::RealName;
327  if (str == "SV") return ClientQueryType::Server;
328  if (str == "ATIS") return ClientQueryType::ATIS;
329  if (str == "IP") return ClientQueryType::PublicIP;
330  if (str == "INF") return ClientQueryType::INF;
331  if (str == "FP") return ClientQueryType::FP;
332  if (str == "ACC") return ClientQueryType::AircraftConfig;
333 
334  // intentionally ignored (ATC ONLY)
335  // discussion: https://discordapp.com/channels/539048679160676382/539925070550794240/669186795790925848
336  if (str == "BC") return ClientQueryType::Unknown;
337  if (str == "BY") return ClientQueryType::Unknown; // CCP_Break
338  if (str == "DI") return ClientQueryType::Unknown; // CCP_ASRC_DI
339  if (str == "DP") return ClientQueryType::Unknown; // CCP_PushToDepartureList
340  if (str == "DR") return ClientQueryType::Unknown; // CCP_DropTrack
341  if (str == "FA") return ClientQueryType::Unknown;
342  if (str == "HC") return ClientQueryType::Unknown; // CCP_HandoffCancelled
343  if (str == "HI") return ClientQueryType::Unknown; // CCP_NoBreak
344  if (str == "HT") return ClientQueryType::Unknown; // CCP_HandoffTo
345  if (str == "ID") return ClientQueryType::Unknown; // CCP_ASRC_ID
346  if (str == "IH") return ClientQueryType::Unknown; // CCP_IHave
347  if (str == "IT") return ClientQueryType::Unknown; // CCP_StartTrack
348  if (str == "PT") return ClientQueryType::Unknown; // CCP_Pointout
349  if (str == "SC") return ClientQueryType::Unknown;
350  if (str == "ST") return ClientQueryType::Unknown; // CCP_PushStrip
351  if (str == "TA") return ClientQueryType::Unknown;
352  if (str == "VT") return ClientQueryType::Unknown;
