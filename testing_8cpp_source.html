41  CAtcStation CTesting::createStation(int index, bool byPropertyIndex)
42  {
43  // from WGS is slow, so static const (only 1 time init)
44  // https://dev.vatsim-germany.org/issues/322#note-2
45  static const CCoordinateGeodetic geoPos = CCoordinateGeodetic::fromWgs84("48° 21′ 13″ N", "11° 47′ 09″ E", CAltitude(index, CLengthUnit::ft()));
46  const QString cs = QStringLiteral("%1MI-SNO_TWR").arg(index);
47  const QString usr = QStringLiteral("Joe %1").arg(index);
48  const QString id = QStringLiteral("00000%1").arg(index).right(6);
49  const double f = 118.0 + (index % 30) * 0.25;
50 
51  const QDateTime dtFrom = QDateTime::currentDateTimeUtc();
52  const QDateTime dtUntil = dtFrom.addSecs(60 * 60); // 1 hour
53  const CUser user(id, usr);
54  const CInformationMessage atis(CInformationMessage::ATIS, "ATIS for " + QString::number(index));
55  const CInformationMessage metar(CInformationMessage::METAR, "Metar for " + QString::number(index));
56 
57  CAtcStation station;
58  if (byPropertyIndex)
59  {
60  station.setPropertyByIndex(CAtcStation::IndexCallsign, CVariant::from(CCallsign(cs)));
61  station.setPropertyByIndex(CAtcStation::IndexController, CVariant::from(user));
62  station.setPropertyByIndex(CAtcStation::IndexFrequency, CVariant::from(CFrequency(f, CFrequencyUnit::MHz())));
63  station.setPropertyByIndex(CAtcStation::IndexRange, CVariant::from(CLength(50, CLengthUnit::km())));
64  station.setPropertyByIndex(CAtcStation::IndexPosition, CVariant::from(geoPos));
65  station.setPropertyByIndex(CAtcStation::IndexIsOnline, CVariant::from(false));
66  station.setPropertyByIndex(CAtcStation::IndexBookedFrom, CVariant::from(dtFrom));
67  station.setPropertyByIndex(CAtcStation::IndexBookedUntil, CVariant::from(dtUntil));
68  station.setPropertyByIndex(CAtcStation::IndexRelativeDistance, CVariant::from(CLength(index + 1, CLengthUnit::NM())));
69  station.setPropertyByIndex(CAtcStation::IndexAtis, CVariant::from(atis));
70  station.setPropertyByIndex(CAtcStation::IndexMetar, CVariant::from(metar));
71  }
72  else
73  {
74  station = CAtcStation(CCallsign(cs, CCallsign::Atc), user,
75  CFrequency(f, CFrequencyUnit::MHz()),
76  geoPos, CLength(50, CLengthUnit::km()), false, dtFrom, dtUntil);
77  station.setRelativeDistance(CLength(index + 1, CLengthUnit::NM()));
78  station.setAtis(atis);
79  station.setMetar(metar);
80  }
81  return station;
82  }
