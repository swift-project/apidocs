26  void CAfvMapReader::updateFromMap()
27  {
28  if (!sApp || !sApp->getNetworkAccessManager() || sApp->isShuttingDown()) { return; }
29 
30  QEventLoop loop(sApp);
31  connect(sApp->getNetworkAccessManager(), &QNetworkAccessManager::finished, &loop, &QEventLoop::quit);
32  connect(sApp, &CApplication::aboutToShutdown, &loop, &QEventLoop::quit);
33  QNetworkReply *reply = sApp->getNetworkAccessManager()->get(QNetworkRequest(QUrl("https://voice1.vatsim.uk/api/v1/network/online/callsigns")));
34  while (reply && !reply->isFinished() && sApp && !sApp->isShuttingDown())
35  {
36  loop.exec();
37  }
38  const QByteArray jsonData = reply ? reply->readAll() : QByteArray {};
39  if (reply) { reply->deleteLater(); }
40 
41  if (jsonData.isEmpty()) { return; }
42 
43  QJsonDocument jsonDoc = QJsonDocument::fromJson(jsonData);
44  if (jsonDoc.isArray())
45  {
46  const QJsonArray rootArray = jsonDoc.array();
47  QVector<CSampleAtcStation> transceivers;
48  QString callsign;
49 
50  for (auto it = rootArray.begin(); it != rootArray.end(); ++it)
51  {
52  if (it->isObject())
53  {
54  const QJsonObject stationObject = it->toObject();
55 
56  if (stationObject.contains("callsign"))
57  {
58  callsign = stationObject.value("callsign").toString();
59  }
60 
61  if (callsign.isEmpty() || !CCallsign::looksLikeAtcCallsign(callsign)) { continue; }
62 
63  if (stationObject.contains("transceivers"))
64  {
65  QJsonArray txArray = stationObject.value("transceivers").toArray();
66  for (auto jt = txArray.begin(); jt != txArray.end(); ++jt)
67  {
68  const TransceiverDto transceiver = TransceiverDto::fromJson(jt->toObject());
69  transceivers.push_back({ callsign, transceiver });
70  }
71  }
72  }
73  }
74 
75  if (transceivers.isEmpty()) { return; }
76  transceivers.erase(std::remove_if(transceivers.begin(), transceivers.end(), [this](const CSampleAtcStation &s) {
77  return s.callsign() == m_callsign;
78  }),
79  transceivers.end());
80  m_model->updateAtcStations(transceivers);
81  }
82  }
