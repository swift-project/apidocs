94 void SwiftGuiStd::attachSimulatorWindow()
95 {
96  this->activateWindow(); // attaching requires active window
97  QWindow *w = CForeignWindows::getFirstFoundSimulatorWindow();
98  if (!w)
99  {
100  CLogMessage(this).warning(u"No simulator window found");
101  return;
102  }
103  const bool a = CForeignWindows::setSimulatorAsParent(w, this);
104  if (a)
105  {
106  CLogMessage(this).info(u"Attached to simulator");
107  }
108  else
109  {
110  CLogMessage(this).warning(u"No simulator window found");
111  }
112 }

126 void SwiftGuiStd::initMenus()
127 {
128  Q_ASSERT_X(ui->menu_InfoAreas, Q_FUNC_INFO, "No menu");
129  Q_ASSERT_X(ui->menu_Window, Q_FUNC_INFO, "No menu");
130  Q_ASSERT_X(ui->comp_MainInfoArea, Q_FUNC_INFO, "no main area");
131  sGui->addMenuFile(*ui->menu_File);
132  sGui->addMenuInternals(*ui->menu_Internals);
133  sGui->addMenuWindow(*ui->menu_Window);
134  ui->menu_Window->addSeparator();
135  QAction *a = ui->menu_Window->addAction("Attach simulator window");
136  bool c = connect(a, &QAction::triggered, this, &SwiftGuiStd::attachSimulatorWindow);
137  Q_ASSERT_X(c, Q_FUNC_INFO, "connect failed");
138  a = ui->menu_Window->addAction("Detach simulator window");
139  c = connect(a, &QAction::triggered, this, &SwiftGuiStd::detachSimulatorWindow);
140  Q_ASSERT_X(c, Q_FUNC_INFO, "connect failed");
141 
142  sGui->addMenuHelp(*ui->menu_Help);
143  ui->menu_InfoAreas->addActions(ui->comp_MainInfoArea->getInfoAreaSelectActions(true, ui->menu_InfoAreas));
144 
145  if (CBuildConfig::isLocalDeveloperDebugBuild() && ui->menu_File && ui->menu_File->actions().size() > 5)
146  {
147  QAction *act = new QAction(CIcons::swift16(), "Copy XSwiftBus dialog");
148  ui->menu_File->insertAction(ui->menu_File->actions().at(5), act);
149  c = connect(
150  act, &QAction::triggered, this, [=] {
151  this->copyXSwiftBusDialog(false);
152  },
153  Qt::QueuedConnection);
154  Q_ASSERT_X(c, Q_FUNC_INFO, "connect failed");
155  }
156 
157  // for hotkeys
158  const QString swift(CGuiActionBindHandler::pathSwiftPilotClient());
159  static const CActionBind swiftRoot(swift, CIcons::Swift16); // inserts action for root folder
160  Q_UNUSED(swiftRoot)
161  m_menuHotkeyHandlers.append(CGuiActionBindHandler::bindMenu(ui->menu_InfoAreas, swift + "Info areas"));
162  m_menuHotkeyHandlers.append(CGuiActionBindHandler::bindMenu(ui->menu_File, swift + "File"));
163  m_menuHotkeyHandlers.append(CGuiActionBindHandler::bindMenu(ui->menu_Window, swift + "Window"));
164 }

166 void SwiftGuiStd::copyXSwiftBusDialog(bool checkFileTimestamp)
167 {
168  const QString xPlaneRootDir = ui->comp_MainInfoArea->getSettingsComponent()->getSimulatorSettings(CSimulatorInfo::XPLANE).getSimulatorDirectoryOrDefault();
169  const bool xpDirExists = !xPlaneRootDir.isEmpty() && QDir().exists(xPlaneRootDir);
170  if (!xpDirExists)
171  {
172  if (checkFileTimestamp) { return; }
173  QMessageBox::warning(this, tr("Copy XSwiftBus"), tr("XPlane directory does not exists!"), QMessageBox::Close);
174  return;
175  }
176 
177  const int c = CCopyXSwiftBusDialog::displayDialogAndCopyBuildFiles(xPlaneRootDir, checkFileTimestamp, this);
178  if (c > 0) { CLogMessage(this).info(u"Copied %1 files from build directory") << c; }
179 }
