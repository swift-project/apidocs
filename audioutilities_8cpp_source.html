83  QAudioDeviceInfo getLowestLatencyDevice(const CAudioDeviceInfo &device, QAudioFormat &format)
84  {
85  if (device.isDefault() || !device.isValid())
86  {
87  const QAudioDeviceInfo defDevice = device.isInputDevice() ? QAudioDeviceInfo::defaultInputDevice() : QAudioDeviceInfo::defaultOutputDevice();
88  if (!defDevice.isFormatSupported(format))
89  {
90  format = defDevice.nearestFormat(format);
91  }
92  return defDevice;
93  }
94 
95  const QList<QAudioDeviceInfo> allQtDevices =
96  device.isInputDevice() ?
97  CAudioDeviceInfoList::allQtInputDevices() :
98  CAudioDeviceInfoList::allQtOutputDevices();
99 
100  // Find the one with lowest latency.
101  QList<QAudioDeviceInfo> supportedDevices;
102  for (const QAudioDeviceInfo &d : allQtDevices)
103  {
104  if (d.deviceName() == device.getName())
105  {
106  if (!d.isFormatSupported(format))
107  {
108  // Check whether the nearest format is acceptable for our needs
109  const QAudioFormat nearestFormat = d.nearestFormat(format);
110  if (nearestFormat.sampleRate() != format.sampleRate() ||
111  nearestFormat.sampleSize() != format.sampleSize() ||
112  nearestFormat.sampleType() != format.sampleType() ||
113  nearestFormat.byteOrder() != format.byteOrder() ||
114  nearestFormat.codec() != format.codec())
115  {
116  continue;
117  }
118  }
119  supportedDevices.push_back(d);
120  }
121  }
122 
123  if (supportedDevices.isEmpty()) { return {}; }
124 
125  QAudioDeviceInfo deviceWithLowestLatency = supportedDevices.at(0);
126 
127  if (supportedDevices.size() > 1)
128  {
129  QAudioFormat nearestFormat = format;
130  int lowestBufferSize = std::numeric_limits<int>::max();
131  for (const QAudioDeviceInfo &d : supportedDevices)
132  {
133  int bufferSize = 0;
134  if (device.getType() == CAudioDeviceInfo::InputDevice)
135  {
136  QAudioInput input(d, d.nearestFormat(format));
137  input.start();
138  input.stop();
139  bufferSize = input.bufferSize();
140  }
141  else
142  {
143  QAudioOutput output(d, d.nearestFormat(format));
144  output.start();
145  output.stop();
146  bufferSize = output.bufferSize();
147  }
148 
149  if (bufferSize < lowestBufferSize)
150  {
151  deviceWithLowestLatency = d;
152  nearestFormat = d.nearestFormat(format);
153  lowestBufferSize = bufferSize;
154  }
155  }
156  format = nearestFormat;
157  }
158  return deviceWithLowestLatency;
159  }

161  QAudioDeviceInfo getHighestCompatibleOutputDevice(const CAudioDeviceInfo &device, QAudioFormat &format)
162  {
163  if (device.isDefault()) { return CAudioDeviceInfoList::defaultQtOutputDevice(); }
164  const QList<QAudioDeviceInfo> allQtDevices = CAudioDeviceInfoList::allQtOutputDevices();
165 
166  QList<QAudioDeviceInfo> supportedDevices;
167  for (const QAudioDeviceInfo &d : allQtDevices)
168  {
169  if (d.deviceName() == device.getName())
170  {
171  // exact match, format supported
172  if (d.isFormatSupported(format)) { return d; }
173  }
174  supportedDevices.push_back(d);
175  }
176 
177  // no suitable device
178  if (supportedDevices.isEmpty())
179  {
180  format = QAudioFormat();
181  return QAudioDeviceInfo();
182  }
183 
184  // here we could "search the best device", currently only first is taken
185  QAudioDeviceInfo usedDevice = supportedDevices.front();
186  if (!usedDevice.isFormatSupported(format))
187  {
188  format = usedDevice.nearestFormat(format);
189  }
190  return usedDevice;
191  }
