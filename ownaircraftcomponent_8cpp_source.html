106  void COwnAircraftComponent::onSimulatorModelChanged(const CAircraftModel &model)
107  {
108  if (!sGui || !sGui->getIContextNetwork() || sApp->isShuttingDown()) { return; }
109  const bool isNetworkConnected = sGui && sGui->getIContextNetwork()->isConnected();
110  if (isNetworkConnected) { return; }
111 
112  // update with latest DB data
113  CAircraftModel reverseModel(model);
114  if (sGui->hasWebDataServices())
115  {
116  reverseModel = sGui->getWebDataServices()->getModelForModelString(model.getModelString());
117  if (!reverseModel.isLoadedFromDb()) { reverseModel = model; } // reset if not found
118  }
119 
120  const QString modelStr(reverseModel.hasModelString() ? reverseModel.getModelString() : "<unknown>");
121  if (!reverseModel.hasModelString())
122  {
123  CLogMessage(this).validationInfo(u"Invalid lookup for '%1' successful: %2") << modelStr << reverseModel.toQString();
124  CLogMessage(this).validationInfo(u"Hint: Are you using the emulated driver? Set a model if so!");
125  return;
126  }
127  this->setOwnModelAndIcaoValues(reverseModel);
128 
129  // open dialog for model mapping
130  if (m_autoPopupWizard && !reverseModel.isLoadedFromDb())
131  {
132  this->mappingWizard();
133  }
134 
135  emit this->aircraftDataChanged();
136  }

159  CStatusMessageList COwnAircraftComponent::validate() const
160  {
161  CGuiAircraftValues values = this->getAircraftValuesFromGui();
162  CStatusMessageList msgs;
163 
164  // fill in combined type if empty
165  if (ui->le_AircraftCombinedType->text().isEmpty() && values.ownAircraftIcao.isLoadedFromDb())
166  {
167  ui->le_AircraftCombinedType->setText(values.ownAircraftIcao.getCombinedType());
168  values.ownAircraftCombinedType = values.ownAircraftIcao.getCombinedType();
169  }
170 
171  const bool validCombinedType = CAircraftIcaoCode::isValidCombinedType(values.ownAircraftCombinedType);
172  ui->lblp_AircraftCombinedType->setTicked(validCombinedType);
173  if (!validCombinedType) { msgs.addValidationMessage("Invalid combined type", CStatusMessage::SeverityError); }
174 
175  // airline is optional, e.g. C172 has no airline
176  const bool validAirlineDesignator = values.ownAirlineIcao.hasValidDesignator() || values.ownAirlineIcao.getDesignator().isEmpty();
177  ui->lblp_AirlineIcao->setTicked(validAirlineDesignator);
178  if (!validAirlineDesignator) { msgs.addValidationMessage("Invalid airline designator", CStatusMessage::SeverityError); }
179 
180  const bool validAircraftDesignator = values.ownAircraftIcao.hasValidDesignator();
181  ui->lblp_AircraftIcao->setTicked(validAircraftDesignator);
182  if (!validAircraftDesignator) { msgs.addValidationMessage("Invalid aircraft designator", CStatusMessage::SeverityError); }
183 
184  const bool validCallsign = CCallsign::isValidAircraftCallsign(values.ownCallsign);
185  ui->lblp_Callsign->setTicked(validCallsign);
186  if (!validCallsign) { msgs.addValidationMessage("Invalid callsign", CStatusMessage::SeverityError); }
187 
188  // model intentionally ignored
189  // return validCombinedType && validAirlineDesignator && validAircraftDesignator && validCallsign;
190 
191  return msgs;
192  }

230  bool COwnAircraftComponent::setGuiIcaoValues(const CAircraftModel &model, bool onlyIfEmpty)
231  {
232  bool changed = false;
233  if (!onlyIfEmpty || !ui->selector_AircraftIcao->isSet())
234  {
235  changed = ui->selector_AircraftIcao->setAircraftIcao(model.getAircraftIcaoCode());
236  }
237  if (!onlyIfEmpty || !ui->selector_AirlineIcao->isSet())
238  {
239  const bool c = ui->selector_AirlineIcao->setAirlineIcao(model.getAirlineIcaoCode());
240  changed |= c;
241  }
242  if (!onlyIfEmpty || ui->le_AircraftCombinedType->text().trimmed().isEmpty())
243  {
244  const QString combined(model.getAircraftIcaoCode().getCombinedType());
245  if (ui->le_AircraftCombinedType->text() != combined)
246  {
247  ui->le_AircraftCombinedType->setText(combined);
248  changed = true;
249  }
250  }
251  if (!onlyIfEmpty || ui->le_SendLivery->text().trimmed().isEmpty())
252  {
253  const QString ls(model.getSwiftLiveryString());
254  if (ui->le_SendLivery->text() != ls)
255  {
256  ui->le_SendLivery->setText(ls);
257  changed = true;
258  }
259  }
260 
261  const bool valid = this->validate().isSuccess();
262  return valid ? changed : false;
263  }

295  void COwnAircraftComponent::setOwnModelAndIcaoValues(const CAircraftModel &ownModel)
296  {
297  if (!this->hasValidContexts()) { return; }
298  CAircraftModel model = ownModel;
299  const bool simulating = sGui->getIContextSimulator() &&
300  (sGui->getIContextSimulator()->getSimulatorStatus() & ISimulator::Simulating);
301  if (simulating)
302  {
303  if (!model.hasModelString())
304  {
305  model = sGui->getIContextOwnAircraft()->getOwnAircraft().getModel();
306  }
307  const QString modelAndKey(model.getModelStringAndDbKey());
308 
309  ui->le_SimulatorModel->setText(modelAndKey);
310  ui->le_SimulatorModel->home(false);
311  this->highlightModelField(model);
312 
313  const CSimulatorInfo sim = sGui->getIContextSimulator()->getSimulatorPluginInfo().getSimulator();
314  ui->comp_ModelStringCompleter->setSimulator(sim);
315  m_lastAircraftModel.set(model);
316 
317  const CSimulatorInternals simulatorInternals = sGui->getIContextSimulator()->getSimulatorInternals();
318  const QString simStr = sim.toQString() + QStringLiteral(" ") + simulatorInternals.getSimulatorVersion();
319  CCrashHandler::instance()->crashAndLogInfoSimulator(simStr);
320  }
321  else
322  {
323  if (!model.hasModelString())
324  {
325  model = this->getPrefillModel();
326  }
327  ui->le_SimulatorModel->clear();
328  this->highlightModelField();
329  }
330  ui->le_SimulatorModel->setToolTip(model.asHtmlSummary());
331  ui->comp_ModelStringCompleter->setText(model.getModelString());
332  ui->le_SendLivery->setText(model.getSwiftLiveryString());
333 
334  // reset the model
335  if (model.isLoadedFromDb() || (model.getAircraftIcaoCode().isLoadedFromDb() && model.getLivery().isLoadedFromDb()))
336  {
337  // full model from DB, take all values
338  this->setGuiIcaoValues(model, false);
339  }
340  else
341  {
342  if (sGui->getIContextSimulator()->isSimulatorAvailable())
343  {
344  // sim. attached, but no model data from DB
345  ui->le_AircraftCombinedType->clear();
346  ui->selector_AircraftIcao->clear();
347  ui->selector_AirlineIcao->clear();
348  }
349  }
350  }

377  bool COwnAircraftComponent::updateOwnAircaftIcaoValuesFromGuiValues()
378  {
379  if (!this->hasValidContexts()) { return false; }
380  const CSimulatedAircraft ownAircraft(sGui->getIContextOwnAircraft()->getOwnAircraft());
381  const CGuiAircraftValues aircraftValues = this->getAircraftValuesFromGui();
382 
383  CAircraftIcaoCode aircraftCode(ownAircraft.getAircraftIcaoCode());
384  CAirlineIcaoCode airlineCode(ownAircraft.getAirlineIcaoCode());
385 
386  bool changedIcaoCodes = false;
387  if (aircraftValues.ownAircraftIcao.hasValidDesignator() && aircraftValues.ownAircraftIcao != aircraftCode)
388  {
389  aircraftCode = aircraftValues.ownAircraftIcao;
390  changedIcaoCodes = true;
391  }
392  if (aircraftValues.ownAirlineIcao.hasValidDesignator() && aircraftValues.ownAirlineIcao != airlineCode)
393  {
394  airlineCode = aircraftValues.ownAirlineIcao;
395  changedIcaoCodes = true;
396  }
397 
398  if (changedIcaoCodes)
399  {
400  sGui->getIContextOwnAircraft()->updateOwnIcaoCodes(aircraftCode, airlineCode);
401  }
402 
403  return changedIcaoCodes;
404  }
