253  void CDockWidget::toggleFloating()
254  {
255  const bool changeToFloating = !this->isFloating();
256  if (!changeToFloating) { this->setFrameless(false); } // remove frameless if not floating
257 
258  // disable the interactive docking
259  if (changeToFloating)
260  {
261  m_originalAreas = this->allowedAreas();
262  this->setAllowedAreas(Qt::NoDockWidgetArea);
263  }
264  else
265  {
266  this->setAllowedAreas(m_originalAreas);
267  }
268 
269  this->setAlwaysOnTop(m_alwaysOnTop && changeToFloating);
270  this->setFloating(changeToFloating);
271 
272  const Qt::KeyboardModifiers km = QGuiApplication::queryKeyboardModifiers();
273  const bool shift = km.testFlag(Qt::ShiftModifier);
274 
275  // with shift ALWAYS reset
276  if (shift && changeToFloating) { this->resetPosition(); }
277  if (changeToFloating)
278  {
279  // check where we are, otherwise reset if NOT appropriate
280  const QPoint p = this->rect().topLeft();
281  if (p.x() < 1 || p.y() < 1)
282  {
283  this->resetPosition();
284  }
285  }
286  }

340  bool CDockWidget::restoreFromSettings()
341  {
342  const CDockWidgetSettings s = this->getSettings();
343  if (s.isFloating() != this->isFloating())
344  {
345  this->toggleFloating();
346  QPointer<CDockWidget> myself(this);
347  QTimer::singleShot(500, this, [=] {
348  if (myself) { myself->restoreFromSettings(); }
349  });
350  return true;
351  }
352 
353  if (s.isFramless() != this->isFrameless())
354  {
355  this->toggleFrameless();
356  QPointer<CDockWidget> myself(this);
357  QTimer::singleShot(500, this, [=] {
358  if (myself) { myself->restoreFromSettings(); }
359  });
360  return true;
361  }
362 
363  // now frameless and floating is correct
364  const QByteArray geo(s.getGeometry());
365  if (!geo.isEmpty())
366  {
367  const bool ok = this->restoreGeometry(geo);
368  if (ok) { this->rememberFloatingSizeAndPosition(); }
369  }
370 
371  this->setMargins();
372  return true;
373  }

486  void CDockWidget::onTopLevelChanged(bool topLevel)
487  {
488 #ifdef Q_OS_LINUX
489  // Give XCB platforms enough time to handle window events before adjusting it.
490  // close T593 if this is no longer needed
491  QThread::msleep(100);
492 #endif
493 
494  this->setMargins(); // from settings or default
495  if (topLevel)
496  {
497  if (m_windowTitleBackup != QDockWidget::windowTitle())
498  {
499  QDockWidget::setWindowTitle(m_windowTitleBackup);
500  }
501  this->setNullTitleBarWidget();
502  if (m_wasFrameless) { this->setFrameless(true); }
503 
516  m_statusBar.show();
517  m_wasAlreadyFloating = true;
518  }
519  else
520  {
521  // frameless
522  this->setFrameless(false);
523 
524  if (!m_windowTitleWhenDocked) { QDockWidget::setWindowTitle(""); }
525  m_statusBar.hide();
526  this->setEmptyTitleBar();
527 
528  // sometimes floating sets a new minimum size, here we reset it
529  if (this->minimumHeight() > m_initialDockedMinimumSize.height())
530  {
531  this->setMinimumSize(m_initialDockedMinimumSize);
532  }
533  }
534 
535  // relay
536  emit this->widgetTopLevelChanged(this, topLevel);
537  }

546  void CDockWidget::initStatusBarAndProperties()
547  {
548  if (m_statusBar.getStatusBar()) { return; }
549 
550  // Typical reasons for asserts here
551  // 1) Check the structure, we expect the following hierarchy:
552  // QDockWidget (CDockWidget/CDockWidgetInfoArea) -> QWidget (outer widget) -> QFrame (inner widget)
553  // Structure used for frameless floating windows
554  // 2) Check if the "floating" flag is accidentally set for the dock widget in the GUI builder
555  // 3) Is the dock widget promoted BlackGui::CDockWidgetInfoArea?
556  QWidget *outerWidget = this->widget(); // the outer widget containing the layout
557  Q_ASSERT_X(outerWidget, "CDockWidget::initStatusBar", "No outer widget");
558  if (!outerWidget) { return; }
559  outerWidget->setProperty("dockwidget", propertyOuterWidget());
560 
561  Q_ASSERT_X(outerWidget->layout(), "CDockWidget::initStatusBar", "No outer widget layout");
562  if (!outerWidget->layout()) { return; }
563  Q_ASSERT_X(outerWidget->layout()->itemAt(0) && outerWidget->layout()->itemAt(0)->widget(), "CDockWidget::initStatusBar", "No outer widget layout item");
564  if (!outerWidget->layout()->itemAt(0) || !outerWidget->layout()->itemAt(0)->widget())
565  {
566  m_allowStatusBar = false;
567  return;
568  }
569 
570  // Inner widget is supposed to be a QFrame / promoted QFrame
571  QFrame *innerWidget = qobject_cast<QFrame *>(outerWidget->layout()->itemAt(0)->widget()); // the inner widget containing the layout
572  Q_ASSERT_X(innerWidget, "CDockWidget::initStatusBar", "No inner widget");
573  if (!innerWidget)
574  {
575  m_allowStatusBar = false;
576  return;
577  }
578  innerWidget->setProperty("dockwidget", propertyInnerWidget());
579 
580  // status bar
581  if (!m_allowStatusBar) { return; }
582  m_statusBar.initStatusBar();
583 
584  // layout
585  QVBoxLayout *vLayout = qobject_cast<QVBoxLayout *>(innerWidget->layout());
586  Q_ASSERT_X(vLayout, "CDockWidget::initStatusBar", "No outer widget layout");
587  if (!vLayout)
588  {
589  m_allowStatusBar = false;
590  return;
591  }
592  vLayout->addWidget(m_statusBar.getStatusBar(), 0, Qt::AlignBottom); // 0->vertical stretch minimum
593 
594  // adjust stretching of the original widget. It was the only widget so far
595  // and should occupy maximum space
596  QWidget *compWidget = innerWidget->findChild<QWidget *>(QString(), Qt::FindDirectChildrenOnly);
597  Q_ASSERT(compWidget);
598  if (!compWidget) { return; }
599  QSizePolicy sizePolicy = compWidget->sizePolicy();
600  sizePolicy.setVerticalStretch(1); // make the original widget occupying maximum space
601  compWidget->setSizePolicy(sizePolicy);
602 
603  // hide status bar if not floating T592 no longer init floating
604  if (!this->isFloating()) { m_statusBar.hide(); }
605  }
