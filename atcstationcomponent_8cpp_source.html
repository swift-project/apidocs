187  void CAtcStationComponent::update()
188  {
189  if (!this->canAccessContext()) { return; }
190  Q_ASSERT(ui->tvp_AtcStationsBooked);
191  Q_ASSERT(ui->tvp_AtcStationsOnline);
192 
193  // check if component is visible, if we have already data then skip udpate
194  const bool hasData = this->countBookedStations() > 0 || this->countOnlineStations() > 0;
195  if (hasData && !this->isVisibleWidget())
196  {
197  // Update skipped, as not visible
198  ui->tvp_AtcStationsBooked->hideLoadIndicator();
199  ui->tvp_AtcStationsOnline->hideLoadIndicator();
200  return;
201  }
202 
203  // bookings
204  if (m_timestampBookedStationsChanged > m_timestampLastReadBookedStations)
205  {
206  this->reloadAtcStationsBooked();
207  }
208 
209  // online stations, only when connected
210  if (sGui->getIContextNetwork()->isConnected())
211  {
212  // update
213  if (m_timestampOnlineStationsChanged > m_timestampLastReadOnlineStations)
214  {
215  const CAtcStationsSettings settings = ui->comp_AtcStationsSettings->getSettings();
216  CAtcStationList onlineStations = sGui->getIContextNetwork()->getAtcStationsOnline(true);
217  const int allStationsCount = onlineStations.sizeInt();
218  int inRangeCount = -1;
219 
220  if (settings.showOnlyWithValidFrequency()) { onlineStations = onlineStations.stationsWithValidFrequency(); }
221  if (settings.showOnlyInRange())
222  {
223  onlineStations.removeIfOutsideRange();
224  inRangeCount = onlineStations.sizeInt();
225  }
226 
227  const int stationsCount = onlineStations.sizeInt();
228  ui->tvp_AtcStationsOnline->updateContainerMaybeAsync(onlineStations);
229  m_timestampLastReadOnlineStations = QDateTime::currentDateTimeUtc();
230  m_timestampOnlineStationsChanged = m_timestampLastReadOnlineStations;
231  this->updateTreeView();
232  this->setOnlineTabs(allStationsCount, stationsCount);
233  ui->comp_AtcStationsSettings->setCounts(allStationsCount, inRangeCount);
234 
235  if (stationsCount < 1 && allStationsCount > 0)
236  {
237  const QString msg = QStringLiteral("All %1 ATC stations are filtered").arg(allStationsCount);
238  ui->tvp_AtcStationsOnline->showOverlayHTMLMessage(msg, 5000);
239  ui->tvp_AtcStationsOnlineTree->showOverlayHTMLMessage(msg, 5000);
240  }
241 
242  if (stationsCount < 1)
243  {
244  m_selectedCallsign.clear();
245  }
246  else if (!m_selectedCallsign.isEmpty() && onlineStations.containsCallsign(m_selectedCallsign))
247  {
248  const CAtcStation lastSelected = onlineStations.findFirstByCallsign(m_selectedCallsign);
249  if (lastSelected.hasCallsign())
250  {
251  // avoid override of manually entered METAR callsigns
252  const CCallsign currentCs(ui->le_AtcStationsOnlineMetar->text());
253  if (currentCs.isEmpty() || currentCs == lastSelected.getCallsign())
254  {
255  this->triggerOnlineAtcStationSelected(lastSelected);
256  }
257  }
258  } // stations
259  }
260  }
261  else
262  {
263  m_selectedCallsign.clear();
264  ui->tvp_AtcStationsOnline->clear();
265  this->updateTreeView();
266  this->setOnlineTabs(0, 0);
267  }
268  }

299  void CAtcStationComponent::reloadAtcStationsBooked()
300  {
301  Q_ASSERT(ui->tvp_AtcStationsBooked);
302  if (!this->canAccessContext()) { return; }
303 
304  QObject *sender = QObject::sender();
305  if (sender == ui->tvp_AtcStationsBooked)
306  {
307  // trigger new read, which takes some time. A signal will be received when this is done
308  CLogMessage(this).info(u"Requested new bookings");
309  sGui->getIContextNetwork()->requestAtcBookingsUpdate();
310  }
311  else
312  {
313  ui->tvp_AtcStationsBooked->updateContainerMaybeAsync(sGui->getIContextNetwork()->getAtcStationsBooked(false));
314  m_timestampLastReadBookedStations = QDateTime::currentDateTimeUtc();
315  }
316  }

325  void CAtcStationComponent::changedAtcStationsBooked()
