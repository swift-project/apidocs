55  bool CBackgroundValidation::triggerValidation(const CSimulatorInfo &simulator, const QString &simDirectory)
56  {
57  const QPointer<CBackgroundValidation> myself(this);
58  if (simulator.isNoSimulator())
59  {
60  return this->requestLastValidationResults();
61  }
62 
63  {
64  QWriteLocker l(&m_lock);
65  if (m_inWork) { return false; }
66  m_simulator = simulator;
67  m_simDirectory = simDirectory;
68  m_checkedSimulatorMsgs.remove(simulator);
69  }
70  QTimer::singleShot(5, this, [=] {
71  if (!myself) { return; }
72  myself->doWork();
73  });
74  return true;
75  }

104  void CBackgroundValidation::doWork()
105  {
106  if (m_inWork) { return; }
107  m_inWork = true;
108  emit this->validating(true);
109 
110  const bool isTimerBased = (QObject::sender() == &m_updateTimer);
111  CAircraftModelList valid;
112  CAircraftModelList invalid;
113  CStatusMessageList msgs;
114  bool validated = false;
115  bool onlyErrorsAndWarnings = false;
116  const CSimulatorInfo simulator = this->getCurrentSimulator();
117  const qint64 started = QDateTime::currentMSecsSinceEpoch();
118  m_wasStopped = false;
119 
120  do
121  {
122  if (!simulator.isSingleSimulator()) { break; }
123  if (this->wasAlreadyChecked(simulator)) { break; }
124 
125  const CAircraftMatcherSetup setup = m_matchingSettings.get();
126  if (!setup.doVerificationAtStartup()) { break; }
127 
128  onlyErrorsAndWarnings = setup.onlyShowVerificationWarningsAndErrors();
129  const CAircraftModelList models = m_modelSets.getCachedModels(simulator);
130  const qint64 now = QDateTime::currentMSecsSinceEpoch();
131  validated = true;
132 
133  if (models.isEmpty())
134  {
135  msgs.push_back(CStatusMessage(this, CStatusMessage::SeverityWarning, QStringLiteral("No models in set for '%1'").arg(simulator.toQString(true))));
136  }
137  else
138  {
139  msgs = CAircraftModelUtilities::validateModelFiles(simulator, models, valid, invalid, false, 25, m_wasStopped, m_simDirectory);
140  }
141 
142  const qint64 deltaTimeMs = now - started;
143  msgs.push_back(CStatusMessage(this, CStatusMessage::SeverityInfo, QStringLiteral("Validated in %1ms").arg(deltaTimeMs)));
144  msgs.sortBySeverityHighestFirst();
145  msgs.freezeOrder();
146 
147  QWriteLocker l(&m_lock);
148  m_lastResultValid = valid;
149  m_lastResultInvalid = invalid;
150  m_lastResultWasStopped = m_wasStopped;
151  m_lastResultSimulator = simulator;
152  m_lastResultMsgs = msgs;
153  m_checkedSimulatorMsgs.insert(simulator, msgs);
154  }
155  while (false);
156 
157  m_inWork = false;
158  if (isTimerBased)
159  {
160  m_timerBasedRuns++;
161 
162  // stop timer after some runs
163  if (m_timerBasedRuns > 3)
164  {
165  m_updateTimer.stop();
166  }
167  }
168 
169  emit this->validating(false);
170  if (validated)
171  {
172  const bool e = !onlyErrorsAndWarnings || (!invalid.isEmpty() || msgs.hasWarningOrErrorMessages());
173  if (e || !isTimerBased) { emit this->validated(simulator, valid, invalid, m_wasStopped, msgs); }
174  }
175  }
176 } // ns
