73  msgBox.setStandardButtons(QMessageBox::Close | QMessageBox::Cancel);

77 
78  result = QMessageBox::Close; // msgBox.exec();

80  {

82  event->accept();
83  break;
84  case QMessageBox::Cancel:

86  break;
87  default:
