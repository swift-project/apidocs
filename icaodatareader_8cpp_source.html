168  void CIcaoDataReader::read(CEntityFlags::Entity entities, CDbFlags::DataRetrievalModeFlag mode, const QDateTime &newerThan)
169  {
170  this->threadAssertCheck(); // runs in background thread
171  if (!this->doWorkCheck()) { return; }
172  entities &= CEntityFlags::AllIcaoCountriesCategory;
173 
174  if (!this->isInternetAccessible())
175  {
176  emit this->dataRead(entities, CEntityFlags::ReadSkipped, 0, {});
177  return;
178  }
179 
180  CEntityFlags::Entity entitiesTriggered = CEntityFlags::NoEntity;
181  CUrl url;
182  if (entities.testFlag(CEntityFlags::AircraftIcaoEntity))
183  {
184  url = this->getAircraftIcaoUrl(mode);
185  if (!url.isEmpty())
186  {
187  url.appendQuery(queryLatestTimestamp(newerThan));
188  this->getFromNetworkAndLog(url, { this, &CIcaoDataReader::parseAircraftIcaoData });
189  entitiesTriggered |= CEntityFlags::AircraftIcaoEntity;
190  }
191  else
192  {
193  this->logNoWorkingUrl(CEntityFlags::AircraftIcaoEntity);
194  }
195  }
196 
197  if (entities.testFlag(CEntityFlags::AirlineIcaoEntity))
198  {
199  url = this->getAirlineIcaoUrl(mode);
200  if (!url.isEmpty())
201  {
202  url.appendQuery(queryLatestTimestamp(newerThan));
203  this->getFromNetworkAndLog(url, { this, &CIcaoDataReader::parseAirlineIcaoData });
204  entitiesTriggered |= CEntityFlags::AirlineIcaoEntity;
205  }
206  else
207  {
208  this->logNoWorkingUrl(CEntityFlags::AirlineIcaoEntity);
209  }
210  }
211 
212  if (entities.testFlag(CEntityFlags::CountryEntity))
213  {
214  url = this->getCountryUrl(mode);
215  if (!url.isEmpty())
216  {
217  url.appendQuery(queryLatestTimestamp(newerThan));
218  this->getFromNetworkAndLog(url, { this, &CIcaoDataReader::parseCountryData });
219  entitiesTriggered |= CEntityFlags::CountryEntity;
220  }
221  else
222  {
223  this->logNoWorkingUrl(CEntityFlags::CountryEntity);
224  }
225  }
226 
227  if (entities.testFlag(CEntityFlags::AircraftCategoryEntity))
228  {
229  url = this->getAircraftCategoryUrl(mode);
230  if (!url.isEmpty())
231  {
232  url.appendQuery(queryLatestTimestamp(newerThan));
233  this->getFromNetworkAndLog(url, { this, &CIcaoDataReader::parseAircraftCategoryData });
234  entitiesTriggered |= CEntityFlags::AircraftCategoryEntity;
235  }
236  else
237  {
238  this->logNoWorkingUrl(CEntityFlags::AircraftCategoryEntity);
239  }
240  }
241 
242  if (entitiesTriggered != CEntityFlags::NoEntity)
243  {
244  emit this->dataRead(entitiesTriggered, CEntityFlags::ReadStarted, 0, url);
245  }
246  }

284  void CIcaoDataReader::parseAircraftIcaoData(QNetworkReply *nwReplyPtr)
285  {
286  // wrap pointer, make sure any exit cleans up reply
287  // required to use delete later as object is created in a different thread
288  QScopedPointer<QNetworkReply, QScopedPointerDeleteLater> nwReply(nwReplyPtr);
289  if (!this->doWorkCheck()) { return; }
290 
291  const CDatabaseReader::JsonDatastoreResponse res = this->setStatusAndTransformReplyIntoDatastoreResponse(nwReply.data());
292  const QUrl url = nwReply->url();
293 
294  if (res.hasErrorMessage())
295  {
296  CLogMessage::preformatted(res.lastWarningOrAbove());
297  emit this->dataRead(CEntityFlags::AircraftIcaoEntity, CEntityFlags::ReadFailed, 0, url);
298  return;
299  }
300 
301  emit this->dataRead(CEntityFlags::AircraftIcaoEntity, CEntityFlags::ReadParsing, 0, url);
302  CAircraftIcaoCodeList codes;
303  CAircraftIcaoCodeList inconsistent;
304  const CAircraftCategoryList categories = this->getAircraftCategories();
305  if (res.isRestricted())
306  {
307  // create full list if it was just incremental
308  const CAircraftIcaoCodeList incrementalCodes(CAircraftIcaoCodeList::fromDatabaseJson(res, categories, true, &inconsistent));
309  if (incrementalCodes.isEmpty()) { return; } // currently ignored
310  codes = this->getAircraftIcaoCodes();
311  codes.replaceOrAddObjectsByKey(incrementalCodes);
312  }
313  else
314  {
315  // normally read from special DB view which already filters incomplete
316  QElapsedTimer time;
317  time.start();
318  codes = CAircraftIcaoCodeList::fromDatabaseJson(res, categories, true, &inconsistent);
319  this->logParseMessage("aircraft ICAO", codes.size(), static_cast<int>(time.elapsed()), res);
320  }
321 
322  if (!inconsistent.isEmpty())
323  {
324  logInconsistentData(
325  CStatusMessage(this, CStatusMessage::SeverityInfo, u"Inconsistent aircraft codes: " % inconsistent.dbKeysAsString(", ")),
326  Q_FUNC_INFO);
327  }
328 
329  if (!this->doWorkCheck()) { return; }
330  const int n = codes.size();
331  qint64 latestTimestamp = codes.latestTimestampMsecsSinceEpoch(); // ignores duplicates
332  if (n > 0 && latestTimestamp < 0)
333  {
334  CLogMessage(this).error(u"No timestamp in aircraft ICAO list, setting to last modified value");
335  latestTimestamp = this->lastModifiedMsSinceEpoch(nwReply.data());
336  }
337 
338  m_aircraftIcaoCache.set(codes, latestTimestamp);
339  this->updateReaderUrl(this->getBaseUrl(CDbFlags::DbReading));
340 
341  this->emitAndLogDataRead(CEntityFlags::AircraftIcaoEntity, n, res);
342  }

494  CStatusMessageList CIcaoDataReader::readFromJsonFiles(const QString &dir, CEntityFlags::Entity whatToRead, bool overrideNewerOnly)
495  {
496  const QDir directory(dir);
497  if (!directory.exists())
498  {
499  return CStatusMessage(this).error(u"Missing directory '%1'") << dir;
500  }
501 
502  // Hint: Do not emit while locked -> deadlock
503  CStatusMessageList msgs;
504  whatToRead &= CEntityFlags::AllIcaoCountriesCategory;
505  CEntityFlags::Entity reallyRead = CEntityFlags::NoEntity;
506  if (whatToRead.testFlag(CEntityFlags::CountryEntity))
507  {
508  const QString fileName = CFileUtils::appendFilePaths(directory.absolutePath(), CDbInfo::entityToSharedName(CEntityFlags::CountryEntity));
509  const QFileInfo fi(fileName);
510  if (!fi.exists())
511  {
512  msgs.push_back(CStatusMessage(this).warning(u"File '%1' does not exist") << fileName);
513  }
514  else if (!this->overrideCacheFromFile(overrideNewerOnly, fi, CEntityFlags::CountryEntity, msgs))
515  {
516  // void
517  }
518  else
519  {
520  const QUrl url = QUrl::fromLocalFile(fi.absoluteFilePath());
521  const QJsonObject countriesJson(CDatabaseUtils::readQJsonObjectFromDatabaseFile(fileName));
522  if (countriesJson.isEmpty())
523  {
524  msgs.push_back(CStatusMessage(this).error(u"Failed to read from file/empty file '%1'") << fileName);
525  }
526  else
527  {
528  try
529  {
530  const CCountryList countries = CCountryList::fromMultipleJsonFormats(countriesJson);
531  const int c = countries.size();
532  msgs.push_back(m_countryCache.set(countries, fi.birthTime().toUTC().toMSecsSinceEpoch()));
533  reallyRead |= CEntityFlags::CountryEntity;
534  emit this->dataRead(CEntityFlags::CountryEntity, CEntityFlags::ReadFinished, c, url);
535  }
536  catch (const CJsonException &ex)
537  {
538  emit this->dataRead(CEntityFlags::CountryEntity, CEntityFlags::ReadFailed, 0, url);
539  msgs.push_back(CStatusMessage::fromJsonException(ex, this, QStringLiteral("Reading countries from '%1'").arg(fileName)));
540  }
541  }
542  }
543  } // country
544 
545  if (whatToRead.testFlag(CEntityFlags::AircraftIcaoEntity))
546  {
547  const QString fileName = CFileUtils::appendFilePaths(directory.absolutePath(), CDbInfo::entityToSharedName(CEntityFlags::AircraftIcaoEntity));
548  const QFileInfo fi(fileName);
549  if (!fi.exists())
550  {
551  msgs.push_back(CStatusMessage(this).warning(u"File '%1' does not exist") << fileName);
552  }
553  else if (!this->overrideCacheFromFile(overrideNewerOnly, fi, CEntityFlags::AircraftIcaoEntity, msgs))
554  {
555  // void
556  }
557  else
558  {
559  const QUrl url = QUrl::fromLocalFile(fi.absoluteFilePath());
560  const QJsonObject aircraftJson(CDatabaseUtils::readQJsonObjectFromDatabaseFile(fileName));
561  if (aircraftJson.isEmpty())
562  {
563  msgs.push_back(CStatusMessage(this).error(u"Failed to read from file/empty file '%1'") << fileName);
564  }
565  else
566  {
567  try
568  {
569  const CAircraftIcaoCodeList aircraftIcaos = CAircraftIcaoCodeList::fromMultipleJsonFormats(aircraftJson);
570  const int c = aircraftIcaos.size();
571  msgs.push_back(m_aircraftIcaoCache.set(aircraftIcaos, fi.birthTime().toUTC().toMSecsSinceEpoch()));
572  reallyRead |= CEntityFlags::AircraftIcaoEntity;
573  emit this->dataRead(CEntityFlags::AircraftIcaoEntity, CEntityFlags::ReadFinished, c, url);
574  }
575  catch (const CJsonException &ex)
576  {
577  emit this->dataRead(CEntityFlags::AircraftIcaoEntity, CEntityFlags::ReadFailed, 0, url);
578  msgs.push_back(CStatusMessage::fromJsonException(ex, this, QStringLiteral("Reading aircraft ICAOs from '%1'").arg(fileName)));
579  }
580  }
581  }
582  } // aircraft
583 
584  if (whatToRead.testFlag(CEntityFlags::AirlineIcaoEntity))
585  {
586  const QString fileName = CFileUtils::appendFilePaths(directory.absolutePath(), CDbInfo::entityToSharedName(CEntityFlags::AirlineIcaoEntity));
587  const QFileInfo fi(fileName);
588  if (!fi.exists())
589  {
590  msgs.push_back(CStatusMessage(this).warning(u"File '%1' does not exist") << fileName);
591  }
592  else if (!this->overrideCacheFromFile(overrideNewerOnly, fi, CEntityFlags::AirlineIcaoEntity, msgs))
593  {
594  // void
595  }
596  else
597  {
598  const QUrl url = QUrl::fromLocalFile(fi.absoluteFilePath());
599  const QJsonObject airlineJson(CDatabaseUtils::readQJsonObjectFromDatabaseFile(fileName));
600  if (airlineJson.isEmpty())
601  {
602  msgs.push_back(CStatusMessage(this).error(u"Failed to read from file/empty file '%1'") << fileName);
603  }
604  else
605  {
606  try
607  {
608  const CAirlineIcaoCodeList airlineIcaos = CAirlineIcaoCodeList::fromMultipleJsonFormats(airlineJson);
