29  // Returns the last path from filePath, which does exist on the file system
30  QString getLastExistingPathFromFile(const QString &filePath)
31  {
32  const QFileInfo fileInfo(filePath);
33  if (!fileInfo.exists()) { return {}; }
34  QFile file(fileInfo.absoluteFilePath());
35  if (!file.open(QIODevice::ReadOnly))
36  {
37  return {};
38  }
39 
40  QString lastLine;
41  QTextStream ts(&file);
42  while (!ts.atEnd())
43  {
44  QString pathToCheck = ts.readLine();
45  if (QFileInfo::exists(pathToCheck)) { lastLine = pathToCheck; }
46  }
47  return lastLine;
48  }

233  QStringList CXPlaneUtil::findAllXplFiles(const QString &pluginDir)
234  {
235  const QString dirName = CFileUtils::fixWindowsUncPath(pluginDir.isEmpty() ? xplaneRootDir() : pluginDir);
236  const QDir directory(dirName);
237  if (!directory.exists()) { return QStringList(); }
238 
239  // this finds the current levels XPLs
240  QStringList files = directory.entryList(xplFileFilter(), QDir::Files, QDir::Name | QDir::IgnoreCase);
241  const QStringList dirs = directory.entryList(QDir::Dirs | QDir::NoDotAndDotDot, QDir::Name | QDir::IgnoreCase);
242  for (const QString &dir : dirs)
243  {
244  const QString subDir = CFileUtils::appendFilePaths(dirName, dir);
245  const QStringList subDirFiles = CXPlaneUtil::findAllXplFiles(subDir);
246  if (subDirFiles.isEmpty()) { continue; }
247  for (const QString &file : subDirFiles)
248  {
249  files.push_back(CFileUtils::appendFilePaths(dir, file));
250  }
251  }
252  return files;
253  }

286  CStatusMessageList CXPlaneUtil::validateModelDirectories(const QString &simDir, const QStringList &modelDirectories)
287  {
288  if (simDir.isEmpty())
289  {
290  return CStatusMessage(static_cast<CXPlaneUtil *>(nullptr), CStatusMessage::SeverityWarning, u"no simulator directory", true);
291  }
292 
293  CStatusMessageList msgs;
294  if (modelDirectories.isEmpty()) { return msgs; }
295  const QDir sd(simDir);
296  const bool simDirExists = sd.exists();
297  for (const QString &modelDir : modelDirectories)
298  {
299  const bool exists = simDirExists ?
300  CDirectoryUtils::isSameOrSubDirectoryOf(modelDir, sd) :
301  CDirectoryUtils::isSameOrSubDirectoryOfStringBased(modelDir, sd.absolutePath());
302  const CStatusMessage m = exists ?
303  CStatusMessage(static_cast<CXPlaneUtil *>(nullptr)).info(u"Model directory '%1' inside '%2'") << modelDir << sd.absolutePath() :
304  CStatusMessage(static_cast<CXPlaneUtil *>(nullptr)).error(u"Model directory '%1' NOT inside '%2'") << modelDir << sd.absolutePath();
305  msgs.push_back(m);
306  }
307  msgs.addValidationCategory();
308  return msgs;
309  }
