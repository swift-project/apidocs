50  bool CSimulatorFsCommon::parseDetails(const CSimpleCommandParser &parser)
51  {
52  // .driver fsuipc on|off
53  if (parser.matchesPart(1, "fsuipc") && parser.hasPart(2))
54  {
55  const bool on = parser.toBool(2);
56  const bool s = this->useFsuipc(on);
57  CLogMessage(this, CLogCategories::cmdLine()).info(u"FSUIPC is '%1'") << boolToOnOff(s);
58  return s;
59  }
60  return CSimulatorPluginCommon::parseDetails(parser);

94  bool CSimulatorFsCommon::useFsuipc(bool on)
95  {
96  if (!m_fsuipc) { return false; } // no FSUIPC available
97  if (m_useFsuipc == on) { return m_useFsuipc; } // nothing changed
98  m_useFsuipc = on;
99  if (on)
100  {
101  m_useFsuipc = m_fsuipc->open();
102  }
103  else
104  {
105  m_fsuipc->close();
106  }
107 
108  this->initSimulatorInternals(); // update internals
109  return m_useFsuipc;
110  }
