46  void CConfigSimulatorComponent::preselectSimulators()
47  {
48  CSimulatorInfo sims = m_enabledSimulators.get();
49  if (!sims.isAnySimulator())
50  {
51  CSimulatorInfo simWithModels = m_modelSets.simulatorsWithModels();
52  sims = simWithModels;
53  }
54 
55  // no x64 check as we would allow to config 32bit with launcher x64 and vice versa
56  const bool p3d = (sims.isP3D() || !CFsDirectories::p3dDir().isEmpty()) && CBuildConfig::isCompiledWithP3DSupport();
57  const bool fsx = (sims.isFSX() || !CFsDirectories::fsxDir().isEmpty()) && CBuildConfig::isCompiledWithFsxSupport();
58  const bool fs9 = (sims.isFS9() || !CFsDirectories::fs9Dir().isEmpty()) && CBuildConfig::isCompiledWithFs9Support();
59  const bool xp = sims.isXPlane() && CBuildConfig::isCompiledWithXPlaneSupport();
60  const bool fg = sims.isFG() && CBuildConfig::isCompiledWithFGSupport();
61 
62  ui->cb_P3D->setChecked(p3d);
63  ui->cb_FSX->setChecked(fsx);
64  ui->cb_FS9->setChecked(fs9);
65  ui->cb_XP->setChecked(xp);
66  ui->cb_FG->setChecked(fg);
67 
68  ui->cb_P3D->setEnabled(CBuildConfig::isCompiledWithP3DSupport());
69  ui->cb_FSX->setEnabled(CBuildConfig::isCompiledWithFsxSupport());
70  ui->cb_FS9->setEnabled(CBuildConfig::isCompiledWithFs9Support());
71  ui->cb_XP->setEnabled(CBuildConfig::isCompiledWithXPlaneSupport());
72  ui->cb_FG->setEnabled(CBuildConfig::isCompiledWithFGSupport());
73 
74  CGuiUtility::checkBoxReadOnly(ui->cb_P3D, !CBuildConfig::isCompiledWithP3DSupport());
75  CGuiUtility::checkBoxReadOnly(ui->cb_FSX, !CBuildConfig::isCompiledWithFsxSupport());
76  CGuiUtility::checkBoxReadOnly(ui->cb_FS9, !CBuildConfig::isCompiledWithFs9Support());
77  CGuiUtility::checkBoxReadOnly(ui->cb_XP, !CBuildConfig::isCompiledWithXPlaneSupport());
78  CGuiUtility::checkBoxReadOnly(ui->cb_FG, !CBuildConfig::isCompiledWithFGSupport());
79 
89  if (p3d) { ui->comp_SettingsSimulator->setSimulator(CSimulatorInfo(CSimulatorInfo::P3D)); }
90  else if (fsx) { ui->comp_SettingsSimulator->setSimulator(CSimulatorInfo(CSimulatorInfo::FSX)); }
91  else if (fs9) { ui->comp_SettingsSimulator->setSimulator(CSimulatorInfo(CSimulatorInfo::FS9)); }
92  else if (xp) { ui->comp_SettingsSimulator->setSimulator(CSimulatorInfo(CSimulatorInfo::XPLANE)); }
93  else if (fg) { ui->comp_SettingsSimulator->setSimulator(CSimulatorInfo(CSimulatorInfo::FG)); }
94  }

102  if (ui->cb_FSX->isChecked()) { ids << CSimulatorPluginInfo::fsxPluginIdentifier(); }
103  if (ui->cb_P3D->isChecked()) { ids << CSimulatorPluginInfo::p3dPluginIdentifier(); }
104  if (ui->cb_XP->isChecked()) { ids << CSimulatorPluginInfo::xplanePluginIdentifier(); }
105  if (ui->cb_FG->isChecked()) { ids << CSimulatorPluginInfo::fgPluginIdentifier(); }
106 
107  return ids;
108  }
