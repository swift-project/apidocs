254  {
255  QJsonValue value = json.value("associativecontainerbase");

274  jsonKey >> key;
275  }
276  {
277  CJsonScope scope("associativecontainerbase", 2 * index++ + 1);

507  template <class Map1, class Map2, class F>
508  void forEachIntersection(const Map1 &map1, const Map2 &map2, F functor)
509  {
510  static_assert(std::is_same_v<typename Map1::key_type, typename Map2::key_type>, "Maps must have the same key type");
511  if (map1.empty() || map2.empty()) { return; }
512  auto it1 = implementationOf(map1).lowerBound(map2.cbegin().key());
513  auto end1 = implementationOf(map1).upperBound((map2.cend() - 1).key());
514  auto it2 = implementationOf(map2).lowerBound(map1.cbegin().key());
515  auto end2 = implementationOf(map2).upperBound((map1.cend() - 1).key());
516  while (it1 != end1 && it2 != end2)
517  {
518  if (it1.key() < it2.key()) { ++it1; }
519  else if (it2.key() < it1.key()) { ++it2; }
520  else
521  {
522  functor(it1.key(), it1.value(), it2);
523  ++it1;
524  ++it2;
525  }
526  }
