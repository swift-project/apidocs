84  bool CCompressUtils::whichZip7(QStringList *stdOutAndError)
85  {
86  const QString cmd("which 7za");
87  QProcess zipProcess;
88  zipProcess.start(cmd);
89  if (!zipProcess.waitForStarted()) { return false; }
90  if (!zipProcess.waitForFinished()) { return false; }
91 
92  const QString pStdout = zipProcess.readAllStandardOutput();
93  const QString pStderr = zipProcess.readAllStandardError();
94  if (stdOutAndError)
95  {
96  stdOutAndError->clear();
97  stdOutAndError->push_back(pStdout);
98  stdOutAndError->push_back(pStderr);
99  }
100  const int r = zipProcess.exitCode();
101  return r == 0 && pStdout.contains("7za", Qt::CaseInsensitive);
102  }

104  bool CCompressUtils::runZip7Process(QProcess *zipProcess, QStringList *stdOutAndError)
105  {
106  zipProcess->start();
107 
108  // If process does not even start, e.g. because no 7za exe found.
109  if (!zipProcess->waitForStarted())
110  {
111  if (stdOutAndError)
112  {
113  stdOutAndError->push_back("7za");
114  stdOutAndError->push_back("Command not found");
115  }
116  return false;
117  }
118 
119  // If process does not finish. Very unlikely.
120  if (!zipProcess->waitForFinished())
121  {
122  if (stdOutAndError)
123  {
124  stdOutAndError->push_back("7za");
125  stdOutAndError->push_back("Process did not finish.");
126  }
127  return false;
128  }
129 
130  if (stdOutAndError)
131  {
132  stdOutAndError->clear();
133  const QString pStdout = zipProcess->readAllStandardOutput();
134  const QString pStderr = zipProcess->readAllStandardError();
135  stdOutAndError->push_back(pStdout);
136  stdOutAndError->push_back(pStderr);
137  }
138 
139  return zipProcess->exitStatus() == QProcess::NormalExit;
140  }
