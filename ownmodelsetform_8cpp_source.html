101  void COwnModelSetForm::changeDistributorDisplay()
102  {
103  if (ui->rb_DisplayAllDistributors->isChecked())
104  {
105  ui->tvp_Distributors->updateContainerMaybeAsync(this->getAllDistributors());
106  ui->cb_SortByPreferences->setChecked(false);
107  CGuiUtility::checkBoxReadOnly(ui->cb_SortByPreferences, true);
108  this->setDistributorView(false);
109  }
110  else
111  {
112  ui->tvp_Distributors->updateContainerMaybeAsync(this->getDistributorsFromPreferences());
113  ui->cb_SortByPreferences->setChecked(true);
114  CGuiUtility::checkBoxReadOnly(ui->cb_SortByPreferences, false);
115  this->setDistributorView(true);
116  }
117  }
