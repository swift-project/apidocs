62  void CSimulatorP3D::SimConnectProc(SIMCONNECT_RECV *pData, DWORD cbData, void *pContext)
63  {
64  CSimulatorP3D *simulatorP3D = static_cast<CSimulatorP3D *>(pContext);
65  Q_ASSERT_X(simulatorP3D, Q_FUNC_INFO, "Cannot convert context to CSimulatorP3D");
66 
67  switch (pData->dwID)
68  {
69  // case SIMCONNECT_RECV_ID_CAMERA_6DOF: break;
70  // case SIMCONNECT_RECV_ID_CAMERA_FOV: break;
71  // case SIMCONNECT_RECV_ID_CAMERA_SENSOR_MODE: break;
72  // case SIMCONNECT_RECV_ID_CAMERA_WINDOW_POSITION: break;
73  // case SIMCONNECT_RECV_ID_CAMERA_WINDOW_SIZE: break;
74 
75  case SIMCONNECT_RECV_ID_GROUND_INFO:
76  {
77  // https://www.prepar3d.com/SDKv4/sdk/simconnect_api/references/structures_and_enumerations.html#SIMCONNECT_RECV_GROUND_INFO
78  const SIMCONNECT_RECV_GROUND_INFO *pObjData = static_cast<SIMCONNECT_RECV_GROUND_INFO *>(pData);
79  const DWORD requestId = pObjData->dwRequestID;
80  if (!CSimulatorFsxCommon::isRequestForSimObjTerrainProbe(requestId)) { break; }
81  // valid elevation request
82  // https://www.prepar3d.com/SDKv4/sdk/simconnect_api/references/structures_and_enumerations.html#SIMCONNECT_DATA_GROUND_INFO
83  if (pObjData->dwArraySize != 1) { break; }
84  const SIMCONNECT_DATA_GROUND_INFO gi = pObjData->rgData[0];
85  if (!gi.bIsValid) { break; }
86  const CLatitude lat(gi.fLat, CAngleUnit::deg());
87  const CLongitude lng(gi.fLon, CAngleUnit::deg());
88  const CAltitude alt(gi.fAlt, CAltitude::MeanSeaLevel, CAltitude::TrueAltitude, CLengthUnit::ft());
89  const CCoordinateGeodetic coordinate(lat, lng, alt);
90  const CElevationPlane ep(coordinate, CElevationPlane::singlePointRadius());
91 
92  const CCallsign cs(simulatorP3D->getCallsignForPendingProbeRequests(requestId, true));
93  simulatorP3D->callbackReceivedRequestedElevation(ep, cs, false);
94  }
95  break;
96  default:
97  CSimulatorFsxCommon::SimConnectProc(pData, cbData, pContext);
98  break;
99  }
100  }

103  bool CSimulatorP3D::requestElevation(const ICoordinateGeodetic &reference, const CCallsign &callsign)
104  {
105  if (reference.isNull()) { return false; }
106  if (this->isShuttingDown()) { return false; }
107  if (!this->isConnected()) { return false; }
108 
109  Q_ASSERT_X(CThreadUtils::isInThisThread(this), Q_FUNC_INFO, "thread");
110  const bool hasHeight = reference.hasMSLGeodeticHeight();
111  const double latDeg = reference.latitude().value(CAngleUnit::deg());
112  const double lngDeg = reference.longitude().value(CAngleUnit::deg());
113  const double maxAltFt = hasHeight ? reference.geodeticHeight().value(CLengthUnit::ft()) : 50000;
114  const DWORD dwGridWidth = 1.0;
115  const DWORD dwGridHeight = 1.0;
116 
117  const SIMCONNECT_DATA_REQUEST_ID requestId = this->obtainRequestIdForSimObjTerrainProbe(); // P3D we use new request id each time (no simobject)
118 
119  // returns SIMCONNECT_RECV_GROUND_INFO -> SIMCONNECT_DATA_GROUND_INFO
120  const HRESULT hr = this->logAndTraceSendId(
121  SimConnect_RequestGroundInfo(
122  m_hSimConnect, requestId, latDeg, lngDeg, 0, latDeg, lngDeg, maxAltFt,
123  dwGridWidth, dwGridHeight,
124  SIMCONNECT_GROUND_INFO_LATLON_FORMAT_DEGREES,
125  SIMCONNECT_GROUND_INFO_ALT_FORMAT_FEET,
126  SIMCONNECT_GROUND_INFO_SOURCE_FLAG_PLATFORMS),
127  Q_FUNC_INFO, "SimConnect_RequestGroundInfo");
128 
129  bool ok = false;
130  if (isOk(hr))
131  {
132  ok = true;
133  emit this->requestedElevation(callsign);
134  m_pendingProbeRequests.insert(requestId, callsign);
135  }
136  else
137  {
138  const CStatusMessage msg = CStatusMessage(this).error(u"SimConnect, can not request ground info: '%1' '%2'") << requestId << callsign.asString();
139  CLogMessage::preformatted(msg);
140  }
141 
142  return ok;
143  }

145  bool CSimulatorP3D::followAircraft(const CCallsign &callsign)
146  {
147  if (this->isShuttingDownOrDisconnected()) { return false; }
148 
149  CSimConnectObject &simObject = m_simConnectObjects[callsign];
150  if (!simObject.hasValidRequestAndObjectId()) { return false; }
151  if (simObject.getCallsignByteArray().isEmpty()) { return false; }
152  HRESULT hr = s_false();
153 
154  //
155  // Experimental code
156  //
157 
194  // Observer is P3D only, not FSX
195  const CAircraftSituation situation = m_lastSentSituations[callsign];
196  if (situation.isNull()) { return false; }
197  SIMCONNECT_DATA_OBSERVER obs;
198  SIMCONNECT_DATA_PBH pbh;
199  pbh.Pitch = pbh.Bank = pbh.Heading = 0;
200  obs.Rotation = pbh;
201  obs.Position = coordinateToFsxLatLonAlt(situation);
202  obs.Regime = SIMCONNECT_OBSERVER_REGIME_GHOST;
203  obs.RotateOnTarget = TRUE;
204  obs.FocusFixed = TRUE;
205  obs.FieldOfViewH = 30; // deg.
206  obs.FieldOfViewV = 30; // deg.
207  obs.LinearStep = 20; // meters
208  obs.AngularStep = 10; // deg.
209 
210  const char *observerName = simObject.getCallsignByteArray().constData();
211  hr = SimConnect_CreateObserver(m_hSimConnect, observerName, obs);
212  if (isOk(hr))
213  {
214  SIMCONNECT_DATA_XYZ offset;
215  offset.x = offset.y = 30;
216  offset.z = 0;
217  hr = SimConnect_ObserverAttachToEntityOn(m_hSimConnect, observerName, simObject.getObjectId(), offset);
218  if (isOk(hr))
219  {
220  SimConnect_SetObserverLookAt(m_hSimConnect, observerName, obs.Position);
221 
222  // const QByteArray viewName = QStringLiteral("Observer %1").arg(callsign.asString()).toLatin1();
223  hr = SimConnect_OpenView(m_hSimConnect, observerName);
224 
225  simObject.setObserverName(callsign.asString());
226  }
227  }
228 
229  return isOk(hr);
230  }

267  const HRESULT hr2 = this->logAndTraceSendId(
268  SimConnect_TransmitClientEvent(m_hSimConnect, objectId, EventFreezeLatLng, 1, SIMCONNECT_GROUP_PRIORITY_HIGHEST, SIMCONNECT_EVENT_FLAG_GROUPID_IS_PRIORITY),
269  simObject, "EventFreezeLatLng", Q_FUNC_INFO, "SimConnect_TransmitClientEvent");
270  const HRESULT hr3 = this->logAndTraceSendId(
271  SimConnect_TransmitClientEvent(m_hSimConnect, objectId, EventFreezeAlt, 1, SIMCONNECT_GROUP_PRIORITY_HIGHEST, SIMCONNECT_EVENT_FLAG_GROUPID_IS_PRIORITY),
272  simObject, "EventFreezeAlt", Q_FUNC_INFO, "SimConnect_TransmitClientEvent");
273  const HRESULT hr4 = this->logAndTraceSendId(
274  SimConnect_TransmitClientEvent(m_hSimConnect, objectId, EventFreezeAtt, 1, SIMCONNECT_GROUP_PRIORITY_HIGHEST, SIMCONNECT_EVENT_FLAG_GROUPID_IS_PRIORITY),
275  simObject, "EventFreezeAtt", Q_FUNC_INFO, "SimConnect_TransmitClientEvent");
276 
277  return isOk(hr1, hr2, hr3, hr4);
