296  CStatusMessageList CFlightPlanComponent::validateAndInitializeFlightPlan(CFlightPlan &flightPlan)
297  {
298  CStatusMessageList messages;
299  const bool strict = ui->cb_StrictCheck->isChecked();
300  const bool vfr = this->isVfr();
301  const CStatusMessage::StatusSeverity severity = strict ? CStatusMessage::SeverityError : CStatusMessage::SeverityWarning;
302  messages.push_back(CStatusMessage(this).validationInfo(strict ? QStringLiteral("Strict validation") : QStringLiteral("Lenient validation")));
303 
304  const CFlightPlan::FlightRules rules = this->getFlightRules();
305  flightPlan.setFlightRule(rules);
306 
307  // callsign
308  QString v;
309  v = ui->le_Callsign->text().trimmed().toUpper();
310  if (v.isEmpty())
311  {
312  // messages.push_back(CStatusMessage(this).validationError(u"Missing '%1'") << ui->lbl_Callsign->text());
313  }
314  else if (!CCallsign::isValidAircraftCallsign(v))
315  {
316  messages.push_back(CStatusMessage(this).validationError(u"Invalid callsign '%1'") << v);
317  }
318  flightPlan.setCallsign(CCallsign(v));
319 
320  // aircraft ICAO / aircraft type
321  v = ui->le_AircraftType->text().trimmed().toUpper();
322  if (v.isEmpty())
323  {
324  messages.push_back(CStatusMessage(this).validationError(u"Missing '%1'") << ui->lbl_AircraftType->text());
325  }
326  else if (!CAircraftIcaoCode::isValidDesignator(v))
327  {
328  messages.push_back(CStatusMessage(this).validationError(u"Invalid aircraft ICAO code '%1'") << v);
329  }
330  else if (sApp && sApp->hasWebDataServices() && sApp->getWebDataServices()->hasDbAircraftData() && !sApp->getWebDataServices()->containsAircraftIcaoDesignator(v))
331  {
332  messages.push_back(CStatusMessage(this).validationWarning(u"Are you sure '%1' is a valid type?") << v);
333  }
334  flightPlan.setAircraftIcao(this->getAircraftIcaoCode());
335 
336  // prefix / equipment code
337  v = this->getPrefix();
338  if (!v.isEmpty() && !CFlightPlan::prefixCodes().contains(v))
339  {
340  messages.push_back(CStatusMessage(this).validation(severity, u"Invalid prefix"));
341  }
342  flightPlan.setPrefix(v);
343 
344  v = this->getEquipmentSuffix();
345  if (!v.isEmpty() && !CFlightPlan::equipmentCodes().contains(v))
346  {
347  messages.push_back(CStatusMessage(this).validation(severity, u"Invalid equipment code"));
348  }
349  flightPlan.setEquipmentSuffix(v);
350 
351  // route
352  v = ui->pte_Route->toPlainText().trimmed();
353  const int routeLength = v.length();
354  if (v.isEmpty())
355  {
356  messages.push_back(CStatusMessage(this).validation(
357  vfr ?
358  CStatusMessage::SeverityInfo :
359  CStatusMessage::SeverityWarning,
360  u"Missing '%1'")
361  << ui->lbl_Route->text());
362  }
363  else if (routeLength > CFlightPlan::MaxRouteLength)
364  {
365  messages.push_back(CStatusMessage(this).validationError(u"Flight plan route length exceeded (%1 chars max.)") << CFlightPlan::MaxRouteLength);
366  }
367  else
368  {
369  flightPlan.setRoute(v);
370  }
371 
372  // remarks
373  v = ui->pte_Remarks->toPlainText().trimmed();
374  const int remarksLength = v.length();
375  if (v.isEmpty())
376  {
377  messages.push_back(CStatusMessage(this).validationError(u"No '%1', voice capabilities are mandatory") << ui->pb_Remarks->text());
378  }
379  else if (remarksLength > CFlightPlan::MaxRemarksLength)
380  {
381  messages.push_back(CStatusMessage(this).validationError(u"Flight plan remarks length exceeded (%1 chars max.)") << CFlightPlan::MaxRemarksLength);
382  }
383  else
384  {
385  flightPlan.setRemarks(v);
386  }
387 
388  // Total length
389  if ((remarksLength + routeLength) > CFlightPlan::MaxRouteAndRemarksLength)
390  {
391  messages.push_back(CStatusMessage(this).validationError(u"Flight plan route (%1) and remarks (%2) length exceeded (%3 chars max.)") << routeLength << remarksLength << CFlightPlan::MaxRemarksLength);
392  }
393 
394  // time enroute
395  v = ui->le_EstimatedTimeEnroute->text();
396  if (v.isEmpty() || v == defaultTime())
397  {
398  messages.push_back(CStatusMessage(this).validation(severity, u"Missing '%1'") << ui->lbl_EstimatedTimeEnroute->text());
399  }
400  flightPlan.setEnrouteTime(v);
401 
402  // fuel
403  v = ui->le_FuelOnBoard->text();
404  if (v.isEmpty() || v == defaultTime())
405  {
406  messages.push_back(CStatusMessage(this).validation(severity, u"Missing '%1'") << ui->lbl_FuelOnBoard->text());
407  }
408  flightPlan.setFuelTime(v);
409 
410  // take off time
411  v = ui->le_TakeOffTimePlanned->text();
412  if (v.isEmpty() || v == defaultTime())
413  {
414  messages.push_back(CStatusMessage(this).validation(severity, u"Missing '%1'") << ui->lbl_TakeOffTimePlanned->text());
415  }
416  flightPlan.setTakeoffTimePlanned(v);
417 
418  // cruising alt
419  if (ui->lep_CrusingAltitude->isValid(&messages))
420  {
421  const CAltitude cruisingAltitude = ui->lep_CrusingAltitude->getAltitude();
422  flightPlan.setCruiseAltitude(cruisingAltitude);
423  }
424 
425  // destination airport
426  v = ui->le_DestinationAirport->text();
427  if (v.isEmpty() || v.endsWith(defaultIcao(), Qt::CaseInsensitive))
428  {
429  messages.push_back(CStatusMessage(this).validationError(u"Missing '%1'") << ui->lbl_DestinationAirport->text());
430  flightPlan.setDestinationAirportIcao(QString());
431  }
432  else
433  {
434  flightPlan.setDestinationAirportIcao(v);
435  if (strict && !flightPlan.getDestinationAirportIcao().hasValidIcaoCode(true))
436  {
437  messages.push_back(CStatusMessage(this).validationError(u"Invalid destination ICAO code '%1'") << v);
438  }
439  else if (!flightPlan.getDestinationAirportIcao().hasValidIcaoCode(false))
440  {
441  messages.push_back(CStatusMessage(this).validationWarning(u"Wrong or missing '%1'") << ui->lbl_DestinationAirport->text());
442  }
443  }
444 
445  // origin airport
446  v = ui->le_OriginAirport->text();
447  if (v.isEmpty() || v.endsWith(defaultIcao(), Qt::CaseInsensitive))
448  {
449  messages.push_back(CStatusMessage(this).validationError(u"Missing '%1'") << ui->lbl_OriginAirport->text());
450  flightPlan.setOriginAirportIcao(defaultIcao());
451  }
452  else
453  {
454  flightPlan.setOriginAirportIcao(v);
455  if (strict && !flightPlan.getOriginAirportIcao().hasValidIcaoCode(true))
456  {
457  messages.push_back(CStatusMessage(this).validationError(u"Invalid origin ICAO code '%1'") << v);
458  }
459  else if (!flightPlan.getOriginAirportIcao().hasValidIcaoCode(false))
460  {
461  messages.push_back(CStatusMessage(this).validationWarning(u"Wrong or missing '%1'") << ui->lbl_DestinationAirport->text());
462  }
463  }
464 
465  // TAS
466  v = ui->le_CruiseTrueAirspeed->text();
467  CSpeed cruiseTAS;
468  cruiseTAS.parseFromString(v, CPqString::SeparatorBestGuess);
469  if (cruiseTAS.isNull())
470  {
471  messages.push_back(CStatusMessage(this).validationError(u"Wrong TAS, %1. Try adding a unit like '100kts' or '150km/h'") << ui->lbl_CruiseTrueAirspeed->text());
472  flightPlan.setDestinationAirportIcao(defaultIcao());
473  }
474  else
475  {
476  flightPlan.setCruiseTrueAirspeed(cruiseTAS);
477  }
478 
479  // Optional fields
480  v = ui->le_AlternateAirport->text();
481  if (v.isEmpty() || v.endsWith(defaultIcao(), Qt::CaseInsensitive))
482  {
483  if (!messages.hasWarningOrErrorMessages())
484  {
485  messages.push_back(CStatusMessage(this).validationInfo(u"Missing %1") << ui->lbl_AlternateAirport->text());
486  }
487  flightPlan.setAlternateAirportIcao(QString());
488  }
489  else
490  {
491  flightPlan.setAlternateAirportIcao(v);
492  }
493 
494  // OK
495  if (!messages.isFailure())
496  {
497  messages.push_back(CStatusMessage(this).validationInfo(u"Flight plan validation passed"));
498  }
499  return messages;
500  }

502  void CFlightPlanComponent::sendFlightPlan()
503  {
504  CFlightPlan flightPlan;
505  const CStatusMessageList messages = this->validateAndInitializeFlightPlan(flightPlan);
506  if (messages.isSuccess())
507  {
508  // no error, send if possible
509  CStatusMessage m;
510  QString lastSent;
511  if (sGui && sGui->getIContextNetwork()->isConnected())
512  {
513  flightPlan.setWhenLastSentOrLoaded(QDateTime::currentDateTimeUtc());
514  sGui->getIContextNetwork()->sendFlightPlan(flightPlan);
515  lastSent = flightPlan.whenLastSentOrLoaded().toString();
516  m = CStatusMessage(this).validationInfo(u"Sent flight plan");
