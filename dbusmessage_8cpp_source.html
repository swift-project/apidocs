107  void CDBusMessage::appendArgument(const std::vector<bool> &array)
108  {
109  // array.data() not existing for bool
110  // changing dbus_message_iter_open_container here affects DBus signature
111  // using int16 creates signature "n", mismatch on swift side
112  // using int32 creates signature "i"
113  // there are also xml files for the signature: src/xswiftbus/org.swift_project.xswiftbus.traffic.xml
114  // discussion: https://discordapp.com/channels/539048679160676382/539925070550794240/698552502831939676
115 
116  DBusMessageIter arrayIterator;
117  dbus_message_iter_open_container(&m_messageIterator, DBUS_TYPE_ARRAY, DBUS_TYPE_BOOLEAN_AS_STRING, &arrayIterator);
118 
119  const std::vector<unsigned> ints(array.begin(), array.end());
120  const unsigned *ptr = ints.data();
121  dbus_message_iter_append_fixed_array(&arrayIterator, DBUS_TYPE_BOOLEAN, &ptr, static_cast<int>(array.size()));
122  dbus_message_iter_close_container(&m_messageIterator, &arrayIterator);
123  }

255  {
