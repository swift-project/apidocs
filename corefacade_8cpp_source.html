55  CStatusMessage CCoreFacade::tryToReconnectWithDBus()
56  {
57  if (m_shuttingDown) { return CStatusMessage(this, CStatusMessage::SeverityInfo, u"Shutdown"); }
58  if (!m_config.requiresDBusConnection()) { return CStatusMessage(this, CStatusMessage::SeverityInfo, u"Not DBus based"); }
59  const QString dBusAddress = this->getDBusAddress();
60  if (dBusAddress.isEmpty()) { return CStatusMessage(this, CStatusMessage::SeverityInfo, u"Not DBus based, no address"); }
61  QString connectMsg;
62  if (!CContextApplicationProxy::isContextResponsive(dBusAddress, connectMsg))
63  {
64  return CStatusMessage(this, CStatusMessage::SeverityError,
65  u"Cannot connect DBus at '" % dBusAddress % u"', reason: " % connectMsg);
66  }
67 
68  // re-init
69  m_initalized = false;
70  this->init();
71 
72  // success
73  return CStatusMessage(this, CStatusMessage::SeverityInfo, u"Re-initialized via '%1'") << dBusAddress;
74  }

76  void CCoreFacade::init()
77  {
78  if (m_initalized || m_shuttingDown) { return; }
79 
80  QMap<QString, qint64> times;
81  QElapsedTimer time;
82  CCoreFacade::registerMetadata();
83 
84  // either use explicit setting or last value
85  const QString dbusAddress = this->getDBusAddress();
86  m_launcherSetup.setProperty(CLauncherSetup::IndexDBusAddress, dbusAddress);
87 
88  // DBus
89  time.start();
90  if (m_config.requiresDBusSever()) { this->initDBusServer(dbusAddress); }
91  if (m_config.requiresDBusConnection())
92  {
93  this->initDBusConnection(dbusAddress);
94  if (!m_dbusConnection.isConnected())
95  {
96  const QString e = m_dbusConnection.lastError().message();
97  BLACK_VERIFY_X(false, "CRuntime::init DBus problem", e.toUtf8().constData());
98  CLogMessage(this).error(u"DBus connection failed: '%1'") << e;
99  return;
100  }
101  }
102  times.insert("DBus", time.restart());
103 
104  // shared state infrastructure
105  m_dataLinkDBus = new SharedState::CDataLinkDBus(this);
106  switch (m_config.getModeApplication())
107  {
108  case CCoreFacadeConfig::NotUsed:
109  case CCoreFacadeConfig::Local:
110  m_dataLinkDBus->initializeLocal(nullptr);
111  break;
112  case CCoreFacadeConfig::LocalInDBusServer:
113  m_dataLinkDBus->initializeLocal(m_dbusServer);
114  break;
115  case CCoreFacadeConfig::Remote:
116  m_dataLinkDBus->initializeRemote(m_dbusConnection, CDBusServer::coreServiceName(m_dbusConnection));
117  break;
118  default:
119  qFatal("Invalid application context mode");
120  }
121 
122  // shared log history
123  m_logHistorySource = new CLogHistorySource(this);
124  m_logHistorySource->initialize(m_dataLinkDBus);
125  if (m_config.hasLocalCore())
126  {
127  m_logHistory = new CLogHistory(this);
128  m_logHistory->initialize(m_dataLinkDBus);
129  }
130 
131  if (m_config.all(CCoreFacadeConfig::NotUsed))
132  {
133  m_initalized = true;
134  return;
135  }
136 
137  // contexts
138  if (m_contextApplication) { m_contextApplication->deleteLater(); }
139  m_contextApplication = IContextApplication::create(this, m_config.getModeApplication(), m_dbusServer, m_dbusConnection);
140  times.insert("Application", time.restart());
141 
142  if (m_contextAudio) { m_contextAudio->deleteLater(); }
143  m_contextAudio = qobject_cast<CContextAudioBase *>(IContextAudio::create(this, m_config.getModeAudio(), m_dbusServer, m_dbusConnection));
144  times.insert("Audio", time.restart());
145 
146  if (m_contextOwnAircraft) { m_contextOwnAircraft->deleteLater(); }
147  m_contextOwnAircraft = IContextOwnAircraft::create(this, m_config.getModeOwnAircraft(), m_dbusServer, m_dbusConnection);
148  times.insert("Own aircraft", time.restart());
149 
150  if (m_contextSimulator) { m_contextSimulator->deleteLater(); }
151  m_contextSimulator = IContextSimulator::create(this, m_config.getModeSimulator(), m_dbusServer, m_dbusConnection);
152  times.insert("Simulator", time.restart());
153 
154  // depends on own aircraft and simulator context, which is bad style
155  if (m_contextNetwork) { m_contextNetwork->deleteLater(); }
156  m_contextNetwork = IContextNetwork::create(this, m_config.getModeNetwork(), m_dbusServer, m_dbusConnection);
157  times.insert("Network", time.restart());
158 
159  // checks --------------
160  // 1. own aircraft and simulator should reside in same location
161  Q_ASSERT(!m_contextSimulator || (m_contextOwnAircraft->isUsingImplementingObject() == m_contextSimulator->isUsingImplementingObject()));
162 
163  // 2. own aircraft and network should reside in same location
164  Q_ASSERT(!m_contextNetwork || (m_contextOwnAircraft->isUsingImplementingObject() == m_contextNetwork->isUsingImplementingObject()));
165 
166  // post inits, wiring things among context (e.g. signal slots)
167  time.restart();
168  this->initPostSetup(times);
169  times.insert("Post setup", time.restart());
170  CLogMessage(this).info(u"Init times: %1") << qmapToString(times);
171 
172  // flag
173  m_initalized = true;
174  }

188  bool CCoreFacade::parseCommandLine(const QString &commandLine, const CIdentifier &originator)
189  {
190  bool handled = false;
191  // audio can be empty depending on whre it runs
192  if (this->getIContextAudio() && !this->getIContextAudio()->isEmptyObject()) { handled = handled || this->getIContextAudio()->parseCommandLine(commandLine, originator); }
193  if (this->getIContextNetwork()) { handled = handled || this->getIContextNetwork()->parseCommandLine(commandLine, originator); }
194  if (this->getIContextOwnAircraft()) { handled = handled || this->getIContextOwnAircraft()->parseCommandLine(commandLine, originator); }
195  if (this->getIContextSimulator()) { handled = handled || this->getIContextSimulator()->parseCommandLine(commandLine, originator); }
196  return handled;
197  }

207  void CCoreFacade::initPostSetup(QMap<QString, qint64> &times)
208  {
209  bool c = false;
210  Q_UNUSED(c) // for release version
211  QElapsedTimer time;
212  time.start();
213 
214  times.insert("Post setup, connects first", time.restart());
215 
216  // local simulator?
217  if (m_contextSimulator && m_contextSimulator->isUsingImplementingObject())
218  {
219  // only connect if network runs locally, no round trips
220  // remark: from a design perspective it would be nice if those could be avoided (seperation of concerns)
221  // those connects here reprent cross context dependencies
222  if (m_contextNetwork && m_contextNetwork->isUsingImplementingObject())
223  {
224  Q_ASSERT_X(this->getCContextNetwork(), Q_FUNC_INFO, "No local network object");
225  Q_ASSERT_X(this->getCContextNetwork()->airspace(), Q_FUNC_INFO, "No airspace object");
226 
227  c = connect(m_contextNetwork, &IContextNetwork::textMessagesReceived,
228  this->getCContextSimulator(), &CContextSimulator::xCtxTextMessagesReceived, Qt::QueuedConnection);
229  Q_ASSERT(c);
230 
231  // use readyForModelMatching instead of CAirspaceMonitor::addedAircraft, as it contains client information
232  // ready for model matching is sent delayed when all information are available
233  c = connect(m_contextNetwork, &IContextNetwork::readyForModelMatching,
234  this->getCContextSimulator(), &CContextSimulator::xCtxAddedRemoteAircraftReadyForModelMatching, Qt::QueuedConnection);
235  Q_ASSERT(c);
236  c = connect(m_contextNetwork, &IContextNetwork::removedAircraft,
237  this->getCContextSimulator(), &CContextSimulator::xCtxRemovedRemoteAircraft, Qt::QueuedConnection);
238  Q_ASSERT(c);
239  c = connect(m_contextNetwork, &IContextNetwork::changedRemoteAircraftModel,
240  this->getCContextSimulator(), &CContextSimulator::xCtxChangedRemoteAircraftModel, Qt::QueuedConnection);
241  Q_ASSERT(c);
242  c = connect(m_contextNetwork, &IContextNetwork::changedRemoteAircraftEnabled,
243  this->getCContextSimulator(), &CContextSimulator::xCtxChangedRemoteAircraftEnabled, Qt::QueuedConnection);
244  Q_ASSERT(c);
245  c = connect(m_contextNetwork, &IContextNetwork::connectionStatusChanged,
246  this->getCContextSimulator(), &CContextSimulator::xCtxNetworkConnectionStatusChanged, Qt::QueuedConnection);
247  Q_ASSERT(c);
248  c = connect(this->getCContextNetwork()->airspace(), &CAirspaceMonitor::requestedNewAircraft,
249  this->getCContextSimulator(), &CContextSimulator::xCtxNetworkRequestedNewAircraft, Qt::QueuedConnection);
250  Q_ASSERT(c);
251  c = connect(this->getCContextSimulator(), &CContextSimulator::renderRestrictionsChanged,
252  this->getCContextNetwork(), &CContextNetwork::xCtxSimulatorRenderRestrictionsChanged, Qt::QueuedConnection);
253  Q_ASSERT(c);
254  c = connect(this->getCContextSimulator(), &CContextSimulator::simulatorStatusChanged,
255  this->getCContextNetwork(), &CContextNetwork::xCtxSimulatorStatusChanged, Qt::QueuedConnection);
256  Q_ASSERT(c);
257 
258  // set provider
259  this->getCContextNetwork()->setSimulationEnvironmentProvider(this->getCContextSimulator()->simulator());
260  }
261 
262  // only if own aircraft runs locally
263  if (m_contextOwnAircraft && m_contextOwnAircraft->isUsingImplementingObject())
264  {
265  c = connect(m_contextOwnAircraft, &IContextOwnAircraft::changedAircraftCockpit,
266  this->getCContextSimulator(), &CContextSimulator::xCtxUpdateSimulatorCockpitFromContext);
267  Q_ASSERT(c);
268  c = connect(m_contextOwnAircraft, &IContextOwnAircraft::changedSelcal,
269  this->getCContextSimulator(), &CContextSimulator::xCtxUpdateSimulatorSelcalFromContext);
270  Q_ASSERT(c);
271 
272  // relay changed aircraft to own aircraft provider but with identifier
273  // identifier is needed because own aircraft context also reports changed aircraft to xCtxChangedOwnAircraftModel
274  // and we avoid roundtrips
275  c = connect(this->getCContextSimulator(), &CContextSimulator::ownAircraftModelChanged,
276  this->getCContextOwnAircraft(), [=](const CAircraftModel &changedModel) {
277  if (!this->getIContextOwnAircraft()) { return; }
278  if (!this->getCContextSimulator()) { return; }
279  this->getCContextOwnAircraft()->xCtxChangedSimulatorModel(changedModel, this->getCContextSimulator()->identifier());
280  });
281 
282  Q_ASSERT(c);
283  c = connect(this->getCContextSimulator(), &CContextSimulator::simulatorStatusChanged,
284  this->getCContextOwnAircraft(), &CContextOwnAircraft::xCtxChangedSimulatorStatus);
285  Q_ASSERT(c);
286 
287  // this is used if the value in own aircraft is changed, to callback simulator
288  c = connect(this->getCContextOwnAircraft(), &CContextOwnAircraft::ps_changedModel,
289  this->getCContextSimulator(), &CContextSimulator::xCtxChangedOwnAircraftModel);
290  Q_ASSERT(c);
291  }
292 
293  // times
294  times.insert("Post setup, sim.connects", time.restart());
295  }
296 
297  // only where network and(!) own aircraft run locally
298  // -> in the core or an all local implementation
299  if (m_contextNetwork && m_contextOwnAircraft && m_contextNetwork->isUsingImplementingObject() && m_contextOwnAircraft->isUsingImplementingObject())
300  {
301  c = connect(m_contextNetwork, &IContextNetwork::changedAtcStationOnlineConnectionStatus,
302  this->getCContextOwnAircraft(), &CContextOwnAircraft::xCtxChangedAtcStationOnlineConnectionStatus);
303  Q_ASSERT(c);
304  times.insert("Post setup, connects network", time.restart());
305  }
306 
307  // connection status of network changed
308  // with AFV no longer use m_contextAudio->isUsingImplementingObject() as audio can run on both sides
309  if (this->getCContextAudioBase() && m_contextNetwork)
310  {
311  Q_ASSERT(m_contextApplication);
312  c = connect(m_contextNetwork, &IContextNetwork::connectionStatusChanged,
313  this->getCContextAudioBase(), &CContextAudio::xCtxNetworkConnectionStatusChanged, Qt::QueuedConnection);
314  Q_ASSERT(c);
315  times.insert("Post setup, connects audio", time.restart());
316  }
317  }

334  void CCoreFacade::gracefulShutdown()
335  {
336  if (!m_initalized) { return; }
337  if (m_shuttingDown) { return; }
338  m_shuttingDown = true;
339 
340  // disable all signals towards runtime
341  disconnect(this);
342 
343  // tear down shared state infrastructure
344  delete m_logHistory;
345  m_logHistory = nullptr;
346  delete m_logHistorySource;
347  m_logHistorySource = nullptr;
348  delete m_dataLinkDBus;
349  m_dataLinkDBus = nullptr;
350 
351  // unregister all from DBus
352  if (m_dbusServer) { m_dbusServer->removeAllObjects(); }
353 
354  // handle contexts
355 
356  if (this->getCContextAudioBase())
357  {
358  // there is no empty audio context since AFV
359  disconnect(this->getCContextAudioBase());
360  this->getCContextAudioBase()->gracefulShutdown();
361  this->getIContextAudio()->deleteLater();
362  m_contextAudio = nullptr;
363  }
364 
365  // log off from network, if connected
366  if (this->getIContextNetwork())
367  {
368  disconnect(this->getIContextNetwork());
369  this->getIContextNetwork()->disconnectFromNetwork();
370  if (m_contextNetwork->isUsingImplementingObject())
371  {
372  this->getCContextNetwork()->gracefulShutdown(); // for threads
373  }
374  this->getIContextNetwork()->deleteLater();
375  // replace by dummy object avoiding nullptr issues during shutdown phase
376  QDBusConnection defaultConnection("default");
377  m_contextNetwork = IContextNetwork::create(this, CCoreFacadeConfig::NotUsed, nullptr, defaultConnection);
378  }
379 
380  if (this->getIContextSimulator())
381  {
382  disconnect(this->getIContextSimulator());
383  if (this->getIContextSimulator()->isUsingImplementingObject())
384  {
385  // shutdown the plugins
386  this->getCContextSimulator()->gracefulShutdown();
387  }
388  this->getIContextSimulator()->deleteLater();
389  QDBusConnection defaultConnection("default");
390  m_contextSimulator = IContextSimulator::create(this, CCoreFacadeConfig::NotUsed, nullptr, defaultConnection);
391  }
392 
393  if (this->getIContextOwnAircraft())
394  {
395  disconnect(this->getIContextOwnAircraft());
396  this->getIContextOwnAircraft()->deleteLater();
397  QDBusConnection defaultConnection("default");
398  m_contextOwnAircraft = IContextOwnAircraft::create(this, CCoreFacadeConfig::NotUsed, nullptr, defaultConnection);
399  }
400 
401  if (this->getIContextApplication())
402  {
403  disconnect(this->getIContextApplication());
404  this->getIContextApplication()->deleteLater();
405  QDBusConnection defaultConnection("default");
406  m_contextApplication = IContextApplication::create(this, CCoreFacadeConfig::NotUsed, nullptr, defaultConnection);
407  }
408  }

410  void CCoreFacade::initDBusConnection(const QString &address)
411  {
412  if (m_initDBusConnection) { return; }
413  if (address.isEmpty() || address == CDBusServer::sessionBusAddress())
414  {
415  QDBusConnection::disconnectFromBus(m_dbusConnection.name());
416  m_dbusConnection = QDBusConnection::sessionBus();
417  }
418  else if (address == CDBusServer::systemBusAddress())
419  {
420  QDBusConnection::disconnectFromBus(m_dbusConnection.name());
421  m_dbusConnection = QDBusConnection::systemBus();
422  }
423  else
424  {
425  const QString name(CDBusServer::p2pConnectionName() + " " + address);
426  QDBusConnection::disconnectFromPeer(name);
427  m_dbusConnection = QDBusConnection::connectToPeer(address, name);
428  }
429  }
