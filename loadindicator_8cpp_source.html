48  int CLoadIndicator::startAnimation(int timeoutMs, bool processEvents)
49  {
50  m_angle = 0;
51  this->show();
52  this->setEnabled(true);
53 
54  QPointer<CLoadIndicator> myself(this);
55  if (m_timerId == -1) { m_timerId = startTimer(m_delayMs); }
56  if (processEvents && sGui)
57  {
58  sGui->processEventsToRefreshGui();
59  if (!myself) { return -1; } // cppcheck-suppress knownConditionTrueFalse // deleted in meantime (process events)
60  }
61 
62  const int stopId = m_currentId++; // copy
63  if (timeoutMs > 0)
64  {
65  QTimer::singleShot(timeoutMs, this, [=] {
66  if (!myself) { return; }
67 
68  // only timeout myself id
69  this->stopAnimation(stopId);
70  emit this->timedOut();
71  });
72  }
73  m_pendingIds.push_back(stopId);
74  return stopId;
75  }

77  void CLoadIndicator::stopAnimation(int indicatorId)
78  {
79  if (indicatorId > 0)
80  {
81  m_pendingIds.removeOne(indicatorId);
82  // if others pending do not stop
83  if (!m_pendingIds.isEmpty()) { return; }
84  }
85  m_pendingIds.clear();
86  if (m_timerId != -1)
87  {
88  BLACK_AUDIT_X(CThreadUtils::isInThisThread(this), Q_FUNC_INFO, "Try to kill timer from another thread");
89  this->killTimer(m_timerId);
90  }
91  m_timerId = -1;
92  this->hide();
93  this->setEnabled(false);
94  this->update();
95  }

144  void CLoadIndicator::paint(QPainter &painter) const
145  {
146  if (!m_displayedWhenStopped && !isAnimated()) { return; }
147  if (!this->isVisible() || !this->isEnabled()) { return; }
148  if (!isParentVisible()) { return; }
149 
150  int width = qMin(this->width(), this->height());
151  painter.setRenderHint(QPainter::Antialiasing);
152 
153  // painter.setBrush(QBrush(QColor(0, 0, 255)));
154  // painter.drawEllipse(0, 0, width, width);
155 
156  int outerRadius = qRound((width - 1) * 0.5);
157  int innerRadius = qRound((width - 1) * 0.5 * 0.38);
158 
159  int capsuleHeight = outerRadius - innerRadius;
160  int capsuleWidth = (width > 32) ? qRound(capsuleHeight * .23) : qRound(capsuleHeight * .35);
161  int capsuleRadius = capsuleWidth / 2;
162 
163  for (int i = 0; i < 12; i++)
164  {
165  QColor color = m_color;
166  color.setAlphaF(qRound(1.0f - (i / 12.0f)));
167  painter.setPen(Qt::NoPen);
168  painter.setBrush(color);
169  painter.save();
170  painter.translate(rect().center());
171  painter.rotate(m_angle - qRound(i * 30.0f));
172  painter.drawRoundedRect(-qRound(capsuleWidth * 0.5), -(innerRadius + capsuleHeight), capsuleWidth, capsuleHeight, capsuleRadius, capsuleRadius);
173  painter.restore();
174  }
175  }
