209  void CInterpolationLogDisplay::onCallsignEntered()
210  {
211  if (!this->canLog())
212  {
213  this->stop();
214  CLogMessage(this).warning(u"Stopping logging (log.display), no simulator");
215  return;
216  }
217 
218  const CCallsign cs = ui->comp_CallsignCompleter->getCallsign();
219  if (m_callsign == cs) { return; }
220 
221  // empty callsign, just stop
222  if (cs.isEmpty())
223  {
224  this->stop();
225  m_callsign.clear();
226  return;
227  }
228 
229  // clear last callsign
230  if (!m_callsign.isEmpty())
231  {
232  m_simulator->setLogInterpolation(false, m_callsign); // stop logging "old" callsign
233  m_callsign.clear(); // clear callsign
234  this->clear();
235  }
236 
237  // set new callsign or stop
238  m_callsign = cs;
239  m_simulator->setLogInterpolation(true, cs);
240  ui->comp_Parts->setCallsign(cs);
241  if (!this->start())
242  {
243  this->initPartsView();
244  }
245  CLogMessage(this).info(u"Starting logging (log.display) of '%1'") << m_callsign;
246  }

362  void CInterpolationLogDisplay::onElevationReceived(const CElevationPlane &elevationPlane, const CCallsign &callsign)
363  {
364  m_elvReceived++;
365  if (m_elvHistoryCount > 0)
366  {
367  const QString history = callsign.asString() %
368  QStringLiteral(": ") % QTime::currentTime().toString("hh:mm:ss.zzz") %
369  QStringLiteral(" ") % elevationPlane.toQString(true);
370  m_elvHistoryModel->insertRow(0);
371  const QModelIndex index = m_elvHistoryModel->index(0, 0);
372  m_elvHistoryModel->setData(index, history);
373 
374  const int c = m_elvHistoryModel->rowCount();
375  if (m_elvHistoryCount < c)
376  {
377  m_elvHistoryModel->removeRows(m_elvHistoryCount, c - m_elvHistoryCount);
378  }
379  }
380 
381  // not for a real plane, but to get elevation at any position for testing
382  if (callsign == CInterpolationLogDisplay::pseudoCallsignElevation())
383  {
384  this->displayArbitraryElevation(elevationPlane);
385  return;
386  }
387 
388  if (!this->logCallsign(callsign)) { return; }
389 
390  // for logged callsign
391  m_elvReceivedLoggedCs++;
392  ui->le_Elevation->setText(elevationPlane.toQString());
393  this->displayElevationRequestReceive();
394  ui->led_Elevation->blink();
395  }

502  bool CInterpolationLogDisplay::checkLogPrerequisites()
503  {
504  CStatusMessage m;
505  do
506  {
507  if (!this->isVisible()) { return false; } // silently return
508  if (!sGui || sGui->isShuttingDown()) { break; } // stop and return
509  if (m_callsign.isEmpty())
510  {
511  // static const CStatusMessage ms = CStatusMessage(this).validationError(u"No callsign for logging");
512  // m = ms;
513  break;
514  }
515 
516  const bool canUpdateLog = m_airspaceMonitor && m_simulator && m_simulator->isConnected() && !m_simulator->isShuttingDown();
517  if (!canUpdateLog)
518  {
519  static const CStatusMessage ms = CStatusMessage(this).validationError(u"No airspace monitor or simulator or shutting down");
520  m = ms;
521  break;
522  }
523 
524  if (!m_simulator->isLogCallsign(m_callsign))
525  {
526  m = CStatusMessage(this).validationError(u"No longer logging callsign '%1'") << m_callsign;
527  break;
528  }
529  return true;
530  }
531  while (false);
532 
533  this->stop();
534  if (!m.isEmpty()) { this->showOverlayMessage(m, 5000); }
535  return false;
536  }
