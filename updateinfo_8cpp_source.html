61  CDistribution CUpdateInfo::anticipateOwnDistribution() const
62  {
63  if (this->isEmpty()) { return CDistribution(); }
64  const CArtifactList ownArtifacts = this->getArtifactsPilotClientForCurrentPlatform();
65  if (ownArtifacts.isEmpty()) { return CDistribution(); }
66 
67  const QVersionNumber myVersion = CBuildConfig::getVersion();
68  const QVersionNumber latestVersion = ownArtifacts.getLatestQVersion();
69  if (myVersion > latestVersion)
70  {
71  // dev. version
72  return ownArtifacts.getDistributions().getLeastStableOrDefault();
73  }
74 
75  const CArtifact exactVersion = ownArtifacts.findFirstByVersionOrDefault(myVersion);
76  if (!exactVersion.isUnknown()) { return exactVersion.getDistributions().getMostStableOrDefault(); }
77 
78  return CDistribution::localDeveloperBuild();
79  }

154  CUpdateInfo CUpdateInfo::fromGitHubReleasesJson(const QByteArray &jsonData)
155  {
156  // https://docs.github.com/en/rest/reference/repos#releases
157 
158  const QString url = CBuildConfig::gitHubRepoUrl() + QStringLiteral("releases/download");
159  CDistribution alphaDistribution("ALPHA", 5, false);
160  CDistribution betaDistribution("BETA", 10, false);
161  alphaDistribution.addDownloadUrl(url);
162  betaDistribution.addDownloadUrl(url);
163 
164  CUpdateInfo result;
165  result.m_distributions = { alphaDistribution, betaDistribution };
166 
167  for (const QJsonValue release : QJsonDocument::fromJson(jsonData).array())
168  {
169  QString version = release[QLatin1String("tag_name")].toString();
170  if (version.isEmpty() || version[0] != 'v') { continue; }
171  version.remove(0, 1);
172  if (containsChar(version, [](QChar c) { return c != '.' && !is09(c); })) { continue; }
173 
174  bool existing = !release[QLatin1String("draft")].toBool();
175  bool alpha = release[QLatin1String("prerelease")].toBool();
176 
177  for (const QJsonValue asset : release[QLatin1String("assets")].toArray())
178  {
179  QString name = asset[QLatin1String("name")].toString();
180  QString filename = QStringLiteral("v%1/%2").arg(version, name);
181  int size = asset[QLatin1String("size")].toInt();
182 
183  CArtifact::ArtifactType type = CArtifact::UnknownArtifact;
184  if (name.startsWith(QStringLiteral("swiftinstaller"))) { type = CArtifact::PilotClientInstaller; }
185  else if (name.startsWith(QStringLiteral("swiftsymbols"))) { type = CArtifact::Symbols; }
186  else if (name.startsWith(QStringLiteral("xswiftbus"))) { type = CArtifact::XSwiftBus; }
187 
188  CPlatform platform;
189  if (name.contains(QStringLiteral("windows-32"))) { platform = CPlatform::win32Platform(); }
190  else if (name.contains(QStringLiteral("windows-64"))) { platform = CPlatform::win64Platform(); }
191  else if (name.contains(QStringLiteral("linux-64"))) { platform = CPlatform::linuxPlatform(); }
192  else if (name.contains(QStringLiteral("macos-64"))) { platform = CPlatform::macOSPlatform(); }
193  else if (name.contains(QStringLiteral("allos"))) { platform = CPlatform::allOs(); }
194 
195  CArtifact artifact(filename, version, {}, type, size, existing, platform);
196  artifact.setDistributions({ alpha ? alphaDistribution : betaDistribution });
197 
198  if (type == CArtifact::PilotClientInstaller) { result.m_artifactsPilotClient.push_back(artifact); }
199  else if (type == CArtifact::XSwiftBus) { result.m_artifactsXSwiftBus.push_back(artifact); }
200  }
201  }
202  return result;
203  }
