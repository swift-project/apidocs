479  bool CSimulatorMessagesSettings::relayThisTextMessage(const Network::CTextMessage &msg, const BlackMisc::Simulation::CSimulatedAircraft &aircraft) const
480  {
481  if (msg.isEmpty()) { return false; }
482  if (!this->isRelayGloballyEnabled()) { return false; }
483  if (m_messageType == NoTextMessages) { return false; }
484 
485  const TextMessageType mt = static_cast<TextMessageType>(m_messageType);
486  if (msg.isPrivateMessage() && mt.testFlag(TextMessagePrivate)) { return true; }
487  if (msg.isSupervisorMessage() && (mt.testFlag(TextMessagePrivate) || mt.testFlag(TextMessageSupervisor))) { return true; }
488  if (msg.isSendToUnicom() && mt.testFlag(TextMessagesUnicom)) { return true; }
489 
490  if (msg.isRadioMessage())
491  {
492  const CFrequency f(msg.getFrequency());
493  if (mt.testFlag(TextMessagesCom1))
494  {
495  if (aircraft.getCom1System().isActiveFrequencySameFrequency(f)) { return true; }
496  }
497  if (mt.testFlag(TextMessagesCom2))
498  {
499  if (aircraft.getCom2System().isActiveFrequencySameFrequency(f)) { return true; }
500  }
501  }
502  return false;
503  }

586  QStringList CSpecializedSimulatorSettings::getModelDirectoriesFromSimulatorDirectoy() const
587  {
588  if (!m_genericSettings.hasSimulatorDirectory()) { return QStringList(); }
589  const QString s(m_genericSettings.getSimulatorDirectory());
590  QStringList dirs;
591  switch (m_simulator.getSimulator())
592  {
593  case CSimulatorInfo::FG: dirs = QStringList(CFlightgearUtil::modelDirectoriesFromSimDir(s)); break;
594  case CSimulatorInfo::FS9: dirs = QStringList({ CFsDirectories::fs9AircraftDirFromSimDir(s) }); break;
595  case CSimulatorInfo::FSX:
596  dirs = CFsDirectories::fsxSimObjectsDirPlusAddOnXmlSimObjectsPaths(CFsDirectories::fsxSimObjectsDirFromSimDir(s));
597  break;
598  case CSimulatorInfo::P3D:
599  {
600  const QString versionHint = CFsDirectories::guessP3DVersion(s);
601  dirs = CFsDirectories::p3dSimObjectsDirPlusAddOnXmlSimObjectsPaths(CFsDirectories::p3dSimObjectsDirFromSimDir(s), versionHint);
602  }
603  break;
604  case CSimulatorInfo::XPLANE: dirs = QStringList({ CXPlaneUtil::modelDirectoriesFromSimDir(s) }); break;
605  default: break;
606  }
607 
608  dirs.removeAll({}); // remove empty
609  dirs.removeDuplicates();
610  return dirs;
611  }
