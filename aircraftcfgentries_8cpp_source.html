114  CAircraftModel CAircraftCfgEntries::toAircraftModel() const
115  {
116  // creates raw, unconsolidated data
117  Q_ASSERT_X(m_timestampMSecsSinceEpoch >= 0, Q_FUNC_INFO, "Missing file timestamp");
118  CAircraftModel model(this->getTitle(), CAircraftModel::TypeOwnSimulatorModel);
119  model.setDescription(this->getUiCombinedDescription()); // Manufacturer, variation, type
120  model.setFileName(this->getFileName());
121  model.setName(this->getSimName());
122  model.setMSecsSinceEpoch(m_timestampMSecsSinceEpoch); // aircraft.cfg file timestamp
123  model.setFileTimestamp(m_timestampMSecsSinceEpoch);
124  model.setIconFile(this->getThumbnailFileNameChecked());
125 
126  const QString designator(CAircraftIcaoCode::normalizeDesignator(this->getAtcModel()));
127  CAircraftIcaoCode aircraft(
128  CAircraftIcaoCode::isValidDesignator(designator) ?
129  designator :
130  CAircraftIcaoCode::getUnassignedDesignator());
131  aircraft.setManufacturer(this->getUiManufacturer());
132  model.setAircraftIcaoCode(aircraft);
133 
134  // livery
135  CLivery livery;
136  livery.setCombinedCode(this->getTexture());
137 
138  CAirlineIcaoCode airline;
139  airline.setName(this->getAtcAirline()); // descriptive name like "BATAVIA"
140  const QString airlineDesignator(this->getAtcParkingCode());
141  if (CAirlineIcaoCode::isValidAirlineDesignator(airlineDesignator))
142  {
143  airline.setDesignator(airlineDesignator);
144  }
145  livery.setAirlineIcaoCode(airline);
146  model.setLivery(livery);
147 
148  // distributor
149  const CDistributor distributor(this->getCreatedBy());
150  model.setDistributor(distributor);
151  return model;
152  }
