48  bool CAtcStationList::updateIfMessageChanged(const CInformationMessage &im, const CCallsign &callsign, bool overrideWithNewer)
49  {
50  const CInformationMessage::InformationType type = im.getType();
51 
52  // for loop just to get reference
53  bool unequal = false;
54  for (CAtcStation &station : *this)
55  {
56  if (station.getCallsign() != callsign) { continue; }
57 
58  const CInformationMessage m = station.getInformationMessage(type);
59  if (m.getType() == CInformationMessage::Unspecified) { break; }
60 
61  if (m.getMessage() == im.getMessage())
62  {
63  if (!overrideWithNewer) { break; }
64  if (!im.isNewerThan(m)) { break; }
65  }
66  else
67  {
68  unequal = true;
69  }
70  station.setMessage(im);
71  break; // only count unequals
72  }
73  return unequal;
74  }

110  int CAtcStationList::synchronizeWithBookedStation(CAtcStation &bookedAtcStation)
111  {
112  int c = 0;

114  if (this->isEmpty()) return 0;

141  QHash<QString, CAtcStationList> CAtcStationList::splitPerSuffix(bool sort) const
142  {
143  if (this->isEmpty()) { return QHash<QString, CAtcStationList>(); }
144  const CAtcStationList stations = sort ? this->sortedByAtcSuffixSortOrderAndDistance() : *this;
145 
146  QString suffix;
147  QHash<QString, CAtcStationList> split;
148  for (const CAtcStation &s : stations)
149  {
150  const QString currentSuffix = s.getCallsignSuffix();
151  if (suffix != currentSuffix)
152  {
153  suffix = currentSuffix;
154  split[currentSuffix] = CAtcStationList();
155  }
156  split[currentSuffix].push_back(s);
157  }
158  return split;
159  }
