95  void CInstallXSwiftBusComponent::installXSwiftBus()
96  {
97  const CRemoteFile rf = this->getRemoteFileSelected();
98  const QString downloadFileName = CFileUtils::appendFilePathsAndFixUnc(this->downloadDir(), rf.getBaseName());
99  QPointer<CInstallXSwiftBusComponent> myself(this);
100  QFile downloadFile(downloadFileName);
101 
102  if (!downloadFile.exists())
103  {
104  const CStatusMessage msg = CStatusMessage(this, CLogCategories::validation()).error(u"Cannot read downloaded file '%1'") << downloadFileName;
105  this->showOverlayMessage(msg, CInstallXSwiftBusComponent::OverlayMsgTimeoutMs);
106  return;
107  }
108 
109  const QString xSwiftBusDirectory = this->xSwiftBusDir();
110  if (xSwiftBusDirectory.isEmpty())
111  {
112  const CStatusMessage msg = CStatusMessage(this, CLogCategories::validation()).error(u"No directory to install to'");
113  this->showOverlayMessage(msg, CInstallXSwiftBusComponent::OverlayMsgTimeoutMs);
114  return;
115  }
116 
117  const QDir installDir(xSwiftBusDirectory);
118  if (!installDir.exists())
119  {
120  const CStatusMessage msg = CStatusMessage(this, CLogCategories::validation()).error(u"Directory '%1' does not exist") << xSwiftBusDirectory;
121  this->showOverlayMessage(msg, CInstallXSwiftBusComponent::OverlayMsgTimeoutMs);
122  return;
123  }
124 
125  const QString destFileName = CFileUtils::appendFilePathsAndFixUnc(xSwiftBusDirectory, rf.getBaseName());
126  {
127  QFile destFile(destFileName);
128  if (destFile.exists())
129  {
130  const bool removed = destFile.remove();
131  if (!removed)
132  {
133  const CStatusMessage msg = CStatusMessage(this, CLogCategories::validation()).error(u"Cannot remove '%1'") << destFileName;
134  this->showOverlayMessage(msg, CInstallXSwiftBusComponent::OverlayMsgTimeoutMs);
135  return;
136  }
137  }
138  }
139 
140  const bool copied = QFile::copy(downloadFileName, destFileName);
141  if (!copied)
142  {
143  const CStatusMessage msg = CStatusMessage(this, CLogCategories::validation()).error(u"Cannot copy '%1' to '%2'") << downloadFileName << destFileName;
144  this->showOverlayMessage(msg, CInstallXSwiftBusComponent::OverlayMsgTimeoutMs);
145  return;
146  }
147 
148  // we need to unzip the destination file
149  const QFileInfo destFile(destFileName);
150  if (!destFile.exists())
151  {
152  const CStatusMessage msg = CStatusMessage(this, CLogCategories::validation()).error(u"XSwiftBus file '%1' does not exist") << destFileName;
153  this->showOverlayMessage(msg, CInstallXSwiftBusComponent::OverlayMsgTimeoutMs);
154  return;
155  }
156 
157  // if possible we will unzip
158  QStringList stdOutAndError;
159  if (CCompressUtils::zip7Uncompress(destFile.absoluteFilePath(), xSwiftBusDirectory, &stdOutAndError))
160  {
161  // capture values by copy!
162  const CStatusMessage msg = CStatusMessage(this, CLogCategories::validation()).info(u"Uncompressed XSwiftBus in '%1'") << xSwiftBusDirectory;
163  this->showOverlayMessagesWithConfirmation(msg, false, "Delete downloaded file?", [=] {
164  if (!myself) { return; }
165  QFile downloadFile(downloadFileName);
166  if (!downloadFile.exists()) { return; } // removed in meantime
167  const bool removed = downloadFile.remove();
168  Q_UNUSED(removed);
169  });
170  return;
171  }
172  else
173  {
174  const CStatusMessage msg = CStatusMessage(this, CLogCategories::validation()).warning(u"Unzip failed: stdout '%1' stderr '%2'") << safeAt(stdOutAndError, 0) << safeAt(stdOutAndError, 1);
175  this->showOverlayMessage(msg);
176  }
177 
179  const QMessageBox::StandardButton reply = QMessageBox::question(this,
180  "Install XSwiftXBus",
181  "You need to manually unzip XSwiftBus into the plugins directory.\nIt needs to look like 'plugin/xswiftbus'.\n\nOpen the archive?",
182  QMessageBox::Yes | QMessageBox::No);
183 
184  if (reply == QMessageBox::Yes)
185  {
186  QDesktopServices::openUrl(QUrl::fromLocalFile(destFile.absoluteFilePath()));
187  }
188  }

190  void CInstallXSwiftBusComponent::triggerDownloadingOfXSwiftBusFile()
191  {
192  if (!sGui || !sGui->hasWebDataServices() || sGui->isShuttingDown()) { return; }
193  const CRemoteFile rf = this->getRemoteFileSelected();
194  if (!rf.getBaseName().contains(CBuildConfig::getVersionString()))
195  {
196  const QMessageBox::StandardButton reply = QMessageBox::question(this,
197  "Download XSwiftBus",
198  QStringLiteral(
199  u"The XSwiftBus versions seems to be for a different version\n"
200  u"Your version is '%1'. Use this version.\n\n"
201  u"If not available, you can try the version next to your version number.\n\n"
202  u"Continue with this version?")
203  .arg(CBuildConfig::getVersionString()),
204  QMessageBox::Yes | QMessageBox::No);
205  if (reply != QMessageBox::Yes) { return; }
206  }
207 
208  if (!this->existsDownloadDir())
209  {
210  const CStatusMessage msg = CStatusMessage(this, CLogCategories::validation()).error(u"Invalid download directory");
211  this->showOverlayMessage(msg, CInstallXSwiftBusComponent::OverlayMsgTimeoutMs);
212  return;
213  }
214 
215  const CUrl download = rf.getSmartUrl();
216  if (download.isEmpty())
217  {
218  const CStatusMessage msg = CStatusMessage(this, CLogCategories::validation()).error(u"No download URL for file name '%1'") << rf.getBaseNameAndSize();
219  this->showOverlayMessage(msg, CInstallXSwiftBusComponent::OverlayMsgTimeoutMs);
220  return;
221  }
222 
223  const QString saveAsFile = CFileUtils::appendFilePathsAndFixUnc(ui->le_DownloadDir->text(), rf.getBaseName());
224  const QFile saveFile(saveAsFile);
225  if (saveFile.exists())
226  {
227  const QMessageBox::StandardButton reply = QMessageBox::question(this,
228  "The file already exists",
229  "Do you want to use the existing '" + saveAsFile + "'?",
230  QMessageBox::Yes | QMessageBox::No | QMessageBox::Cancel);
231  if (reply == QMessageBox::Cancel) { return; }
232  if (reply == QMessageBox::Yes)
233  {
234  const CStatusMessage msg = CStatusMessage(this).info(u"Using existing file '%1'") << saveAsFile;
235  const QPointer<CInstallXSwiftBusComponent> guard(this);
236  QTimer::singleShot(100, this, [=] {
237  if (guard.isNull()) { return; }
238  this->downloadedXSwiftBusFile(msg);
239  });
240  return;
241  }
242  }
243 
244  const QNetworkReply *r = sGui->downloadFromNetwork(download, saveAsFile, { this, &CInstallXSwiftBusComponent::downloadedXSwiftBusFile });
245  if (r)
246  {
247  CLogMessage(this).info(u"Triggered downloading of XSwiftBus file from '%1'") << download.getHost();
248  this->showLoading(120 * 1000); // timeout in any case
249  }
250  else
251  {
252  const CStatusMessage msg = CStatusMessage(this, CLogCategories::validation()).error(u"Starting download for '%1' failed") << download.getFullUrl();
253  this->showOverlayMessage(msg, CInstallXSwiftBusComponent::OverlayMsgTimeoutMs);
254  }
255  }

315  void CInstallXSwiftBusComponent::updatesChanged()
316  {
317  const CUpdateInfo updateInfo = m_updates.get();
318  if (updateInfo.getArtifactsXSwiftBus().isEmpty()) { return; }
319  const CArtifactList artifacts = updateInfo.getArtifactsXSwiftBusLatestVersionFirst().findWithUnrestrictedDistributions();
320  if (artifacts.isEmpty()) { return; }
321 
322  const CRemoteFileList remoteFiles = artifacts.asRemoteFiles();
323  if (!remoteFiles.isEmpty())
324  {
325  const QStringList xSwiftBusFiles(remoteFiles.getBaseNamesPlusSize(false));
326  m_xSwiftBusArtifacts = artifacts;
327  ui->cb_DownloadFile->addItems(xSwiftBusFiles);
328 
329  // current text
330  QString current = xSwiftBusFiles.front(); // default latest first
331  if (m_defaultDownloadName.isEmpty())
332  {
333  const CRemoteFile rf = remoteFiles.findFirstContainingNameOrDefault(CBuildConfig::getVersionString(), Qt::CaseInsensitive);
334  if (rf.hasName()) { current = rf.getBaseNameAndSize(); }
335  }
336  else
337  {
338  const CRemoteFile rf = remoteFiles.findFirstByMatchingBaseNameOrDefault(m_defaultDownloadName);
339  if (rf.hasName()) { current = rf.getBaseNameAndSize(); }
340  }
341 
342  ui->cb_DownloadFile->setCurrentText(
343  current.isEmpty() ?
344  remoteFiles.frontOrDefault().getBaseNameAndSize() :
345  current); // latest version
346  }
347  ui->cb_DownloadFile->setEnabled(!remoteFiles.isEmpty());
348  }
