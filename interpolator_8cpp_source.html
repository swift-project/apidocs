115  template <typename Derived>
116  bool CInterpolator<Derived>::presetGroundElevation(CAircraftSituation &situationToPreset, const CAircraftSituation &oldSituation, const CAircraftSituation &newSituation, const CAircraftSituationChange &change)
117  {
118  // IMPORTANT: we do not know what the situation will be (interpolated to), so we cannot transfer
119  situationToPreset.resetGroundElevation();
120  do
121  {
122  if (oldSituation.equalNormalVectorDouble(newSituation))
123  {
124  if (oldSituation.hasGroundElevation())
125  {
126  // same positions, we can use existing elevation
127  // means we were not moving between old an new
128  situationToPreset.transferGroundElevationToMe(oldSituation, true);
129  break;
130  }
131  }
132 
133  const CLength distance = newSituation.calculateGreatCircleDistance(oldSituation);
134  if (distance < newSituation.getDistancePerTime250ms(CElevationPlane::singlePointRadius()))
135  {
136  if (oldSituation.hasGroundElevation())
137  {
138  // almost same positions, we can use existing elevation
139  situationToPreset.transferGroundElevationToMe(oldSituation, true);
140  break;
141  }
142  }
143 
144  if (change.hasElevationDevWithinAllowedRange())
145  {
146  // not much change in known elevations
147  const CAltitudePair elvDevMean = change.getElevationStdDevAndMean();
148  situationToPreset.setGroundElevation(elvDevMean.second, CAircraftSituation::SituationChange);
149  break;
150  }
151 
152  const CElevationPlane epInterpolated = CAircraftSituation::interpolatedElevation(CAircraftSituation::null(), oldSituation, newSituation, distance);
153  if (!epInterpolated.isNull())
154  {
155  situationToPreset.setGroundElevation(epInterpolated, CAircraftSituation::Interpolated);
156  break;
157  }
158  }
159  while (false);
160  return situationToPreset.hasGroundElevation();
161  }

170  template <typename Derived>
171  bool CInterpolator<Derived>::verifyInterpolationSituations(const CAircraftSituation &oldest, const CAircraftSituation &newer, const CAircraftSituation &latest, const CInterpolationAndRenderingSetupPerCallsign &setup)
172  {
173  if (!CBuildConfig::isLocalDeveloperDebugBuild()) { return true; }
174  CAircraftSituationList situations;
175 
176  // oldest last, null ignored
177  if (!latest.isNull()) { situations.push_back(latest); }
178  if (!newer.isNull()) { situations.push_back(newer); }
179  if (!oldest.isNull()) { situations.push_back(oldest); }
180 
181  const bool sorted = situations.isSortedAdjustedLatestFirstWithoutNullPositions();
182  if (setup.isNull() || !setup.isAircraftPartsEnabled()) { return sorted; }
183 
184  bool details = false;
185  if (situations.containsOnGroundDetails(CAircraftSituation::InFromParts))
186  {
187  // if a client supports parts, all ground situations are supposed to be parts based
188  details = situations.areAllOnGroundDetailsSame(CAircraftSituation::InFromParts);
189  BLACK_VERIFY_X(details, Q_FUNC_INFO, "Once gnd.from parts -> always gnd. from parts");
190  }
191 
192  for (const CAircraftSituation &s : situations)
193  {
194  if (!s.hasGroundElevation()) { continue; }
195  BLACK_VERIFY_X(!s.getGroundElevation().isZeroEpsilonConsidered(), Q_FUNC_INFO, "Suspicous 0 gnd. value");
196  }
197 
198  // check if middle situation is missing
199  if (latest.hasGroundElevation() && oldest.hasGroundElevation())
200  {
201  BLACK_VERIFY_X(newer.hasGroundElevation(), Q_FUNC_INFO, "Middle ground elevation is missing");
202  }
203 
204  // result
205  return sorted && details;
206  }

215  template <typename Derived>
216  CInterpolationResult CInterpolator<Derived>::getInterpolation(qint64 currentTimeSinceEpoc, const CInterpolationAndRenderingSetupPerCallsign &setup, int aircraftNumber)
217  {
218  CInterpolationResult result;
219  do
220  {
221  // make sure we can also interpolate parts only (needed in unit tests)
222  if (aircraftNumber < 0) { aircraftNumber = 0; }
223  const bool init = this->initIniterpolationStepData(currentTimeSinceEpoc, setup, aircraftNumber);
224  Q_ASSERT_X(!m_currentInterpolationStatus.isInterpolated(), Q_FUNC_INFO, "Expect reset status");
225  if (!m_unitTest && !init) { break; } // failure in real scenarios, unit tests move on
226  Q_ASSERT_X(m_currentTimeMsSinceEpoch > 0, Q_FUNC_INFO, "No valid timestamp, interpolator initialized?");
227  const CAircraftSituation interpolatedSituation = this->getInterpolatedSituation();
228  const CAircraftParts interpolatedParts = this->getInterpolatedOrGuessedParts(aircraftNumber);
229  result.setValues(interpolatedSituation, interpolatedParts);
230  }
231  while (false);
232 
233  result.setStatus(m_currentInterpolationStatus, m_currentPartsStatus);
234  return result;
235  }

237  template <typename Derived>
238  CAircraftSituation CInterpolator<Derived>::getInterpolatedSituation()
239  {
240  Q_ASSERT_X(!m_currentInterpolationStatus.isInterpolated(), Q_FUNC_INFO, "Expect reset status");
241  if (m_currentSituations.isEmpty())
242  {
243  m_lastSituation = CAircraftSituation::null();
244  return CAircraftSituation::null();
245  }
246 
247  // interpolant as function of derived class
248  // CInterpolatorLinear::Interpolant or CInterpolatorSpline::Interpolant
249  SituationLog log;
250  const auto interpolant = derived()->getInterpolant(log);
251  const bool isValidInterpolant = interpolant.isValid();
252 
253  CAircraftSituation currentSituation = m_lastSituation;
254  CAircraftSituation::AltitudeCorrection altCorrection = CAircraftSituation::NoCorrection;
255 
256  bool isValidInterpolation = false;
257  do
258  {
259  if (!isValidInterpolant) { break; }
260  const CInterpolatorPbh pbh = interpolant.pbh();
261 
262  // init interpolated situation
263  currentSituation = this->initInterpolatedSituation(pbh.getOldSituation(), pbh.getNewSituation());
264 
265  // Pitch bank heading first, so follow up steps could use those values
266  currentSituation.setHeading(pbh.getHeading());
267  currentSituation.setPitch(pbh.getPitch());
268  currentSituation.setBank(pbh.getBank());
269  currentSituation.setGroundSpeed(pbh.getGroundSpeed());
270 
271  // use derived interpolant function
272  const bool interpolateGndFlag = pbh.getNewSituation().hasGroundDetailsForGndInterpolation() && pbh.getOldSituation().hasGroundDetailsForGndInterpolation();
273  currentSituation = interpolant.interpolatePositionAndAltitude(currentSituation, interpolateGndFlag);
274  if (currentSituation.isNull()) { break; }
275 
276  // if we get here and the vector is invalid it means we haven't handled it correctly in one of the interpolators
277  if (!currentSituation.isValidVectorRange())
278  {
279  if (CBuildConfig::isLocalDeveloperDebugBuild()) { BLACK_VERIFY_X(false, Q_FUNC_INFO, "Invalid interpolation situation"); }
280  return CAircraftSituation::null();
281  }
282 
283  // GND flag.
284  if (!interpolateGndFlag) { CAircraftSituationChange::null().guessOnGround(currentSituation, m_model); }
285 
286  // as we now have the position and can interpolate elevation
287  currentSituation.interpolateElevation(pbh.getOldSituation(), pbh.getNewSituation());
288  if (!currentSituation.hasGroundElevation())
289  {
290  // we still have no elevation
291  const CLength radius = currentSituation.getDistancePerTime250ms(CElevationPlane::singlePointRadius());
292  if (!m_lastSituation.transferGroundElevationFromMe(currentSituation, radius))
293  {
294  if (currentSituation.canLikelySkipNearGroundInterpolation())
295  {
296  // skipped
297  }
298  else
299  {
300  const CElevationPlane groundElevation = this->findClosestElevationWithinRange(currentSituation, radius);
301  m_lastSituation.setGroundElevationChecked(groundElevation, CAircraftSituation::FromCache);
302  }
303  }
304  }
305 
306  // correct altitude itself
307  if (!interpolateGndFlag && currentSituation.getOnGroundDetails() != CAircraftSituation::OnGroundByGuessing)
308  {
309  // just in case
310  altCorrection = currentSituation.correctAltitude(true); // we have CG set
311  }
312 
313  // correct pitch on ground
314  if (currentSituation.isOnGround())
315  {
316  const CAngle correctedPitchOnGround = m_currentSetup.getPitchOnGround();
317  if (!correctedPitchOnGround.isNull())
318  {
319  currentSituation.setPitch(correctedPitchOnGround);
320  }
321  }
322 
323  isValidInterpolation = true;
324  }
325  while (false);
326 
327  const bool valid = isValidInterpolant && isValidInterpolation;
328  if (!valid)
329  {
330  // further handling could go here, mainly we continue with last situation
331  m_invalidSituations++;
332 
333  // avoid flooding of log.
334  if (m_currentTimeMsSinceEpoch - m_lastInvalidLogTs > m_lastSituation.getTimeOffsetMs())
335  {
336  m_lastInvalidLogTs = m_currentTimeMsSinceEpoch;
337  const bool noSituation = m_lastSituation.isNull();
338 
339  // Problem 1, we have no "last situation"
340  // Problem 2, "it takes too long to recover"
341  CStatusMessage m;
342  if (noSituation)
343  {
344  m = CStatusMessage(this).warning(u"No situation #%1 for interpolation reported for '%2' (Interpolant: %3 interpolation: %4)") << m_invalidSituations << m_callsign.asString() << boolToTrueFalse(isValidInterpolant) << boolToTrueFalse(isValidInterpolation);
345  }
346  else
347  {
348  const qint64 diff = m_currentTimeMsSinceEpoch - currentSituation.getAdjustedMSecsSinceEpoch();
349  m = CStatusMessage(this).warning(u"Invalid situation, diff. %1ms #%2 for interpolation reported for '%3' (Interpolant: %4 interpolation: %5)") << diff << m_invalidSituations << m_callsign.asString() << boolToTrueFalse(isValidInterpolant) << boolToTrueFalse(isValidInterpolation);
350  }
351  if (!m.isEmpty())
352  {
353  if (m_interpolationMessages.sizeInt() == 2)
354  {
355  // display second message as a hint in the general log
356  // we DO NOT display the first message, as this can happen due to pilot logging off
357  // if it happens twice we consider it worth displaying
358  CLogMessage::preformatted(m);
359  }
360  m_interpolationMessages.push_back(m);
361  }
362  }
363  } // valid?
364 
365  // situation and status
366  if (valid)
367  {
368  Q_ASSERT_X(currentSituation.hasMSLGeodeticHeight(), Q_FUNC_INFO, "No MSL altitude");
369  m_lastSituation = currentSituation;
370  m_currentInterpolationStatus.setInterpolatedAndCheckSituation(valid, currentSituation);
371  }
372  else
373  {
374  currentSituation = m_lastSituation;
375  m_currentInterpolationStatus.setSameSituation(true);
376  m_currentInterpolationStatus.setInterpolatedAndCheckSituation(valid, currentSituation);
377  }
378 
379  // logging
380  if (this->doLogging())
381  {
382  log.tsCurrent = m_currentTimeMsSinceEpoch;
383  log.callsign = m_callsign;
384  log.groundFactor = currentSituation.getOnGroundFactor();
385  log.altCorrection = CAircraftSituation::altitudeCorrectionToString(altCorrection);
386  log.situationCurrent = currentSituation;
387  log.interpolantRecalc = interpolant.isRecalculated();
388  log.change = m_pastSituationsChange;
389  log.usedSetup = m_currentSetup;
390  log.elevationInfo = this->getElevationsFoundMissedInfo();
391  log.cgAboveGround = currentSituation.getCG();
392  log.sceneryOffset = m_currentSceneryOffset;
393  log.noInvalidSituations = m_invalidSituations;
394  log.noNetworkSituations = m_currentSituations.sizeInt();
395  log.useParts = this->isRemoteAircraftSupportingParts(m_callsign);
396  m_logger->logInterpolation(log);
397 
398  // if (log.interpolantRecalc) { CLogMessage(this).debug(u"Recalc %1") << log.callsign.asString(); }
399  }
400 
401  // bye
402  return currentSituation;
403  }

405  template <typename Derived>
406  CAircraftParts CInterpolator<Derived>::getInterpolatedParts()
407  {
408  // (!) this code is used by linear and spline interpolator
409 
410  // Parts are supposed to be in correct order, latest first
411  const CAircraftPartsList validParts = this->remoteAircraftParts(m_callsign);
412 
413  // log for empty parts aircraft parts
414  if (validParts.isEmpty())
415  {
416  static const CAircraftParts emptyParts;
417  this->logParts(emptyParts, validParts.size(), true);
418  return emptyParts;
419  }
420 
421  m_currentPartsStatus.setSupportsParts(true);
422  CAircraftParts currentParts;
423  do
424  {
425  // find the first parts earlier than the current time
426  const auto pivot = std::partition_point(validParts.begin(), validParts.end(), [=](auto &&p) { return p.getAdjustedMSecsSinceEpoch() > m_currentTimeMsSinceEpoch; });
427  const auto partsNewer = makeRange(validParts.begin(), pivot).reverse();
428  const auto partsOlder = makeRange(pivot, validParts.end());
429 
430  // if (partsOlder.isEmpty()) { currentParts = *(partsNewer.end() - 1); break; }
431  if (partsOlder.isEmpty())
432  {
433  currentParts = *(partsNewer.begin());
434  break;
435  }
436  currentParts = partsOlder.front(); // latest older parts
437  }
438  while (false);
439 
440  this->logParts(currentParts, validParts.size(), false);
441  return currentParts;
442  }

444  template <typename Derived>
445  CAircraftParts CInterpolator<Derived>::getInterpolatedOrGuessedParts(int aircraftNumber)
446  {
447  Q_ASSERT_X(m_partsToSituationInterpolationRatio >= 1 && m_partsToSituationInterpolationRatio < 11, Q_FUNC_INFO, "Wrong ratio");
448  const bool needParts = m_unitTest || m_lastParts.isNull();
449  const bool doInterpolation = needParts || ((m_interpolatedSituationsCounter + aircraftNumber) % m_partsToSituationInterpolationRatio == 0);
450  const bool doGuess = needParts || ((m_interpolatedSituationsCounter + aircraftNumber) % m_partsToSituationGuessingRatio == 0);
451 
452  if (!doGuess && !doInterpolation)
453  {
454  // reuse
455  return this->logAndReturnNullParts("neither guess nor interpolation", true);
456  }
457 
458  CAircraftParts parts = CAircraftParts::null();
459  if (m_currentSetup.isAircraftPartsEnabled())
460  {
461  // this already logs and sets status
462  parts = this->getInterpolatedParts();
463  }
464 
465  // if we have supported parts, we skip this step, but it can happen the parts are still empty
466  if (!m_currentPartsStatus.isSupportingParts())
467  {
468  if (!doGuess)
469  {
470  return this->logAndReturnNullParts("not supporting parts, and marked for guessing", true);
471  }
472 
473  // check if model has been thru model matching
474  if (!m_lastSituation.isNull())
475  {
476  parts = guessParts(m_lastSituation, m_pastSituationsChange, m_model);
477  this->logParts(parts, 0, false);
478  }
479  else
480  {
481  // quite normal initial situation, just return NULL
482  return this->logAndReturnNullParts("guessing, but no situation yet", false);
483  }
484  }
485 
486  m_lastParts = parts;
487  m_lastPartsStatus = m_currentPartsStatus;
488  return parts;
489  }

517  template <typename Derived>
518  CAircraftParts CInterpolator<Derived>::guessParts(const CAircraftSituation &situation, const CAircraftSituationChange &change, const CAircraftModel &model)
519  {
520  CAircraftParts parts;
521  parts.setMSecsSinceEpoch(situation.getMSecsSinceEpoch());
522  parts.setTimeOffsetMs(situation.getTimeOffsetMs());
523  parts.setPartsDetails(CAircraftParts::GuessedParts);
524  parts.setLights(situation.guessLights());
525 
526  QString *details = /*CBuildConfig::isLocalDeveloperDebugBuild() ? &parts.m_guessingDetails :*/ nullptr;
527 
528  CAircraftEngineList engines;
529  const bool vtol = model.isVtol();
530  const int engineCount = model.getEngineCount();
531  CSpeed guessedVRotate = CSpeed::null();
532  CLength guessedCG = model.getCG();
533  model.getAircraftIcaoCode().guessModelParameters(guessedCG, guessedVRotate);
534 
535  if (situation.getOnGroundDetails() != CAircraftSituation::NotSetGroundDetails)
536  {
537  do
538  {
539  // set some reasonable values
540  const bool isOnGround = situation.isOnGround();
541  engines.initEngines(engineCount, !isOnGround || situation.isMoving());
542  parts.setGearDown(isOnGround);
543  parts.setSpoilersOut(false);
544  parts.setEngines(engines);
545 
546  if (!change.isNull())
547  {
548  if (change.isConstDecelarating())
549  {
550  parts.setSpoilersOut(true);
551  parts.setFlapsPercent(10);
552  break;
553  }
554  }
555 
556  const CSpeed slowSpeed = guessedVRotate * 0.30;
557  if (situation.getGroundSpeed() < slowSpeed)
558  {
559  if (details) { *details += u"slow speed <" % slowSpeed.valueRoundedWithUnit(1) % u" on ground"; }
560  parts.setFlapsPercent(0);
561  break;
562  }
563  else
564  {
565  if (details) { *details += u"faster speed >" % slowSpeed.valueRoundedWithUnit(1) % u" on ground"; }
566  parts.setFlapsPercent(0);
567  break;
568  }
569  }
570  while (false);
571  }
572  else
573  {
574  if (details) { *details = QStringLiteral("no ground info"); }
575 
576  // no idea if on ground or not
577  engines.initEngines(engineCount, true);
578  parts.setEngines(engines);
579  parts.setGearDown(true);
580  parts.setSpoilersOut(false);
581  }
582 
583  const double pitchDeg = situation.getPitch().value(CAngleUnit::deg());
584  const bool isLikelyTakeOffOrClimbing = change.isNull() ? pitchDeg > 20 : (change.isRotatingUp() || change.isConstAscending());
585  const bool isLikelyLanding = change.isNull() ? false : change.isConstDescending();
586 
587  if (situation.hasGroundElevation())
588  {
589  const CLength aboveGnd = situation.getHeightAboveGround();
590  if (aboveGnd.isNull() || std::isnan(aboveGnd.value()))
591  {
592  BLACK_VERIFY_X(false, Q_FUNC_INFO, "above gnd.is null");
593  return parts;
594  }
595 
596  const double nearGround1Ft = 300;
597  const double nearGround2Ft = isLikelyTakeOffOrClimbing ? 500 : 1000;
598  const double aGroundFt = aboveGnd.value(CLengthUnit::ft());
599  static const QString detailsInfo("above ground: %1ft near grounds: %2ft %3ft likely takeoff: %4 likely landing: %5");
600 
601  if (details) { *details = detailsInfo.arg(aGroundFt).arg(nearGround1Ft).arg(nearGround2Ft).arg(boolToYesNo(isLikelyTakeOffOrClimbing), boolToYesNo(isLikelyLanding)); }
602  if (aGroundFt < nearGround1Ft)
603  {
604  if (details) { details->prepend(QStringLiteral("near ground: ")); }
605  parts.setGearDown(true);
606  parts.setFlapsPercent(25);
607  }
608  else if (aGroundFt < nearGround2Ft)
609  {
610  if (details) { details->prepend(QStringLiteral("2nd layer: ")); }
611  const bool gearDown = !isLikelyTakeOffOrClimbing && (situation.getGroundSpeed() < guessedVRotate || isLikelyLanding);
612  parts.setGearDown(gearDown);
613  parts.setFlapsPercent(10);
614  }
615  else
616  {
617  if (details) { details->prepend(QStringLiteral("airborne: ")); }
618  parts.setGearDown(false);
619  parts.setFlapsPercent(0);
620  }
621  }
622  else
623  {
624  if (situation.getOnGroundDetails() != CAircraftSituation::NotSetGroundDetails)
625  {
626  // we have no ground elevation but a ground info
627  if (situation.getOnGroundDetails() == CAircraftSituation::OnGroundByGuessing)
628  {
629  // should be OK
630  if (details) { *details = QStringLiteral("on ground, no elv."); }
631  }
632  else
633  {
634  if (!vtol)
635  {
636  const bool gearDown = situation.getGroundSpeed() < guessedVRotate;
637  parts.setGearDown(gearDown);
638  if (details) { *details = QStringLiteral("not on ground elv., gs < ") + guessedVRotate.valueRoundedWithUnit(1); }
639  }
640  }
641  }
642  }

644  return parts;

698  template <typename Derived>
699  bool CInterpolator<Derived>::initIniterpolationStepData(qint64 currentTimeSinceEpoc, const CInterpolationAndRenderingSetupPerCallsign &setup, int aircraftNumber)
700  {
701  Q_ASSERT_X(!m_callsign.isEmpty(), Q_FUNC_INFO, "Missing callsign");
702 
703  const qint64 lastModifed = this->situationsLastModified(m_callsign);
704  const bool slowUpdateStep = (((m_interpolatedSituationsCounter + aircraftNumber) % 25) == 0); // flag when parts are updated, which need not to be updated every time
705  const bool changedSituations = lastModifed > m_situationsLastModified;
706 
707  m_currentTimeMsSinceEpoch = currentTimeSinceEpoc;
708  m_currentInterpolationStatus.reset();
709  m_currentPartsStatus.reset();
710  m_currentSetup = setup;
711 
712  if (changedSituations)
713  {
714  m_situationsLastModified = lastModifed;
715  m_currentSituations = this->remoteAircraftSituationsAndChange(setup); // only update when needed
716  }
717 
718  if (!m_model.hasCG() || slowUpdateStep)
719  {
720  this->getAndFetchModelCG(CLength::null()); // update CG
721  }
722 
723  bool success = false;
724  const int situationsSize = m_currentSituations.sizeInt();
725  m_currentInterpolationStatus.setSituationsCount(situationsSize);
726  if (m_currentSituations.isEmpty())
727  {
728  const bool inRange = this->isAircraftInRange(m_callsign);
729  m_lastSituation = CAircraftSituation::null(); // no interpolation possible for that step
730  static const QString extraNoSituations("No situations, but remote aircraft '%1'");
731  static const QString extraNoRemoteAircraft("Unknown remote aircraft: '%1'");
732  m_currentInterpolationStatus.setExtraInfo((inRange ? extraNoSituations : extraNoRemoteAircraft).arg(m_callsign.asString()));
733  }
734  else
735  {
736  success = true;
737  m_interpolatedSituationsCounter++; // counter updated in initIniterpolationStepData
738 
739  // with the latest updates of T243 the order and the offsets are supposed to be correct
740  // so even mixing fast/slow updates shall work
741  if (!CBuildConfig::isReleaseBuild())
742  {
743  Q_ASSERT_X(m_currentSituations.isSortedAdjustedLatestFirstWithoutNullPositions(), Q_FUNC_INFO, "Wrong sort order");
744  Q_ASSERT_X(m_currentSituations.size() <= IRemoteAircraftProvider::MaxSituationsPerCallsign, Q_FUNC_INFO, "Wrong size");
745  }
746  }
747 
748  return success;
