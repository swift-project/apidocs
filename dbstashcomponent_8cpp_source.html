91  CStatusMessage CDbStashComponent::stashModel(const CAircraftModel &model, bool replace, bool consolidateWithDbData, bool clearHighlighting)
92  {
93  const CAircraftModel stashModel(consolidateWithDbData ? this->consolidateModel(model) : model);
94  const CStatusMessage m(this->validateStashModel(stashModel, replace));
95  if (!m.isWarningOrAbove())
96  {
97  if (clearHighlighting) { this->clearValidationHighlighting(); }
98  if (replace)
99  {
100  ui->tvp_StashAircraftModels->replaceOrAdd(&CAircraftModel::getModelString, stashModel.getModelString(), stashModel);
101  }
102  else
103  {
104  ui->tvp_StashAircraftModels->insert(stashModel);
105  }
106  }
107  return m;
108  }

110  CStatusMessageList CDbStashComponent::stashModels(const CAircraftModelList &models, bool replace, bool consolidateWithDbData, bool clearHighlighting)
111  {
112  if (models.isEmpty()) { return CStatusMessageList(); }
113  CStatusMessageList msgs;
114  int successfullyAdded = 0;
115  for (const CAircraftModel &model : models)
116  {
117  const CStatusMessage m(this->stashModel(model, replace, consolidateWithDbData, false));
118  if (m.isWarningOrAbove())
119  {
120  msgs.push_back(m);
121  }
122  else
123  {
124  successfullyAdded++;
125  }
126  }
127  if (successfullyAdded > 0 && clearHighlighting) { this->clearValidationHighlighting(); }
128  return msgs;
129  }

259  void CDbStashComponent::onPublishPressed()
260  {
261  if (!sGui || sGui->isShuttingDown() || !sGui->hasWebDataServices()) { return; }
262  if (ui->tvp_StashAircraftModels->isEmpty()) { return; }
263  if (!sGui->hasMinimumMappingVersion()) { return; }
264 
265  // get models right here, because later steps might affect selection
266  const CAircraftModelList models(getSelectedOrAllModels());
267  if (models.isEmpty()) { return; }
268 
269  // validate
270  CAircraftModelList validModels;
271  CAircraftModelList invalidModels;
272  if (!this->validateAndDisplay(validModels, invalidModels)) { return; }
273  CStatusMessageList msgs;
274  if (validModels.size() > MaxModelPublished)
275  {
276  validModels.truncate(MaxModelPublished);
277  msgs.push_back(CStatusMessage(validationCategories(), CStatusMessage::SeverityWarning, u"More than %1 values, values skipped") << MaxModelPublished);
278  }
279 
280  msgs.push_back(sGui->getWebDataServices()->asyncPublishModels(validModels));
281  if (msgs.hasWarningOrErrorMessages())
282  {
283  this->showOverlayMessages(msgs);
284  }
285  else
286  {
287  ui->tvp_StashAircraftModels->showLoadIndicator();
288  }

319  CStatusMessageList CDbStashComponent::validate(CAircraftModelList &validModels, CAircraftModelList &invalidModels) const
320  {
321  if (ui->tvp_StashAircraftModels->isEmpty()) { return CStatusMessageList(); }
322  Q_ASSERT_X(sGui->getWebDataServices(), Q_FUNC_INFO, "No web services");
323  const CAircraftModelList models(getSelectedOrAllModels());
324  if (models.isEmpty()) { return CStatusMessageList(); }
325  const bool ignoreEqual = ui->cb_ChangedOnly->isChecked();
326  const CStatusMessageList msgs(sGui->getWebDataServices()->validateForPublishing(models, ignoreEqual, validModels, invalidModels));
327 
328  // OK?
329  if (msgs.isEmpty())
330  {
331  return CStatusMessageList(
332  { CStatusMessage(validationCategories(), CStatusMessage::SeverityInfo, u"No errors in %1 model(s)") << models.size() });
333  }
334  else
335  {
336  return msgs;
337  }
338  }

411  CAircraftModel CDbStashComponent::consolidateModel(const CAircraftModel &model) const
412  {
413  CAircraftModel stashModel(model);
414  const bool ownModel = stashModel.getModelType() == CAircraftModel::TypeOwnSimulatorModel;
415 
416  // merge/update with DB data if any
417  // this is a forced update with DB data, as DB data can change
418  stashModel = this->consolidateWithDbData(stashModel, true);
419 
420  // merge with own models if any
421  if (!ownModel)
422  {
423  stashModel = this->consolidateWithOwnModels(stashModel);
424  }
425 
426  return stashModel;
427  }
