60  static QString descriptionForFlyableModel(const CAircraftModel &model)
61  {
62  if (!model.getName().isEmpty())
63  {
64  if (model.getDistributor().hasDescription() && !model.getName().contains(model.getDistributor().getDescription()))
65  {
66  return u"[ACF] " % model.getName() % u" by " % model.getDistributor().getDescription();
67  }
68  else
69  {
70  return u"[ACF] " % model.getName();

81  return u"[ACF] " % model.getAircraftIcaoCodeDesignator();

83  }

96  void CAircraftModelLoaderXPlane::startLoadingFromDisk(LoadMode mode, const ModelConsolidationCallback &modelConsolidation, const QStringList &modelDirectories)
97  {
98  const CSimulatorInfo simulator = CSimulatorInfo::xplane();
99  const QStringList modelDirs = this->getInitializedModelDirectories(modelDirectories, simulator);
100  const QStringList excludedDirectoryPatterns(m_settings.getModelExcludeDirectoryPatternsOrDefault(simulator)); // copy
101 
102  if (modelDirs.isEmpty())
103  {
104  this->clearCachedModels(CSimulatorInfo::xplane());
105  emit this->loadingFinished(CStatusMessage(this, CStatusMessage::SeverityError, u"XPlane model directories '%1' are empty") << modelDirectories.join(", "), simulator, ParsedData);
106  return;
107  }
108 
109  if (mode.testFlag(LoadInBackground))
110  {
111  if (m_parserWorker && !m_parserWorker->isFinished()) { return; }
112  emit this->diskLoadingStarted(simulator, mode);
113 
114  m_parserWorker = CWorker::fromTask(this, "CAircraftModelLoaderXPlane::performParsing",
115  [this, modelDirs, excludedDirectoryPatterns, modelConsolidation]() {
116  auto models = this->performParsing(modelDirs, excludedDirectoryPatterns);
117  if (modelConsolidation) { modelConsolidation(models, true); }
118  return models;
119  });
120  m_parserWorker->thenWithResult<CAircraftModelList>(this, [=](const auto &models) {
121  this->updateInstalledModels(models);
122  m_loadingMessages.freezeOrder();
123  emit this->loadingFinished(m_loadingMessages, simulator, ParsedData);
124  });
125  }
126  else if (mode.testFlag(LoadDirectly))
127  {
128  emit this->diskLoadingStarted(simulator, mode);
129  CAircraftModelList models(this->performParsing(modelDirs, excludedDirectoryPatterns));
130  this->updateInstalledModels(models);
131  }
132  }

177  CAircraftModelList CAircraftModelLoaderXPlane::parseFlyableAirplanes(const QString &rootDirectory, const QStringList &excludeDirectories)
178  {
179  Q_UNUSED(excludeDirectories)
180  if (rootDirectory.isEmpty()) { return {}; }
181 
182  QDir searchPath(rootDirectory, fileFilterFlyable());
183  QDirIterator aircraftIt(searchPath, QDirIterator::Subdirectories | QDirIterator::FollowSymlinks);
184 
185  emit loadingProgress(this->getSimulator(), QStringLiteral("Parsing flyable airplanes in '%1'").arg(rootDirectory), -1);
186 
187  CAircraftModelList installedModels;
188  while (aircraftIt.hasNext())
189  {
190  aircraftIt.next();
191  if (CFileUtils::isExcludedDirectory(aircraftIt.fileInfo(), excludeDirectories, Qt::CaseInsensitive)) { continue; }
192 
193  using namespace BlackMisc::Simulation::XPlane::QtFreeUtils;
194  AcfProperties acfProperties = extractAcfProperties(aircraftIt.filePath().toStdString());
195 
196  const CDistributor dist({}, QString::fromStdString(acfProperties.author), {}, {}, CSimulatorInfo::XPLANE);
197  CAircraftModel model;
198  model.setAircraftIcaoCode(QString::fromStdString(acfProperties.aircraftIcaoCode));
199  model.setDescription(QString::fromStdString(acfProperties.modelDescription));
200  model.setName(QString::fromStdString(acfProperties.modelName));
201  model.setDistributor(dist);
202  model.setModelString(QString::fromStdString(acfProperties.modelString));
203  if (!model.hasDescription()) { model.setDescription(descriptionForFlyableModel(model)); }
204  model.setModelType(CAircraftModel::TypeOwnSimulatorModel);
205  model.setSimulator(CSimulatorInfo::xplane());
206  model.setFileDetailsAndTimestamp(aircraftIt.fileInfo());
207  model.setModelMode(CAircraftModel::Exclude);
208  addUniqueModel(model, installedModels);
209 
210  const QString baseModelString = model.getModelString();
211  QDirIterator liveryIt(CFileUtils::appendFilePaths(aircraftIt.fileInfo().canonicalPath(), QStringLiteral("liveries")), QDir::Dirs | QDir::NoDotAndDotDot);
212  emit this->loadingProgress(this->getSimulator(), QStringLiteral("Parsing flyable liveries in '%1'").arg(aircraftIt.fileInfo().canonicalPath()), -1);
213  while (liveryIt.hasNext())
214  {
215  liveryIt.next();
216  model.setModelString(baseModelString % u' ' % liveryIt.fileName());
217  addUniqueModel(model, installedModels);
218  }
219  }
220  return installedModels;
221  }

223  CAircraftModelList CAircraftModelLoaderXPlane::parseCslPackages(const QString &rootDirectory, const QStringList &excludeDirectories)
224  {
225  Q_UNUSED(excludeDirectories);
226  if (rootDirectory.isEmpty()) { return {}; }
227 
228  m_cslPackages.clear();
229 
230  QDir searchPath(rootDirectory, fileFilterCsl());
231  QDirIterator it(searchPath, QDirIterator::Subdirectories);
232  while (it.hasNext())
233  {
234  QString packageFile = it.next();
235  if (CFileUtils::isExcludedDirectory(it.filePath(), excludeDirectories)) { continue; }
236 
237  const QString packageFilePath = it.fileInfo().absolutePath();
238  QFile file(packageFile);
239  file.open(QIODevice::ReadOnly);
240  QString content;
241 
242  QTextStream ts(&file);
243  content.append(ts.readAll());
244  file.close();
245 
246  const auto package = parsePackageHeader(packageFilePath, content);
247  if (package.hasValidHeader()) m_cslPackages.push_back(package);
248  }
249 
250  CAircraftModelList installedModels;
251 
252  // Now we do a full run
253  for (auto &package : m_cslPackages)
254  {
255  const QString packageFile = CFileUtils::appendFilePaths(package.path, QStringLiteral("xsb_aircraft.txt"));
256  emit this->loadingProgress(this->getSimulator(), QStringLiteral("Parsing CSL '%1'").arg(packageFile), -1);
257 
258  QFile file(packageFile);
259  file.open(QIODevice::ReadOnly);
260  QString content;
261 
262  QTextStream ts(&file);
263  content.append(ts.readAll());
264  file.close();
265  parseFullPackage(content, package);
266 
267  for (const auto &plane : std::as_const(package.planes))
268  {
269  if (installedModels.containsModelString(plane.getModelName()))
270  {
271  const CStatusMessage msg = CStatusMessage(this).warning(u"XPlane model '%1' exists already! Potential model string conflict! Ignoring it.") << plane.getModelName();
272  m_loadingMessages.push_back(msg);
273  continue;
274  }
275 
276  CAircraftModel model(plane.getModelName(), CAircraftModel::TypeOwnSimulatorModel);
277  const CAircraftIcaoCode icao(plane.icao);
278  const QFileInfo modelFileInfo(plane.filePath);
279  model.setFileDetailsAndTimestamp(modelFileInfo);
280  model.setAircraftIcaoCode(icao);
281 
282  if (CBuildConfig::isLocalDeveloperDebugBuild())
283  {
284  BLACK_AUDIT_X(modelFileInfo.exists(), Q_FUNC_INFO, "Model does NOT exist");
285  }
286 
287  CLivery livery;
288  livery.setCombinedCode(plane.livery);
289  CAirlineIcaoCode airline(plane.airline);
290  livery.setAirlineIcaoCode(airline);
291  model.setLivery(livery);
292 
293  model.setSimulator(CSimulatorInfo::xplane());
294  QString modelDescription("[CSL]");
295  if (plane.objectVersion == CSLPlane::OBJ7) { modelDescription += "[OBJ7]"; }
296  else if (plane.objectVersion == CSLPlane::OBJ8) { modelDescription += "[OBJ8]"; }
297  model.setDescription(modelDescription);
298  installedModels.push_back(model);
299  }
300  }
301  return installedModels;
302  }

318  bool CAircraftModelLoaderXPlane::parseExportCommand(const QStringList &tokens, CSLPackage &package, const QString &path, int lineNum)
319  {
320  if (tokens.size() != 2)
321  {

333  }
334  else

336  const CStatusMessage m = CStatusMessage(this).error(u"XPlane package name '%1' already in use by '%2' reqested by use by '%3'") << tokens[1] << p->path << path;
337  m_loadingMessages.push_back(m);
338  return false;

423  bool CAircraftModelLoaderXPlane::parseObj8Command(const QStringList &tokens, CSLPackage &package, const QString &path, int lineNum)
424  {
425  // OBJ8 <group> <animate YES|NO> <filename>
426  if (tokens.size() != 4)
427  {
428  if (tokens.size() == 5 || tokens.size() == 6)
429  {
430  const CStatusMessage m = CStatusMessage(this).error(u"%1/xsb_aircraft.txt Line %2 : Unsupported IVAO CSL format - consider using CSL2XSB.") << path << lineNum;
431  m_loadingMessages.push_back(m);
432  }
433  else
434  {
435  const CStatusMessage m = CStatusMessage(this).error(u"%1/xsb_aircraft.txt Line %2 : OBJ8 command takes 3 arguments.") << path << lineNum;
436  m_loadingMessages.push_back(m);
437  }
438  return false;
439  }
440  if (package.planes.isEmpty())
441  {
442  m_loadingMessages.push_back(CStatusMessage(this).error(u"%1/xsb_aircraft.txt Line %2 : invalid position for command.") << path << lineNum);
443  return false;
444  }
445 
446  if (tokens[1] != "SOLID") { return true; }
447 
448  QString relativePath(tokens[3]);
449  normalizePath(relativePath);
450  QString fullPath(relativePath);
451  if (!doPackageSub(fullPath))
452  {
453  const CStatusMessage m = CStatusMessage(this).error(u"%1/xsb_aircraft.txt Line %2 : package not found.") << path << lineNum;
454  m_loadingMessages.push_back(m);
455  return false;
456  }
457 
458  // just the name of the dir containing xsbaircraft.txt
459  QString packageRootDir = package.path.mid(package.path.lastIndexOf('/') + 1);
460  package.planes.back().dirNames = QStringList { packageRootDir };
461 
462  package.planes.back().filePath = fullPath;
463  return true;
464  }

546  CAircraftModelLoaderXPlane::CSLPackage CAircraftModelLoaderXPlane::parsePackageHeader(const QString &path, const QString &content)
547  {
548  using command = std::function<bool(const QStringList &, CSLPackage &, const QString &, int)>;
549  using namespace std::placeholders;
550 
551  const QMap<QString, command> commands {
552  { "EXPORT_NAME", std::bind(&CAircraftModelLoaderXPlane::parseExportCommand, this, _1, _2, _3, _4) }
553  };
554 
555  CSLPackage package;
556  int lineNum = 0;
557 
558  QString localCopy(content);
559  QTextStream in(&localCopy);
560  while (!in.atEnd())
561  {
562  ++lineNum;
563  QString line = in.readLine();
564  auto tokens = splitString(line, [](QChar c) { return c.isSpace(); });
565  if (!tokens.empty())
566  {
567  auto it = commands.find(tokens[0]);
568  if (it != commands.end())
569  {
570  bool result = it.value()(tokens, package, path, lineNum);
571  // Stop loop once we found EXPORT command
572  if (result) break;
573  }
574  }
575  }
576  return package;
577  }

579  void CAircraftModelLoaderXPlane::parseFullPackage(const QString &content, CSLPackage &package)
580  {
581  using command = std::function<bool(const QStringList &, CSLPackage &, const QString &, int)>;
582  using namespace std::placeholders;
583 
584  const QMap<QString, command> commands {
585  { "EXPORT_NAME", std::bind(&CAircraftModelLoaderXPlane::parseDummyCommand, this, _1, _2, _3, _4) },
586  { "DEPENDENCY", std::bind(&CAircraftModelLoaderXPlane::parseDependencyCommand, this, _1, _2, _3, _4) },
587  { "OBJECT", std::bind(&CAircraftModelLoaderXPlane::parseObjectCommand, this, _1, _2, _3, _4) },
588  { "TEXTURE", std::bind(&CAircraftModelLoaderXPlane::parseTextureCommand, this, _1, _2, _3, _4) },
589  { "AIRCRAFT", std::bind(&CAircraftModelLoaderXPlane::parseAircraftCommand, this, _1, _2, _3, _4) },
