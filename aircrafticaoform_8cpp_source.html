84  void CAircraftIcaoForm::jsonPasted(const QString &json)
85  {
86  if (json.isEmpty()) { return; } // avoid unnecessary conversions
87  try
88  {
89  CVariant jsonVariant;
90  jsonVariant.convertFromJson(Json::jsonObjectFromString(json));
91  if (!jsonVariant.canConvert<CAircraftIcaoCodeList>()) { return; }
92  const CAircraftIcaoCodeList icaos = jsonVariant.value<CAircraftIcaoCodeList>();
93  if (!icaos.isEmpty())
94  {
95  this->setValue(icaos.front());
96  }
97  }
98  catch (const CJsonException &ex)
99  {
100  Q_UNUSED(ex);
101  }
102  }

104  CAircraftIcaoCode CAircraftIcaoForm::getValue() const
105  {
106  CAircraftIcaoCode icao(ui->aircraft_Selector->getAircraftIcao());
107  if (!icao.hasValidDbKey() && sGui && sGui->getWebDataServices())
108  {
109  // not based on DB yet, do we have a DB key
110  int k = this->getDbKeyFromGui();
111  if (k >= 0)
112  {
113  // we got an id, we get the DB object for it
114  CAircraftIcaoCode fromDb(sGui->getWebDataServices()->getAircraftIcaoCodeForDbKey(k));
115  if (fromDb.getDesignator() == icao.getDesignator())
116  {
117  // we replace by DB object
118  icao = fromDb;
119  }
120  }
121  }
122 
123  const QString manufacturer(ui->le_Manufacturer->text().trimmed().toUpper());
124  const QString modelDescription(ui->le_ModelDescription->text().trimmed());
125  const QString iata(ui->le_Iata->text().trimmed().toUpper());
126  const QString family(ui->le_Family->text().trimmed().toUpper());
127  const QString wtc(ui->cb_Wtc->currentText().left(1));
128  const QString combined(ui->combined_TypeSelector->getCombinedType());
129  bool ok;
130  int rank = ui->cb_Rank->currentText().toInt(&ok);
131  if (!ok) { rank = 10; }
132  bool legacy = ui->cb_Legacy->isChecked();
133  bool military = ui->cb_Military->isChecked();
134  bool realWorld = ui->cb_RealWorld->isChecked();
135  icao.setManufacturer(manufacturer);
136  icao.setModelDescription(modelDescription);
137  icao.setWtc(wtc);
138  icao.setCodeFlags(military, legacy, realWorld);
139  icao.setRank(rank);
140  icao.setCombinedType(combined);
141  icao.setFamily(family);
142  icao.setIataCode(iata);
143  return icao;
144  }

207  void CAircraftIcaoForm::droppedCode(const BlackMisc::CVariant &variantDropped)
208  {
209  CAircraftIcaoCode icao;
210  if (variantDropped.canConvert<CAircraftIcaoCode>())
211  {
212  icao = variantDropped.value<CAircraftIcaoCode>();
213  }
214  else if (variantDropped.canConvert<CAircraftIcaoCodeList>())
215  {
216  const CAircraftIcaoCodeList icaoList(variantDropped.value<CAircraftIcaoCodeList>());
217  if (icaoList.isEmpty()) { return; }
218  icao = icaoList.front();
219  }
220  else
221  {
222  return;
223  }
224  this->setValue(icao);
225  }
