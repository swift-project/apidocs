29  {
30  const unsigned int flags = dbus_watch_get_flags(watch);
31  short monitoredEvents = EV_PERSIST;
32 
33  if (flags & DBUS_WATCH_READABLE) { monitoredEvents |= EV_READ; }
34  if (flags & DBUS_WATCH_WRITABLE) { monitoredEvents |= EV_WRITE; }
35 
36  const int fd = dbus_watch_get_unix_fd(watch);
37  m_event.reset(event_new(m_base, fd, monitoredEvents, callback, this));
38  event_add(m_event.get(), nullptr);
39  }

49  static void callback(evutil_socket_t fd, short event, void *data)
50  {
51  (void)fd; // Not really unused, but GCC/Clang still complain about it.
52  auto *watchHandler = static_cast<WatchHandler *>(data);
53 
54  unsigned int flags = 0;
55  if (event & EV_READ) { flags |= DBUS_WATCH_READABLE; }
56  if (event & EV_WRITE) { flags |= DBUS_WATCH_WRITABLE; }
57  dbus_watch_handle(watchHandler->m_watch, flags);

179  dbus_bool_t CDBusDispatcher::dbusAddWatch(DBusWatch *watch)

189  {

210  void CDBusDispatcher::dbusRemoveTimeout(DBusTimeout *timeout)
211  {

214  };
215
