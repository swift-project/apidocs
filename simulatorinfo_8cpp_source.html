214  {

233  CSimulatorInfo::Simulator CSimulatorInfo::identifierToSimulator(const QString &identifier)
234  {
235  const QString i(identifier.toLower().trimmed().remove(' ').remove('-'));
236  if (i.isEmpty()) { return None; }
237 
238  Simulator s = None;
239  if (i.contains("fsx") || i.contains("fs10")) { s |= FSX; }
240  if (i.contains("fs9") || i.contains("2004")) { s |= FS9; }
241  if (i.contains("plane") || i.contains("xp")) { s |= XPLANE; }
242  if (i.contains("gear") || stringCompare(QStringLiteral("fg"), identifier, Qt::CaseInsensitive)) { s |= FG; }
243  if (i.contains("3d") || i.contains("prepar") || i.contains("martin") || i.contains("lm") || i.contains("lock"))
244  {
245  s |= P3D;
246  }
247  if (i.contains("msfs")) { s |= MSFS; }
248  return s;
249  }
