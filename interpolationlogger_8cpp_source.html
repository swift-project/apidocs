45  QList<PartsLog> parts;
46  {

61  });

65  QStringList CInterpolationLogger::getLatestLogFiles()
66  {
67  QStringList files({ "", "" });

69  QDir logs(logDir);

78  if (!parts.isEmpty())

90  CStatusMessageList CInterpolationLogger::writeLogFiles(const QList<SituationLog> &interpolation, const QList<PartsLog> &parts)
91  {
92  if (parts.isEmpty() && interpolation.isEmpty()) { return CStatusMessage(static_cast<CInterpolationLogger *>(nullptr)).warning(u"No data for log"); }
93  static const QString html = QStringLiteral("Entries: %1\n\n%2");
94  const QString htmlTemplate = CFileUtils::readFileToString(CSwiftDirectories::htmlTemplateFilePath());
95 
96  CStatusMessageList msgs;
97  const QString ts = QDateTime::currentDateTimeUtc().toString("yyyyMMddhhmmss");
98 
99  const QString htmlInterpolation = CInterpolationLogger::getHtmlInterpolationLog(interpolation);
100  if (!htmlInterpolation.isEmpty())
101  {
102  QString file = filePatternInterpolationLog();
103  file.remove('*');
104  const QString fn = CFileUtils::appendFilePaths(CSwiftDirectories::logDirectory(), QStringLiteral("%1 %2").arg(ts, file));
105  const bool s = CFileUtils::writeStringToFile(htmlTemplate.arg(html.arg(interpolation.size()).arg(htmlInterpolation)), fn);
106  msgs.push_back(CInterpolationLogger::logStatusFileWriting(s, fn));
107  }
108 
109  const QString htmlParts = CInterpolationLogger::getHtmlPartsLog(parts);
110  if (!htmlParts.isEmpty())
111  {
112  QString file = filePatternPartsLog();
113  file.remove('*');
114  const QString fn = CFileUtils::appendFilePaths(CSwiftDirectories::logDirectory(), QStringLiteral("%1 %2").arg(ts, file));
115  const bool s = CFileUtils::writeStringToFile(htmlTemplate.arg(html.arg(parts.size()).arg(htmlParts)), fn);
116  msgs.push_back(CInterpolationLogger::logStatusFileWriting(s, fn));
117  }
118 
119  QString kml = CKmlUtils::wrapAsKmlDocument(CInterpolationLogger::getKmlChangedSituations(interpolation));
120  if (!kml.isEmpty())
121  {
122  const QString fn = CFileUtils::appendFilePaths(CSwiftDirectories::logDirectory(), QStringLiteral("%1_changedSituations.kml").arg(ts));
123  const bool s = CFileUtils::writeStringToFile(kml, fn);
124  msgs.push_back(CInterpolationLogger::logStatusFileWriting(s, fn));
125  }
126 
127  kml = CKmlUtils::wrapAsKmlDocument(CInterpolationLogger::getKmlInterpolatedSituations(interpolation));
128  if (!kml.isEmpty())
129  {
130  const QString fn = CFileUtils::appendFilePaths(CSwiftDirectories::logDirectory(), QStringLiteral("%1_interpolatedSituations.kml").arg(ts));
131  const bool s = CFileUtils::writeStringToFile(kml, fn);
132  msgs.push_back(CInterpolationLogger::logStatusFileWriting(s, fn));
133  }
134 
135  kml = CKmlUtils::wrapAsKmlDocument(CInterpolationLogger::getKmlElevations(interpolation));
136  if (!kml.isEmpty())
137  {
138  const QString fn = CFileUtils::appendFilePaths(CSwiftDirectories::logDirectory(), QStringLiteral("%1_elevations.kml").arg(ts));
139  const bool s = CFileUtils::writeStringToFile(kml, fn);
140  msgs.push_back(CInterpolationLogger::logStatusFileWriting(s, fn));
141  }
142 
143  return msgs;
144  }

380  QString CInterpolationLogger::getKmlChangedSituations(const QList<SituationLog> &logs)
381  {
382  if (logs.isEmpty()) { return {}; }
383 
384  QString kml;
385  qint64 newPosTs = -1;
386  int n = 1;
387  const CKmlUtils::KMLSettings s(true, true);
388 
389  for (const SituationLog &log : logs)
390  {
391  const CAircraftSituation situationNew = log.newestInterpolationSituation();
392  const bool changedNewPosition = (newPosTs != situationNew.getMSecsSinceEpoch());
393  const bool recalc = log.interpolantRecalc;
394  if (!changedNewPosition && !recalc) { continue; }
395  newPosTs = situationNew.getMSecsSinceEpoch();
396  kml += CKmlUtils::asPlacemark(
397  QStringLiteral("%1: %2 new pos: %3 recalc: %4").arg(n++).arg(situationNew.getFormattedUtcTimestampHmsz(), boolToYesNo(changedNewPosition), boolToYesNo(recalc)),
398  situationNew.toQString(true),
399  situationNew, s) %
400  u"\n";
401  }
402 
403  return kml;
404  }
