67  void CSettingsNetworkServersComponent::alterTrafficServer()
68  {
69  const QObject *sender = QObject::sender();
70  const CServer server(ui->form_Server->getServer());
71 
72  CServerList serverList(m_trafficNetworkServers.getThreadLocal());
73  CStatusMessage msg;
74  if (sender == ui->pb_RemoveServer)
75  {
76  // lenient name removal
77  serverList.removeByName(server.getName());
78  }
79  else if (sender == ui->pb_SaveServer)
80  {
81  CStatusMessageList msgs = server.validate();
82  if (!msgs.isEmpty()) { msgs.addCategories(this); }
83 
84  // error?
85  if (msgs.isFailure())
86  {
87  CLogMessage::preformatted(msgs);
88  return;
89  }
90 
91  serverList.replaceOrAdd(&CServer::getName, server.getName(), server);
92  }
93  else
94  {
95  qFatal("Wrong sender");
96  return;
97  }
98 
99  msg = m_trafficNetworkServers.setAndSave(serverList);
100  this->reloadSettings(); // call manually as local object
101 
102  if (!msg.isEmpty())
103  {
104  msg.addCategories(this);
105  CLogMessage::preformatted(msg);
106  }
107  }
