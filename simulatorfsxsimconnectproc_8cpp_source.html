30  void CALLBACK CSimulatorFsxCommon::SimConnectProc(SIMCONNECT_RECV *pData, DWORD cbData, void *pContext)
31  {
32  // IMPORTANT:
33  // all tasks called in this function (ie all called functions) must perform fast or shall be called asynchronously
34 
35  const qint64 procTimeStart = QDateTime::currentMSecsSinceEpoch();
36  CSimulatorFsxCommon *simulatorFsxP3D = static_cast<CSimulatorFsxCommon *>(pContext);
37  const SIMCONNECT_RECV_ID recvId = static_cast<SIMCONNECT_RECV_ID>(pData->dwID);
38  simulatorFsxP3D->m_dispatchReceiveIdLast = recvId;
39  simulatorFsxP3D->m_dispatchProcCount++;
40  switch (recvId)
41  {
42  case SIMCONNECT_RECV_ID_OPEN:
43  {
44  SIMCONNECT_RECV_OPEN *event = static_cast<SIMCONNECT_RECV_OPEN *>(pData);
45  const QString simConnectVersion = QStringLiteral("%1.%2.%3.%4").arg(event->dwSimConnectVersionMajor).arg(event->dwSimConnectVersionMinor).arg(event->dwSimConnectBuildMajor).arg(event->dwSimConnectBuildMinor);
46  const QString version = QStringLiteral("%1.%2.%3.%4").arg(event->dwApplicationVersionMajor).arg(event->dwApplicationVersionMinor).arg(event->dwApplicationBuildMajor).arg(event->dwApplicationBuildMinor);
47  const QString name = CSimulatorFsxCommon::fsxCharToQString(event->szApplicationName);
48  const QString details = QStringLiteral("Name: '%1' Version: %2 SimConnect: %3").arg(name, version, simConnectVersion);
49  simulatorFsxP3D->setSimulatorDetails(name, details, version);
50  simulatorFsxP3D->m_simConnectVersion = simConnectVersion;
51  CLogMessage(simulatorFsxP3D).info(u"Connected to %1: '%2'") << simulatorFsxP3D->getSimulatorPluginInfo().getIdentifier() << details;
52  simulatorFsxP3D->setSimConnected();
53  break; // SIMCONNECT_RECV_ID_OPEN
54  }
55  case SIMCONNECT_RECV_ID_EXCEPTION:
56  {
57  if (!simulatorFsxP3D->stillDisplayReceiveExceptions()) { break; }
58  simulatorFsxP3D->triggerAutoTraceSendId();
59 
60  SIMCONNECT_RECV_EXCEPTION *exception = static_cast<SIMCONNECT_RECV_EXCEPTION *>(pData);
61  const DWORD exceptionId = exception->dwException;
62  const DWORD sendId = exception->dwSendID;
63  const DWORD index = exception->dwIndex; // index of parameter that was source of error, 4294967295/0xFFFFFFFF means unknown, 0 means also UNKNOWN INDEX
64  const DWORD data = cbData;
65  const TraceFsxSendId trace = simulatorFsxP3D->getSendIdTrace(sendId);
66  bool logGenericExceptionInfo = true;
67 
68  switch (exceptionId)
69  {
70  case SIMCONNECT_EXCEPTION_OPERATION_INVALID_FOR_OBJECT_TYPE: break;
71  case SIMCONNECT_EXCEPTION_UNRECOGNIZED_ID: break; // Specifies that the client event, request ID, data definition ID, or object ID was not recognized
72  case SIMCONNECT_EXCEPTION_CREATE_OBJECT_FAILED:
73  {
74  if (trace.isValid())
75  {
76  // it can happen the object is not yet existing
77  CSimConnectObject simObject = simulatorFsxP3D->getSimObjectForTrace(trace);
78  if (simObject.isInvalid()) { simObject = trace.simObject; } // take the one in the trace
79  if (simObject.isValid())
80  {
81  if (simObject.isAircraft())
82  {
83  simulatorFsxP3D->addingAircraftFailed(simObject);
84  logGenericExceptionInfo = false;
85  }
86  else
87  {
88  const bool removed = simulatorFsxP3D->m_simConnectObjects.remove(simObject.getCallsign());
89  Q_UNUSED(removed);
90  CLogMessage(simulatorFsxP3D).warning(u"Adding probe failed: %1 %2") << simObject.getCallsign().asString() << simObject.getAircraftModelString();
91  if (simulatorFsxP3D->isUsingFsxTerrainProbe())
92  {
93  CLogMessage(simulatorFsxP3D).warning(u"Disabling terrain probe");
94  simulatorFsxP3D->setUsingFsxTerrainProbe(false);
95  }
96  logGenericExceptionInfo = false;
97  } // aircraft
98  } // valid
99  } // trace
100  } // SIMCONNECT_EXCEPTION_CREATE_OBJECT_FAILED:
101  break;
102  default:
103  break;
104  } // switch exception id
105 
106  // generic exception warning
107  if (logGenericExceptionInfo)
108  {
109  QString ex = QString::asprintf("Exception=%lu | SendID=%lu | Index=%lu | cbData=%lu", exceptionId, sendId, index, data);
110  const QString exceptionString(CSimConnectUtilities::simConnectExceptionToString(static_cast<DWORD>(exception->dwException)));
111  const QString sendIdDetails = simulatorFsxP3D->getSendIdTraceDetails(sendId);
112  CLogMessage(simulatorFsxP3D).warning(u"Caught simConnect exception: '%1' '%2' | send details: '%3'")
113  << exceptionString << ex
114  << (sendIdDetails.isEmpty() ? "N/A" : sendIdDetails);
115  }
116  break; // SIMCONNECT_RECV_ID_EXCEPTION
117  }
118  case SIMCONNECT_RECV_ID_QUIT:
119  {
120  simulatorFsxP3D->onSimExit();
121  break;
122  }
123  case SIMCONNECT_RECV_ID_EVENT:
124  {
125  const SIMCONNECT_RECV_EVENT *event = static_cast<SIMCONNECT_RECV_EVENT *>(pData);
126  switch (event->uEventID)
127  {
128  case SystemEventSimStatus:
129  {
130  const bool running = event->dwData ? true : false;
131  if (running)
132  {
133  simulatorFsxP3D->onSimRunning();
134  }
135  else
136  {
137  simulatorFsxP3D->onSimStopped();
138  }
139  break;
140  }
141  case SystemEventPause:
142  {
143  const bool paused = event->dwData ? true : false;
144  if (simulatorFsxP3D->m_simPaused != paused)
145  {
146  simulatorFsxP3D->m_simPaused = paused;
147  simulatorFsxP3D->emitSimulatorCombinedStatus();
148  }
149  break;
150  }
151  default:
152  break;
153  }
154  break; // SIMCONNECT_RECV_ID_EVENT
155  }
156  case SIMCONNECT_RECV_ID_EVENT_OBJECT_ADDREMOVE:
157  {
158  const SIMCONNECT_RECV_EVENT_OBJECT_ADDREMOVE *event = static_cast<SIMCONNECT_RECV_EVENT_OBJECT_ADDREMOVE *>(pData);
159  const DWORD objectId = event->dwData;
160  const SIMCONNECT_SIMOBJECT_TYPE objectType = event->eObjType;
161  if (objectType != SIMCONNECT_SIMOBJECT_TYPE_AIRCRAFT && objectType != SIMCONNECT_SIMOBJECT_TYPE_HELICOPTER)
162  {
163  break; // SIMCONNECT_RECV_ID_EVENT_OBJECT_ADDREMOVE
164  }
165 
166  // such an object is not necessarily one of ours
167  // for instance, I always see object "5" when I start the simulator
168  if (simulatorFsxP3D->getSimConnectObjects().isKnownSimObjectId(objectId))
169  {
170  switch (event->uEventID)
171  {
172  case SystemEventObjectRemoved:
173  simulatorFsxP3D->simulatorReportedObjectRemoved(objectId);
174  break;
175  case SystemEventObjectAdded:
176  // added in SIMCONNECT_RECV_ID_ASSIGNED_OBJECT_ID
177  // this event here is normally received before SIMCONNECT_RECV_ID_ASSIGNED_OBJECT_ID
178  break;
179  default:
180  break;
181  }
182  }
183  break; // SIMCONNECT_RECV_ID_EVENT_OBJECT_ADDREMOVE
184  }
185  case SIMCONNECT_RECV_ID_EVENT_FRAME:
186  {
187  const SIMCONNECT_RECV_EVENT_FRAME *event = static_cast<SIMCONNECT_RECV_EVENT_FRAME *>(pData);
188  switch (event->uEventID)
189  {
190  case SystemEventFrame:
191  // doing interpolation
192  simulatorFsxP3D->onSimFrame();
193  break;
194  default:
195  break;
196  }
197  break; // SIMCONNECT_RECV_ID_EVENT_FRAME
198  }
199  case SIMCONNECT_RECV_ID_ASSIGNED_OBJECT_ID:
200  {
201  const SIMCONNECT_RECV_ASSIGNED_OBJECT_ID *event = static_cast<SIMCONNECT_RECV_ASSIGNED_OBJECT_ID *>(pData);
202  const DWORD requestId = event->dwRequestID;
203  const DWORD objectId = event->dwObjectID;
204  simulatorFsxP3D->m_dispatchRequestIdLast = requestId;
205 
206  if (CSimulatorFsxCommon::isRequestForSimConnectObject(requestId))
207  {
208  bool success = simulatorFsxP3D->setSimConnectObjectId(requestId, objectId);
209  if (!success) { break; } // not an request ID of ours
210  success = simulatorFsxP3D->simulatorReportedObjectAdded(objectId); // adding failed (no IDs), trigger follow up actions
211  if (!success)
212  {
213  // getting here would mean object was removed in the meantime
214  // otherwise we will detect it in verification
215  const CSimConnectObject simObject = simulatorFsxP3D->getSimObjectForObjectId(objectId);
216  const CSimulatedAircraft remoteAircraft(simObject.getAircraft());
217  const CStatusMessage msg = CStatusMessage(simulatorFsxP3D).error(u"Cannot add object %1, cs: '%2' model: '%3'") << objectId << remoteAircraft.getCallsignAsString() << remoteAircraft.getModelString();
218  CLogMessage::preformatted(msg);
219  emit simulatorFsxP3D->physicallyAddingRemoteModelFailed(remoteAircraft, false, false, msg);
220  }
221  }
222  break; // SIMCONNECT_RECV_ID_ASSIGNED_OBJECT_ID
223  }
224  case SIMCONNECT_RECV_ID_SIMOBJECT_DATA_BYTYPE:
225  {
226  // SIMCONNECT_RECV_SIMOBJECT_DATA_BYTYPE *pObjData = (SIMCONNECT_RECV_SIMOBJECT_DATA_BYTYPE *)pData;
227  break;
228  }
229  case SIMCONNECT_RECV_ID_SIMOBJECT_DATA:
230  {
231  const SIMCONNECT_RECV_SIMOBJECT_DATA *pObjData = static_cast<SIMCONNECT_RECV_SIMOBJECT_DATA *>(pData);
232  const DWORD requestId = pObjData->dwRequestID;
233  simulatorFsxP3D->m_dispatchRequestIdLast = requestId;
234 
235  switch (requestId)
236  {
237  case CSimConnectDefinitions::RequestOwnAircraft:
238  {
239  static_assert(sizeof(DataDefinitionOwnAircraft) == 45 * sizeof(double), "DataDefinitionOwnAircraft has an incorrect size.");
240  const DataDefinitionOwnAircraft *ownAircaft = reinterpret_cast<const DataDefinitionOwnAircraft *>(&pObjData->dwData);
241  simulatorFsxP3D->updateOwnAircraftFromSimulator(*ownAircaft);
242  break;
243  }
244  case CSimConnectDefinitions::RequestOwnAircraftTitle:
245  {
246  const DataDefinitionOwnAircraftModel *dataDefinitionModel = reinterpret_cast<const DataDefinitionOwnAircraftModel *>(&pObjData->dwData);
247  const CAircraftModel model(dataDefinitionModel->title, CAircraftModel::TypeOwnSimulatorModel);
248  simulatorFsxP3D->reverseLookupAndUpdateOwnAircraftModel(model);
249  break;
250  }
251  case CSimConnectDefinitions::RequestSimEnvironment:
252  {
253  const DataDefinitionSimEnvironment *simEnv = reinterpret_cast<const DataDefinitionSimEnvironment *>(&pObjData->dwData);
254  simulatorFsxP3D->synchronizeTime(simEnv);
255  break;
256  }
257  default:
258  {
259  const DWORD objectId = pObjData->dwObjectID;
260  if (CSimulatorFsxCommon::isRequestForSimObjAircraft(requestId))
261  {
262  const CSimConnectObject simObject = simulatorFsxP3D->getSimObjectForObjectId(objectId);
263  if (!simObject.hasValidRequestAndObjectId()) { break; }
264  const CSimConnectDefinitions::SimObjectRequest subRequest = CSimulatorFsxCommon::requestToSimObjectRequest(requestId);
265 
266  if (subRequest == CSimConnectDefinitions::SimObjectPositionData)
267  {
268  static_assert(sizeof(DataDefinitionPosData) == 5 * sizeof(double), "DataDefinitionPosData has an incorrect size.");
269  const DataDefinitionPosData *remoteAircraftSimData = reinterpret_cast<const DataDefinitionPosData *>(&pObjData->dwData);
270  // extra check, but ids should be the same
271  if (objectId == simObject.getObjectId())
272  {
273  simulatorFsxP3D->triggerUpdateRemoteAircraftFromSimulator(simObject, *remoteAircraftSimData);
274  }
275  } // position
276  else if (subRequest == CSimConnectDefinitions::SimObjectModel)
277  {
278  static_assert(sizeof(DataDefinitionRemoteAircraftModel) == sizeof(double) + 168 + 256, "DataDefinitionRemoteAircraftModel has an incorrect size.");
279  const DataDefinitionRemoteAircraftModel *remoteAircraftModel = reinterpret_cast<const DataDefinitionRemoteAircraftModel *>(&pObjData->dwData);
280  // extra check, but ids should be the same
281  if (objectId == simObject.getObjectId())
282  {
283  simulatorFsxP3D->triggerUpdateRemoteAircraftFromSimulator(simObject, *remoteAircraftModel);
284  }
285  } // model
286  else if (subRequest == CSimConnectDefinitions::SimObjectLights)
287  {
288  static_assert(sizeof(DataDefinitionRemoteAircraftLights) == 8 * sizeof(double), "DataDefinitionRemoteAircraftLights has an incorrect size.");
289  const DataDefinitionRemoteAircraftLights *remoteAircraftLights = reinterpret_cast<const DataDefinitionRemoteAircraftLights *>(&pObjData->dwData);
290  // extra check, but ids should be the same
291  if (objectId == simObject.getObjectId())
292  {
293  const CCallsign callsign(simObject.getCallsign());
294  const CAircraftLights lights(remoteAircraftLights->toLights()); // as in simulator
295  simulatorFsxP3D->setCurrentLights(callsign, lights);
296  if (simObject.getLightsAsSent().isNull())
297  {
298  // allows to compare for toggle
299  simulatorFsxP3D->setLightsAsSent(callsign, lights);
300  }
301  }
302  break;
303  } // lights
304  else
305  {
306  if (CBuildConfig::isLocalDeveloperDebugBuild())
307  {
308  CLogMessage(simulatorFsxP3D).error(u"Unknown subrequest (aircraft): '%1' %2")
309  << CSimConnectDefinitions::simObjectRequestToString(subRequest)
310  << simObject.toQString();
311  }
312  }
313  }
314  else if (CSimulatorFsxCommon::isRequestForSimObjTerrainProbe(requestId))
315  {
316  const CSimConnectObject probeObj = simulatorFsxP3D->getSimObjectForObjectId(objectId);
317  if (!probeObj.hasValidRequestAndObjectId()) { break; }
318  Q_ASSERT_X(probeObj.isTerrainProbe(), Q_FUNC_INFO, "No probe");
319  const CSimConnectDefinitions::SimObjectRequest subRequest = CSimulatorFsxCommon::requestToSimObjectRequest(requestId);
320 
321  if (subRequest == CSimConnectDefinitions::SimObjectPositionData)
322  {
323  static_assert(sizeof(DataDefinitionPosData) == 5 * sizeof(double), "DataDefinitionRemoteAircraftSimData has an incorrect size.");
324  const DataDefinitionPosData *probeSimData = reinterpret_cast<const DataDefinitionPosData *>(&pObjData->dwData);
325  // extra check, but ids should be the same
326  if (objectId == probeObj.getObjectId())
327  {
328  const CCallsign cs = simulatorFsxP3D->m_pendingProbeRequests.value(requestId);
329  if (cs.isEmpty()) { break; }
330  simulatorFsxP3D->updateProbeFromSimulator(cs, *probeSimData);
331  }
332  }
333  else
334  {
335  if (CBuildConfig::isLocalDeveloperDebugBuild())
336  {
337  CLogMessage(simulatorFsxP3D).error(u"Unknown subrequest (probe): '%1' %2")
338  << CSimConnectDefinitions::simObjectRequestToString(subRequest)
339  << probeObj.toQString();
340  }
341  }
342  } // probe
343  }
344  break; // default (SIMCONNECT_RECV_ID_SIMOBJECT_DATA)
345  }
346  break; // SIMCONNECT_RECV_ID_SIMOBJECT_DATA
347  }
348  case SIMCONNECT_RECV_ID_AIRPORT_LIST:
349  {
350  const SIMCONNECT_RECV_AIRPORT_LIST *pAirportList = static_cast<SIMCONNECT_RECV_AIRPORT_LIST *>(pData);
351  CAirportList simAirports;
352  for (unsigned i = 0; i < pAirportList->dwArraySize; ++i)
353  {
354  const SIMCONNECT_DATA_FACILITY_AIRPORT *pFacilityAirport = pAirportList->rgData + i;
355  if (!pFacilityAirport) { break; }
356  const QString icao(pFacilityAirport->Icao);
357  if (icao.isEmpty()) { continue; } // airfield without ICAO code
358  if (!CAirportIcaoCode::isValidIcaoDesignator(icao, true)) { continue; } // tiny airfields/strips in simulator
359  if (CAirportIcaoCode::containsNumbers(icao)) { continue; } // tiny airfields/strips in simulator
360  const CCoordinateGeodetic pos(pFacilityAirport->Latitude, pFacilityAirport->Longitude, pFacilityAirport->Altitude);
361  const CAirport airport(CAirportIcaoCode(icao), pos);
362  simAirports.push_back(airport);
363  }
364  if (!simAirports.isEmpty())
365  {
366  simulatorFsxP3D->triggerUpdateAirports(simAirports); // real "work" outside SimConnectProc
367  }
368  break; // SIMCONNECT_RECV_ID_AIRPORT_LIST
369  }
370  case SIMCONNECT_RECV_ID_CLIENT_DATA:
371  {
372  if (!simulatorFsxP3D->m_useSbOffsets) { break; }
373  simulatorFsxP3D->m_sbDataReceived++;
374  const SIMCONNECT_RECV_CLIENT_DATA *clientData = static_cast<SIMCONNECT_RECV_CLIENT_DATA *>(pData);
375  if (clientData->dwRequestID == CSimConnectDefinitions::RequestSbData)
376  {
378  DataDefinitionClientAreaSb sbData;
379  std::memcpy(&sbData.data, &clientData->dwData, 128);
380  simulatorFsxP3D->updateOwnAircraftFromSimulator(sbData);
381  }
382  break; // SIMCONNECT_RECV_ID_CLIENT_DATA
383  }
384  case SIMCONNECT_RECV_ID_EVENT_FILENAME:
385  {
386  const SIMCONNECT_RECV_EVENT_FILENAME *event = static_cast<SIMCONNECT_RECV_EVENT_FILENAME *>(pData);
387  switch (event->uEventID)
388  {
389  case SystemEventFlightLoaded: break;
390  default: break;
391  }
392  break; // SIMCONNECT_RECV_ID_EVENT_FILENAME
393  }
394  default:
395  simulatorFsxP3D->m_dispatchProcEmptyCount++;
396  break;
397  } // main switch
398 
399  // performance stats
400  const qint64 procTimeEnd = QDateTime::currentMSecsSinceEpoch();
401  simulatorFsxP3D->m_dispatchProcTimeMs = procTimeEnd - procTimeStart;
402  if (simulatorFsxP3D->m_dispatchProcTimeMs > simulatorFsxP3D->m_dispatchProcMaxTimeMs)
403  {
404  simulatorFsxP3D->m_dispatchProcMaxTimeMs = simulatorFsxP3D->m_dispatchProcTimeMs;
405  }
406  } // method
