217  void CMappingComponent::onChangedSimulatedAircraftInView(const CVariant &object, const CPropertyIndex &index)
218  {
219  if (!index.contains(CSimulatedAircraft::IndexEnabled)) { return; } // we only deal with enabled/disabled here
220  const CSimulatedAircraft sa = object.to<CSimulatedAircraft>(); // changed in GUI
221  const CSimulatedAircraft saFromBackend = sGui->getIContextNetwork()->getAircraftInRangeForCallsign(sa.getCallsign());
222  if (!saFromBackend.hasValidCallsign()) { return; } // obviously deleted
223  const bool nowEnabled = sa.isEnabled();
224  if (saFromBackend.isEnabled() == nowEnabled) { return; } // already same value
225  CLogMessage(this).info(u"Request to %1 aircraft '%2'") << (nowEnabled ? "enable" : "disable") << saFromBackend.getCallsign().toQString();
226  sGui->getIContextNetwork()->updateAircraftEnabled(saFromBackend.getCallsign(), nowEnabled);
227  }

317  void CMappingComponent::onSaveAircraft()
318  {
319  if (!sGui || !sGui->getIContextSimulator() || !sGui->getIContextSimulator()->isSimulatorSimulating()) { return; }
320  const CCallsign callsign(this->validateRenderedCallsign());
321  if (callsign.isEmpty()) { return; }
322  const QString modelString = ui->completer_ModelStrings->getModelString();
323  if (modelString.isEmpty())
324  {
325  this->showOverlayHTMLMessage(CStatusMessage(this).validationError(u"Missing model for mapping"), OverlayMessageMs);
326  return;
327  }
328 
329  // model in current view (data already in UI)
330  bool hasModel = ui->tvp_AircraftModels->container().containsModelString(modelString);
331  if (!hasModel && sGui && sGui->getIContextSimulator())
332  {
333  hasModel = sGui->getIContextSimulator()->isKnownModelInSet(modelString);
334  }
335 
336  if (!hasModel)
337  {
338  this->showOverlayMessage(CStatusMessage(this).validationError(u"Invalid model for mapping, reloading model set"), OverlayMessageMs);
339  if (ui->tvp_AircraftModels->isEmpty())
340  {
341  this->onModelsUpdateRequested();
342  }
343  return;
344  }
345 
346  const CSimulatedAircraft aircraftFromBackend = sGui->getIContextNetwork()->getAircraftInRangeForCallsign(callsign);
347  const bool enabled = ui->cb_AircraftEnabled->isChecked();
348  bool changed = false;
349 
350  // changed model?
351  if (aircraftFromBackend.getModelString() != modelString)
352  {
353  const CAircraftModelList models = sGui->getIContextSimulator()->getModelSetModelsStartingWith(modelString);
354  if (models.isEmpty())
355  {
356  const CStatusMessage msg = CStatusMessage(this).validationError(u"No model for title: '%1'") << modelString;
357  this->showOverlayMessage(msg, OverlayMessageMs);
358  return;
359  }
360 
361  CAircraftModel model(models.front());
362  if (models.size() > 1)
363  {
364  if (models.containsModelString(modelString))
365  {
366  model = models.findByModelString(modelString).front(); // exact match
367  }
368  else
369  {
370  const CStatusMessage msg = CStatusMessage(this).validationInfo(u"Ambigious title: '%1', using '%2'") << modelString << model.getModelString();
371  this->showOverlayMessage(msg, OverlayMessageMs);
372  }
373  }
374  model.setModelType(CAircraftModel::TypeManuallySet);
375  CLogMessage(this).info(u"Requesting changes for '%1'") << callsign.asString();
376 
377  // enable in any case
378  sGui->getIContextNetwork()->updateAircraftEnabled(aircraftFromBackend.getCallsign(), true);
379  changed = sGui->getIContextNetwork()->updateAircraftModel(aircraftFromBackend.getCallsign(), model, identifier());
380  }
381  if (aircraftFromBackend.isEnabled() != enabled)
382  {
383  changed = sGui->getIContextNetwork()->updateAircraftEnabled(aircraftFromBackend.getCallsign(), enabled);
384  }
385 
386  if (!changed)
387  {
388  this->showOverlayHTMLMessage(CLogMessage(this).info(u"Model mapping, nothing to change"), OverlayMessageMs);
389  }
390  }
