38  void CDbAircraftCategoryComponent::onCategoryRead(CEntityFlags::Entity entity, CEntityFlags::ReadState readState, int count)
39  {
40  Q_UNUSED(count);
41  if (!sGui || sGui->isShuttingDown() || !sGui->getWebDataServices()) { return; }
42  if (!entity.testFlag(CEntityFlags::AircraftCategoryEntity)) { return; }
43 
44  if (CEntityFlags::isFinishedReadState(readState))
45  {
46  this->showOverlayHTMLMessage(QStringLiteral("Updating %1").arg(CEntityFlags::entitiesToString(entity)), 2000);
47  const CAircraftCategoryList categories = sGui->getWebDataServices()->getAircraftCategories();
48  ui->tvp_AircraftCategoryView->updateContainerMaybeAsync(categories);
49  ui->tvp_AircraftCategoryTree->updateContainer(categories);
50  }
51  else
52  {
53  this->showOverlayHTMLMessage(u"Current state: " % CEntityFlags::entitiesToString(entity) % u" " % CEntityFlags::stateToString(readState), 10000);
54  }
55  }
