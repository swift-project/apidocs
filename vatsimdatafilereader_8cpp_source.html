173  void CVatsimDataFileReader::read()
174  {
175  this->threadAssertCheck();
176  if (!this->doWorkCheck()) { return; }
177  if (!this->isInternetAccessible("No network/internet access, cannot read VATSIM data file")) { return; }
178  if (this->isNotVATSIMEcosystem()) { return; }
179 
180  // round robin for load balancing
181  // remark: Don't use QThread to run network operations in the background
182  // see http://qt-project.org/doc/qt-4.7/qnetworkaccessmanager.html
183  Q_ASSERT_X(sApp, Q_FUNC_INFO, "Missing application");
184  CFailoverUrlList urls(sApp->getVatsimDataFileUrls());
185  const QUrl url(urls.obtainNextWorkingUrl(true));
186  if (url.isEmpty()) { return; }
187  this->getFromNetworkAndLog(url, { this, &CVatsimDataFileReader::parseVatsimFile });
188  }

190  void CVatsimDataFileReader::parseVatsimFile(QNetworkReply *nwReplyPtr)
191  {
192  // wrap pointer, make sure any exit cleans up reply
193  // required to use delete later as object is created in a different thread
194  QScopedPointer<QNetworkReply, QScopedPointerDeleteLater> nwReply(nwReplyPtr);
195  this->threadAssertCheck();
196  if (this->isNotVATSIMEcosystem()) { return; }
197 
198  // Worker thread, make sure to write only synced here!
199  if (!this->doWorkCheck())
200  {
201  CLogMessage(this).info(u"Terminated VATSIM file parsing process");
202  return; // stop, terminate straight away, ending thread
203  }
204 
205  this->logNetworkReplyReceived(nwReplyPtr);
206  QStringList illegalEquipmentCodes;
207  const QUrl url = nwReply->url();
208  const QString urlString = url.toString();
209 
210  if (nwReply->error() == QNetworkReply::NoError)
211  {
212  const QString dataFileData = nwReply->readAll();
213  nwReply->close(); // close asap
214 
215  if (dataFileData.isEmpty()) { return; }
216  if (!this->didContentChange(dataFileData)) // Quick check by hash
217  {
218  CLogMessage(this).info(u"VATSIM file '%1' has same content, skipped") << urlString;
219  return;
220  }
221  auto jsonDoc = QJsonDocument::fromJson(dataFileData.toUtf8());
222  if (jsonDoc.isEmpty()) { return; }
223 
224  // build on local vars for thread safety
225  CServerList fsdServers;
226  CAtcStationList atcStations;
227  CSimulatedAircraftList aircraft;
228  QMap<CCallsign, CFlightPlanRemarks> flightPlanRemarksMap;
229  auto updateTimestampFromFile = QDateTime::fromString(jsonDoc["general"]["update_timestamp"].toString(), Qt::ISODateWithMs);
230 
231  const bool alreadyRead = (updateTimestampFromFile == this->getUpdateTimestamp());
232  if (alreadyRead)
233  {
234  CLogMessage(this).info(u"VATSIM file has same timestamp, skipped");
235  return;
236  }
237 
238  for (QJsonValueRef pilot : jsonDoc["pilots"].toArray())
239  {
240  if (!this->doWorkCheck())
241  {
242  CLogMessage(this).info(u"Terminated VATSIM file parsing process");
243  return;
244  }
245  aircraft.push_back(parsePilot(pilot.toObject(), illegalEquipmentCodes));
246  flightPlanRemarksMap.insert(aircraft.back().getCallsign(), parseFlightPlanRemarks(pilot.toObject()));
247  }
248  for (QJsonValueRef controller : jsonDoc["controllers"].toArray())
249  {
250  if (!this->doWorkCheck())
251  {
252  CLogMessage(this).info(u"Terminated VATSIM file parsing process");
253  return;
254  }
255  atcStations.push_back(parseController(controller.toObject()));
256  }
257  for (QJsonValueRef atis : jsonDoc["atis"].toArray())
258  {
259  if (!this->doWorkCheck())
260  {
261  CLogMessage(this).info(u"Terminated VATSIM file parsing process");
262  return;
263  }
264  atcStations.push_back(parseController(atis.toObject()));
265  }
266 
267  // Setup for VATSIM servers and sorting for comparison
268  fsdServers.sortBy(&CServer::getName, &CServer::getDescription);
269 
270  // this part needs to be synchronized
271  {
272  QWriteLocker wl(&m_lock);
273  this->setUpdateTimestamp(updateTimestampFromFile);
274  m_aircraft = aircraft;
275  m_atcStations = atcStations;
276  m_flightPlanRemarks = flightPlanRemarksMap;
277  }
278 
279  // warnings, if required
280  if (!illegalEquipmentCodes.isEmpty())
281  {
282  CVatsimDataFileReader::logInconsistentData(
283  CStatusMessage(this, CStatusMessage::SeverityInfo, u"Illegal / ignored equipment code(s) in VATSIM data file: %1") << illegalEquipmentCodes.join(", "));
284  }
285 
286  // data read finished
287  emit this->dataFileRead(dataFileData.size() / 1000);
288  emit this->dataRead(CEntityFlags::VatsimDataFile, CEntityFlags::ReadFinished, dataFileData.size() / 1000, url);
289  }
290  else
291  {
292  // network error
293  CLogMessage(this).warning(u"Reading VATSIM data file failed '%1' '%2'") << nwReply->errorString() << urlString;
294  nwReply->abort();
295  emit this->dataRead(CEntityFlags::VatsimDataFile, CEntityFlags::ReadFailed, 0, url);
296  }
297  }

299  CSimulatedAircraft CVatsimDataFileReader::parsePilot(const QJsonObject &pilot, QStringList &o_illegalEquipmentCodes) const
300  {
301  const CCallsign callsign(pilot["callsign"].toString());
302  const CUser user(pilot["cid"].toString(), pilot["name"].toString(), callsign);
303  const CCoordinateGeodetic position(pilot["latitude"].toDouble(), pilot["longitude"].toDouble(), pilot["altitude"].toInt());
304  const CHeading heading(pilot["heading"].toInt(), CAngleUnit::deg());
305  const CSpeed groundspeed(pilot["groundspeed"].toInt(), CSpeedUnit::kts());
306  const CAircraftSituation situation(callsign, position, heading, {}, {}, groundspeed);
307  CSimulatedAircraft aircraft(callsign, user, situation);
308  const QString icaoAndEquipment(pilot["flight_plan"]["aircraft"].toString().trimmed());
309  const QString icao(CFlightPlan::aircraftIcaoCodeFromEquipmentCode(icaoAndEquipment));
310  if (CAircraftIcaoCode::isValidDesignator(icao))
311  {
312  aircraft.setAircraftIcaoCode(icao);
313  }
314  else if (!icaoAndEquipment.isEmpty())
315  {
316  o_illegalEquipmentCodes.push_back(icaoAndEquipment);
317  }
318  aircraft.setTransponderCode(pilot["transponder"].toString().toInt());
319  return aircraft;
320  }

327  CAtcStation CVatsimDataFileReader::parseController(const QJsonObject &controller) const
328  {
329  const CCallsign callsign(controller["callsign"].toString());
330  const CUser user(controller["cid"].toString(), controller["name"].toString(), callsign);
331  const CFrequency freq(controller["frequency"].toString().toDouble(), CFrequencyUnit::kHz());
332  const CLength range(controller["visual_range"].toInt(), CLengthUnit::NM());
333  const QJsonArray atisLines = controller["text_atis"].toArray();
334  const auto atisText = makeRange(atisLines).transform([](auto line) { return line.toString(); });
335  const CInformationMessage atis(CInformationMessage::ATIS, atisText.to<QStringList>().join('\n'));
336  return CAtcStation(callsign, user, freq, {}, range, true, {}, {}, atis);
337  }
