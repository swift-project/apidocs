205  void CAltitude::convertToPressureAltitude(const CPressure &seaLevelPressure)
206  {
207  if (m_altitudeType == PressureAltitude) { return; }
208  if (this->isNull()) { return; }
209  const CPressure deltaPressure = standardISASeaLevelPressure() - seaLevelPressure;
210  const double deltaPressureV = deltaPressure.value(CPressureUnit::mbar());
211  const double deltaAltitudeV = deltaPressureV * 30.0; // 30.0 ft per mbar
212  CLength deltaAltitude(deltaAltitudeV, CLengthUnit::ft());
213  *this += deltaAltitude;
214  m_altitudeType = PressureAltitude;
215  }

231  void CAltitude::parseFromString(const QString &value, CPqString::SeparatorMode mode)
232  {
233  QString v = value.trimmed();
234 
235  // special case FL
236  if (v.contains("FL", Qt::CaseInsensitive) || v.startsWith("F"))
237  {
238  v = char09OnlyString(value);
239  bool ok = false;
240  const int dv = v.toInt(&ok) * 100;
241  const CAltitude a(ok ? dv : 0.0, FlightLevel, ok ? CLengthUnit::ft() : nullptr);
242  *this = a;
243  return;
244  }
245 
246  // special case A (altitude
247  if (v.contains("ALT", Qt::CaseInsensitive) || v.startsWith("A"))
248  {
249  v = char09OnlyString(value);
250  bool ok = false;
251  const int dv = v.toInt(&ok) * 100;
252  const CAltitude a(ok ? dv : 0.0, MeanSeaLevel, ok ? CLengthUnit::ft() : nullptr);
253  *this = a;
254  return;
255  }
256 
257  // normal altitude, AGL/MSL
258  ReferenceDatum rd = MeanSeaLevel;
259  if (v.contains("MSL", Qt::CaseInsensitive))
260  {
261  v = v.replace("MSL", "", Qt::CaseInsensitive).trimmed();
262  rd = MeanSeaLevel;
263  }
264  else if (v.contains("AGL"))
265  {
266  v = v.replace("AGL", "", Qt::CaseInsensitive).trimmed();
267  rd = AboveGround;
268  }
269 
270  const CLength l = CPqString::parse<CLength>(v, mode);
271  *this = CAltitude(l, rd);
272  }

274  bool CAltitude::parseFromFpAltitudeString(const QString &value, CStatusMessageList *msgs)
275  {
276  QString v(value.trimmed()); // do not convert case because of units
277  if (v.startsWith("VFR", Qt::CaseInsensitive))
278  {
279  // we set a more or less meaningful value
280  *this = CAltitude(5000, MeanSeaLevel, CLengthUnit::ft());
281  return true;
282  }
283 
284  this->setNull();
285  if (v.length() < 3)
286  {
287  if (msgs) { msgs->push_back(CStatusMessage(this).validationError(u"Altitude empty or too short")); }
288  return false;
289  }
290 
291  if (!fpAltitudeRegExp().globalMatch(v).hasNext())
292  {
293  if (msgs) { msgs->push_back(CStatusMessage(this).validationError(u"Altitude '%1' needs to match format '%2'") << v << fpAltitudeExamples()); }
294  return false;
295  }
296 
297  // normalize characters to upper/lower
298  // in same step get numeric value only
299  bool beforeDigit = true;
300  QString numericPart;
301  for (int i = 0; i < v.length(); i++)
302  {
303  const QChar c = v[i];
304  if (c.isDigit())
305  {
306  beforeDigit = false;
307  numericPart.push_back(c);
308  continue;
309  }
310  v[i] = beforeDigit ? c.toUpper() : c.toLower();
311  }
312 
313  if (numericPart.isEmpty())
314  {
315  if (msgs) { msgs->push_back(CStatusMessage(this).validationError(u"Altitude '%1' has no numeric part")); }
316  return false;
317  }
318 
319  bool ok;
320  if (v.startsWith("F", Qt::CaseInsensitive))
321  {
322  this->setUnit(CLengthUnit::ft());
323  this->setValueSameUnit(numericPart.toInt(&ok) * 100);
324  m_datum = FlightLevel;
325  }
326  else if (v.startsWith("S", Qt::CaseInsensitive))
327  {
328  this->setUnit(CLengthUnit::m());
329  this->setValueSameUnit(numericPart.toInt(&ok) * 10);
330  m_datum = FlightLevel;
331  }
332  else if (v.startsWith("A", Qt::CaseInsensitive))
333  {
334  this->setUnit(CLengthUnit::ft());
335  this->setValueSameUnit(numericPart.toInt(&ok) * 100);
336  m_datum = MeanSeaLevel;
337  }
338  else if (v.startsWith("M", Qt::CaseInsensitive))
339  {
340  this->setUnit(CLengthUnit::m());
341  this->setValueSameUnit(numericPart.toInt(&ok) * 10);
342  m_datum = MeanSeaLevel;
343  }
344  else if (v.endsWith(CLengthUnit::m().getSymbol()))
345  {
346  this->setUnit(CLengthUnit::m());
347  this->setValueSameUnit(numericPart.toInt(&ok));
348  m_datum = MeanSeaLevel;
349  }
350  else if (v.endsWith(CLengthUnit::ft().getSymbol()))
351  {
352  this->setUnit(CLengthUnit::ft());
353  this->setValueSameUnit(numericPart.toInt(&ok));
354  m_datum = MeanSeaLevel;
355  }
356  else
357  {
358  if (msgs) { msgs->push_back(CStatusMessage(this).validationError(u"Altitude '%1' needs to match format '%2'") << v << fpAltitudeExamples()); }
359  return false;
360  }
361 
362  // further checks
363  const bool valid = this->isValidFpAltitude(msgs);
364  if (!valid) { this->setNull(); }
365  return valid;
366  }

368  bool CAltitude::isValidFpAltitude(CStatusMessageList *msgs) const
369  {
370  if (this->isNull())
371  {
372  if (msgs) { msgs->push_back(CStatusMessage(this).validationError(u"Altitude NULL value")); }
373  return false;
374  }
375  if (!(this->getReferenceDatum() == FlightLevel || this->getReferenceDatum() == MeanSeaLevel))
376  {
377  if (msgs) { msgs->push_back(CStatusMessage(this).validationError(u"Altitude, must be FL or MSL")); }
378  return false;
379  }
380  if (!(this->getUnit() == CLengthUnit::m() || this->getUnit() == CLengthUnit::ft()))
381  {
382  if (msgs) { msgs->push_back(CStatusMessage(this).validationError(u"Altitude, valid unit must be 'ft' or 'm'")); }
383  return false;
384  }
385  if (this->isNegativeWithEpsilonConsidered())
386  {
387  if (msgs) { msgs->push_back(CStatusMessage(this).validationError(u"Altitude must be positive")); }
388  return false;
389  }
390  if (this->isFlightLevel())
391  {
392  if (!this->isInteger())
393  {
394  if (msgs) { msgs->push_back(CStatusMessage(this).validationError(u"FL needs to be positive integer")); }
395  return false;
396  }
397 
398  if (this->getUnit() == CLengthUnit::ft())
399  {
400  const int flInt = this->valueInteger() / 100; // internally represented as ft: FL10->1000
401  if (flInt < 10 || flInt >= 1000)
402  {
403  if (msgs) { msgs->push_back(CStatusMessage(this).validationError(u"FL range is 10-999")); }
404  return false;
405  }
406  if (!CMathUtils::epsilonZero(fmod(flInt, 5)))
407  {
408  if (msgs) { msgs->push_back(CStatusMessage(this).validationWarning(u"FL should end with 0 or 5")); }
409  }
410  }
411  }
412  return true;
413  }
