40  CUrl CUrlList::getRandomWorkingUrl(int maxTrials, int timeoutMs) const
41  {
42  if (this->isEmpty()) { return CUrl(); }
43  if (maxTrials < 1) { return CUrl(); }
44  CUrlList trials;
45 
46  if (timeoutMs < 0) { timeoutMs = CNetworkUtils::getTimeoutMs(); }

48  {

171  int CFailoverUrlList::numberOfStillValidUrls() const
172  {
173  if (this->isEmpty()) { return 0; }
174  if (m_failedUrls.size() >= m_maxTrials) { return 0; }
175  const int trailsLeft1 = qMax(m_maxTrials - m_failedUrls.size(), 0);
176  const int trailsLeft2 = qMax(this->size() - m_failedUrls.size(), 0);
177  return qMin(trailsLeft1, trailsLeft2);

180  CUrl CFailoverUrlList::obtainNextWorkingUrl(bool random, int connectTimeoutMs)
181  {
182  if (!hasMoreUrlsToTry()) { return CUrl(); }
183  const CUrl url(this->obtainNextUrlWithout(random, m_failedUrls));
184  QString msg;
185  if (CNetworkUtils::canConnect(url, msg, connectTimeoutMs)) { return url; }
186  if (addFailedUrl(url))
187  {
188  if (!msg.isEmpty())
189  {
190  m_errorMsgs.append(QStringLiteral("URL: %1 error: %2").arg(url.toQString(), msg));
191  }
192  return obtainNextWorkingUrl(random, connectTimeoutMs);
193  }
194  return CUrl();

197  CUrl CFailoverUrlList::obtainNextUrl(bool randomStart)
198  {
199  if (this->isEmpty()) { return CUrl(); }
200  if (this->size() == 1) { return this->front(); }
201  if (m_currentIndexDistributedLoad < 0)
202  {
203  // random start point
204  m_currentIndexDistributedLoad = randomStart ?
205  CMathUtils::randomInteger(0, this->size() - 1) :
206  0;
207  }
208  else
209  {
210  m_currentIndexDistributedLoad++;
211  if (m_currentIndexDistributedLoad >= this->size())
212  {
213  m_currentIndexDistributedLoad = 0;
214  }
215  }
