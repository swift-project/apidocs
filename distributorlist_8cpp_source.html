47  CDistributor CDistributorList::smartDistributorSelector(const CDistributor &distributor) const
48  {
49  // key is not necessarily a DB key, so use complete data, happens when key is set from raw data
50  if (distributor.isLoadedFromDb()) { return distributor; }
51  if (distributor.hasValidDbKey())
52  {
53  const QString key(distributor.getDbKey());
54  CDistributor d(this->findByKey(key));
55  if (d.hasCompleteData()) { return d; }
56 
57  // more lenient search
58  return this->findByKeyOrAlias(key);
59  }
60  return CDistributor();
61  }
