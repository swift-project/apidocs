35  QStringList EuroscopeSimData::toTokens() const
36  {
37  static constexpr auto toFlags = [](const CAircraftLights &lights) {
38  int n = 0;
39  if (lights.isBeaconOn()) { n |= 0x002; }
40  if (lights.isCabinOn()) { n |= 0x400; }
41  if (lights.isLandingOn()) { n |= 0x020; }
42  if (lights.isLogoOn()) { n |= 0x200; }
43  if (lights.isNavOn()) { n |= 0x008; }
44  if (lights.isRecognitionOn()) { n |= 0x080; }
45  if (lights.isStrobeOn()) { n |= 0x004; }
46  if (lights.isTaxiOn()) { n |= 0x040; }
47  return n;
48  };
49 
50  QStringList tokens;
51  tokens.push_back({}); // first token is empty
52  tokens.push_back(m_sender);
53  tokens.push_back(m_model);
54  tokens.push_back(m_livery);
55  tokens.push_back(QString::number(m_timestamp));
56  tokens.push_back(QString::number(m_latitude, 'f', 7));
57  tokens.push_back(QString::number(m_longitude, 'f', 7));
58  tokens.push_back(QString::number(m_altitude, 'f', 1));
59  tokens.push_back(QString::number(m_heading, 'f', 2));
60  tokens.push_back(QString::number(m_bank));
61  tokens.push_back(QString::number(m_pitch));
62  tokens.push_back(QString::number(m_groundSpeed));
63  tokens.push_back(m_onGround ? QStringLiteral("1") : QStringLiteral("0"));
64  tokens.push_back(QString::number(m_gearPercent));
65  tokens.push_back(QString::number(m_thrustPercent));
66  tokens.push_back(QStringLiteral("0")); // emergency flags
67  tokens.push_back(QStringLiteral("0.0")); // airport altitude
68  tokens.push_back(QString::number(toFlags(m_lights)));
69  return tokens;
70  }
