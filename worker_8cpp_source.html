32  CRegularThread::~CRegularThread()
33  {
34  const QString name = this->objectName();
35 
36 #ifdef Q_OS_WIN32
37  auto handle = m_handle.load();
38  if (handle)
39  {
40  const auto status = WaitForSingleObject(handle, 0);
41  if (isRunning())
42  {
43  switch (status)
44  {
45  default:
46  case WAIT_FAILED: qWarning() << "Thread" << name << "unspecified error"; break;
47  case WAIT_OBJECT_0: qWarning() << "Thread" << name << "unsafely terminated by program shutdown"; break;
48  case WAIT_TIMEOUT: break;
49  }
50  }
51  }
52 #endif
53  quit();
54 
55  // the wait avoids: QThread: Destroyed while thread is still running
56  const unsigned long timeoutMs = 5 * 1000;
57  const bool ok = wait(timeoutMs);
58  const QString as = QStringLiteral("Wait timeout after %1ms for '%2'").arg(timeoutMs).arg(name);
59  const QByteArray asBA = as.toLatin1();
60  BLACK_AUDIT_X(ok, Q_FUNC_INFO, asBA); // MS 2018-09 assert because we want a stack trace of all threads, via breakpad
61  Q_UNUSED(ok)
62  }

64  CWorker *CWorker::fromTaskImpl(QObject *owner, const QString &name, int typeId, const std::function<QVariant()> &task)
65  {
66  auto *worker = new CWorker(task);
67  emit worker->aboutToStart();
68  worker->setStarted();
69  auto *thread = new CRegularThread(owner);
70 
71  if (typeId != QMetaType::Void) { worker->m_result = QVariant(typeId, nullptr); }
72 
73  const QString ownerName = owner->objectName().isEmpty() ? owner->metaObject()->className() : owner->objectName();
74  thread->setObjectName(ownerName + ":" + name);
75  worker->setObjectName(name);
76 
77  worker->moveToThread(thread);
78  const bool s = QMetaObject::invokeMethod(worker, &CWorker::ps_runTask);
79  Q_ASSERT_X(s, Q_FUNC_INFO, "cannot invoke");
80  Q_UNUSED(s)
81  thread->start();
82  return worker;
83  }

85  void CWorker::ps_runTask()
86  {
87  m_result = m_task();
88 
89  this->setFinished();
90 
91  QThread *workerThread = this->thread();
92  Q_ASSERT_X(workerThread->thread()->isRunning(), Q_FUNC_INFO, "Owner thread's event loop already ended");
93 
94  // MS 2018-09 Now we post the DeferredDelete event from within the worker thread, but rely on it being dispatched
95  // by the owner thread. Posted events are moved along with the object when moveToThread is called.
96  this->deleteLater();
97 
98  this->moveToThread(workerThread->thread()); // move worker back to the thread which constructed it, so there is no race on deletion
99  // must not access the worker beyond this point, as it now lives in the owner's thread and could be deleted at any moment
100 
101  QMetaObject::invokeMethod(workerThread, [workerThread] {
102  // quit and wait is redundant as the CRegularThread dtor will do that anyway, but put here for debugging
103  workerThread->quit();
104  const bool ok = workerThread->wait(5000);
105  const QString as = QStringLiteral("Worker thread '%2' refuses to stop after worker finished").arg(workerThread->objectName());
106  const QByteArray asBA = as.toLatin1();
107  BLACK_AUDIT_X(ok, Q_FUNC_INFO, asBA);
108 
109  workerThread->deleteLater();
110  });
111  }

161  void CContinuousWorker::start(QThread::Priority priority)
162  {
163  BLACK_VERIFY_X(!hasStarted(), Q_FUNC_INFO, "Tried to start a worker that was already started");
164  if (hasStarted()) { return; }
165 
166  // avoid message "QObject: Cannot create children for a parent that is in a different thread"
167  Q_ASSERT_X(CThreadUtils::isInThisThread(m_owner), Q_FUNC_INFO, "Needs to be started in owner thread");
168  emit this->aboutToStart();
169  setStarted();
170  auto *thread = new CRegularThread(m_owner);
171 
172  Q_ASSERT(m_owner); // must not be null, see (9) https://dev.vatsim-germany.org/issues/402
173  if (m_owner)
174  {
175  const QString ownerName = m_owner->objectName().isEmpty() ? m_owner->metaObject()->className() : m_owner->objectName();
176  thread->setObjectName(ownerName + ": " + m_name);
177  }
178 
179  moveToThread(thread);
180  connect(thread, &QThread::started, this, &CContinuousWorker::initialize);
181  connect(thread, &QThread::finished, &m_updateTimer, &QTimer::stop);
182  connect(thread, &QThread::finished, this, &CContinuousWorker::cleanup);
183  connect(thread, &QThread::finished, this, &CContinuousWorker::finish);
184  thread->start(priority);
185  }

191  // already in owner's thread? then return

199  void CContinuousWorker::quitAndWait() noexcept
200  {
201  this->setEnabled(false);
202 
203  // already in owner's thread? then return
204  if (this->thread() == m_owner->thread()) { return; }
205 
206  // called by own thread, will deadlock, return
207  if (CThreadUtils::isInThisThread(this)) { return; }
208 
209  QThread *workerThread = thread(); // must be before quit()
210  this->quit();
211 
212  // T647, discussed here: https://discordapp.com/channels/539048679160676382/539925070550794240/573260844004016148
213  const unsigned long waitTimeoutMs = this->waitTimeoutMs();
214  const QString name(this->getName());
215  qint64 waitTime = QDateTime::currentMSecsSinceEpoch();
216  const bool ok = workerThread->wait(waitTimeoutMs);
217  waitTime = QDateTime::currentMSecsSinceEpoch() - waitTime;
218  const QString msg = QStringLiteral("Waiting for quitAndWait of '%1' for %2ms").arg(name).arg(waitTime);
219  const QByteArray msgBA = msg.toLatin1();
220  BLACK_AUDIT_X(ok, Q_FUNC_INFO, msgBA); // MS 2019-05 AUDIT because we want a stack trace of all threads, via breakpad
221  CLogMessage(this).info(msg);
222  Q_UNUSED(ok)
223  }

225  void CContinuousWorker::startUpdating(int updateTimeSecs)
226  {
227  Q_ASSERT_X(this->hasStarted(), Q_FUNC_INFO, "Worker not yet started");
228  if (!CThreadUtils::isInThisThread(this))
229  {
230  // shift in correct thread
231  QPointer<CContinuousWorker> myself(this);
232  QTimer::singleShot(0, this, [=] {
233  if (!myself) { return; }
234  this->doIfNotFinished([=] { startUpdating(updateTimeSecs); });
235  });
236  return;
237  }
238 
239  // here in correct timer thread
240  if (updateTimeSecs < 0)
241  {
242  this->setEnabled(false);
243  m_updateTimer.stop();
244  }
245  else
246  {
247  this->setEnabled(true);
248  m_updateTimer.start(1000 * updateTimeSecs);
249  }
250  }

252  void CContinuousWorker::stopUpdateTimer()
253  {
254  if (!m_updateTimer.isActive()) { return; }
255 
256  // avoid "Timers cannot be stopped from another thread"
257  if (CThreadUtils::isInThisThread(&m_updateTimer))
258  {
259  m_updateTimer.stop();
260  }
261  else
262  {
263  QPointer<CContinuousWorker> myself(this);
264  QTimer::singleShot(0, &m_updateTimer, [=] {
265  // stop timer in timer thread
266  if (!myself) { return; }
267  m_updateTimer.stop();
268  });
269  }
270  }
