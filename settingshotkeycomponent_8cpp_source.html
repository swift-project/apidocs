84  void CSettingsHotkeyComponent::editEntry()
85  {
86  const auto index = ui->tv_Hotkeys->selectionModel()->currentIndex();
87  if (!index.isValid()) { return; }
88 
89  const auto model = ui->tv_Hotkeys->model();
90  const QModelIndex indexHotkey = model->index(index.row(), 0, QModelIndex());
91  Q_ASSERT_X(indexHotkey.data(CActionHotkeyListModel::ActionHotkeyRole).canConvert<CActionHotkey>(), Q_FUNC_INFO, "No action hotkey");
92  const CActionHotkey actionHotkey = indexHotkey.data(CActionHotkeyListModel::ActionHotkeyRole).value<CActionHotkey>();
93  const CActionHotkey selectedActionHotkey = CHotkeyDialog::getActionHotkey(actionHotkey, getAllIdentifiers(), this);
94  if (selectedActionHotkey.isValid() && checkAndConfirmConflicts(selectedActionHotkey, { actionHotkey }))
95  {
96  updateHotkeyInSettings(actionHotkey, selectedActionHotkey);
97  m_model.setData(indexHotkey, QVariant::fromValue(selectedActionHotkey), CActionHotkeyListModel::ActionHotkeyRole);
98 
99  // T784, further info about the "key"/button
100  CLogMessage(this).debug(u"%1, edited key: '%2'") << classNameShort(this) << selectedActionHotkey.toQString(true);
101  }
102  this->resizeView();
103  }

142  bool CSettingsHotkeyComponent::checkAndConfirmConflicts(const CActionHotkey &actionHotkey, const CActionHotkeyList &ignore)
143  {
144  // check the hotkeys of the same machine only
145  // and avoid duplicates (replace or add)
146  const CActionHotkeyList configuredHotkeysSameMachine = m_actionHotkeys.getThreadLocal().findBySameMachine(actionHotkey);
147  CActionHotkeyList conflicts = configuredHotkeysSameMachine.findSupersetsOf(actionHotkey);
148  conflicts.replaceOrAdd(configuredHotkeysSameMachine.findSubsetsOf(actionHotkey));
149  conflicts.removeIfIn(ignore.findBySameMachine(actionHotkey));
150 
151  if (!conflicts.isEmpty())
152  {
153  QString message = QStringLiteral("The selected combination conflicts with the following %1 combination(s):\n\n").arg(conflicts.size());
154  for (const CActionHotkey &conflict : conflicts)
155  {
156  message += conflict.toQString();
157  message += "\n";
158  }
159  message += "\n Do you want to use it anway?";
160  const auto reply = QMessageBox::warning(this, "SettingsHotkeyComponent", message, QMessageBox::Yes | QMessageBox::No, QMessageBox::No);
161  if (reply == QMessageBox::No) { return false; }
162  }
163  return true;
164  }
