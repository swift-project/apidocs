67  void CDbAutoStashingComponent::accept()
68  {
69  switch (m_state)
70  {
71  case Running: return;
72  case Completed:
73  {
74  if (!m_modelsToStash.isEmpty())
75  {
76  // this removes previously stashed models
77  this->getMappingComponent()->replaceStashedModelsUnvalidated(m_modelsToStash);
78  if (ui->cb_RemovedChecked->isChecked())
79  {
80  this->currentModelView()->removeModelsWithModelString(m_modelsToStash);
81  }
82  const CStatusMessage stashedMsg(this, CStatusMessage::SeverityInfo, QStringLiteral("Auto stashed %1 models").arg(m_modelsToStash.size()));
83  this->addStatusMessage(stashedMsg);
84  m_modelsToStash.clear();
85  }
86  QDialog::accept();
87  break;
88  }
89  default:
90  {
91  if (this->getSelectedOrAllCount() < 1)
92  {
93  const CStatusMessage m(this, CStatusMessage::SeverityError, u"No data, nothing to do");
94  this->addStatusMessage(m);
95  QDialog::accept();
96  }
97  this->tryToStashModels();
98  }
99  }
100  }

220  void CDbAutoStashingComponent::tryToStashModels()
221  {
222  Q_ASSERT_X(this->currentModelView(), Q_FUNC_INFO, "No view");
223  const CAircraftModelList models(ui->rb_Selected->isChecked() ? this->currentModelView()->selectedObjects() : this->currentModelView()->containerOrFilteredContainer());
224  if (models.isEmpty()) { return; }
225  if (!sGui || sGui->isShuttingDown()) { return; }
226 
227  // we have data and are good to go
228  m_state = Running;
229  const int all = models.size();
230 
231  // maximum
232  int max = CDbStashComponent::MaxModelPublished;
233  const QString maxStr(ui->le_MaxModelsStashed->text());
234  bool okMaxStr = true;
235  if (!maxStr.isEmpty()) { max = maxStr.toInt(&okMaxStr); }
236  if (!okMaxStr || max > all) { max = all; }
237 
238  // override description
239  const QString description(ui->le_Description->text().trimmed());
240 
241  // temp livery if applicable
242  const CLivery tempLivery(ui->cb_UseTempLivery->isChecked() ? getTempLivery() : CLivery());
243 
244  int c = 0;
245  CAircraftModelList autoStashed;
246  for (const CAircraftModel &model : models)
247  {
248  CAircraftModel stashModel(model);
249  const bool stashed = this->tryToStashModel(stashModel, tempLivery);
250  if (stashed)
251  {
252  if (!description.isEmpty())
253  {
254  this->setModelDescription(stashModel, description);
255  }
256  autoStashed.push_back(stashModel);
257  }
258 
259  c++;
260  if (c % 25 == 0)
261  {
262  Q_ASSERT_X(c <= all, Q_FUNC_INFO, "illegal numbers");
263  sGui->processEventsToRefreshGui();
264 
265  int percent = c * 100 / all;
266  if (max < all)
267  {
268  int maxPercent = autoStashed.size() * 100 / max;
269  if (maxPercent > percent) { percent = maxPercent; }
270  }
271  this->updateProgressIndicator(percent);
272  }
273  if (autoStashed.size() >= max) { break; }
274  }
275 
276  this->updateProgressIndicator(100);
277  sGui->processEventsToRefreshGui();
278 
279  const CStatusMessage stashedMsg(this, CStatusMessage::SeverityInfo, QStringLiteral("Ready to auto stash %1 models").arg(autoStashed.size()));
280  this->addStatusMessage(stashedMsg);
281  m_modelsToStash = autoStashed;
282  m_state = Completed;
283  }

285  bool CDbAutoStashingComponent::tryToStashModel(CAircraftModel &model, const CLivery &tempLivery)
286  {
287  const bool useTempLivery = tempLivery.isLoadedFromDb();
288 
289  // no airline and no livery, here replaced by temp livery
290  if (useTempLivery && !model.hasAirlineDesignator() && !model.getLivery().hasValidDbKey())
291  {
292  model.setLivery(tempLivery);
293  }
294 
296  if (!model.hasModelString())
297  {
298  this->addStatusMessage(CStatusMessage(this, CStatusMessage::SeverityError, u"No model string"));
299  m_noData++;
300  return false;
301  }
302 
303  if (!model.hasAircraftDesignator())
304  {
305  this->addStatusMessage(CStatusMessage(this, CStatusMessage::SeverityError, u"No aircraft designator"), model);
306  m_noData++;
307  return false;
308  }
309 
310  if (!model.hasAirlineDesignator() && !model.getLivery().hasValidDbKey())
311  {
312  // no valid airline and NO DB livery
313  // we try one fallback
314  const QString liveryCombinedCode = model.getLivery().getCombinedCode();
315  bool fallback = false;
316  if (liveryCombinedCode.length() == 3 || liveryCombinedCode.length() == 4)
317  {
318  // could we use the combined code as airline
319  if (CAirlineIcaoCode::isValidAirlineDesignator(liveryCombinedCode))
320  {
321  model.setAirlineIcaoDesignator(liveryCombinedCode);
322  fallback = true;
323  }
324  }
325 
326  // if there is no livery (normal) we need an airline
327  if (!fallback)
328  {
329  this->addStatusMessage(CStatusMessage(this, CStatusMessage::SeverityError, u"No airline designator"), model);
330  m_noData++;
331  return false;
332  }
333  }
334 
335  // stash here consolidates with DB data and validates
336  bool stashed = false;
337  const CAircraftModel stashModel(this->getMappingComponent()->consolidateModel(model));
338  CStatusMessageList validationMsgs(stashModel.validate(true));
339  validationMsgs.removeWarningsAndBelow();
340  CStatusMessage msg = validationMsgs.toSingleMessage();
341  if (msg.getSeverity() == CStatusMessage::SeverityError)
342  {
343  m_noValidationFailed++;
344  }
345  else
346  {
347  msg = CStatusMessage(this, CStatusMessage::SeverityInfo, u"Stashed succesfully");
348  stashed = true;
349  m_noStashed++;
350  model = stashModel;
351  }
352  this->addStatusMessage(msg, stashModel);
353  return stashed;
354  }
