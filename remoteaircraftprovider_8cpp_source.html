256  CAircraftSituation CRemoteAircraftProvider::storeAircraftSituation(const CAircraftSituation &situation, bool allowTestAltitudeOffset)
257  {
258  const CCallsign cs = situation.getCallsign();
259  if (cs.isEmpty()) { return situation; }
260 
261  // testing
262  if (CBuildConfig::isLocalDeveloperDebugBuild())
263  {
264  BLACK_VERIFY_X(situation.getTimeOffsetMs() > 0, Q_FUNC_INFO, "Missing offset");
265  BLACK_VERIFY_X(situation.isValidVectorRange(), Q_FUNC_INFO, "Invalid vector");
266  }
267 
268  // add altitude offset (for testing only)
269  CAircraftSituation situationCorrected(allowTestAltitudeOffset ? this->addTestAltitudeOffsetToSituation(situation) : situation);
270 
271  // CG, model
272  const CAircraftModel aircraftModel = this->getAircraftInRangeModelForCallsign(cs);
273  if (situation.hasCG() && aircraftModel.getCG() != situation.getCG())
274  {
275  this->updateCG(cs, situation.getCG());
276  }
277 
278  // list from new to old
279  CAircraftSituationList updatedSituations; // copy of updated situations
280  {
281  const qint64 now = QDateTime::currentMSecsSinceEpoch();
282  QWriteLocker lock(&m_lockSituations);
283  m_situationsAdded++;
284  m_situationsLastModified[cs] = now;
285  CAircraftSituationList &newSituationsList = m_situationsByCallsign[cs];
286  newSituationsList.setAdjustedSortHint(CAircraftSituationList::AdjustedTimestampLatestFirst);
287  const int situations = newSituationsList.size();
288  if (situations < 1)
289  {
290  newSituationsList.prefillLatestAdjustedFirst(situationCorrected, IRemoteAircraftProvider::MaxSituationsPerCallsign);
291  }
292  else if (!situationCorrected.hasVelocity() && newSituationsList.front().hasVelocity())
293  {
294  return situationCorrected;
295  }
296  else
297  {
298  // newSituationsList.push_frontKeepLatestFirstIgnoreOverlapping(situationCorrected, true, IRemoteAircraftProvider::MaxSituationsPerCallsign);
299  newSituationsList.push_frontKeepLatestFirstAdjustOffset(situationCorrected, true, IRemoteAircraftProvider::MaxSituationsPerCallsign);
300  newSituationsList.setAdjustedSortHint(CAircraftSituationList::AdjustedTimestampLatestFirst);
301  newSituationsList.transferElevationForward(); // transfer elevations, will do nothing if elevations already exist
302 
303  // unify all inbound ground information
304  if (situation.hasInboundGroundDetails())
305  {
306  newSituationsList.setOnGroundDetails(situation.getOnGroundDetails());
307  }
308  }
309  m_latestSituationByCallsign[cs] = situationCorrected;
310 
311  // check sort order
312  if (CBuildConfig::isLocalDeveloperDebugBuild())
313  {
314  BLACK_VERIFY_X(newSituationsList.isSortedAdjustedLatestFirstWithoutNullPositions(), Q_FUNC_INFO, "wrong adjusted sort order");
315  BLACK_VERIFY_X(newSituationsList.isSortedLatestFirst(), Q_FUNC_INFO, "wrong sort order");
316  BLACK_VERIFY_X(newSituationsList.size() <= IRemoteAircraftProvider::MaxSituationsPerCallsign, Q_FUNC_INFO, "Wrong size");
317  }
318 
319  if (!situation.hasInboundGroundDetails())
320  {
321  // first use a version without standard deviations to guess "on ground
322  const CAircraftSituationChange simpleChange(updatedSituations, situationCorrected.getCG(), aircraftModel.isVtol(), true, false);
323 
324  // guess GND
325  simpleChange.guessOnGround(newSituationsList.front(), aircraftModel);
326  }
327  updatedSituations = m_situationsByCallsign[cs];
328 
329  } // lock
330 
331  // calculate change AFTER gnd. was guessed
332  Q_ASSERT_X(!updatedSituations.isEmpty(), Q_FUNC_INFO, "Missing situations");
333  const CAircraftSituationChange change(updatedSituations, situationCorrected.getCG(), aircraftModel.isVtol(), true, true);
334  this->storeChange(change);
335 
336  if (change.hasSceneryDeviation())
337  {
338  const CLength offset = change.getGuessedSceneryDeviation();
339  situationCorrected.setSceneryOffset(offset);
340 
341  QWriteLocker lock(&m_lockSituations);
342  m_latestSituationByCallsign[cs].setSceneryOffset(offset);
343  m_situationsByCallsign[cs].front().setSceneryOffset(offset);
344  }
345 
346  // situation has been added
347  emit this->addedAircraftSituation(situationCorrected);
348 
349  // bye
350  return situationCorrected;
351  }

353  void CRemoteAircraftProvider::storeAircraftParts(const CCallsign &callsign, const CAircraftParts &parts, bool removeOutdated)
354  {
355  BLACK_VERIFY_X(!callsign.isEmpty(), Q_FUNC_INFO, "empty callsign");
356  if (callsign.isEmpty()) { return; }
357 
358  // list sorted from new to old
359  const qint64 ts = QDateTime::currentMSecsSinceEpoch();
360  CAircraftPartsList correctiveParts;
361  {
362  QWriteLocker lock(&m_lockParts);
363  m_partsAdded++;
364  m_partsLastModified[callsign] = ts;
365  CAircraftPartsList &partsList = m_partsByCallsign[callsign];
366  partsList.push_frontKeepLatestFirstAdjustOffset(parts, true, IRemoteAircraftProvider::MaxPartsPerCallsign);
367  partsList.setAdjustedSortHint(CAircraftPartsList::AdjustedTimestampLatestFirst);
368 
369  // remove outdated parts (but never remove the most recent one)
370  if (removeOutdated) { IRemoteAircraftProvider::removeOutdatedParts(partsList); }
371  correctiveParts = partsList;
372 
373  // check sort order
374  Q_ASSERT_X(partsList.isSortedAdjustedLatestFirst(), Q_FUNC_INFO, "wrong sort order");
375  Q_ASSERT_X(partsList.size() <= IRemoteAircraftProvider::MaxPartsPerCallsign, Q_FUNC_INFO, "Wrong size");
376  } // lock
377 
378  // adjust gnd.flag from parts
379  if (!correctiveParts.isEmpty())
380  {
381  QWriteLocker lock(&m_lockSituations);
382  CAircraftSituationList &situationList = m_situationsByCallsign[callsign];
383  const int c = situationList.adjustGroundFlag(parts);
384  if (c > 0) { m_situationsLastModified[callsign] = ts; }
385  }
386 
387  // update aircraft
388  {
389  QWriteLocker l(&m_lockAircraft);
390  if (m_aircraftInRange.contains(callsign))
391  {
392  CSimulatedAircraft &aircraft = m_aircraftInRange[callsign];
393  aircraft.setParts(parts);
394  aircraft.setPartsSynchronized(true);
395  }
396  }
397 
398  // update parts
399  {
400  // aircraft supporting parts
401  QWriteLocker l(&m_lockParts);
402  m_aircraftWithParts.insert(callsign); // mark as callsign which supports parts
403  }
404 
405  emit this->addedAircraftParts(callsign, parts);
406  }

408  void CRemoteAircraftProvider::storeAircraftParts(const CCallsign &callsign, const QJsonObject &jsonObject, qint64 currentOffsetMs)
409  {
410  const CSimulatedAircraft remoteAircraft(this->getAircraftInRangeForCallsign(callsign));
411  const bool isFull = jsonObject.value(CAircraftParts::attributeNameIsFullJson()).toBool();
412  const bool validCs = remoteAircraft.hasValidCallsign();
413  if (!validCs)
414  {
415  if (!isFull) { return; } // incremental parts broadcasting
416  return; // suspicious
417  }
418 
419  // If we are not yet synchronized, we throw away any incremental packet
420  if (!remoteAircraft.isPartsSynchronized() && !isFull) { return; }
421 
422  CAircraftParts parts;
423  try
424  {
425  if (isFull)
426  {
427  if (CBuildConfig::isLocalDeveloperDebugBuild())
428  {
429  // validation in dev.env.
430  const int attributes = jsonObject.size();
431  const bool correctCount = (attributes == CAircraftParts::attributesCountFullJson);
432  BLACK_VERIFY_X(correctCount || !CBuildConfig::isLocalDeveloperDebugBuild(), Q_FUNC_INFO, "Wrong full aircraft parts");
433  if (!correctCount)
434  {
435  CLogMessage(this).warning(u"Wrong full parts attributes, %1 (expected %2)") << attributes << CAircraftParts::attributesCountFullJson;
437  if (attributes < 3)
438  {
439  // EXPERIMENTAL
440  if (attributes < 1) { return; }
441 
442  // treat as incremental
443  CLogMessage(this).warning(u"Treating %1 attributes as incremental") << attributes;
444  parts = this->remoteAircraftParts(callsign).frontOrDefault(); // latest
445  const QJsonObject config = applyIncrementalObject(parts.toJson(), jsonObject);
446  parts.convertFromJson(config);
447  }
448  }
449  }
450  parts.convertFromJson(jsonObject);
451  }
452  else
453  {
454  // incremental update
455  parts = this->remoteAircraftParts(callsign).frontOrDefault(); // latest
456  const QJsonObject config = applyIncrementalObject(parts.toJson(), jsonObject);
457  parts.convertFromJson(config);
458  }
459  }
460  catch (const CJsonException &ex)
461  {
462  CStatusMessage message = CStatusMessage::fromJsonException(ex, this, "Invalid parts packet");
463  message.setSeverity(CStatusMessage::SeverityDebug);
464  CLogMessage::preformatted(message);
465  }
466 
467  // make sure in any case right time and correct details
468  parts.setCurrentUtcTime();
469  parts.setTimeOffsetMs(currentOffsetMs);
470  parts.setPartsDetails(CAircraftParts::FSDAircraftParts);
471 
472  // store part history (parts always absolute)
473  this->storeAircraftParts(callsign, parts, false);
474 
475  // history
476  if (this->isAircraftPartsHistoryEnabled())
477  {
478  const QJsonDocument doc(jsonObject);
479  const QString partsAsString = doc.toJson(QJsonDocument::Compact);
480  const CStatusMessage message(this, CStatusMessage::SeverityInfo, callsign.isEmpty() ? callsign.toQString() + ": " + partsAsString.trimmed() : partsAsString.trimmed());
481 
482  QReadLocker l(&m_lockPartsHistory);
483  if (m_aircraftPartsMessages.contains(callsign))
484  {
485  CStatusMessageList &msgs = m_aircraftPartsMessages[callsign];
486  msgs.push_back(message);
487  }
488  else
489  {
490  m_aircraftPartsMessages.insert(callsign, message);
491  }
492  }
493  }

579  int CRemoteAircraftProvider::updateAircraftGroundElevation(const CCallsign &callsign, const CElevationPlane &elevation, CAircraftSituation::GndElevationInfo info, bool *setForOnGroundPosition)
580  {
581  if (!this->isAircraftInRange(callsign)) { return 0; }
582 
583  // update aircraft situation
584  const qint64 now = QDateTime::currentMSecsSinceEpoch();
585  const CAircraftModel model = this->getAircraftInRangeModelForCallsign(callsign);
586  CAircraftSituationChange change;
587  bool setForOnGndPosition = false;
588 
589  int updated = 0;
590  {
591  QWriteLocker l(&m_lockSituations);
592  CAircraftSituationList &situations = m_situationsByCallsign[callsign];
593  if (situations.isEmpty()) { return 0; }
594  updated = setGroundElevationCheckedAndGuessGround(situations, elevation, info, model, &change, &setForOnGndPosition);
595  if (updated < 1) { return 0; }
596  m_situationsLastModified[callsign] = now;
597  const CAircraftSituation latestSituation = situations.front();
598  if (info == CAircraftSituation::FromProvider && latestSituation.isOnGround())
599  {
600  m_latestOnGroundProviderElevation[callsign] = latestSituation;
601  }
602  }
603 
604  // update change
605  if (!change.isNull())
606  {
607  this->storeChange(change);
608  }
609 
610  // aircraft updates
611  QWriteLocker l(&m_lockAircraft);
612  if (m_aircraftInRange.contains(callsign))
613  {
614  m_aircraftInRange[callsign].setGroundElevationChecked(elevation, info);
615  }
616 
617  if (setForOnGroundPosition) { *setForOnGroundPosition = setForOnGndPosition; }

777  CAircraftSituationList &situations, const CElevationPlane &elevationPlane, CAircraftSituation::GndElevationInfo info, const CAircraftModel &model,
778  CAircraftSituationChange *changeOut, bool *setForOnGroundPosition)
779  {
780  if (setForOnGroundPosition) { *setForOnGroundPosition = false; } // set a default
781  if (elevationPlane.isNull()) { return 0; }

783 
784  // the change has the timestamps of the latest situation
785  // Q_ASSERT_X(situations.m_tsAdjustedSortHint == CAircraftSituationList::AdjustedTimestampLatestFirst || situations.isSortedAdjustedLatestFirstWithoutNullPositions(), Q_FUNC_INFO, "Need sorted situations without NULL positions");

787  int c = 0; // changed elevations
788  bool latest = true;
789  bool setForOnGndPosition = false;
790 
791  for (CAircraftSituation &s : situations)
792  {
793  const bool set = s.setGroundElevationChecked(elevationPlane, info);
794  if (set)
795  {
796  // simpleChange is only valid for the latest situation
797  // this will do nothing if not appropriate!
798  const bool guessed = (latest ? simpleChange : CAircraftSituationChange::null()).guessOnGround(s, model);
799  Q_UNUSED(guessed)
800  c++;
801 
802  // if not guessed and "on ground" we mark the "elevation"
803  // as an elevation for a ground position
804  if (!setForOnGndPosition && s.hasInboundGroundDetails() && s.isOnGround())
805  {
806  setForOnGndPosition = true;
807  }
808  }
809  latest = false; // only first pos. is "the latest" one
810  }
811 
812  if (setForOnGroundPosition) { *setForOnGroundPosition = setForOnGndPosition; }
813  if (changeOut)
814  {
815  const CAircraftSituationChange change(situations, model.getCG(), model.isVtol(), true, true);
816  *changeOut = change;
817  }
818 
819  return c;
