67  bool CCoordinateForm::setCoordinate(const ICoordinateGeodetic &coordinate)
68  {
69  if (coordinate == m_coordinate) { return false; }
70  m_coordinate = coordinate;
71 
72  const CLatitude lat = coordinate.latitude();
73  const QString latWgs = lat.toWgs84();
74  ui->le_Latitude->setText(latWgs);
75  ui->le_Latitude->setToolTip(QString::number(lat.value(CAngleUnit::deg())));
76  ui->lblp_LatCheck->setTicked(!lat.isNull());
77  if (latWgs.contains('S'))
78  {
79  ui->rb_S->setChecked(true);
80  }
81  else
82  {
83  ui->rb_N->setChecked(true);
84  }
85 
86  const CLongitude lng = coordinate.longitude();
87  const QString lngWgs = lng.toWgs84();
88  ui->le_Longitude->setText(lngWgs);
89  ui->le_Longitude->setToolTip(QString::number(lng.value(CAngleUnit::deg())));
90  ui->lblp_LngCheck->setTicked(!lng.isNull());
91  if (lngWgs.contains('W'))
92  {
93  ui->rb_W->setChecked(true);
94  }
95  else
96  {
97  ui->rb_E->setChecked(true);
98  }
99 
100  const CLatitude::DegMinSecFractionalSec latParts = lat.asSexagesimalDegMinSec(true);
101  ui->le_LatDeg->setText(latParts.degAsString());
102  ui->le_LatMin->setText(latParts.minAsString());
103  ui->le_LatSec->setText(latParts.secAsString());
104  ui->le_LatSecFrag->setText(latParts.fractionalSecAsString(6));
105 
106  const CLongitude::DegMinSecFractionalSec lngParts = lng.asSexagesimalDegMinSec(true);
107  ui->le_LngDeg->setText(lngParts.degAsString());
108  ui->le_LngMin->setText(lngParts.minAsString());
109  ui->le_LngSec->setText(lngParts.secAsString());
110  ui->le_LngSecFrag->setText(lngParts.fractionalSecAsString(6));
111 
112  const QString elvString = coordinate.geodeticHeightAsString();
113  ui->le_Elevation->setText(elvString);
114  ui->lblp_ElvCheck->setTicked(!elvString.isEmpty());
115 
116  std::array<double, 3> v = coordinate.normalVectorDouble();
117  ui->le_X->setText(QString::number(v[0]));
118  ui->le_Y->setText(QString::number(v[1]));
119  ui->le_Z->setText(QString::number(v[2]));
120  ui->le_X->home(false);
121  ui->le_Y->home(false);
122  ui->le_Z->home(false);
123 
124  return true;
125  }

174  void CCoordinateForm::locationEntered()
175  {
176  const QString l = ui->le_Location->text().trimmed().simplified().toUpper();
177 
178  // location based on swift data
179  if (sApp && sApp->hasWebDataServices())
180  {
181  CAirport airport;
182 
183  // airport ICAO
184  if (l.length() == 4 && sApp->getWebDataServices()->getAirportsCount() > 0)
185  {
186  airport = sApp->getWebDataServices()->getAirportForIcaoDesignator(l);
187  if (airport.hasValidDbKey())
188  {
189  this->setCoordinate(airport);
190  return;
191  }
192  }
193 
194  airport = sApp->getWebDataServices()->getAirportForNameOrLocation(l);
195  if (airport.hasValidDbKey())
196  {
197  this->setCoordinate(airport);
198  return;
199  }
200  }
201  // 33°59′42″S 150°57′06″E
202  if (l.contains(' '))
203  {
204  QString lat, lng;
205  const QStringList parts = l.split(' ');
206  for (const QString &p : parts)
207  {
208  if (p.contains('S') || p.contains('N'))
209  {
210  lat = p;
211  }
212  else if (p.contains('E') || p.contains('W'))
213  {
214  lng = p;
215  }
216  }
217  if (!lat.isEmpty() && !lng.isEmpty())
218  {
219  CCoordinateGeodetic c = m_coordinate;
220  c.setLatLongFromWgs84(lat, lng);
221  this->setCoordinate(c);
222  }
223  }
224  }

236  void CCoordinateForm::latCombinedEntered()
237  {
238  bool ok;
239  int deg = ui->le_LatDeg->text().trimmed().toInt(&ok);
240  if (!ok) return;
241 
242  int min = ui->le_LatMin->text().trimmed().toInt(&ok);
243  if (!ok) return;
244 
245  const QString secStr = ui->le_LatSec->text().trimmed() + "." + ui->le_LatSecFrag->text().trimmed();
246  double sec = secStr.toDouble(&ok);
247  if (!ok) return;
248 
249  CAngle::unifySign(deg, min, sec);
250  const CAngle a(deg, min, sec);
251  CLatitude lat(a);
252  lat.roundToEpsilon();
253  CCoordinateGeodetic c = m_coordinate;
254  c.setLatitude(lat);
255  this->setCoordinate(c);
256  }
