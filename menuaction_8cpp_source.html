128  void CMenuActions::splitSubMenus(const QString &key, QList<CMenuAction> &actions, QList<CMenuAction> &menus) const
129  {
130  QList<CMenuAction> myActions(m_actions.values(key));
131  QList<CMenuAction> checkableActions;
132  std::reverse(myActions.begin(), myActions.end()); // the order is reverse because of the insert multi value
133  for (const CMenuAction &action : myActions)
134  {
135  if (action.isSeparator())
136  {
137  actions.append(action);
138  }
139  else if (action.isSubMenu() || !action.getQAction())
140  {
141  menus.append(action);
142  }
143  else
144  {
145  if (action.isCheckableQAction())
146  {
147  checkableActions.append(action);
148  }
149  else
150  {
151  actions.append(action);
152  }
153  }
154  }
155  actions.append(checkableActions); // checkable actions at end
156  }

335  void CMenuActions::toQMenu(QMenu &menu, bool separateGroups) const
336  {
337  if (m_actions.isEmpty()) { return; }
338  const QStringList keys = m_actions.uniqueKeys(); // Sorted ascending keys, we need a menu for all those keys
339 
340  QMap<QString, QMenu *> subMenus; // all sub menus
341  QString lastKey;
342 
343  for (const QString &key : keys)
344  {
345  bool handledSeparator = false;
346  const int pathDepth = CMenuActions::pathDepth(key); // 0 based
347 
348  QList<CMenuAction> actions;
349  QList<CMenuAction> menus;
350  this->splitSubMenus(key, actions, menus); // splits actions and (sub) menus for that key
351  if (actions.isEmpty())
352  {
353  // No actions directly for that level
354  continue;
355  }
356  if (!menu.isEmpty() && separateGroups)
357  {
358  // no separator a) if there is already one b) key roots are the same (such entries belong together)
359  const bool noSeparator = (!menu.actions().isEmpty() && menu.actions().last()->isSeparator()) ||
360  (isSameKeyRoot(key, lastKey));
361  if (!noSeparator) { menu.addSeparator(); }
362  handledSeparator = true;
363  }
364 
365  int noActionsWithoutPath = 0;
366  QMenu *currentMenu = nullptr;
367 
368  // reverse iteration because same key values are inserted and have reverse order
369  for (const CMenuAction &menuAction : std::as_const(actions))
370  {
371  // create submenu if required
372  if (!currentMenu)
373  {
374  currentMenu = currentMenuForAction(menu, menuAction, menus, subMenus, key, pathDepth);
375  }
376  Q_ASSERT_X(currentMenu, Q_FUNC_INFO, "Missing menu");
377  Q_ASSERT_X(menuAction.isSubMenu() || menuAction.isSeparator() || menuAction.getQAction(), Q_FUNC_INFO, "Wrong type");
378 
379  if (menuAction.isSeparator())
380  {
381  if (menu.isEmpty()) continue;
382  currentMenu->addSeparator();
383  continue;
384  }
385 
386  if (menuAction.hasNoPath())
387  {
388  noActionsWithoutPath++;
389 
390  // separator for unclassfied items
391  if ((!handledSeparator || noActionsWithoutPath > 1) && menuAction.hasNoPathWithSeparator())
392  {
393  menu.addSeparator();
394  handledSeparator = false;
395  }
396  }
397 
398  // either add to submenu or menu
399  currentMenu->addAction(menuAction.getQAction());
400 
401  // menu has ownership if there is no other parent
402  if (menuAction.getQAction() && !menuAction.getQAction()->parent())
403  {
404  menuAction.getQAction()->setParent(&menu);
405  }
406  } // actions
407 
408  // clean up empty sub menus
409  if (currentMenu && currentMenu->isEmpty())
410  {
411  menu.removeAction(currentMenu->menuAction());
412  }
413 
414  // remember last key
415  lastKey = key;
416 
417  } // keys
418  }

504  QMenu *CMenuActions::currentMenuForAction(QMenu &menu, const CMenuAction &menuAction, const QList<CMenuAction> &menus, QMap<QString, QMenu *> &subMenus, const QString &key, int pathDepth)
505  {
506  if (pathDepth < 1) { return &menu; }
507  QMenu *parentMenu = &menu;
508  if (pathDepth > 1)
509  {
510  // find the corresponding submenu. If this is empty the next higher level will be choosen
511  // if not found at all, use top level menu
512  parentMenu = findUpwardsInMenus(key, subMenus);
513  if (!parentMenu)
514  {
515  parentMenu = &menu;
516  }
517  }
518 
519  // explicity menu?
520  QMenu *subMenu = nullptr;
521  if (menus.isEmpty())
522  {
523  subMenu = parentMenu->addMenu(menuAction.getLastPathPart());
524  }
525  else
526  {
527  const CMenuAction menuFound(menus.first());
528  subMenu = parentMenu->addMenu(menuFound.getIcon(), menuFound.getTitle());
529  }
530  Q_ASSERT_X(subMenu, Q_FUNC_INFO, "Could not create sub menu");
531 
532  if (subMenu)
533  {
534  subMenu->setParent(parentMenu);
535  }
536 
537  if (pathDepth > 0 && subMenu)
538  {
539  subMenus.insert(key, subMenu);
540  }
541  return subMenu;
542  }

574  QMenu *CMenuActions::findUpwardsInMenus(const QString &key, const QMap<QString, QMenu *> &menus)
575  {
