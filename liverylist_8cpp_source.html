26  CLivery CLiveryList::findStdLiveryByAirlineIcaoVDesignator(const CAirlineIcaoCode &icao) const
27  {
28  if (this->isEmpty() || !icao.hasValidDesignator()) { return CLivery(); }
29  CLiveryList candidates;
30  for (const CLivery &livery : *this)
31  {
32  if (!livery.isAirlineStandardLivery()) { continue; }
33  const CAirlineIcaoCode livIcao = livery.getAirlineIcaoCode();
34  if (livIcao.isDbEqual(icao)) { return livery; }
35  if (livIcao.getVDesignator() != icao.getVDesignator()) { continue; }
36  if (icao.getName().size() > 5 && livery.getDescription().contains(icao.getName(), Qt::CaseInsensitive)) { return livery; }
37  candidates.push_back(livery);
38  }
39 
40  if (candidates.size() < 2) { return candidates.frontOrDefault(); }
41  const CLiveryList operatingAirlines = candidates.findBy(&CLivery::isAirlineOperating, true);
42  if (!operatingAirlines.isEmpty()) { return operatingAirlines.frontOrDefault(); }
43  return candidates.frontOrDefault();
44  }

85  CLivery CLiveryList::findClosestColorLiveryOrDefault(const CRgbColor &fuselage, const CRgbColor &tail) const
86  {
87  if (!fuselage.isValid() || !tail.isValid()) { return CLivery(); }
88  CLivery bestMatch;
89  double bestDistance = 1.0;
90  for (const CLivery &livery : *this)
91  {
92  if (!livery.isColorLivery()) { continue; }
93  const double d = livery.getColorDistance(fuselage, tail);
94  if (d == 0.0) { return livery; } // exact match
95  if (d < bestDistance)
96  {
97  bestMatch = livery;
98  bestDistance = d;
99  }
100  }
101  return bestMatch;
102  }

146  CLivery CLiveryList::smartLiverySelector(const CLivery &liveryPattern) const
147  {
148  // multiple searches are slow, maybe we can performance optimize this
149  // in the future
150 
151  // first try on id, that would be perfect
152  if (liveryPattern.hasValidDbKey()) { return liveryPattern; }
153 
154  // by combined code
155  if (liveryPattern.hasCombinedCode())
156  {
157  const QString cc(liveryPattern.getCombinedCode());
158  const CLivery l(this->findByCombinedCode(cc));
159  if (l.hasCompleteData()) { return l; }
160  }
161 
162  // by airline
163  if (liveryPattern.hasValidAirlineDesignator())
164  {
165  const QString icao(liveryPattern.getAirlineIcaoCodeDesignator());
166  const CLivery l(this->findStdLiveryByAirlineIcaoVDesignator(icao));
167  if (l.hasCompleteData()) { return l; }
168 
169  // lenient search by assuming that a virtual airline is not annotated by "V"
170  // VHDU not found, but HDU
171  const CLiveryList liveries = this->findByAirlineIcaoDesignator(icao);
172  if (liveries.size() == 1) { return liveries.front(); }
173  if (liveries.size() > 1)
174  {
175  if (liveryPattern.hasAirlineName())
176  {
177  // reduce by name
178  const CLiveryList liveriesByName = liveries.findStdLiveriesBySimplifiedAirlineName(liveryPattern.getAirlineName());
179  if (!liveriesByName.isEmpty()) { return liveriesByName.front(); }
180  }
181  return liveries.front();
182  }
183  }
184 
185  // lenient search by name contained (slow)
186  if (liveryPattern.getAirlineIcaoCode().hasName())
187  {
188  const QString search(liveryPattern.getAirlineIcaoCode().getSimplifiedName());
189  const CLiveryList liveries(this->findStdLiveriesByNamesOrTelephonyDesignator(search));
190  if (!liveries.isEmpty())
191  {
192  return liveries.front();
193  }
194  }
195  return CLivery();
196  }
