37  CAircraftSituationChange::CAircraftSituationChange(const CAircraftSituationList &situations, const PhysicalQuantities::CLength &cg, bool isVtol, bool alreadySortedLatestFirst, bool calcStdDeviations)
38  {
39  if (situations.size() < 2) { return; }
40  const CAircraftSituationList sorted(alreadySortedLatestFirst ? situations : situations.getSortedAdjustedLatestFirst());
41 
42  if (CBuildConfig::isLocalDeveloperDebugBuild())
43  {
44  Q_ASSERT_X(sorted.isSortedAdjustedLatestFirstWithoutNullPositions(), Q_FUNC_INFO, "Wrong sort order or NULL position");
45  }
46 
47  const CAircraftSituation latest(sorted.front());
48  const CAircraftSituation oldest(sorted.back());
49  const CCallsign cs(latest.getCallsign());
50 
51  m_situationsCount = situations.size();
52  m_correspondingCallsign = cs;
53  m_timestampMSecsSinceEpoch = latest.getMSecsSinceEpoch();
54  m_timeOffsetMs = latest.getTimeOffsetMs();
55  m_oldestTimestampMSecsSinceEpoch = oldest.getMSecsSinceEpoch();
56  m_latestAdjustedTimestampMSecsSinceEpoch = latest.getAdjustedMSecsSinceEpoch();
57  m_oldestAdjustedTimestampMSecsSinceEpoch = oldest.getAdjustedMSecsSinceEpoch();
58  m_constAscending = sorted.isConstAscending(true);
59  m_constDescending = sorted.isConstDescending(true);
60  m_constOnGround = sorted.isConstOnGround();
61  m_constNotOnGround = sorted.isConstNotOnGround();
62  m_justTakeoff = sorted.isJustTakingOff(true);
63  m_justTouchdown = sorted.isJustTouchingDown(true);
64  m_constAccelerating = sorted.isConstAccelerating(true);
65  m_constDecelerating = sorted.isConstDecelarating(true);
66  m_containsPushBack = !isVtol && sorted.containsPushBack();
67 
68  if (sorted.size() >= 3)
69  {
70  const CAircraftSituationList sortedWithout = sorted.withoutFrontSituation();
71  m_wasNotOnGround = sortedWithout.isConstNotOnGround();
72  m_wasOnGround = sortedWithout.isConstOnGround();
73  }
74 
75  if (calcStdDeviations)
76  {
77  this->calculateStdDeviations(situations, cg);
78  m_rotateUp = sorted.front().getPitch() > (m_pitchMean + m_pitchStdDev);
79  }
80  else
81  {
82  m_rotateUp = sorted.isRotatingUp(true);
83  }
84  }

86  bool CAircraftSituationChange::guessOnGround(CAircraftSituation &situation, const Simulation::CAircraftModel &model) const
87  {
88  if (!situation.shouldGuessOnGround()) { return false; }
89 
90  // for debugging purposed
91  QString *details = /*CBuildConfig::isLocalDeveloperDebugBuild() ? &m_onGroundGuessingDetails :*/ nullptr;
92 
93  // Non VTOL aircraft have to move to be not on ground
94  const bool vtol = model.isVtol();
95  if (!vtol)
96  {
97  if (situation.getGroundSpeed().isNegativeWithEpsilonConsidered())
98  {
99  situation.setOnGround(CAircraftSituation::OnGround, CAircraftSituation::OnGroundByGuessing);
100  if (details) { *details = QStringLiteral("No VTOL, push back"); }
101  return true;
102  }
103 
104  if (!situation.isMoving())
105  {
106  situation.setOnGround(CAircraftSituation::OnGround, CAircraftSituation::OnGroundByGuessing);
107  if (details) { *details = QStringLiteral("No VTOL, not moving => on ground"); }
108  return true;
109  }
110  }
111 
112  // not on ground is default
113  situation.setOnGround(CAircraftSituation::NotOnGround, CAircraftSituation::OnGroundByGuessing);
114 
115  CLength cg = situation.hasCG() ? situation.getCG() : model.getCG();
116  CSpeed guessedRotateSpeed = CSpeed::null();
117  CSpeed sureRotateSpeed = CSpeed(130, CSpeedUnit::kts());
118  model.getAircraftIcaoCode().guessModelParameters(cg, guessedRotateSpeed);
119  if (!guessedRotateSpeed.isNull())
120  {
121  // does the value make any sense?
122  const bool validGuessedSpeed = (guessedRotateSpeed.value(CSpeedUnit::km_h()) > 5.0);
123  BLACK_VERIFY_X(validGuessedSpeed, Q_FUNC_INFO, "Wrong guessed value for lift off");
124  if (!validGuessedSpeed) { guessedRotateSpeed = CSpeed(80, CSpeedUnit::kts()); } // fix
125  sureRotateSpeed = guessedRotateSpeed * 1.25;
126  }
127 
128  // "extreme" values for which we are surely not on ground
129  if (qAbs(situation.getPitch().value(CAngleUnit::deg())) > 20)
130  {
131  if (details) { *details = QStringLiteral("max.pitch"); }
132  return true;
133  } // some tail wheel aircraft already have 11° pitch on ground
134  if (qAbs(situation.getBank().value(CAngleUnit::deg())) > 10)
135  {
136  if (details) { *details = QStringLiteral("max.bank"); }
137  return true;
138  }
139  if (situation.getGroundSpeed() > sureRotateSpeed)
140  {
141  if (details) { *details = u"gs. > vr " % sureRotateSpeed.valueRoundedWithUnit(1); }
142  return true;
143  }
144 
145  // use the most accurate or reliable guesses here first
146  // ------------------------------------------------------
147  // by elevation
148  // we can detect "on ground" (underflow, near ground), but not "not on ground" because of overflow
149 
150  // we can detect on ground for underflow, but not for overflow (so we can not rely on NotOnGround)
151  CAircraftSituation::IsOnGround og = situation.isOnGroundByElevation(cg);
152  if (og == CAircraftSituation::OnGround)
153  {
154  if (details) { *details = QStringLiteral("elevation on ground"); }
155  situation.setOnGround(og, CAircraftSituation::OnGroundByGuessing);
156  return true;
157  }
158 
159  if (!isNull())
160  {
161  if (!vtol && wasConstOnGround())
162  {
163  if (isRotatingUp())
164  {
165  // not OG
166  if (details) { *details = QStringLiteral("rotating up detected"); }
167  return true;
168  }
169 
170  // here we stick to ground until we detect rotate up
171  situation.setOnGround(CAircraftSituation::OnGround, CAircraftSituation::OnGroundByGuessing);
172  if (details) { *details = QStringLiteral("waiting for rotating up"); }
173  return true;
174  }
175 
176  if (isConstAscending())
177  {
178  // not OG
179  if (details) { *details = QStringLiteral("const ascending"); }
180  return true;
181  }
182  }
183 
184  // on VTOL we stop here
185  if (vtol)
186  {
187  // no idea
188  situation.setOnGround(CAircraftSituation::OnGroundSituationUnknown, CAircraftSituation::NotSetGroundDetails);
189  return false;
190  }
191 
192  // guessed speed null -> vtol
193  if (!guessedRotateSpeed.isNull())
194  {
195  // does the value make any sense?
196  if (situation.getGroundSpeed() < guessedRotateSpeed)
197  {
198  situation.setOnGround(CAircraftSituation::OnGround, CAircraftSituation::OnGroundByGuessing);
199  if (details) { *details = QStringLiteral("Guessing, max.guessed gs.") + guessedRotateSpeed.valueRoundedWithUnit(CSpeedUnit::kts(), 1); }
200  return true;
201  }
202  }
203 
204  // not sure, but this is a guess
205  if (details) { *details = QStringLiteral("Fall through"); }
206  return true;
207  }

342  bool CAircraftSituationChange::calculateStdDeviations(const CAircraftSituationList &situations, const CLength &cg)
343  {
344  if (situations.isEmpty()) { return false; }
345 
346  const QPair<CSpeed, CSpeed> gsStdDevMean = situations.groundSpeedStandardDeviationAndMean();
347  m_gsStdDev = gsStdDevMean.first;
348  m_gsMean = gsStdDevMean.second;
349 
350  const QPair<CAngle, CAngle> pitchStdDevMean = situations.pitchStandardDeviationAndMean();
351  m_pitchStdDev = pitchStdDevMean.first;
352  m_pitchMean = pitchStdDevMean.second;
353 
354  const QList<double> altValues = situations.altitudeValues(CAltitude::defaultUnit());
355  if (altValues.size() == situations.size())
356  {
357  const QPair<double, double> altDevMean = CMathUtils::standardDeviationAndMean(altValues);
358  m_altStdDev = CAltitude(altDevMean.first, CAltitude::MeanSeaLevel, CAltitude::defaultUnit());
359  m_altMean = CAltitude(altDevMean.second, CAltitude::MeanSeaLevel, CAltitude::defaultUnit());
360  }
361 
362  const QList<double> elvValues = situations.elevationValues(CAltitude::defaultUnit());
363  if (elvValues.size() == situations.size())
364  {
365  const QPair<double, double> elvDevMean = CMathUtils::standardDeviationAndMean(elvValues);
366  m_elvStdDev = CAltitude(elvDevMean.first, CAltitude::MeanSeaLevel, CAltitude::defaultUnit());
367  m_elvMean = CAltitude(elvDevMean.second, CAltitude::MeanSeaLevel, CAltitude::defaultUnit());
368 
369  if (altValues.size() == situations.size())
370  {
371  QList<double> gndDistance;
372  for (int i = 0; i < altValues.size(); i++)
373  {
374  const double delta = altValues[i] - elvValues[i];
375  gndDistance.push_back(delta);
376  }
377  const QPair<double, double> gndDistanceDevMean = CMathUtils::standardDeviationAndMean(gndDistance);
378  m_gndDistStdDev = CLength(gndDistanceDevMean.first, CAltitude::defaultUnit());
379  m_gndDistMean = CLength(gndDistanceDevMean.second, CAltitude::defaultUnit());
380 
381  const auto gndDistMinMax = std::minmax_element(gndDistance.constBegin(), gndDistance.constEnd());
382  const double gndDistMin = *gndDistMinMax.first;
383  const double gndDistMax = *gndDistMinMax.second;
384  m_minGroundDistance = CLength(gndDistMin, CAltitude::defaultUnit());
385  m_maxGroundDistance = CLength(gndDistMax, CAltitude::defaultUnit());
386  this->guessSceneryDeviation(cg);
387  }
388  }
389  return true;
390  }

423  void CAircraftSituationChange::guessSceneryDeviation(const CLength &cg)
424  {
425  m_guessedSceneryDeviation = CLength::null();
426  this->setSceneryDeviationHint(NoDeviationInfo);
427  if (m_gndDistStdDev.isNull()) { return; }
428  if (m_gndDistMean.isNull()) { return; }
429 
430  // only for a small deviation we can calculate scenery differemce
431  static const CLength maxDeviation(2, CLengthUnit::ft());
432 
433  // Small deviation means "const" AGL
434  if (m_gndDistStdDev <= maxDeviation)
435  {
436  do
437  {
438  if (this->isConstOnGround())
439  {
440  this->setSceneryDeviation(m_gndDistMean, cg, AllOnGround);
441  break;
442  }
443  if (this->wasConstNotOnGround())
444  {
445  this->setSceneryDeviation(m_gndDistMean, cg, WasOnGround);
446  break;
447  }
448  if (!m_altStdDev.isNull() && m_altStdDev <= maxDeviation)
449  {
450  // small alt.deviation too!
451  if (!m_maxGroundDistance.isNull() && m_maxGroundDistance < cg)
452  {
453  if (this->isConstOnGround())
454  {
455  this->setSceneryDeviation(m_gndDistMean, cg, SmallAGLDeviationNearGround);
456  break;
457  }
458  }
459  }
460  }
461  while (false);
462  }
