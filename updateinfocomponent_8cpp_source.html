53  bool CUpdateInfoComponent::isNewPilotClientVersionAvailable() const
54  {
55  const QStringList settings = m_updateSettings.get();
56  Q_ASSERT_X(settings.size() == 2, Q_FUNC_INFO, "wrong setting");
57  const QString channel = settings.front();
58  if (channel.isEmpty()) { return false; }
59 
60  const CUpdateInfo info = m_updateInfo.get();
61  const CDistribution d = info.getDistributions().findFirstByChannelOrDefault(channel);
62  const QVersionNumber vCurrentChannelPlatform = info.getArtifactsPilotClient().findByDistribution(d).getLatestQVersion();
63  const QVersionNumber vCurrent = CBuildConfig::getVersion();
64  if (vCurrentChannelPlatform.isNull()) { return false; }
65  return (vCurrentChannelPlatform > vCurrent);
66  }

83  void CUpdateInfoComponent::changedUpdateInfo()
84  {
85  ui->lbl_CurrentVersionDisplay->setText(CBuildConfig::getVersionString());
86 
87  const CUpdateInfo updateInfo(m_updateInfo.get());
88  const QStringList settings = m_updateSettings.get();
89  Q_ASSERT_X(settings.size() == 2, Q_FUNC_INFO, "Settings");
90 
91  const CPlatformSet platforms = updateInfo.getArtifactsPilotClient().getPlatforms();
92  CDistributionList distributions = updateInfo.getArtifactsPilotClient().getDistributions();
93  distributions.sortByStability(Qt::DescendingOrder);
94 
95  int i = 0;
96  ui->cb_Platforms->disconnect();
97  ui->cb_Platforms->clear();
98  for (const CPlatform &platform : platforms)
99  {
100  ui->cb_Platforms->insertItem(i++, CIcon(platform.toIcon()).toPixmap(), platform.getPlatformName());
101  }
102  if (platforms.contains(settings.last())) { ui->cb_Platforms->setCurrentText(settings.last()); }
103  else if (platforms.contains(CPlatform::currentPlatform().getPlatformName())) { ui->cb_Platforms->setCurrentText(CPlatform::currentPlatform().getPlatformName()); }
104 
105  i = 0;
106  ui->cb_Channels->disconnect();
107  ui->cb_Channels->clear();
108  for (const CDistribution &distribution : distributions)
109  {
110  ui->cb_Channels->insertItem(i++, CIcon(distribution.getRestrictionIcon()).toPixmap(), distribution.getChannel());
111  }
112  if (distributions.containsChannel(settings.front())) { ui->cb_Channels->setCurrentText(settings.front()); }
113 
114  this->uiSelectionChanged();
115  connect(ui->cb_Platforms, &QComboBox::currentTextChanged, this, &CUpdateInfoComponent::platformChanged, Qt::QueuedConnection);
116  connect(ui->cb_Channels, &QComboBox::currentTextChanged, this, &CUpdateInfoComponent::channelChanged, Qt::QueuedConnection);
117 
118  // emit via digest signal
119  m_dsDistributionAvailable.inputSignal();
120  }

144  void CUpdateInfoComponent::downloadInstallerDialog()
145  {
146  const CUpdateInfo update(m_updateInfo.get());
147  const QString currentVersion = ui->cb_ArtifactsPilotClient->currentText();
148  const QString platform = ui->cb_Platforms->currentText();
149  if (!CPlatform::isCurrentPlatform(platform))
150  {
151  const QMessageBox::StandardButton ret = QMessageBox::warning(this, "Download installer",
152  QStringLiteral(
153  "The platform '%1' does not match your current platform '%2'.\n"
154  "Do you want to continue?")
155  .arg(platform, CPlatform::currentPlatform().getPlatformName()),
156  QMessageBox::Yes | QMessageBox::No);
157  if (ret != QMessageBox::Yes) { return; }
158  }
159 
160  // find artifcat
161  const CArtifact artifact = update.getArtifactsPilotClient().findByMatchingPlatform(platform).findFirstByVersionOrDefault(currentVersion);
162 
163  if (!m_downloadDialog)
164  {
165  m_downloadDialog.reset(new CDownloadDialog(this));
166  m_downloadDialog->setModal(true);
167  }
168 
169  const CRemoteFile rf = artifact.asRemoteFile();
170  if (rf.getUrl().isHavingHtmlSuffix())
171  {
172  QDesktopServices::openUrl(rf.getUrl());
173  }
174  else
175  {
176  m_downloadDialog->setMode(CDownloadComponent::SwiftInstaller);
177  m_downloadDialog->setDownloadFile(artifact.asRemoteFile());
178  m_downloadDialog->showAndStartDownloading();
179  }

204  void CUpdateInfoComponent::uiSelectionChanged()
205  {
206  const CDistribution selectedDistribution(this->getSelectedOrDefaultDistribution());
207  const CPlatform selectedPlatform(this->getSelectedOrDefaultPlatform());
208 
209  // for XSwiftBus we only show public (unrestricted) ones, as the follow up dialog will only show unrestricted
210  const CUpdateInfo updateInfo(m_updateInfo.get());
211  const CArtifactList artifactsPilotClient = updateInfo.getArtifactsPilotClient().findByDistributionAndPlatform(selectedDistribution, selectedPlatform, true);
212  const CArtifactList artifactsXsb = updateInfo.getArtifactsXSwiftBus().findWithUnrestrictedDistributions().findByDistributionAndPlatform(selectedDistribution, selectedPlatform, true);
213  const CArtifact latestPilotClient = artifactsPilotClient.getLatestArtifactOrDefault();
214 
215  const QStringList sortedPilotClientVersions = artifactsPilotClient.getSortedVersions();
216  ui->cb_ArtifactsPilotClient->clear();
217  ui->cb_ArtifactsPilotClient->insertItems(0, sortedPilotClientVersions);
218  ui->pb_DownloadInstaller->setEnabled(!artifactsPilotClient.isEmpty());
219 
220  const QStringList sortedXsbVersions = artifactsXsb.getSortedVersions();
221  ui->cb_ArtifactsXsb->clear();
222  ui->cb_ArtifactsXsb->insertItems(0, sortedXsbVersions);
223  ui->pb_DownloadXSwiftBus->setEnabled(!artifactsXsb.isEmpty());
224 
225  // save the settings as this is needed afterwards
226  this->saveSettings();
227 
229  const bool newer = this->isNewPilotClientVersionAvailable();
230  ui->lbl_StatusInfo->setText(newer ? "New version available" : "Nothing new");
231  ui->lbl_StatusInfo->setStyleSheet(newer ?
232  "background-color: green; color: white;" :
233  "background-color: red; color: white;");
234 
235  emit this->selectionChanged();
236 
237  if (newer && latestPilotClient.isLoadedFromDb())
238  {
239  emit this->newerPilotClientAvailable(latestPilotClient);
240  }
241  }
