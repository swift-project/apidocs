22  void CNotificationPlayer::play(CNotificationSounds::NotificationFlag notification, int volume)
23  {
24  QSoundEffect *effect = m_effects.value(notification, nullptr);
25  if (effect)
26  {
27  if (!m_playingEffect && effect->isLoaded() && !effect->isPlaying())
28  {
29  const int currentEffect = ++m_playingEffectCounter;
30  const qreal v = volume / 100.0f;
31  m_playingEffect = effect;
32  effect->setVolume(v); // 0..1
33  effect->play();
34 
41  // used for too long or hanging sounds
42  QPointer<CNotificationPlayer> myself(this);
43  QTimer::singleShot(3000, effect, [=] {
44  if (!myself || !m_playingEffect) { return; }
45  if (currentEffect != m_playingEffectCounter) { return; }
46  m_playingEffect->stop();
47  });
48  }
49  }

82  void CNotificationPlayer::updateEffect(CNotificationSounds::NotificationFlag f, const QString &directory, const QString &name)
83  {
84  QSoundEffect *e = nullptr;
85  if (m_effects.contains(f)) { e = m_effects[f]; }
86  if (e) { e->deleteLater(); }
87 
88  // file if existing
89  const QUrl url = QUrl::fromLocalFile(CSwiftDirectories::soundFilePathOrDefaultPath(directory, name));
90  if (url.isEmpty() || !url.isLocalFile())
91  {
92  // remove notification as not existing
93  m_effects.remove(f);
94  return;
95  }
96 
97  // new effect
98  // QString fn = url.toLocalFile();
99  QSoundEffect *effect = new QSoundEffect(this);

102  effect->setMuted(false);
103  m_effects[f] = effect;
