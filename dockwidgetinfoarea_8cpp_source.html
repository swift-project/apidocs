61  void CDockWidgetInfoArea::addToContextMenu(QMenu *contextMenu) const
62  {
63  QList<const CInfoArea *> parentInfoAreas = this->findParentInfoAreas();
64  Q_ASSERT(!parentInfoAreas.isEmpty());
65  if (parentInfoAreas.isEmpty()) return;
66 
67  // Dockable widget's context menu
68  CDockWidget::addToContextMenu(contextMenu);
69  if (!contextMenu->isEmpty()) { contextMenu->addSeparator(); }
70 
71  // first info area, myself's direct parent info area
72  parentInfoAreas.first()->addToContextMenu(contextMenu);
73 
74  // top info areas other than direct parent
75  // (parent's parent when nested info areas are used)
76  if (parentInfoAreas.size() < 2) { return; }
77  contextMenu->addSeparator();
78  for (int i = 1; i < parentInfoAreas.size(); i++)
79  {
80  const CInfoArea *infoArea = parentInfoAreas.at(i);
81  QString title(infoArea->windowTitle());
82  if (title.isEmpty())
83  {
84  title = infoArea->objectName();
85  }
86  QMenu *m = contextMenu->addMenu(title);
87  infoArea->addToContextMenu(m);
88  }
89  }

104  QList<CEnableForDockWidgetInfoArea *> CDockWidgetInfoArea::findEmbeddedDockWidgetInfoAreaComponents()
105  {
106  QList<QWidget *> widgets = this->findChildren<QWidget *>(); // must not use Qt::FindDirectChildrenOnly here
107  QList<CEnableForDockWidgetInfoArea *> widgetsWithDockWidgetInfoAreaComponent;
108  for (QWidget *w : widgets)
109  {
110  Q_ASSERT(w);
111 
112  // CEnableForDockWidgetInfoArea is no QObject, so we use dynamic_cast
113  CEnableForDockWidgetInfoArea *dwc = dynamic_cast<CEnableForDockWidgetInfoArea *>(w);
114  if (dwc)
115  {
116  widgetsWithDockWidgetInfoAreaComponent.append(dwc);
117  }
118  }
119  QList<CDockWidgetInfoArea *> nestedInfoAreas = this->findNestedInfoAreas();
120  if (nestedInfoAreas.isEmpty()) return widgetsWithDockWidgetInfoAreaComponent;
121 
122  // we have to exclude the nested embedded areas
123  for (CDockWidgetInfoArea *ia : nestedInfoAreas)
124  {
125  QList<CEnableForDockWidgetInfoArea *> nestedInfoAreaComponents = ia->findEmbeddedDockWidgetInfoAreaComponents();
126  if (nestedInfoAreaComponents.isEmpty()) { continue; }
127  for (CEnableForDockWidgetInfoArea *iac : nestedInfoAreaComponents)
128  {
129  const bool r = widgetsWithDockWidgetInfoAreaComponent.removeOne(iac);
130  Q_ASSERT(r); // why is the nested component not in the child list?
131  Q_UNUSED(r)
132  }
133  }
134  return widgetsWithDockWidgetInfoAreaComponent;
135  }

144  QList<const CInfoArea *> CDockWidgetInfoArea::findParentInfoAreas() const
145  {
146  QList<const CInfoArea *> parents;
147  QWidget *currentWidget = this->parentWidget();
148  while (currentWidget)
149  {
150  const CInfoArea *ia = qobject_cast<CInfoArea *>(currentWidget);

152  currentWidget = currentWidget->parentWidget();
