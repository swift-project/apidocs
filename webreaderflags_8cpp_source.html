10  CWebReaderFlags::WebReader CWebReaderFlags::entitiesToReaders(CEntityFlags::Entity entities)
11  {
12  WebReader f = None;
13  if (
14  entities.testFlag(CEntityFlags::AircraftIcaoEntity) || entities.testFlag(CEntityFlags::AirlineIcaoEntity) ||
15  entities.testFlag(CEntityFlags::CountryEntity) || entities.testFlag(CEntityFlags::AircraftCategoryEntity))
16  {
17  f |= IcaoDataReader;
18  }
19 
20  if (entities.testFlag(CEntityFlags::ModelEntity) || entities.testFlag(CEntityFlags::DistributorEntity) || entities.testFlag(CEntityFlags::LiveryEntity))
21  {
22  f |= ModelReader;
23  }
24 
25  if (entities.testFlag(CEntityFlags::AirportEntity))
26  {
27  f |= AirportReader;
28  }
29 
30  if (entities.testFlag(CEntityFlags::DbInfoObjectEntity)) { f |= DbInfoDataReader; }
31  if (entities.testFlag(CEntityFlags::BookingEntity)) { f |= VatsimBookingReader; }
32  if (entities.testFlag(CEntityFlags::VatsimDataFile)) { f |= VatsimDataReader; }
33  if (entities.testFlag(CEntityFlags::VatsimStatusFile)) { f |= VatsimStatusReader; }
34  if (entities.testFlag(CEntityFlags::MetarEntity)) { f |= VatsimMetarReader; }
35 
36  return f;
37  }

44  CEntityFlags::Entity CWebReaderFlags::allEntitiesForReaders(WebReader readers)
45  {
46  CEntityFlags::Entity entities = CEntityFlags::NoEntity;
47  if (readers.testFlag(IcaoDataReader)) { entities |= CEntityFlags::AllIcaoCountriesCategory; }
48  if (readers.testFlag(ModelReader)) { entities |= CEntityFlags::DistributorLiveryModel; }
49  if (readers.testFlag(AirportReader)) { entities |= CEntityFlags::AirportEntity; }
50  if (readers.testFlag(DbInfoDataReader)) { entities |= CEntityFlags::DbInfoObjectEntity; }
51  if (readers.testFlag(VatsimBookingReader)) { entities |= CEntityFlags::BookingEntity; }
52  if (readers.testFlag(VatsimMetarReader)) { entities |= CEntityFlags::MetarEntity; }
53  if (readers.testFlag(VatsimDataReader)) { entities |= CEntityFlags::VatsimDataFile; }
54  if (readers.testFlag(VatsimStatusReader)) { entities |= CEntityFlags::VatsimStatusFile; }
