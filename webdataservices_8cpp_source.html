68  if (!readFromSwiftDb && readers.testFlag(CWebReaderFlags::DbInfoDataReader))

246  CEntityFlags::Entity CWebDataServices::triggerRead(CEntityFlags::Entity whatToRead, const QDateTime &newerThan)
247  {
248  if (m_shuttingDown) { return CEntityFlags::NoEntity; }
249 
250  m_initialRead = true; // read started
251  Q_ASSERT_X(!whatToRead.testFlag(CEntityFlags::DbInfoObjectEntity), Q_FUNC_INFO, "Info object must be read upfront");
252  CEntityFlags::Entity triggeredRead = CEntityFlags::NoEntity;
253  if (m_vatsimDataFileReader)
254  {
255  if (whatToRead.testFlag(CEntityFlags::VatsimDataFile))
256  {
257  m_vatsimDataFileReader->readInBackgroundThread();
258  triggeredRead |= CEntityFlags::VatsimDataFile;
259  }
260  }
261 
262  if (m_vatsimBookingReader)
263  {
264  if (whatToRead.testFlag(CEntityFlags::BookingEntity))
265  {
266  m_vatsimBookingReader->readInBackgroundThread();
267  triggeredRead |= CEntityFlags::BookingEntity;
268  }
269  }
270 
271  if (m_vatsimMetarReader)
272  {
273  if (whatToRead.testFlag(CEntityFlags::MetarEntity))
274  {
275  m_vatsimMetarReader->readInBackgroundThread();
276  triggeredRead |= CEntityFlags::MetarEntity;
277  }
278  }
279 
280  if (m_airportDataReader)
281  {
282  if (whatToRead.testFlag(CEntityFlags::AirportEntity))
283  {
284  CEntityFlags::Entity airportEntities = whatToRead & CEntityFlags::AirportEntity;
285  m_airportDataReader->readInBackgroundThread(airportEntities, newerThan);
286  triggeredRead |= CEntityFlags::AirportEntity;
287  }
288  }
289 
290  if (m_icaoDataReader)
291  {
292  if (whatToRead.testFlag(CEntityFlags::AircraftIcaoEntity) || whatToRead.testFlag(CEntityFlags::AircraftCategoryEntity) || whatToRead.testFlag(CEntityFlags::AirlineIcaoEntity) || whatToRead.testFlag(CEntityFlags::CountryEntity))
293  {
294  CEntityFlags::Entity icaoEntities = whatToRead & CEntityFlags::AllIcaoCountriesCategory;
295  m_icaoDataReader->readInBackgroundThread(icaoEntities, newerThan);
296  triggeredRead |= icaoEntities;
297  }
298  }
299 
300  if (m_modelDataReader)
301  {
302  if (whatToRead.testFlag(CEntityFlags::LiveryEntity) || whatToRead.testFlag(CEntityFlags::DistributorEntity) || whatToRead.testFlag(CEntityFlags::ModelEntity))
303  {
304  CEntityFlags::Entity modelEntities = whatToRead & CEntityFlags::DistributorLiveryModel;
305  m_modelDataReader->readInBackgroundThread(modelEntities, newerThan);
306  triggeredRead |= modelEntities;
307  }
308  }
309 
310  return triggeredRead;
311  }

900  CStatusMessageList CWebDataServices::validateForPublishing(const CAircraftModelList &modelsToBePublished, bool ignoreEqual, CAircraftModelList &validModels, CAircraftModelList &invalidModels) const
901  {
902  CStatusMessageList msgs(modelsToBePublished.validateForPublishing(validModels, invalidModels)); // technical validation
903 
904  // check against existing distributors
905  const CDistributorList distributors(this->getDistributors());
906  if (!distributors.isEmpty())
907  {
908  // only further check the valid ones
909  CAircraftModelList newValidModels;
910  const CStatusMessageList msgsDistributors(validModels.validateDistributors(distributors, newValidModels, invalidModels));
911  validModels = newValidModels;
912  msgs.push_back(msgsDistributors);
913  }
914 
915  // check if model is changed
916  // in case of not ignoreEqual we just check create the messages
917  {
918  CAircraftModelList newValidModels;
919  for (const CAircraftModel &publishModel : validModels)
920  {
921  CStatusMessageList equalMessages;
922  const bool changed = !this->isDbModelEqualForPublishing(publishModel, &equalMessages);
923  if (changed)
924  {
925  // all good
926  newValidModels.push_back(publishModel);
927  continue;
928  }
929  if (ignoreEqual) { equalMessages.warningToError(); }
930  msgs.push_back(CStatusMessage(this, ignoreEqual ? CStatusMessage::SeverityError : CStatusMessage::SeverityWarning, u"Model: '%1', there is no change") << publishModel.getModelString());
931  if (ignoreEqual)
932  {
933  invalidModels.push_back(publishModel);
934  }
935  else
936  {
937  newValidModels.push_back(publishModel);
938  }
939  }
940  validModels = newValidModels;
941  }
942  return msgs;
943  }

1049  {
1050  Q_ASSERT_X(CThreadUtils::thisIsMainThread(), Q_FUNC_INFO, "shall run in main application thread");
1051 
1052  //
1053  // ---- "metadata" reader, 1 will trigger read directly during init
1054  //
1055  CDatabaseReaderConfigList dbReaderConfig(m_dbReaderConfig);
1056  const CEntityFlags::Entity dbEntities = entities & CEntityFlags::AllDbEntitiesNoInfoObjects;
1057  const bool anyDbEntities = CEntityFlags::anySwiftDbEntity(dbEntities); // contains any DB entities
1058  const bool needsSharedInfoObjects = dbReaderConfig.needsSharedInfoObjects(dbEntities);
1059  const bool needsDbInfoObjects = dbReaderConfig.possiblyReadsFromSwiftDb();
1060  bool c = false; // for signal connect
1061  const Qt::ConnectionType typeReaderReadSignals = Qt::QueuedConnection; // Decouple processing of the reader signals from the readers
1062 
1063  // 1a. If any DB data, read the info objects upfront
1064  if (needsDbInfoObjects)
1065  {
1066  const bool databaseUp = sApp->isSwiftDbAccessible();
1067  if (!databaseUp) { dbReaderConfig.markAsDbDown(); }
1068 
1069  if (anyDbEntities && readersNeeded.testFlag(CWebReaderFlags::WebReaderFlag::DbInfoDataReader))
1070  {
1071  // info data reader has a special role, it will not be triggered in triggerRead()
1072  if (databaseUp)
1073  {
1074  this->initDbInfoObjectReaderAndTriggerRead();
1075  }
1076  else
1077  {
1078  CLogMessage(this).warning(u"DB unreachable, skipping read from DB info data reader");
1079  }
1080  }
1081  }
1082 
1083  // 1b. Read info objects if needed
1084  if (needsSharedInfoObjects)
1085  {
1086  this->initSharedInfoObjectReaderAndTriggerRead();
1087  }
1088 
1089  // 2. Status and server file, updating the VATSIM related caches
1090  // Read as soon as initReaders is done
1091  if (readersNeeded.testFlag(CWebReaderFlags::VatsimStatusReader) || readersNeeded.testFlag(CWebReaderFlags::VatsimDataReader) || readersNeeded.testFlag(CWebReaderFlags::VatsimMetarReader))
1092  {
1093  m_vatsimStatusReader = new CVatsimStatusFileReader(this);
1094  c = connect(m_vatsimStatusReader, &CVatsimStatusFileReader::dataFileRead, this, &CWebDataServices::vatsimStatusFileRead, typeReaderReadSignals);
1095  CLogMessage(this).info(u"Trigger read of VATSIM status file");
1096  m_vatsimStatusReader->start(QThread::LowPriority);
1097 
1098  // run single shot in main loop, so readInBackgroundThread is not called before initReaders completes
1099  const QPointer<CWebDataServices> myself(this);
1100  QTimer::singleShot(0, this, [=]() {
1101  if (!myself || m_shuttingDown) { return; }
1102  if (!sApp || sApp->isShuttingDown()) { return; }
1103  m_vatsimStatusReader->readInBackgroundThread();
1104  });
1105 
1106  startVatsimServerFileReader();
1107  }
1108 
1109  // ---- "normal data", triggerRead will start read, not starting directly
1110 
1111  // 3. VATSIM bookings
1112  if (readersNeeded.testFlag(CWebReaderFlags::WebReaderFlag::VatsimBookingReader))
1113  {
1114  m_vatsimBookingReader = new CVatsimBookingReader(this);
1115  c = connect(m_vatsimBookingReader, &CVatsimBookingReader::atcBookingsRead, this, &CWebDataServices::receivedBookings, typeReaderReadSignals);
1116  Q_ASSERT_X(c, Q_FUNC_INFO, "VATSIM booking reader signals");
1117  c = connect(m_vatsimBookingReader, &CVatsimBookingReader::dataRead, this, &CWebDataServices::dataRead, typeReaderReadSignals);
1118  Q_ASSERT_X(c, Q_FUNC_INFO, "connect failed bookings");
1119  m_entitiesPeriodicallyRead |= CEntityFlags::BookingEntity;
1120  m_vatsimBookingReader->start(QThread::LowPriority);
1121  m_vatsimBookingReader->startReader();
1122  }
1123 
1124  // 4. VATSIM data file
1125  if (readersNeeded.testFlag(CWebReaderFlags::WebReaderFlag::VatsimDataReader))
1126  {
1127  m_vatsimDataFileReader = new CVatsimDataFileReader(this);
1128  c = connect(m_vatsimDataFileReader, &CVatsimDataFileReader::dataFileRead, this, &CWebDataServices::vatsimDataFileRead, typeReaderReadSignals);
1129  Q_ASSERT_X(c, Q_FUNC_INFO, "VATSIM data reader signals");
1130  c = connect(m_vatsimDataFileReader, &CVatsimDataFileReader::dataRead, this, &CWebDataServices::dataRead, typeReaderReadSignals);
1131  Q_ASSERT_X(c, Q_FUNC_INFO, "connect failed VATSIM data file");
1132  m_entitiesPeriodicallyRead |= CEntityFlags::VatsimDataFile;
1133  m_vatsimDataFileReader->start(QThread::LowPriority);
1134  m_vatsimDataFileReader->startReader();
1135  }
1136 
1137  // 5. VATSIM METAR data
1138  if (readersNeeded.testFlag(CWebReaderFlags::WebReaderFlag::VatsimMetarReader))
1139  {
1140  m_vatsimMetarReader = new CVatsimMetarReader(this);

1517  bool CWebDataServices::waitForInfoObjectsThenRead(CEntityFlags::Entity entities, const QString &info, CInfoDataReader *infoReader, QDateTime &timeOut)
1518  {
1519  if (m_shuttingDown) { return false; }
1520 
1521  // this will called for each entity readers, i.e. model reader, ICAO reader ...
1522  Q_ASSERT_X(infoReader, Q_FUNC_INFO, "Need info data reader");
1523  const int waitForInfoObjectsMs = 1000; // ms
1524 
1525  if (infoReader->areAllInfoObjectsRead())
1526  {
1527  // we have all data and carry on
1528  CLogMessage(this).info(u"Info objects (%1) triggered for '%2' loaded from '%3'") << info << CEntityFlags::flagToString(entities) << infoReader->getInfoObjectsUrl().toQString();
1529  timeOut = QDateTime(); // reset to null
1530  return true; // no need to wait any longer
1531  }
1532 
1533  // try to read if not timed out
1534  if (timeOut.isValid() && QDateTime::currentDateTimeUtc() > timeOut)
1535  {
1536  const QString timeOutString = timeOut.toString();
1537  const CStatusMessage m = CLogMessage(this).warning(u"Could not read '%1' info objects for '%2' from '%3', time out '%4'. Marking reader '%5' as failed and continue.")
1538  << info << CEntityFlags::flagToString(entities)
1539  << infoReader->getInfoObjectsUrl().toQString() << timeOutString
1540  << infoReader->getName();
1541  emit this->databaseReaderMessages(m);
1542 
1543  // continue here and read data without info objects
1544  infoReader->setMarkedAsFailed(true);
1545  // no timeout reset here
1546  return true; // carry on, regardless of situation
1547  }
1548 
1549  if (infoReader->hasReceivedFirstReply())
1550  {
1551  // we have received a response, but not all data yet
1552  if (infoReader->hasReceivedOkReply())
1553  {
1554  // ok, this means we are parsing
1555  this->readDeferredInBackground(entities, waitForInfoObjectsMs);
1556  const CStatusMessage m = CLogMessage(this).info(u"Parsing objects (%1) for '%2' from '%3'") << info << CEntityFlags::flagToString(entities) << infoReader->getInfoObjectsUrl().toQString();
1557  emit this->databaseReaderMessages(m);
1558  return false; // wait
1559  }
1560  else
1561  {
1562  // we have a response, but a failure, means server is alive, but responded with error
1563  // such an error (access, ...) normally will not go away
1564  const CStatusMessage m = CLogMessage(this).error(u"Info objects (%1) loading for '%2' failed from '%3', '%4'") << info << CEntityFlags::flagToString(entities) << infoReader->getInfoObjectsUrl().toQString() << infoReader->getStatusMessage();
1565  infoReader->setMarkedAsFailed(true);
1566  emit this->databaseReaderMessages(m);
1567  return true; // carry on, regardless of situation
1568  }
1569  }
1570  else
1571  {
1572  // wait for 1st reply
1573  // we call read again in some time
1574  this->readDeferredInBackground(entities, waitForInfoObjectsMs);
1575  return false; // wait
1576  }
