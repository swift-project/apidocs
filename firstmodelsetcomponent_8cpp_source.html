68  void CFirstModelSetComponent::onSimulatorChanged(const CSimulatorInfo &simulator)
69  {
70  if (!simulator.isSingleSimulator())
71  {
73  if (CBuildConfig::isLocalDeveloperDebugBuild()) { BLACK_VERIFY_X(false, Q_FUNC_INFO, "Need single simulator"); }
74  CLogMessage(this).error(u"Changing to non-single simulator %1 ignored") << simulator.toQString();
75  return;
76  }
77 
78  Q_ASSERT_X(m_modelsDialog, Q_FUNC_INFO, "No models dialog");
79  m_modelsDialog->setSimulator(simulator);
80 
81  Q_ASSERT_X(m_modelSetDialog, Q_FUNC_INFO, "No model set dialog");
82  m_modelSetDialog->setSimulator(simulator);
83 
84  // distributor component
85  ui->comp_Distributors->filterBySimulator(simulator);
86 
87  const QStringList dirs = m_simulatorSettings.getModelDirectoriesOrDefault(simulator);
88  ui->le_ModelDirectories->setText(dirs.join(", "));
89 
90  // kind of hack, but simplest solution
91  // we us the loader of the components directly,
92  // avoid to fully init a loader logic here
93  static const QString modelsNo("No models so far");
94  const int modelsCount = this->modelLoader()->getCachedModelsCount(simulator);
95  if (modelsCount > 0)
96  {
97  static const QString modelsInfo("%1 included %2 DB key %3");
98  const CAircraftModelList modelsInCache = this->modelLoader()->getCachedModels(simulator);
99  const int modelsIncluded = modelsInCache.countByMode(CAircraftModel::Include);
100  const int modelsDbKey = modelsInCache.countWithValidDbKey(true);
101  ui->le_ModelsInfo->setText(modelsInfo.arg(this->modelLoader()->getCacheCountAndTimestamp(simulator)).arg(modelsIncluded).arg(modelsDbKey));
102  }
103  else
104  {
105  ui->le_ModelsInfo->setText(modelsNo);
106  }
107 
108  ui->pb_CreateModelSet->setEnabled(modelsCount > 0);
109 
110  static const QString modelsSetNo("Model set is empty");
111  const int modelsSetCount = m_modelSetDialog->modelSetComponent()->getModelSetCount();
112  ui->le_ModelSetInfo->setText(modelsSetCount > 0 ? m_modelSetDialog->modelSetComponent()->getModelCacheCountAndTimestamp() : modelsSetNo);
113  }

160  void CFirstModelSetComponent::openOwnModelsDialog()
161  {
162  if (!m_modelsDialog) { return; }
163  if (!sGui || sGui->isShuttingDown() || !sGui->getWebDataServices()) { return; }
164  const bool reload = (QObject::sender() == ui->pb_ModelsTriggerReload);
165 
166  const CSimulatorInfo simulator = ui->comp_SimulatorSelector->getValue();
167  m_modelsDialog->setSimulator(simulator);
168 
169  if (reload)
170  {
171  if (!sGui->getWebDataServices()->hasDbModelData())
172  {
173  const QMessageBox::StandardButton reply = QMessageBox::warning(this->mainWindow(), "DB data", "No DB data, models cannot be consolidated. Load anyway?", QMessageBox::Yes | QMessageBox::No);
174  if (reply != QMessageBox::Yes) { return; }
175  }
176 
177  bool loadOnlyIfNotEmpty = true;
178  if (m_modelsDialog->getOwnModelsCount() > 0)
179  {
180  const QMessageBox::StandardButton reply = QMessageBox::warning(this->mainWindow(), "Model loading", "Reload the models?\nThe existing cache data will we overridden.", QMessageBox::Yes | QMessageBox::No);
181  if (reply == QMessageBox::Yes) { loadOnlyIfNotEmpty = false; }
182  }
183  m_modelsDialog->requestModelsInBackground(simulator, loadOnlyIfNotEmpty);
184  }
185  m_modelsDialog->exec();
186 
187  // force UI update
188  this->triggerSettingsChanged(simulator);
189  }

202  void CFirstModelSetComponent::changeModelDirectory()
203  {
204  if (!sGui || sGui->isShuttingDown()) { return; }
205  const CSimulatorInfo simulator = ui->comp_SimulatorSelector->getValue();
206  CSpecializedSimulatorSettings settings = m_simulatorSettings.getSpecializedSettings(simulator);
207  const bool clear = (QObject::sender() == ui->pb_ClearModelDir);
208 
209  if (clear)
210  {
211  settings.clearModelDirectories();
212  }
213  else
214  {
215  const QString dirOld = settings.getFirstModelDirectoryOrDefault();
216  const QString newDir = QFileDialog::getExistingDirectory(this->mainWindow(), tr("Open model directory"), dirOld, QFileDialog::ShowDirsOnly | QFileDialog::DontResolveSymlinks);
217  if (newDir.isEmpty() || CDirectoryUtils::isSameExistingDirectory(dirOld, newDir)) { return; }
218  settings.addModelDirectory(newDir);
219  }
220 
221  const CStatusMessage msg = m_simulatorSettings.setAndSaveSettings(settings, simulator);
222  if (msg.isSuccess())
223  {
224  this->triggerSettingsChanged(simulator);
225  }
226  else
227  {
228  this->showOverlayMessage(msg, 4000);
229  }

232  void CFirstModelSetComponent::createModelSet()
233  {
234  const CSimulatorInfo simulator = ui->comp_SimulatorSelector->getValue();
235  const int modelsCount = this->modelLoader()->getCachedModelsCount(simulator);
236  if (modelsCount < 1)
237  {
238  static const CStatusMessage msg = CStatusMessage(this).validationError(u"No models indexed so far. Try 'reload'!");
239  this->showOverlayMessage(msg, 4000);
240  return;
241  }
242 
243  bool useAllModels = false;
244  if (!ui->comp_Distributors->hasSelectedDistributors())
245  {
246  const QMessageBox::StandardButton reply = QMessageBox::question(this->mainWindow(), "Models", "No distributors selected, use all models?", QMessageBox::Yes | QMessageBox::No);
247  if (reply == QMessageBox::Yes)
248  {
249  useAllModels = true;
250  }
251  else
252  {
253  static const CStatusMessage msg = CStatusMessage(this).validationError(u"No distributors selected");
254  this->showOverlayMessage(msg, 4000);
255  return;
256  }
257  }
258  CAircraftModelList modelsForSet = this->modelLoader()->getCachedModels(simulator);
259  if (!useAllModels)
260  {
261  const CDistributorList distributors = ui->comp_Distributors->getSelectedDistributors();
262  modelsForSet = modelsForSet.findByDistributors(distributors);
263  }
264 
265  if (ui->cb_DbDataOnly->isChecked()) { modelsForSet.removeObjectsWithoutDbKey(); }
266  if (modelsForSet.isEmpty())
267  {
268  this->showOverlayHTMLMessage("Selection yielded no result!");
269  return;
270  }
271 
272  // just in case, paranoia
273  if (!m_modelSetDialog || !m_modelSetDialog->modelSetComponent())
274  {
275  this->showOverlayHTMLMessage("No model set dialog, cannot continue");
276  return;
277  }
278 
279  const int modelsSetCount = m_modelSetDialog->modelSetComponent()->getModelSetCount();
280  if (modelsSetCount > 0)
281  {
282  QMessageBox::StandardButton override = QMessageBox::question(this, "Override", "Override existing model set?", QMessageBox::Yes | QMessageBox::No);
283  if (override != QMessageBox::Yes) { return; }
284  }
285 
286  m_modelSetDialog->modelSetComponent()->setModelSet(modelsForSet, simulator);
287  ui->pb_ModelSet->click();
288  }
