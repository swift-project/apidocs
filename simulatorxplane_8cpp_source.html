179  bool CSimulatorXPlane::handleProbeValue(const CElevationPlane &plane, const CCallsign &callsign, bool waterFlag, const QString &hint, bool ignoreOutsideRange)
180  {
181  // XPlane specific checks for T778
182  // https://discordapp.com/channels/539048679160676382/577213275184562176/696780159969132626
183  if (!plane.hasMSLGeodeticHeight()) { return false; }
184  if (isSuspiciousTerrainValue(plane))
185  {
186  // ignore values up to +- 1.0meters, those most likely mean no scenery
187  const CLength distance = this->getDistanceToOwnAircraft(plane);
188  if (ignoreOutsideRange && distance > maxTerrainRequestDistance()) { return false; }
189 
190  static const CLength threshold = maxTerrainRequestDistance() * 0.50;
191  if (distance > threshold)
192  {
193  // we expect scenery to be loaded within threshold range
194  // outside that range we assue a suspicous value "represents no scenery"
195  // of course this can be wrong, but in that case we would geth those values
196  // once we get inside range
197  this->setMinTerrainProbeDistance(distance);
198  CLogMessage(this).debug(u"Suspicous XPlane probe [%1] value %2 for '%3' ignored, distance: %4 min.disance: %5 water: %6")
199  << hint
200  << plane.getAltitude().valueRoundedAsString(CLengthUnit::m(), 4)
201  << callsign.asString()
202  << distance.valueRoundedAsString(CLengthUnit::NM(), 2)
203  << m_minSuspicousTerrainProbe.valueRoundedAsString(CLengthUnit::NM(), 2)
204  << boolToYesNo(waterFlag);
205  return false;
206  }
207  }
208  return true;
209  }

251  bool CSimulatorXPlane::requestElevation(const ICoordinateGeodetic &reference, const CCallsign &callsign)
252  {
253  if (this->isShuttingDownOrDisconnected()) { return false; }
254  if (reference.isNull()) { return false; }
255 
256  // avoid requests for NON exising aircraft (based on LINUX crashes)
257  if (callsign.isEmpty()) { return false; }
258  if (!this->isAircraftInRange(callsign)) { return false; }
259 
260  const CLength d = this->getDistanceToOwnAircraft(reference);
261  if (!d.isNull() && d > maxTerrainRequestDistance())
262  {
263  // no request, too far away
264  return false;
265  }
266 
267  CCoordinateGeodetic pos(reference);
268  if (!pos.hasMSLGeodeticHeight())
269  {
270  // testing showed: height has an influence on the returned result
271  // - the most accurate value seems to be returned if the height is close to the elevation
272  // - in normal scenarios there is no much difference of the results if 0 is used
273  // - in Lukla (9200ft MSL) the difference between 0 and 9200 is around 1ft
274  // - in the LOWW scenario using 50000ft MSL results in around 3ft too low elevation
275  static const CAltitude alt(0, CAltitude::MeanSeaLevel, CLengthUnit::ft());
276  pos.setGeodeticHeight(alt);
277  }
278 
279  using namespace std::placeholders;
280  auto callback = std::bind(&CSimulatorXPlane::callbackReceivedRequestedElevation, this, _1, _2, _3);
281 
282  // Request
283  m_trafficProxy->getElevationAtPosition(callsign,
284  pos.latitude().value(CAngleUnit::deg()),
285  pos.longitude().value(CAngleUnit::deg()),
286  pos.geodeticHeight().value(CLengthUnit::m()),
287  callback);
288  emit this->requestedElevation(callsign);
289  return true;
290  }

306  void CSimulatorXPlane::fastTimerTimeout()
307  {
308  if (!this->isShuttingDownOrDisconnected())
309  {
310  m_fastTimerCalls++;
311 
312  m_serviceProxy->getOwnAircraftSituationDataAsync(&m_xplaneData);
313  m_serviceProxy->getOwnAircraftVelocityDataAsync(&m_xplaneData);
314  m_serviceProxy->getOwnAircraftCom1DataAsync(&m_xplaneData);
315  m_serviceProxy->getOwnAircraftCom2DataAsync(&m_xplaneData);
316  m_serviceProxy->getOwnAircraftXpdrAsync(&m_xplaneData);
317  m_serviceProxy->getAllWheelsOnGroundAsync(&m_xplaneData.onGroundAll);
318  m_serviceProxy->getHeightAglMAsync(&m_xplaneData.heightAglM);
319  m_serviceProxy->getPressureAltitudeFtAsync(&m_xplaneData.pressureAltitudeFt);
320 
321  CAircraftSituation situation;
322  situation.setPosition({ m_xplaneData.latitudeDeg, m_xplaneData.longitudeDeg, 0 });
323  const CAltitude altitude { m_xplaneData.altitudeM, CAltitude::MeanSeaLevel, CLengthUnit::m() };
324  const CPressure seaLevelPressure({ m_xplaneData.seaLevelPressureInHg, CPressureUnit::inHg() });
325  const CAltitude pressureAltitude(altitude.toPressureAltitude(seaLevelPressure));
326  if (std::isnan(m_xplaneData.pressureAltitudeFt))
327  {
328  m_altitudeDelta = {};
329  situation.setAltitude(altitude);
330  situation.setPressureAltitude(pressureAltitude);
331  }
332  else
333  {
334  const CAltitude pressureAltitudeXP12(m_xplaneData.pressureAltitudeFt, CAltitude::MeanSeaLevel, CAltitude::PressureAltitude, CLengthUnit::ft());
335  m_altitudeDelta = pressureAltitude - pressureAltitudeXP12;
336 
337  situation.setAltitude({ altitude - m_altitudeDelta, CAltitude::MeanSeaLevel });
338  situation.setPressureAltitude(pressureAltitudeXP12);
339  }
340  situation.setHeading({ m_xplaneData.trueHeadingDeg, CHeading::True, CAngleUnit::deg() });
341  situation.setPitch({ m_xplaneData.pitchDeg, CAngleUnit::deg() });
342  situation.setBank({ m_xplaneData.rollDeg, CAngleUnit::deg() });
343  situation.setGroundSpeed({ m_xplaneData.groundspeedMs, CSpeedUnit::m_s() });
344  const CAltitude elevation { m_xplaneData.altitudeM - m_xplaneData.heightAglM, CAltitude::MeanSeaLevel, CLengthUnit::m() };
345  situation.setGroundElevation(elevation, CAircraftSituation::FromProvider);
346  situation.setVelocity({ m_xplaneData.localXVelocityMs, m_xplaneData.localYVelocityMs, m_xplaneData.localZVelocityMs,
347  CSpeedUnit::m_s(), m_xplaneData.pitchRadPerSec, m_xplaneData.rollRadPerSec, m_xplaneData.headingRadPerSec,
348  CAngleUnit::rad(), CTimeUnit::s() });
349 
350  // Updates
351  // Do not update ICAO codes, as this overrides reverse lookups
352  // updateOwnIcaoCodes(m_xplaneData.aircraftIcaoCode, CAirlineIcaoCode());
353  this->updateOwnSituationAndGroundElevation(situation);
354 
355  // defaults
356  CSimulatedAircraft myAircraft(getOwnAircraft());
357  CComSystem com1(myAircraft.getCom1System()); // set defaults
358  CComSystem com2(myAircraft.getCom2System());
359  CTransponder transponder(myAircraft.getTransponder());
360 
361  // updates
362  com1.setFrequencyActive(CFrequency(m_xplaneData.com1ActiveKhz, CFrequencyUnit::kHz()));
363  com1.setFrequencyStandby(CFrequency(m_xplaneData.com1StandbyKhz, CFrequencyUnit::kHz()));
364  const int v1 = qRound(m_xplaneData.com1Volume * 100);
365  com1.setVolumeReceive(v1);
366  com1.setReceiveEnabled(m_xplaneData.isCom1Receiving);
367  com1.setTransmitEnabled(m_xplaneData.isCom1Transmitting);
368  const bool changedCom1 = myAircraft.getCom1System() != com1;
369 
370  com2.setFrequencyActive(CFrequency(m_xplaneData.com2ActiveKhz, CFrequencyUnit::kHz()));
371  com2.setFrequencyStandby(CFrequency(m_xplaneData.com2StandbyKhz, CFrequencyUnit::kHz()));
372  const int v2 = qRound(m_xplaneData.com2Volume * 100);
373  com2.setVolumeReceive(v2);
374  com2.setReceiveEnabled(m_xplaneData.isCom2Receiving);
375  com2.setTransmitEnabled(m_xplaneData.isCom2Transmitting);
376  const bool changedCom2 = myAircraft.getCom2System() != com2;
377 
378  transponder = CTransponder::getStandardTransponder(m_xplaneData.xpdrCode, xpdrMode(m_xplaneData.xpdrMode, m_xplaneData.xpdrIdent));
379  const bool changedXpr = (myAircraft.getTransponder() != transponder);
380 
381  if (changedCom1 || changedCom2 || changedXpr)
382  {
383  this->updateCockpit(com1, com2, transponder, identifier());
384  }
385 
386  if (m_isWeatherActivated)
387  {
388  if (CWeatherScenario::isRealWeatherScenario(m_weatherScenarioSettings.get()))
389  {
390  if (m_lastWeatherPosition.isNull() ||
391  calculateGreatCircleDistance(m_lastWeatherPosition, situation).value(CLengthUnit::mi()) > 20)
392  {
393  m_lastWeatherPosition = situation;
394  requestWeatherGrid(situation, this->identifier());
395  }
396  }
397  } // weather
398  }

401  void CSimulatorXPlane::slowTimerTimeout()
402  {
403  if (!this->isShuttingDownOrDisconnected())
404  {
405  m_slowTimerCalls++;
406 
407  // own aircraft data
408  m_serviceProxy->getOwnAircraftModelDataAsync(&m_xplaneData);
409  m_serviceProxy->getOwnAircraftLightsAsync(&m_xplaneData);
410  m_serviceProxy->getOwnAircraftPartsAsync(&m_xplaneData);
411 
412  CAircraftEngineList engines;
413  for (int engineNumber = 0; engineNumber < m_xplaneData.enginesN1Percentage.size(); ++engineNumber)
414  {
415  // Engine number start counting at 1
416  // We consider the engine running when N1 is bigger than 5 %
417  const CAircraftEngine engine { engineNumber + 1, m_xplaneData.enginesN1Percentage.at(engineNumber) > 5.0 };
418  engines.push_back(engine);
419  }
420 
421  const CAircraftLights lights(
422  m_xplaneData.strobeLightsOn, m_xplaneData.landingLightsOn, m_xplaneData.taxiLightsOn,
423  m_xplaneData.beaconLightsOn, m_xplaneData.navLightsOn, false);
424 
425  const CAircraftParts parts {
426  lights,
427  m_xplaneData.gearDeployRatio > 0, qRound(m_xplaneData.flapsDeployRatio * 100.0),
428  m_xplaneData.speedBrakeRatio > 0.5, engines, m_xplaneData.onGroundAll
429  };
430 
431  this->updateOwnParts(parts);
432 
433  CCallsignSet invalid;
434  for (CXPlaneMPAircraft &xplaneAircraft : m_xplaneAircraftObjects)
435  {
436  // Update remote aircraft to have the latest transponder modes, codes etc.
437  const CCallsign cs = xplaneAircraft.getCallsign();
438  const CSimulatedAircraft simulatedAircraft = this->getAircraftInRangeForCallsign(cs);
439  if (!simulatedAircraft.hasCallsign())
440  {
441  // removed in provider
442  if (!cs.isEmpty()) { invalid.insert(cs); }
443  continue;
444  }
445  xplaneAircraft.setSimulatedAircraft(simulatedAircraft);
446  }
447 
448  // remove the invalid ones
449  int i = 0;
450  if (this->isTestMode()) { invalid.clear(); } // skip this in test mode
451  for (const CCallsign &cs : invalid)
452  {
453  this->triggerRemoveAircraft(cs, ++i * 100);
454  }
455 
456  // KB: IMHO those data are pretty useless for XPlane
457  // no need to request them all the times
458  if ((m_slowTimerCalls % 3u) == 0u)
459  {
460  this->requestRemoteAircraftDataFromXPlane();
461  }
462 
463  // FPS
464  // reading FPS resets average, so we only monitor over some time
465  if ((m_slowTimerCalls % 5u) == 0u)
466  {
467  constexpr double warningMiles = 1;
468  constexpr double disconnectMiles = 2;
469  const double previousMiles = m_trackMilesShort;
470 
471  m_serviceProxy->getFrameStats(&m_averageFps, &m_simTimeRatio, &m_trackMilesShort, &m_minutesLate);
472 
473  if (previousMiles < disconnectMiles && m_trackMilesShort >= disconnectMiles)
474  {
475  emit insufficientFrameRateDetected(true);
476  }
477  else if (previousMiles < warningMiles && m_trackMilesShort >= warningMiles)
478  {
479  emit insufficientFrameRateDetected(false);
480  }
481  }
482  }
483  }

490  bool CSimulatorXPlane::connectTo()
491  {
492  if (this->isConnected()) { return true; }
493  const QString dBusServerAddress = m_xSwiftBusServerSettings.getThreadLocal().getDBusServerAddressQt();
494 
495  if (CDBusServer::isSessionOrSystemAddress(dBusServerAddress))
496  {
497  m_dBusConnection = QDBusConnection::sessionBus();
498  m_dbusMode = Session;
499  }
500  else if (CDBusServer::isQtDBusAddress(dBusServerAddress))
501  {
502  m_dBusConnection = QDBusConnection::connectToPeer(dBusServerAddress, "xswiftbus");
503  if (!m_dBusConnection.isConnected()) { return false; }
504  m_dbusMode = P2P;
505  }
506 
507  m_serviceProxy = new CXSwiftBusServiceProxy(m_dBusConnection, this);
508  m_trafficProxy = new CXSwiftBusTrafficProxy(m_dBusConnection, this);
509  m_weatherProxy = new CXSwiftBusWeatherProxy(m_dBusConnection, this);
510 
511  // hook up disconnected slot of connection
512  bool s = m_dBusConnection.connect(QString(), DBUS_PATH_LOCAL, DBUS_INTERFACE_LOCAL,
513  "Disconnected", this, SLOT(onDBusServiceUnregistered()));
514  Q_ASSERT(s);
515  if (!m_serviceProxy->isValid() || !m_trafficProxy->isValid() || !m_weatherProxy->isValid())
516  {
517  this->disconnectFrom();
518  return false;
519  }
520 
521  emitOwnAircraftModelChanged(m_serviceProxy->getAircraftModelPath(), m_serviceProxy->getAircraftModelFilename(), m_serviceProxy->getAircraftLivery(),
522  m_serviceProxy->getAircraftIcaoCode(), m_serviceProxy->getAircraftModelString(), m_serviceProxy->getAircraftName(), m_serviceProxy->getAircraftDescription());
523  QString xplaneVersion = QStringLiteral("%1.%2").arg(m_serviceProxy->getXPlaneVersionMajor()).arg(m_serviceProxy->getXPlaneVersionMinor());
524  setSimulatorDetails("X-Plane", {}, xplaneVersion);
525  connect(m_serviceProxy, &CXSwiftBusServiceProxy::aircraftModelChanged, this, &CSimulatorXPlane::emitOwnAircraftModelChanged);
526  connect(m_serviceProxy, &CXSwiftBusServiceProxy::airportsInRangeUpdated, this, &CSimulatorXPlane::setAirportsInRange);
527  m_serviceProxy->updateAirportsInRange();
528  connect(m_trafficProxy, &CXSwiftBusTrafficProxy::simFrame, this, &CSimulatorXPlane::updateRemoteAircraft);
529  connect(m_trafficProxy, &CXSwiftBusTrafficProxy::remoteAircraftAdded, this, &CSimulatorXPlane::onRemoteAircraftAdded);
530  connect(m_trafficProxy, &CXSwiftBusTrafficProxy::remoteAircraftAddingFailed, this, &CSimulatorXPlane::onRemoteAircraftAddingFailed);
531  if (m_watcher) { m_watcher->setConnection(m_dBusConnection); }
532  m_trafficProxy->removeAllPlanes();
533 
534  // send the settings
535  this->sendXSwiftBusSettings();
536 
537  // load CSL
538  this->loadCslPackages();
539 
540  // finish

708  void CSimulatorXPlane::loadCslPackages()
709  {
710  // An ad-hoc type for keeping track of packages as they are discovered.
711  // A model is a member of a package if the model path starts with the package path.
712  // A trailing separator is appended only for checking if a model path starts with this path.
713  struct Prefix
714  {
715  Prefix(const QString &p) : s(p + '/') {}
716  QString parent() const { return s.section('/', 0, -2, QString::SectionSkipEmpty | QString::SectionIncludeLeadingSep); }
717  bool isPrefixOf(const QString &o) const { return o.startsWith(s); }
718  QString s;
719  };
720 
721  // Heterogeneous comparison so a package can be found by binary search
722  // (e.g. std::lower_bound) using a model path as the search key.
723  struct PrefixComparator
724  {
725  bool operator()(const Prefix &a, const QString &b) const { return QStringRef(&a.s) < b.leftRef(a.s.size()); }
726  bool operator()(const QString &a, const Prefix &b) const { return a.leftRef(b.s.size()) < QStringRef(&b.s); }
727  };
728 
729  // The list of packages discovered so far.
730  QList<Prefix> packages;
731 
732  Q_ASSERT(this->isConnected());
733  const CAircraftModelList models = this->getModelSet();
734 
735  // Find the CSL packages for all models in the list
736  for (const auto &model : models)
737  {
738  const QString &modelFile = model.getFileName();
739  if (modelFile.isEmpty() || !QFile::exists(modelFile)) { continue; }
740 
741  // Check if this model's package was already found
742  auto it = std::lower_bound(packages.begin(), packages.end(), modelFile, PrefixComparator());
743  if (it != packages.end() && it->isPrefixOf(modelFile)) { continue; }
744 
745  // This model's package was not already found, so find it and add it to the list
746  QString package = findCslPackage(modelFile);
747  if (package.isEmpty()) { continue; }
748  packages.insert(it, package);
749  }
750 
751  // comment KB 2019-06
752  // a package is one xsb_aircraft.txt file BB has 9, X-CSL has 76
753  CSetBuilder<QString> superpackages;
754  for (const Prefix &package : std::as_const(packages))
755  {
756  superpackages.insert(package.parent());
757  }
758  QStringList superpackagesList = superpackages;
759 
760  const QDir simDir = getSimulatorSettings().getSimulatorDirectoryOrDefault();
761  for (const QString &package : superpackagesList)
762  {
763  if (CDirectoryUtils::isSameOrSubDirectoryOf(package, simDir))
764  {
765  const QString message = m_trafficProxy->loadPlanesPackage(package);
766  if (!message.isEmpty())
767  {
768  CLogMessage(this).validationError(u"CSL package '%1' xpmp error: %2") << package << message;
769  }
770  }
771  else
772  {
773  CLogMessage(this).validationError(u"CSL package '%1' can not be loaded as it is outside the X-Plane installation directory") << package;
774  }
775  }
776  }

1057  const bool hasCallsign = !callsign.isEmpty();
1058  if (!hasCallsign)

1066  if (!callsignsInRange.contains(callsign)) { continue; }
