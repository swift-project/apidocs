13  {
14  if (hostAddress.isEmpty()) { return false; }
15  QProcess process;
16  process.setProgram("ping");
17  if (CBuildConfig::isRunningOnWindowsNtPlatform())
18  {
19  process.setArguments({ "-n", "1", hostAddress });

21  else

25  }

34  if (err.contains("unreachable", Qt::CaseInsensitive)) { return false; }
35  return true;
36  }
