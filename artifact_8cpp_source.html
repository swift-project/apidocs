58  bool CArtifact::isWithDistribution(const CDistribution &distribution, bool acceptMoreStableDistributions) const
59  {
60  if (distribution.isEmpty() || !this->hasDistributions()) { return false; }
61  for (const CDistribution &dist : this->getDistributions())
62  {
63  if (dist == distribution) { return true; }
64  if (acceptMoreStableDistributions && dist.isStabilityBetter(distribution)) { return true; }
65  }
66  return false;
67  }

69  CRemoteFile CArtifact::asRemoteFile() const
70  {
71  if (!this->hasDistributions()) { return CRemoteFile(); }

73  const CDistribution d = this->getMostStableDistribution();
74  const CUrl url = d.getDownloadUrls().getRandomUrl();
75  if (url.isEmpty()) { return CRemoteFile(); }
76  rf.setUtcTimestamp(this->getUtcTimestamp());
77  rf.setUrl(url);
78  rf.setDescription(this->getPlatform().toQString() + " " + d.getChannel());

110  QVariant CArtifact::propertyByIndex(CPropertyIndexRef index) const
111  {
112  if (index.isMyself()) { return QVariant::fromValue(*this); }
113  if (IDatastoreObjectWithIntegerKey::canHandleIndex(index)) { return IDatastoreObjectWithIntegerKey::propertyByIndex(index); }
114 
115  const ColumnIndex i = index.frontCasted<ColumnIndex>();
116  switch (i)
117  {
118  case IndexName: return QVariant::fromValue(m_name);
119  case IndexMd5: return QVariant::fromValue(m_md5);
120  case IndexPlatform: return m_platform.propertyByIndex(index.copyFrontRemoved());
121  case IndexType: return QVariant::fromValue(m_type);
122  case IndexSize: return QVariant::fromValue(m_size);
123  case IndexSizeHumanReadable: return QVariant::fromValue(this->getFileSizeHumanReadable());
124  case IndexVersionString: return QVariant::fromValue(m_version);
125  case IndexQVersion: return QVariant::fromValue(this->getQVersion());
126  case IndexDistributions: return QVariant::fromValue(m_distributions);
127  default: return CValueObject::propertyByIndex(index);
128  }

131  void CArtifact::setPropertyByIndex(CPropertyIndexRef index, const QVariant &variant)
132  {
133  if (index.isMyself())
134  {
135  (*this) = variant.value<CArtifact>();
136  return;
137  }
138  if (IDatastoreObjectWithIntegerKey::canHandleIndex(index))
139  {
140  IDatastoreObjectWithIntegerKey::setPropertyByIndex(index, variant);
141  return;
142  }
143 
144  const ColumnIndex i = index.frontCasted<ColumnIndex>();
145  switch (i)
146  {
147  case IndexName: this->setName(variant.toString()); break;
148  case IndexMd5: m_md5 = variant.toString(); break;
149  case IndexPlatform: m_platform.setPropertyByIndex(index.copyFrontRemoved(), variant); break;
150  case IndexType: m_type = variant.toInt(); break;
151  case IndexSize: m_size = variant.toInt(); break;
152  case IndexVersionString: m_version = variant.toString(); break;
153  case IndexDistributions: m_distributions = variant.value<CDistributionList>(); break;
154  default: CValueObject::setPropertyByIndex(index, variant); break;
155  }

178  artifact.setDistributions(distributions);

180  }

202  CPlatform CArtifact::artifactNameToPlatform(const QString &name)
203  {
204  if (name.isEmpty()) { return CPlatform::unknownOs(); }
205  const QString n(name.toLower().trimmed());
206  if (n.contains("-windows-") || n.endsWith(".exe"))
207  {
208  if (n.contains("-64-")) { return CPlatform::win64Platform(); }
209  if (n.contains("-32-")) { return CPlatform::win32Platform(); }
210  return CPlatform::unknownOs();
211  }
212 
213  if (n.contains("-macos-") || n.endsWith(".dmg")) { return CPlatform::macOSPlatform(); }
214  if (n.contains("-linux-") || n.endsWith(".run")) { return CPlatform::linuxPlatform(); }
215  if (n.contains("-allos-")) { return CPlatform::allOs(); }
216 
217  return CPlatform::unknownOs();
218  }

240  QString CArtifact::trimVersionString(const QString &version)
241  {
242  if (version.count('.') != 3) return version; // no 4th segment
243  QStringList parts = version.split('.');
244  const QString p4 = trim4thSegment(parts[3]);
245  if (p4 == parts[3]) { return version; } // nothing changed
246  parts[3] = p4;
247  const QString v = parts.join('.');
248  return v;
249  }

255  return seg;
