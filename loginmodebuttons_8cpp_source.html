64  void CLoginModeButtons::configureLoginModes()
65  {
66  if (!sGui || sGui->isShuttingDown()) { return; }
67 
68  // we have no idea how we can get here without the context existing Ref T389
69  if (CBuildConfig::isLocalDeveloperDebugBuild() && !sGui->getIContextSimulator())
70  {
71  // how is this possible? In debug builds I do crosscheck
72  Q_ASSERT_X(false, Q_FUNC_INFO, "No context or sGUI");
73  }
74 
75  if (!sGui->getIContextSimulator() || (!sGui->isDeveloperFlagSet() && !sGui->getIContextSimulator()->isSimulatorSimulating()))
76  {
77  // Disable pilot login modes, only observer
78  ui->rb_LoginNormal->setEnabled(false);
79  ui->rb_LoginStealth->setEnabled(false);
80  ui->rb_LoginNormal->setToolTip("No simulator available");
81  ui->rb_LoginStealth->setToolTip("No simulator available");
82  ui->rb_LoginObserver->setChecked(true);
83  ui->lbl_NoSimulator->setVisible(true);
84  }
85  else
86  {
87  ui->rb_LoginNormal->setEnabled(true);
88  ui->rb_LoginStealth->setEnabled(true);
89  ui->rb_LoginNormal->setToolTip({});
90  ui->rb_LoginStealth->setToolTip({});
91  ui->rb_LoginNormal->setChecked(true);
92  ui->lbl_NoSimulator->setVisible(false);
93  }
94 
95  ui->rb_LoginStealth->setVisible(false); // 2019-01 hide as based on discussion with RR
96  }
