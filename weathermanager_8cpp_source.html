48  void CWeatherManager::appendRequest(const WeatherRequest &request)
49  {
50  const qint64 now = QDateTime::currentMSecsSinceEpoch();
51  if (!m_pendingRequests.isEmpty() && m_lastPendingRequestTs > 0)
52  {
53  const qint64 pendingMs = now - m_lastPendingRequestTs;
54  if (pendingMs > 30 * 1000)
55  {
56  BLACK_VERIFY_X(!CBuildConfig::isLocalDeveloperDebugBuild(), Q_FUNC_INFO, "Time out of pending weather request");
57  CLogMessage(this).warning(u"Time out of pending weather request, cleaning all requests!");
58  m_pendingRequests.clear();
59  }
60  }
61 
62  m_pendingRequests.append(request);
63  m_lastPendingRequestTs = now;
64  }

129  bool CWeatherManager::loadWeatherDataPlugins()
130  {
131  const CWeatherDataPluginInfoList weatherDataPluginInfos = m_pluginManagerWeatherData.getAvailableWeatherDataPlugins();
132  if (weatherDataPluginInfos.isEmpty())
133  {
134  CLogMessage(this).warning(u"No weather data plugin found!");
135  return false;
136  }
137 
138  for (const auto &pluginInfo : weatherDataPluginInfos)
139  {
140  IWeatherDataFactory *factory = m_pluginManagerWeatherData.getPluginById<IWeatherDataFactory>(pluginInfo.getIdentifier());
141  if (!factory)
142  {
143  CLogMessage(this).error(u"Failed to create IWeatherDataFactory for %1") << pluginInfo.getIdentifier();
144  return false;
145  }
146 
147  IWeatherData *weatherData = factory->create(this);
148  if (!weatherData)
149  {
150  CLogMessage(this).error(u"Failed to create IWeatherData instance for %1") << pluginInfo.getIdentifier();
151  return false;
152  }
153 
154  connect(weatherData, &IWeatherData::fetchingFinished, this, &CWeatherManager::handleNextRequest, Qt::QueuedConnection);
155  m_weatherDataPlugins.append(weatherData);
156  delete factory;
157  }
158  return true;
159  }

161  void CWeatherManager::fetchNextWeatherData()
162  {
163  const WeatherRequest weatherRequest = m_pendingRequests.constFirst();
164  const CLength maxDistance(100.0, CLengthUnit::km());
165 
166  for (IWeatherData *plugin : std::as_const(m_weatherDataPlugins))
167  {
168  if (!plugin) { continue; }
169  if (weatherRequest.filePath.isEmpty())
170  {
171  plugin->fetchWeatherData(weatherRequest.weatherGrid, maxDistance);
172  }
173  else
174  {
175  plugin->fetchWeatherDataFromFile(weatherRequest.filePath, weatherRequest.weatherGrid, maxDistance);
176  }
177  }
178  }

189  void CWeatherManager::handleNextRequest()
190  {
191  if (m_pendingRequests.isEmpty())
192  {
193  // we should normally never get here, only if a request timedout
194  CLogMessage(this).warning(u"No pending weather request, ignoring ....");
195  BLACK_VERIFY_X(!CBuildConfig::isLocalDeveloperDebugBuild(), Q_FUNC_INFO, "No pending weather request, ignoring ....");
196  return;
197  }
198 
199  IWeatherData *weatherDataPlugin = qobject_cast<IWeatherData *>(sender());
200  Q_ASSERT(weatherDataPlugin);
201  const auto fetchedWeatherGrid = weatherDataPlugin->getWeatherData();
202 
203  const WeatherRequest weatherRequest = m_pendingRequests.front();
204  CWeatherGrid requestedWeatherGrid = weatherRequest.weatherGrid;
205 
206  // Interpolation. So far it just picks the closest point without interpolation.
207  for (CGridPoint &gridPoint : requestedWeatherGrid)
208  {
209  const auto nearestGridPoint = fetchedWeatherGrid.findClosest(1, gridPoint.getPosition()).frontOrDefault();
210  gridPoint.copyWeatherDataFrom(nearestGridPoint);
211  gridPoint.setPosition(nearestGridPoint.getPosition());
212  }
213 
214  if (weatherRequest.callback)
215  {
216  weatherRequest.callback.singleShot(requestedWeatherGrid);
217  }
218 
219  if (!weatherRequest.identifier.isAnonymous()) { emit weatherGridReceived(requestedWeatherGrid, weatherRequest.identifier); }
220  m_pendingRequests.pop_front();
221 
222  // In case there are pending requests, start over again
223  if (!m_pendingRequests.isEmpty())
224  {
225  fetchNextWeatherData();
226  }
227  else
228  {
229  m_lastPendingRequestTs = -1;
230  }
231  }
