282  bool CGuiUtility::lenientTitleComparison(const QString &title, const QString &comparison)
283  {
284  if (title == comparison) { return true; }
285 
286  QString t(title.trimmed().toLower().simplified());
287  QString c(comparison.trimmed().toLower().simplified());
288 
289  // we should not have empty titles
290  BLACK_VERIFY_X(!t.isEmpty(), Q_FUNC_INFO, "missing title");
291  BLACK_VERIFY_X(!c.isEmpty(), Q_FUNC_INFO, "missing comparison value");
292  if (t.isEmpty() || c.isEmpty()) { return false; }
293 
294  // same?
295  if (t == c) { return true; }
296 
297  // further unify
298  if (!tsRegex.hasLocalData()) { tsRegex.setLocalData(QRegularExpression("[^a-z0-9\\s]")); }
299  const QRegularExpression &regexp = tsRegex.localData();
300  t = t.remove(regexp);
301  c = c.remove(regexp);
302  return t == c;
303  }

386  const int typeId = QMetaType::type(qPrintable(typeName));

537  {
