38  void CApiServerConnection::connectTo(const QString &username, const QString &password, const QString &client, const QUuid &networkVersion, ConnectionCallback callback)
39  {
40  if (isShuttingDown()) { return; }
41 
42  m_username = username;
43  m_password = password;
44  m_client = client;
45  m_networkVersion = networkVersion;
46  m_isAuthenticated = false;
47 
48  QUrl url(m_addressUrl);
49  url.setPath("/api/v1/auth");
50 
51  QJsonObject obj {
52  { "username", username },
53  { "password", password },
54  { "networkversion", networkVersion.toString() },
55  { "client", client }
56  };
57 
58  QNetworkRequest request(url);
59  QPointer<CApiServerConnection> myself(this);
60  request.setHeader(QNetworkRequest::ContentTypeHeader, "application/json");
61 
62  // posted in QAM thread, reply is nullptr if called from another thread
63  sApp->postToNetwork(request, CApplication::NoLogRequestId, QJsonDocument(obj).toJson(),
64  { this, [=](QNetworkReply *nwReply) {
65  // called in "this" thread
66  const QScopedPointer<QNetworkReply, QScopedPointerDeleteLater> reply(nwReply);
67  if (!myself || isShuttingDown()) { return; } // cppcheck-suppress knownConditionTrueFalse
68 
69  this->logRequestDuration(reply.data(), "authentication");
70  if (reply->error() != QNetworkReply::NoError)
71  {
72  this->logReplyErrorMessage(reply.data(), "authentication error");
73  callback(false);
74  return;
75  }
76 
77  // JWT authentication token
78  m_serverToUserOffsetMs = 0;
79  m_expiryLocalUtc = QDateTime(); // clean up
80 
81  m_jwt = reply->readAll().trimmed();
82  qint64 lifeTimeSecs = -1;
83  qint64 serverToUserOffsetSecs = -1;
84  do
85  {
86  const QString jwtToken(m_jwt);
87  const QJsonWebToken token = QJsonWebToken::fromTokenAndSecret(jwtToken, "");
88 
89  // get decoded header and payload
90  // QString strHeader = token.getHeaderQStr();
91  // QString strPayload = token.getPayloadQStr();
92  const QJsonDocument doc = token.getPayloadJDoc();
93  if (doc.isEmpty() || !doc.isObject()) { break; }
94  const qint64 validFromSecs = doc.object().value("nbf").toInt(-1);
95  if (validFromSecs < 0) { break; }
96  const qint64 localSecsSinceEpoch = QDateTime::currentSecsSinceEpoch();
97  serverToUserOffsetSecs = validFromSecs - localSecsSinceEpoch;
98  const qint64 serverExpirySecs = doc.object().value("exp").toInt();
99  const qint64 expiryLocalUtc = serverExpirySecs - serverToUserOffsetSecs;
100  lifeTimeSecs = expiryLocalUtc - localSecsSinceEpoch;
101  }
102  while (false);
103 
104  if (lifeTimeSecs > 0)
105  {
106  m_serverToUserOffsetMs = serverToUserOffsetSecs * 1000;
107  m_expiryLocalUtc = QDateTime::currentDateTimeUtc().addSecs(lifeTimeSecs);
108  m_isAuthenticated = true;
109  }
110 
111  // connected, callback
112  callback(m_isAuthenticated);
113  } });
114  }

157  {
158  if (isShuttingDown()) { return {}; }
159 
160  QPointer<QEventLoop> loop(this->newEventLoop());
161  QByteArray receivedData;
162 
163  // posted in QAM thread, reply is nullptr if called from another thread
164  sApp->getFromNetwork(request,
165  { this, [=, &receivedData](QNetworkReply *nwReply) {
166  const QScopedPointer<QNetworkReply, QScopedPointerDeleteLater> reply(nwReply);

220  void CApiServerConnection::postNoResponse(const QString &resource, const QJsonDocument &json)
221  {

223  this->checkExpiry();
224 
225  QUrl url(m_addressUrl);
226  url.setPath(resource);
227  QNetworkRequest request(url);
228  request.setRawHeader("Authorization", "Bearer " + m_jwt);
229  request.setHeader(QNetworkRequest::ContentTypeHeader, "application/json");
230 
231  // posted in QAM thread, reply is nullptr if called from another thread
232  sApp->postToNetwork(request, CApplication::NoLogRequestId, json.toJson(),
233  { this, [=](QNetworkReply *nwReply) {
234  // called in "this" thread

269  void CApiServerConnection::checkExpiry()
270  {
271  if (!m_expiryLocalUtc.isValid() || QDateTime::currentDateTimeUtc() > m_expiryLocalUtc.addSecs(-5 * 60))
272  {
273  QPointer<CApiServerConnection> myself(this);
274  this->connectTo(m_username, m_password, m_client, m_networkVersion,
275  { this, [=](bool authenticated) {
276  if (!myself) { return; } // cppcheck-suppress knownConditionTrueFalse
277  CLogMessage(this).info(u"API server authenticated '%1': %2") << m_username << boolToYesNo(authenticated);
278  } });
279  }

295  void CApiServerConnection::logRequestDuration(const QNetworkReply *reply, const QString &addMsg)
296  {
297  if (!CBuildConfig::isLocalDeveloperDebugBuild()) { return; }
298  if (!reply) { return; }
299 
300  const qint64 d = CNetworkUtils::requestDuration(reply);
301  if (d < 0) { return; }
302  if (addMsg.isEmpty())
303  {
304  CLogMessage(this).info(u"AFV network request for '%1': %2ms") << reply->url().toString() << d;
305  }
306  else
307  {
308  CLogMessage(this).info(u"AFV network request (%1) for '%2': '%3'") << addMsg << reply->url().toString() << d;
