121  void CInternalsComponent::sendTextMessage()
122  {
123  if (!sGui || !sGui->getIContextNetwork()) { return; }
124  if (ui->le_TxtMsgFrom->text().isEmpty()) { return; }
125  if (ui->pte_TxtMsg->toPlainText().isEmpty()) { return; }
126  if (ui->le_TxtMsgTo->text().isEmpty() && ui->dsb_TxtMsgFrequency->text().isEmpty()) { return; }
127 
128  // send in some time
129  const QObject *senderObj = QObject::sender();
130  if (senderObj == ui->pb_SendTextMessageDeferred)
131  {
132  QTimer::singleShot(5000, this, &CInternalsComponent::sendTextMessage);
133  return;
134  }
135 
136  const CCallsign sender(ui->le_TxtMsgFrom->text().trimmed());
137  const CCallsign recipient(ui->le_TxtMsgTo->text().trimmed());
138  const QString msgTxt(ui->pte_TxtMsg->toPlainText().trimmed());
139  const double freqMHz = CMathUtils::round(ui->dsb_TxtMsgFrequency->value(), 3);
140  CTextMessage tm;
141  if (recipient.isEmpty() && freqMHz >= 117.995) // 118.000 actually plus epsilon
142  {
143  // remark: the internal double value can be still something like 127.999997
144  CFrequency f = CFrequency(freqMHz, CFrequencyUnit::MHz());
145  CComSystem::roundToChannelSpacing(f, CComSystem::ChannelSpacing8_33KHz);
146  tm = CTextMessage(msgTxt, f, sender);
147  }
148  else
149  {
150  tm = CTextMessage(msgTxt, sender, recipient);
151  }
152  tm.setCurrentUtcTime();
153  sGui->getIContextNetwork()->testReceivedTextMessages(CTextMessageList({ tm }));
154  }
