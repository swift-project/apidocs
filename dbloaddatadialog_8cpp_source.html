79  void CDbLoadDataDialog::onButtonClicked(QAbstractButton *button)
80  {
81  if (!button) { return; }
82  if (button == ui->bb_loadDataDialog->button(QDialogButtonBox::Apply))
83  {
84  const QStringList entityList = this->selectedEntities();
85  if (entityList.isEmpty()) { return; }
86  const CEntityFlags::Entity loadEntities = CEntityFlags::multipleEntitiesByNames(entityList);
87  m_pendingEntities = sGui->getWebDataServices()->triggerLoadingDirectlyFromSharedFiles(loadEntities, false);
88  const int pending = CEntityFlags::numberOfEntities(m_pendingEntities);
89  m_pendingEntitiesCount = sGui->getWebDataServices()->getDbInfoObjectsCount(loadEntities);
90  ui->pb_Loading->setMaximum(m_pendingEntitiesCount > 0 ? m_pendingEntitiesCount : pending);
91  ui->pb_Loading->setValue(0);
92  ui->wi_WorkStatus->setVisible(pending > 0);
93  ui->wi_Consolidate->setVisible(false);
94  ui->le_Info->setText("Loading started ...");
95  }
96  }

98  void CDbLoadDataDialog::onDataRead(CEntityFlags::Entity entity, CEntityFlags::ReadState state, int number, const QUrl &url)
99  {
100  if (m_pendingEntities == CEntityFlags::NoEntity) { return; } // not triggered from here
101  if (!m_pendingEntities.testFlag(CEntityFlags::entityToEntityFlag(entity))) { return; }
102 
103  const QString e = CEntityFlags::entitiesToString(entity);
104  const QString s = CEntityFlags::stateToString(state);
105 
106  ui->le_Info->setText(e % u" " % s);
107  if (!CEntityFlags::isFinishedReadStateOrFailure(state)) { return; }
108  if (state == CEntityFlags::ReadFailed)
109  {
110  CLogMessage(this).warning(u"Read failed for %1 from '%2'") << e << url.toString();
111  }
112 
113  m_pendingEntities &= ~entity;
114  const int pending = CEntityFlags::numberOfEntities(m_pendingEntities);
115  const int max = ui->pb_Loading->maximum();
116  if (m_pendingEntitiesCount < 0)
117  {
118  ui->pb_Loading->setValue(max - pending);
119  }
120  else
121  {
122  m_pendingEntitiesCount -= number;
123  ui->pb_Loading->setValue(max - m_pendingEntitiesCount);
124  }
125  if (pending < 1)
126  {
127  m_pendingEntitiesCount = -1;
128  const bool defaultConsolidate = !ui->cb_AllModels->isChecked() && ui->cb_ModelSet->isChecked();
129 
130  QPointer<CDbLoadDataDialog> myself(this);
131  QTimer::singleShot(2000, this, [=] {
132  if (!myself) { return; }
133  ui->wi_Consolidate->setVisible(true);
134  ui->wi_WorkStatus->setVisible(false);
135  if (defaultConsolidate)
136  {
137  m_autoConsolidate = true;
138  QPointer<CDbLoadDataDialog> self(this); // 2nd "self"/"myself" for cppcheck identicalConditionAfterEarlyExit
139  QTimer::singleShot(1000, this, [=] {
140  if (!self) { return; }
141  self->consolidate();
142  });
143  }
144  });
145  }
146  }

166  void CDbLoadDataDialog::consolidate()
167  {
168  const bool set = ui->cb_ModelSet->isChecked();
169  const bool all = ui->cb_AllModels->isChecked();
170  if (m_consolidating) { return; }
171  if (!set && !all) { return; }
172  ui->wi_WorkStatus->setVisible(true);
173  ui->pb_Loading->setValue(0);
174  ui->pb_Loading->setMaximum(0); // 0/0 causing busy indicator
175  const CSimulatorInfo simulator = ui->comp_SimulatorSelector->getValue();
176 
177  do
178  {
179  if (set)
180  {
181  ui->le_Info->setText("Model set");
182  CAircraftModelList models = m_sets.getCachedModels(simulator);
183  const int c = CDatabaseUtils::consolidateModelsWithDbDataAllowsGuiRefresh(models, true, true);
184  if (c > 0) { m_sets.setCachedModels(models, simulator); }
185  }
186 
187  if (!this->isVisible()) { break; } // dialog closed?
188  if (all)
189  {
190  ui->le_Info->setText("All models");
191  CAircraftModelList models = m_models.getCachedModels(simulator);
192  const int c = CDatabaseUtils::consolidateModelsWithDbDataAllowsGuiRefresh(models, true, true);
193  if (c > 0) { m_models.setCachedModels(models, simulator); }
194  }
195  }
196  while (false);
197 
198  m_consolidating = false;
199 
200  QPointer<CDbLoadDataDialog> myself(this);
201  QTimer::singleShot(2000, this, [=] {
202  if (!myself) { return; }
203  ui->pb_Loading->setMaximum(100);
204  ui->wi_WorkStatus->setVisible(false);
205  if (m_autoConsolidate)
206  {
207  m_autoConsolidate = false;
208  this->accept();
209  }
210  });
211  }
