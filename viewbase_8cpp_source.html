39  template <class T>
40  int CViewBase<T>::updateContainer(const ContainerType &container, bool sort, bool resize)
41  {
42  Q_ASSERT_X(m_model, Q_FUNC_INFO, "Missing model");
43  if (container.isEmpty())
44  {
45  // shortcut
46  this->clear();
47  return 0;
48  }
49 
50  // we have data
51  this->showLoadIndicator(container.size());
52  const bool reallyResize = resize && isResizeConditionMet(container.size()); // do we really perform resizing
53  bool presize = (m_resizeMode == PresizeSubset) &&
54  this->isEmpty() && // only when no data yet
55  !reallyResize; // not when we resize later
56  presize = presize || (this->isEmpty() && resize && !reallyResize); // we presize if we wanted to resize but actually do not because of condition
57  const bool presizeThresholdReached = presize && container.size() > ResizeSubsetThreshold; // only when size making sense
58 
59  // when we will not resize, we might presize
60  if (presizeThresholdReached)
61  {
62  const int presizeRandomElements = this->getPresizeRandomElementsSize(container.size());
63  if (presizeRandomElements > 0)
64  {
65  m_model->update(container.sampleElements(presizeRandomElements), false);
66  this->fullResizeToContents();
67  }
68  }
69 
70  const int c = m_model->update(container, sort);
71 
72  // resize after real update according to mode
73  if (presizeThresholdReached) // cppcheck-suppress knownConditionTrueFalse
74  {
75  // currently no furhter actions
76  }
77  else if (reallyResize)
78  {
79  this->resizeToContents(); // mode based resize
80  }
81  else if (presize && !presizeThresholdReached) // cppcheck-suppress knownConditionTrueFalse
82  {
83  // small amount of data not covered before
84  this->fullResizeToContents();
85  }
86  this->updateSortIndicator(); // make sure sort indicator represents sort order
87  this->hideLoadIndicator();
88  return c;
89  }

92  CWorker *CViewBase<T>::updateContainerAsync(const ContainerType &container, bool sort, bool resize)
93  {
94  // avoid unnecessary effort when empty
95  if (container.isEmpty())
96  {
97  this->clear();
98  return nullptr;
99  }
100 
101  Q_UNUSED(sort)
102  ModelClass *model = this->derivedModel();
103  const auto sortColumn = model->getSortColumn();
104  const auto sortOrder = model->getSortOrder();
105  this->showLoadIndicator(container.size());
106  CWorker *worker = CWorker::fromTask(this, "ViewSort", [model, container, sortColumn, sortOrder]() {
107  return model->sortContainerByColumn(container, sortColumn, sortOrder);
108  });
109  worker->thenWithResult<ContainerType>(this, [this, resize](const ContainerType &sortedContainer) {
110  this->updateContainer(sortedContainer, false, resize);
111  });
112  worker->then(this, &CViewBase::asyncUpdateFinished);
113  return worker;
114  }

287  template <class T>
288  int CViewBase<T>::updateSelected(const CPropertyIndexVariantMap &vm)
289  {
290  if (vm.isEmpty()) { return 0; }
291  if (!hasSelection()) { return 0; }
292  int c = 0;
293 
294  int lastUpdatedRow = -1;
295  int firstUpdatedRow = -1;
296  const CPropertyIndexList propertyIndexes(vm.indexes());
297  const QModelIndexList indexes = this->selectedRows();
298 
299  for (const QModelIndex &i : indexes)
300  {
301  if (i.row() == lastUpdatedRow) { continue; }
302  lastUpdatedRow = i.row();
303  if (firstUpdatedRow < 0 || lastUpdatedRow < firstUpdatedRow) { firstUpdatedRow = lastUpdatedRow; }
304  ObjectType obj(this->at(i));
305 
306  // update all properties in map
307  for (const CPropertyIndex &pi : propertyIndexes)
308  {
309  obj.setPropertyByIndex(pi, vm.value(pi));
310  }
311 
312  // and update container
313  if (this->derivedModel()->setInContainer(i, obj))
314  {
315  c++;
316  }
317  }
318 
319  if (c > 0)
320  {
321  this->derivedModel()->emitDataChanged(firstUpdatedRow, lastUpdatedRow);
322  }
323  return c;
324  }

340  template <class T>
341  int CViewBase<T>::removeSelectedRows()
342  {
343  if (!this->hasSelection()) { return 0; }
344  if (this->isEmpty()) { return 0; }
345 
346  const int currentRows = this->rowCount();
347  const ContainerType selected(this->selectedObjects());
348  const CVariant deletedObjsVariant = CVariant::from(selected);
349  int delta = 0;
350 
351  if (!this->hasFilter() && currentRows == this->selectedRowCount())
352  {
353  // shortcut if all are selected
354  this->clear(); // clear all
355  delta = currentRows;
356  }
357  else
358  {
359  ContainerType unselectedObjects(this->container());
360  unselectedObjects.removeIfInSubset(selected);
361  this->updateContainerMaybeAsync(unselectedObjects);
362  delta = currentRows - unselectedObjects.size();
363  }
364  emit this->objectsDeleted(deletedObjsVariant);
365  return delta;
366  }

370  {
371  const int rc = this->rowCount();
372  if (rc > ResizeSubsetThreshold)

375  if (presizeRandomElements > 0)

378  m_model->update(containerBackup.sampleElements(presizeRandomElements), false);
379  this->fullResizeToContents();
380  m_model->update(containerBackup, false);
381  }
382  }
383  else
384  {
385  this->fullResizeToContents();
386  }

728  template <class T>
729  CStatusMessage CViewBase<T>::loadJsonFile(const QString &fileName)
730  {
731  CStatusMessage m;
732  do
733  {
734  if (fileName.isEmpty())
735  {
736  m = CStatusMessage(this).error(u"Load canceled, no file name");
737  break;
738  }
739 
740  const QString json(CFileUtils::readFileToString(fileName));
741  if (json.isEmpty())
742  {
743  m = CStatusMessage(this).warning(u"Reading '%1' yields no data") << fileName;
744  break;
745  }
746  if (!Json::looksLikeSwiftJson(json))
747  {
748  m = CStatusMessage(this).warning(u"No swift JSON '%1'") << fileName;
749  break;

752  try
753  {
754  const bool allowCacheFormat = this->allowCacheFileFormatJson();
755  const QJsonObject jsonObject = Json::jsonObjectFromString(json, allowCacheFormat);

759  break;
760  }
761 
762  ContainerType container;
763 
764  if (jsonObject.contains("type") && jsonObject.contains("value"))

766  // read from variant format

768  containerVariant.convertFromJson(jsonObject);
769  if (!containerVariant.canConvert<ContainerType>())

771  m = CStatusMessage(this).warning(u"No valid swift JSON '%1'") << fileName;
772  break;
773  }
774  container = containerVariant.value<ContainerType>();
775  }
776  else

778  // container format directly
779  container.convertFromJson(jsonObject);
780  }

784  if (m.isFailure()) { break; } // modification error
785  if (countBefore > 0 && container.isEmpty()) { break; }
786  m = this->validateLoadedJsonData(container);
787  if (m.isFailure()) { break; } // validaton error
788  this->updateContainerMaybeAsync(container);
789  m = CStatusMessage(this, CStatusMessage::SeverityInfo, "Reading " + fileName + " completed", true);
790  this->jsonLoadedAndModelUpdated(container);
791  this->rememberLastJsonDirectory(fileName);
792  }
793  catch (const CJsonException &ex)
794  {
795  m = CStatusMessage::fromJsonException(ex, this, QString("Reading JSON from '%1'").arg(fileName));
796  break;
797  }
798  }
