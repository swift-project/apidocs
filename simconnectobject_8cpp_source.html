69  DWORD CSimConnectObject::getRequestId(CSimConnectDefinitions::SimObjectRequest offset) const
70  {
71  if (CBuildConfig::isLocalDeveloperDebugBuild())
72  {
73  const SimObjectType type = requestIdToType(m_requestId);
74  const bool same = CSimConnectObject::isSameTypeGroup(type, this->getType());
75  Q_ASSERT_X(same, Q_FUNC_INFO, "Type mismatch");
76  }
77 
78  DWORD os = 0;
79  switch (this->getType())
80  {
81  case TerrainProbe:
82  os = static_cast<DWORD>(CSimulatorFsxCommon::offsetSimObjTerrainProbe(offset));
83  break;
84  case AircraftNonAtc:
85  case AircraftSimulatedObject:
86  default:
87  os = static_cast<DWORD>(CSimulatorFsxCommon::offsetSimObjAircraft(offset));
88  break;
89  }
90  return os + m_requestId;
91  }

104  bool CSimConnectObject::isOutdatedPendingAdded(qint64 thresholdMs, qint64 currentMsSinceEpoch) const
105  {
106  if (!this->isPendingAdded()) { return false; }
107  if (currentMsSinceEpoch < 0) { currentMsSinceEpoch = QDateTime::currentMSecsSinceEpoch(); }
108  if (m_tsCreated < 0) { return true; } // no valid timestamp
109  const qint64 delta = currentMsSinceEpoch - m_tsCreated;
110  return delta > thresholdMs;
111  }

536  CSimConnectObjects CSimConnectObjects::removeOutdatedPendingAdded(CSimConnectObject::SimObjectType type)
537  {
538  CCallsignSet removeCallsigns;
539  CSimConnectObjects removedObjects;
540 
541  const qint64 ts = QDateTime::currentMSecsSinceEpoch();
542  for (const CSimConnectObject &simObject : std::as_const(*this))
543  {
544  // verification takes at least a second, so we need some time before outdating
545  if (type != CSimConnectObject::AllTypes && simObject.getType() != type) { continue; }
546  if (!simObject.isOutdatedPendingAdded(5000, ts)) { continue; }
547  removedObjects.insert(simObject);
548  removeCallsigns.insert(simObject.getCallsign());
549  }
550  if (!removeCallsigns.isEmpty())
551  {
552  this->removeCallsigns(removeCallsigns);
553  }
554  return removedObjects;
555  }
