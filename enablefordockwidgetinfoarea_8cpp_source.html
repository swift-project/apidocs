40  m_parentDockableInfoArea = parentDockableWidget;
41  QMetaObject::Connection con = QDockWidget::connect(parentDockableWidget, &QDockWidget::destroyed, [this] {
42  // break dependency to dockable widget
43  m_parentDockableInfoArea = nullptr;

45  Q_ASSERT_X(con, Q_FUNC_INFO, "Connection failed");
