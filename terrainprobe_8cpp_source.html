23  std::array<double, 3> CTerrainProbe::getElevation(double degreesLatitude, double degreesLongitude, double metersAltitude, const std::string &callsign, bool &o_isWater) const
24  {
25  double x, y, z;
26  XPLMWorldToLocal(degreesLatitude, degreesLongitude, metersAltitude, &x, &y, &z);
27 
28  XPLMProbeInfo_t probe;
29  probe.structSize = sizeof(probe);
30  auto result = XPLMProbeTerrainXYZ(m_ref, static_cast<float>(x), static_cast<float>(y), static_cast<float>(z), &probe);
31  if (result != xplm_ProbeHitTerrain)
32  {
33  if (m_logMessageCount < 100)
34  {
35  m_logMessageCount++;
36  std::string error;
37  if (result == xplm_ProbeError) { error = "probe error"; }
38  else if (result == xplm_ProbeMissed) { error = "probe missed"; }
39  else { error = "unknown probe result"; }
40  WARNING_LOG(callsign + " " + error + " at " + std::to_string(degreesLatitude) + ", " + std::to_string(degreesLongitude) + ", " + std::to_string(metersAltitude));
41  }
42 
43  o_isWater = false;
44  return { { std::numeric_limits<double>::quiet_NaN(), degreesLatitude, degreesLongitude } };
45  }
46  XPLMLocalToWorld(probe.locationX, probe.locationY, probe.locationZ, &degreesLatitude, &degreesLongitude, &metersAltitude);
47 
48  if (probe.is_wet && m_logMessageCount < 100)
49  {
50  m_logMessageCount++;
51  DEBUG_LOG(callsign + " probe hit water at " + std::to_string(degreesLatitude) + ", " + std::to_string(degreesLongitude) + ", " + std::to_string(metersAltitude));
52  }
53  if (std::isnan(metersAltitude) && m_logMessageCount < 100)
54  {
55  m_logMessageCount++;
56  DEBUG_LOG(callsign + " probe returned NaN at " + std::to_string(degreesLatitude) + ", " + std::to_string(degreesLongitude) + ", " + std::to_string(metersAltitude));
57  }
58  o_isWater = probe.is_wet;
59  return { { metersAltitude, degreesLatitude, degreesLongitude } };
60  }
