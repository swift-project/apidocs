60  CAircraftSituation CSituationForm::getSituation() const
61  {
62  const CCoordinateGeodetic position = ui->comp_Coordinate->getCoordinate();
63  const CAltitude pressureAltitude(position.geodeticHeight().toPressureAltitude(this->getBarometricPressureMsl()));
64 
65  CAircraftSituation s(position);
66  s.setBank(this->getBankAngle());
67  s.setPitch(this->getPitchAngle());
68  s.setHeading(CHeading(this->getHeadingAngle(), CHeading::True));
69  s.setGroundSpeed(this->getGroundSpeed());
70 
71  if (!pressureAltitude.isNull() && pressureAltitude.getAltitudeType() == CAltitude::PressureAltitude)
72  {
73  s.setPressureAltitude(pressureAltitude);
74  }
75  return s;
76  }
