172  void CTextMessageComponent::displayTextMessage(const CTextMessageList &messages)
173  {
174  if (messages.isEmpty()) { return; }
175  if (!sGui || sGui->isShuttingDown()) { return; }
176  const CSimulatedAircraft ownAircraft(this->getOwnAircraft());
177  const CTextMessageSettings msgSettings(m_messageSettings.getThreadLocal());
178  const bool playNotification = sGui && sGui->getIContextAudio();
179  const bool audioCsMentioned = playNotification && m_audioSettings.get().textCallsignMentioned();
180 
181  bool addedToAllMessages = false;
182  for (const CTextMessage &message : messages)
183  {
184  bool relevantForMe = false;
185  CNotificationSounds::NotificationFlag notification = CNotificationSounds::NoNotifications;
186 
187  // SELCAL
188  if (!m_usedAsOverlayWidget && message.isSelcalMessage() && ownAircraft.isSelcalSelected(message.getSelcalCode()))
189  {
190  // this is SELCAL for me
191  if (playNotification)
192  {
193  sGui->getCContextAudioBase()->playSelcalTone(message.getSelcalCode());
194  }
195 
196  if (msgSettings.popupSelcalMessages())
197  {
198  CStatusMessage msg = CLogMessage(this).info(u"SELCAL received");
199  this->emitDisplayInInfoWindow(CVariant::from(msg), 3 * 1000);
200  }
201  continue;
202  }
203 
204  // UNICOM
205  if (message.isSendToUnicom())
206  {
207  ui->tep_TextMessagesUnicom->insertTextMessage(message);
208 
209  // Message was received from others
210  if (!message.wasSent())
211  {
212  notification = CNotificationSounds::NotificationTextMessageUnicom;
213  }
214  relevantForMe = true;
215  }
216 
217  // check message, handle special cases first
218  if (message.isServerMessage())
219  {
220  // void
221  }
222  else if (message.isBroadcastMessage())
223  {
224  // FAKE private message
225  this->addPrivateChannelTextMessage(message);
226  relevantForMe = true;
227  }
228  else if (message.isRadioMessage())
229  {
230  // check for own COM frequencies
231  if (message.isSendToFrequency(ownAircraft.getCom1System().getFrequencyActive()))
232  {
233  ui->tep_TextMessagesCOM1->insertTextMessage(message);
234  if (!message.isSendToUnicom())
235  {
236  notification = CNotificationSounds::NotificationTextMessageFrequency;
237  }
238  relevantForMe = true;
239  }
240  if (message.isSendToFrequency(ownAircraft.getCom2System().getFrequencyActive()))
241  {
242  ui->tep_TextMessagesCOM2->insertTextMessage(message);
243  if (!message.isSendToUnicom())
244  {
245  notification = CNotificationSounds::NotificationTextMessageFrequency;
246  }
247  relevantForMe = true;
248  }
249 
250  // callsign mentioned notification
251  if (relevantForMe && audioCsMentioned && ownAircraft.hasCallsign() && message.mentionsCallsign(ownAircraft.getCallsign()))
252  {
253  notification = CNotificationSounds::NotificationTextCallsignMentioned;
254  // Flash taskbar icon
255  QApplication::alert(QWidget::topLevelWidget());
256  }
257  }
258  else if (message.isPrivateMessage())
259  {
260  // private message
261  this->addPrivateChannelTextMessage(message);
262  relevantForMe = true;
263  // Flash taskbar icon
264  QApplication::alert(QWidget::topLevelWidget());
265  }
266  else
267  {
268  BLACK_AUDIT_X(false, Q_FUNC_INFO, "Wrong message type");
269  continue;
270  }
271 
272  // message for me? right frequency? otherwise quit
273  if (this->hasAllMessagesTab() && (relevantForMe || message.isServerMessage()))
274  {
275  ui->tvp_TextMessagesAll->push_back(message); // no sorting
276  ui->tvp_TextMessagesAll->resort();
277  addedToAllMessages = true;
278  }
279  if (!relevantForMe) { continue; }
280 
281  // Play notification
282  if (playNotification && notification != CNotificationSounds::NoNotifications)
283  {
284  sGui->getCContextAudioBase()->playNotification(notification, true);
285  }
286 
287  // overlay message if this channel is not selected
288  if (message.isServerMessage()) { continue; }
289  if (message.isBroadcastMessage()) { continue; }
290 
291  if (!message.wasSent() && !message.isSendToUnicom())
292  {
293  // if the channel is selected, do nothing
294  if (!this->isCorrespondingTextMessageTabSelected(message))
295  {
296  if (msgSettings.popup(message, ownAircraft))
297  {
298  this->emitDisplayInInfoWindow(CVariant::from(message), 15 * 1000);
299  }
300  }
301  } // message
302  } // for
303 
304  if (addedToAllMessages && ui->tvp_TextMessagesAll->isSortedByTimestampPropertyLatestLast())
305  {
306  ui->tvp_TextMessagesAll->scrollToBottom();
307  }
308  }

393  bool CTextMessageComponent::isCorrespondingTextMessageTabSelected(const CTextMessage &textMessage) const
394  {
395  if (!this->isVisibleWidgetHack()) { return false; }
396  if (!textMessage.hasValidRecipient()) { return false; }
397  if (textMessage.isEmpty()) { return false; } // ignore empty message
398  if (textMessage.isPrivateMessage())
399  {
400  // private message
401  const CCallsign cs = textMessage.getSenderCallsign();
402  if (cs.isEmpty()) { return false; }
403  const QWidget *tab = this->findTextMessageTabByCallsign(cs, false);
404  if (!tab) { return false; }
405  return ui->tw_TextMessages->currentWidget() == tab;
406  }
407  else
408  {
409  // frequency message
410  const CSimulatedAircraft ownAircraft(this->getOwnAircraft());
411  if (ui->tw_TextMessages->currentWidget() == ui->tb_TextMessagesAll) { return true; }
412  if (textMessage.isSendToFrequency(ownAircraft.getCom1System().getFrequencyActive()))
413  {
414  return ui->tw_TextMessages->currentWidget() == ui->tb_TextMessagesCOM1;
415  }
416  if (textMessage.isSendToFrequency(ownAircraft.getCom2System().getFrequencyActive()))
417  {
418  return ui->tw_TextMessages->currentWidget() == ui->tb_TextMessagesCOM2;
419  }
420  return false;
421  }
422  }

435  void CTextMessageComponent::showCurrentFrequenciesFromCockpit(const CSimulatedAircraft &ownAircraft)
436  {
437  const CFrequency freq1 = ownAircraft.getCom1System().getFrequencyActive();
438  const CFrequency freq2 = ownAircraft.getCom2System().getFrequencyActive();
439 
440  CAtcStationList f1Stations;
441  CAtcStationList f2Stations;
442  if (sGui && sGui->getIContextNetwork())
443  {
444  f1Stations = sGui->getIContextNetwork()->getOnlineStationsForFrequency(freq1);
445  f2Stations = sGui->getIContextNetwork()->getOnlineStationsForFrequency(freq2);
446  }
447 
448  const QString f1n = QString::asprintf("%03.3f", freq1.valueRounded(CFrequencyUnit::MHz(), 3));
449  const QString f2n = QString::asprintf("%03.3f", freq2.valueRounded(CFrequencyUnit::MHz(), 3));
450  QString f1 = QStringLiteral("COM1: %1").arg(f1n);
451  QString f2 = QStringLiteral("COM2: %1").arg(f2n);
452  if (f1Stations.size() == 1)
453  {
454  f1 += u' ' % f1Stations.front().getCallsignAndControllerRealName();
455  }
456  if (f2Stations.size() == 1)
457  {
458  f2 += u' ' % f2Stations.front().getCallsignAndControllerRealName();
459  }
460 
461  ui->tb_TextMessagesCOM1->setToolTip(f1);
462  ui->tb_TextMessagesCOM1->setToolTip(f2);
463 
464  ui->tw_TextMessages->setTabText(ui->tw_TextMessages->indexOf(ui->tb_TextMessagesCOM1), f1);
465  ui->tw_TextMessages->setTabText(ui->tw_TextMessages->indexOf(ui->tb_TextMessagesCOM2), f2);
466 
467  // update tabs
468  this->updateAllTabs();
469  }

482  newTabWidget->setProperty("callsign", callsign.asString());

484  QVBoxLayout *layout = new QVBoxLayout(newTabWidget);

489  newTabWidget->layout()->setContentsMargins(marginLeft, marginTop, marginRight, 2);
490  layout->addWidget(textEdit);
491  layout->addWidget(closeButton);
492  newTabWidget->setLayout(layout);
493  textEdit->setContextMenuPolicy(Qt::CustomContextMenu);
494  textEdit->setProperty("supervisormsg", supervisor);
495  textEdit->setStyleSheetForContent(style);
496 
497  const int index = ui->tw_TextMessages->addTab(newTabWidget, this->getCallsignAndRealName(callsign));
498  QToolButton *closeButtonInTab = new QToolButton(newTabWidget);
499  closeButtonInTab->setText("[X]");
500  closeButtonInTab->setProperty("supervisormsg", supervisor);
501  QTabBar *bar = ui->tw_TextMessages->tabBar();

503  if (supervisor)
504  {

508  }
509  ui->tw_TextMessages->setCurrentIndex(index);
510  closeButtonInTab->setProperty("tabName", tabName);
511  closeButton->setProperty("tabName", tabName);
512 
513  connect(closeButton, &QPushButton::released, this, &CTextMessageComponent::closeTextMessageTab);
514  connect(closeButtonInTab, &QPushButton::released, this, &CTextMessageComponent::closeTextMessageTab);
515 
516  this->setTabWidgetDescription(callsign, index);
517  return newTabWidget;
518  }

561  void CTextMessageComponent::addPrivateChannelTextMessage(const CTextMessage &textMessage)
562  {
563  if (!textMessage.isPrivateMessage()) { return; }
564  CCallsign cs = textMessage.wasSent() ? textMessage.getRecipientCallsign() : textMessage.getSenderCallsign();
565  if (cs.isEmpty()) { return; }
566 
567  const bool isBroadcast = textMessage.isBroadcastMessage();
568  if (isBroadcast) { cs.markAsBroadcastCallsign(); }
569 
570  const bool isWallopMessage = textMessage.isWallopMessage();
571  if (isWallopMessage) { cs.markAsWallopCallsign(); }
572 
573  const QWidget *tab = this->findTextMessageTabByCallsign(cs);
574  if (!tab) { tab = this->addNewTextMessageTab(cs); }
575  Q_ASSERT_X(tab, Q_FUNC_INFO, "Missing tab");
576  CTextMessageTextEdit *textEdit = tab->findChild<CTextMessageTextEdit *>();
577  BLACK_VERIFY_X(textEdit, Q_FUNC_INFO, "Missing text edit");
578  if (!textEdit) { return; } // do not crash, though this situation should not happen
579  textEdit->insertTextMessage(textMessage);
580 
581  // sound
582  if (textMessage.isServerMessage()) { return; }
583  if (isBroadcast) { return; }

736  QString CTextMessageComponent::textMessageToCommand(const QString &enteredLine)
737  {
738  // only if visible
739  if (enteredLine.isEmpty()) { return {}; }
740 
741  const int index = ui->tw_TextMessages->currentIndex();
742  QString cmd(".msg ");
743  if (index < 0 || index == ui->tw_TextMessages->indexOf(ui->tb_TextMessagesAll))
744  {
745  CLogMessage(this).validationError(u"Incorrect message channel");
746  return {};
747  }
748  else if (ui->tw_TextMessages->tabText(index) == "SUP")
749  {
750  CLogMessage(this).validationError(u"Message cannot be send to SUP channel. To send another wallop message use .wallop instead");
751  return {};
752  }
753  else
754  {
755  if (index == ui->tw_TextMessages->indexOf(ui->tb_TextMessagesCOM1))
756  {
757  cmd.append(QString::number(this->getOwnAircraft().getCom1System().getFrequencyActive().valueRounded(CFrequencyUnit::MHz(), 3)));
758  }
759  else if (index == ui->tw_TextMessages->indexOf(ui->tb_TextMessagesCOM2))
760  {
761  cmd.append(QString::number(this->getOwnAircraft().getCom2System().getFrequencyActive().valueRounded(CFrequencyUnit::MHz(), 3)));
762  }
763  else if (index == ui->tw_TextMessages->indexOf(ui->tb_TextMessagesUnicom))
764  {
765  cmd.append(QString::number(CPhysicalQuantitiesConstants::FrequencyUnicom().valueRounded(CFrequencyUnit::MHz(), 3)));
766  }
767  else
768  {
769  // not a standard channel
770  bool isNumber;
771  const QString selectedTabText = firstPartOfTabText(ui->tw_TextMessages->tabText(index).trimmed());
772  const double frequency = selectedTabText.toDouble(&isNumber);
773  if (isNumber)
774  {
775  const CFrequency radioFrequency = CFrequency(frequency, CFrequencyUnit::MHz());
776  if (CComSystem::isValidCivilAviationFrequency(radioFrequency))
777  {
778  cmd.append(QString::number(radioFrequency.valueRounded(CFrequencyUnit::MHz(), 3)));
779  }
780  else
781  {
782  // selectedTabText expected to be the callsign
783  // with T664 we resolve against the callsigns in the context if possible
784  const CCallsign cs = this->getCallsignPropertyForTab(index, true);
785  if (cs.isEmpty())
786  {
787  cmd.append(selectedTabText);
788  }
789  else
790  {
791  cmd.append(cs.isAtcCallsign() ? cs.getStringAsSet() : cs.asString());
792  }
793  }
794  }
795  else
796  {
797  cmd.append(selectedTabText);
798  }
799  }
800  return cmd % u" " % enteredLine;
801  }
802  }
