45  QString CAutoPublishData::toDatabaseJson() const
46  {
47  // used simple string JSON generation as it is faster
48  QString json;
49 
50  for (const auto [string, cg] : makePairsRange(m_modelStringVsCG))
51  {
52  json += QStringLiteral("{ \"type\": \"cg\", \"modelstring\": \"%1\", \"cgft\": %2 },\n").arg(string, cg.valueRoundedAsString(CLengthUnit::ft(), 1));
53  }
54 
55  for (const auto [string, sim] : makePairsRange(m_modelStringVsSimulatorInfo))
56  {
57  json += QStringLiteral("{ \"type\": \"simulatorupdate\", \"modelstring\": \"%1\", \"simulator\": \"%2\" },\n").arg(string, sim.toQString(false));
58  }
59 
60  if (json.isEmpty()) { return {}; }
61  json.chop(2); // remove 2 chars at end
62  return u"[\n" % json % u"\n]\n";
63  }

65  int CAutoPublishData::fromDatabaseJson(const QString &jsonData, bool clear)
66  {
67  if (clear) { this->clear(); }
68  if (jsonData.isEmpty()) { return 0; }
69  const QJsonArray array = Json::jsonArrayFromString(jsonData);
70  if (array.isEmpty()) { return 0; }
71 
72  for (const QJsonValue &value : array)
73  {
74  const QJsonObject obj = value.toObject();
75  const QString t = obj["type"].toString();
76  const QString m = obj["modelstring"].toString();
77  if (m.isEmpty()) { continue; }
78 
79  if (t.startsWith("simulator", Qt::CaseInsensitive))
80  {
81  const QString simulator = obj["simulator"].toString();
82  const CSimulatorInfo si(simulator);
83  if (si.isSingleSimulator())
84  {
85  this->insert(m, si);
86  }
87  }
88  else if (t.startsWith("cg", Qt::CaseInsensitive))
89  {
90  const double cgFt = obj["cgft"].toDouble(-1);
91  if (cgFt < 0) { continue; }
92  this->insert(m, CLength(cgFt, CLengthUnit::ft()));
93  }
94  }
95 
96  return m_modelStringVsCG.size() + m_modelStringVsSimulatorInfo.size();
97  }

136  CStatusMessageList CAutoPublishData::analyzeAgainstDBData(const CAircraftModelList &dbModels)
137  {
138  static const CLogCategoryList cats({ CLogCategories::mapping(), CLogCategories::webservice() });
139  if (dbModels.isEmpty()) { return CStatusMessage(this).validationError(u"No DB data"); }
140  if (this->isEmpty()) { return CStatusMessage(this).validationWarning(u"No data"); }
141 
142  CStatusMessageList msgs;
143  msgs.push_back(CStatusMessage(cats).validationInfo(u"DB models: %1") << dbModels.size());
144 
145  CSetBuilder<QString> unchangedCG;
146  for (const QString &modelString : m_modelStringVsCG.keys())
147  {
148  const CAircraftModel dbModel = dbModels.findFirstByModelStringOrDefault(modelString);
149  if (dbModel.hasValidDbKey())
150  {
151  if (dbModel.getCG() == m_modelStringVsCG[modelString])
152  {
153  unchangedCG.insert(modelString);
154  }
155  }
156  }
157 
158  CSetBuilder<QString> unchangedSim;
159  for (const QString &modelString : m_modelStringVsSimulatorInfo.keys())
160  {
161  const CAircraftModel dbModel = dbModels.findFirstByModelStringOrDefault(modelString);
162  if (dbModel.hasValidDbKey())
163  {
164  if (dbModel.getSimulator().matchesAny(m_modelStringVsSimulatorInfo[modelString]))
165  {
166  unchangedSim.insert(modelString);
167  }
168  }
169  }
170 
171  // remove
172  if (!unchangedCG.isEmpty())
173  {
174  QList<QString> unchangedCGList = std::move(unchangedCG);
175  msgs.push_back(CStatusMessage(cats).validationInfo(u"Removing unchanged CGs: %1") << unchangedCGList.size());
176  for (const QString &m : unchangedCGList) { m_modelStringVsCG.remove(m); }
177  }
178 
179  if (!unchangedSim.isEmpty())
180  {
181  QList<QString> unchangedSimList = std::move(unchangedSim);
182  msgs.push_back(CStatusMessage(cats).validationInfo(u"Removing unchanged simulators: %1") << unchangedSimList.size());
183  for (const QString &m : unchangedSimList) { m_modelStringVsSimulatorInfo.remove(m); }
184  }
185 
186  msgs.push_back(CStatusMessage(this).validationInfo(this->getSummary()));
187  return msgs;
188  }

239  int CAutoPublishData::deleteAutoPublishFiles(const QString &dirPath)
240  {
241  const QStringList fileList = findAndCleanupPublishFiles(dirPath);
242  if (fileList.isEmpty()) { return 0; }
243 
244  int c = 0;

255  QStringList CAutoPublishData::findAndCleanupPublishFiles(const QString &dirPath)
256  {
257  QDir dir(dirPath);
258  if (!dir.exists()) { return {}; }
259 
260  const QString filter = fileBaseName() % u'*' % fileAppendix();
261  const QStringList filters({ filter });
262  dir.setNameFilters(filters);
263  dir.setFilter(QDir::Files | QDir::NoDotAndDotDot | QDir::NoSymLinks);
264  const QStringList fileList = dir.entryList();
265  if (fileList.isEmpty()) { return fileList; }
266 
267  // avoid outdated files (e.g. if format changes)
268  QStringList correctedList;
269  const QDateTime deadline = QDateTime::currentDateTimeUtc().addDays(-30);
270  for (const QString &fn : fileList)
271  {
272  const QFileInfo fi(fn.contains(dir.absolutePath()) ? fn : CFileUtils::appendFilePathsAndFixUnc(dir.absolutePath(), fn));
273  if (!fi.exists()) { continue; }
274  const QDateTime created = fi.birthTime().toUTC();
275  if (deadline < created)
276  {
277  correctedList << fn;
278  }
279  else
280  {
281  QFile deleteFile(fn);
282  deleteFile.remove();
283  }
284  }
285 
286  return correctedList;
287  }
