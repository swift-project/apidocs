118  void CInfoBarWebReadersStatusBase::setLedReadState(CLedWidget *led, CEntityFlags::ReadState readState)
119  {
120  Q_ASSERT_X(led, Q_FUNC_INFO, "no LED");
121  const int resetTimeMs = qRound(2.5 * 1000);
122  switch (readState)
123  {
124  case CEntityFlags::ReadSkipped:
125  case CEntityFlags::ReadFinished:
126  case CEntityFlags::ReadFinishedRestricted:
127  led->setOn(false);
128  led->blink(resetTimeMs); // blink here (temp. "on"), since read from cache results in immediate ReadFinished
129  break;
130  case CEntityFlags::ReadStarted:
131  case CEntityFlags::ReadParsing:
132  led->setOn(true);
133  break;
134  case CEntityFlags::ReadFailed:
135  led->setTriState(2 * resetTimeMs);
136  break;
137  default:
138  BLACK_VERIFY_X(false, Q_FUNC_INFO, "read state not handled");
139  break;
140  }
141  }

143  QList<CLedWidget *> CInfoBarWebReadersStatusBase::entitiesToLeds(CEntityFlags::Entity entities) const
144  {
145  QList<CLedWidget *> leds;
146  if (entities.testFlag(CEntityFlags::CountryEntity)) { leds << m_ledCountries; }
147  if (entities.testFlag(CEntityFlags::DistributorEntity)) { leds << m_ledDistributors; }
148  if (entities.testFlag(CEntityFlags::AircraftIcaoEntity)) { leds << m_ledIcaoAircraft; }
149  if (entities.testFlag(CEntityFlags::AirlineIcaoEntity)) { leds << m_ledIcaoAirline; }
150  if (entities.testFlag(CEntityFlags::LiveryEntity)) { leds << m_ledLiveries; }
151  if (entities.testFlag(CEntityFlags::ModelEntity)) { leds << m_ledModels; }
152  return leds;
153  }
