127  QStringList CSimConnectUtilities::getSimConnectIniFileDirectories()
128  {
129  const QString docDir = QStandardPaths::locate(QStandardPaths::DocumentsLocation, "", QStandardPaths::LocateDirectory);
130  if (docDir.isEmpty()) return QStringList();
131 
132  QDir directory(docDir);
133  directory.setFilter(QDir::Dirs | QDir::NoDotAndDotDot | QDir::NoSymLinks);
134  const QStringList dirList = directory.entryList();
135  QStringList simDirs;
136  for (const QString &dir : dirList)
137  {
138  if (dir.contains("Flight Simulator", Qt::CaseInsensitive) || dir.contains("Prepar3D", Qt::CaseInsensitive))
139  {
140  simDirs.push_back(CFileUtils::appendFilePaths(docDir, dir));
141  }
142  }
143 
144  // gets the latest P3D as first
145  simDirs.sort();
146  std::reverse(std::begin(simDirs), std::end(simDirs));
147  return simDirs;
148  }

203  QString CSimConnectUtilities::convertToSimConnectMetar(const CGridPoint &gridPoint, bool isFSX, bool useWindLayers, bool useVisibilityLayers, bool useCloudLayers, bool useTempLayers)
204  {
205  // STATION ID
206  Q_ASSERT(!gridPoint.getIdentifier().isEmpty());
207  QString simconnectMetar = gridPoint.getIdentifier();
208 
209  // SURFACE WINDS/WINDS ALOFT
210  const CWindLayerList windLayers = useWindLayers ? gridPoint.getWindLayers().sortedBy(&CWindLayer::getLevel) : CWindLayerList();
211  simconnectMetar += windsToSimConnectMetar(windLayers, isFSX);
212 
213  // VISIBILITY
214  const CVisibilityLayerList visibilityLayers = useVisibilityLayers ? gridPoint.getVisibilityLayers().sortedBy(&CVisibilityLayer::getBase) : CVisibilityLayerList();
215  simconnectMetar += visibilitiesToSimConnectMetar(visibilityLayers);
216 
217  // PRESENT CONDITIONS
218  // todo
219 
220  // PARTIAL OBSCURATION
221  // todo
222 
223  // SKY CONDITIONS
224  const CCloudLayerList cloudLayers = useCloudLayers ? gridPoint.getCloudLayers().sortedBy(&CCloudLayer::getBase) : CCloudLayerList();
225  simconnectMetar += cloudsToSimConnectMetar(cloudLayers);
226 
227  // TEMPERATURE
228  const CTemperatureLayerList temperatureLayers = useTempLayers ? gridPoint.getTemperatureLayers().sortedBy(&CTemperatureLayer::getLevel) : CTemperatureLayerList();
229  simconnectMetar += temperaturesToSimConnectMetar(temperatureLayers);
230 
231  // ALTIMETER
232  // Format:
233  // QNNNN
234  // Q = specifier for altimeter in millibars
235  // NNNN = altimeter in millibars
236  if (!gridPoint.getPressureAtMsl().isNull())
237  {
238  static const QString arg1s(" Q%1");
239  const int altimeter = gridPoint.getPressureAtMsl().valueInteger(CPressureUnit::mbar());
240  simconnectMetar += arg1s.arg(altimeter, 4, 10, QLatin1Char('0'));
241  }
242  return simconnectMetar.simplified();
243  }

251  QString CSimConnectUtilities::windsToSimConnectMetar(const CWindLayerList &windLayers, bool isFSX)
252  {
253  static const QString arg1s("%1");
254 
255  QString simConnectWinds;
256  bool surface = true;
257  for (const CWindLayer &windLayer : windLayers)
258  {
259  simConnectWinds += QLatin1Char(' ');
260 
261  // Format:
262  // DDDSSSUUU (steady)
263  // DDDSSSGXXUUU (gusts)
264  if (windLayer.isDirectionVariable())
265  {
266  // DDD = VRB for variable
267  simConnectWinds += QLatin1String("VRB");
268  }
269  else
270  {
271  if (!windLayer.getSpeed().isNull() && !windLayer.getDirection().isNull())
272  {
273  const int speedKts = windLayer.getSpeed().valueInteger(CSpeedUnit::kts());
274  const int directionDeg = windLayer.getDirection().valueInteger(CAngleUnit::deg());
275 
276  simConnectWinds += arg1s.arg(directionDeg, 3, 10, QLatin1Char('0')) % // DDD = Direction (0-360 degrees)
277  arg1s.arg(speedKts, 3, 10, QLatin1Char('0')); // SSS = Speed
278  }
279  }
280  // XX = Gust speed
281  const int gustSpeedKts = windLayer.getGustSpeed().valueInteger(CSpeedUnit::kts());
282  if (gustSpeedKts > 0) { simConnectWinds += u'G' % arg1s.arg(gustSpeedKts, 2, 10, QLatin1Char('0')); }
283 
284  // UUU = Speed units
285  simConnectWinds += QStringLiteral("KT");
286 
287  if (surface)
288  {
289  // Surface extension:
290  // &DNNNNTS
291  static const QString surfaceWinds =
292  "&D" // D = specifier for surface layer
293  "305" // Surface default depth is 1000 feet or 305m
294  "NG"; // We don't have turbulence or wind shear information, hence we use the defaults
295  simConnectWinds += surfaceWinds;
296  surface = false;
297  }
298  else
299  {
300  CAltitude altitude = windLayer.getLevel();
301 
302  // this seems to crash FSX, P3D works
303  // https://www.fsdeveloper.com/forum/threads/setting-winds-aloft.18862/
304  if (!altitude.isNull() && !isFSX)
305  {
306  altitude.toMeanSeaLevel();
307  int altitudeValueMeters = altitude.valueInteger(CLengthUnit::m());
308 
309  // Winds aloft extension:
310  // &ANNNNTS
311  simConnectWinds +=
312  u"&A" % // A = specifier for altitude above mean sea-level (MSL)
313  arg1s.arg(altitudeValueMeters, 4, 10, QLatin1Char('0')) % // NNNN = depth (height) in meters.
314  u"NG"; // We don't have turbulence or wind shear information, hence we use the defaults
315  }
316  }
317  }
318 
319  return QStringLiteral(" ") % simConnectWinds.simplified();
320  }

322  QString CSimConnectUtilities::visibilitiesToSimConnectMetar(const CVisibilityLayerList &visibilityLayers)
323  {
324  // There are several format options, we use the meter format:
325  // NNNND&BXXXX&DYYYY
326  QString simconnectVisibilities;
327  for (const auto &visibilityLayer : visibilityLayers)
328  {
329  simconnectVisibilities += QLatin1Char(' ');
330 
331  // NNNN = in meters
332  auto visibility = visibilityLayer.getVisibility().valueInteger(CLengthUnit::m());
333  visibility = qMin(9999, visibility);
334  simconnectVisibilities += QStringLiteral("%1").arg(visibility, 4, 10, QLatin1Char('0'));
335 
336  // D = directional variation
337  // We set NDV - no directional variation
338  simconnectVisibilities += QLatin1String("NDV");
339 
340  // XXXX = base of visibility layer in meters
341  const auto base = visibilityLayer.getBase().valueInteger(CLengthUnit::m());
342  simconnectVisibilities += QStringLiteral("&B%1").arg(base, 4, 10, QLatin1Char('0'));
343 
344  // YYYY = depth of visibility layer in meters
345  const auto depth = visibilityLayer.getTop().valueInteger(CLengthUnit::m());
346  simconnectVisibilities += QStringLiteral("&D%1").arg(depth, 4, 10, QLatin1Char('0'));
347  }
348  return simconnectVisibilities;
349  }

351  QString CSimConnectUtilities::cloudsToSimConnectMetar(const CCloudLayerList &cloudLayers)
352  {
353  // Format:
354  // CCCNNN&BXXXX&DYYYY
355  QString simconnectClouds;
356  static const QString arg1s("%1");
357  for (const CCloudLayer &cloudLayer : cloudLayers)
358  {
359  simconnectClouds += QLatin1Char(' ');
360 
361  // CCC = Coverage string
362  switch (cloudLayer.getCoverage())
363  {
364  case CCloudLayer::None: simconnectClouds += QLatin1String("CLR"); break;
365  case CCloudLayer::Few: simconnectClouds += QLatin1String("FEW"); break;
366  case CCloudLayer::Broken: simconnectClouds += QLatin1String("BKN"); break;
367  case CCloudLayer::Overcast: simconnectClouds += QLatin1String("OVC"); break;
368  case CCloudLayer::Scattered:
369  default:
370  simconnectClouds += QLatin1String("SCT");
371  }
372 
373  // NNN = coded height
374  // If NNN is 999 the level is 100,000 feet, otherwise it is 100 x NNN in feet
375  auto level = cloudLayer.getTop().valueInteger(CLengthUnit::ft()) / 100;
376  // Ignore clouds higher than 99900 feet
377  if (level > 999) { continue; }
378  simconnectClouds += arg1s.arg(level, 3, 10, QLatin1Char('0')) %
379  u'&';
380 
381  // TT = Cloud type
382  switch (cloudLayer.getClouds())
383  {
384  case CCloudLayer::Cirrus: simconnectClouds += QLatin1String("CI"); break;
385  case CCloudLayer::Stratus: simconnectClouds += QLatin1String("ST"); break;
386  case CCloudLayer::Thunderstorm: simconnectClouds += QLatin1String("CB"); break;
387  case CCloudLayer::Cumulus:
388  default:
389  simconnectClouds += QLatin1String("CU");
390  }
391 
392  // 000 - Unused.
393  simconnectClouds += QLatin1String("000");
394 
395  // F = Top of cloud
396  // Default to F - flat
397  simconnectClouds += QLatin1Char('F');
398  // T = Turbulence
399  // N - None (default)
400  simconnectClouds += QLatin1Char('N');
401 
402  // P = precipitation rate
403  // http://wiki.sandaysoft.com/a/Rain_measurement#Rain_Rate
404  auto precipitationRate = cloudLayer.getPrecipitationRate();
405  // Very light rain: precipitation rate is < 0.25 mm/hour
406  if (precipitationRate < 0.25) { simconnectClouds += QLatin1Char('V'); }
407  // Light rain: precipitation rate is between 0.25mm/hour and 1.0mm/hour
408  else if (precipitationRate >= 0.25 && precipitationRate < 1.0) { simconnectClouds += QLatin1Char('L'); }
409  // Moderate rain: precipitation rate is between 1.0 mm/hour and 4.0 mm/hour
410  else if (precipitationRate >= 1.0 && precipitationRate < 4.0) { simconnectClouds += QLatin1Char('M'); }
411  // Heavy rain: recipitation rate is between 4.0 mm/hour and 16.0 mm/hour
412  else if (precipitationRate >= 4.0 && precipitationRate < 16.0) { simconnectClouds += QLatin1Char('H'); }
413  // Very heavy rain: precipitation rate is > 16.0 mm/hour
414  else if (precipitationRate >= 16.0) { simconnectClouds += QLatin1Char('D'); }
415 
416  // Q = Type of precipitation
417  switch (cloudLayer.getPrecipitation())
418  {
419  case CCloudLayer::Rain: simconnectClouds += QLatin1Char('R'); break;
420  case CCloudLayer::Snow: simconnectClouds += QLatin1Char('S'); break;
421  default: simconnectClouds += QLatin1Char('N');
422  }
423 
424  // BBB = Coded base height
425  // the precipitation ends at this height, set to 0 for it to land on the ground
426  simconnectClouds += QLatin1String("000");
427 
428  // I = icing rate
429  // Set to None for now
430  simconnectClouds += QLatin1String("N");
431  }
432  return simconnectClouds;
433  }

435  QString CSimConnectUtilities::temperaturesToSimConnectMetar(const CTemperatureLayerList &temperatureLayers)
436  {
437  // Format:
438  // TT/DD&ANNNNN
439  QString simconnectTemperatures;
440  static const QString arg1s("%1");
441 
442  for (const CTemperatureLayer &temperatureLayer : temperatureLayers)
443  {
444  simconnectTemperatures += QLatin1Char(' ');
445 
446  const int temperature = temperatureLayer.getTemperature().valueInteger(CTemperatureUnit::C());
447  const int dewPoint = temperatureLayer.getDewPoint().valueInteger(CTemperatureUnit::C());
448  const int altitude = temperatureLayer.getLevel().valueInteger(CLengthUnit::m());
449 
450  simconnectTemperatures += arg1s.arg(temperature, 2, 10, QLatin1Char('0')) % // TT = temperature in Celsius
451  u'/' %
452  arg1s.arg(dewPoint, 2, 10, QLatin1Char('0')) % // DD = dewpoint in Celsius
453  u"&A" %
454  arg1s.arg(altitude, 5, 10, QLatin1Char('0')); // NNNNN = altitude of the temperatures in meters.
455  }
456  return simconnectTemperatures;
457  }
