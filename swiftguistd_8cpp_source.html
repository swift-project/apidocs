106 void SwiftGuiStd::performGracefulShutdown()
107 {
108  if (!m_init) { return; }
109  m_init = false;
110 
111  Q_ASSERT_X(CThreadUtils::thisIsMainThread(), Q_FUNC_INFO, "Should shutdown in main thread");
112 
113  // shut down all timers
114  this->stopAllTimers(true);
115 
116  // if we have a context, we shut some things down
117  if (m_contextNetworkAvailable)
118  {
119  if (sGui && sGui->getIContextNetwork() && sGui->getIContextNetwork()->isConnected())
120  {
121  if (m_contextAudioAvailable)
122  {
123  sGui->getIContextAudio()->disconnect(this); // break down signal / slots
124  }
125  sGui->getIContextNetwork()->disconnectFromNetwork();
126  sGui->getIContextNetwork()->disconnect(this); // avoid any status update signals, etc.
127  }
128  }
129 
130  // clean up GUI
131  ui->comp_MainInfoArea->dockAllWidgets();
132 
133  // allow some other parts to react
134  const QPointer<SwiftGuiStd> myself(this);
135  if (sGui) { sGui->processEventsToRefreshGui(); }
136  if (!sGui || !myself) { return; } // killed in meantime?
137 
138  // tell context GUI is going down
139  if (sGui->getIContextApplication())
140  {
141  sGui->getIContextApplication()->unregisterApplication(identifier());
142  }
143 
144  // allow some other parts to react
145  if (sGui) { sGui->processEventsToRefreshGui(); }
146 }

148 void SwiftGuiStd::closeEvent(QCloseEvent *event)
149 {
150  if (sGui)
151  {
152  if (sGui->getIContextNetwork() && sGui->getIContextNetwork()->isConnected())
153  {
154  // we do not just logoff, but give the user a chance to respond
155  event->ignore();
156  QPointer<SwiftGuiStd> myself(this);
157  QTimer::singleShot(500, this, [=] {
158  if (!myself) { return; }
159  myself->loginRequested();
160  });
161  return;
162  }
163 
164  if (this->triggerAutoPublishDialog())
165  {
166  event->ignore();
167  return;
168  }
169 
170  // save settings
171  if (sGui->showCloseDialog(this, event) == QDialog::Rejected)
172  {
173  // already ignored
174  return;
175  }
176  }
177  this->performGracefulShutdown();
178 }

301 void SwiftGuiStd::setContextAvailability()
302 {
303  const bool corePreviouslyAvailable = m_coreAvailable;
304  const bool isShuttingDown = !sGui || sGui->isShuttingDown();
305  if (!isShuttingDown && sGui->getIContextApplication() && !sGui->getIContextApplication()->isEmptyObject())
306  {
307  // ping to check if core is still alive
308  m_coreAvailable = this->isMyIdentifier(sGui->getIContextApplication()->registerApplication(identifier()));
309  }
310  else
311  {
312  m_coreAvailable = false;
313  }
314  if (isShuttingDown) { return; }
315  if (m_coreAvailable && m_coreFailures > 0)
316  {
317  m_coreFailures--;
318  }
319  else if (!m_coreAvailable && m_coreFailures < MaxCoreFailures)
320  {
321  m_coreFailures++;
322  }
323  else if (!m_coreAvailable && !m_displayingDBusReconnect)
324  {
325  this->displayDBusReconnectDialog();
326  }
327  m_contextNetworkAvailable = m_coreAvailable && sGui->getIContextNetwork() && !sGui->getIContextNetwork()->isEmptyObject();
328  m_contextAudioAvailable = m_coreAvailable && sGui->getIContextAudio() && !sGui->getIContextAudio()->isEmptyObject();
329 
330  // react to a change in core's availability
331  if (m_coreAvailable != corePreviouslyAvailable)
332  {
333  if (m_coreAvailable)
334  {
335  // core has just become available (startup)
336  // this HERE is called with and without DBus
337  sGui->getIContextApplication()->synchronizeLocalSettings();
338  }
339  }
340 }

566 void SwiftGuiStd::displayDBusReconnectDialog()
567 {
568  if (m_displayingDBusReconnect) { return; }
569  if (!sGui || sGui->isShuttingDown()) { return; }
570  if (!sGui->getCoreFacade()) { return; }
571  if (!sGui->getCoreFacadeConfig().requiresDBusConnection()) { return; }
572  m_displayingDBusReconnect = true;
573  const QString dBusAddress = sGui->getCoreFacade()->getDBusAddress();
574  static const QString informativeText("Do you want to try to reconnect? 'Abort' will close the GUI.\n\nDBus: '%1'");
575  QMessageBox msgBox(this);
576  msgBox.setIcon(QMessageBox::Critical);
577  msgBox.setText("swift core not reachable!");
578  msgBox.setInformativeText(informativeText.arg(dBusAddress));
579  msgBox.setStandardButtons(QMessageBox::Retry | QMessageBox::Abort);
580  msgBox.setDefaultButton(QMessageBox::Retry);
581  const int ret = msgBox.exec();
582  if (ret == QMessageBox::Abort)
583  {
584  m_coreFailures = 0;
585  sGui->gracefulShutdown();
586  CGuiApplication::exit(EXIT_FAILURE);
587  return;
588  }
589 
590  m_displayingDBusReconnect = false;
591  CStatusMessage msg = sGui->getCoreFacade()->tryToReconnectWithDBus();
592  if (msg.isSuccess()) { m_coreFailures = 0; }
593  msg.clampSeverity(CStatusMessage::SeverityWarning);
594  CLogMessage::preformatted(msg);
595 }

615 bool SwiftGuiStd::triggerAutoPublishDialog()
616 {
617  if (!CAutoPublishData::existAutoPublishFiles()) { return false; }
618 
619  constexpr qint64 deltaT = 48 * 60 * 60 * 1000;
620  const qint64 lastDialogTs = m_lastAutoPublish.get();
621  bool showAutoPublish = lastDialogTs < 0 || (QDateTime::currentMSecsSinceEpoch() - lastDialogTs) > deltaT;
622  if (!showAutoPublish) { return false; }
623 
624  const QMessageBox::StandardButton reply = QMessageBox::question(
625  this,
626  QStringLiteral("Upload data?"),
627  QStringLiteral("Do you want to help improving swift by uploading anonymized data?"),
628  QMessageBox::Yes | QMessageBox::No);
629 
630  if (reply != QMessageBox::Yes)
631  {
632  m_lastAutoPublish.set(QDateTime::currentMSecsSinceEpoch());
633  return false;
634  }
635 
636  this->autoPublishDialog(); // updates m_lastAutoPublish
637  return true;
638 }
