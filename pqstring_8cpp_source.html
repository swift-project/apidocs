108  double CPqString::parseNumber(const QString &number, bool &success, CPqString::SeparatorMode mode)
109  {
110  double numberD = -1;
111  switch (mode)
112  {
113  case SeparatorLocale:
114  numberD = QLocale::system().toDouble(number, &success);
115  break;
116  case SeparatorQtDefault:
117  numberD = number.toDouble(&success);
118  break;
119  case SeparatorBestGuess:
120  numberD = number.toDouble(&success);
121  if (!success) { numberD = QLocale::system().toDouble(number, &success); }
122  break;
123  default:
124  qFatal("Wrong mode");
125  break;
126  }
127  return numberD;
128  }
