16  CXSwiftBusServiceProxy::CXSwiftBusServiceProxy(QDBusConnection &connection, QObject *parent, bool dummy) : QObject(parent)
17  {
18  m_dbusInterface = new BlackMisc::CGenericDBusInterface(XSWIFTBUS_SERVICE_SERVICENAME, ObjectPath(), InterfaceName(), connection, this);
19  if (!dummy)
20  {
21  bool s;
22  s = connection.connect(QString(), "/xswiftbus/service", "org.swift_project.xswiftbus.service",
23  "aircraftModelChanged", this,
24  SIGNAL(aircraftModelChanged(QString, QString, QString, QString, QString, QString, QString)));
25  Q_ASSERT(s);
26 
27  s = connection.connect(QString(), "/xswiftbus/service", "org.swift_project.xswiftbus.service",
28  "airportsInRangeUpdated", this,
29  SIGNAL(airportsInRangeUpdated(QStringList, QStringList, QList<double>, QList<double>, QList<double>)));
30  Q_ASSERT(s);
31 
32  s = connection.connect(QString(), "/xswiftbus/service", "org.swift_project.xswiftbus.service",
33  "sceneryLoaded", this,
34  SIGNAL(sceneryLoaded()));
35  Q_ASSERT(s);
36  }
37  }

49  void CXSwiftBusServiceProxy::getOwnAircraftSituationDataAsync(XPlaneData *o_xplaneData)
50  {
51  if (!o_xplaneData) { return; }
52  QPointer<CXSwiftBusServiceProxy> myself(this);
53  std::function<void(QDBusPendingCallWatcher *)> callback = [=](QDBusPendingCallWatcher *watcher) {
54  if (!myself) { return; }
55  QDBusPendingReply<double, double, double, double, double, double, double, double> reply = *watcher;
56  if (!reply.isError())
57  {
58  o_xplaneData->latitudeDeg = reply.argumentAt<0>();
59  o_xplaneData->longitudeDeg = reply.argumentAt<1>();
60  o_xplaneData->altitudeM = reply.argumentAt<2>();
61  o_xplaneData->groundspeedMs = reply.argumentAt<3>();
62  o_xplaneData->pitchDeg = reply.argumentAt<4>();
63  o_xplaneData->rollDeg = reply.argumentAt<5>();
64  o_xplaneData->trueHeadingDeg = reply.argumentAt<6>();
65  o_xplaneData->seaLevelPressureInHg = reply.argumentAt<7>();
66  }
67  watcher->deleteLater();
68  };
69  m_dbusInterface->callDBusAsync(QLatin1String("getOwnAircraftSituationData"), callback);
70  }
