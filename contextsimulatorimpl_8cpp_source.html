57  CContextSimulator::CContextSimulator(CCoreFacadeConfig::ContextMode mode, CCoreFacade *runtime) : IContextSimulator(mode, runtime),
58  CIdentifiable(this),
59  m_plugins(new CPluginManagerSimulator(this))
60  {
61  this->setObjectName("CContextSimulator");
62  CContextSimulator::registerHelp();
63 
64  Q_ASSERT_X(sApp, Q_FUNC_INFO, "Need sApp");
65  MatchingLog logMatchingMessages = CBuildConfig::isLocalDeveloperDebugBuild() ? MatchingLogAll : MatchingLogSimplified;
66  m_logMatchingMessages = logMatchingMessages;
67  m_plugins->collectPlugins();
68  this->restoreSimulatorPlugins();
69 
70  connect(&m_weatherManager, &CWeatherManager::weatherGridReceived, this, &CContextSimulator::onWeatherGridReceived, Qt::QueuedConnection);
71  connect(&m_aircraftMatcher, &CAircraftMatcher::setupChanged, this, &CContextSimulator::matchingSetupChanged);
72  connect(&CCentralMultiSimulatorModelSetCachesProvider::modelCachesInstance(), &CCentralMultiSimulatorModelSetCachesProvider::cacheChanged, this, &CContextSimulator::modelSetChanged);
73 
74  // deferred init of last model set, if no other data are set in meantime
75  const QPointer<CContextSimulator> myself(this);
76  QTimer::singleShot(2500, this, [=] {
77  if (!myself) { return; }
78  this->initByLastUsedModelSet();
79  m_aircraftMatcher.setSetup(m_matchingSettings.get());
80  if (m_aircraftMatcher.getModelSetCount() <= MatchingLogMaxModelSetSize)
81  {
82  this->enableMatchingMessages(logMatchingMessages);
83  }
84  });
85 
86  // Validation
87  m_validator = new CBackgroundValidation(this);
88  this->setValidator(this->getSimulatorPluginInfo().getSimulator());
89  connect(this, &CContextSimulator::simulatorChanged, this, &CContextSimulator::setValidator);
90  connect(m_validator, &CBackgroundValidation::validated, this, &CContextSimulator::validatedModelSet, Qt::QueuedConnection);
91 
92  m_validator->start(QThread::LowestPriority);
93  m_validator->startUpdating(60);
94  }

132  m_validator->deleteLater();

134  }

300  {
301  if (m_debugEnabled) { CLogMessage(this, CLogCategories::contextSlot()).debug() << Q_FUNC_INFO; }
302  if (removeModels.isEmpty()) { return 0; }

305 
306  CCentralMultiSimulatorModelSetCachesProvider::modelCachesInstance().synchronizeCache(simulator);
307  CAircraftModelList models = CCentralMultiSimulatorModelSetCachesProvider::modelCachesInstance().getCachedModels(simulator);
308  const int removed = models.removeModelsWithString(removeModels, Qt::CaseInsensitive);
309  if (removed > 0)
310  {
311  CCentralMultiSimulatorModelSetCachesProvider::modelCachesInstance().setCachedModels(models, simulator);
312  }

448  bool CContextSimulator::loadSimulatorPlugin(const CSimulatorPluginInfo &simulatorPluginInfo)
449  {
450  Q_ASSERT(this->getIContextApplication());
451  Q_ASSERT(this->getIContextApplication()->isUsingImplementingObject());
452  Q_ASSERT(!simulatorPluginInfo.isUnspecified());
453  Q_ASSERT(CThreadUtils::thisIsMainThread()); // only run in main thread
454 
455  // Is a plugin already loaded?
456  if (!m_simulatorPlugin.first.isUnspecified())
457  {
458  // This can happen, if a listener emitted simulatorStarted twice or two different simulators
459  // are running at the same time. In this case, we leave the loaded plugin and just return.
460  return false;
461  }
462 
463  if (!simulatorPluginInfo.isValid())
464  {
465  CLogMessage(this).error(u"Illegal plugin");
466  return false;
467  }
468 
469  ISimulatorFactory *factory = m_plugins->getFactory(simulatorPluginInfo.getIdentifier());
470  Q_ASSERT_X(factory, Q_FUNC_INFO, "no factory");
471 
472  // We assume we run in the same process as the own aircraft context
473  // Hence we pass in memory reference to own aircraft object
474  Q_ASSERT_X(this->getIContextOwnAircraft()->isUsingImplementingObject(), Q_FUNC_INFO, "Need implementing object");
475  Q_ASSERT_X(this->getIContextNetwork()->isUsingImplementingObject(), Q_FUNC_INFO, "Need implementing object");
476  IOwnAircraftProvider *ownAircraftProvider = this->getRuntime()->getCContextOwnAircraft();
477  IRemoteAircraftProvider *renderedAircraftProvider = this->getRuntime()->getCContextNetwork();
478  IClientProvider *clientProvider = this->getRuntime()->getCContextNetwork();
479  ISimulator *simulator = factory->create(simulatorPluginInfo, ownAircraftProvider, renderedAircraftProvider, &m_weatherManager, clientProvider);
480  Q_ASSERT_X(simulator, Q_FUNC_INFO, "no simulator driver can be created");
481 
482  this->setRemoteAircraftProvider(renderedAircraftProvider);
483 
484  // use simulator info from ISimulator as it can access the emulated driver settings
485  CSimulatorInfo simInfo = simulator->getSimulatorInfo();
486  if (!simInfo.isSingleSimulator())
487  {
488  BLACK_VERIFY_X(false, Q_FUNC_INFO, "Invalid simulator from plugin");
489  simInfo = CSimulatorInfo::p3d();
490  if (simulator->isEmulatedDriver())
491  {
492  CLogMessage(this).error(u"Emulated driver does NOT provide valid simulator, using default '%1', plugin: '%2'") << simInfo.toQString(true) << simulatorPluginInfo.toQString(true);
493  }
494  else
495  {
496  CLogMessage(this).error(u"Invalid driver using default '%1', plugin: '%2'") << simInfo.toQString(true) << simulatorPluginInfo.toQString(true);
497  }
498  }
499  Q_ASSERT_X(simInfo.isSingleSimulator(), Q_FUNC_INFO, "need single simulator");
500 
501  m_modelSetSimulator.set(simInfo);
502  const CAircraftModelList modelSetModels = this->getModelSet(); // synced
503  m_aircraftMatcher.setModelSet(modelSetModels, simInfo, true);
504  m_aircraftMatcher.setDefaultModel(simulator->getDefaultModel());
505 
506  bool c = connect(simulator, &ISimulator::simulatorStatusChanged, this, &CContextSimulator::onSimulatorStatusChanged);
507  Q_ASSERT(c);
508  c = connect(simulator, &ISimulator::physicallyAddingRemoteModelFailed, this, &CContextSimulator::onAddingRemoteAircraftFailed);
509  Q_ASSERT(c);
510  c = connect(simulator, &ISimulator::ownAircraftModelChanged, this, &CContextSimulator::onOwnSimulatorModelChanged);
511  Q_ASSERT(c);
512  c = connect(simulator, &ISimulator::aircraftRenderingChanged, this, &IContextSimulator::aircraftRenderingChanged);
513  Q_ASSERT(c);
514  c = connect(simulator, &ISimulator::renderRestrictionsChanged, this, &IContextSimulator::renderRestrictionsChanged);
515  Q_ASSERT(c);
516  c = connect(simulator, &ISimulator::interpolationAndRenderingSetupChanged, this, &IContextSimulator::interpolationAndRenderingSetupChanged);
517  Q_ASSERT(c);
518  c = connect(simulator, &ISimulator::airspaceSnapshotHandled, this, &IContextSimulator::airspaceSnapshotHandled);
519  Q_ASSERT(c);

1077  if (parser.matchesCommand("ris"))

1079  CInterpolationAndRenderingSetupGlobal rs = this->getInterpolationAndRenderingSetupGlobal();

1089  if (!parser.hasPart(2)) { return false; }
1090  const bool on = stringToBool(parser.part(2));
1091  if (p1 == "debug") { rs.setSimulatorDebuggingMessages(on); }
1092  else if (p1 == "parts") { rs.setEnabledAircraftParts(on); }
