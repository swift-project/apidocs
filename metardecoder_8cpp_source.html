40  class IMetarDecoderPart
41  {
42  public:
44  virtual ~IMetarDecoderPart() {}
45 
47  virtual QString getDecoderType() const = 0;
48 
49  protected:
51  virtual const QRegularExpression &getRegExp() const = 0;
52 
53  virtual bool isRepeatable() const { return false; }
54  virtual bool validateAndSet(const QRegularExpressionMatch &match, CMetar &metar) const = 0;
55  virtual bool isMandatory() const = 0;
56 
57  public:
59  bool parse(QString &metarString, CMetar &metar)
60  {
61  bool isValid = false;
62  const QRegularExpression re(getRegExp());
63  Q_ASSERT(re.isValid());
64  // Loop stop condition:
65  // - Invalid data
66  // - One match found and token not repeatable
67  do
68  {
69  QRegularExpressionMatch match = re.match(metarString);
70  if (match.hasMatch())
71  {
72  // If invalid data, we return straight away
73  if (!validateAndSet(match, metar))
74  {
75  return false;
76  }
77 
78  // Remove the captured part
79  metarString.replace(re, QString());
80  isValid = true;
81  }
82  else
83  {
84  // No (more) match found.
85  if (!isMandatory()) { isValid = true; }
86  break;
87  }
88  }
89  while (isRepeatable());
90 
91  if (!isValid)
92  {
93  CLogMessage(static_cast<CMetarDecoder *>(nullptr)).debug() << "Failed to match" << getDecoderType() << "in remaining METAR:" << metarString;
94  }
95  return isValid;
96  }
97  };

249  bool validateAndSet(const QRegularExpressionMatch &match, CMetar &metar) const override
250  {
251  bool ok = false;
252  QString directionAsString = match.captured("direction");
253  if (directionAsString == "///") return true;
254  int direction = 0;
255  bool directionVariable = false;
256  if (directionAsString == "VRB")
257  {
258  directionVariable = true;
259  }
260  else
261  {
262  direction = directionAsString.toInt(&ok);
263  if (!ok) return false;
264  }
265 
266  QString speedAsString = match.captured("speed");
267  if (speedAsString == "//") return true;
268  int speed = speedAsString.toInt(&ok);
269  if (!ok) return false;
270  QString gustAsString = match.captured("gustSpeed");
271  int gustSpeed = 0;
272  if (!gustAsString.isEmpty())
273  {
274  gustSpeed = gustAsString.toInt(&ok);
275  if (!ok) return false;
276  }
277  QString unitAsString = match.captured("unit");
278  if (!getWindUnitHash().contains(unitAsString)) return false;
279 
280  CWindLayer windLayer(CAltitude(0, CAltitude::AboveGround, CLengthUnit::ft()), CAngle(direction, CAngleUnit::deg()), CSpeed(speed, getWindUnitHash().value(unitAsString)),
281  CSpeed(gustSpeed, getWindUnitHash().value(unitAsString)));
282  windLayer.setDirectionVariable(directionVariable);
283  metar.setWindLayer(windLayer);
284  return true;
285  }
286 
287  virtual bool isMandatory() const override { return false; }
288 
289  private:
290  QString getRegExpImpl() const
291  {
292  // Wind direction in three digits, 'VRB' or /// if no info available
293  const QString direction = QStringLiteral("(?<direction>\\d{3}|VRB|/{3})");
294  // Wind speed in two digits (or three digits if required)
295  const QString speed = QStringLiteral("(?<speed>\\d{2,3}|/{2})");
296  // Optional: Gust in two digits (or three digits if required)
297  const QString gustSpeed = QStringLiteral("(G(?<gustSpeed>\\d{2,3}))?");
298  // Unit
299  const QString unit = QStringLiteral("(?<unit>") + QStringList(getWindUnitHash().keys()).join('|') + ")";
300  // Regexp
301  const QString regexp = "^" + direction + speed + gustSpeed + unit + " ?";
302  return regexp;
303  }

352  class CMetarDecoderVisibility : public IMetarDecoderPart
353  {
354  public:
355  virtual QString getDecoderType() const override { return "Visibility"; }
356 
357  protected:
358  const QHash<QString, QString> &getCardinalDirections() const
359  {
360  static const QHash<QString, QString> hash = {
361  { "N", "north" },
362  { "NE", "north-east" },
363  { "E", "east" },
364  { "SE", "south-east" },
365  { "S", "south" },
366  { "SW", "south-west" },
367  { "W", "west" },
368  { "NW", "north-west" },
369  };
370  return hash;
371  }
372 
373  const QRegularExpression &getRegExp() const override
374  {
375  static const QRegularExpression re(getRegExpImpl());
376  return re;
377  }
378 
379  bool validateAndSet(const QRegularExpressionMatch &match, CMetar &metar) const override
380  {
381  bool ok = false;
382  if (!match.captured("cavok").isEmpty())
383  {
384  metar.setCavok();
385  return true;
386  }
387  QString visibilityAsString = match.captured("visibility");
388  if (visibilityAsString == "////") return true;
389 
390  double visibility = 0;
391  if (!visibilityAsString.isEmpty())
392  {
393  visibility = visibilityAsString.toDouble(&ok);
394  if (!ok) return false;
395  metar.setVisibility(CLength(visibility, CLengthUnit::m()));
396  return true;
397  }
398 
399  QString distanceAsString = match.captured("distance");
400  if (!distanceAsString.isEmpty())
401  {
402  visibility += distanceAsString.toDouble(&ok);
403  if (!ok) return false;
404  }
405  QString numeratorAsString = match.captured("numerator");
406  QString denominatorAsString = match.captured("denominator");
407  if (!numeratorAsString.isEmpty() && !denominatorAsString.isEmpty())
408  {
409 
410  double numerator = numeratorAsString.toDouble(&ok);
411  if (!ok) return false;
412 
413  double denominator = denominatorAsString.toDouble(&ok);
414  if (!ok) return false;
415  if (denominator < 1 || numerator < 1) return false;
416  visibility += (numerator / denominator);
417  }
418 
419  QString unitAsString = match.captured("unit");
420  CLengthUnit unit = CLengthUnit::SM();
421  if (unitAsString == "KM") unit = CLengthUnit::km();
422 
423  metar.setVisibility(CLength(visibility, unit));
424 
425  return true;
426  }
427 
428  virtual bool isMandatory() const override { return false; }
429 
430  private:
431  QString getRegExpImpl() const
432  {
433  // CAVOK
434  const QString cavok = QStringLiteral("(?<cavok>CAVOK)");
435  // European version:
436  // Visibility of 4 digits in meter
437  // Cardinal directions N, NE etc.
438  // "////" in case no info is available
439  // NDV = No Directional Variation
440  const QString visibility_eu = QStringLiteral("(?<visibility>\\d{4}|/{4})(NDV)?") + "(" + QStringList(getCardinalDirections().keys()).join('|') + ")?";
441  // US/Canada version:
442  // Surface visibility reported in statute miles.
443  // A space divides whole miles and fractions.
444  // Group ends with SM to indicate statute miles. For example,
445  // 1 1/2SM.
446  // Auto only: M prefixed to value < 1/4 mile, e.g., M1/4S
447  const QString visibility_us = QStringLiteral("(?<distance>\\d{0,2}) ?M?((?<numerator>\\d)/(?<denominator>\\d))?(?<unit>SM|KM)");
448  const QString regexp = "^(" + cavok + "|" + visibility_eu + "|" + visibility_us + ") ";
449  return regexp;
450  }

486  private:
487  QString getRegExpImpl() const
488  {
489  // 10 Minute RVR value: Reported in hundreds of feet if visibility is one statute mile
490  // or RVR is 6000 feet. Group ends with FT to indicate feet. For example, R06L/2000FT.
491  // The RVR value is prefixed with either M or P to indicate the value is lower or higher
492  // than the RVR reportable values, e.g., R06L/P6000FT. If the RVR is variable during the 10
493  // minute evaluation period, the variability is reported, e.g., R06L/2000V4000FT
494 
495  // Runway
496  const QString runway = QStringLiteral("R(?<runway>\\d{2}[LCR]*)");
497  // Visibility
498  const QString visibility = QStringLiteral("/[PM]?(?<rwy_visibility>\\d{4})");
499  // Variability
500  static const QString variability = QStringLiteral("V?(?<variability>\\d{4})?");
501  // Unit
502  const QString unit = QStringLiteral("(?<unit>FT)?");
503  // Trend
504  const QString trend = QStringLiteral("/?(?<trend>[DNU])?");
505  const QString regexp = "^" + runway + visibility + variability + unit + trend + " ";
506  return regexp;
