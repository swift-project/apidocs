24  void CConfig::parse()
25  {
26  std::ifstream configFile(m_filePath);
27  if (!configFile.is_open()) { return; }
28 
29  std::string line;
30  for (int lineNo = 1; std::getline(configFile, line); ++lineNo)
31  {
32  line.erase(std::remove_if(line.begin(), line.end(), isspace), line.end());
33  if (line.empty() || line[0] == '#') { continue; }
34 
35  auto delimiterPos = line.find("=");
36  if (delimiterPos == std::string::npos)
37  {
38  WARNING_LOG("xswiftbus.conf line " + std::to_string(lineNo) + ": Skipping invalid line!");
39  continue;
40  }
41 
42  std::string key = line.substr(0, delimiterPos);
43  std::string value = line.substr(delimiterPos + 1);
44 
45  if (key.empty() || value.empty())
46  {
47  WARNING_LOG("xswiftbus.conf line " + std::to_string(lineNo) + ": Skipping invalid line!");
48  continue;
49  }
50 
51  bool valid = true;
52  if (stringCompareCaseInsensitive(key, "dbusMode")) { valid = parseDBusMode(value); }
53  else if (stringCompareCaseInsensitive(key, "dbusAddress")) { valid = parseDBusAddress(value); }
54  else if (stringCompareCaseInsensitive(key, "dbusPort")) { valid = parseDBusPort(value); }
55  else if (stringCompareCaseInsensitive(key, "debug")) { valid = parseDebug(value); }
56  else if (stringCompareCaseInsensitive(key, "tcas")) { valid = parseTcas(value); }
57  else
58  {
59  WARNING_LOG("xswiftbus.conf line " + std::to_string(lineNo) + ": Unknown variable " + key + "!");
60  continue;
61  }
62 
63  if (!valid)
64  {
65  WARNING_LOG("xswiftbus.conf line " + std::to_string(lineNo) + ": Skipping invalid line!");
66  continue;
67  }
68  }
69  }

86  bool CConfig::writeConfigFile() const
87  {
88  std::ofstream configFile(m_filePath, std::ofstream::out | std::ofstream::trunc);
89  if (!configFile.is_open()) { return false; }
90 
91  // this code should be similar to CXSwiftBusConfigWriter
92  configFile << "# DBus Mode - Options: p2p, session" << std::endl;
93  configFile << "dbusMode = " << toLower(dbusModeToString(m_dbusMode)) << std::endl;
94  configFile << std::endl;
95  configFile << "# DBus server address - relevant for P2P mode only" << std::endl;
96  configFile << "dbusAddress = " << m_dbusAddress << std::endl;
97  configFile << std::endl;
98  configFile << "# DBus server port - relevant for P2P mode only" << std::endl;
99  configFile << "dbusPort = " << m_dbusPort << std::endl;
100  configFile << std::endl;
101  configFile << "# Render phase debugging - to help diagnose crashes" << std::endl;
102  configFile << "debug = " << boolToOnOff(m_debug) << std::endl;
103  configFile << std::endl;
104  configFile << "# TCAS traffic - to disable in case of crashes" << std::endl;
105  configFile << "tcas = " << boolToOnOff(m_tcas) << std::endl;
106 
107  // for info
108  const auto clockNow = std::chrono::system_clock::now();
109  const time_t now = std::chrono::system_clock::to_time_t(clockNow);
110  struct tm tms;
111 #if defined(IBM)
112  localtime_s(&tms, &now);
113 #else
114  localtime_r(&now, &tms);
115 #endif
116  configFile << std::endl;
117  configFile << "# Updated by XSwiftBus plugin " << std::put_time(&tms, "%T");
118  configFile << std::endl;
119  configFile.close();
120  return true;
121  }
