106  QVariant CApplicationInfo::propertyByIndex(CPropertyIndexRef index) const
107  {
108  if (index.isMyself()) { return QVariant::fromValue(*this); }
109  const ColumnIndex i = index.frontCasted<ColumnIndex>();
110  switch (i)
111  {
112  case IndexApplication: return QVariant::fromValue(m_app);
113  case IndexApplicationAsString: return QVariant::fromValue(this->getApplicationAsString());
114  case IndexApplicationDataPath: return QVariant::fromValue(this->getApplicationDataDirectory());
115  case IndexCompileInfo: return QVariant::fromValue(this->getCompileInfo());
116  case IndexExecutablePath: return QVariant::fromValue(this->getExecutablePath());
117  case IndexExecutablePathExisting: return QVariant::fromValue(this->isExecutablePathExisting());
118  case IndexPlatformInfo: return QVariant::fromValue(this->getPlatform());
119  case IndexProcessInfo: return m_process.propertyByIndex(index.copyFrontRemoved());
120  case IndexVersionString: return QVariant::fromValue(this->getVersionString());
121  case IndexWordSize: return QVariant::fromValue(this->getWordSize());
122  default: break;
123  }
124  return CValueObject::propertyByIndex(index);
125  }

127  void CApplicationInfo::setPropertyByIndex(CPropertyIndexRef index, const QVariant &variant)
128  {
129  if (index.isMyself())
130  {
131  (*this) = variant.value<CApplicationInfo>();
132  return;
133  }
134  const ColumnIndex i = index.frontCasted<ColumnIndex>();
135  switch (i)
136  {
137  case IndexApplication: this->setApplication(static_cast<Application>(variant.toInt())); break;
138  case IndexApplicationAsString: break;
139  case IndexApplicationDataPath: this->setApplicationDataDirectory(variant.toString()); break;
140  case IndexCompileInfo: this->setCompileInfo(variant.toString()); break;
141  case IndexExecutablePath: this->setExecutablePath(variant.toString()); break;
142  case IndexExecutablePathExisting: break;
143  case IndexPlatformInfo: this->setPlatformInfo(variant.toString()); break;
144  case IndexProcessInfo: m_process.setPropertyByIndex(index.copyFrontRemoved(), variant); break;
145  case IndexVersionString: this->setVersionString(variant.toString()); break;
146  case IndexWordSize: this->setWordSize(variant.toInt()); break;
147  default: break;
148  }
149  CValueObject::setPropertyByIndex(index, variant);
150  }

152  int CApplicationInfo::comparePropertyByIndex(CPropertyIndexRef index, const CApplicationInfo &compareValue) const
153  {
154  if (index.isMyself()) { return this->getExecutablePath().compare(compareValue.getExecutablePath()); }
155  const ColumnIndex i = index.frontCasted<ColumnIndex>();
156  switch (i)
157  {
158  case IndexApplicationDataPath: return this->getApplicationDataDirectory().compare(compareValue.getApplicationDataDirectory());
159  case IndexCompileInfo: return this->getCompileInfo().compare(compareValue.getCompileInfo());
160  case IndexExecutablePath: return this->getExecutablePath().compare(compareValue.getExecutablePath());
161  case IndexExecutablePathExisting: return Compare::compare(this->isExecutablePathExisting(), compareValue.isExecutablePathExisting());
162  case IndexPlatformInfo: return this->getPlatform().compare(compareValue.getPlatform());
163  case IndexProcessInfo: return this->getProcessInfo().processName().compare(compareValue.getProcessInfo().processName());
164  case IndexVersionString: return this->getVersionString().compare(compareValue.getVersionString());
165  case IndexWordSize: return Compare::compare(this->getWordSize(), compareValue.getWordSize());
166  case IndexApplication:
167  case IndexApplicationAsString:
168  return Compare::compare(m_app, compareValue.m_app);
169  default: return CValueObject::comparePropertyByIndex(index.copyFrontRemoved(), compareValue);
170  }
171  }

215  CApplicationInfo::Application CApplicationInfo::guessApplication()
216  {
217  const QString a(QCoreApplication::instance()->applicationName().toLower());
218  if (a.contains("test")) { return CApplicationInfo::UnitTest; } // names like testcore
219  if (a.contains("sample")) { return CApplicationInfo::Sample; }
220  if (a.contains("core")) { return CApplicationInfo::PilotClientCore; }
221  if (a.contains("launcher")) { return CApplicationInfo::Laucher; }
222  if (a.contains("gui")) { return CApplicationInfo::PilotClientGui; }
223  if (a.contains("data") || a.contains("mapping")) { return CApplicationInfo::MappingTool; }
224  return CApplicationInfo::Unknown;
225  }
