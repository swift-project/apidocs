138  void CRadarComponent::refreshTargets()
139  {
140  if (!sGui || sGui->isShuttingDown()) { return; }
141 
142  qDeleteAll(m_radarTargets.childItems());
143 
144  if (sGui->getIContextNetwork() && sGui->getIContextNetwork()->isConnected())
145  {
146  if (isVisibleWidget())
147  {
148  const CSimulatedAircraftList aircraft = sGui->getIContextNetwork()->getAircraftInRange();
149  for (const CSimulatedAircraft &sa : aircraft)
150  {
151  const double distanceNM = sa.getRelativeDistance().value(CLengthUnit::NM());
152  const double bearingRad = sa.getRelativeBearing().value(CAngleUnit::rad());
153  const int groundSpeedKts = sa.getGroundSpeed().valueInteger(CSpeedUnit::kts());
154 
155  QPointF position(polarPoint(distanceNM, bearingRad));
156 
157  QGraphicsEllipseItem *dot = new QGraphicsEllipseItem(-2.0, -2.0, 4.0, 4.0, &m_radarTargets);
158  dot->setPos(position);
159  dot->setPen(m_radarTargetPen);
160  dot->setBrush(m_radarTargetPen.color());
161  dot->setFlags(QGraphicsItem::ItemIgnoresTransformations);
162 
163  QGraphicsTextItem *tag = new QGraphicsTextItem(&m_radarTargets);
164  QString tagText;
165  if (ui->cb_Callsign->isChecked())
166  {
167  tagText += sa.getCallsignAsString() % u"\n";
168  }
169  if (ui->cb_Altitude->isChecked())
170  {
171  int flightLeveL = sa.getAltitude().valueInteger(CLengthUnit::ft()) / 100;
172  tagText += u"FL" % QStringLiteral("%1").arg(flightLeveL, 3, 10, QChar('0'));
173  }
174  if (ui->cb_GroundSpeed->isChecked())
175  {
176  if (!tagText.isEmpty()) tagText += QStringLiteral(" ");
177  tagText += QString::number(groundSpeedKts) % u" kt";
178  }
179 
180  tag->setPlainText(tagText);
181  tag->setFont(m_tagFont);
182  tag->setPos(position);
183  tag->setDefaultTextColor(Qt::green);
184  tag->setFlags(QGraphicsItem::ItemIgnoresTransformations);
185 
186  if (ui->cb_Heading->isChecked() && groundSpeedKts > 3.0)
187  {
188  const double headingRad = sa.getHeading().value(CAngleUnit::rad());
189  QPen pen(Qt::green, 1);
190  pen.setCosmetic(true);
191  QGraphicsLineItem *li = new QGraphicsLineItem(QLineF({ 0.0, 0.0 }, polarPoint(5.0, headingRad)), &m_radarTargets);
192  li->setPos(position);
193  li->setPen(pen);
194  }
195  }
196  }
197  }

200  void CRadarComponent::rotateView()
201  {
202  if (sGui->getIContextOwnAircraft())
203  {
204  if (isVisibleWidget())
205  {
206  int headingDegree = 0;
207  if (!ui->cb_LockNorth->isChecked())
208  {
209  headingDegree = sGui->getIContextOwnAircraft()->getOwnAircraftSituation().getHeading().valueInteger(CAngleUnit::deg());
210  }
211 
212  if (m_rotatenAngle != headingDegree)
213  {
214  // Rotations are summed up, hence rotate back before applying the new rotation.
215  // Doing a global transformation reset will not work as it resets also zooming.
216  ui->gv_RadarView->rotate(m_rotatenAngle);
217  ui->gv_RadarView->rotate(-headingDegree);
218  m_rotatenAngle = headingDegree;
219  }
220  }
221  }

236  void CRadarComponent::changeRangeInSteps(bool zoomIn)
237  {
238  qreal direction = zoomIn ? 1.0 : -1.0;
239  double factor = 10.0;
240  if (m_rangeNM < 10.0 || (qFuzzyCompare(m_rangeNM, 10.0) && zoomIn))
241  {
242  factor = 1.0;
243  }
244 
245  if (m_rangeNM < 1.0 || (qFuzzyCompare(m_rangeNM, 1.0) && zoomIn))
246  {
247  factor = 0.5;
248  }
249 
250  m_rangeNM = m_rangeNM - direction * factor;
251  m_rangeNM = qMin(90.0, qMax(0.5, m_rangeNM));
252  ui->cb_RadarRange->setCurrentText(QString::number(m_rangeNM) % u" nm");
253  fitInView();
254  }

289  QPointF CRadarComponent::polarPoint(double distance, double angleRadians)
290  {
291  angleRadians = -angleRadians; // conversion assumes angles are counterclockwise
292 
293  // standard conversion from https://en.wikipedia.org/wiki/Polar_coordinate_system
294  QPointF p(distance * qCos(angleRadians), distance * qSin(angleRadians));
295 
296  // conversion yields a coordinate system
297  // in which North=(1,0) and East=(-1,0)
298  // but we want North=(0,-1) and East=(0,1)
299  // (QGraphicsView y axis increases downwards)
300  std::swap(p.rx(), p.ry());
301  p.setX(-p.x());
302  p.setY(-p.y());
303  return p;
304  }
