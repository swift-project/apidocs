97  bool CThreadedReader::didContentChange(const QString &content, int startPosition)
98  {
99  uint oldHash = 0;
100  {
101  QReadLocker rl(&m_lock);
102  oldHash = m_contentHash;
103  }
104  uint newHash = qHash(startPosition < 0 ? content : content.mid(startPosition));
105  if (oldHash == newHash) { return false; }
106  {
107  QWriteLocker wl(&m_lock);
108  m_contentHash = newHash;
109  }
110  return true;
111  }

135  void CThreadedReader::setInitialAndPeriodicTime(int initialTime, int periodicTime)
136  {
137  m_initialTime = initialTime;
138  m_periodicTime = periodicTime;
139 
140  // if timer is active start with delta time
141  // remark: will be reset in doWork
142  if (m_updateTimer.isActive())
143  {
144  const int oldPeriodicTime = m_updateTimer.interval();
145  const int delta = m_periodicTime - oldPeriodicTime + m_updateTimer.remainingTime();
146  m_updateTimer.start(qMax(delta, 0));
147  }
148  }

158  bool CThreadedReader::doWorkCheck() const
159  {
160  // sApp->hasWebDataServices() cannot be used, as some readers are already used during init phase
161  if (!this->isEnabled()) { return false; }
162 
163  // MS 2019-02-23 isAbandoned() check only makes sense when called by worker thread (T541)
164  if (CThreadUtils::isInThisThread(this) && this->isAbandoned()) { return false; }
165 
166  if (!m_unitTest && (!sApp || sApp->isShuttingDown())) { return false; }
167  return true;
168  }

170  QNetworkReply *CThreadedReader::getFromNetworkAndLog(const CUrl &url, const CSlot<void(QNetworkReply *)> &callback)
171  {
172  QWriteLocker wl(&m_lock);
173  const CUrlLogList outdatedPendingUrls = m_urlReadLog.findOutdatedPending(OutdatedPendingCallMs);
174  if (!outdatedPendingUrls.isEmpty())
175  {
176  CLogMessage(this).warning(u"Detected outdated reader pending calls: '%1'") << outdatedPendingUrls.toQString(true);
177  m_urlReadLog.removeOlderThanNowMinusOffset(OutdatedPendingCallMs); // clean up
178  }
179 
180  const int id = m_urlReadLog.addPendingUrl(url);
181  wl.unlock();
182 
183  // returned QNetworkReply normally nullptr since QAM is in different thread
184  QNetworkReply *nr = sApp->getFromNetwork(url, id, callback, { this, &CThreadedReader::networkReplyProgress });
185  return nr;
186  }
