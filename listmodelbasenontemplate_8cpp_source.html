159  void CListModelBaseNonTemplate::emitDataChanged(int startRowIndex, int endRowIndex)
160  {
161  BLACK_VERIFY_X(startRowIndex <= endRowIndex, Q_FUNC_INFO, "check rows");
162  BLACK_VERIFY_X(startRowIndex >= 0 && endRowIndex >= 0, Q_FUNC_INFO, "check rows");
163 
164  const int columns = columnCount();
165  const int rows = rowCount();
166  if (columns < 1) { return; }
167  if (startRowIndex < 0) { startRowIndex = 0; }
168  if (endRowIndex >= rows) { endRowIndex = rows - 1; }
169  const QModelIndex topLeft(createIndex(startRowIndex, 0));
170  const QModelIndex bottomRight(createIndex(endRowIndex, columns - 1));
171  emit this->dataChanged(topLeft, bottomRight);
172  }
