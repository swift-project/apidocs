37  void CManagedStatusBar::initStatusBar(QStatusBar *statusBar)
38  {
39  if (m_statusBar) { return; }
40  m_ownedStatusBar = statusBar ? false : true;
41  m_statusBar = statusBar ? statusBar : new QStatusBar();
42  if (m_statusBar->objectName().isEmpty()) { m_statusBar->setObjectName("sb_ManagedStatusBar"); }
43  // if (m_ownedStatusBar) { m_statusBar->setSizePolicy(QSizePolicy::MinimumExpanding, QSizePolicy::Preferred);}
44 
45  m_statusBarIcon = new QLabel(m_statusBar);
46  m_statusBarIcon->setObjectName(QString("lbl_StatusBarIcon").append(m_statusBar->objectName()));
47  m_statusBarLabel = new QLabel(m_statusBar);
48  m_statusBarLabel->setSizePolicy(QSizePolicy::MinimumExpanding, QSizePolicy::Preferred);
49  m_statusBarLabel->setObjectName(QString("lbl_StatusBarLabel").append(m_statusBar->objectName()));
50 
51  // use insert to insert from left to right
52  // this keeps any grip on the right size
53  m_statusBar->insertPermanentWidget(0, m_statusBarIcon, 0); // status icon
54  m_statusBar->insertPermanentWidget(1, m_statusBarLabel, 1); // status text
55 
56  // timer
57  m_timerStatusBar.setObjectName(this->objectName().append(":m_timerStatusBar"));
58  m_timerStatusBar.setSingleShot(true);
59  connect(&m_timerStatusBar, &QTimer::timeout, this, &CManagedStatusBar::clearStatusBar);
60 
61  // done when injected status bar
62  if (m_ownedStatusBar)
63  {
64  // self created status bar
65  QSizePolicy sizePolicy(QSizePolicy::MinimumExpanding, QSizePolicy::Preferred);
66  sizePolicy.setHorizontalStretch(0);
67  sizePolicy.setVerticalStretch(0);
68  sizePolicy.setHeightForWidth(m_statusBar->sizePolicy().hasHeightForWidth());
69 
70  m_statusBar->setMinimumHeight(24); // with no minimum height the layout always adjustes when displaying a status message
71  m_statusBar->setSizePolicy(sizePolicy);
72  m_statusBar->setSizeGripEnabled(false);
73  }
74  }

94  void CManagedStatusBar::displayStatusMessage(const CStatusMessage &statusMessage)
95  {
96  Q_ASSERT_X(m_statusBarIcon, Q_FUNC_INFO, "Missing status bar icon");
97  Q_ASSERT_X(m_statusBar, Q_FUNC_INFO, "Missing status bar");
98  Q_ASSERT_X(m_statusBarLabel, Q_FUNC_INFO, "Missing label");
99 
100  // already displaying a message with severity higher than this one?
101  if (statusMessage.isEmpty()) { return; }
102  if (!statusMessage.isSeverityHigherOrEqual(m_currentSeverity)) { return; }
103 
104  // used with log subscriber, make sure it is not displayed twice
105  if (statusMessage.wasHandledBy(this)) { return; }
106  statusMessage.markAsHandledBy(this);
107 
108  this->show();
109  m_timerStatusBar.start(3000); // start / restart
110  m_statusBarIcon->setPixmap(CIcon(statusMessage.toIcon()));
111 
112  // restrict size for own status bars
113  const QSize size = m_statusBar->window()->size();
114  const int w = qRound(0.95 * size.width());
115  if (m_ownedStatusBar) { m_statusBar->setMaximumWidth(w); }
116 
117  if (m_elideMode != Qt::ElideNone)
118  {
119  const QFontMetrics metrics(m_statusBarLabel->font());
120  const QString elidedText = metrics.elidedText(statusMessage.getMessage(), m_elideMode, qRound(0.90 * w));
121  m_statusBarLabel->setText(elidedText);
122  }
123  else
124  {
125  m_statusBarLabel->setText(statusMessage.getMessage());
126  }
127  m_currentSeverity = statusMessage.getSeverity();
128  }
