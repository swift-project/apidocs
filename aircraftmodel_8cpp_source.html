93  obj.insert("simp3d", QJsonValue(flag));
94  flag = CDatastoreUtility::boolToDbYN(sim.isFS9());
95  obj.insert("simfs9", QJsonValue(flag));
96  flag = CDatastoreUtility::boolToDbYN(sim.isXPlane());
97  obj.insert("simxplane", QJsonValue(flag));

104  obj.insert("idlivery", this->getLivery().getDbKeyAsJsonValue());
105  obj.insert("idairlineicao", this->getLivery().getAirlineIcaoCode().getDbKeyAsJsonValue()); // not really needed if livery is complete

263  void CAircraftModel::setPropertyByIndex(CPropertyIndexRef index, const QVariant &variant)
264  {
265  if (index.isMyself())
266  {
267  (*this) = variant.value<CAircraftModel>();
268  return;
269  }
270  if (IOrderable::canHandleIndex(index))
271  {
272  IOrderable::setPropertyByIndex(index, variant);
273  return;
274  }
275  if (IDatastoreObjectWithIntegerKey::canHandleIndex(index))
276  {
277  IDatastoreObjectWithIntegerKey::setPropertyByIndex(index, variant);
278  return;
279  }
280 
281  const ColumnIndex i = index.frontCasted<ColumnIndex>();
282  switch (i)
283  {
284  case IndexModelString: m_modelString = variant.toString(); break;
285  case IndexModelStringAlias: m_modelStringAlias = variant.toString(); break;
286  case IndexAircraftIcaoCode: m_aircraftIcao.setPropertyByIndex(index.copyFrontRemoved(), variant); break;
287  case IndexLivery: m_livery.setPropertyByIndex(index.copyFrontRemoved(), variant); break;
288  case IndexDistributor: m_distributor.setPropertyByIndex(index.copyFrontRemoved(), variant); break;
289  case IndexDescription: m_description = variant.toString(); break;
290  case IndexSimulatorInfo: m_simulator.setPropertyByIndex(index.copyFrontRemoved(), variant); break;
291  case IndexName: m_name = variant.toString(); break;
292  case IndexIconPath: m_iconFile = variant.toString(); break;
293  case IndexCG: m_cg.setPropertyByIndex(index.copyFrontRemoved(), variant); break;
294  case IndexSupportedParts: this->setSupportedParts(variant.toString()); break;
295  case IndexModelType: m_modelType = variant.value<ModelType>(); break;
296  case IndexFileName: m_fileName = variant.toString(); break;
297  case IndexCallsign:
298  m_callsign.setPropertyByIndex(index.copyFrontRemoved(), variant);
299  m_callsign.setTypeHint(CCallsign::Aircraft);
300  break;
301  case IndexFileTimestamp:
302  if (variant.canConvert<QDateTime>())
303  {
304  this->setFileTimestamp(variant.value<QDateTime>());
305  }
306  else if (variant.canConvert<qint64>())
307  {
308  m_fileTimestamp = variant.value<qint64>();
309  }
310  break;
311  case IndexModelMode:
312  if (static_cast<QMetaType::Type>(variant.type()) == QMetaType::QString)
313  {
314  this->setModelModeAsString(variant.toString());
315  }
316  else
317  {
318  m_modelMode = variant.value<ModelMode>();
319  }
320  break;
321  // no setter indexes ignored
322  case IndexHasQueriedModelString: break;
323  case IndexModelTypeAsString: break;
324  case IndexModelModeAsString: break;
325  case IndexModelModeAsIcon: break;
326  case IndexFileTimestampFormattedYmdhms: break;
327  case IndexSimulatorInfoAsString: break;
328  case IndexMembersDbStatus: break;
329 
330  default: CValueObject::setPropertyByIndex(index, variant); break;
331  }

334  int CAircraftModel::comparePropertyByIndex(CPropertyIndexRef index, const CAircraftModel &compareValue) const
335  {
336  if (IDatastoreObjectWithIntegerKey::canHandleIndex(index)) { return IDatastoreObjectWithIntegerKey::comparePropertyByIndex(index, compareValue); }
337  if (IOrderable::canHandleIndex(index)) { return IOrderable::comparePropertyByIndex(index, compareValue); }
338  if (index.isMyself()) { return m_modelString.compare(compareValue.getModelString(), Qt::CaseInsensitive); }
339  const ColumnIndex i = index.frontCasted<ColumnIndex>();
340  switch (i)
341  {
342  case IndexModelString: return m_modelString.compare(compareValue.getModelString(), Qt::CaseInsensitive);
343  case IndexModelStringAlias: return m_modelStringAlias.compare(compareValue.getModelStringAlias(), Qt::CaseInsensitive);
344  case IndexAllModelStrings: return this->getAllModelStringsAndAliases().compare(compareValue.getAllModelStringsAndAliases(), Qt::CaseInsensitive);
345  case IndexHasQueriedModelString: return Compare::compare(this->hasQueriedModelString(), compareValue.hasQueriedModelString());
346  case IndexAircraftIcaoCode: return m_aircraftIcao.comparePropertyByIndex(index.copyFrontRemoved(), compareValue.getAircraftIcaoCode());
347  case IndexLivery: return m_livery.comparePropertyByIndex(index.copyFrontRemoved(), compareValue.getLivery());
348  case IndexDistributor: return m_distributor.comparePropertyByIndex(index.copyFrontRemoved(), compareValue.getDistributor());
349  case IndexDescription: return m_description.compare(compareValue.getDescription(), Qt::CaseInsensitive);
350  case IndexName: return m_name.compare(compareValue.getName(), Qt::CaseInsensitive);
351  case IndexCallsign: return m_callsign.comparePropertyByIndex(index.copyFrontRemoved(), compareValue.getCallsign());
352  case IndexFileName: return m_fileName.compare(compareValue.getFileName(), Qt::CaseInsensitive);
353  case IndexIconPath: return m_iconFile.compare(compareValue.getIconFile(), Qt::CaseInsensitive);
354  case IndexCG: return m_cg.comparePropertyByIndex(index.copyFrontRemoved(), compareValue.getCG());
355  case IndexSupportedParts: return m_supportedParts.compare(compareValue.getSupportedParts());
356  case IndexModelTypeAsString:
357  case IndexModelType: return Compare::compare(m_modelType, compareValue.getModelType());
358  case IndexSimulatorInfoAsString:
359  case IndexSimulatorInfo: return m_simulator.comparePropertyByIndex(index.copyFrontRemoved(), compareValue.getSimulator());
360  case IndexFileTimestamp:
361  case IndexFileTimestampFormattedYmdhms: return Compare::compare(m_fileTimestamp, compareValue.m_fileTimestamp);
362  case IndexModelMode:
363  case IndexModelModeAsString:
364  case IndexModelModeAsIcon: return Compare::compare(m_modelMode, compareValue.getModelMode());
365  case IndexMembersDbStatus: return getMembersDbStatus().compare(compareValue.getMembersDbStatus());
366  default: break;
367  }
368  Q_ASSERT_X(false, Q_FUNC_INFO, "No comparison");
369  return 0;
370  }
