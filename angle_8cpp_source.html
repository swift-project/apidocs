50  CAngle::DegMinSecFractionalSec CAngle::asSexagesimalDegMinSec(bool range180Degrees) const
51  {
52  double dms = this->value(CAngleUnit::sexagesimalDeg());
53 
54  if (range180Degrees)
55  {
56  dms = std::fmod(dms + 180.0, 360.0);
57  dms += (dms < 0) ? 180.0 : -180.0;
58  }
59 
60  DegMinSecFractionalSec values;
61  if (dms < 0)
62  {
63  values.sign = -1;
64  dms *= -1.0;
65  }
66 
67  QString str = QStringLiteral("%1").arg(dms, 14, 'f', 10, '0'); // 000.0000000000
68  values.deg = str.midRef(0, 3).toInt();
69  values.min = str.midRef(4, 2).toInt();
70  values.sec = str.midRef(6, 2).toInt();
71  values.fractionalSec = str.midRef(8, 6).toInt() / 1000000.0;
72  return values;
73  }
