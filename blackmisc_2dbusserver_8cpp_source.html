19  CDBusServer::CDBusServer(const QString &service, const QString &address, QObject *parent) : QObject(parent)
20  {
21  // Application Options:
22  // -h, –host=HOSTNAME Hostname or IP of the remote host
23  // -p, –port-ssh=PORT-SSH SSH port on the remote host
24  // -u, –username=USERNAME SSH username on the remote host
25  // -w, –password=PASSWORD SSH password on the remote host
26  // -m, –method=DBUS_TRANSPORT_METHOD The D-Bus transport method to use (TCP, UNIX, abstract-UNIX)
27  // -b, –bind=HOSTNAME The bind-address to listen for D-Bus client connections on
28  // -d, –bus-address=BUS_ADDRESS The DBus session bus address of the remote D-Bus daemon
29  // -t, –port-tcp=PORT-TCP The TCP port to listen for DBus client connections on
30  // -v, –verbose=VERBOSE Set verbosity level (3, 2, 1, 0, -1)=(packet,protocol,functions,important,none)
31 
32  static const QString desc("Mode: %1 Address: '%2' Service: '%3'");
33  m_serverMode = CDBusServer::modeOfAddress(address);
34  this->setObjectName(desc.arg(CDBusServer::modeToString(m_serverMode), address, service.isEmpty() ? "-" : service));
35  switch (m_serverMode)
36  {
37  case SERVERMODE_SESSIONBUS:
38  {
39  QDBusConnection connection = QDBusConnection::connectToBus(QDBusConnection::SessionBus, coreServiceName());
40  connection.unregisterService(service); // allow reconnecting by removing still registered service
41  if (!connection.isConnected() || !connection.registerService(service))
42  {
43  // registration fails can either mean something wrong with DBus or service already exists
44  CLogMessage(this).warning(u"DBus registration: %1") << connection.lastError().message();
45  CLogMessage(this).warning(u"Cannot register DBus service, check server running: dbus-daemon.exe --session --address=tcp:host=192.168.0.133,port=45000");
46  }
47  }
48  break;
49  case SERVERMODE_SYSTEMBUS:
50  {
51  QDBusConnection connection = QDBusConnection::connectToBus(QDBusConnection::SystemBus, coreServiceName());
52  connection.unregisterService(service); // allow reconnecting by removing still registered service
53  if (!connection.isConnected() || !connection.registerService(service))
54  {
55  // registration fails can either mean something wrong with DBus or service already exists
56  CLogMessage(this).warning(u"DBus registration: %1") << connection.lastError().message();
57  CLogMessage(this).warning(u"Cannot register DBus service, check server running: dbus-daemon.exe --system --address=tcp:host=192.168.0.133,port=45000");
58  }
59  }
60  break;
61  case SERVERMODE_P2P:
62  default:
63  {
64  QString dbusAddress = isQtDBusAddress(address) ? address : "tcp:host=127.0.0.1,port=45000";
65  dbusAddress = dbusAddress.toLower().trimmed().replace(' ', "");
66  if (!dbusAddress.contains("bind=")) { dbusAddress = dbusAddress.append(",bind=*"); } // bind to all network interfaces
67 
68  m_busServer.reset(new QDBusServer(dbusAddress, this));
69  m_busServer->setObjectName("QDBusServer: " + this->objectName());
70  m_busServer->setAnonymousAuthenticationAllowed(true);
71 
72  // Note: P2P has no service name
73  if (m_busServer->isConnected())
74  {
75  CLogMessage(this).info(u"DBus P2P Server listening on address: '%1'") << m_busServer->address();
76  }
77  else
78  {
79  CLogMessage(this).warning(u"DBus P2P connection failed: %1") << lastQDBusServerError().message();
80  }
81  connect(m_busServer.data(), &QDBusServer::newConnection, this, &CDBusServer::registerObjectsWithP2PConnection);
82  }
83  break;
84  }
85  }

122  bool CDBusServer::dBusAddressToHostAndPort(const QString &address, QString &host, int &port)
123  {
124  const QString canonicalAddress = address.trimmed().toLower().replace(' ', "");
125  if (canonicalAddress.contains("host=") || canonicalAddress.contains("port="))
126  {
127  // "tcp:host=foo.com,port=123"
128  const QStringList parts(canonicalAddress.split(','));
129  for (const QString &part : parts)
130  {
131  // "host=" or "tcp:host="
132  if (part.contains("host=", Qt::CaseInsensitive))
133  {
134  const QString h = part.mid(part.lastIndexOf("=") + 1).trimmed();
135  host = h;
136  }
137  else if (part.contains("port=", Qt::CaseInsensitive))
138  {
139  const QString p = part.mid(part.lastIndexOf("=") + 1).trimmed();
140  bool ok;
141  port = p.toInt(&ok);
142  if (!ok) { port = -1; }
143  }
144  }
145  if (port < 0) { port = 45000; }
146  if (host.isEmpty()) { host = "127.0.0.1"; }
147  return true;
148  }
149  else
150  {
151  host = "";
152  port = -1;
153  return false;
154  }
155  }

182  {

185  for (int i = 0; i < mo->classInfoCount(); i++)
186  {
187  const QMetaClassInfo ci = mo->classInfo(i);

200  bool CDBusServer::registerObjectsWithP2PConnection(QDBusConnection connection)
201  {
202  Q_ASSERT(!m_objects.isEmpty());
203  m_connections.insert(connection.name(), connection);
204  CLogMessage(this).info(u"New Connection from: '%1'") << connection.name();
205  bool success = true;
206  for (auto i = m_objects.cbegin(); i != m_objects.cend(); ++i)
207  {
208  const QString key(i.key());
209  const bool ok = connection.registerObject(key, i.value(), registerOptions());
210  if (ok)
211  {
212  CLogMessage(this).info(u"Adding '%1' to the new connection '%2'") << key << this->getDBusInterfaceFromClassInfo(i.value());
213  }
214  else
215  {
216  CLogMessage(this).info(u"Adding '%1' failed, connection '%2', error '%3'") << key << this->getDBusInterfaceFromClassInfo(i.value()) << connection.lastError().message();
217  success = false;
218  }
219  }
220  return success;
221  }

223  void CDBusServer::addObject(const QString &path, QObject *object)
224  {
225  if (!object) { return; }
226  m_objects.insert(path, object); // will be registered when P2P connection is established
227 
228  QObject::connect(object, &QObject::destroyed, this, [this, path] { m_objects.remove(path); });
229 
230  switch (m_serverMode)
231  {
232  case SERVERMODE_SESSIONBUS:
233  {
234  QDBusConnection connection = QDBusConnection::connectToBus(QDBusConnection::SessionBus, coreServiceName());
235  if (connection.registerObject(path, object, registerOptions()))
236  {
237  CLogMessage(this).info(u"Adding '%1' '%2' to session DBus") << path << getDBusInterfaceFromClassInfo(object);
238  }
239  else
240  {
241  CLogMessage(this).error(u"Error adding '%1' '%2' to session DBus: '%3'") << path << getDBusInterfaceFromClassInfo(object) << connection.lastError().message();
242  }
243  }
244  break;
245  case SERVERMODE_SYSTEMBUS:
246  {
247  QDBusConnection connection = QDBusConnection::connectToBus(QDBusConnection::SystemBus, coreServiceName());
248  if (connection.registerObject(path, object, registerOptions()))
249  {
250  CLogMessage(this).info(u"Adding '%1' '%2' to system DBus") << path << getDBusInterfaceFromClassInfo(object);
251  }
252  else
253  {
254  CLogMessage(this).error(u"Error adding '%1' '%2' to system DBus: '%3'") << path << getDBusInterfaceFromClassInfo(object) << connection.lastError().message();
255  }
256  }
257  break;
258  case SERVERMODE_P2P:
259  {
260  for (QDBusConnection connection : std::as_const(m_connections))
261  {
262  if (connection.registerObject(path, object, registerOptions()))
263  {
264  CLogMessage(this).info(u"Adding '%1' '%2' to P2P DBus '%3'") << path << getDBusInterfaceFromClassInfo(object) << connection.name();
265  }
266  else
267  {
268  CLogMessage(this).error(u"Error adding '%1' '%2' to P2P DBus '%3': '%4'") << path << getDBusInterfaceFromClassInfo(object) << connection.name() << connection.lastError().message();
269  }
270  }
271  }
272  break;
273  default:
274  Q_ASSERT_X(false, Q_FUNC_INFO, "Wrong server mode");
275  }
276  }

295  {

297  {
298  switch (m_serverMode)

302  break;
303  case SERVERMODE_SYSTEMBUS:
304  QDBusConnection::systemBus().unregisterObject(path);
305  break;

314  m_objects.clear();

335  QDBusConnection CDBusServer::connectToDBus(const QString &dBusAddress, const QString &name)
336  {
337  if (dBusAddress == sessionBusAddress())
338  {
339  if (name.isEmpty()) return QDBusConnection::sessionBus();
340  return QDBusConnection::connectToBus(QDBusConnection::SessionBus, name);
341  }
342  else if (dBusAddress == systemBusAddress())
343  {
344  if (name.isEmpty()) return QDBusConnection::systemBus();
345  return QDBusConnection::connectToBus(QDBusConnection::SystemBus, name);
346  }
347  else if (isP2PAddress(dBusAddress))
348  {
349  return QDBusConnection::connectToPeer(dBusAddress,
350  name.isEmpty() ? CDBusServer::p2pConnectionName() : name);
351  }
352  return QDBusConnection("invalid");
353  }

368  QString CDBusServer::p2pAddress(const QString &host, const QString &port)
369  {
370  QString h = host.trimmed().toLower().remove(' ');
371  if (h.isEmpty()) { h = "127.0.0.1"; }
372 
373  // check port
374  bool ok = false;
375  QString p = port.toLower().trimmed();
376  if (!p.isEmpty())
377  {
378  p.toShort(&ok); // cppcheck-suppress ignoredReturnValue
379  if (!ok)
380  {
381  p = ""; // was not a number
382  }
383  }
384 
385  // can handle host and port separately or combined, e.g. "myhost:1234"
386  if (port.isEmpty())
387  {
388  if (h.startsWith("tcp:") && h.contains("host=") && h.contains("port="))
389  {
390  // looks we already got a full string
391  return h;
392  }
393 
394  // 192.168.5.3:9300 style
395  if (h.contains(":"))
396  {
397  const QStringList parts = h.split(":");
398  h = parts.at(0).trimmed();
399  p = parts.at(1).trimmed();
400  }
401  else
402  {
403  p = "45000";
404  }
405  }
406 
407  // todo: Replace assert with input validation
408  Q_ASSERT_X(CNetworkUtils::isValidIPv4Address(p), "CDBusServer::p2pAddress", "Wrong IP in String");
409  return QStringLiteral("tcp:host=%1,port=%2").arg(h, p);
410  }

418  QString CDBusServer::normalizeAddress(const QString &address)
419  {
420  const QString lc(address.toLower().trimmed());
421 
422  if (lc.isEmpty()) { return sessionBusAddress(); }
423  if (lc == sessionBusAddress() || lc == systemBusAddress()) { return lc; }
424 
425  // some aliases
426  if (lc.startsWith("sys")) { return systemBusAddress(); }
427  if (lc.startsWith("ses")) { return sessionBusAddress(); }
428 
429  // Qt / P2P
430  if (isQtDBusAddress(address)) { return address; }
431  return p2pAddress(address);
432  }

469  bool CDBusServer::isDBusAvailable(const QString &dBusAddress, QString &message, int timeoutMs)
470  {
471  if (dBusAddress.isEmpty())
472  {
473  message = "No address.";
474  return false;
475  }
476  if (isP2PAddress(dBusAddress))
477  {
478  QString host;
479  int port = -1;
480  return CDBusServer::dBusAddressToHostAndPort(dBusAddress, host, port) ?
481  CDBusServer::isDBusAvailable(host, port, message, timeoutMs) :
482  false;
483  }
484  else
485  {
486  QDBusConnection connection = CDBusServer::connectToDBus(dBusAddress);
487  const bool isConnected = connection.isConnected();
488  message = connection.lastError().message();
489  CDBusServer::disconnectFromDBus(connection, dBusAddress);
490  return isConnected;
491  }
492  }
