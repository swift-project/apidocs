44  void CSimulatorXPlaneConfigWindow::onSettingsAccepted()
45  {
46  if (!sGui || sGui->isShuttingDown()) { return; }
47 
48  const CXSwiftBusSettings s = m_xSwiftBusServerSettings.getThreadLocal();
49  const CXSwiftBusSettings changed = this->getSettingsFromUI();
50  if (s != changed)
51  {
52  m_xSwiftBusServerSettings.set(changed);
53 
54  // this writes to a local XPlane directory
55  // if swift runs distributed it does nothing
56  // if XPlane is connected the settings will be written from config.cpp
57  if (!sGui->getIContextSimulator() || !sGui->getIContextSimulator()->isSimulatorSimulating())
58  {
59  CXSwiftBusConfigWriter xswiftbusConfigWriter;
60  xswiftbusConfigWriter.setDBusAddress(changed.getDBusServerAddressQt());
61  xswiftbusConfigWriter.setDebugMode(changed.isLogRenderPhases());
62  xswiftbusConfigWriter.setTcasEnabled(changed.isTcasEnabled());
63  xswiftbusConfigWriter.updateInAllXPlaneVersions();
64  }
65  }
66  this->close();
67  }

100  void CSimulatorXPlaneConfigWindow::setUiValues(const CXSwiftBusSettings &settings)
101  {
102  ui->comp_SettingsXSwiftBus->set(settings.getDBusServerAddressQt());
103  ui->sb_MaxAircraft->setValue(settings.getMaxPlanes());
104  ui->sb_FollowAircraftDistanceM->setValue(settings.getFollowAircraftDistanceM());
105  ui->ds_MaxDrawDistanceNM->setValue(settings.getMaxDrawDistanceNM());
106  ui->cb_DrawLabels->setChecked(settings.isDrawingLabels());
107  ui->cs_LabelColor->setColor(CRgbColor::fromPacked(settings.getLabelColor()));
108  ui->cb_BundleTaxiLandingLights->setChecked(settings.isBundlingTaxiAndLandingLights());
109  ui->cb_TcasEnabled->setChecked(settings.isTcasEnabled());
110  ui->cb_TerrainProbeEnabled->setChecked(settings.isTerrainProbeEnabled());
111  ui->cb_LogRenderPhases->setChecked(settings.isLogRenderPhases());
112 
113  const QString s = settings.getNightTextureModeQt().left(1);
114  if (!s.isEmpty())
115  {
116  for (int i = 0; i < ui->cb_NightTextureMode->count(); i++)
117  {
118  if (ui->cb_NightTextureMode->itemText(i).startsWith(s, Qt::CaseInsensitive))
119  {
120  ui->cb_NightTextureMode->setCurrentIndex(i);
121  break;
122  }
123  }
124  }
125 
126  const std::vector<int> values = settings.getMessageBoxValuesVector();
127  if (values.size() >= 6)
128  {
129  // left, top, right, bottom, height, duration
130  ui->le_MsgBoxMarginsLeft->setText(QString::number(values[0]));
131  ui->le_MsgBoxMarginsTop->setText(QString::number(values[1]));
132  ui->le_MsgBoxMarginsRight->setText(QString::number(values[2]));
133  ui->le_MsgBoxMarginsBottom->setText(QString::number(values[3]));
134  ui->sb_MessageBoxLines->setValue(values[4]);
135  ui->sb_MessageBoxDuration->setValue(values[5]);
136  }
137  if (values.size() >= 11)
138  {
139  // freq, priv, serv, stat, sup
140  ui->cs_ColorFreq->setColor(CRgbColor::fromPacked(values[6]));
141  ui->cs_ColorPriv->setColor(CRgbColor::fromPacked(values[7]));
142  ui->cs_ColorServ->setColor(CRgbColor::fromPacked(values[8]));
143  ui->cs_ColorStat->setColor(CRgbColor::fromPacked(values[9]));
144  ui->cs_ColorSup->setColor(CRgbColor::fromPacked(values[10]));
145  }
146  }
