58  void CVatsimStatusFileReader::read()
59  {
60  this->threadAssertCheck();
61  if (!this->doWorkCheck()) { return; }
62  if (!this->isInternetAccessible("No network/internet access, cannot read VATSIM status file")) { return; }
63 
64  Q_ASSERT_X(sApp, Q_FUNC_INFO, "Missing application");
65  const CUrlList urls(sApp->getGlobalSetup().getVatsimStatusFileUrls());
66  const CUrl url = urls.getRandomUrl();
67  if (url.isEmpty()) { return; }
68  CLogMessage(this).info(u"Trigger read of VATSIM status file from '%1'") << url.toQString(true);
69  this->getFromNetworkAndLog(url, { this, &CVatsimStatusFileReader::parseVatsimFile });
70 
71  if (urls.size() < 2) { return; }
72  const CUrl secondary = urls.getRandomWithout(url);
73  if (secondary.isEmpty()) { return; }
74 
75  constexpr int DelayMs = 5000;
76  const QPointer<CVatsimStatusFileReader> myself(this);
77  QTimer::singleShot(DelayMs, this, [=] {
78  if (!myself) { return; }
79  const CVatsimSetup vs(m_lastGoodSetup.get());
80  if (vs.getTimeDifferenceToNowMs() > 2 * DelayMs)
81  {
82  // not yet read
83  this->getFromNetworkAndLog(url, { this, &CVatsimStatusFileReader::parseVatsimFile });
84  }
85  });
86  }

88  void CVatsimStatusFileReader::parseVatsimFile(QNetworkReply *nwReplyPtr)
89  {
90  // wrap pointer, make sure any exit cleans up reply
91  // required to use delete later as object is created in a different thread
92  QScopedPointer<QNetworkReply, QScopedPointerDeleteLater> nwReply(nwReplyPtr);
93 
94  // Worker thread, make sure to write only synced here!
95  this->threadAssertCheck();
96  if (!this->doWorkCheck())
97  {
98  CLogMessage(this).debug() << Q_FUNC_INFO;
99  CLogMessage(this).info(u"Terminated VATSIM status file parsing process"); // for users
100  return; // stop, terminate straight away, ending thread
101  }
102 
103  this->logNetworkReplyReceived(nwReplyPtr);
104  const QString urlString = nwReply->url().toString();
105 
106  if (nwReply->error() == QNetworkReply::NoError)
107  {
108  const QString dataFileData = nwReply->readAll();
109  nwReply->close(); // close asap
110 
111  if (dataFileData.isEmpty()) return;
112  const QList<QStringRef> lines = splitLinesRefs(dataFileData);
113  if (lines.isEmpty()) { return; }
114 
115  CUrlList dataFileUrls;
116  CUrlList serverFileUrls;
117  CUrlList metarFileUrls;
118 
119  QString currentLine; // declared outside of the for loop, to amortize the cost of allocation
120  for (const QStringRef &clRef : lines)
121  {
122  if (!this->doWorkCheck())
123  {
124  CLogMessage(this).debug() << Q_FUNC_INFO;
125  CLogMessage(this).info(u"Terminated status parsing process"); // for users
126  return; // stop, terminate straight away, ending thread
127  }
128 
129  // parse lines
130  currentLine = clRef.toString().trimmed();
131  if (currentLine.isEmpty()) { continue; }
132  if (currentLine.startsWith(";")) { continue; }
133  if (!currentLine.contains("=")) { continue; }
134 
135  const QStringList parts(currentLine.split('='));
136  if (parts.length() != 2) { continue; }
137  const QString key(parts[0].trimmed().toLower());
138  const QString value(parts[1].trimmed());
139  const CUrl url(value);
140  if (key.startsWith("json3"))
141  {
142  dataFileUrls.push_back(url);
143  }
144  else if (key.startsWith("url1"))
145  {
146  serverFileUrls.push_back(url);
147  }
148  else if (key.startsWith("metar"))
149  {
150  metarFileUrls.push_back(url);
151  }
152  else if (key.startsWith("atis"))
153  {
154  // not yet used
155  }
156  } // for each line
157 
158  // cache itself is thread safe, avoid writing with unchanged data
159  CVatsimSetup vs(m_lastGoodSetup.get());
160  const bool changed = vs.setUrls(dataFileUrls, serverFileUrls, metarFileUrls);
161  vs.setUtcTimestamp(QDateTime::currentDateTime());
162  const CStatusMessage cacheMsg = m_lastGoodSetup.set(vs);
163  if (cacheMsg.isFailure()) { CLogMessage::preformatted(cacheMsg); }
164  else
165  {
166  CLogMessage(this).info(u"Read VATSIM status file from '%1', %2 data file URLs, %3 server file URLs, %4 METAR file URLs")
167  << urlString << dataFileUrls.size() << serverFileUrls.size() << metarFileUrls.size();
168  }
169  Q_UNUSED(changed);
170 
171  // data read finished
172  emit this->dataFileRead(lines.count());
173  emit this->dataRead(CEntityFlags::VatsimStatusFile, CEntityFlags::ReadFinished, lines.count());
174  }
175  else
176  {
177  // network error
178  CLogMessage(this).warning(u"Reading VATSIM status file failed '%1' '%2'") << nwReply->errorString() << urlString;
179  nwReply->abort();
180  emit this->dataRead(CEntityFlags::VatsimStatusFile, CEntityFlags::ReadFailed, 0);
181  }
182  }
183 } // ns
