31  bool CJoystickDevice::init(HWND helperWindow)
32  {
33  HRESULT hr;
34  // Create device
35  {
36  IDirectInputDevice8 *diDevice = nullptr;
37  if (FAILED(hr = m_directInput->CreateDevice(m_guidDevice, &diDevice, nullptr)))
38  {
39  CLogMessage(this).warning(u"IDirectInput8::CreateDevice failed: ") << hr;
40  return false;
41  }
42  m_directInputDevice.reset(diDevice);
43  }
44 
45  // Set cooperative level
46  if (!helperWindow) { return false; }
47  if (FAILED(hr = m_directInputDevice->SetCooperativeLevel(helperWindow, DISCL_NONEXCLUSIVE | DISCL_BACKGROUND)))
48  {
49  CLogMessage(this).warning(u"IDirectInputDevice8::SetCooperativeLevel failed: ") << hr;
50  return false;
51  }
52 
53  // Set data format to c_dfDIJoystick2
54  if (FAILED(hr = m_directInputDevice->SetDataFormat(&c_dfDIJoystick2)))
55  {
56  CLogMessage(this).warning(u"IDirectInputDevice8::SetDataFormat failed: ") << hr;
57  return false;
58  }
59 
60  DIDEVCAPS deviceCaps;
61  deviceCaps.dwSize = sizeof(DIDEVCAPS);
62  // Get device capabilities - we are interested in the number of buttons.
63  if (FAILED(hr = m_directInputDevice->GetCapabilities(&deviceCaps)))
64  {
65  CLogMessage(this).warning(u"IDirectInputDevice8::GetCapabilities failed: ") << hr;
66  return false;
67  }
68 
69  // Filter devices with 0 buttons
70  if (deviceCaps.dwButtons == 0) { return false; }
71 
72  // fix for the toggle button issue T585
73  // if (FAILED(hr = m_directInputDevice->EnumObjects(enumObjectsCallback, this, DIDFT_BUTTON)))
74  if (FAILED(hr = m_directInputDevice->EnumObjects(enumObjectsCallback, this, DIDFT_PSHBUTTON)))
75  {
76  CLogMessage(this).warning(u"IDirectInputDevice8::EnumObjects failed: ") << hr;
77  return false;
78  }
79 
80  CLogMessage(this).info(u"Created joystick device '%1' with %2 buttons") << m_deviceName << deviceCaps.dwButtons;
81  this->startTimer(50);
82  return true;
83  }

101  HRESULT CJoystickDevice::pollDeviceState()
102  {
103  m_directInputDevice->Poll();
104 
105  DIJOYSTATE2 state;
106  HRESULT hr = m_directInputDevice->GetDeviceState(sizeof(DIJOYSTATE2), &state);
107  if (hr == DIERR_INPUTLOST || hr == DIERR_NOTACQUIRED)
108  {
109  m_directInputDevice->Acquire();
110  m_directInputDevice->Poll();
111  hr = m_directInputDevice->GetDeviceState(sizeof(DIJOYSTATE2), &state);
112  }
113 
114  if (FAILED(hr))
115  {
116  CLogMessage(this).warning(u"Cannot acquire and poll joystick device %1. Removing it.") << m_deviceName;
117  emit connectionLost(m_guidDevice);
118  return hr;
119  }
120 
121  for (const CJoystickDeviceInput &input : std::as_const(m_joystickDeviceInputs))
122  {
123  const qint32 buttonIndex = input.m_offset - DIJOFS_BUTTON0;
124  const bool isPressed = state.rgbButtons[buttonIndex] & 0x80;
125  emit this->buttonChanged(input.m_button, isPressed);
126  }
127  return hr;
128  }

153  return DIENUM_CONTINUE;

156  CJoystickWindows::CJoystickWindows(QObject *parent) : IJoystick(parent)
157  {
158  // Initialize COM.
159  // https://docs.microsoft.com/en-us/windows/desktop/api/combaseapi/nf-combaseapi-coinitializeex
160  HRESULT hr = CoInitializeEx(nullptr, COINIT_MULTITHREADED);
161 
162  // RPC_E_CHANGED_MODE: CoInitializeEx was already called by someone else in this thread with a different mode.
163  if (hr == RPC_E_CHANGED_MODE)
164  {
165  CLogMessage(this).debug(u"CoInitializeEx was already called with a different mode. Trying again.");
166  hr = CoInitializeEx(nullptr, COINIT_APARTMENTTHREADED);
167  }
168 
169  // Continue here only if CoInitializeEx was successful
170  // S_OK: The COM library was initialized successfully on this thread.
171  // S_FALSE: The COM library is already initialized on this thread. Reference count was incremented. This is not an error.
172  if (hr == S_OK || hr == S_FALSE)
173  {
174  m_coInitializeSucceeded = true;
175  this->createHelperWindow();
176 
177  if (helperWindow)
178  {
179  this->initDirectInput();
180  this->enumJoystickDevices();
181  this->requestDeviceNotification();
182  }
183  }
184  else
185  {
186  CLogMessage(this).warning(u"CoInitializeEx returned error code %1");
187  }
188  }

254  int CJoystickWindows::createHelperWindow()
255  {
256  // Make sure window isn't created twice
257  if (helperWindow != nullptr)
258  {
259  return 0;
260  }
261 
262  HINSTANCE hInstance = GetModuleHandle(nullptr);
263  WNDCLASSEX wce;
264  ZeroMemory(&wce, sizeof(wce));
265  wce.cbSize = sizeof(wce);
266  wce.lpfnWndProc = windowProc;
267  wce.lpszClassName = static_cast<LPCWSTR>(helperWindowClassName);
268  wce.hInstance = hInstance;
269 
270  /* Register the class. */
271  if (!RegisterClassEx(&wce))
272  {
273  return -1;
274  }
275 
276  /* Create the window. */
277  helperWindow = CreateWindowEx(0, helperWindowClassName,
278  helperWindowName,
279  WS_OVERLAPPED, CW_USEDEFAULT,
280  CW_USEDEFAULT, CW_USEDEFAULT,
281  CW_USEDEFAULT, HWND_MESSAGE, nullptr,
282  hInstance, nullptr);
283  if (helperWindow == nullptr)
284  {
285  UnregisterClass(helperWindowClassName, hInstance);
286  return -1;
287  }
288 
289  SetProp(helperWindow, L"CJoystickWindows", this);
290 
291  return 0;
