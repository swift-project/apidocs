32  void CClientConnection::connectTo(const QString &userName, const QString &password, const QString &callsign, const QString &client, ConnectionCallback callback)
33  {
34  if (m_connection.isConnected())
35  {
36  CLogMessage(this).debug(u"Client already connected to network");
37  return;
38  }
39 
40  m_connection.setUserName(userName);
41  m_connection.setCallsign(callsign);
42 
43  QPointer<CClientConnection> myself(this);
44  m_apiServerConnection->connectTo(userName, password, client, m_networkVersion,
45  { // callback called when connected
46  this, [=](bool authenticated) {
47  // callback when connection has been established
48  if (!myself) { return; } // cppcheck-suppress knownConditionTrueFalse
49 
50  if (authenticated)
51  {
52  const QString cs = m_connection.getCallsign();
53  m_connection.setTokens(m_apiServerConnection->addCallsign(cs));
54  m_connection.setTsAuthenticatedToNow();
55  m_connection.createCryptoChannels();
56  m_connection.setTsHeartbeatToNow();
57  this->connectToVoiceServer();
58  // taskServerConnectionCheck.Start();
59 
60  CLogMessage(this).info(u"Connected: '%1' to voice server, socket open: %2") << cs << boolToYesNo(m_udpSocket->isOpen());
61  }
62  else
63  {
64  m_connection.reset();
65  }
66 
67  // Make sure crypto channels etc. are created
68  m_connection.setConnected(authenticated);
69 
70  // callback of the calling parent
71  if (callback) { callback(authenticated); }
72  } });
73  }

148  void CClientConnection::processMessage(const QByteArray &messageDdata, bool loopback)
149  {
150  if (!m_connection.m_voiceCryptoChannel)
151  {
152  BLACK_VERIFY_X(false, Q_FUNC_INFO, "processMessage used without crypto channel");
153  return;
154  }
155 
156  CryptoDtoSerializer::Deserializer deserializer = CryptoDtoSerializer::deserialize(*m_connection.m_voiceCryptoChannel, messageDdata, loopback);
157 
158  if (deserializer.m_dtoNameBuffer == AudioRxOnTransceiversDto::getShortDtoName())
159  {
160  // qDebug() << "Received audio data";
161  const AudioRxOnTransceiversDto audioOnTransceiverDto = deserializer.getDto<AudioRxOnTransceiversDto>();
162  if (m_connection.isReceivingAudio() && m_connection.isConnected())
163  {
164  emit audioReceived(audioOnTransceiverDto);
165  }
166  }
167  else if (deserializer.m_dtoNameBuffer == HeartbeatAckDto::getShortDtoName())
168  {
169  m_connection.setTsHeartbeatToNow();
170  if (CBuildConfig::isLocalDeveloperDebugBuild()) { CLogMessage(this).debug(u"Received voice server heartbeat"); }
171  }
172  else
173  {
174  CLogMessage(this).warning(u"Received unknown data: %1 %2") << QString(deserializer.m_dtoNameBuffer) << deserializer.m_dataLength;
175  }
176  }

184  void CClientConnection::voiceServerHeartbeat()
185  {
186  if (!m_connection.m_voiceCryptoChannel || !m_udpSocket)
187  {
188  BLACK_VERIFY_X(false, Q_FUNC_INFO, "voiceServerHeartbeat used without crypto channel or socket");
189  return;
190  }
191 
192  const QUrl voiceServerUrl("udp://" + m_connection.getTokens().VoiceServer.addressIpV4);
193  if (CBuildConfig::isLocalDeveloperDebugBuild()) { CLogMessage(this).debug(u"Sending voice server heartbeat to '%1'") << voiceServerUrl.host(); }
194  HeartbeatDto keepAlive;
195  keepAlive.callsign = m_connection.getCallsign().toStdString();
196  const QByteArray dataBytes = CryptoDtoSerializer::serialize(*m_connection.m_voiceCryptoChannel, CryptoDtoMode::AEAD_ChaCha20Poly1305, keepAlive);
197  m_udpSocket->writeDatagram(dataBytes, QHostAddress(voiceServerUrl.host()), static_cast<quint16>(voiceServerUrl.port()));
198  }
