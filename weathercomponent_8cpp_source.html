178  void CWeatherComponent::updateWeatherInformation(bool forceRealWeatherReload)
179  {
180  setWeatherGrid({});
181  ui->lbl_Status->setText({});
182  const bool useOwnAcftPosition = ui->cb_UseOwnAcftPosition->isChecked();
183  CCoordinateGeodetic position;
184  Q_ASSERT(ui->cb_weatherScenario->currentData().canConvert<CWeatherScenario>());
185  const CWeatherScenario scenario = ui->cb_weatherScenario->currentData().value<CWeatherScenario>();
186 
187  if (useOwnAcftPosition)
188  {
189  Q_ASSERT(sGui->getIContextOwnAircraft());
190  position = sGui->getIContextOwnAircraft()->getOwnAircraft().getPosition();
191  m_coordinateDialog->setCoordinate(position);
192  }
193  else
194  {
195  position = m_coordinateDialog->getCoordinate();
196  if (position.isNull())
197  {
198  ui->lbl_Status->setText("No position selected.");
199  return;
200  }
201  }
202 
203  if (CWeatherScenario::isRealWeatherScenario(scenario))
204  {
205  if (m_lastOwnAircraftPosition.isNull() || forceRealWeatherReload ||
206  calculateGreatCircleDistance(position, m_lastOwnAircraftPosition).value(CLengthUnit::km()) > 20)
207  {
208  this->requestWeatherGrid(position);
209  m_lastOwnAircraftPosition = position;
210  showOverlayHTMLMessage("Weather loading,<br>this may take a while", 7500);
211  }
212  }
213  else
214  {
215  setWeatherGrid(CWeatherGrid::getByScenario(scenario));
216  }
217  }

219  void CWeatherComponent::onWeatherGridReceived(const CWeatherGrid &weatherGrid, const CIdentifier &identifier)
220  {
221  if (!isMyIdentifier(identifier))
222  {
223  // not from myself
224  const CWeatherScenario scenario = ui->cb_weatherScenario->currentData().value<CWeatherScenario>();
225  if (!CWeatherScenario::isRealWeatherScenario(scenario)) { return; }
226 
227  // we have received weather grid data and assume those are real weather updates
228  closeOverlay();
229  }
230  ui->lbl_Status->setText({});
231  setWeatherGrid(weatherGrid);
232  }

248  void CWeatherComponent::setWeatherGrid(const CWeatherGrid &weatherGrid)
249  {
250  const CGridPoint gridPoint = weatherGrid.frontOrDefault();
251  ui->tvp_TemperatureLayers->updateContainer(gridPoint.getTemperatureLayers());
252  ui->tvp_CloudLayers->updateContainer(gridPoint.getCloudLayers());
253  ui->tvp_WindLayers->updateContainer(gridPoint.getWindLayers());
254  const CCoordinateGeodetic position = gridPoint.getPosition();
255  const double pressureAtMslHpa = gridPoint.getPressureAtMsl().value(CPressureUnit::hPa());
256  const double pressureAtMslInHg = gridPoint.getPressureAtMsl().value(CPressureUnit::inHg());
257  const QString status = QStringLiteral("Weather Position: %1 %2\nPressure (MSL): %3 hPa / %4 inHg").arg(position.latitude().toWgs84(), position.longitude().toWgs84()).arg(pressureAtMslHpa, 0, 'f', 0).arg(pressureAtMslInHg, 0, 'f', 2);
258  ui->lbl_Status->setText(status);
259  }
