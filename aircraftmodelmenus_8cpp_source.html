75  void CShowSimulatorFileMenu::customMenu(CMenuActions &menuActions)
76  {
77  CAircraftModelView *mv = modelView();
78  Q_ASSERT_X(mv, Q_FUNC_INFO, "no view");
79 
80  if (mv->hasSingleSelectedRow())
81  {
82  const CAircraftModel model(mv->selectedObject());
83  bool added = false;

85  {
86  menuActions.addMenuSimulator();

90  {
91  m_modelConverterX = menuActions.addAction(m_modelConverterX, CIcons::modelConverterX(), "ModelConverterX", CMenuAction::pathSimulator(), { this, &CShowSimulatorFileMenu::startModelConverterX });
92  }
93  }
94 
95  if (m_messageFrame && !model.getIconFile().isEmpty())
96  {
97  added = true;
98  menuActions.addMenuSimulator();
99  m_iconAction = menuActions.addAction(m_iconAction, CIcons::appAircraft16(), "Display icon", CMenuAction::pathSimulator(), { this, &CShowSimulatorFileMenu::displayIcon });
100  }
101 
102  if (added)
103  {
104  menuActions.addSeparator(CMenuAction::pathSimulator());
105  }
106  }
107  this->nestedCustomMenu(menuActions);
108  }

110  void CShowSimulatorFileMenu::showSimulatorFile()
111  {
112  const CAircraftModelView *mv = modelView();
113  if (!mv->hasSingleSelectedRow()) { return; }
114  const CAircraftModel model(mv->selectedObject());
115  if (!model.hasFileName()) { return; }
116  if (QFile::exists(model.getFileName()))
117  {
118  const QUrl url = QUrl::fromLocalFile(model.getFileName());
119  QDesktopServices::openUrl(url);
120  }
121  else
122  {
123  const CStatusMessage m = CStatusMessage(this, CStatusMessage::SeverityError, QStringLiteral("No file for model '&1'").arg(model.getFileName()), true);
124  if (m_messageFrame) { m_messageFrame->showOverlayHTMLMessage(m); }
125  CLogMessage::preformatted(m);
126  }

199  void CConsolidateWithDbDataMenu::consolidateData()
200  {
201  if (!sGui || sGui->isShuttingDown() || !sGui->hasWebDataServices()) { return; }
202  const CAircraftModelList dbModels(sGui->getWebDataServices()->getModels());
203  if (dbModels.isEmpty())
204  {
205  CLogMessage(this).warning(u"No DB models to consolidate with");
206  return;
207  }
208  if (!this->modelsTargetSetable())
209  {
210  CLogMessage(this).warning(u"No setable target");
211  return;
212  }
213 
214  this->modelView()->showLoadIndicator();
215  bool filtered = false;
216  CAircraftModelList models(this->getAllOrAllFilteredAircraftModels(&filtered));
217 
218  const int c = CDatabaseUtils::consolidateModelsWithDbDataAllowsGuiRefresh(models, true, true);
219  if (c > 0 && this->modelsTargetSetable() && this->modelsTargetUpdatable())
220  {
221  if (filtered)
222  {
223  this->modelsTargetUpdatable()->updateModels(models);
224  CLogMessage(this).info(u"Consolidated %1/%2 filtered models with DB") << c << models.size();
225  }
226  else
227  {
228  this->modelsTargetSetable()->setModels(models);
229  CLogMessage(this).info(u"Consolidated %1/%2 models with DB") << c << models.size();
230  }
231  }
232  else
233  {
234  CLogMessage(this).info(u"No data consolidated with DB");
235  this->modelView()->hideLoadIndicator();
236  }
237  }

314  void CConsolidateWithSimulatorModels::consolidateData()
315  {
316  bool filtered = false;
317  const CAircraftModelList models(this->getAllOrAllFilteredAircraftModels(&filtered));
318  if (models.isEmpty()) { return; }
319  QStringList removedModelStrings;
320  const int i = this->modelView()->showLoadIndicator();
321  const CAircraftModelList consolidated = CDatabaseUtils::consolidateModelsWithSimulatorModelsAllowsGuiRefresh(models, this->getSimulatorModels(), removedModelStrings, true);
322  const CSimulatorInfo sim(this->getSimulator());
323 
324  if (!filtered)
325  {
326  this->modelsTargetSetable()->setModelsForSimulator(consolidated, sim);
327  }
328  else
329  {
330  if (!this->modelsTargetUpdatable())
331  {
332  CLogMessage(this).warning(u"No updatable target");
333  }
334  else
335  {
336  this->modelsTargetUpdatable()->updateModelsForSimulator(consolidated, sim);
337  }
338  }
339  this->modelView()->hideLoadIndicator(i);
340  if (!removedModelStrings.isEmpty() && this->getMappingComponent())
341  {
342  const CStatusMessage m = CStatusMessage(this).info(u"Removed %1 model(s)") << removedModelStrings.size();
343  this->getMappingComponent()->showOverlayMessage(m, 5000);
344  }
345  }
