71  template <typename T, typename K, bool UseCompare>
72  void COrderableListModelDbObjects<T, K, UseCompare>::moveItems(const ContainerType &items, int position)
73  {
74  if (items.isEmpty()) { return; }
75  ContainerType container(this->container());
76  int order = 0;
77  if (position >= 0 && position < container.size())
78  {
79  order = container[position].getOrder();
80  }
81  this->setSortColumnToOrder();
82  container.moveTo(items, order);
83 
84  // update container
85  this->updateContainerMaybeAsync(container);
86  }
