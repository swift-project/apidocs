189  void CAircraftModelListModel::replaceOrAddByModelString(const CAircraftModelList &models)
190  {

192  CAircraftModelList currentModels(container());
193  currentModels.removeModelsWithString(models.getModelStringList(true), Qt::CaseInsensitive);

198  QVariant CAircraftModelListModel::data(const QModelIndex &index, int role) const
199  {
200  if (role == Qt::BackgroundRole)
201  {
202  const bool ms = highlightModels() && !m_highlightStrings.isEmpty();
203  if (!ms) { return CListModelDbObjects::data(index, role); }
204 
205  // the underlying model object
206  const CAircraftModel model(this->at(index));
207 
208  // highlight stashed first
209  if (m_highlightStrings.contains(model.getModelString(), Qt::CaseInsensitive))
210  {
211  return m_highlightColor;
212  }
213 
214  return QVariant();
215  }
216  else if (role == Qt::ToolTipRole)
217  {
218  // the underlying model object as summary
219  const CAircraftModel model(this->at(index));
220  return model.asHtmlSummary("<br>");
221  }
222  return CListModelDbObjects::data(index, role);
223  }
