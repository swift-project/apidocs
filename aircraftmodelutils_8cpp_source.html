15  bool CAircraftModelUtilities::mergeWithVPilotData(CAircraftModelList &modelToBeModified, const CAircraftModelList &vPilotModels, bool force)
16  {
17  if (vPilotModels.isEmpty() || modelToBeModified.isEmpty()) { return false; }
18  for (CAircraftModel &simModel : modelToBeModified)
19  {
20  if (!force && simModel.hasValidAircraftAndAirlineDesignator()) { continue; } // already done
21  const CAircraftModel vPilotModel(vPilotModels.findFirstByModelStringOrDefault(simModel.getModelString()));
22  if (!vPilotModel.hasValidDbKey())
23  {
24  continue; // not found
25  }
26  simModel.updateMissingParts(vPilotModel, false);
27  }
28  return true;
29  }

31  QString CAircraftModelUtilities::createIcaoAirlineAircraftHtmlMatrix(const CAircraftModelList &models)
32  {
33  if (models.isEmpty()) { return {}; }
34  static const QString emptyDesignator = "----";
35  static const QString colorLiveryDesignator = "-C-";
36 
37  QMap<QString, QMap<QString, CAircraftModelList>> modelsByDesignator;
38 
39  // create an empty map of all airlines
40  const QMap<QString, CAircraftModelList> emptyAirlineDesignatorMap;
41  CAircraftModelList sortedByAircraft(models);
42  sortedByAircraft.sortBy(&CAircraftModel::getAircraftIcaoCodeDesignator);
43 
44  for (const CAircraftModel &model : std::as_const(sortedByAircraft))
45  {
46  const QString aircraftIcao(model.hasAircraftDesignator() ? model.getAircraftIcaoCodeDesignator() : emptyDesignator);
47  if (!modelsByDesignator.contains(aircraftIcao))
48  {
49  modelsByDesignator.insert(aircraftIcao, emptyAirlineDesignatorMap);
50  }
51  QMap<QString, CAircraftModelList> &airlineModels = modelsByDesignator[aircraftIcao];
52  const QString airlineIcao(
53  model.getLivery().isColorLivery() ? colorLiveryDesignator :
54  model.hasAirlineDesignator() ? model.getAirlineIcaoCodeDesignator() :
55  emptyDesignator);
56  if (airlineModels.contains(airlineIcao))
57  {
58  airlineModels[airlineIcao].push_back(model);
59  }
60  else
61  {
62  airlineModels.insert(airlineIcao, CAircraftModelList({ model }));
63  }
64  }
65 
66  // to HTML
67  QString html("<table>\n");
68  QStringList airlineIcaos = models.getAirlineVDesignators().values();
69  airlineIcaos.sort();
70  airlineIcaos.push_front(colorLiveryDesignator);
71  airlineIcaos.push_back(emptyDesignator);
72  QStringList aircraftIcaos = modelsByDesignator.keys();
73  aircraftIcaos.sort();
74 
75  // header
76  html += "<thead><tr>\n"
77  "<th></th>";
78  for (const QString &airline : std::as_const(airlineIcaos))
79  {
80  html += "<th>";
81  html += airline;
82  html += "</th>";
83  }
84  html += "\n</tr></thead>\n"
85  "<tbody>\n";
86 
87  // fill data
88  for (const QString &aircraftIcao : std::as_const(aircraftIcaos))
89  {
90  html += "<tr>\n"
91  " <th>";
92  html += aircraftIcao;
93  html += "</th>\n";
94 
95  const QMap<QString, CAircraftModelList> &airlineModels = modelsByDesignator[aircraftIcao];
96  for (const QString &airline : std::as_const(airlineIcaos))
97  {
98  if (airlineModels.contains(airline))
99  {
100  html += " <td>";
101  const CAircraftModelList &models(airlineModels[airline]);
102  html += "<a>";
103  html += QString::number(models.size());
104  html += "</a><div class=\"mouseoverdisplay\">";
105  html += models.asHtmlSummary();
106  html += "</div>"
107  "</td>\n";
108  }
109  else
110  {
111  html += " <td></td>\n";
112  }
113  }
114  html += "</tr>\n";
115  }
116  html += "</tbody>\n"
117  "</table>\n";
118  return html;
119  }

121  QString CAircraftModelUtilities::createIcaoAirlineAircraftHtmlMatrixFile(const CAircraftModelList &models, const QString &tempDir)
122  {
123  Q_ASSERT_X(!tempDir.isEmpty(), Q_FUNC_INFO, "Need directory");
124  if (models.isEmpty()) { return {}; }
125  const QString html = createIcaoAirlineAircraftHtmlMatrix(models);
126  if (html.isEmpty()) { return {}; }
127 
128  QDir dir(tempDir);
129  BLACK_VERIFY_X(dir.exists(), Q_FUNC_INFO, "Directory does not exist");
130  if (!dir.exists()) { return {}; }
131 
132  const QString htmlTemplate = CFileUtils::readFileToString(CSwiftDirectories::htmlTemplateFilePath());
133  const QString fn("airlineAircraftMatrix.html");
134  const bool ok = CFileUtils::writeStringToFile(htmlTemplate.arg(html), dir.absoluteFilePath(fn));
135  return ok ? dir.absoluteFilePath(fn) : "";
136  }

138  CStatusMessageList CAircraftModelUtilities::validateModelFiles(const CSimulatorInfo &simulator, const CAircraftModelList &models, CAircraftModelList &validModels, CAircraftModelList &invalidModels, bool ignoreEmpty, int stopAtFailedFiles, std::atomic_bool &wasStopped, const QString &simulatorDir)
139  {
140  // some generic tests
141  CLogCategoryList cats = { CLogCategories::matching() };
142  CStatusMessageList msgs;
143  if (models.isEmpty())
144  {
145  msgs.push_back(CStatusMessage(cats, CStatusMessage::SeverityError, u"No models", true));
146  return msgs;
147  }
148 
149  const int noDb = models.size() - models.countWithValidDbKey();
150  if (noDb > 0)
151  {
152  msgs.push_back(CStatusMessage(cats, CStatusMessage::SeverityWarning, QStringLiteral("%1 models without DB data, is this intended?").arg(noDb), true));
153  const QString ms = models.findWithoutValidDbKey(5).getModelStringList().join(", ");
154  msgs.push_back(CStatusMessage(cats, CStatusMessage::SeverityWarning, QStringLiteral("Some of the non DB models are: '%1'").arg(ms), true));
155  }
156 
157  const int noExcluded = models.countByMode(CAircraftModel::Exclude);
158  if (noExcluded > 0)
159  {
160  msgs.push_back(CStatusMessage(cats, CStatusMessage::SeverityWarning, QStringLiteral("%1 models marked as excluded, is this intended?").arg(noExcluded), true));
161  const QString ms = models.findAllExcludedModels(5).getModelStringList().join(", ");
162  msgs.push_back(CStatusMessage(cats, CStatusMessage::SeverityWarning, QStringLiteral("Some of the excluded models are: '%1'").arg(ms), true));
163  }
164 
165  // specific checks for FSX/XPlane/FG
166  CStatusMessageList specificTests;
167  if (simulator.isMicrosoftOrPrepare3DSimulator() || models.isLikelyFsFamilyModelList())
168  {
169  const CStatusMessageList specificTests1 = FsCommon::CFsCommonUtil::validateAircraftConfigFiles(models, validModels, invalidModels, ignoreEmpty, stopAtFailedFiles, wasStopped);
170  specificTests.push_back(specificTests1);
171 
172  if (simulator.isP3D())
173  {
174  const CStatusMessageList specificTests2 = FsCommon::CFsCommonUtil::validateP3DSimObjectsPath(models, validModels, invalidModels, ignoreEmpty, stopAtFailedFiles, wasStopped, simulatorDir);
175  specificTests.push_back(specificTests2);
176  }
177  else if (simulator.isFSX())
178  {
179  const CStatusMessageList specificTests2 = FsCommon::CFsCommonUtil::validateFSXSimObjectsPath(models, validModels, invalidModels, ignoreEmpty, stopAtFailedFiles, wasStopped, simulatorDir);
180  specificTests.push_back(specificTests2);
181  }
182  }
183  else if (simulator.isXPlane() || models.isLikelyXPlaneModelList())
184  {
185  specificTests = models.validateFiles(validModels, invalidModels, ignoreEmpty, stopAtFailedFiles, wasStopped, simulatorDir);
186  }
187  else
188  {
189  specificTests = models.validateFiles(validModels, invalidModels, ignoreEmpty, stopAtFailedFiles, wasStopped, {});
190  }
191 
192  msgs.push_back(specificTests);
193  return msgs;
194  }
195 } // ns
