68  template <typename T>
69  std::function<void(QDBusPendingCallWatcher *)> setterCallback(T *obj)
70  {
71  return [this, obj](QDBusPendingCallWatcher *watcher) {
72  QDBusPendingReply<T> reply = *watcher;
73  if (reply.isError()) { emit this->asyncMethodError(reply.error()); }
74  else { *obj = reply; }
75  watcher->deleteLater();
76  };
77  }
78 
79  template <typename T>
80  std::function<void(QDBusPendingCallWatcher *)> setterCallbackWithDefault(T *obj, T defaultValue)
81  {
82  return [this, obj, defaultValue](QDBusPendingCallWatcher *watcher) {
83  QDBusPendingReply<T> reply = *watcher;
84  if (reply.isError())
85  {
86  if (reply.error().type() == QDBusError::UnknownMethod)
87  *obj = defaultValue;
88  else
89  emit this->asyncMethodError(reply.error());
90  }
91  else { *obj = reply; }
92  watcher->deleteLater();
93  };
