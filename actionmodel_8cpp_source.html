93  void CActionModel::setupModelData()
94  {
95  m_rootItem.reset(new CActionItem(QString(), QString()));
96 
97  Q_ASSERT_X(sApp && sApp->getInputManager(), Q_FUNC_INFO, "Missing input manager");
98  const QMap<QString, BlackMisc::CIcons::IconIndex> availableActionsAndIcons = sApp->getInputManager()->allAvailableActionsAndIcons();
99  QStringList keys = availableActionsAndIcons.keys();
100  keys.sort();
101  for (const QString &actionPath : std::as_const(keys))
102  {
103  QString currentPath;
104  const QStringList tokens = actionPath.split("/", Qt::SkipEmptyParts);
105  CActionItem *parentItem = m_rootItem.data();
106  for (const auto &token : tokens)
107  {
108  currentPath += u'/' % token;
109  CActionItem *child = parentItem->findChildByName(token);
110  if (child == nullptr)
111  {
112  const bool isAction = currentPath == actionPath; // action istelf, or just a node?
113  const BlackMisc::CIcons::IconIndex icon = isAction ? availableActionsAndIcons[actionPath] : CIcons::StandardIconEmpty16;
114  child = new CActionItem(isAction ? actionPath : "", token, icon, parentItem);
115  parentItem->appendChild(child);
116  }
117  Q_ASSERT(child);
118  parentItem = child;
119  }
120  }
121  }
