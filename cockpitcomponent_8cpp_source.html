63  void CCockpitComponent::toggleShowHideDetails(bool show, bool considerCurrentSize)
64  {
65  Q_ASSERT(this->isParentDockWidgetFloating()); // show hide should not be visible if docked
66  Q_ASSERT(this->window());
67  if (!this->isParentDockWidgetFloating()) { return; }
68 
69  // manually setting size, all other approaches failed
70  static const QSize defaultSizeShown(500, 600);
71  static const QSize defaultSizeHidden(300, 150);
72 
73  // keep old size
74  const QSize manuallySetSize = this->window()->size();
75 
76  // hide area
77  ui->comp_CockpitInfoArea->setVisible(show);
78 
79  // adjust size
80  if (show)
81  {
82  ui->comp_CockpitInfoArea->setMinimumHeight(m_minHeightInfoArea);
83  if (m_sizeFloatingShown.isValid())
84  {
85  this->window()->resize(m_sizeFloatingShown);
86  if (considerCurrentSize) { m_sizeFloatingHidden = manuallySetSize; } // for next time
87  }
88  else
89  {
90  // manually setting size, all other approaches failed
91  this->window()->resize(defaultSizeShown);
92  m_sizeFloatingShown = this->window()->size();
93  }
94  }
95  else
96  {
97  ui->comp_CockpitInfoArea->setMinimumHeight(0);
98  this->window()->setMinimumSize(defaultSizeHidden);
99  if (m_sizeFloatingHidden.isValid())
100  {
101  this->window()->resize(m_sizeFloatingHidden);
102  if (considerCurrentSize) { m_sizeFloatingShown = manuallySetSize; }
103  }
104  else
105  {
106  // manually setting size, all other approaches failed
107  this->window()->resize(defaultSizeHidden);
108  m_sizeFloatingHidden = this->window()->size();
109  }
110  }
111  }
