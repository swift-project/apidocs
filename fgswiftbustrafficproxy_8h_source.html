61  this->headingsDeg.push_back(situation.getHeading().value(BlackMisc::PhysicalQuantities::CAngleUnit::deg()));
62  this->onGrounds.push_back(situation.getOnGround() == BlackMisc::Aviation::CAircraftSituation::OnGround);
63  this->groundSpeedKts.push_back(situation.getGroundSpeed().value(BlackMisc::PhysicalQuantities::CSpeedUnit::kts()));

87  this->gears.push_back(parts.isFixedGearDown() ? 1 : 0);
88  this->flaps.push_back(parts.getFlapsPercent() / 100.0);
89  this->spoilers.push_back(parts.isSpoilersOut() ? 1 : 0);
90  this->speedBrakes.push_back(parts.isSpoilersOut() ? 1 : 0);
91  this->slats.push_back(parts.getFlapsPercent() / 100.0);
92  this->wingSweeps.push_back(0.0);
93  this->thrusts.push_back(parts.isAnyEngineOn() ? 0 : 0.75);
94  this->elevators.push_back(0.0);
95  this->rudders.push_back(0.0);
96  this->ailerons.push_back(0.0);
97  this->landLights.push_back(parts.getLights().isLandingOn());
98  this->beaconLights.push_back(parts.getLights().isBeaconOn());
99  this->strobeLights.push_back(parts.getLights().isStrobeOn());
100  this->navLights.push_back(parts.getLights().isNavOn());
101  this->lightPatterns.push_back(0);
102  this->taxiLights.push_back(parts.getLights().isTaxiOn());
103  }
