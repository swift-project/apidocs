145  bool CDbOwnModelsComponent::setSimulator(const CSimulatorInfo &simulator, bool forced)
146  {
147  Q_ASSERT_X(simulator.isSingleSimulator(), Q_FUNC_INFO, "Need single simulator");
148 
149  const bool simUnchanged = (m_simulator == simulator);
150 
151  if (!simUnchanged) { emit this->ownModelsSimulatorChanged(simulator); }
152  if (!forced && simUnchanged) { return false; }
153 
154  // changed simulator
155  m_simulator = simulator;
156  this->requestSimulatorModelsWithCacheInBackground(simulator);
157  ui->comp_SimulatorSelector->setValue(simulator);
158  this->setUiSimulatorString(simulator);
159  ui->tvp_OwnAircraftModels->setCorrespondingSimulator(simulator, m_simulatorSettings.getSimulatorDirectoryOrDefault(simulator));
160  return true;
161  }

228  bool CDbOwnModelsComponent::initModelLoader(const CSimulatorInfo &simulator, IAircraftModelLoader::LoadMode load)
229  {
230  // called when simulator is changed / init
231  Q_ASSERT_X(simulator.isSingleSimulator(), Q_FUNC_INFO, "Need single simulator");
232 
233  // already loaded
234  if (!m_modelLoader || !m_modelLoader->supportsSimulator(simulator))
235  {
236  m_modelLoader = CMultiAircraftModelLoaderProvider::multiModelLoaderInstance().loaderInstance(simulator);
237  if (m_modelLoader) { m_modelLoader->startLoading(load); }
238  }
239  this->setSaveFileName(simulator);
240  return m_modelLoader && m_modelLoader->supportsSimulator(simulator);
241  }

663  void CDbOwnModelsComponent::onModelLoaderLoadingFinished(const CStatusMessageList &statusMessages, const CSimulatorInfo &simulator, IAircraftModelLoader::LoadFinishedInfo info)
664  {
665  Q_ASSERT_X(simulator.isSingleSimulator(), Q_FUNC_INFO, "Expect single simulator");
666 
667  bool hideIndicator = false; // hide in case loading failed
668  CStatusMessage summaryMsg;
669 
670  if (IAircraftModelLoader::isLoadedInfo(info) && m_modelLoader)
671  {
672  const CAircraftModelList models(m_modelLoader->getCachedModels(simulator));
673  const int modelsLoaded = models.size();
674  ui->tvp_OwnAircraftModels->updateContainerMaybeAsync(models);
675 
676  if (modelsLoaded < 1)
677  {
678  // loading ok, but no data
679  summaryMsg = CLogMessage(this).warning(u"Loading completed for simulator '%1', but no models") << simulator;
680  hideIndicator = true;
681  }
682  else
683  {
684  summaryMsg = CLogMessage(this).info(u"Loading completed for simulator '%1' with %2 models") << simulator << modelsLoaded;
685  }
686 
687  // overlay
688  if (!summaryMsg.isEmpty() && info == IAircraftModelLoader::ParsedData)
689  {
690  this->showOverlayHTMLMessage(summaryMsg, 5000);
691  }
692 
693  // signal
694  emit this->successfullyLoadedModels(simulator, modelsLoaded);
695  }
696  else if (info == IAircraftModelLoader::LoadingSkipped)
697  {
698  CLogMessage(this).error(u"Loading of models skipped, simulator '%1'") << simulator.toQString();
699  hideIndicator = true;
700  }
701  else
702  {
703  ui->tvp_OwnAircraftModels->clear();
704  hideIndicator = true;
705  CLogMessage(this).error(u"Loading of models failed, simulator '%1'") << simulator.toQString();
706  }
707 
708  // with errors we make sure errors are on top
709  if (statusMessages.hasErrorMessages())
710  {
711  this->setOverlayMessagesSorting(CStatusMessage::IndexSeverityAsIcon, Qt::DescendingOrder);
712  this->showOverlayMessagesOrHTMLMessage(statusMessages, false, -1);
713  }
714  else
715  {
716  // no issues, directly hide
717  const int timeoutMs = statusMessages.hasWarningOrErrorMessages() ? -1 : 7500;
718  if (statusMessages.size() < 50)
719  {
720  // small number of messages
721  this->showOverlayMessagesOrHTMLMessage(statusMessages, false, timeoutMs);
722  }
723  else
724  {
725  QPointer<CDbOwnModelsComponent> myself(this);
726  const QString confirmMessage = QStringLiteral("Do you want to see the %1 detailled messages?").arg(statusMessages.size());
727  this->showOverlayMessagesWithConfirmation(summaryMsg, false, confirmMessage, [=] {
728  if (!myself) { return; }
729  myself->showOverlayMessagesOrHTMLMessage(statusMessages);
730  });
731  }
732  }
733 
734  if (hideIndicator) { ui->tvp_OwnAircraftModels->hideLoadIndicatorForced(); }
735 
736  // cache loads may occur in background, do not adjust UI settings
737  if (info == IAircraftModelLoader::CacheLoaded) { return; }
738 
739  // parsed loads normally explicit displaying this simulator
740  this->setSimulator(simulator);
741  }
