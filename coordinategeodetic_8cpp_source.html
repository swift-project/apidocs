41  CLength calculateGreatCircleDistance(const ICoordinateGeodetic &coordinate1, const ICoordinateGeodetic &coordinate2)
42  {
43  if (coordinate1.isNull() || coordinate2.isNull()) { return CLength::null(); }
44  // if (coordinate1.equalNormalVectorDouble(coordinate2)) { return CLength(0, CLengthUnit::defaultUnit()); }
45  constexpr float earthRadiusMeters = 6371000.8f;
46 
47  const QVector3D v1 = coordinate1.normalVector();
48  const QVector3D v2 = coordinate2.normalVector();
49  Q_ASSERT_X(std::isfinite(v1.x()) && std::isfinite(v1.y()) && std::isfinite(v1.z()), Q_FUNC_INFO, "Distance calculation: v1 non-finite argument");
50  Q_ASSERT_X(std::isfinite(v2.x()) && std::isfinite(v2.y()) && std::isfinite(v2.z()), Q_FUNC_INFO, "Distance calculation: v2 non-finite argument");
51 
52  const float d = earthRadiusMeters * std::atan2(QVector3D::crossProduct(v1, v2).length(), QVector3D::dotProduct(v1, v2));
53 
54  BLACK_VERIFY_X(!std::isnan(d), Q_FUNC_INFO, "Distance calculation: NaN in result");
55  if (std::isnan(d))
56  {
57  CLogMessage().debug(u"Distance calculation: NaN in result (given arguments %1 %2 %3; %4 %5 %6)") << static_cast<double>(v1.x()) << static_cast<double>(v1.y()) << static_cast<double>(v1.z()) << static_cast<double>(v2.x()) << static_cast<double>(v2.y()) << static_cast<double>(v2.z());
58  return CLength::null();
59  }
60  return { static_cast<double>(d), CLengthUnit::m() };
61  }

63  CAngle calculateBearing(const ICoordinateGeodetic &coordinate1, const ICoordinateGeodetic &coordinate2)
64  {
65  if (coordinate1.isNull() || coordinate2.isNull()) { return CAngle::null(); }
66  // if (coordinate1.equalNormalVectorDouble(coordinate2)) { return CAngle(0, CAngleUnit::defaultUnit()); } // null or 0?
67  static const QVector3D northPole { 0, 0, 1 };
68  const QVector3D c1 = QVector3D::crossProduct(coordinate1.normalVector(), coordinate2.normalVector());
69  const QVector3D c2 = QVector3D::crossProduct(coordinate1.normalVector(), northPole);
70  const QVector3D cross = QVector3D::crossProduct(c1, c2);
71  const float sinTheta = std::copysign(cross.length(), QVector3D::dotProduct(cross, coordinate1.normalVector()));
72  const float cosTheta = QVector3D::dotProduct(c1, c2);
73  const float theta = std::atan2(sinTheta, cosTheta);
74  return { static_cast<double>(theta), CAngleUnit::rad() };
75  }

317  CCoordinateGeodetic CCoordinateGeodetic::calculatePosition(const CLength &distance, const CAngle &relBearing) const
318  {
319  if (this->isNull()) { return CCoordinateGeodetic::null(); }
320  if (distance.isNull() || distance.isNegativeWithEpsilonConsidered() || relBearing.isNull()) { return CCoordinateGeodetic::null(); }
321  if (distance.isZeroEpsilonConsidered()) { return *this; }
322 
323  // http://www.movable-type.co.uk/scripts/latlong.html#destPoint
324  // https://stackoverflow.com/a/879531/356726
325  // https://www.cosmocode.de/en/blog/gohr/2010-06/29-calculate-a-destination-coordinate-based-on-distance-and-bearing-in-php
326  constexpr double earthRadiusMeters = 6371000.8;
327  const double startLatRad = this->latitude().value(CAngleUnit::rad());
328  const double startLngRad = this->longitude().value(CAngleUnit::rad());
329  const double bearingRad = relBearing.value(CAngleUnit::rad());
330  const double distRatio = distance.value(CLengthUnit::m()) / earthRadiusMeters;
331 
332  const double newLatRad = asin(sin(startLatRad) * cos(distRatio) + cos(startLatRad) * sin(distRatio) * cos(bearingRad));
333  double newLngRad = 0;
334 
335  constexpr double epsilon = 1E-06;
336  if (cos(newLatRad) == 0 || qAbs(cos(newLatRad)) < epsilon)
337  newLngRad = startLngRad;
338  else
339  {
340  // λ1 + Math.atan2(Math.sin(brng)*Math.sin(d/R)*Math.cos(φ1), Math.cos(d/R)-Math.sin(φ1)*Math.sin(φ2));
341  newLngRad = startLngRad + atan2(sin(bearingRad) * sin(distRatio) * cos(startLatRad), cos(distRatio) - sin(startLatRad) * sin(newLatRad));
342  newLngRad = fmod(newLngRad + 3 * M_PI, 2 * M_PI) - M_PI; // normalize +-180deg
343  }
344 
345  CCoordinateGeodetic copy = *this;
346  const CLatitude lat(newLatRad, CAngleUnit::rad());
347  const CLongitude lng(newLngRad, CAngleUnit::rad());
348  copy.setLatLong(lat, lng);
349  return copy;
350  }
