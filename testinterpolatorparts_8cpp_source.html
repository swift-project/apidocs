42  void CTestInterpolatorParts::groundFlagInterpolation()
43  {
44  const CCallsign cs("SWIFT");
45  const CInterpolationAndRenderingSetupGlobal gSetup;
46  const CInterpolationAndRenderingSetupPerCallsign setup(cs, gSetup);
47 
48  CRemoteAircraftProviderDummy provider;
49  CInterpolatorSpline interpolator(cs, nullptr, nullptr, &provider);
50  interpolator.markAsUnitTest();
51 
52  // fixed time so everything can be debugged
53  const qint64 ts = 1425000000000; // QDateTime::currentMSecsSinceEpoch()
54  const qint64 deltaT = 5000; // ms
55  const int number = CRemoteAircraftProviderDummy::MaxPartsPerCallsign;
56  // const int numberHalf = number / 2;
57  const qint64 farFuture = ts + 3 * number * deltaT;
58  const qint64 farPast = ts - 4 * number * deltaT;
59 
60  CAircraftPartsList parts;
61  for (int i = 0; i < number; i++)
62  {
63  const CAircraftParts p = createTestParts(i, ts, deltaT, true);
64  parts.push_back(p);
65  }
66  QVERIFY2(parts.size() == number, "Wrong parts size of list");
67 
68  // interpolation functional check
69  const qint64 oldestTs = parts.oldestTimestampMsecsSinceEpoch();
70 
71  // Testing for a time >> last time
72  // all on ground flags true
73  provider.insertNewAircraftParts(cs, parts, false); // we work with 0 offsets here
74  QVERIFY2(provider.remoteAircraftPartsCount(cs) == parts.size(), "Wrong parts size");
75 
76  CInterpolationResult result = interpolator.getInterpolation(farFuture, setup);
77  CAircraftParts p = result;
78  qint64 pTs = p.getAdjustedMSecsSinceEpoch();
79  QVERIFY2(result.getPartsStatus().isSupportingParts(), "Should support parts");
80  QVERIFY2(pTs == ts, "Expect latest ts");
81  result = interpolator.getInterpolation(farPast, setup);
82  p = result;
83  pTs = p.getAdjustedMSecsSinceEpoch();
84  QVERIFY2(result.getPartsStatus().isSupportingParts(), "Should support parts");
85  QVERIFY2(pTs == oldestTs, "Expect oldest ts");
86 
87  // Testing for a time >> last time
88  // all on ground flags true
89  interpolator.clear();
90  provider.clear();
91 
92  parts.setOnGround(false);
93  provider.insertNewAircraftParts(cs, parts, false); // we work with 0 offsets here
94  result = interpolator.getInterpolation(farFuture, setup);
95  p = result;
96  pTs = p.getAdjustedMSecsSinceEpoch();
97  QVERIFY2(result.getPartsStatus().isSupportingParts(), "Should support parts");
98  QVERIFY2(p.getAdjustedMSecsSinceEpoch() == pTs, "Expect latest ts");
99  }

101  void CTestInterpolatorParts::partsToSituationGndFlag()
102  {
103  CCallsign cs("SWIFT");
104 
105  // fixed time so everything can be debugged
106  const qint64 ts = 1425000000000; // QDateTime::currentMSecsSinceEpoch()
107  const qint64 deltaT = 5000; // ms
108  const int number = 10;
109 
110  CAircraftPartsList partsOnGround;
111  for (int i = 0; i < number; i++)
112  {
113  const CAircraftParts p = createTestParts(i, ts, deltaT, true);
114  partsOnGround.push_back(p);
115  }
116 
117  CAircraftPartsList partsNotOnGround;
118  for (int i = 0; i < number; i++)
119  {
120  const CAircraftParts p = createTestParts(i, ts, deltaT, false);
121  partsNotOnGround.push_back(p);
122  }
123 
124  CAircraftSituationList situations;
125  for (int i = 0; i < number; i++)
126  {
127  CAircraftSituation s = createTestSituation(cs, i, ts, deltaT, 0);
128  s.setOnGround(CAircraftSituation::OnGroundSituationUnknown, CAircraftSituation::NotSetGroundDetails);
129  situations.push_back(s);
130  }
131 
132  CAircraftSituation s0 = situations[0];
133  s0.adjustGroundFlag(partsOnGround, true);
134  QVERIFY2(s0.getOnGround(), "Supposed to be on ground");
135 
136  s0 = situations[0];
137  s0.adjustGroundFlag(partsNotOnGround, true);
138  QVERIFY2(!s0.getOnGround(), "Supposed to be not on ground");
139 
140  qint64 distanceMs = -1;
141  const qint64 Offset = 33;
142  partsOnGround.addMsecsToOffset(Offset);
143  CAircraftSituation s1 = situations[1];
144  s1.setOnGroundDetails(CAircraftSituation::NotSetGroundDetails);
145  s1.adjustGroundFlag(partsOnGround, true, 0.1, &distanceMs);
146  QVERIFY2(s1.getOnGround(), "Supposed to be on ground");
147  QVERIFY2(distanceMs == deltaT - Offset, "Offset time wrong");
148  QVERIFY2(s1.getOnGroundDetails() == CAircraftSituation::InFromParts, "Wrong details");
149  }

162  CAircraftSituation CTestInterpolatorParts::createTestSituation(const CCallsign &callsign, int number, qint64 ts, qint64 deltaT, qint64 offset)
163  {
164  const CAltitude alt(number, CAltitude::MeanSeaLevel, CLengthUnit::m());

166  const CLongitude lng(180.0 + number, CAngleUnit::deg());

172  const CCoordinateGeodetic c(lat, lng, alt);
173  CAircraftSituation s(callsign, c, heading, pitch, bank, gs);
174  s.setGroundElevation(gndElev, CAircraftSituation::Test);
175  s.setMSecsSinceEpoch(ts - deltaT * number); // values in past
176  s.setTimeOffsetMs(offset);
