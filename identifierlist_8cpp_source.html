44  int CIdentifierList::removeDuplicates()
45  {

47  CIdentifierList il;
48  for (const CIdentifier &identifier : *this)
49  {
50  if (il.contains(identifier)) continue;
51  il.push_back(identifier);
52  }
53  const int delta = this->size() - il.size();
54  if (delta == 0) { return 0; }
