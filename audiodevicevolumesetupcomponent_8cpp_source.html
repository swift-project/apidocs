100  void CAudioDeviceVolumeSetupComponent::init()
101  {
102  if (!sGui || sGui->isShuttingDown() || !sGui->getCContextAudioBase()) { return; }
103 
104  // audio is optional
105  const bool audio = this->hasAudio();
106  this->setEnabled(audio);
107  this->reloadSettings();
108 
109  bool c = connect(ui->cb_SetupAudioLoopback, &QCheckBox::toggled, this, &CAudioDeviceVolumeSetupComponent::onLoopbackToggled);
110  Q_ASSERT(c);
111  c = connect(ui->cb_DisableAudioEffects, &QCheckBox::toggled, this, &CAudioDeviceVolumeSetupComponent::onDisableAudioEffectsToggled);
112  Q_ASSERT(c);
113 
114  if (audio)
115  {
116  this->setAudioRunsWhere();
117  this->initAudioDeviceLists();
118 
119  // default
120  ui->cb_SetupAudioLoopback->setChecked(sGui->getCContextAudioBase()->isAudioLoopbackEnabled());
121 
122  // the connects depend on initAudioDeviceLists
123  c = connect(ui->cb_SetupAudioInputDevice, qOverload<int>(&QComboBox::currentIndexChanged), this, &CAudioDeviceVolumeSetupComponent::onAudioDeviceSelected, Qt::QueuedConnection);
124  Q_ASSERT(c);
125  c = connect(ui->cb_SetupAudioOutputDevice, qOverload<int>(&QComboBox::currentIndexChanged), this, &CAudioDeviceVolumeSetupComponent::onAudioDeviceSelected, Qt::QueuedConnection);
126  Q_ASSERT(c);
127 
128  // context
129  c = connect(sGui->getCContextAudioBase(), &CContextAudioBase::changedLocalAudioDevices, this, &CAudioDeviceVolumeSetupComponent::onAudioDevicesChanged, Qt::QueuedConnection);
130  Q_ASSERT(c);
131  c = connect(sGui->getCContextAudioBase(), &CContextAudioBase::startedAudio, this, &CAudioDeviceVolumeSetupComponent::onAudioStarted, Qt::QueuedConnection);
132  Q_ASSERT(c);
133  c = connect(sGui->getCContextAudioBase(), &CContextAudioBase::stoppedAudio, this, &CAudioDeviceVolumeSetupComponent::onAudioStopped, Qt::QueuedConnection);
134  Q_ASSERT(c);
135 
136  this->initWithAfvClient();
137  m_init = true;
138  }
139  Q_UNUSED(c)
140  }

142  void CAudioDeviceVolumeSetupComponent::initWithAfvClient()
143  {
144  if (!afvClient()) { return; }
145  m_afvConnections.disconnectAll();
146 
148  CAfvClient *afv = afvClient();
149  const Qt::ConnectionType ct = Qt::QueuedConnection;
150  QMetaObject::Connection c;
151  c = connect(afv, &CAfvClient::outputVolumePeakVU, this, &CAudioDeviceVolumeSetupComponent::onOutputVU, ct);
152  Q_ASSERT(c);
153  m_afvConnections.append(c);
154  c = connect(afv, &CAfvClient::inputVolumePeakVU, this, &CAudioDeviceVolumeSetupComponent::onInputVU, ct);
155  Q_ASSERT(c);
156  m_afvConnections.append(c);
157  c = connect(afv, &CAfvClient::receivedCallsignsChanged, this, &CAudioDeviceVolumeSetupComponent::onReceivingCallsignsChanged, ct);
158  Q_ASSERT(c);
159  m_afvConnections.append(c);
160  c = connect(afv, &CAfvClient::updatedFromOwnAircraftCockpit, this, &CAudioDeviceVolumeSetupComponent::onUpdatedClientWithCockpitData, ct);
161  Q_ASSERT(c);
162  m_afvConnections.append(c);
163 
164  // default values for RX/TX
165  afv->setRxTx(true, true, true, false);
166 
167  QPointer<CAudioDeviceVolumeSetupComponent> myself(this);
168  c = connect(
169  afv, &CAfvClient::connectionStatusChanged, this, [=](CAfvClient::ConnectionStatus status) {
170  if (!myself || !sGui || sGui->isShuttingDown()) { return; }
171  myself->setTransmitReceiveInUiFromVoiceClient();
172  Q_UNUSED(status)
173  },
174  ct);
175  Q_ASSERT(c);
176  m_afvConnections.append(c);
177 
178  this->setTransmitReceiveInUiFromVoiceClient();
179  }

377  void CAudioDeviceVolumeSetupComponent::saveVolumes()
378  {
379  CSettings as(m_audioSettings.getThreadLocal());
380  const int i = this->getInValue();
381  const int o = this->getOutValue();
382  const int o1 = this->getOutValueCom1();
383  const int o2 = this->getOutValueCom2();
384  if (as.getInVolume() == i && o == as.getOutVolume() &&
385  as.getOutVolumeCom1() == o1 && as.getOutVolumeCom2() == o2) { return; }
386  as.setInVolume(i);
387  as.setOutVolume(o);
388  as.setOutVolumeCom1(o1);
389  as.setOutVolumeCom2(o2);
390  m_audioSettings.setAndSave(as);
