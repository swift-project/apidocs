66  QString convertToUnicodeEscaped(const QString &str)
67  {
68  QString escaped;
69  for (const auto &ch : str)
70  {
71  const ushort code = ch.unicode();
72  if (code < 0x80)
73  {
74  escaped += ch;
75  }
76  else
77  {
78  escaped += "\\u" % QString::number(code, 16).rightJustified(4, '0');
79  }
80  }
81  return escaped;
82  }

232  void CFSDClient::connectToServer()
233  {
234  if (!CThreadUtils::isInThisThread(this))
235  {
236  QMetaObject::invokeMethod(this, [=] {
237  if (sApp && !sApp->isShuttingDown()) { connectToServer(); }

240  }
241 
242  if (m_socket->isOpen()) { return; }
243  Q_ASSERT(!m_clientName.isEmpty());
244  Q_ASSERT((m_versionMajor + m_versionMinor) > 0);
245  Q_ASSERT(m_capabilities != Capabilities::None);
246 
247  if (m_hostApplication.isEmpty()) { m_hostApplication = this->getSimulatorNameAndVersion().replace(':', ' '); }
248 
249  this->clearState();

251 
252  m_loginSince = QDateTime::currentMSecsSinceEpoch();
253  const qint64 timerMs = qRound(PendingConnectionTimeoutMs * 1.25);
254 
255  const QPointer<CFSDClient> myself(this);
256  QTimer::singleShot(timerMs, this, [=] {
257  if (!myself || !sApp || sApp->isShuttingDown()) { return; }
258  this->pendingTimeoutCheck();
259  });
260 
261  this->updateConnectionStatus(CConnectionStatus::Connecting);
262 
263  initiateConnection();

393  void CFSDClient::sendVisualPilotDataUpdate(bool slowUpdate)
394  {
395  if (this->getConnectionStatus().isDisconnected() && !m_unitTestMode) { return; }
396  if (m_loginMode == CLoginMode::Observer || !isVisualPositionSendingEnabledForServer()) { return; }
397  const CSimulatedAircraft myAircraft(getOwnAircraft());
398 
399  if (!slowUpdate)
400  {
401  static constexpr double minVelocity = 0.00005;
402  if (std::abs(myAircraft.getVelocity().getVelocityX(CSpeedUnit::m_s())) < minVelocity &&
403  std::abs(myAircraft.getVelocity().getVelocityY(CSpeedUnit::m_s())) < minVelocity &&
404  std::abs(myAircraft.getVelocity().getVelocityZ(CSpeedUnit::m_s())) < minVelocity &&
405  std::abs(myAircraft.getVelocity().getPitchVelocity(CAngleUnit::rad(), CTimeUnit::s())) < minVelocity &&
406  std::abs(myAircraft.getVelocity().getRollVelocity(CAngleUnit::rad(), CTimeUnit::s())) < minVelocity &&
407  std::abs(myAircraft.getVelocity().getHeadingVelocity(CAngleUnit::rad(), CTimeUnit::s())) < minVelocity)
408  {
409  if (m_stoppedSendingVisualPositions) { return; }
410  m_stoppedSendingVisualPositions = true;
411  m_visualPositionUpdateSentCount = 0;
412  }
413  else
414  {
415  m_stoppedSendingVisualPositions = false;
416  }
417 
418  if (!m_serverWantsVisualPositions)
419  {
420  return;
421  }
422  }
423  VisualPilotDataUpdate visualPilotDataUpdate(getOwnCallsignAsString(),
424  myAircraft.latitude().value(CAngleUnit::deg()),
425  myAircraft.longitude().value(CAngleUnit::deg()),
426  myAircraft.getAltitude().value(CLengthUnit::ft()),
427  myAircraft.getAltitude().value(CLengthUnit::ft()) - myAircraft.getGroundElevation().value(CLengthUnit::ft()),
428  myAircraft.getPitch().value(CAngleUnit::deg()),
429  myAircraft.getBank().value(CAngleUnit::deg()),
430  myAircraft.getHeading().normalizedTo360Degrees().value(CAngleUnit::deg()),
431  myAircraft.getVelocity().getVelocityX(CSpeedUnit::m_s()),
432  myAircraft.getVelocity().getVelocityY(CSpeedUnit::m_s()),
433  myAircraft.getVelocity().getVelocityZ(CSpeedUnit::m_s()),
434  myAircraft.getVelocity().getPitchVelocity(CAngleUnit::rad(), CTimeUnit::s()),
435  myAircraft.getVelocity().getRollVelocity(CAngleUnit::rad(), CTimeUnit::s()),
436  myAircraft.getVelocity().getHeadingVelocity(CAngleUnit::rad(), CTimeUnit::s()));
437 
438  if (m_stoppedSendingVisualPositions)
439  {
440  sendQueudedMessage(visualPilotDataUpdate.toStopped());
441  }
442  else if (m_visualPositionUpdateSentCount++ % 25 == 0)
443  {
444  sendQueudedMessage(visualPilotDataUpdate.toPeriodic());

448  sendQueudedMessage(visualPilotDataUpdate);
449  }

512  void CFSDClient::sendClientQuery(ClientQueryType queryType, const CCallsign &receiver, const QStringList &queryData)
513  {
514  if (queryType == ClientQueryType::Unknown) { return; }
515  if (!CThreadUtils::isInThisThread(this))
516  {
517  QMetaObject::invokeMethod(this, [=] {
518  if (sApp && !sApp->isShuttingDown()) { sendClientQuery(queryType, receiver, queryData); }
519  });
520  return;
521  }
522 
523  const QString reveiverCallsign = receiver.getFsdCallsignString();
524  if (queryType == ClientQueryType::IsValidATC)
525  {
526  const ClientQuery clientQuery(getOwnCallsignAsString(), "SERVER", ClientQueryType::IsValidATC, queryData);
527  sendQueudedMessage(clientQuery);
528  }
529  else if (queryType == ClientQueryType::Capabilities)
530  {
531  const ClientQuery clientQuery(getOwnCallsignAsString(), reveiverCallsign, ClientQueryType::Capabilities);
532  sendQueudedMessage(clientQuery);
533  }
534  else if (queryType == ClientQueryType::Com1Freq)
535  {
536  const ClientQuery clientQuery(getOwnCallsignAsString(), reveiverCallsign, ClientQueryType::Com1Freq);
537  sendQueudedMessage(clientQuery);
538  }
539  else if (queryType == ClientQueryType::RealName)
540  {
541  const ClientQuery clientQuery(getOwnCallsignAsString(), reveiverCallsign, ClientQueryType::RealName);
542  sendQueudedMessage(clientQuery);
543  }
544  else if (queryType == ClientQueryType::Server)
545  {
546  ClientQuery clientQuery(getOwnCallsignAsString(), reveiverCallsign, ClientQueryType::Server);
547  sendQueudedMessage(clientQuery);
548  }
549  else if (queryType == ClientQueryType::ATIS)
550  {
551  const ClientQuery clientQuery(getOwnCallsignAsString(), reveiverCallsign, ClientQueryType::ATIS);
552  sendQueudedMessage(clientQuery);
553  if (m_serverType != ServerType::Vatsim)
554  {
555  m_pendingAtisQueries.insert(receiver, {});
556  }
557  }
558  else if (queryType == ClientQueryType::PublicIP)
559  {
560  const ClientQuery clientQuery(getOwnCallsignAsString(), reveiverCallsign, ClientQueryType::PublicIP);
561  sendQueudedMessage(clientQuery);
562  }
563  else if (queryType == ClientQueryType::INF)
564  {
565  const ClientQuery clientQuery(getOwnCallsignAsString(), reveiverCallsign, ClientQueryType::INF);
566  sendQueudedMessage(clientQuery);
567  }
568  else if (queryType == ClientQueryType::FP)
569  {
570  if (queryData.isEmpty()) { return; }
