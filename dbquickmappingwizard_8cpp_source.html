217  void CDbQuickMappingWizard::currentWizardPageChanged(int id)
218  {
219  const bool forward = id > m_lastId;
220  const bool colorMode = ui->rb_ColorLivery->isChecked();
221  m_lastId = id;
222 
223  const Pages page = static_cast<Pages>(id);
224  switch (page)
225  {
226  case PageAircraftSelect:
227  {
228  this->setAircraftIcaoFilter();
229  }
230  break;
231  case PageColor:
232  if (!colorMode)
233  {
234  forward ? this->next() : this->back();
235  }
236  break;
237  case PageLiverySelect:
238  {
239  if (colorMode)
240  {
241  this->setColorFilter();
242  }
243  else
244  {
245  this->setAirlineIcaoFilter();
246  }
247  }
248  break;
249  case PageDistributorSelect:
250  {
251  this->setDistributorFilter();
252  }
253  break;
254  case PageConfirmation:
255  {
256  const CSimulatorInfo sims = this->guessSimulator();
257  ui->selector_Simulator->setValue(sims);
258  ui->editor_AircraftModel->setLivery(this->getFirstSelectedOrDefaultLivery());
259  ui->editor_AircraftModel->setDistributor(this->getFirstSelectedOrDefaultDistributor());
260  ui->editor_AircraftModel->setAircraftIcao(this->getFirstSelectedOrDefaultAircraftIcao());
261  const CStatusMessageList msgs(this->validateData());
262  const bool errorFree = !msgs.hasWarningOrErrorMessages();
263  ui->fr_ConfirmationOk->setVisible(errorFree);
264  ui->fr_ConfirmationStillErrors->setVisible(!errorFree);
265  if (!errorFree)
266  {
267  ui->wp6_Confirmation->showOverlayMessages(msgs);
268  }
269  }
270  break;
271  case PageCredentials:
272  {
273  ui->comp_Log->clear();
274  }
275  break;
276  case PageLastConfirmation:
277  {
278  // void
279  }
280  break;
281  case PageSendStatus:
282  {
283  this->writeModelToDb();
284  this->button(BackButton)->hide();
285  }
286  break;
287  default:
288  break;
289  }
290  }

292  bool CDbQuickMappingWizard::validateCurrentPage()
293  {
294  const Pages page = static_cast<Pages>(this->currentId());
295  bool ok = false;
296  switch (page)
297  {
298  case PageConfirmation:
299  {
300  const CStatusMessageList msgs(this->validateData());
301  if (!msgs.isEmpty())
302  {
303  ui->wp6_Confirmation->showOverlayMessages(msgs);
304  }
305  ok = !msgs.hasWarningOrErrorMessages();
306  }
307  break;
308  case PageCredentials:
309  {
310  ok = ui->comp_DbLogin->isUserAuthenticated();
311  if (!ok)
312  {
313  ui->wp7_Credentials->showOverlayHTMLMessage("No user credentials, read login hints!", 10 * 1000);
314  }
315  }
316  break;
317  default:
318  {
319  ok = true;
320  }
321  break;
322  }
323 
324  return ok;
325  }

355  CSimulatorInfo CDbQuickMappingWizard::guessSimulator() const
356  {
357  CSimulatorInfo sims = m_model.getSimulator();
358  if (!sims.isAnySimulator() && m_model.hasModelString() && sGui && sGui->hasWebDataServices())
359  {
360  const CAircraftModel m = sGui->getWebDataServices()->getModelForModelString(m_model.getModelString());
361  if (m.isLoadedFromDb()) { sims = m.getSimulator(); }
362  }
363 
364  if (sGui && !sGui->isShuttingDown() && sGui->getIContextSimulator() && sGui->getIContextSimulator()->isSimulatorAvailable())
365  {
366  sims.add(sGui->getIContextSimulator()->getSimulatorPluginInfo().getSimulator());
367  }
368  return sims;
369  }

371  void CDbQuickMappingWizard::writeModelToDb()
372  {
373  if (!sGui || sGui->isShuttingDown() || !sGui->getWebDataServices() || !sGui->getWebDataServices()->getDatabaseWriter()) { return; }
374  this->consolidateModelWithUIData();
375 
376  // make sure the model is correctly excluded for XP etc.
377  static const QString qmw(u"[swift QMW " % CBuildConfig::getVersionString() % u"]");
378  if (m_model.getSimulator().isXPlane()) { m_model.setModelMode(CAircraftModel::Exclude); }
379  m_model.setVersion(CBuildConfig::getVersionString());
380  const QString extraInfo = QString(m_model.getDescription() % u" " % qmw).simplified().trimmed();
381 
382  const CStatusMessageList msgs = sGui->getWebDataServices()->getDatabaseWriter()->asyncPublishModel(m_model, extraInfo);
383  ui->comp_Log->appendStatusMessagesToList(msgs);
384  }
