105  u' ' % m_position.toQString(i18n) %
106  u" online: " % boolToYesNo(m_isOnline) %

108  // controller

110  u' ' % m_controller.toQString(i18n)) %
111 
112  // frequency
113  u' ' % m_frequency.valueRoundedWithUnit(3, i18n) %
114 
115  // ATIS
116  (!this->hasAtis() ? QString() :
117  u' ' % m_atis.toQString(i18n)) %
118 
119  // METAR
120  (!this->hasMetar() ? QString() :
121  u' ' % m_metar.toQString(i18n)) %
122 
123  // range
124  u' ' % (i18n ? rangeI18n : QStringLiteral("range")) %
125  u' ' % m_range.toQString(i18n) %
126 
127  // distance / bearing
128  u' ' % ICoordinateWithRelativePosition::convertToQString(i18n) %
129 
130  // booking from/until
131  u' ' %
132  (i18n ? fromUtcI18n : QStringLiteral("from(UTC)")) %
133  u' ' %
134  (m_bookedFromUtc.isNull() ? QStringLiteral("-") : m_bookedFromUtc.toString("yy-MM-dd HH:mm")) %

136  u' ' %

138  u' ' %

167  void CAtcStation::synchronizeWithBookedStation(CAtcStation &bookedStation)
168  {
169  if (bookedStation.getCallsign() != this->getCallsign()) { return; }
170 
171  // from online to booking
172  bookedStation.setOnline(true);
173  bookedStation.setFrequency(this->getFrequency());
174 
175  // Logoff Zulu Time set?
176  // comes directly from the online controller and is most likely more accurate
177  if (!this->getBookedUntilUtc().isNull())
178  {
179  bookedStation.setBookedUntilUtc(this->getBookedUntilUtc());
180  }
181 
182  // from booking to online
183  // booked now stations have valid data and need no update
184  if (!this->isBookedNow() && bookedStation.hasValidBookingTimes())
185  {
186  if (this->hasValidBookingTimes())
187  {
188  if (bookedStation.isBookedNow())
189  {
190  // can't get any better, we just copy from / to over
191  this->setBookedFromUntil(bookedStation);
192  }
193  else
194  {
195  // we already have some booking dates, we will verify those now
196  // and will set the most appropriate booking dates
197  const CTime timeDiffBooking(bookedStation.bookedWhen());
198  const CTime timeDiffOnline(this->bookedWhen()); // diff to now
199  if (timeDiffBooking.isNegativeWithEpsilonConsidered() && timeDiffOnline.isNegativeWithEpsilonConsidered())
200  {
201  // both in past
202  if (timeDiffBooking > timeDiffOnline)
203  {
204  this->setBookedFromUntil(bookedStation);
205  }
206  }
207  else if (timeDiffBooking.isPositiveWithEpsilonConsidered() && timeDiffOnline.isPositiveWithEpsilonConsidered())
208  {
209  // both in future
210  if (timeDiffBooking < timeDiffOnline)
211  {
212  this->setBookedFromUntil(bookedStation);
213  }
214  }
215  else if (timeDiffBooking.isPositiveWithEpsilonConsidered() && timeDiffOnline.isNegativeWithEpsilonConsidered())
216  {
217  // future booking is better than past booking
218  this->setBookedFromUntil(bookedStation);
219  }
220  }
221  }
222  else
223  {
224  // no booking info so far, so we just copy over
225  this->setBookedFromUntil(bookedStation);
226  }
227  }
228 
229  // both ways
230  this->synchronizeControllerData(bookedStation);
231  if (this->hasValidRelativeDistance())
232  {
233  bookedStation.setRelativeDistance(this->getRelativeDistance());
234  bookedStation.setRelativeBearing(this->getRelativeBearing());
235  }
236  else if (bookedStation.hasValidRelativeDistance())
237  {
238  this->setRelativeDistance(bookedStation.getRelativeDistance());
239  this->setRelativeBearing(bookedStation.getRelativeBearing());
240  }
241  }

268  bool CAtcStation::isBookedNow() const
269  {
270  if (!this->hasValidBookingTimes()) { return false; }
271  QDateTime now = QDateTime::currentDateTimeUtc();
272  if (m_bookedFromUtc > now) { return false; }
273  if (now > m_bookedUntilUtc) { return false; }
274  return true;

287  CTime CAtcStation::bookedWhen() const
288  {
289  if (!this->hasValidBookingTimes()) { return CTime(0, nullptr); }
290  QDateTime now = QDateTime::currentDateTimeUtc();
291  qint64 diffMs;
292  if (m_bookedFromUtc > now)
293  {
294  // future
295  diffMs = now.msecsTo(m_bookedFromUtc);
296  return CTime(diffMs / 1000.0, CTimeUnit::s());
297  }
298  else if (m_bookedUntilUtc > now)
299  {
300  // now
301  return CTime(0.0, CTimeUnit::s());
302  }
303  else
304  {
305  // past
306  diffMs = m_bookedUntilUtc.msecsTo(now);
307  return CTime(-diffMs / 1000.0, CTimeUnit::s());
308  }

358  QVariant CAtcStation::propertyByIndex(CPropertyIndexRef index) const
359  {
360  if (index.isMyself()) { return QVariant::fromValue(*this); }
361  const ColumnIndex i = index.frontCasted<ColumnIndex>();
362  switch (i)

365  case IndexBookedUntil: return QVariant::fromValue(m_bookedUntilUtc);
366  case IndexCallsign: return m_callsign.propertyByIndex(index.copyFrontRemoved());
367  case IndexCallsignString: return this->getCallsignAsString();
368  case IndexCallsignStringCrossCopuled: return this->getCallsignAsStringCrossCoupled();
369  case IndexController: return m_controller.propertyByIndex(index.copyFrontRemoved());
370  case IndexFrequency: return m_frequency.propertyByIndex(index.copyFrontRemoved());
371  case IndexIsOnline: return QVariant::fromValue(m_isOnline);
372  case IndexLatitude: return this->latitude().propertyByIndex(index.copyFrontRemoved());
373  case IndexLongitude: return this->longitude().propertyByIndex(index.copyFrontRemoved());
374  case IndexPosition: return m_position.propertyByIndex(index.copyFrontRemoved());
375  case IndexRange: return m_range.propertyByIndex(index.copyFrontRemoved());
376  case IndexIsInRange: return QVariant::fromValue(isInRange());
377  case IndexAtis: return m_atis.propertyByIndex(index.copyFrontRemoved());
378  case IndexMetar: return m_metar.propertyByIndex(index.copyFrontRemoved());
379  case IndexIsAfvCrossCoupled: return QVariant::fromValue(m_isAfvCrossCoupled);
380  default:
381  return (ICoordinateWithRelativePosition::canHandleIndex(index)) ?
382  ICoordinateWithRelativePosition::propertyByIndex(index) :
383  CValueObject::propertyByIndex(index);

387  void CAtcStation::setPropertyByIndex(CPropertyIndexRef index, const QVariant &variant)
388  {
389  if (index.isMyself())
390  {
391  (*this) = variant.value<CAtcStation>();
392  return;
393  }
394  const ColumnIndex i = index.frontCasted<ColumnIndex>();
395  switch (i)
396  {
397  case IndexBookedFrom: this->setBookedFromUtc(variant.value<QDateTime>()); break;
398  case IndexBookedUntil: this->setBookedUntilUtc(variant.value<QDateTime>()); break;
399  case IndexCallsign: m_callsign.setPropertyByIndex(index.copyFrontRemoved(), variant); break;
400  case IndexController: m_controller.setPropertyByIndex(index.copyFrontRemoved(), variant); break;
401  case IndexFrequency: m_frequency.setPropertyByIndex(index.copyFrontRemoved(), variant); break;
402  case IndexIsOnline: this->setOnline(variant.value<bool>()); break;
403  case IndexPosition: m_position.setPropertyByIndex(index.copyFrontRemoved(), variant); break;
404  case IndexRange: m_range.setPropertyByIndex(index.copyFrontRemoved(), variant); break;
405  case IndexAtis: m_atis.setPropertyByIndex(index.copyFrontRemoved(), variant); break;
406  case IndexMetar: m_metar.setPropertyByIndex(index.copyFrontRemoved(), variant); break;
407  case IndexIsAfvCrossCoupled: this->setAfvCrossCoupled(variant.value<bool>()); break;
408  case IndexCallsignString:
409  case IndexCallsignStringCrossCopuled:
410  {
411  const QString cs = variant.toString();
412  *this = CAtcStation();
413  this->setAfvCrossCoupled(cs.startsWith('*'));
414  }
415  break;
416  default:
417  if (ICoordinateWithRelativePosition::canHandleIndex(index))
418  {
419  ICoordinateWithRelativePosition::setPropertyByIndex(index, variant);
420  }
421  else
422  {
423  CValueObject::setPropertyByIndex(index, variant);
424  }
425  break;

429  int CAtcStation::comparePropertyByIndex(CPropertyIndexRef index, const CAtcStation &compareValue) const
430  {
431  if (index.isMyself()) { return this->getCallsign().comparePropertyByIndex(CPropertyIndexRef::empty(), compareValue.getCallsign()); }
432  const ColumnIndex i = index.frontCasted<ColumnIndex>();
433  switch (i)
434  {
435  case IndexBookedFrom: return Compare::compare(this->getBookedFromUtc(), compareValue.getBookedFromUtc());
436  case IndexBookedUntil: return Compare::compare(this->getBookedUntilUtc(), compareValue.getBookedUntilUtc());
437  case IndexCallsignString:
438  case IndexCallsignStringCrossCopuled:
439  return m_callsign.comparePropertyByIndex(CPropertyIndexRef::empty(), compareValue.getCallsign());
440  case IndexCallsign: return m_callsign.comparePropertyByIndex(index.copyFrontRemoved(), compareValue.getCallsign());
441  case IndexController: return m_controller.comparePropertyByIndex(index.copyFrontRemoved(), compareValue.getController());
442  case IndexFrequency: return m_frequency.comparePropertyByIndex(index.copyFrontRemoved(), compareValue.getFrequency());
443  case IndexIsOnline: return Compare::compare(this->isOnline(), compareValue.isOnline());
444  case IndexLatitude: return this->latitude().comparePropertyByIndex(index.copyFrontRemoved(), compareValue.latitude());
445  case IndexLongitude: return this->longitude().comparePropertyByIndex(index.copyFrontRemoved(), compareValue.longitude());
446  case IndexPosition: return m_position.comparePropertyByIndex(index.copyFrontRemoved(), compareValue.getPosition());
447  case IndexRange: return m_range.comparePropertyByIndex(index.copyFrontRemoved(), compareValue.getRange());
448  case IndexIsInRange: return Compare::compare(this->isInRange(), compareValue.isInRange());
449  case IndexAtis: return m_atis.comparePropertyByIndex(index.copyFrontRemoved(), compareValue.getAtis());
450  case IndexMetar: return m_metar.comparePropertyByIndex(index.copyFrontRemoved(), compareValue.getMetar());
451  case IndexIsAfvCrossCoupled: return Compare::compare(this->isAfvCrossCoupled(), compareValue.isAfvCrossCoupled());
452  default:
453  if (ICoordinateWithRelativePosition::canHandleIndex(index))
454  {
455  return ICoordinateWithRelativePosition::comparePropertyByIndex(index, compareValue);
456  }
457  return CValueObject::comparePropertyByIndex(index, *this);
458  }
459  Q_ASSERT_X(false, Q_FUNC_INFO, "Compare failed");
460  return 0;
