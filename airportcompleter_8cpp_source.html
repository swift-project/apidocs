33  void CAirportCompleter::setAirport(const CAirport &airport)
34  {
35  if (m_current == airport) { return; }
36  m_current = airport;
37  if (airport.getIcaoAsString() != ui->le_Icao->text()) { ui->le_Icao->setText(airport.getIcaoAsString()); }
38  if (airport.getDescriptiveName() != ui->le_Name->text()) { ui->le_Name->setText(airport.getDescriptiveName()); }
39  if (airport.getLocation() != ui->le_Location->text()) { ui->le_Location->setText(airport.getLocation()); }
40  emit this->changedAirport(m_current);
41  }

87  void CAirportCompleter::onAirportsChanged()

92  ui->le_Name->setCompleter(new QCompleter(airports.allDescriptivesNames(true), ui->le_Name));

112  ui->le_Location->completer()->popup()->setObjectName("AirportCompleter.Location");
113  const int w5chars = ui->le_Location->completer()->popup()->fontMetrics().size(Qt::TextSingleLine, "FooBa").width();
114  ui->le_Location->completer()->popup()->setMinimumWidth(w5chars * 10);
115  }
116 
117  // turn into airport when it was not possible before
118  if (m_current.isNull())
119  {
120  const QString icao = this->getIcaoText();
121  if (CAirportIcaoCode::isValidIcaoDesignator(icao, true))
122  {
123  const CAirport airport = sGui->getWebDataServices()->getAirportForIcaoDesignator(icao);
124  if (!airport.isNull()) { this->setAirport(airport); }
