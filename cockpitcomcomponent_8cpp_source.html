100  void CCockpitComComponent::updateCockpitFromContext(const CSimulatedAircraft &ownAircraft, const CIdentifier &originator)
101  {
102  if (isMyIdentifier(originator)) { return; } // comes from myself
103 
104  // update GUI elements
105  // avoid unnecessary change events as far as possible
106  const CComSystem com1 = ownAircraft.getCom1System(); // aircraft just updated or set from context
107  const CComSystem com2 = ownAircraft.getCom2System();
108  const CTransponder transponder = ownAircraft.getTransponder();
109 
110  // update the frequencies
111  ui->editor_Com->setFrequencies(com1, com2);
112 
113  // update transponder
114  ui->editor_Com->setTransponder(transponder);
115 
116  // selected stations
117  if (sGui)
118  {
119  if (sGui->getIContextNetwork())
120  {
121  const CAtcStationList selectedStations = sGui->getIContextNetwork()->getSelectedAtcStations();
122  ui->editor_Com->setSelectedAtcStations(selectedStations);
123  }
124  }
125  }
