35  CAircraftLights CAircraftSituation::guessLights() const
36  {
37  const bool isOnGround = getOnGround() == CAircraftSituation::OnGround;
38  const double gsKts = getGroundSpeed().value(CSpeedUnit::kts());
39  CAircraftLights lights;
40  lights.setCabinOn(true);
41  lights.setRecognitionOn(true);
42 
43  // when first detected moving, lights on
44  if (isOnGround)
45  {
46  lights.setTaxiOn(true);
47  lights.setBeaconOn(true);
48  lights.setNavOn(true);
49 
50  if (gsKts > 30)
51  {
52  // mode accelaration for takeoff
53  lights.setTaxiOn(false);
54  lights.setLandingOn(true);
55  }
56  else if (gsKts > 5)
57  {
58  // mode taxi
59  lights.setTaxiOn(true);
60  lights.setLandingOn(false);
61  }
62  else
63  {
64  // slow movements or parking
65  lights.setTaxiOn(false);
66  lights.setLandingOn(false);
67  }
68  }
69  else
70  {
71  // not on ground
72  lights.setTaxiOn(false);
73  lights.setBeaconOn(true);
74  lights.setNavOn(true);
75  // landing lights for < 10000ft (normally MSL, here ignored)
76  lights.setLandingOn(getAltitude().value(CLengthUnit::ft()) < 10000);
77  }
78  return lights;
79  }

257  CElevationPlane CAircraftSituation::interpolatedElevation(const CAircraftSituation &situation, const CAircraftSituation &oldSituation, const CAircraftSituation &newSituation, const CLength &distance)
258  {
259  if (oldSituation.isNull() || newSituation.isNull()) { return CElevationPlane::null(); }
260  if (!oldSituation.hasGroundElevation() || !newSituation.hasGroundElevation()) { return CElevationPlane::null(); }
261  if (oldSituation.equalNormalVectorDouble(newSituation)) { return newSituation.getGroundElevationPlane(); }
262 
263  const double newElvFt = newSituation.getGroundElevation().value(CLengthUnit::ft());
264  const double oldElvFt = oldSituation.getGroundElevation().value(CLengthUnit::ft());
265  const double deltaElvFt = newElvFt - oldElvFt;
266  if (deltaElvFt > MaxDeltaElevationFt) { return CElevationPlane::null(); } // threshold, interpolation not possible
267 
268  if (!situation.isNull())
269  {
270  if (CBuildConfig::isLocalDeveloperDebugBuild())
271  {
272  Q_ASSERT_X(situation.isValidVectorRange(), Q_FUNC_INFO, "Invalid range");
273  Q_ASSERT_X(oldSituation.isValidVectorRange(), Q_FUNC_INFO, "Invalid range");
274  Q_ASSERT_X(newSituation.isValidVectorRange(), Q_FUNC_INFO, "Invalid range");
275  }
276 
277  const double distanceSituationNewM = situation.calculateGreatCircleDistance(newSituation).value(CLengthUnit::m());
278  if (distanceSituationNewM < 5.0) { return newSituation.getGroundElevationPlane(); }
279 
280  const double distanceOldNewM = (distance.isNull() ? oldSituation.calculateGreatCircleDistance(newSituation) : distance).value(CLengthUnit::m());
281  if (distanceOldNewM < 5.0) { return oldSituation.getGroundElevationPlane(); }
282 
283  const double distRatio = distanceSituationNewM / distanceOldNewM;
284 
285  // very close to the situations we return their elevation
286  if (distRatio < 0.05) { return newSituation.getGroundElevationPlane(); }
287  if (distRatio > 0.95) { return oldSituation.getGroundElevationPlane(); }
288 
289  const double situationElvFt = newElvFt - distRatio * deltaElvFt;
290  return CElevationPlane(situation, situationElvFt, CElevationPlane::singlePointRadius());
291  }
292  else
293  {
294  const double elvSumFt = oldElvFt + newElvFt;
295  const double elvFt = 0.5 * elvSumFt;
296  return CElevationPlane(newSituation, elvFt, CElevationPlane::singlePointRadius());
297  }

524  void CAircraftSituation::setOnGroundFactor(double groundFactor)
525  {
526  double gf = groundFactor;
527  do
528  {
529  if (groundFactor < 0.0)
530  {

532  break;
533  }
