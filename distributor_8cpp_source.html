59  QVariant CDistributor::propertyByIndex(CPropertyIndexRef index) const
60  {
61  if (index.isMyself()) { return QVariant::fromValue(*this); }
62  if (IDatastoreObjectWithStringKey::canHandleIndex(index)) { return IDatastoreObjectWithStringKey::propertyByIndex(index); }
63  if (IOrderable::canHandleIndex(index)) { return IOrderable::propertyByIndex(index); }
64 
65  const ColumnIndex i = index.frontCasted<ColumnIndex>();
66  switch (i)
67  {
68  case IndexAlias1: return QVariant::fromValue(m_alias1);
69  case IndexAlias2: return QVariant::fromValue(m_alias2);
70  case IndexDescription: return QVariant::fromValue(m_description);
71  case IndexSimulator: return m_simulator.propertyByIndex(index.copyFrontRemoved());
72  default:
73  return CValueObject::propertyByIndex(index);
74  }
75  }

102  default: CValueObject::setPropertyByIndex(index, variant); break;
103  }

106  int CDistributor::comparePropertyByIndex(CPropertyIndexRef index, const CDistributor &compareValue) const
107  {
108  if (IDatastoreObjectWithStringKey::canHandleIndex(index)) { return IDatastoreObjectWithStringKey::comparePropertyByIndex(index, compareValue); }
109  if (IOrderable::canHandleIndex(index)) { return IOrderable::comparePropertyByIndex(index, compareValue); }
110 
111  const ColumnIndex i = index.frontCasted<ColumnIndex>();
112  switch (i)
113  {
114  case IndexAlias1: return m_alias1.compare(compareValue.m_alias1, Qt::CaseInsensitive);
115  case IndexAlias2: return m_alias2.compare(compareValue.m_alias2, Qt::CaseInsensitive);
116  case IndexDescription: return m_description.compare(compareValue.getDescription(), Qt::CaseInsensitive);
117  case IndexSimulator: return m_simulator.comparePropertyByIndex(index.copyFrontRemoved(), compareValue.m_simulator);
118  default:
119  break;
120  }
121  Q_ASSERT_X(false, Q_FUNC_INFO, "Compare failed");
122  return 0;
123  }

148  void CDistributor::updateMissingParts(const CDistributor &otherDistributor)
149  {
150  if (!this->hasValidDbKey() && otherDistributor.hasValidDbKey())
151  {
152  // we have no DB data, but the other one has
153  // so we change roles. We take the DB object as base, and update our parts
154  CDistributor copy(otherDistributor);
155  copy.updateMissingParts(*this);
156  *this = copy;
157  return;
158  }
159 
160  if (!this->hasAlias1()) { this->setAlias1(otherDistributor.getAlias1()); }
161  if (!this->hasAlias2()) { this->setAlias1(otherDistributor.getAlias2()); }
162  if (!this->hasDescription()) { this->setDescription(otherDistributor.getDescription()); }
163  }

165  CDistributor CDistributor::fromDatabaseJson(const QJsonObject &json, const QString &prefix)
166  {
167  if (!existsKey(json, prefix))
168  {
169  // when using relationship, this can be null
170  return CDistributor();
171  }
172 
173  const QString description(json.value(prefix % u"description").toString());
174  if (description.isEmpty())
175  {
176  // stub, only key, maybe also timestamps
177  CDistributor distributorStub;
178  distributorStub.setKeyVersionTimestampFromDatabaseJson(json, prefix);
179  return distributorStub;
180  }
181 
182  const CSimulatorInfo simulator = CSimulatorInfo::fromDatabaseJson(json, prefix);
183  const QString alias1(json.value(prefix % u"alias1").toString());
184  const QString alias2(json.value(prefix % u"alias2").toString());
185  Q_ASSERT_X(!description.isEmpty(), Q_FUNC_INFO, "Missing description");
186  CDistributor distributor("", description, alias1, alias2, simulator);
187  distributor.setKeyVersionTimestampFromDatabaseJson(json, prefix);
188  distributor.setLoadedFromDb(true);
189  return distributor;
190  }
