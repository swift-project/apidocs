57  void CUserComponent::update()
58  {
59  if (!sGui || sGui->isShuttingDown() || !sGui->getIContextNetwork()) { return; }
60  Q_ASSERT(ui->tvp_AllUsers);
61  Q_ASSERT(ui->tvp_Clients);
62 
63  if (sGui->getIContextNetwork()->isConnected())
64  {
65  bool withData = countUsers() > 0 || countClients() > 0;
66  if (withData && !isVisibleWidget())
67  {
68  // Skip update, invisible
69  return;
70  }
71 
72  // load data
73  const CUserList users = sGui->getIContextNetwork()->getUsers();
74  ui->tvp_AllUsers->updateContainer(users);
75  ui->tvp_Clients->updateContainer(sGui->getIContextNetwork()->getClientsForCallsigns(users.getCallsigns()));
76  }
77  }
