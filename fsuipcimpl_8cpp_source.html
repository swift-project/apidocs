69  bool CFsuipc::open(bool force)
70  {
71  Q_ASSERT_X(CThreadUtils::isInThisThread(this), Q_FUNC_INFO, "Open not threadsafe");
72  DWORD dwResult;
73  m_lastErrorMessage = "";
74  m_lastErrorIndex = 0;
75  if (!force && m_opened) { return m_opened; } // already connected
76 
77  if (FSUIPC_Open(SIM_ANY, &dwResult))
78  {
79  m_opened = true; // temp status
80  m_openCount++;
81 
82  if (this->isOpen())
83  {
84  const int simIndex = static_cast<int>(FSUIPC_FS_Version);
85  const QString sim = CFsuipc::simulator(simIndex);
86  const QString ver = QStringLiteral("%1.%2.%3.%4%5")
87  .arg(QLatin1Char(48 + (0x0f & (FSUIPC_Version >> 28))))
88  .arg(QLatin1Char(48 + (0x0f & (FSUIPC_Version >> 24))))
89  .arg(QLatin1Char(48 + (0x0f & (FSUIPC_Version >> 20))))
90  .arg(QLatin1Char(48 + (0x0f & (FSUIPC_Version >> 16))))
91  .arg((FSUIPC_Version & 0xffff) ? QString(QLatin1Char('a' + static_cast<char>(FSUIPC_Version & 0xff) - 1)) : "");
92  m_fsuipcVersion = QStringLiteral("FSUIPC %1 (%2)").arg(ver, sim);
93  CLogMessage(this).info(u"FSUIPC connected: %1") << m_fsuipcVersion;
94  }
95  else
96  {
97  CLogMessage(this).warning(u"FSUIPC opened, but verification failed");
98  m_opened = false;
99  FSUIPC_Close(); // under any circumstances close
100  }
101  }
102  else
103  {
104  const int index = static_cast<int>(dwResult);
105  m_lastErrorIndex = index;
106  m_lastErrorMessage = CFsuipc::errorMessages().at(index);
107  CLogMessage(this).warning(u"FSUIPC not connected: %1") << m_lastErrorMessage;
108  m_opened = false;
109  FSUIPC_Close(); // under any circumstances close
110  }
111 
112  return m_opened;
113  }

116  {
117  Q_ASSERT_X(CThreadUtils::isInThisThread(this), Q_FUNC_INFO, "Open not threadsafe");
118  if (m_opened)

120  CLogMessage(this).info(u"Closing FSUIPC: %1") << m_fsuipcVersion;

123  m_closeCount++;

146  bool CFsuipc::write(const CSimulatedAircraft &aircraft)
147  {
148  Q_ASSERT_X(CThreadUtils::isInThisThread(this), Q_FUNC_INFO, "Open not threadsafe");
149  if (!this->isOpened()) { return false; }
150 
151  quint16 com1ActiveRaw = static_cast<quint16>(aircraft.getCom1System().getFrequencyActive().value(CFrequencyUnit::MHz()) * 100);
152  quint16 com2ActiveRaw = static_cast<quint16>(aircraft.getCom2System().getFrequencyActive().value(CFrequencyUnit::MHz()) * 100);
153  quint16 com1StandbyRaw = static_cast<quint16>(aircraft.getCom1System().getFrequencyStandby().value(CFrequencyUnit::MHz()) * 100);
154  quint16 com2StandbyRaw = static_cast<quint16>(aircraft.getCom2System().getFrequencyStandby().value(CFrequencyUnit::MHz()) * 100);
155  com1ActiveRaw = static_cast<quint16>(CBcdConversions::dec2Bcd(com1ActiveRaw - 10000));
156  com2ActiveRaw = static_cast<quint16>(CBcdConversions::dec2Bcd(com2ActiveRaw - 10000));
157  com1StandbyRaw = static_cast<quint16>(CBcdConversions::dec2Bcd(com1StandbyRaw - 10000));
158  com2StandbyRaw = static_cast<quint16>(CBcdConversions::dec2Bcd(com2StandbyRaw - 10000));
159  quint16 transponderCodeRaw = static_cast<quint16>(aircraft.getTransponderCode());
160  transponderCodeRaw = static_cast<quint16>(CBcdConversions::dec2Bcd(transponderCodeRaw));
161 
162  DWORD dwResult = 0;
163  bool ok =
164  FSUIPC_Write(0x034e, 2, &com1ActiveRaw, &dwResult) &&
165  FSUIPC_Write(0x3118, 2, &com2ActiveRaw, &dwResult) &&
166  FSUIPC_Write(0x311a, 2, &com1StandbyRaw, &dwResult) &&
167  FSUIPC_Write(0x311c, 2, &com2StandbyRaw, &dwResult) &&
168  FSUIPC_Write(0x0354, 2, &transponderCodeRaw, &dwResult);
169  if (ok) { FSUIPC_Process(&dwResult); }
170 
171  ok = ok && write(aircraft.getTransponder());
172 
173  return ok && dwResult == 0;
174  }

192  bool CFsuipc::write(const CWeatherGrid &weatherGrid)
193  {
194  Q_ASSERT_X(CThreadUtils::isInThisThread(this), Q_FUNC_INFO, "Open not threadsafe");
195  if (!this->isOpened()) { return false; }
196 
197  if (weatherGrid.isEmpty()) { return false; }
198  this->clearAllWeather();
199  CLogMessage(this).debug(u"FSUIPC cleared weather");
200 
201  const CGridPoint gridPoint = weatherGrid.front();
202  NewWeather nw;
203 
204  // Clear new weather
205  nw.uCommand = NW_SET;
206  nw.uFlags = 0;
207  nw.ulSignature = 0;
208  nw.uDynamics = 0;
209  for (std::size_t i = 0; i < sizeof(nw.uSpare) / sizeof(nw.uSpare[0]); i++) { nw.uSpare[i] = 0; }
210 
211  nw.dLatitude = 0.0;
212  nw.dLongitude = 0.0;
213  nw.nElevation = 0;
214  nw.ulTimeStamp = 0;
215  nw.nTempCtr = 0;
216  nw.nWindsCtr = 0;
217  nw.nCloudsCtr = 0;
218  nw.nElevation = 0; // meters * 65536;
219  nw.nUpperVisCtr = 0;
220 
221  // todo: Take station from weather grid
222  memcpy(nw.chICAO, "GLOB", 4);
223 
224  const CVisibilityLayerList visibilityLayers = gridPoint.getVisibilityLayers().sortedBy(&CVisibilityLayer::getBase);
225  const auto surfaceVisibility = visibilityLayers.frontOrDefault();
226  NewVis vis;
227  vis.LowerAlt = static_cast<short>(surfaceVisibility.getBase().valueInteger(CLengthUnit::m()));
228  vis.UpperAlt = static_cast<ushort>(surfaceVisibility.getTop().valueInteger(CLengthUnit::m()));
229  // Range is measured in: 1/100ths sm
230  vis.Range = static_cast<ushort>(surfaceVisibility.getVisibility().value(CLengthUnit::SM()) * 100);
231  vis.Spare = 0;
232  nw.Vis = vis;
233 
234  for (const auto &visibilityLayer : visibilityLayers)
235  {
236  vis.LowerAlt = static_cast<short>(visibilityLayer.getBase().valueInteger(CLengthUnit::m()));
237  vis.UpperAlt = static_cast<ushort>(visibilityLayer.getTop().valueInteger(CLengthUnit::m()));
238  vis.Range = static_cast<ushort>(visibilityLayer.getVisibility().value(CLengthUnit::SM()) * 100);
239  vis.Spare = 0;
240  nw.UpperVis[nw.nUpperVisCtr++] = vis;
241  }
242  CLogMessage(this).debug(u"FSUIPC updated %1 visibility layers") << visibilityLayers.size();
243 
244  const CTemperatureLayerList temperatureLayers = gridPoint.getTemperatureLayers().sortedBy(&CTemperatureLayer::getLevel);
245  for (const auto &temperatureLayer : temperatureLayers)
246  {
247  NewTemp temp;
248  temp.Alt = static_cast<ushort>(temperatureLayer.getLevel().valueInteger(CLengthUnit::m()));
249  temp.Day = static_cast<short>(temperatureLayer.getTemperature().valueInteger(CTemperatureUnit::C()));
250  temp.DayNightVar = 3;
251  temp.DewPoint = static_cast<short>(temperatureLayer.getDewPoint().value(CTemperatureUnit::C()));
252  nw.Temp[nw.nTempCtr++] = temp;
253  }
254  CLogMessage(this).debug(u"FSUIPC updated %1 temperature layers") << temperatureLayers.size();
255 
256  const CCloudLayerList cloudLayers = gridPoint.getCloudLayers().sortedBy(&CCloudLayer::getBase);
257  for (const auto &cloudLayer : cloudLayers)
258  {
259  NewCloud cloud;
260  switch (cloudLayer.getCoverage())
261  {
262  case CCloudLayer::None: cloud.Coverage = 0; break;
263  case CCloudLayer::Few: cloud.Coverage = 2; break;
264  case CCloudLayer::Scattered: cloud.Coverage = 4; break;
265  case CCloudLayer::Broken: cloud.Coverage = 6; break;
266  case CCloudLayer::Overcast: cloud.Coverage = 8; break;
267  default: cloud.Coverage = 0; break;
268  }
269 
270  cloud.Deviation = 0;
271  cloud.Icing = 0;
272  cloud.LowerAlt = static_cast<ushort>(cloudLayer.getBase().valueInteger(CLengthUnit::m()));
273  cloud.PrecipBase = 0;
274 
275  // Light rain - when the precipitation rate is < 2.5 mm (0.098 in) per hour
276  // Moderate rain - when the precipitation rate is between 2.5 mm (0.098 in) - 7.6 mm (0.30 in) or 10 mm (0.39 in) per hour
277  // Heavy rain - when the precipitation rate is > 7.6 mm (0.30 in) per hour, or between 10 mm (0.39 in) and 50 mm (2.0 in) per hour
278  // Violent rain - when the precipitation rate is > 50 mm (2.0 in) per hour
279 
280  cloud.PrecipRate = 2 * static_cast<unsigned char>(cloudLayer.getPrecipitationRate());
281  cloud.PrecipType = static_cast<unsigned char>(cloudLayer.getPrecipitation());
282  cloud.TopShape = 0;
283  cloud.Turbulence = 0;
284 
285  switch (cloudLayer.getClouds())
286  {
287  case CCloudLayer::NoClouds: cloud.Type = 0; break;
288  case CCloudLayer::Cirrus: cloud.Type = 1; break;
289  case CCloudLayer::Stratus: cloud.Type = 8; break;
290  case CCloudLayer::Cumulus: cloud.Type = 9; break;
291  case CCloudLayer::Thunderstorm: cloud.Type = 10; break;
292  default: cloud.Type = 0;
293  }
294 
295  cloud.UpperAlt = static_cast<ushort>(cloudLayer.getTop().valueInteger(CLengthUnit::m()));
296  nw.Cloud[nw.nCloudsCtr++] = cloud;
297  }
298  CLogMessage(this).debug(u"FSUIPC updated %1 cloud layers") << cloudLayers.size();
299 
300  const CWindLayerList windLayers = gridPoint.getWindLayers().sortedBy(&CWindLayer::getLevel);
301  for (const auto &windLayer : windLayers)
302  {
303  NewWind wind;
304  wind.Direction = static_cast<ushort>(windLayer.getDirection().value(CAngleUnit::deg()) * 65536 / 360.0);
305  wind.GapAbove = 0;
306  wind.Gust = static_cast<ushort>(windLayer.getGustSpeed().valueInteger(CSpeedUnit::kts()));
307  wind.Shear = 0;
308  wind.Speed = static_cast<ushort>(windLayer.getSpeed().valueInteger(CSpeedUnit::kts()));
309  wind.SpeedFract = 0;
310  wind.Turbulence = 0;
311  wind.UpperAlt = static_cast<ushort>(windLayer.getLevel().valueInteger(CLengthUnit::m()));
312  wind.Variance = 0;
313  nw.Wind[nw.nWindsCtr++] = wind;
314  }
315  CLogMessage(this).debug(u"FSUIPC updated %1 wind layers") << windLayers.size();
316 
317  NewPress press;
318  press.Drift = 0;
319  // Pressure is measured in: 16 x mb
320  press.Pressure = static_cast<ushort>(gridPoint.getPressureAtMsl().value(CPressureUnit::mbar()) * 16);
321  nw.Press = press;
322 
323  // writing will take place in
324  QByteArray weatherData(reinterpret_cast<const char *>(&nw), sizeof(NewWeather));
325  m_weatherMessageQueue.append(FsuipcWeatherMessage(0xC800, weatherData, 5));
326  return true;
327  }

351  bool CFsuipc::read(CSimulatedAircraft &aircraft, bool cockpit, bool situation, bool aircraftParts)
352  {
353  Q_ASSERT_X(CThreadUtils::isInThisThread(this), Q_FUNC_INFO, "Open not threadsafe");
354  DWORD dwResult = 0;
355  char localFsTimeRaw[3];
356  char modelNameRaw[256];
357  qint16 com1ActiveRaw = 0, com2ActiveRaw = 0, com1StandbyRaw = 0, com2StandbyRaw = 0;
358  qint16 transponderCodeRaw = 0;
359  byte xpdrModeSb3Raw = 1, xpdrIdentSb3Raw = 1;
360  qint32 groundspeedRaw = 0, pitchRaw = 0, bankRaw = 0, headingRaw = 0;
361  qint64 altitudeRaw = 0;
362  double pressureAltitudeRaw = 0; // 34B0
363  qint32 groundAltitudeRaw = 0;
364  qint64 latitudeRaw = 0, longitudeRaw = 0;
365  qint16 lightsRaw = 0;
366  qint16 onGroundRaw = 0;
367  qint32 flapsControlRaw = 0, gearControlRaw = 0, spoilersControlRaw = 0;
368  qint16 numberOfEngines = 0;
369  qint16 engine1CombustionFlag = 0, engine2CombustionFlag = 0, engine3CombustionFlag = 0, engine4CombustionFlag = 0;
370  double velocityWorld[3];
371  double rotationVelocityBody[3];
372 
373  // http://www.projectmagenta.com/all-fsuipc-offsets/
374  // https://www.ivao.aero/softdev/ivap/fsuipc_sdk.asp
375  // http://squawkbox.ca/doc/sdk/fsuipc.php
376 
377  if (!this->isOpened()) { return false; }
378  if (!(aircraftParts || situation || cockpit)) { return false; }
379 
380  bool read = false;
381  bool cockpitN = !cockpit;
382  bool situationN = !situation;
383  bool aircraftPartsN = !aircraftParts;
384 
391 
404  if (
405  FSUIPC_Read(0x0238, 3, localFsTimeRaw, &dwResult) &&
406 
407  // COM settings
408  (cockpitN || FSUIPC_Read(0x034e, 2, &com1ActiveRaw, &dwResult)) &&
409  (cockpitN || FSUIPC_Read(0x3118, 2, &com2ActiveRaw, &dwResult)) &&
410  (cockpitN || FSUIPC_Read(0x311a, 2, &com1StandbyRaw, &dwResult)) &&
411  (cockpitN || FSUIPC_Read(0x311c, 2, &com2StandbyRaw, &dwResult)) &&
412  (cockpitN || FSUIPC_Read(0x0354, 2, &transponderCodeRaw, &dwResult)) &&
413 
414  // COM Settings, transponder, SB3
415  (cockpitN || FSUIPC_Read(0x7b91, 1, &xpdrModeSb3Raw, &dwResult)) &&
416  (cockpitN || FSUIPC_Read(0x7b93, 1, &xpdrIdentSb3Raw, &dwResult)) &&
417 
418  // Speeds, situation
419  (situationN || FSUIPC_Read(0x02b4, 4, &groundspeedRaw, &dwResult)) &&
420  (situationN || FSUIPC_Read(0x0578, 4, &pitchRaw, &dwResult)) &&
421  (situationN || FSUIPC_Read(0x057c, 4, &bankRaw, &dwResult)) &&
422  (situationN || FSUIPC_Read(0x0580, 4, &headingRaw, &dwResult)) &&
423  (situationN || FSUIPC_Read(0x0570, 8, &altitudeRaw, &dwResult)) &&
424 
425  (situationN || FSUIPC_Read(0x3198, 8, &velocityWorld[0], &dwResult)) &&
426  (situationN || FSUIPC_Read(0x31a0, 8, &velocityWorld[1], &dwResult)) &&
427  (situationN || FSUIPC_Read(0x3190, 8, &velocityWorld[2], &dwResult)) &&
428 
429  (situationN || FSUIPC_Read(0x30A8, 8, &rotationVelocityBody[0], &dwResult)) &&
430  (situationN || FSUIPC_Read(0x30B0, 8, &rotationVelocityBody[1], &dwResult)) &&
431  (situationN || FSUIPC_Read(0x30B8, 8, &rotationVelocityBody[2], &dwResult)) &&
432 
433  // Position
434  (situationN || FSUIPC_Read(0x0560, 8, &latitudeRaw, &dwResult)) &&
435  (situationN || FSUIPC_Read(0x0568, 8, &longitudeRaw, &dwResult)) &&
436  (situationN || FSUIPC_Read(0x0020, 4, &groundAltitudeRaw, &dwResult)) &&
437  (situationN || FSUIPC_Read(0x34B0, 8, &pressureAltitudeRaw, &dwResult)) &&
438 
439  // model name
440  FSUIPC_Read(0x3d00, 256, &modelNameRaw, &dwResult) &&
441 
442  // aircraft parts
443  (aircraftPartsN || FSUIPC_Read(0x0D0C, 2, &lightsRaw, &dwResult)) &&
444  (aircraftPartsN || FSUIPC_Read(0x0366, 2, &onGroundRaw, &dwResult)) &&
445  (aircraftPartsN || FSUIPC_Read(0x0BDC, 4, &flapsControlRaw, &dwResult)) &&
446  (aircraftPartsN || FSUIPC_Read(0x0BE8, 4, &gearControlRaw, &dwResult)) &&
447  (aircraftPartsN || FSUIPC_Read(0x0BD0, 4, &spoilersControlRaw, &dwResult)) &&
448 
449  // engines
450  (aircraftPartsN || FSUIPC_Read(0x0AEC, 2, &numberOfEngines, &dwResult)) &&
451  (aircraftPartsN || FSUIPC_Read(0x0894, 2, &engine1CombustionFlag, &dwResult)) &&
452  (aircraftPartsN || FSUIPC_Read(0x092C, 2, &engine2CombustionFlag, &dwResult)) &&
453  (aircraftPartsN || FSUIPC_Read(0x09C4, 2, &engine3CombustionFlag, &dwResult)) &&
454  (aircraftPartsN || FSUIPC_Read(0x0A5C, 2, &engine4CombustionFlag, &dwResult)) &&
455 
456  // If we wanted other reads/writes at the same time, we could put them here
457  FSUIPC_Process(&dwResult))
458  {
459  read = true;
460 
461  if (cockpit)
462  {
463  // COMs
464  CComSystem com1 = aircraft.getCom1System();
465  CComSystem com2 = aircraft.getCom2System();
466  CTransponder xpdr = aircraft.getTransponder();
467 
468  // 2710 => 12710 => / 100.0 => 127.1
469  com1ActiveRaw = static_cast<short>(10000 + CBcdConversions::bcd2Dec(com1ActiveRaw));
470  com2ActiveRaw = static_cast<short>(10000 + CBcdConversions::bcd2Dec(com2ActiveRaw));
471  com1StandbyRaw = static_cast<short>(10000 + CBcdConversions::bcd2Dec(com1StandbyRaw));
472  com2StandbyRaw = static_cast<short>(10000 + CBcdConversions::bcd2Dec(com2StandbyRaw));
473  com1.setFrequencyActiveMHz(com1ActiveRaw / 100.0);
474  com2.setFrequencyActiveMHz(com2ActiveRaw / 100.0);
475  com1.setFrequencyStandbyMHz(com1StandbyRaw / 100.0);
476  com2.setFrequencyStandbyMHz(com2StandbyRaw / 100.0);
477 
478  transponderCodeRaw = static_cast<qint16>(CBcdConversions::bcd2Dec(transponderCodeRaw));
479  xpdr.setTransponderCode(transponderCodeRaw);
480 
481  // Mode by SB3
482  xpdr.setTransponderMode(xpdrModeSb3Raw == 0 ? CTransponder::ModeC : CTransponder::StateStandby);
483  if (xpdrIdentSb3Raw != 0)
484  {
485  // will be reset in CFsuipc::write
486  xpdr.setTransponderMode(CTransponder::StateIdent);
487  }
488  aircraft.setCockpit(com1, com2, xpdr);
489  } // cockpit
490 
491  if (situation)
492  {
493  // position
494  const double latCorrectionFactor = 90.0 / (10001750.0 * 65536.0 * 65536.0);
495  const double lonCorrectionFactor = 360.0 / (65536.0 * 65536.0 * 65536.0 * 65536.0);
496  CAircraftSituation situation = aircraft.getSituation(); // cppcheck-suppress shadowArgument
497  CCoordinateGeodetic position = situation.getPosition();
498  CLatitude lat(latitudeRaw * latCorrectionFactor, CAngleUnit::deg());
499  CLongitude lon(longitudeRaw * lonCorrectionFactor, CAngleUnit::deg());
500  CAltitude groundAltitude(groundAltitudeRaw / 256.0, CAltitude::MeanSeaLevel, CLengthUnit::m());
501  position.setLatitude(lat);
502  position.setLongitude(lon);
503  position.setGeodeticHeight(groundAltitude);
504  situation.setPosition(position);
505 
506  const double angleCorrectionFactor = 360.0 / 65536.0 / 65536.0; // see FSUIPC docu
507  pitchRaw = qRound(std::floor(pitchRaw * angleCorrectionFactor));
508  bankRaw = qRound(std::floor(bankRaw * angleCorrectionFactor));
509 
510  // MSFS has inverted pitch and bank angles
511  pitchRaw = ~pitchRaw;
512  bankRaw = ~bankRaw;

604  void CFsuipc::processWeatherMessages()
605  {
606  if (m_weatherMessageQueue.isEmpty()) { return; }
607  if (!m_opened) { return; }
608  FsuipcWeatherMessage &weatherMessage = m_weatherMessageQueue.first();
609 
610  DWORD dwResult;
611  weatherMessage.m_leftTrials--;
612  FSUIPC_Write(static_cast<DWORD>(weatherMessage.m_offset), static_cast<DWORD>(weatherMessage.m_messageData.size()), reinterpret_cast<void *>(weatherMessage.m_messageData.data()), &dwResult);
613 
614  unsigned int timeStamp = 0;
615  FSUIPC_Read(0xC824, sizeof(timeStamp), &timeStamp, &dwResult);
616  FSUIPC_Process(&dwResult);
617  if (timeStamp > m_lastTimestamp)
618  {
619  m_weatherMessageQueue.removeFirst();
620  m_lastTimestamp = timeStamp;
621  return;
622  }
623 
624  if (weatherMessage.m_leftTrials < 1)
625  {
626  CLogMessage(this).debug() << "Number of trials reached for weather message. Dropping it.";
627  m_weatherMessageQueue.removeFirst();
628  }
