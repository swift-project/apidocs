48  bool CAirportDataReader::readFromJsonFilesInBackground(const QString &dir, CEntityFlags::Entity whatToRead, bool overrideNewerOnly)
49  {
50  if (dir.isEmpty() || whatToRead == CEntityFlags::NoEntity) { return false; }
51 
52  QPointer<CAirportDataReader> myself(this);
53  QTimer::singleShot(0, this, [=]() {
54  if (!myself) { return; }
55  const CStatusMessageList msgs = this->readFromJsonFiles(dir, whatToRead, overrideNewerOnly);
56  if (msgs.isFailure())
57  {
58  CLogMessage::preformatted(msgs);
59  }
60  });
61  return true;

64  CStatusMessageList CAirportDataReader::readFromJsonFiles(const QString &dir, CEntityFlags::Entity whatToRead, bool overrideNewerOnly)
65  {
66  const QDir directory(dir);
67  if (!directory.exists())
68  {
69  return CStatusMessage(this).error(u"Missing directory '%1'") << dir;
70  }
71 
72  whatToRead &= CEntityFlags::AirportEntity; // can handle these entities
73  if (whatToRead == CEntityFlags::NoEntity)
74  {
75  return CStatusMessage(this).info(u"'%1' No entity for this reader") << CEntityFlags::flagToString(whatToRead);
76  }
77 
78  int c = 0;
79  CEntityFlags::Entity reallyRead = CEntityFlags::NoEntity;
80 
81  CStatusMessageList msgs;
82  const QString fileName = CFileUtils::appendFilePaths(dir, "airports.json");
83  const QFileInfo fi(fileName);
84  if (!fi.exists())
85  {
86  msgs.push_back(CStatusMessage(this).warning(u"File '%1' does not exist") << fileName);
87  }
88  else if (!this->overrideCacheFromFile(overrideNewerOnly, fi, CEntityFlags::AirportEntity, msgs))
89  {
90  // void
91  }
92  else
93  {
94  const QUrl url = QUrl::fromLocalFile(fi.absoluteFilePath());
95  const QJsonObject airportsJson(CDatabaseUtils::readQJsonObjectFromDatabaseFile(fileName));
96  if (!airportsJson.isEmpty())
97  {
98  try
99  {
100  const CAirportList airports = CAirportList::fromMultipleJsonFormats(airportsJson);
101  c = airports.size();
102  msgs.push_back(m_airportCache.set(airports, fi.birthTime().toUTC().toMSecsSinceEpoch()));
103 
104  emit dataRead(CEntityFlags::AirportEntity, CEntityFlags::ReadFinished, c, url);
105  reallyRead |= CEntityFlags::AirportEntity;
106  }
107  catch (const CJsonException &ex)
108  {
109  emit dataRead(CEntityFlags::AirportEntity, CEntityFlags::ReadFailed, 0, url);
110  return CStatusMessage::fromJsonException(ex, this, QStringLiteral("Reading airports from '%1'").arg(fileName));
111  }
112  }
113  }
114  return msgs;
115  }

184  void CAirportDataReader::parseAirportData(QNetworkReply *nwReplyPtr)
185  {
186  // wrap pointer, make sure any exit cleans up reply
187  // required to use delete later as object is created in a different thread
188  QScopedPointer<QNetworkReply, QScopedPointerDeleteLater> nwReply(nwReplyPtr);
189  if (!this->doWorkCheck()) { return; }
190 
191  const CDatabaseReader::JsonDatastoreResponse res = this->setStatusAndTransformReplyIntoDatastoreResponse(nwReplyPtr);
192  const QUrl url = nwReply->url();
193 
194  if (res.hasErrorMessage())
195  {
196  CLogMessage::preformatted(res.lastWarningOrAbove());
197  emit this->dataRead(CEntityFlags::AirportEntity, CEntityFlags::ReadFailed, 0, url);
198  return;
199  }
200 
201  // parsing
202  emit this->dataRead(CEntityFlags::AircraftIcaoEntity, CEntityFlags::ReadParsing, 0, url);
203  CAirportList airports;
204  CAirportList inconsistent;
205  if (res.isRestricted())
206  {
207  const CAirportList incrementalAirports(CAirportList::fromDatabaseJson(res, &inconsistent));
208  if (incrementalAirports.isEmpty()) { return; } // currently ignored
209  airports = this->getAirports();
210  airports.replaceOrAddObjectsByKey(incrementalAirports);
211  }
212  else
213  {
214  QElapsedTimer time;
215  time.start();
216  airports = CAirportList::fromDatabaseJson(res, &inconsistent);
217  this->logParseMessage("airports", airports.size(), static_cast<int>(time.elapsed()), res);
218  }
219 
220  if (!inconsistent.isEmpty())
221  {
222  logInconsistentData(
223  CStatusMessage(this, CStatusMessage::SeverityInfo, u"Inconsistent airports: " % inconsistent.dbKeysAsString(", ")),
224  Q_FUNC_INFO);
225  }
226 
227  if (!this->doWorkCheck()) { return; }
228  const int size = airports.size();
229  qint64 latestTimestamp = airports.latestTimestampMsecsSinceEpoch();
230  if (size > 0 && latestTimestamp < 0)
231  {
232  CLogMessage(this).error(u"No timestamp in airport list, setting to last modified value");
233  latestTimestamp = lastModifiedMsSinceEpoch(nwReply.data());
234  }
235 
236  m_airportCache.set(airports, latestTimestamp);
237  this->updateReaderUrl(getBaseUrl(CDbFlags::DbReading));
238 
239  this->emitAndLogDataRead(CEntityFlags::AirportEntity, size, res);
240  }

242  void CAirportDataReader::read(CEntityFlags::Entity entity, CDbFlags::DataRetrievalModeFlag mode, const QDateTime &newerThan)
243  {
244  this->threadAssertCheck();
245  if (!this->doWorkCheck()) { return; }
246  entity &= CEntityFlags::AirportEntity;
247  if (!this->isInternetAccessible())
248  {
249  emit this->dataRead(entity, CEntityFlags::ReadSkipped, 0, {});
250  return;
251  }
252 
253  if (entity.testFlag(CEntityFlags::AirportEntity))
254  {
255  CUrl url = this->getAirportsUrl(mode);
256  if (!url.isEmpty())
257  {
258  url.appendQuery(queryLatestTimestamp(newerThan));
259  this->getFromNetworkAndLog(url, { this, &CAirportDataReader::parseAirportData });
260  emit dataRead(CEntityFlags::AirportEntity, CEntityFlags::ReadStarted, 0, url);
261  }
262  else
263  {
264  this->logNoWorkingUrl(CEntityFlags::AirportEntity);
265  }
266  }
267  }
