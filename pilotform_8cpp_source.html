107  CStatusMessageList CPilotForm::validate(bool nested) const
108  {
109  Q_UNUSED(nested);
110 
111  CStatusMessageList msgs;
112  const CUser user(this->getUser());
113  const bool validId = m_vatsim ? user.hasValidVatsimId() : user.hasId();
114  ui->lblp_Id->setTicked(validId);
115  if (!validId) { msgs.push_back(CStatusMessage(this).validationError(u"Invalid id")); }
116 
117  // check against user and if the data entry contains a valid code (or is empty)
118  const bool validHomeAirport = user.hasValidOrEmptyHomeBase() && (ui->comp_HomeAirport->isEmpty() || ui->comp_HomeAirport->getAirportIcaoCode().hasValidIcaoCode(false));
119  ui->lblp_HomeAirport->setTicked(validHomeAirport);
120  if (!validHomeAirport) { msgs.push_back(CStatusMessage(this).validationError(u"Wromg home airport")); }
121 
122  const bool validCredentials = user.hasCredentials();
123  ui->lblp_Password->setTicked(validCredentials);
124  if (!validCredentials) { msgs.push_back(CStatusMessage(this).validationError(u"Invalid credentials")); }
125 
126  const bool validRealName = user.hasRealName();
127  ui->lblp_RealName->setTicked(validRealName);
128  if (!validRealName) { msgs.push_back(CStatusMessage(this).validationError(u"Invalid real name")); }
129 
130  return msgs;
131  }
