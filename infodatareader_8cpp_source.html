131  void CInfoDataReader::readInfoData()
132  {
133  if (!this->doWorkCheck()) { return; }
134 
135  const CUrl url(this->getInfoObjectsUrl());
136  const CUrlLogList urlLogList(this->getUrlLogList()); // thread safe copy
137 
138  if (urlLogList.hasPending())
139  {
140  CLogMessage(this).info(u"Info data reading still pending, summary: '%1'") << urlLogList.getSummary();
141  return;
142  }
143  if (!url.isEmpty())
144  {
145  this->getFromNetworkAndLog(url, { this, &CInfoDataReader::parseInfoObjectsData });
146  emit this->dataRead(this->getEntityForMode(), CEntityFlags::ReadStarted, 0, url);
147  }
148  else
149  {
150  this->logNoWorkingUrl(this->getEntityForMode());
151  }
152  }

154  void CInfoDataReader::parseInfoObjectsData(QNetworkReply *nwReplyPtr)
155  {
156  // wrap pointer, make sure any exit cleans up reply
157  // required to use delete later as object is created in a different thread
158  QScopedPointer<QNetworkReply, QScopedPointerDeleteLater> nwReply(nwReplyPtr);
159  if (!this->doWorkCheck()) { return; }
160 
161  const CDatabaseReader::JsonDatastoreResponse res = this->setStatusAndTransformReplyIntoDatastoreResponse(nwReply.data());
162  if (res.hasErrorMessage())
163  {
164  CLogMessage::preformatted(res.lastWarningOrAbove());
165  emit this->dataRead(this->getEntityForMode(), CEntityFlags::ReadFailed, 0, res.getUrl());
166  return;
167  }
168 
169  // get all or incremental set
170  const CDbInfoList infoObjects = CDbInfoList::fromDatabaseJson(res.getJsonArray());
171  const int n = infoObjects.size();
172 
173  // Service URL => DB data
174  // DB data directory => shared files
175  const QString urlStr = nwReply->url().toString();
176  Q_UNUSED(urlStr) // debug only
177 
178  // this part needs to be synchronized
179  {
180  QWriteLocker wl(&m_lockInfoObjects);
181  m_infoObjects = infoObjects;
182  }
183 
184  this->emitAndLogDataRead(this->getEntityForMode(), n, res);
185  }
