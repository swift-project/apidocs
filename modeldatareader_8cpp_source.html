180  void CModelDataReader::read(CEntityFlags::Entity entities, CDbFlags::DataRetrievalModeFlag mode, const QDateTime &newerThan)
181  {
182  this->threadAssertCheck();
183  if (!this->doWorkCheck()) { return; }
184  entities &= CEntityFlags::DistributorLiveryModel;
185  if (!this->isInternetAccessible())
186  {
187  emit this->dataRead(entities, CEntityFlags::ReadSkipped, 0, {});
188  return;
189  }
190 
191  CEntityFlags::Entity triggeredRead = CEntityFlags::NoEntity;
192  CUrl url;
193 
194  if (entities.testFlag(CEntityFlags::LiveryEntity))
195  {
196  url = this->getLiveryUrl(mode);
197  if (!url.isEmpty())
198  {
199  url.appendQuery(queryLatestTimestamp(newerThan));
200  this->getFromNetworkAndLog(url, { this, &CModelDataReader::parseLiveryData });
201  triggeredRead |= CEntityFlags::LiveryEntity;
202  }
203  else
204  {
205  this->logNoWorkingUrl(CEntityFlags::LiveryEntity);
206  }
207  }
208 
209  if (entities.testFlag(CEntityFlags::DistributorEntity))
210  {
211  url = this->getDistributorUrl(mode);
212  if (!url.isEmpty())
213  {
214  url.appendQuery(queryLatestTimestamp(newerThan));
215  this->getFromNetworkAndLog(url, { this, &CModelDataReader::parseDistributorData });
216  triggeredRead |= CEntityFlags::DistributorEntity;
217  }
218  else
219  {
220  this->logNoWorkingUrl(CEntityFlags::DistributorEntity);
221  }
222  }
223 
224  if (entities.testFlag(CEntityFlags::ModelEntity))
225  {
226  url = this->getModelUrl(mode);
227  if (!url.isEmpty())
228  {
229  url.appendQuery(queryLatestTimestamp(newerThan));
230  this->getFromNetworkAndLog(url, { this, &CModelDataReader::parseModelData });
231  triggeredRead |= CEntityFlags::ModelEntity;
232  }
233  else
234  {
235  this->logNoWorkingUrl(CEntityFlags::ModelEntity);
236  }
237  }
238 
239  if (triggeredRead != CEntityFlags::NoEntity)
240  {
241  emit dataRead(triggeredRead, CEntityFlags::ReadStarted, 0, url);
242  }
243  }

288  void CModelDataReader::parseLiveryData(QNetworkReply *nwReplyPtr)
289  {
290  // wrap pointer, make sure any exit cleans up reply
291  // required to use delete later as object is created in a different thread
292  QScopedPointer<QNetworkReply, QScopedPointerDeleteLater> nwReply(nwReplyPtr);
293  if (!this->doWorkCheck()) { return; }
294  CDatabaseReader::JsonDatastoreResponse res = this->setStatusAndTransformReplyIntoDatastoreResponse(nwReply.data());
295  if (res.hasErrorMessage())
296  {
297  CLogMessage::preformatted(res.lastWarningOrAbove());
298  emit dataRead(CEntityFlags::LiveryEntity, CEntityFlags::ReadFailed, 0, res.getUrl());
299  return;
300  }
301 
302  // get all or incremental set of distributor
303  emit this->dataRead(CEntityFlags::LiveryEntity, CEntityFlags::ReadParsing, 0, res.getUrl());
304  CLiveryList liveries;
305  if (res.isRestricted())
306  {
307  // create full list if it was just incremental
308  const CLiveryList incrementalLiveries(CLiveryList::fromDatabaseJson(res));
309  if (incrementalLiveries.isEmpty()) { return; } // currenty ignored
310  liveries = this->getLiveries();
311  liveries.replaceOrAddObjectsByKey(incrementalLiveries);
312  }
313  else
314  {
315  QElapsedTimer time;
316  time.start();
317  liveries = CLiveryList::fromDatabaseJson(res);
318  this->logParseMessage("liveries", liveries.size(), static_cast<int>(time.elapsed()), res);
319  }
320 
321  if (!this->doWorkCheck()) { return; }
322  const int n = liveries.size();
323  qint64 latestTimestamp = liveries.latestTimestampMsecsSinceEpoch();
324  if (n > 0 && latestTimestamp < 0)
325  {
326  CLogMessage(this).error(u"No timestamp in livery list, setting to last modified value");
327  latestTimestamp = lastModifiedMsSinceEpoch(nwReply.data());
328  }
329  const CStatusMessage cacheMsg = m_liveryCache.set(liveries, latestTimestamp);
330  CLogMessage::preformatted(cacheMsg);
331 
332  this->updateReaderUrl(getBaseUrl(CDbFlags::DbReading));
333  this->emitAndLogDataRead(CEntityFlags::LiveryEntity, n, res);
334  }

385  void CModelDataReader::parseModelData(QNetworkReply *nwReplyPtr)
386  {
387  // wrap pointer, make sure any exit cleans up reply
388  // required to use delete later as object is created in a different thread
389  QScopedPointer<QNetworkReply, QScopedPointerDeleteLater> nwReply(nwReplyPtr);
390  if (!this->doWorkCheck()) { return; }
391  const CDatabaseReader::JsonDatastoreResponse res = this->setStatusAndTransformReplyIntoDatastoreResponse(nwReply.data());
392  if (res.hasErrorMessage())
393  {
394  CLogMessage::preformatted(res.lastWarningOrAbove());
395  emit this->dataRead(CEntityFlags::ModelEntity, CEntityFlags::ReadFailed, 0, res.getUrl());
396  return;
397  }
398 
399  // get all or incremental set of models
400  emit this->dataRead(CEntityFlags::ModelEntity, CEntityFlags::ReadParsing, 0, res.getUrl());
401 
402  // use prefilled data:
403  // this saves a lot of parsing time as the models do not need to re-parse the sub parts
404  // but can use objects directly
405  const CAircraftCategoryList categories = this->getAircraftCategories();
406  const CLiveryList liveries = this->getLiveries();
407  const CAircraftIcaoCodeList icaos = this->getAircraftAircraftIcaos();
408  const CDistributorList distributors = this->getDistributors();
409 
410  CAircraftModelList models;
411  if (res.isRestricted())
412  {
413  // create full list if it was just incremental
414  const CAircraftModelList incrementalModels(CAircraftModelList::fromDatabaseJsonCaching(res, icaos, categories, liveries, distributors));
415  if (incrementalModels.isEmpty()) { return; } // currently ignored
416  models = this->getModels();
417  models.replaceOrAddObjectsByKey(incrementalModels);
418  }
419  else
420  {
421  QElapsedTimer time;
422  time.start();
423  models = CAircraftModelList::fromDatabaseJsonCaching(res, icaos, categories, liveries, distributors);
424  this->logParseMessage("models", models.size(), static_cast<int>(time.elapsed()), res);
425  }
426 
427  // synchronized update
428  if (!this->doWorkCheck()) { return; }
429  const int n = models.size();
430  qint64 latestTimestamp = models.latestTimestampMsecsSinceEpoch();
431  if (n > 0 && latestTimestamp < 0)
432  {
433  CLogMessage(this).error(u"No timestamp in model list, setting to last modified value");
434  latestTimestamp = lastModifiedMsSinceEpoch(nwReply.data());
435  }
436  const CStatusMessage cacheMsg = m_modelCache.set(models, latestTimestamp);
437  CLogMessage::preformatted(cacheMsg);
438 
439  this->updateReaderUrl(this->getBaseUrl(CDbFlags::DbReading));
440  this->emitAndLogDataRead(CEntityFlags::ModelEntity, n, res);
441  }

443  CStatusMessageList CModelDataReader::readFromJsonFiles(const QString &dir, CEntityFlags::Entity whatToRead, bool overrideNewerOnly)
444  {
445  const QDir directory(dir);
446  if (!directory.exists())
447  {
448  return CStatusMessage(this).error(u"Missing directory '%1'") << dir;
449  }
450 
451  whatToRead &= CEntityFlags::DistributorLiveryModel; // supported
452  CEntityFlags::Entity reallyRead = CEntityFlags::NoEntity;
453 
454  CStatusMessageList msgs;
455  if (whatToRead.testFlag(CEntityFlags::LiveryEntity))
456  {
457  const QString fileName = CFileUtils::appendFilePaths(directory.absolutePath(), "liveries.json");
458  const QFileInfo fi(fileName);
459  if (!fi.exists())
460  {
461  msgs.push_back(CStatusMessage(this).warning(u"File '%1' does not exist") << fileName);
462  }
463  else if (!this->overrideCacheFromFile(overrideNewerOnly, fi, CEntityFlags::LiveryEntity, msgs))
464  {
465  // void
466  }
467  else
468  {
469  const QJsonObject liveriesJson(CDatabaseUtils::readQJsonObjectFromDatabaseFile(fileName));
470  const QUrl url = QUrl::fromLocalFile(fi.absoluteFilePath());
471  if (liveriesJson.isEmpty())
472  {
473  msgs.push_back(CStatusMessage(this).error(u"Failed to read from file/empty file '%1'") << fileName);
474  }
475  else
476  {
477  try
478  {
479  const CLiveryList liveries = CLiveryList::fromMultipleJsonFormats(liveriesJson);
480  const int c = liveries.size();
481  msgs.push_back(m_liveryCache.set(liveries, fi.birthTime().toUTC().toMSecsSinceEpoch()));
482  emit this->dataRead(CEntityFlags::LiveryEntity, CEntityFlags::ReadFinished, c, url);
483  reallyRead |= CEntityFlags::LiveryEntity;
484  }
485  catch (const CJsonException &ex)
486  {
487  emit this->dataRead(CEntityFlags::LiveryEntity, CEntityFlags::ReadFailed, 0, url);
488  msgs.push_back(CStatusMessage::fromJsonException(ex, this, QStringLiteral("Reading liveries from '%1'").arg(fileName)));
489  }
490  }
491  }
492  }
493 
494  if (whatToRead.testFlag(CEntityFlags::ModelEntity))
495  {
496  const QString fileName = CFileUtils::appendFilePaths(directory.absolutePath(), "models.json");
497  const QFileInfo fi(fileName);
498  if (!fi.exists())
499  {
500  msgs.push_back(CStatusMessage(this).warning(u"File '%1' does not exist") << fileName);
501  }
502  else if (!this->overrideCacheFromFile(overrideNewerOnly, fi, CEntityFlags::ModelEntity, msgs))
503  {
504  // void
505  }
506  else
507  {
508  const QJsonObject modelsJson(CDatabaseUtils::readQJsonObjectFromDatabaseFile(fileName));
509  const QUrl url = QUrl::fromLocalFile(fi.absoluteFilePath());
510 
511  if (modelsJson.isEmpty())
512  {
513  msgs.push_back(CStatusMessage(this).error(u"Failed to read from file/empty file '%1'") << fileName);
514  }
515  else
516  {
517  try
518  {
519  const CAircraftModelList models = CAircraftModelList::fromMultipleJsonFormats(modelsJson);
520  const int c = models.size();
521  msgs.push_back(m_modelCache.set(models, fi.birthTime().toUTC().toMSecsSinceEpoch()));
522  emit this->dataRead(CEntityFlags::ModelEntity, CEntityFlags::ReadFinished, c, url);
523  reallyRead |= CEntityFlags::ModelEntity;
524  }
525  catch (const CJsonException &ex)
526  {
527  emit this->dataRead(CEntityFlags::ModelEntity, CEntityFlags::ReadFailed, 0, url);
528  msgs.push_back(CStatusMessage::fromJsonException(ex, this, QStringLiteral("Reading models from '%1'").arg(fileName)));
529  }
530  }
531  }
532  }
533 
534  if (whatToRead.testFlag(CEntityFlags::DistributorEntity))
535  {
536  const QString fileName = CFileUtils::appendFilePaths(directory.absolutePath(), "distributors.json");
537  const QFileInfo fi(fileName);
538  if (!fi.exists())
539  {
540  msgs.push_back(CStatusMessage(this).warning(u"File '%1' does not exist") << fileName);
541  }
542  else if (!this->overrideCacheFromFile(overrideNewerOnly, fi, CEntityFlags::DistributorEntity, msgs))
543  {
544  // void
545  }
546  else
547  {
548  const QJsonObject distributorsJson(CDatabaseUtils::readQJsonObjectFromDatabaseFile(fileName));
549  const QUrl url = QUrl::fromLocalFile(fi.absoluteFilePath());
550 
551  if (distributorsJson.isEmpty())
552  {
553  msgs.push_back(CStatusMessage(this).error(u"Failed to read from file/empty file '%1'") << fileName);
554  }
555  else
556  {
557  try
558  {
559  const CDistributorList distributors = CDistributorList::fromMultipleJsonFormats(distributorsJson);
560  const int c = distributors.size();
561  msgs.push_back(m_distributorCache.set(distributors, fi.birthTime().toUTC().toMSecsSinceEpoch()));
562  emit this->dataRead(CEntityFlags::DistributorEntity, CEntityFlags::ReadFinished, c, url);
563  reallyRead |= CEntityFlags::DistributorEntity;
564  }
565  catch (const CJsonException &ex)
566  {
567  emit this->dataRead(CEntityFlags::DistributorEntity, CEntityFlags::ReadFailed, 0, url);
568  msgs.push_back(CStatusMessage::fromJsonException(ex, this, QStringLiteral("Reading distributors from '%1'").arg(fileName)));
569  }
570  }
571  }
572  }
573 
574  return msgs;
575  }

577  bool CModelDataReader::readFromJsonFilesInBackground(const QString &dir, CEntityFlags::Entity whatToRead, bool overrideNewerOnly)
578  {
579  if (dir.isEmpty() || whatToRead == CEntityFlags::NoEntity) { return false; }
580 
581  QPointer<CModelDataReader> myself(this);
582  QTimer::singleShot(0, this, [=]() {
583  if (!myself) { return; }
584  const CStatusMessageList msgs = this->readFromJsonFiles(dir, whatToRead, overrideNewerOnly);
585  if (msgs.isFailure())
586  {
587  CLogMessage::preformatted(msgs);
588  }
589  });
590  return true;
591  }

593  bool CModelDataReader::writeToJsonFiles(const QString &dir)
594  {
595  QDir directory(dir);
596  if (!directory.exists()) { return false; }
597  QList<QPair<QString, QString>> fileContents;
598  if (this->getLiveriesCount() > 0)
599  {
600  const QString json(QJsonDocument(this->getLiveries().toJson()).toJson());
601  fileContents.push_back({ "liveries.json", json });
602  }
603 
604  if (this->getModelsCount() > 0)
605  {
606  const QString json(QJsonDocument(this->getModels().toJson()).toJson());
607  fileContents.push_back({ "models.json", json });
608  }
609 
610  if (this->getDistributorsCount() > 0)
611  {
612  const QString json(QJsonDocument(this->getDistributors().toJson()).toJson());
613  fileContents.push_back({ "distributors.json", json });
614  }
615 
616  for (const auto &pair : fileContents)
617  {
618  CWorker::fromTask(this, Q_FUNC_INFO, [pair, directory] {
619  CFileUtils::writeStringToFile(CFileUtils::appendFilePaths(directory.absolutePath(), pair.first), pair.second);
620  });
621  }
622  return true;
623  }
