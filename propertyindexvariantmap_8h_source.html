148  template <class Derived>
149  CPropertyIndexList Index<Derived>::apply(const BlackMisc::CPropertyIndexVariantMap &indexMap, bool skipEqualValues)
150  {
151  if (indexMap.isEmpty()) return {};
152 
153  CPropertyIndexList changed;
154  const auto &map = indexMap.map();
155  for (auto it = map.begin(); it != map.end(); ++it)
156  {
157  const CVariant value = it.value();
158  const CPropertyIndex index = it.key();
159  if (skipEqualValues)
160  {
161  const bool equal = derived()->equalsPropertyByIndex(value, index);
162  if (equal) { continue; }
163  }
164  derived()->setPropertyByIndex(index, value);
165  changed.push_back(index);
166  }
167  return changed;
168  }
169  }
