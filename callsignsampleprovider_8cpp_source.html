51  m_simpleCompressorEffect->setMakeUpGain(-5.5);
52

54  m_voiceEqualizer = new CEqualizerSampleProvider(m_simpleCompressorEffect, EqualizerPresets::VHFEmulation, m_mixer);
55 
56  m_mixer->addMixerInput(m_whiteNoise);
57  m_mixer->addMixerInput(m_acBusNoise);
58  m_mixer->addMixerInput(m_hfWhiteNoise);
59  m_mixer->addMixerInput(m_voiceEqualizer);

61  m_timer = new QTimer(this);
62  m_timer->setObjectName(this->objectName() + ":m_timer");
63 
64  m_timer->setInterval(100);

68  int CCallsignSampleProvider::readSamples(QVector<float> &samples, qint64 count)
69  {
70  const int noOfSamples = m_mixer->readSamples(samples, count);
71 
72  if (m_inUse && m_lastPacketLatch && m_audioInput->getBufferedBytes() == 0)
73  {
74  idle();
75  m_lastPacketLatch = false;
76  }
77 
78  if (m_inUse && !m_underflow && m_audioInput->getBufferedBytes() == 0)
79  {
80  if (verbose()) { CLogMessage(this).debug(u"[%1] [Delay++]") << m_callsign; }
81  CallsignDelayCache::instance().underflow(m_callsign);
82  m_underflow = true;
83  }
84 
85  return noOfSamples;
86  }

96  void CCallsignSampleProvider::active(const QString &callsign, const QString &aircraftType)
97  {
98  m_callsign = callsign;
99  CallsignDelayCache::instance().initialise(callsign);
100  m_aircraftType = aircraftType;
101  m_decoder.resetState();
102  m_inUse = true;
103  setEffects();
104  m_underflow = false;
105 
106  const int delayMs = CallsignDelayCache::instance().get(callsign);
107  if (verbose()) { CLogMessage(this).debug(u"[%1] [Delay %2ms]") << m_callsign << delayMs; }
108  if (delayMs > 0)
109  {
110  const int phaseDelayLength = (m_audioFormat.sampleRate() / 1000) * delayMs;
111  const QVector<float> phaseDelay(phaseDelayLength * 2, 0);
112  m_audioInput->addSamples(phaseDelay);
113  }
114  }

174  void CCallsignSampleProvider::setEffects(bool noEffects)
175  {
176  if (noEffects || m_bypassEffects || !m_inUse)
177  {
178  m_crackleSoundProvider->setGain(0.0);
179  m_whiteNoise->setGain(0.0);
180  m_hfWhiteNoise->setGain(0.0);
181  m_acBusNoise->setGain(0.0);
182  m_simpleCompressorEffect->setEnabled(false);
183  m_voiceEqualizer->setBypassEffects(true);
184  }
185  else
186  {
187  if (m_receiver->getFrequencyHz() < 30000000)
188  {
195  m_hfWhiteNoise->setGain(m_hfWhiteNoiseGainMin);
196  m_acBusNoise->setGain(m_acBusGainMin + 0.001f);
197  m_simpleCompressorEffect->setEnabled(true);
198  m_voiceEqualizer->setBypassEffects(false);
199  m_voiceEqualizer->setOutputGain(0.38);
200  m_whiteNoise->setGain(0.0);
201  }
202  else
203  {
204  double crackleFactor = (((qExp(m_distanceRatio) * qPow(m_distanceRatio, -4.0)) / 350.0) - 0.00776652);
205 
206  if (crackleFactor < 0.0) { crackleFactor = 0.0; }
207  if (crackleFactor > 0.20) { crackleFactor = 0.20; }
208 
209  m_crackleSoundProvider->setGain(crackleFactor * 2);
210  m_whiteNoise->setGain(m_whiteNoiseGainMin);
211  m_acBusNoise->setGain(m_acBusGainMin);
212  m_simpleCompressorEffect->setEnabled(true);
213  m_voiceEqualizer->setBypassEffects(false);
214  m_voiceEqualizer->setOutputGain(1.0 - crackleFactor * 3.7);
215  }
216  }
217  }
