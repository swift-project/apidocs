113  CLength CModelMappingForm::getCGFromUI() const
114  {
115  if (ui->le_CG->text().isEmpty()) { return CLength::null(); }
116  const QString v = ui->le_CG->text();
117 
118  // without unit we assume ft
119  if (isDigitsOnlyString(v))
120  {
121  bool ok;
122  const double cgv = v.toDouble(&ok);
123  if (!ok) { return CLength::null(); }
124  return CLength(cgv, CLengthUnit::ft());
125  }
126 
127  CLength cg;
128  cg.parseFromString(v, CPqString::SeparatorBestGuess);
129  return cg;
130  }
