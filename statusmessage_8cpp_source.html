23  QString arg(QStringView format, const QStringList &args)
24  {
25  if (format.isEmpty())
26  {
27  return args.join(u' ');
28  }
29 
30  QString &temp = t_tempBuffer.localData();
31  temp.resize(0); // unlike clear(), resize(0) doesn't release the capacity if there are no implicitly shared copies
32 
33  quint64 unusedArgs = (1ULL << std::min(63, args.size())) - 1;
34  for (auto it = format.begin();;)
35  {
36  const auto pc = std::find(it, format.end(), u'%');
37  temp.append(&*it, std::distance(it, pc));
38  if ((it = pc) == format.end()) { break; }
39  if (++it == format.end())
40  {
41  temp += u'%';
42  break;
43  }
44 
45  if (*it == u'%')
46  {
47  temp += u'%';
48  ++it;
49  continue;
50  }
51  if (is09(*it))
52  {
53  int n = it->unicode() - u'0';
54  BLACK_VERIFY(n >= 0 && n <= 9);
55  if (++it != format.end() && is09(*it))
56  {
57  n = n * 10 + it->unicode() - u'0';
58  ++it;
59  }
60  BLACK_VERIFY(n > 0 && n <= 99);
61  if (n > 0 && n <= args.size())
62  {
63  temp += args[n - 1];
64  unusedArgs &= ~(1ULL << (n - 1));
65  }
66  else { temp += u'%' % QString::number(n); }
67  }
68  else { temp += u'%'; }
69  }
70  if (unusedArgs) { temp += QStringLiteral(" [SOME MESSAGE ARGUMENT(S) UNUSED]"); }
71 
72  QString result = temp;
73  result.squeeze(); // release unused capacity and implicitly detach so temp keeps its capacity for next time
74  return result;
75  }
76  }

327  CStatusMessage::StatusSeverity CStatusMessage::stringToSeverity(const QString &severity)
328  {
329  // pre-check
330  QString severityString(severity.trimmed().toLower());
331  if (severityString.isEmpty()) { return SeverityInfo; }
332 
333  // hard check
334  if (severityString.compare(severityToString(SeverityDebug), Qt::CaseInsensitive) == 0) { return SeverityDebug; }
335  if (severityString.compare(severityToString(SeverityInfo), Qt::CaseInsensitive) == 0) { return SeverityInfo; }
336  if (severityString.compare(severityToString(SeverityWarning), Qt::CaseInsensitive) == 0) { return SeverityWarning; }
337  if (severityString.compare(severityToString(SeverityError), Qt::CaseInsensitive) == 0) { return SeverityError; }
338 
339  // not found yet, lenient checks
340  QChar s = severityString.at(0);
341  if (s == 'd') { return SeverityDebug; }
342  if (s == 'i') { return SeverityInfo; }
343  if (s == 'w') { return SeverityWarning; }
344  if (s == 'e') { return SeverityError; }
345 
346  return SeverityInfo;
347  }

382  QString CStatusMessage::severitiesToString(const QSet<CStatusMessage::StatusSeverity> &severities)
383  {
384  if (severities.isEmpty()) { return {}; }
385  auto minmax = std::minmax_element(severities.begin(), severities.end());
386  auto min = *minmax.first;
387  auto max = *minmax.second;
388  if (min == SeverityDebug && max == SeverityError)
389  {
390  static const QString all("all severities");
391  return all;
392  }
393  if (min == SeverityDebug) { return u"at or below " % severityToString(max); }
394  if (max == SeverityError) { return u"at or above " % severityToString(min); }
395  auto list = severities.values();
396  std::sort(list.begin(), list.end());
397  QStringList ret;
398  std::transform(list.cbegin(), list.cend(), std::back_inserter(ret), severityToString);
399  return ret.join("|");
400  }
