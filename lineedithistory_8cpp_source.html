26  void CLineEditHistory::keyPressEvent(QKeyEvent *event)
27  {
28  const int key = event->key();
29  bool nonEmpty = false;
30 
31  if (key == Qt::Key_Up)
32  {
33  // move back in history
34  if (m_history.size() > m_position)
35  {
36  this->setText(m_history.at(m_position++));
37  }
38  }
39  else if (key == Qt::Key_Down)
40  {
41  // move forward in history
42  if (m_position <= 0)
43  {
44  this->clear();
45  return;
46  }
47  if (m_position == m_history.size()) { --m_position; } // avoid need of 2xKeyDown at end
48  if (m_position > 0 && m_history.size() > --m_position)
49  {
50  this->setText(m_history.at(m_position));
51  }
52  }
53  else if (key == Qt::Key_Return || key == Qt::Key_Enter) // normal and keypad enter
54  {
55  const QString t = this->text().trimmed();
56  if (!t.isEmpty())
57  {
58  m_history.push_front(t);
59  m_position = 0;
60  this->clear();
61  nonEmpty = true;
62  }
63  }
64  // default handler for event
65  QLineEdit::keyPressEvent(event);
66 
67  // signal
68  if (nonEmpty) { emit this->returnPressedUnemptyLine(); }
69  }
