22  // Like invoke() but ignores the first argument if callable is not a member function. For uniform calling of callables with slot semantics.
23  template <typename F, typename T, typename U, size_t... Is>
24  decltype(auto) invokeSlotImpl(F ptr, T *object, U tuple, std::index_sequence<Is...>, std::true_type)
25  {
26  Q_UNUSED(tuple); // in case the pack expansion is empty
27  return (object->*ptr)(std::forward<std::tuple_element_t<Is, U>>(std::get<Is>(tuple))...);
28  }
29  template <typename F, typename T, typename U, size_t... Is>
30  decltype(auto) invokeSlotImpl(F &&func, T *, U tuple, std::index_sequence<Is...>, std::false_type)
31  {
32  Q_UNUSED(tuple); // in case the pack expansion is empty
33  return std::forward<F>(func)(std::forward<std::tuple_element_t<Is, U>>(std::get<Is>(tuple))...);
34  }
35  template <typename F, typename T, typename... Ts>
36  decltype(auto) invokeSlot(F &&func, T *object, Ts &&...args)
37  {
38  using seq = MaskSequence<std::make_index_sequence<sizeof...(Ts)>, !TIsQPrivateSignal<std::decay_t<Ts>>::value...>;
39  return invokeSlotImpl(std::forward<F>(func), object, std::forward_as_tuple(std::forward<Ts>(args)...), seq(), std::is_member_pointer<std::decay_t<F>>());
40  }

42  // Like QMetaObject::invokeMethod but the return value is accessed through a QFuture, and extra arguments can be provided.
43  template <typename T, typename F, typename... Ts>
44  auto invokeMethod(T *object, F &&func, Ts &&...args)
45  {
46  const auto invoker = [](auto &&...x) { return Private::invokeSlot(std::forward<decltype(x)>(x)...); };
47  auto method = std::bind(invoker, std::forward<F>(func), object, std::forward<Ts>(args)...);
48  CPromise<decltype(std::move(method)())> promise;
49  QMetaObject::invokeMethod(object, [promise, method = std::move(method)]() mutable { promise.setResultFrom(std::move(method)); });
50  return promise.future();
51  }
