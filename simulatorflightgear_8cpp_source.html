192  {
193  if (!this->isShuttingDownOrDisconnected())
194  {
195  m_serviceProxy->getOwnAircraftSituationData(&m_flightgearData);
196  m_serviceProxy->getOwnAircraftVelocityData(&m_flightgearData);
197  m_serviceProxy->isPausedAsync(&m_simulatorPaused);
198  m_serviceProxy->getCom1ActiveKhzAsync(&m_flightgearData.com1ActiveKhz);
199  m_serviceProxy->getCom1StandbyKhzAsync(&m_flightgearData.com1StandbyKhz);
200  m_serviceProxy->getCom2ActiveKhzAsync(&m_flightgearData.com2ActiveKhz);
201  m_serviceProxy->getCom2StandbyKhzAsync(&m_flightgearData.com2StandbyKhz);
202  m_serviceProxy->getTransponderCodeAsync(&m_flightgearData.xpdrCode);
203  m_serviceProxy->getTransponderModeAsync(&m_flightgearData.xpdrMode);
204  m_serviceProxy->getTransponderIdentAsync(&m_flightgearData.xpdrIdent);
205  m_serviceProxy->getAllWheelsOnGroundAsync(&m_flightgearData.onGroundAll);
206  m_serviceProxy->getGroundElevationAsync(&m_flightgearData.groundElevation);
207  m_serviceProxy->getCom1VolumeAsync(&m_flightgearData.volumeCom1);
208  m_serviceProxy->getCom2VolumeAsync(&m_flightgearData.volumeCom2);
209 
210  CAircraftSituation situation;
211  situation.setPosition({ m_flightgearData.latitudeDeg, m_flightgearData.longitudeDeg, 0 });
212  situation.setAltitude({ m_flightgearData.altitudeFt, CAltitude::MeanSeaLevel, CLengthUnit::ft() });
213  situation.setPressureAltitude({ m_flightgearData.pressureAltitudeFt, CAltitude::MeanSeaLevel, CAltitude::PressureAltitude, CLengthUnit::ft() });
214  situation.setHeading({ m_flightgearData.trueHeadingDeg, CHeading::True, CAngleUnit::deg() });
215  situation.setPitch({ m_flightgearData.pitchDeg, CAngleUnit::deg() });
216  situation.setBank({ m_flightgearData.rollDeg, CAngleUnit::deg() });
217  situation.setGroundSpeed({ m_flightgearData.groundspeedKts, CSpeedUnit::kts() });
218  situation.setGroundElevation(CAltitude(m_flightgearData.groundElevation, CAltitude::MeanSeaLevel, CLengthUnit::m()), CAircraftSituation::FromProvider);
219 
220  if (!m_simulatorPaused)
221  {
222  situation.setVelocity({ m_flightgearData.velocityXMs, m_flightgearData.velocityYMs, m_flightgearData.velocityZMs,
223  CSpeedUnit::m_s(), m_flightgearData.pitchRateRadPerSec, m_flightgearData.rollRateRadPerSec, m_flightgearData.yawRateRadPerSec,
224  CAngleUnit::rad(), CTimeUnit::s() });
225  }
226  else
227  {
228  situation.setVelocity({ 0.0, 0.0, 0.0, CSpeedUnit::m_s(), 0.0, 0.0, 0.0, CAngleUnit::rad(), CTimeUnit::s() });
229  }
230 
231  // Updates
232  // Do not update ICAO codes, as this overrides reverse lookups
233  // updateOwnIcaoCodes(m_flightgearData.aircraftIcaoCode, CAirlineIcaoCode());
234  this->updateOwnSituationAndGroundElevation(situation);
235 
236  // defaults
237  CSimulatedAircraft myAircraft(getOwnAircraft());
238  CComSystem com1(myAircraft.getCom1System()); // set defaults
239  CComSystem com2(myAircraft.getCom2System());
240  CTransponder transponder(myAircraft.getTransponder());
241 
242  // updates
243  com1.setFrequencyActive(CFrequency(m_flightgearData.com1ActiveKhz, CFrequencyUnit::kHz()));
244  com1.setFrequencyStandby(CFrequency(m_flightgearData.com1StandbyKhz, CFrequencyUnit::kHz()));
245  com1.setVolumeReceive(qRound(m_flightgearData.volumeCom1 * 100));
246  const bool changedCom1 = myAircraft.getCom1System() != com1;
247 
248  com2.setFrequencyActive(CFrequency(m_flightgearData.com2ActiveKhz, CFrequencyUnit::kHz()));
249  com2.setFrequencyStandby(CFrequency(m_flightgearData.com2StandbyKhz, CFrequencyUnit::kHz()));
250  com2.setVolumeReceive(qRound(m_flightgearData.volumeCom2 * 100));
251  const bool changedCom2 = myAircraft.getCom2System() != com2;
252 
253  transponder = CTransponder::getStandardTransponder(m_flightgearData.xpdrCode, xpdrMode(m_flightgearData.xpdrMode, m_flightgearData.xpdrIdent));
254  const bool changedXpr = (myAircraft.getTransponder() != transponder);
255 
256  if (changedCom1 || changedCom2 || changedXpr)
257  {
258  this->updateCockpit(com1, com2, transponder, identifier());
259  }
260  }

263  void CSimulatorFlightgear::slowTimerTimeout()
264  {
265  if (!this->isShuttingDownOrDisconnected())
266  {
267  m_serviceProxy->getAircraftModelPathAsync(&m_flightgearData.aircraftModelPath); // this is NOT the model string
268  m_serviceProxy->getAircraftIcaoCodeAsync(&m_flightgearData.aircraftIcaoCode);
269  m_serviceProxy->getBeaconLightsOnAsync(&m_flightgearData.beaconLightsOn);
270  m_serviceProxy->getLandingLightsOnAsync(&m_flightgearData.landingLightsOn);
271  m_serviceProxy->getNavLightsOnAsync(&m_flightgearData.navLightsOn);
272  m_serviceProxy->getStrobeLightsOnAsync(&m_flightgearData.strobeLightsOn);
273  m_serviceProxy->getTaxiLightsOnAsync(&m_flightgearData.taxiLightsOn);
274  m_serviceProxy->getFlapsDeployRatioAsync(&m_flightgearData.flapsReployRatio);
275  m_serviceProxy->getGearDeployRatioAsync(&m_flightgearData.gearReployRatio);
276  m_serviceProxy->getEngineN1PercentageAsync(&m_flightgearData.enginesN1Percentage);
277  m_serviceProxy->getSpeedBrakeRatioAsync(&m_flightgearData.speedBrakeRatio);
278 
279  CAircraftEngineList engines;
280  for (int engineNumber = 0; engineNumber < m_flightgearData.enginesN1Percentage.size(); ++engineNumber)
281  {
282  // Engine number start counting at 1
283  // We consider the engine running when N1 is bigger than 5 %
284  CAircraftEngine engine { engineNumber + 1, m_flightgearData.enginesN1Percentage.at(engineNumber) > 5.0 };
285  engines.push_back(engine);
286  }
287 
288  CAircraftParts parts { { m_flightgearData.strobeLightsOn, m_flightgearData.landingLightsOn, m_flightgearData.taxiLightsOn,
289  m_flightgearData.beaconLightsOn, m_flightgearData.navLightsOn, false },
290  m_flightgearData.gearReployRatio > 0,
291  static_cast<int>(m_flightgearData.flapsReployRatio * 100),
292  m_flightgearData.speedBrakeRatio > 0.5,
293  engines,
294  m_flightgearData.onGroundAll };
295 
296  this->updateOwnParts(parts);
297  this->requestRemoteAircraftDataFromFlightgear();
298 
299  CCallsignSet invalid;
300  for (CFlightgearMPAircraft &flightgearAircraft : m_flightgearAircraftObjects)
301  {
302  // Update remote aircraft to have the latest transponder modes, codes etc.
303  const CCallsign cs = flightgearAircraft.getCallsign();
304  const CSimulatedAircraft simulatedAircraft = this->getAircraftInRangeForCallsign(cs);
305  if (!simulatedAircraft.hasCallsign())
306  {
307  if (!cs.isEmpty()) { invalid.insert(cs); }
308  continue;
309  }
310  flightgearAircraft.setSimulatedAircraft(simulatedAircraft);
311  }
312 
313  int i = 0;
314 
315  for (const CCallsign &cs : invalid)
316  {
317  this->triggerRemoveAircraft(cs, ++i * 100);
318  }
319  }
320  }

327  bool CSimulatorFlightgear::connectTo()
328  {
329  if (isConnected()) { return true; }
330  QString dbusAddress = m_fgswiftbusServerSetting.getThreadLocal().getDBusServerAddress();
331 
332  if (CDBusServer::isSessionOrSystemAddress(dbusAddress))
333  {
334  m_dBusConnection = QDBusConnection::sessionBus();
335  m_dbusMode = Session;
336  }
337  else if (CDBusServer::isQtDBusAddress(dbusAddress))
338  {
339  m_dBusConnection = QDBusConnection::connectToPeer(dbusAddress, "fgswiftbus");
340  if (!m_dBusConnection.isConnected()) { return false; }
341  m_dbusMode = P2P;
342  }
343 
344  m_serviceProxy = new CFGSwiftBusServiceProxy(m_dBusConnection, this);
345  m_trafficProxy = new CFGSwiftBusTrafficProxy(m_dBusConnection, this);
346 
347  const bool s = m_dBusConnection.connect(QString(), DBUS_PATH_LOCAL, DBUS_INTERFACE_LOCAL,
348  "Disconnected", this, SLOT(onDBusServiceUnregistered()));
349  Q_ASSERT(s);
350  if (!m_serviceProxy->isValid() || !m_trafficProxy->isValid())
351  {
352  this->disconnectFrom();
353  return false;
354  }
355 
356  emitOwnAircraftModelChanged(m_serviceProxy->getAircraftModelPath(), m_serviceProxy->getAircraftModelFilename(), m_serviceProxy->getAircraftLivery(),
357  m_serviceProxy->getAircraftIcaoCode(), m_serviceProxy->getAircraftModelString(), m_serviceProxy->getAircraftName(), m_serviceProxy->getAircraftDescription());
358  setSimulatorDetails("Flightgear", {}, "");
359  connect(m_serviceProxy, &CFGSwiftBusServiceProxy::aircraftModelChanged, this, &CSimulatorFlightgear::emitOwnAircraftModelChanged);
360  connect(m_serviceProxy, &CFGSwiftBusServiceProxy::airportsInRangeUpdated, this, &CSimulatorFlightgear::setAirportsInRange);
361  connect(m_trafficProxy, &CFGSwiftBusTrafficProxy::simFrame, this, &CSimulatorFlightgear::updateRemoteAircraft);
362  connect(m_trafficProxy, &CFGSwiftBusTrafficProxy::remoteAircraftAdded, this, &CSimulatorFlightgear::onRemoteAircraftAdded);
363  connect(m_trafficProxy, &CFGSwiftBusTrafficProxy::remoteAircraftAddingFailed, this, &CSimulatorFlightgear::onRemoteAircraftAddingFailed);
364  if (m_watcher) { m_watcher->setConnection(m_dBusConnection); }
365  m_trafficProxy->removeAllPlanes();
366  this->emitSimulatorCombinedStatus();
367 
368  this->initSimulatorInternals();
369  return true;
370  }

468  bool CSimulatorFlightgear::updateOwnSimulatorCockpit(const Simulation::CSimulatedAircraft &aircraft, const CIdentifier &originator)
469  {
470  if (originator == this->identifier()) { return false; }
471  if (this->isShuttingDownOrDisconnected()) { return false; }
472 
473  auto com1 = CComSystem::getCom1System({ m_flightgearData.com1ActiveKhz, CFrequencyUnit::kHz() }, { m_flightgearData.com1StandbyKhz, CFrequencyUnit::kHz() });
474  auto com2 = CComSystem::getCom2System({ m_flightgearData.com2ActiveKhz, CFrequencyUnit::kHz() }, { m_flightgearData.com2StandbyKhz, CFrequencyUnit::kHz() });
475  auto xpdr = CTransponder::getStandardTransponder(m_flightgearData.xpdrCode, xpdrMode(m_flightgearData.xpdrMode, m_flightgearData.xpdrIdent));
476  if (aircraft.hasChangedCockpitData(com1, com2, xpdr))
477  {
478  m_flightgearData.com1ActiveKhz = aircraft.getCom1System().getFrequencyActive().valueInteger(CFrequencyUnit::kHz());
479  m_flightgearData.com1StandbyKhz = aircraft.getCom1System().getFrequencyStandby().valueInteger(CFrequencyUnit::kHz());
480  m_flightgearData.com2ActiveKhz = aircraft.getCom2System().getFrequencyActive().valueInteger(CFrequencyUnit::kHz());
481  m_flightgearData.com2StandbyKhz = aircraft.getCom2System().getFrequencyStandby().valueInteger(CFrequencyUnit::kHz());
482  m_flightgearData.xpdrCode = aircraft.getTransponderCode();
483  m_flightgearData.xpdrMode = xpdrMode(aircraft.getTransponderMode());
484  m_serviceProxy->setCom1ActiveKhz(m_flightgearData.com1ActiveKhz);
485  m_serviceProxy->setCom1StandbyKhz(m_flightgearData.com1StandbyKhz);
486  m_serviceProxy->setCom2ActiveKhz(m_flightgearData.com2ActiveKhz);

661  if (!callsignsInRange.contains(callsign)) { continue; }

689

694  {
695  this->rememberLastSent(parts, callsign);
696  planesSurfaces.push_back(flightgearAircraft.getCallsign(), parts);
697  }
698  }
699 
700  } // all callsigns
701 
702  if (!planesTransponders.isEmpty() && Flightgear::FGSWIFTBUS_API_VERSION >= 2)
703  {
704  m_trafficProxy->setPlanesTransponders(planesTransponders);
705  }
706

709  if (CBuildConfig::isLocalDeveloperDebugBuild())
710  {

712  }
713  m_trafficProxy->setPlanesPositions(planesPositions);
714  }
715 
716  if (!planesSurfaces.isEmpty() && Flightgear::FGSWIFTBUS_API_VERSION >= 2)
717  {
718  m_trafficProxy->setPlanesSurfaces(planesSurfaces);

760  void CSimulatorFlightgear::updateRemoteAircraftFromSimulator(
761  const QStringList &callsigns, const QDoubleList &latitudesDeg, const QDoubleList &longitudesDeg,
762  const QDoubleList &elevationsMeters, const QDoubleList &verticalOffsetsMeters)
763  {
764  const int size = callsigns.size();
765 
766  // we skip if we are not near ground
767  if (CBuildConfig::isLocalDeveloperDebugBuild())
768  {
769  Q_ASSERT_X(elevationsMeters.size() == size, Q_FUNC_INFO, "Wrong elevations");
770  Q_ASSERT_X(latitudesDeg.size() == size, Q_FUNC_INFO, "Wrong latitudesDeg");
771  Q_ASSERT_X(longitudesDeg.size() == size, Q_FUNC_INFO, "Wrong longitudesDeg");
772  Q_ASSERT_X(verticalOffsetsMeters.size() == size, Q_FUNC_INFO, "Wrong CG");
773  }
774 
775  const CCallsignSet logCallsigns = this->getLogCallsigns();
776  for (int i = 0; i < size; i++)
777  {
778  const bool emptyCs = callsigns[i].isEmpty();
779  BLACK_VERIFY_X(!emptyCs, Q_FUNC_INFO, "Need callsign");
780  if (emptyCs) { continue; }
781  const CCallsign cs(callsigns[i]);
782  if (!m_flightgearAircraftObjects.contains(cs)) { continue; }
783  const CFlightgearMPAircraft fgAircraft = m_flightgearAircraftObjects[cs];
784  BLACK_VERIFY_X(fgAircraft.hasCallsign(), Q_FUNC_INFO, "Need callsign");
785  if (!fgAircraft.hasCallsign()) { continue; }
786 
787  CElevationPlane elevation = CElevationPlane::null();
788  if (!std::isnan(elevationsMeters[i]))
789  {
790  const CAltitude elevationAlt = CAltitude(elevationsMeters[i], CLengthUnit::m(), CLengthUnit::ft());
791  elevation = CElevationPlane(CLatitude(latitudesDeg[i], CAngleUnit::deg()), CLongitude(longitudesDeg[i], CAngleUnit::deg()), elevationAlt, CElevationPlane::singlePointRadius());
792  }
793 
794  const double cgValue = verticalOffsetsMeters[i]; // XP offset is swift CG
795  const CLength cg = std::isnan(cgValue) ?
796  CLength::null() :
797  CLength(cgValue, CLengthUnit::m(), CLengthUnit::ft());
798 
799  // if we knew "on ground" here we could set it as parameter of rememberElevationAndSimulatorCG
800  this->rememberElevationAndSimulatorCG(cs, fgAircraft.getAircraftModel(), false, elevation, cg);
801 
802  // loopback
803  if (logCallsigns.contains(cs))
804  {
805  this->addLoopbackSituation(cs, elevation, cg);
806  }
807  }
