108  void CAirportSmallCompleter::onAirportsChanged()
109  {
110  if (!sGui || !sGui->hasWebDataServices()) { return; }
111  const CAirportList airports = sGui->getWebDataServices()->getAirports();
112  ui->le_Icao->setCompleter(new QCompleter(airports.allIcaoCodes(true), ui->le_Icao));
113 
114  if (ui->le_Icao->completer()->popup())
115  {
116  ui->le_Icao->completer()->popup()->setObjectName("AirportSmallCompleter.ICAO");
117  ui->le_Icao->completer()->popup()->setMinimumWidth(75);
118  ui->le_Icao->completer()->setCaseSensitivity(Qt::CaseInsensitive);
119  }
120 
121  // turn into airport when it was not possible before
122  if (m_current.isNull())
123  {
124  const QString icao = this->getIcaoText();
125  if (CAirportIcaoCode::isValidIcaoDesignator(icao, true))
126  {
127  const CAirport airport = sGui->getWebDataServices()->getAirportForIcaoDesignator(icao);
128  if (!airport.isNull()) { this->setAirport(airport); }
129  }
130  }
131  }
