15  CAngle CInterpolatorPbh::interpolateAngle(const CAngle &begin, const CAngle &end, double timeFraction0to1)
16  {
17  // determine the right direction (to left, to right) we interpolate towards to
18  // -30 -> 30 => 60 (via 0)
19  // 30 -> -30 => -60 (via 0)
20  // 170 -> -170 => -340 (via 180)
21  // -170 -> 170 => 340 (via 180)
22  double deltaDeg = (end - begin).value(CAngleUnit::deg());
23  if (deltaDeg > 180.0) { deltaDeg -= 360; }
24  else if (deltaDeg < -180.0) { deltaDeg += 360; }
25 
26  if (CBuildConfig::isLocalDeveloperDebugBuild())
27  {
28  BLACK_VERIFY_X(isAcceptableTimeFraction(timeFraction0to1), Q_FUNC_INFO, "0..1 fraction needed");
29  }
30 
32  if (timeFraction0to1 >= 1.0) { return begin + CAngle(deltaDeg, CAngleUnit::deg()); }
33  if (timeFraction0to1 <= 0.0) { return begin; }
34  return begin + CAngle(timeFraction0to1 * deltaDeg, CAngleUnit::deg());
35  }

37  CHeading CInterpolatorPbh::getHeading() const
38  {
39  // HINT: VTOL aircraft can change pitch/bank without changing position, planes cannot
40  // Interpolate heading: HDG = (HdgB - HdgA) * t + HdgA
41  const CHeading headingBegin = m_oldSituation.getHeading();
42  const CHeading headingEnd = m_newSituation.getHeading();
43 
44  if (CBuildConfig::isLocalDeveloperDebugBuild())
45  {
46  BLACK_VERIFY_X(headingBegin.getReferenceNorth() == headingEnd.getReferenceNorth(), Q_FUNC_INFO, "Need same reference");
47  }
48  return CHeading(interpolateAngle(headingBegin, headingEnd, m_simulationTimeFraction), headingEnd.getReferenceNorth());
49  }
