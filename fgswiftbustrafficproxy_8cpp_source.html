36  MultiplayerAcquireInfo CFGSwiftBusTrafficProxy::acquireMultiplayerPlanes()
37  {
38  QDBusPendingReply<bool, QString> reply = m_dbusInterface->asyncCall(QLatin1String("acquireMultiplayerPlanes"));
39  reply.waitForFinished();
40  if (reply.isError())
41  {
42  BlackMisc::CLogMessage(this).debug(u"CFGSwiftBusTrafficProxy::acquireMultiplayerPlanes returned: %1") << reply.error().message();
43  }
44  MultiplayerAcquireInfo info;
45  info.hasAcquired = reply.argumentAt<0>();
46  info.owner = reply.argumentAt<1>();
47  return info;
48  }

102  void CFGSwiftBusTrafficProxy::getRemoteAircraftData(const QStringList &callsigns, const RemoteAircraftDataCallback &setter) const
103  {
104  std::function<void(QDBusPendingCallWatcher *)> callback = [=](QDBusPendingCallWatcher *watcher) {
105  QDBusPendingReply<QStringList, QList<double>, QList<double>, QList<double>, QList<double>> reply = *watcher;
106  if (!reply.isError())
107  {
108  const QStringList callsigns = reply.argumentAt<0>();
109  const QList<double> latitudesDeg = reply.argumentAt<1>();
110  const QList<double> longitudesDeg = reply.argumentAt<2>();
111  const QList<double> elevationsM = reply.argumentAt<3>();
112  const QList<double> verticalOffsets = reply.argumentAt<4>();
113  setter(callsigns, latitudesDeg, longitudesDeg, elevationsM, verticalOffsets);
114  }
115  watcher->deleteLater();
116  };
117  m_dbusInterface->callDBusAsync(QLatin1String("getRemoteAircraftData"), callback, callsigns);
118  }

120  void CFGSwiftBusTrafficProxy::getElevationAtPosition(const CCallsign &callsign, double latitudeDeg, double longitudeDeg, double altitudeMeters,
121  const ElevationCallback &setter) const
122  {
123  std::function<void(QDBusPendingCallWatcher *)> callback = [=](QDBusPendingCallWatcher *watcher) {
124  QDBusPendingReply<QString, double> reply = *watcher;
125  if (!reply.isError())
126  {
127  const CCallsign cs(reply.argumentAt<0>());
128  const double elevationMeters = reply.argumentAt<1>();
129  const CAltitude elevationAlt = std::isnan(elevationMeters) ? CAltitude::null() : CAltitude(elevationMeters, CLengthUnit::m(), CLengthUnit::ft());
130  const CElevationPlane elevation(CLatitude(latitudeDeg, CAngleUnit::deg()),
131  CLongitude(longitudeDeg, CAngleUnit::deg()),
132  elevationAlt, CElevationPlane::singlePointRadius());
133  setter(elevation, cs);
134  }
135  watcher->deleteLater();
136  };
137  m_dbusInterface->callDBusAsync(QLatin1String("getElevationAtPosition"), callback, callsign.asString(), latitudeDeg, longitudeDeg, altitudeMeters);
138  }
