181  bool CLogPattern::checkInvariants() const
182  {
183  switch (m_strategy)
184  {
185  case Everything: return m_strings.isEmpty();
186  case ExactMatch: return m_strings.size() == 1;
187  case AnyOf: return m_strings.size() > 1;
188  case AllOf: return m_strings.size() > 1;
189  case StartsWith: return m_strings.size() == 1;
190  case EndsWith: return m_strings.size() == 1;
191  case Contains: return m_strings.size() == 1;
192  case Nothing: return m_strings.isEmpty();
193  default: return false;
194  }
195  }

197  bool CLogPattern::match(const CStatusMessage &message) const
198  {
199  if (!checkInvariants())
200  {
201  Q_ASSERT(false);
202  return true;
203  }
204 
205  if (!m_severities.contains(message.getSeverity()))
206  {
207  return false;
208  }
209 
210  switch (m_strategy)
211  {
212  default:
213  case Everything: return true;
214  case ExactMatch: return message.getCategories().contains(getString());
215  case AnyOf: return std::any_of(m_strings.begin(), m_strings.end(), [&](const QString &s) { return message.getCategories().contains(s); });
216  case AllOf: return std::all_of(m_strings.begin(), m_strings.end(), [&](const QString &s) { return message.getCategories().contains(s); });
217  case StartsWith: return message.getCategories().containsBy([this](const CLogCategory &cat) { return cat.startsWith(getPrefix()); });
218  case EndsWith: return message.getCategories().containsBy([this](const CLogCategory &cat) { return cat.endsWith(getSuffix()); });
219  case Contains: return message.getCategories().containsBy([this](const CLogCategory &cat) { return cat.contains(getSubstring()); });
220  case Nothing: return message.getCategories().isEmpty();
221  }
222  }

224  bool CLogPattern::isProperSubsetOf(const CLogPattern &other) const
225  {
226  if (!(checkInvariants() && other.checkInvariants()))
227  {
228  Q_ASSERT(false);
229  return false;
230  }
231 
232  // For this function to return true, the severities matched by this pattern must be a subset of the severities
233  // matched by the other, and the categories matched by this pattern must be a subset of the categories matched
234  // by the other, and at least one of these two subset relations must be a proper subset relation.
235 
236  if (!other.m_severities.contains(m_severities))
237  {
238  // Severities are not a subset
239  return false;
240  }
241  if (m_strategy == other.m_strategy && m_strings == other.m_strings)
242  {
243  // Severities are a subset, and categories are an improper subset, so it all depends
244  // on whether the subset relation of the severities is proper or improper
245  return m_severities.size() < other.m_severities.size();
246  }
247 
248  // If we got this far then the severity set is a (proper or improper) subset of the other severity set,
249  // and the question of whether this pattern is a proper subset of the other pattern depends upon whether the
250  // set of categories matched by this pattern is a proper subset of the set of categories matched by the other.
251  // The matrix below is a guide to the implementation that follows.
252  //
253  // Matrix of "categories matched by X is proper subset of categories matched by Y" for two strategies X and Y
254  // 0 means always false
255  // 1 means always true
256  // ? means it depends on a particular relation between their string sets
257  //
258  // Y Ev EM An Al SW EW Co No
259  // X
260  // Ev 0 0 0 0 0 0 0 0 (Everything)
261  // EM 1 0 ? 0 ? ? ? 0 (ExactMatch)
262  // An 1 0 ? 0 ? ? ? 0 (AnyOf)
263  // Al 1 ? ? ? ? ? ? 0 (AllOf)
264  // SW 1 0 0 0 ? 0 ? 0 (StartsWith)
265  // EW 1 0 0 0 0 ? ? 0 (EndsWith)
266  // Co 1 0 0 0 0 0 ? 0 (Contains)
267  // No 1 0 0 0 0 0 0 0 (Nothing)
268 
269  if (m_strategy != Everything && other.m_strategy == Everything)
270  {
271  return true;
272  }
273  switch (m_strategy)
274  {
275  case ExactMatch:
276  switch (other.m_strategy)
277  {
278  case AnyOf: return other.m_strings.contains(getString());
279  case StartsWith: return getString().startsWith(other.getPrefix());
280  case EndsWith: return getString().endsWith(other.getSuffix());
281  case Contains: return getString().contains(other.getSubstring());
282  default:;
283  }
284  break;
285  case AnyOf:
286  switch (other.m_strategy)
287  {
288  case AnyOf: return other.m_strings.contains(m_strings) && other.m_strings.size() > m_strings.size();
289  case StartsWith: return std::all_of(m_strings.begin(), m_strings.end(), [&](const QString &s) { return s.startsWith(other.getPrefix()); });
290  case EndsWith: return std::all_of(m_strings.begin(), m_strings.end(), [&](const QString &s) { return s.endsWith(other.getSuffix()); });
291  case Contains: return std::all_of(m_strings.begin(), m_strings.end(), [&](const QString &s) { return s.contains(other.getSubstring()); });
292  default:;
293  }
294  break;
295  case AllOf:
296  switch (other.m_strategy)
297  {
298  case ExactMatch: return m_strings.contains(other.getString());
299  case AnyOf: return !(m_strings & other.m_strings).isEmpty();
300  case AllOf: return m_strings.contains(other.m_strings) && m_strings.size() > other.m_strings.size();
301  case StartsWith: return std::any_of(m_strings.begin(), m_strings.end(), [&](const QString &s) { return s.startsWith(other.getPrefix()); });
302  case EndsWith: return std::any_of(m_strings.begin(), m_strings.end(), [&](const QString &s) { return s.endsWith(other.getSuffix()); });
303  case Contains: return std::any_of(m_strings.begin(), m_strings.end(), [&](const QString &s) { return s.contains(other.getSubstring()); });
304  default:;
305  }
306  break;
307  case StartsWith:
308  switch (other.m_strategy)
309  {
310  case StartsWith: return getPrefix().startsWith(other.getPrefix()) && getPrefix().size() > other.getPrefix().size();
311  case Contains: return getPrefix().contains(other.getSubstring());
312  default:;
313  }
314  break;
315  case EndsWith:
316  switch (other.m_strategy)
317  {
318  case EndsWith: return getSuffix().endsWith(other.getSuffix()) && getSuffix().size() > other.getSuffix().size();
319  case Contains: return getSuffix().contains(other.getSubstring());
320  default:;
321  }
322  break;
323  case Contains:
324  switch (other.m_strategy)
325  {
326  case Contains: return getSubstring().contains(other.getSubstring()) && getSubstring().size() > other.getSubstring().size();
327  default:;
328  }
329  break;
330  default:;
331  }
332  return false;
333  }

335  QString CLogPattern::convertToQString(bool i18n) const
336  {
337  Q_UNUSED(i18n)
338  QString strategy;
339  QString categories = m_strings.values().join("|"); // clazy:exclude=container-anti-pattern
340  switch (m_strategy)
341  {
342  case Everything: strategy = "all"; break;
343  case ExactMatch: strategy = "exact match:" + categories; break;
344  case AnyOf: strategy = "any of:" + categories; break;
345  case AllOf: strategy = "all of:" + categories; break;
346  case StartsWith: strategy = "starts with:" + categories; break;
347  case EndsWith: strategy = "ends with:" + categories; break;
348  case Contains: strategy = "contains:" + categories; break;
349  case Nothing: strategy = "none"; break;
350  default: strategy = "<invalid>"; break;
351  }
352  return "{" + CStatusMessage::severitiesToString(m_severities) + "," + strategy + "}";
353  }

355  void CLogPattern::marshallToDbus(QDBusArgument &argument) const
356  {
357  // a bug in QtDBus prevents us from marshalling m_severities as a list
358  quint8 severities = 0;
359  for (auto s : m_severities) { severities |= (1 << static_cast<int>(s)); }
360 
361  argument << severities << m_strategy << m_strings.values();
362  }

364  void CLogPattern::unmarshallFromDbus(const QDBusArgument &argument)
365  {
366  quint8 severities;
367  QStringList strings;
368  argument >> severities >> m_strategy >> strings;
369  m_strings = QSet<QString>(strings.begin(), strings.end());
370 
371  m_severities.clear();
372  for (int s : { 0, 1, 2, 3 })
373  {
374  if (severities & (1 << s)) { m_severities.insert(static_cast<CStatusMessage::StatusSeverity>(s)); }
375  }
376  }
