100  void IAircraftModelLoader::startLoading(LoadMode mode, const IAircraftModelLoader::ModelConsolidationCallback &modelConsolidation, const QStringList &modelDirectories)
101  {
102  if (m_loadingInProgress) { return; }
103  if (mode == NotSet) { return; }
104  m_loadingInProgress = true;
105  m_loadingMessages.clear();
106 
107  const CSimulatorInfo simulator = this->getSimulator();
108  const bool needsCacheSynced = IAircraftModelLoader::needsCacheSynchronized(mode);
109  if (needsCacheSynced) { this->synchronizeCache(simulator); }
110 
111  const bool useCachedData = !mode.testFlag(CacheSkipped) && this->hasCachedData();
112  if (useCachedData && (mode.testFlag(CacheFirst) || mode.testFlag(CacheOnly)))
113  {
114  // we just just cache data
115  static const CStatusMessage status(this, CStatusMessage::SeverityInfo, u"Using cached data");
116  emit this->loadingFinished(status, simulator, CacheLoaded);
117  return;
118  }
119  if (mode.testFlag(CacheOnly))
120  {
121  // only cache, but we did not find any data yet (still in progress?)
122  // here we rely on the cache load slot, no need to emit here, will
123  // be done later in ps_cacheChanged. An alternative was to sync cache here
124  m_loadingInProgress = false;
125  return;
126  }
127 
128  // really load from disk?
129  const QStringList modelDirs = this->getInitializedModelDirectories(modelDirectories, simulator);
130  if (m_skipLoadingEmptyModelDir && modelDirs.isEmpty())
131  {
132  const CStatusMessage status = CStatusMessage(this, CStatusMessage::SeverityWarning,
133  u"Empty or not existing '%1' directory '%2', skipping read")
134  << simulator.toQString() << modelDirectories.join(", ");
135  m_loadingMessages.push_back(status);
136  m_loadingMessages.freezeOrder();
137  emit this->loadingFinished(m_loadingMessages, simulator, LoadingSkipped);
138  return;
139  }
140 
141  this->setObjectInfo(simulator);
142  this->startLoadingFromDisk(mode, modelConsolidation, modelDirs);
143  }

178  void IAircraftModelLoader::onLoadingFinished(const CStatusMessageList &statusMsgs, const CSimulatorInfo &simulator, IAircraftModelLoader::LoadFinishedInfo info)
179  {
180  if (!this->supportsSimulator(simulator)) { return; } // none of my business
181  this->setObjectInfo(simulator);
182 
183  // remark: in the past status used to be bool, now it is CStatusMessage
184  // so there is some redundancy here between status and m_loadingMessages
185  m_loadingInProgress = false;
186 
187  const QMap<int, int> counts = statusMsgs.countSeverities();
188  const int errors = counts.value(SeverityError);
189  const int warnings = counts.value(SeverityWarning);
190 
191  if (statusMsgs.hasWarningOrErrorMessages())
192  {
193  CLogMessage(this).log(m_loadingMessages.worstSeverity(),
194  u"Message loading produced %1 error and %2 warning messages")
195  << errors << warnings;
196  }
197  else
198  {
199  CLogMessage(this).info(u"Loading '%1' finished, success for '%2'") << IAircraftModelLoader::enumToString(info) << simulator.toQString();
200  }
201  }
