62  void SimpleCrypt::splitKey()
63  {
64  m_keyParts.clear();
65  m_keyParts.resize(8);
66  for (int i = 0; i < 8; i++)
67  {
68  quint64 part = m_key;
69  for (int j = i; j > 0; j--)
70  part = part >> 8;
71  part = part & 0xff;
72  m_keyParts[i] = static_cast<char>(part);
73  }

82  QByteArray SimpleCrypt::encryptToByteArray(const QByteArray &plaintext)
83  {
84  if (m_keyParts.isEmpty())
85  {
86  qWarning() << "No key set.";
87  m_lastError = ErrorNoKeySet;
88  return QByteArray();
89  }
90 
91 
92  QByteArray ba = plaintext;
93 
94  CryptoFlags flags = CryptoFlagNone;
95  if (m_compressionMode == CompressionAlways)
96  {
97  ba = qCompress(ba, 9); //maximum compression
98  flags |= CryptoFlagCompression;
99  }
100  else if (m_compressionMode == CompressionAuto)
101  {
102  QByteArray compressed = qCompress(ba, 9);
103  if (compressed.count() < ba.count())
104  {
105  ba = compressed;
106  flags |= CryptoFlagCompression;
107  }
108  }
109 
110  QByteArray integrityProtection;
111  if (m_protectionMode == ProtectionChecksum)
112  {
113  flags |= CryptoFlagChecksum;
114  QDataStream s(&integrityProtection, QIODevice::WriteOnly);
115  s << qChecksum(ba.constData(), ba.length());
116  }
117  else if (m_protectionMode == ProtectionHash)
118  {
119  flags |= CryptoFlagHash;
120  QCryptographicHash hash(QCryptographicHash::Sha1);
121  hash.addData(ba);
122 
123  integrityProtection += hash.result();
124  }
125 
126  //prepend a random char to the string
127  thread_local QRandomGenerator rng(QRandomGenerator::global()->generate());
128  char randomChar = static_cast<char>(rng.bounded(static_cast<int>(std::numeric_limits<char>::min()), static_cast<int>(std::numeric_limits<char>::max()) + 1));
129  ba = randomChar + integrityProtection + ba;
130 
131  int pos(0);
132  char lastChar(0);
133 
134  int cnt = ba.count();
135 
136  while (pos < cnt)
137  {
138  ba[pos] = ba.at(pos) ^ m_keyParts.at(pos % 8) ^ lastChar;
139  lastChar = ba.at(pos);
140  ++pos;
141  }
142 
143  QByteArray resultArray;
144  resultArray.append(char(0x03)); //version for future updates to algorithm
145  resultArray.append(char(flags)); //encryption flags
146  resultArray.append(ba);
147 
148  m_lastError = ErrorNoError;
149  return resultArray;
150  }

192  QByteArray SimpleCrypt::decryptToByteArray(const QByteArray &cypher)
193  {
194  if (m_keyParts.isEmpty())
195  {
196  qWarning() << "No key set.";
197  m_lastError = ErrorNoKeySet;
198  return QByteArray();
199  }
200 
201  QByteArray ba = cypher;
202 
203  if (cypher.count() < 3)
204  return QByteArray();
205 
206  char version = ba.at(0);
207 
208  if (version != 3) //we only work with version 3
209  {
210  m_lastError = ErrorUnknownVersion;
211  qWarning() << "Invalid version or not a cyphertext.";
212  return QByteArray();
213  }
214 
215  CryptoFlags flags = CryptoFlags(ba.at(1));
216 
217  ba = ba.mid(2);
218  int pos(0);
219  int cnt(ba.count());
220  char lastChar = 0;
221 
222  while (pos < cnt)
223  {
224  char currentChar = ba[pos];
225  ba[pos] = ba.at(pos) ^ lastChar ^ m_keyParts.at(pos % 8);
226  lastChar = currentChar;
227  ++pos;
228  }
229 
230  ba = ba.mid(1); //chop off the random number at the start
231 
232  bool integrityOk(true);
233  if (flags.testFlag(CryptoFlagChecksum))
234  {
235  if (ba.length() < 2)
236  {
237  m_lastError = ErrorIntegrityFailed;
238  return QByteArray();
239  }
240  quint16 storedChecksum;
241  {
242  QDataStream s(&ba, QIODevice::ReadOnly);
243  s >> storedChecksum;
244  }
245  ba = ba.mid(2);
246  quint16 checksum = qChecksum(ba.constData(), ba.length());
247  integrityOk = (checksum == storedChecksum);
248  }
249  else if (flags.testFlag(CryptoFlagHash))
250  {
251  if (ba.length() < 20)
252  {
253  m_lastError = ErrorIntegrityFailed;
254  return QByteArray();
255  }
256  QByteArray storedHash = ba.left(20);
257  ba = ba.mid(20);
258  QCryptographicHash hash(QCryptographicHash::Sha1);
259  hash.addData(ba);
260  integrityOk = (hash.result() == storedHash);
261  }
262 
263  if (!integrityOk)
264  {
265  m_lastError = ErrorIntegrityFailed;
266  return QByteArray();
267  }
268 
269  if (flags.testFlag(CryptoFlagCompression))
270  ba = qUncompress(ba);
271 
272  m_lastError = ErrorNoError;
273  return ba;
274  }
