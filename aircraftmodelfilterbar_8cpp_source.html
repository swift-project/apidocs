69  std::unique_ptr<IModelFilter<CAircraftModelList>> CAircraftModelFilterBar::createModelFilter() const
70  {
71  CAircraftModel::ModelModeFilter mf = CAircraftModel::All;
72  if (ui->cbt_IncludeExclude->checkState() == Qt::Checked)
73  {
74  mf = CAircraftModel::Include;
75  }
76  else if (ui->cbt_IncludeExclude->checkState() == Qt::Unchecked)
77  {
78  mf = CAircraftModel::Exclude;
79  }
80 
81  DbKeyStateFilter dbf = BlackMisc::Db::All;
82  if (ui->cbt_Db->checkState() == Qt::Checked)
83  {
84  dbf = BlackMisc::Db::Valid;
85  }
86  else if (ui->cbt_Db->checkState() == Qt::Unchecked)
87  {
88  dbf = BlackMisc::Db::Invalid;
89  }
90 
91  return std::make_unique<CAircraftModelFilter>(
92  convertDbId(ui->le_Id->text()),
93  ui->le_ModelString->text(),
94  ui->le_ModelDescription->text(),
95  mf,
96  dbf,
97  ui->cbt_Military->checkState(),
98  ui->cbt_ColorLiveries->checkState(),
99  ui->le_AircraftIcao->text(),
100  ui->le_AircraftManufacturer->text(),
101  ui->le_AirlineIcao->text(),
102  ui->le_AirlineName->text(),
103  ui->le_LiveryCode->text(),
104  ui->le_FileName->text(),
105  ui->comp_CombinedType->getCombinedType(),
106  ui->comp_SimulatorSelector->getValue(),
107  ui->comp_DistributorSelector->getDistributor());
108  }
