60  CStatusMessageList CDatabaseAuthenticationService::login(const QString &username, const QString &password)
61  {
62  CStatusMessageList msgs;
63  static const CLogCategoryList cats(CLogCategoryList(this).withValidation());
64 
65  if (!sApp || m_shutdown)
66  {
67  msgs.push_back(CStatusMessage(cats, CStatusMessage::SeverityError, u"Shutdown in progress"));
68  return msgs;
69  }
70 
71  const QString un(username.trimmed());
72  const QString pw(password.trimmed());
73  if (un.isEmpty()) { msgs.push_back(CStatusMessage(cats, CStatusMessage::SeverityError, u"No user name/id")); }
74  if (pw.isEmpty()) { msgs.push_back(CStatusMessage(cats, CStatusMessage::SeverityError, u"No password")); }
75  if (!msgs.isEmpty()) { return msgs; }
76 
77  const CUrl url(sApp->getGlobalSetup().getDbLoginServiceUrl());
78  QString msg;
79  if (!CNetworkUtils::canConnect(url, msg))
80  {
81  msgs.push_back(CStatusMessage(cats, CStatusMessage::SeverityError, msg));
82  return msgs;
83  }
84 
85  QUrlQuery params;
86  params.addQueryItem("username", un);
87  params.addQueryItem("password", pw);
88  if (sApp->getGlobalSetup().dbDebugFlag()) { CNetworkUtils::addDebugFlag(params); }
89 
90  const QString query = params.toString();
91  const QNetworkRequest request(CNetworkUtils::getSwiftNetworkRequest(url, CNetworkUtils::PostUrlEncoded, sApp->getApplicationNameAndVersion()));
92  sApp->postToNetwork(request, CApplication::NoLogRequestId, query.toUtf8(), { this, &CDatabaseAuthenticationService::parseServerResponse });
93  static const QString rm("Sent request to authentication server '%1'");
94  msgs.push_back(CStatusMessage(cats, CStatusMessage::SeverityInfo, rm.arg(url.toQString())));
95  return msgs;
96  }

102  url.setQuery("logoff=true");
103  QNetworkRequest request(CNetworkUtils::getSwiftNetworkRequest(url));
104  sApp->getFromNetwork(request, { this, &CDatabaseAuthenticationService::parseServerResponse });
105  m_swiftDbUser.set(CAuthenticatedUser());

108  void CDatabaseAuthenticationService::parseServerResponse(QNetworkReply *nwReplyPtr)
109  {
110  // always cleanup reply
111  QScopedPointer<QNetworkReply, QScopedPointerDeleteLater> nwReply(nwReplyPtr);
112 
113  if (m_shutdown) { return; }
114  QString urlString(nwReply->url().toString());
115  if (urlString.contains("logoff", Qt::CaseInsensitive))
116  {
117  sApp->deleteAllCookies();
118  emit this->logoffFinished();
119  return;
120  }
121 
122  if (nwReply->error() == QNetworkReply::NoError)
123  {
124  const QString json(nwReply->readAll().trimmed());
125  if (json.isEmpty())
126  {
127  CLogMessage(this).error(u"Authentication failed, no response from '%1'") << urlString;
128  return;
129  }
130  if (!Json::looksLikeJson(json))
131  {
132  CLogMessage(this).error(u"Illegal JSON object: %1") << CNetworkUtils::removeHtmlPartsFromPhpErrorMessage(json);
133  return;
134  }
135 
136  static const CLogCategoryList cats(CLogCategoryList(this).withValidation());
137  const QJsonObject jsonObj(Json::jsonObjectFromString(json));
138  CAuthenticatedUser user = CAuthenticatedUser::fromDatabaseJson(jsonObj.contains("user") ? jsonObj["user"].toObject() : jsonObj);
139  CStatusMessageList msgs;
140  if (jsonObj.contains("messages"))
141  {
142  msgs = CStatusMessageList::fromDatabaseJson(jsonObj["messages"].toArray());
143  msgs.setCategories(cats);
144  }
145 
146  // allow auto enabled for SSO users
147  if (user.isValid() && !user.isEnabled())
148  {
149  if (user.getRoles().hasRole("VATSIMUSER"))
150  {
151  user.setEnabled(true);
152  msgs.push_back(CStatusMessage(cats, CStatusMessage::SeverityInfo, u"Auto enabled SSO user"));
153  }
154  }
155 
156  if (!user.isAuthenticated() || !user.isValid())
157  {
158  if (!msgs.hasErrorMessages())
159  {
160  msgs.push_back(CStatusMessage(cats, CStatusMessage::SeverityError, u"Cannot login, user or password wrong"));
161  }
162  }
163  else
164  {
165  if (!user.isEnabled())
166  {
167  msgs.push_back(CStatusMessage(cats, CStatusMessage::SeverityError, u"User is disabled"));
168  }
169  if (user.getRoles().isEmpty())
170  {
171  msgs.push_back(CStatusMessage(cats, CStatusMessage::SeverityError, u"User has no roles"));
172  }
173  }
174  m_swiftDbUser.set(user);
175  emit userAuthenticationFinished(user, msgs);
176  }
177  else
178  {
179  CLogMessage(this).error(u"Authentication failed, %1") << nwReply->errorString();
180  return;
181  }
182  }
