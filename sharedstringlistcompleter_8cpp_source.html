10  {
11  QStringListModel *model = this->getCompleterModel();
12  Q_ASSERT_X(model, Q_FUNC_INFO, "Model missing");
13  bool updated = false;
14  const qint64 now = QDateTime::currentMSecsSinceEpoch();
15  if (model->rowCount() != data.size() || (now - m_lastUpdated) > cacheTimeMs)
16  {
17  // quick check, different size means data changed -> update
18  // otherwise rely on time
19  m_lastUpdated = now;
20  model->setStringList(data);
21  updated = true;
22  }
23  return updated;
24  }
