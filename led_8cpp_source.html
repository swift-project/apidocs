48  void CLedWidget::setLed(LedColor ledColor)
49  {
50  Q_ASSERT_X(!m_renderer.isNull(), Q_FUNC_INFO, "no renderer");
51  if (!m_renderer) { return; }
52 
53  // load image, init renderer
54  QString ledShapeAndColor(shapes().at(static_cast<int>(m_shape)));
55  if (ledColor == NoColor)
56  {
57  switch (m_blinkState)
58  {
59  case On:
60  m_currentToolTip = m_tooltipOn;
61  ledShapeAndColor.append(CLedWidget::colorString(m_colorOn));
62  break;
63  case TriState:
64  m_currentToolTip = m_tooltipTriState;
65  ledShapeAndColor.append(CLedWidget::colorString(m_colorTriState));
66  break;
67  case Off:
68  default:
69  m_currentToolTip = m_tooltipOff;
70  ledShapeAndColor.append(CLedWidget::colorString(m_colorOff));
71  break;
72  }
73  }
74  else
75  {
76  if (ledColor == m_colorOn)
77  {
78  m_currentToolTip = m_tooltipOn;
79  ledShapeAndColor.append(CLedWidget::colorString(m_colorOn));
80  }
81  else if (ledColor == m_colorOff)
82  {
83  m_currentToolTip = m_tooltipOff;
84  ledShapeAndColor.append(CLedWidget::colorString(m_colorOff));
85  }
86  else
87  {
88  m_currentToolTip = m_tooltipTriState;
89  ledShapeAndColor.append(CLedWidget::colorString(m_colorTriState));
90  }
91  }
92  this->setToolTip(m_currentToolTip); // for widget
93 
94  // init renderer, load led.
95  m_renderer->load(ledShapeAndColor); // load by filename
96 
97  // original size
98  const QSize s = m_renderer->defaultSize();
99  m_whRatio = s.width() / s.height();
100 
101  // size
102  if (m_widthTarget < 0) { m_widthTarget = widths().at(static_cast<int>(m_shape)); }
103  const double h = m_widthTarget / m_whRatio;
104  m_heightCalculated = qRound(h);
105 
106  this->setFixedHeight(m_heightCalculated);
107  this->setFixedWidth(m_widthTarget);
108  this->update();
109  }

228  void CLedWidget::setOn(bool on, int resetTimeMs)
229  {
230  State s = on ? On : Off;
231  if (resetTimeMs > 0)
232  {
233  QPointer<CLedWidget> myself(this);
234  m_resetTimer.singleShot(resetTimeMs, this, [=] {
235  if (!myself) { return; }
236  this->resetState();
237  });
238  }
239  else
240  {
241  m_resetTimer.stop();
242  m_value = s;
243  }
244  if (m_blinkState == s) { return; }
245  m_blinkState = s;
246  this->setLed();
247  }
