145  void CRgbColor::setByString(const QString &color, bool isName)
146  {
147  if (color.isEmpty()) { return; }
148  else if (isName)
149  {
150  const QColor q(color);
151  m_r = q.red();
152  m_g = q.green();
153  m_b = q.blue();
154  }
155  else
156  {
157  const QString c(color.trimmed());
158  QColor q(c);
159  if (this->setQColor(q)) { return; }
160  if (c.startsWith("#"))
161  {
162  this->setInvalid();
163  return;
164  }
165  q.setNamedColor("#" + c);
166  this->setQColor(q);
167  }

175  double CRgbColor::colorDistance(const CRgbColor &color) const
176  {
177  if (!this->isValid() && !color.isValid()) { return 0; }
178  if (!this->isValid() || !color.isValid()) { return 1; }
179  if (*this == color) { return 0.0; } // avoid rounding
180 
181  // all values 0-1
182  const double rd = (normalizedRed() - color.normalizedRed());
183  const double bd = (normalizedBlue() - color.normalizedBlue());
184  const double gd = (normalizedGreen() - color.normalizedGreen());
185  return (rd * rd + bd * bd + gd * gd) / 3.0;
186  }

250  int CRgbColor::compare(const CRgbColor &color) const
251  {
252  int c = Compare::compare(m_r, color.m_r);
253  if (c != 0) { return c; }
254  c = Compare::compare(m_g, color.m_g);

259  double CRgbColor::colorRange() const
260  {
261  if (!this->isValid()) { return 255; }
262  if (m_b < 256 && m_g < 256 && m_r < 256) { return 255; }
263  if (m_b < 4096 && m_g < 4096 && m_r < 4096) { return 4095; }
264  return 65535;
265  }
