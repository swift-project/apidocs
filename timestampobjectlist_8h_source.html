385  MillisecondsMinMaxMean getTimestampDifferenceMinMaxMean() const
386  {
387  MillisecondsMinMaxMean mmm;
388  mmm.reset();
389  const CONTAINER &container = this->container();
390  if (container.size() < 2) { return mmm; }
391 
392  // Do not confuse with adjusted sort hint!
393  if (container.m_tsSortHint == NoTimestampSortHint)
394  {
395  CONTAINER copy(container);
396  copy.sortLatestFirst();
397  copy.m_tsSortHint = TimestampLatestFirst;
398  return copy.getTimestampDifferenceMinMaxMean();
399  }
400 
401  mmm.max = std::numeric_limits<qint64>::min();
402  mmm.min = std::numeric_limits<qint64>::max();
403  qint64 mean = 0;
404  int c = 0;
405  OBJ last;
406 
407  for (const OBJ &object : container)
408  {
409  if (c > 0)
410  {
411  const ITimestampBased &l = last;
412  const ITimestampBased &o = object;
413  const qint64 diff = l.getAbsTimeDifferenceMs(o);
414  if (diff > mmm.max) { mmm.max = diff; }
415  if (diff < mmm.min) { mmm.min = diff; }
416  mean += diff;
417  }
418  c++;
419  last = object;
420  }
421 
422  mmm.mean = mean / c;
423  return mmm;
424  }

542  void push_frontKeepLatestFirstAdjustOffset(const OBJ &value, bool replaceSameTimestamp = true, int maxElements = -1)
543  {
544  ITimestampWithOffsetObjectList<OBJ, CONTAINER>::push_frontKeepLatestFirst(value, replaceSameTimestamp, maxElements);
545 
546  // now sorted by timestamp
547  // this reflects normally the incoming order
548  // ts
549  // 8: os 2 adj 12 => min os 4
550  // 6: os 2 adj 11 => min os 5
551  // 5: os 5 adj 10
552  // 0: os 5 adj 5
553  CONTAINER &c = this->container();
554  if (c.size() < 2) { return; }
555  ITimestampWithOffsetBased &front = c.front();
556  const ITimestampWithOffsetBased &second = c[1];
557  if (!front.isNewerThanAdjusted(second))
558  {
559  // const qint64 minOs = qMin(front.getTimeOffsetMs(), second.getTimeOffsetMs());
560  const qint64 minReqOs = second.getAdjustedMSecsSinceEpoch() - front.getMSecsSinceEpoch(); // minimal required
561  const qint64 avgOs = (front.getTimeOffsetMs() + second.getTimeOffsetMs()) / 2;
562  const qint64 os = qMax(minReqOs + 1, avgOs); // at least +1, as value must be > (greater)
563  front.setTimeOffsetMs(os);
564  }
565 
566  if (BlackConfig::CBuildConfig::isLocalDeveloperDebugBuild())
567  {
568  BLACK_VERIFY_X(front.isNewerThanAdjusted(second), Q_FUNC_INFO, "Front/second timestamp");
569  BLACK_VERIFY_X(this->isSortedAdjustedLatestFirst(), Q_FUNC_INFO, "Wrong sort order");
570  }
571  }
