67  bool CSimulatedAircraftList::updateWithVatsimDataFileData(CSimulatedAircraft &aircraftToBeUpdated) const
68  {
69  if (this->isEmpty()) return false;
70  if (aircraftToBeUpdated.hasRealName() && aircraftToBeUpdated.hasId() && aircraftToBeUpdated.hasAircraftAndAirlineDesignator()) { return false; }
71 
72  CSimulatedAircraft currentDataFileAircraft = this->findFirstByCallsign(aircraftToBeUpdated.getCallsign());
73  if (currentDataFileAircraft.getCallsign().isEmpty()) return false;
74 
75  CUser user = aircraftToBeUpdated.getPilot();
76  user.updateMissingParts(currentDataFileAircraft.getPilot());
77  aircraftToBeUpdated.setPilot(user);
78 
79  CAircraftIcaoCode aircraftIcao = aircraftToBeUpdated.getAircraftIcaoCode();
80  CAirlineIcaoCode airlineIcao = aircraftToBeUpdated.getAirlineIcaoCode();
81  aircraftIcao.updateMissingParts(currentDataFileAircraft.getAircraftIcaoCode());
82  airlineIcao.updateMissingParts(currentDataFileAircraft.getAirlineIcaoCode());
83  aircraftToBeUpdated.setIcaoCodes(aircraftIcao, airlineIcao);
84  return true;
85  }

220  bool CSimulatedAircraftList::replaceOrAddByCallsign(const CSimulatedAircraft &aircraft)
221  {
222  const CCallsign cs(aircraft.getCallsign());
223  if (cs.isEmpty()) { return false; }
224 
225  if (this->containsCallsign(cs))
226  {
227  int c = this->replaceIf(&CSimulatedAircraft::getCallsign, cs, aircraft);
228  return c > 0;
229  }
230  this->push_back(aircraft);

264  void CSimulatedAircraftList::sortByDistanceToReferencePositionRenderedCallsign()
265  {

267  if (a.getRelativeDistance() != b.getRelativeDistance()) { return a.getRelativeDistance() < b.getRelativeDistance(); }
