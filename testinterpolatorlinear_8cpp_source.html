66  void CTestInterpolatorLinear::basicInterpolatorTests()
67  {
68  const CCallsign cs("SWIFT");
69  CRemoteAircraftProviderDummy provider;
70  CInterpolatorLinear interpolator(cs, nullptr, nullptr, &provider);
71  interpolator.markAsUnitTest();
72 
73  // fixed time so everything can be debugged
74  const qint64 ts = 1425000000000; // QDateTime::currentMSecsSinceEpoch();
75  const qint64 deltaT = 5000; // ms
76  const qint64 offset = 5000; // ms
77  for (int i = IRemoteAircraftProvider::MaxSituationsPerCallsign - 1; i >= 0; i--)
78  {
79  const CAircraftSituation s(getTestSituation(cs, i, ts, deltaT, offset));
80 
81  // check height above ground
82  CLength hag = (s.getAltitude() - s.getGroundElevation());
83  QVERIFY2(s.getHeightAboveGround() == hag, "Wrong elevation");
84  provider.insertNewSituation(s);
85  }
86 
87  constexpr int partsCount = 10;
88  for (int i = partsCount - 1; i >= 0; i--)
89  {
90  const CAircraftParts p(getTestParts(i, ts, deltaT));
91  provider.insertNewAircraftParts(cs, p, false);
92  }
93 
94  // make sure signals are processed, if the interpolator depends on those signals
95  QCoreApplication::processEvents(QEventLoop::AllEvents, 1000);
96 
97  // interpolation functional check
98  const CInterpolationAndRenderingSetupPerCallsign setup;
99  double latOld = 360.0;
100  double lngOld = 360.0;
101 
102  qint64 from = ts - 2 * deltaT + offset;
103  qint64 to = ts; // ts + offset is last value, but we have to consider offset
104  qint64 step = deltaT / 20;
105  for (qint64 currentTime = from; currentTime < to; currentTime += step)
106  {
107  // This will use time range
108  // from: ts - 2 * deltaT + offset
109  // to: ts + offset
110 
111  const CInterpolationResult result = interpolator.getInterpolation(currentTime, setup);
112  const CAircraftSituation currentSituation(result);
113  QVERIFY2(result.getInterpolationStatus().isInterpolated(), "Value was not interpolated");
114  const double latDeg = currentSituation.getPosition().latitude().valueRounded(CAngleUnit::deg(), 5);
115  const double lngDeg = currentSituation.getPosition().longitude().valueRounded(CAngleUnit::deg(), 5);
116  QVERIFY2(latDeg < latOld && lngDeg < lngOld, QStringLiteral("Values shall decrease: %1/%2 %3/%4").arg(latDeg).arg(latOld).arg(lngDeg).arg(lngOld).toLatin1());
117  QVERIFY2(latDeg >= 0 && latDeg <= IRemoteAircraftProvider::MaxSituationsPerCallsign, "Values shall be in range");
118  latOld = latDeg;
119  lngOld = lngDeg;
120  }
121 
122  QElapsedTimer timer;
123  timer.start();
124  int interpolationNo = 0;
125  const qint64 startTimeMsSinceEpoch = ts - 2 * deltaT;
126 
127  // Pseudo performance test:
128  // Those make not completely sense, as the performance depends on the implementation of
129  // the dummy provider, which is different from the real provider
130  // With one callsign in the lists (of dummy provider) it is somehow expected to be roughly the same performance
131 
132  interpolator.resetLastInterpolation();
133  interpolator.markAsUnitTest();
134  for (int loops = 0; loops < 20; loops++)
135  {
136  from = startTimeMsSinceEpoch + offset;
137  to = ts; // ts + offset is last value, but we have to consider offset
138  step = deltaT / 20;
139 
140  for (qint64 currentTime = from; currentTime < to; currentTime += step)
141  {
142  // This will use range
143  // from: ts - 2* deltaT + offset
144  // to: ts + offset
145  const CInterpolationResult result = interpolator.getInterpolation(currentTime, setup);
146  const CAircraftSituation currentSituation(result);
147  QVERIFY2(result.getInterpolationStatus().isInterpolated(), "Not interpolated");
148  QVERIFY2(!currentSituation.getCallsign().isEmpty(), "Empty callsign");
149  QVERIFY2(currentSituation.getCallsign() == cs, "Wrong callsign");
150  const double latDeg = currentSituation.getPosition().latitude().valueRounded(CAngleUnit::deg(), 5);
151  const double lngDeg = currentSituation.getPosition().longitude().valueRounded(CAngleUnit::deg(), 5);
152  Q_UNUSED(latDeg);
153  Q_UNUSED(lngDeg);
154  interpolationNo++;
155  }
156  }
157 
158  // check on time just to learn if interpolation suddenly gets very slow
159  // this is a risky test as in some situations the values can be exceeded
160  int timeMs = timer.elapsed();
161  QVERIFY2(timeMs < interpolationNo * 1.5, "Interpolation > 1.5ms");
162  qDebug() << timeMs << "ms"
163  << "for" << interpolationNo << "interpolations";
164 
165  int fetchedParts = 0;
166  timer.start();
167  for (qint64 currentTime = ts - 2 * deltaT; currentTime < ts; currentTime += 250)
168  {
169  const CInterpolationResult result = interpolator.getInterpolation(currentTime, setup);
170  fetchedParts++;
171  QVERIFY2(result.getPartsStatus().isSupportingParts(), "Parts not supported");
172  }
173  timeMs = timer.elapsed();
174  qDebug() << timeMs << "ms"
175  << "for" << fetchedParts << "fetched parts";
176  }

348  CAircraftSituation CTestInterpolatorLinear::getTestSituation(const CCallsign &callsign, int number, qint64 ts, qint64 deltaT, qint64 offset)
349  {
350  const CAltitude alt(number, CAltitude::MeanSeaLevel, CLengthUnit::m());
351  const CLatitude lat(number, CAngleUnit::deg());
352  const CLongitude lng(180.0 + number, CAngleUnit::deg());
353  const CHeading heading(number * 10, CHeading::True, CAngleUnit::deg());
354  const CAngle bank(number, CAngleUnit::deg());
355  const CAngle pitch(number, CAngleUnit::deg());
356  const CSpeed gs(number * 10, CSpeedUnit::km_h());
357  const CAltitude gndElev({ 0, CLengthUnit::m() }, CAltitude::MeanSeaLevel);
358  const CCoordinateGeodetic c(lat, lng, alt);
359  CAircraftSituation s(callsign, c, heading, pitch, bank, gs);
360  s.setGroundElevation(gndElev, CAircraftSituation::Test);
361  s.setMSecsSinceEpoch(ts - deltaT * number); // values in past
362  s.setTimeOffsetMs(offset);
363  return s;
364  }
