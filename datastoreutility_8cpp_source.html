40  int CDatastoreUtility::extractIntegerKey(const QString &stringWithKey)
41  {
42  QString ks(stringWithKey.trimmed());
43  if (ks.isEmpty()) { return -1; }
44  bool ok = false;
45  int key = ks.toInt(&ok);
46  if (ok) { return key; } // only a number
47 
48  // key in string with ()
49  int i1 = ks.lastIndexOf('(');
50  if (i1 < 0) { return -1; }
51  int i2 = ks.lastIndexOf(')');
52  if (i2 <= i1 + 1) { return -1; }
53  const QString n(ks.mid(i1 + 1, i2 - i1 - 1));
54  ok = false;
55  key = n.toInt(&ok);
56  return ok ? key : -1;
57  }

73  bool CDatastoreUtility::parseAutoPublishResponse(const QString &jsonResponse, CStatusMessageList &messages)
74  {
75  if (jsonResponse.isEmpty())
76  {
77  messages.push_back(CStatusMessage(static_cast<CDatastoreUtility *>(nullptr), CStatusMessage::SeverityError, u"Empty JSON data for published models"));
78  return false;
79  }
80 
81  const QJsonDocument jsonDoc(QJsonDocument::fromJson(jsonResponse.toUtf8()));
82 
83  // array of messages only
84  if (jsonDoc.isArray())
85  {
86  const CStatusMessageList msgs(CStatusMessageList::fromDatabaseJson(jsonDoc.array()));
87  messages.push_back(msgs);
88  return true;
89  }
90 
91  // no object -> most likely some fucked up HTML string with the PHP error
92  if (!jsonDoc.isObject())
93  {
94  const QString phpError(CNetworkUtils::removeHtmlPartsFromPhpErrorMessage(jsonResponse));
95  messages.push_back(CStatusMessage(static_cast<CDatastoreUtility *>(nullptr), CStatusMessage::SeverityError, phpError));
96  return false;
97  }
98 
99  QJsonObject json(jsonDoc.object());
100  if (json.contains("msgs"))
101  {
102  const QJsonValue msgJson(json.take("msgs"));
103  const CStatusMessageList msgs(CStatusMessageList::fromDatabaseJson(msgJson.toArray()));
104  if (!msgs.isEmpty())
105  {
106  messages.push_back(msgs);
107  return !messages.hasErrorMessages();
108  }
109  }
110  return false;
111  }
