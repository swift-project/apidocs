89  void CContextApplication::callHotkeyActionRemotely(const QString &action, bool argument, const CIdentifier &origin)
90  {
91  if (origin.hasApplicationProcessId())
92  {
93  // If it originated from this process, then we are going to emit a signal
94  emit this->remoteHotkeyAction(action, argument, origin);
95  }
96  else
97  {
98  // action came from a different process but on the same machine. Ignore
99  if (origin.isFromLocalMachine()) { return; }
100 
101  // Different process and different machine. Process it.
102  // However, it should not emit a remote action itself.
103  sApp->getInputManager()->callFunctionsBy(action, argument, false);
104  }
105  }

121  CIdentifier CContextApplication::registerApplication(const CIdentifier &application)
122  {
123  if (m_debugEnabled) { CLogMessage(this, CLogCategories::contextSlot()).debug() << Q_FUNC_INFO << application; }
124 
125  if (!m_registeredApplications.contains(application))
126  {
127  m_registeredApplications.push_back(application);
128  emit this->registrationChanged();
129  emit this->hotkeyActionsRegistered(sApp->getInputManager()->allAvailableActions(), {});
130  }
131 
132  this->cleanupRegisteredApplications();
133  return application;
134  }
