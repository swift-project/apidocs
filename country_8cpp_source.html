87  bool CCountry::matchesCountryName(const QString &name) const
88  {
89  if (name.isEmpty() || m_name.isEmpty()) { return false; }
90  if (caseInsensitiveStringCompare(name, this->getDbKey())) { return true; } // exact ISO match
91  if (caseInsensitiveStringCompare(name, this->getIso3Code())) { return true; } // exact ISO3 match
92  if (name.length() < 5)
93  {
94  // contains would be too fuzzy for short names
95  return caseInsensitiveStringCompare(name, m_name) || caseInsensitiveStringCompare(name, m_simplifiedName);
96  }
97  else
98  {
99  return m_name.contains(name, Qt::CaseInsensitive) || m_simplifiedName.contains(name, Qt::CaseInsensitive);
100  }
101  }

164  {
165  if (index.isMyself()) { return getIsoCode().compare(compareValue.getIsoCode(), Qt::CaseInsensitive); }
166  if (IDatastoreObjectWithStringKey::canHandleIndex(index)) { return IDatastoreObjectWithStringKey::comparePropertyByIndex(index, compareValue); }
167  const ColumnIndex i = index.frontCasted<ColumnIndex>();
168  switch (i)

170  case IndexIsoCode: return getIsoCode().compare(compareValue.getIsoCode(), Qt::CaseInsensitive);
171  case IndexIso3Code: return getIso3Code().compare(compareValue.getIsoCode(), Qt::CaseInsensitive);
172  case IndexName: return getName().compare(compareValue.getName(), Qt::CaseInsensitive);
173  case IndexAlias1: return this->getAlias1().compare(compareValue.getAlias1(), Qt::CaseInsensitive);

175  default: break;

180  CCountry CCountry::fromDatabaseJson(const QJsonObject &json, const QString &prefix)
181  {
182  if (!existsKey(json, prefix))
183  {
184  // when using relationship, this can be null
185  return CCountry();
186  }

188  const QString name(json.value(prefix % u"country").toString());
189  const QString alias1(json.value(prefix % u"alias1").toString());
190  const QString alias2(json.value(prefix % u"alias2").toString());
191  const QString iso3(json.value(prefix % u"iso3").toString());
192  const QString historic(json.value(prefix % u"historic").toString());
193  CCountry country(iso, name);
194  country.setLoadedFromDb(true);
195  country.setAlias1(alias1);
196  country.setAlias2(alias2);
197  country.setIso3Code(iso3);
198  country.setHistoric(stringToBool(historic));
199  country.setKeyVersionTimestampFromDatabaseJson(json, prefix);
