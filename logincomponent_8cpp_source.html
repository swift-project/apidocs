231  void CLoginComponent::toggleNetworkConnection()
232  {
233  if (!sGui || sGui->isShuttingDown()) { return; }
234  if (!sGui->getIContextNetwork() || !sGui->getIContextAudio()) { return; }
235 
236  m_networkConnected = sGui && sGui->getIContextNetwork()->isConnected();
237  const bool vatsimLogin = this->isVatsimNetworkTabSelected();
238  m_networkSetup.setAutoLogoff(ui->cb_AutoLogoff->isChecked());
239 
240  ui->form_Pilot->setVatsimValidation(vatsimLogin);
241  this->updateUiConnectState();
242 
243  // reset time
244  this->setLogoffCountdown();
245 
246  CServer currentServer; // used for login
247  CSimulatedAircraft ownAircraft; // used own aircraft
248  CStatusMessage msg;
249  if (!m_networkConnected)
250  {
251  if (!this->validateAircraftValues())
252  {
253  this->showOverlayHTMLMessage(CStatusMessage(this).validationWarning(u"Invalid aircraft data, login not possible"), OverlayMessageMs);
254  return;
255  }
256 
257  const CStatusMessageList pilotMsgs = ui->form_Pilot->validate();
258  if (pilotMsgs.isFailure())
259  {
260  this->showOverlayHTMLMessage(CStatusMessage(this).validationWarning(u"Invalid pilot data, login not possible"), OverlayMessageMs);
261  return;
262  }
263 
264  // sync values with GUI values
265  this->updateOwnAircraftCallsignAndPilotFromGuiValues();
266  this->updateOwnAircaftIcaoValuesFromGuiValues();
267 
268  // Login mode
269  const CLoginMode mode = ui->frp_LoginMode->getLoginMode();
270  if (mode.isObserver()) { CLogMessage(this).info(u"Login in observer mode"); }
271 
272  // Server
273  currentServer = this->getCurrentServer();
274  const CUser user = this->getUserFromPilotGuiValues();
275  currentServer.setUser(user);
276 
277  // FSD setup, then override
278  if (ui->form_FsdDetails->isFsdSetupEnabled())
279  {
280  const CFsdSetup fsd = ui->form_FsdDetails->getValue();
281  currentServer.setFsdSetup(fsd);
282  }
283 
284  if (ui->form_Voice->isVoiceSetupEnabled())
285  {
286  const CVoiceSetup voice = ui->form_Voice->getValue();
287  currentServer.setVoiceSetup(voice);
288  }
289 
290  ui->frp_CurrentServer->setServer(currentServer);
291  sGui->getIContextOwnAircraft()->updateOwnAircraftPilot(currentServer.getUser());
292 
293  // set own aircraft from all values
294  ownAircraft = sGui->getIContextOwnAircraft()->getOwnAircraft();
295 
296  // Login
297  if (sGui->getCContextAudioBase())
298  {
299  sGui->getCContextAudioBase()->setVoiceSetup(currentServer.getVoiceSetup());
300  }
301 
302  msg = sGui->getIContextNetwork()->connectToNetwork(currentServer, {}, true, {}, true, {}, mode);
303  if (msg.isSuccess())
304  {
305  Q_ASSERT_X(currentServer.isValidForLogin(), Q_FUNC_INFO, "invalid server");
306  sGui->setExtraWindowTitle(QStringLiteral("[%1]").arg(ownAircraft.getCallsignAsString()));
307  CCrashHandler::instance()->crashAndLogInfoUserName(currentServer.getUser().getRealNameAndId());
308  CCrashHandler::instance()->crashAndLogInfoFlightNetwork(currentServer.getEcosystem().toQString(true));
309  CCrashHandler::instance()->crashAndLogAppendInfo(currentServer.getServerSessionId(false));
310  m_networkSetup.setLastServer(currentServer);
311 
312  const CAircraftModel ownModel = ownAircraft.getModel();
313  m_lastAircraftModel.set(ownModel);
314  ui->le_LoginCallsign->setText(ownAircraft.getCallsignAsString());
315  ui->le_LoginHomeBase->setText(currentServer.getUser().getHomeBase().asString());
316  if (vatsimLogin) { m_networkSetup.setLastVatsimServer(currentServer); }
317  }
318  else
319  {
320  sGui->setExtraWindowTitle("");
321  }
322  }
323  else
324  {
325  // disconnect from network
326  sGui->setExtraWindowTitle("");
327  msg = sGui->getIContextNetwork()->disconnectFromNetwork();
328  }
329 
330  // log message and trigger events
331  msg.addCategories(this);
332  CLogMessage::preformatted(msg);
333  if (msg.isSuccess())
334  {
335  this->setGuiLoginAsValues(ownAircraft);
336  emit this->loginOrLogoffSuccessful();
337  }
338  else
339  {
340  emit this->loginOrLogoffCancelled();
341  }
342  }

344  void CLoginComponent::onWebServiceDataRead(CEntityFlags::Entity entity, CEntityFlags::ReadState state, int number, const QUrl &url)
345  {
346  if (!CEntityFlags::isFinishedReadState(state)) { return; }
347  if (!sGui || !sGui->getIContextNetwork() || sGui->isShuttingDown()) { return; }
348 
349  Q_UNUSED(number)
350  Q_UNUSED(url)
351 
352  if (entity == CEntityFlags::VatsimDataFile)
353  {
354  CServerList vatsimFsdServers = sGui->getIContextNetwork()->getVatsimFsdServers();
355  if (vatsimFsdServers.isEmpty()) { return; }
356  vatsimFsdServers.sortBy(&CServer::getName);
357  const CServer currentServer = m_networkSetup.getLastVatsimServer();
358  ui->comp_VatsimServers->setServers(vatsimFsdServers, true);
359  ui->comp_VatsimServers->preSelect(currentServer.getName());
360  }
361  }

531  void CLoginComponent::setOwnModelAndIcaoValues(const CAircraftModel &ownModel)
532  {
533  if (!this->hasValidContexts()) { return; }
534  CAircraftModel model = ownModel;
535  const bool simulating = sGui->getIContextSimulator() &&
536  (sGui->getIContextSimulator()->getSimulatorStatus() & ISimulator::Simulating);
537 
538  // fill simulator related values
539  if (simulating)
540  {
541  if (!model.hasModelString())
542  {
543  model = sGui->getIContextOwnAircraft()->getOwnAircraft().getModel();
544  }
545  const QString modelAndKey(model.getModelStringAndDbKey());
546  ui->le_SimulatorModel->setText(modelAndKey);
547  ui->le_SimulatorModel->home(false);
548  this->highlightModelField(model);
549 
550  const CSimulatorInfo sim = sGui->getIContextSimulator()->getSimulatorPluginInfo().getSimulator();
551  const CSimulatorInternals simulatorInternals = sGui->getIContextSimulator()->getSimulatorInternals();
552  const QString simStr = sim.toQString() + QStringLiteral(" ") + simulatorInternals.getSimulatorVersion();
553  CCrashHandler::instance()->crashAndLogInfoSimulator(simStr);
554  }
555  else
556  {
557  ui->le_SimulatorModel->clear();
558  this->highlightModelField();
559  }
560  ui->le_SimulatorModel->setToolTip(model.asHtmlSummary());
561 
562  // reset the model
563  bool changedOwnAircraftIcaoValues = false;
564 
565  if (model.isLoadedFromDb() || (model.getAircraftIcaoCode().isLoadedFromDb() && model.getLivery().isLoadedFromDb()))
566  {
567  // full model from DB, take all values
568  this->setGuiIcaoValues(model, false);
569  }
570  else
571  {
572  // we have a model, which is not from DB
573  model = this->getPrefillModel(); // manually entered values
574  if (model.getLivery().hasValidDbKey() && model.getLivery().isColorLivery())
575  {
576  // special case for color liveries/NO airline
577  ui->selector_AirlineIcao->clear();
578  }
579 
580  this->setGuiIcaoValues(model, true);
