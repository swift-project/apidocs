139  bool CKeyboardWindows::init()
140  {
141  if (useWindowsHook) // cppcheck-suppress knownConditionTrueFalse
142  {

146  GetModuleHandleEx(GET_MODULE_HANDLE_EX_FLAG_FROM_ADDRESS | GET_MODULE_HANDLE_EX_FLAG_UNCHANGED_REFCOUNT, reinterpret_cast<LPCTSTR>(&CKeyboardWindows::keyboardProc), &module);
147  m_keyboardHook = SetWindowsHookEx(WH_KEYBOARD_LL, CKeyboardWindows::keyboardProc, module, 0);

151  }

155  void CKeyboardWindows::processKeyEvent(DWORD vkcode, WPARAM event)
156  {
157  CHotkeyCombination oldCombination(m_keyCombination);
158  if ((event == WM_KEYDOWN) || (event == WM_SYSKEYDOWN))
159  {
160  auto key = keyMapping().value(static_cast<int>(vkcode));
161  if (key == Key_Unknown) { return; }
162  m_keyCombination.addKeyboardKey(CKeyboardKey(key));
163  }
164  else if ((event == WM_KEYUP) || (event == WM_SYSKEYUP))
165  {
166  auto key = keyMapping().value(static_cast<int>(vkcode));
167  if (key == Key_Unknown) { return; }
168  m_keyCombination.removeKeyboardKey(CKeyboardKey(key));
169  }
170 
171  if (oldCombination != m_keyCombination)
172  {
173  emit keyCombinationChanged(m_keyCombination);
174  }
175  }

177  void CKeyboardWindows::pollKeyboardState()
178  {
179  CHotkeyCombination oldCombination(m_keyCombination);
180  QList<int> vkeys = keyMapping().keys();
181  for (int vkey : vkeys)
182  {
183  if ((GetKeyState(vkey) & 0x8000) && !m_pressedKeys.contains(vkey))
184  {
185  // key down
186  auto key = keyMapping().value(vkey);
187  if (key == Key_Unknown) { return; }
188  m_pressedKeys.push_back(vkey);
189  m_keyCombination.addKeyboardKey(CKeyboardKey(key));
190  }
191  else if (!(GetKeyState(vkey) & 0x8000) && m_pressedKeys.contains(vkey))
192  {
193  // key up
194  auto key = keyMapping().value(vkey);
195  if (key == Key_Unknown) { return; }
196  m_pressedKeys.removeAll(vkey);
197  m_keyCombination.removeKeyboardKey(CKeyboardKey(key));
198  }
199  }
200 
201  if (oldCombination != m_keyCombination)
202  {
203  emit keyCombinationChanged(m_keyCombination);
204  }
205  }

208  {
209  if (nCode == HC_ACTION)

211  KBDLLHOOKSTRUCT *keyboardEvent = reinterpret_cast<KBDLLHOOKSTRUCT *>(lParam);
