29  qint64 CAudioOutputBuffer::readData(char *data, qint64 maxlen)
30  {
31  const int sampleBytes = m_outputFormat.sampleSize() / 8;
32  const int channelCount = m_outputFormat.channelCount();
33  const qint64 count = maxlen / (sampleBytes * channelCount);
34  QVector<float> buffer;
35  m_sampleProvider->readSamples(buffer, count);
36 
37  for (float sample : std::as_const(buffer))
38  {
39  const float absSample = qAbs(sample);
40  if (absSample > m_maxSampleOutput) { m_maxSampleOutput = absSample; }
41  }
42 
43  m_sampleCount += buffer.size();
44  if (m_sampleCount >= SampleCountPerEvent)
45  {
46  OutputVolumeStreamArgs outputVolumeStreamArgs;
47  outputVolumeStreamArgs.PeakRaw = m_maxSampleOutput / 1.0;
48  outputVolumeStreamArgs.PeakDb = static_cast<float>(20 * std::log10(outputVolumeStreamArgs.PeakRaw));
49  const double db = qBound(m_minDb, outputVolumeStreamArgs.PeakDb, m_maxDb);
50  double ratio = (db - m_minDb) / (m_maxDb - m_minDb);
51  if (ratio < 0.30) { ratio = 0.0; }
52  if (ratio > 1.0) { ratio = 1.0; }
53  outputVolumeStreamArgs.PeakVU = ratio;
54  emit outputVolumeStream(outputVolumeStreamArgs);
55  m_sampleCount = 0;
56  m_maxSampleOutput = 0;
57  }
58 
59  if (channelCount == 2)
60  {
61  buffer = convertFromMonoToStereo(buffer);
62  }
63 
64  memcpy(data, buffer.constData(), static_cast<size_t>(maxlen));
65  return maxlen;
66  }

81  {

84  BLACK_VERIFY_X(outputDevice.isValid() && outputDevice.isOutputDevice(), Q_FUNC_INFO, "Wrong output device");
85

87  m_audioOutputBuffer = new CAudioOutputBuffer(sampleProvider, this);

89

97  outputFormat.setByteOrder(QAudioFormat::LittleEndian);
98  outputFormat.setCodec("audio/pcm");
99 
100  const QString format = toQString(outputFormat);
101  const QAudioDeviceInfo selectedDevice = getLowestLatencyDevice(outputDevice, outputFormat);
102  CLogMessage(this).info(u"Starting: '%1' with: %2") << selectedDevice.deviceName() << format;
103 
104  m_audioOutput.reset(new QAudioOutput(selectedDevice, outputFormat));
105  m_audioOutputBuffer->open(QIODevice::ReadWrite | QIODevice::Unbuffered);
106  m_audioOutputBuffer->setAudioFormat(outputFormat);
107  m_audioOutput->start(m_audioOutputBuffer);
