113  void CModelMatcherComponent::onSimulatorChanged(const CSimulatorInfo &simulator)
114  {
115  Q_ASSERT_X(simulator.isSingleSimulator(), Q_FUNC_INFO, "Need single simulator");
116 
117  ui->tvp_ResultMessages->clear();
118  if (this->useWorkbench())
119  {
120  const CAircraftModelList models = m_workbenchView->container();
121  if (models.isEmpty())
122  {
123  CStatusMessage m(this, CStatusMessage::SeverityWarning, u"No models in workbench, disabled.");
124  ui->tvp_ResultMessages->insert(m);
125  return;
126  }
127  CSimulatorInfo si = models.simulatorsWithMaxEntries();
128  m_matcher.setModelSet(models, si, true);
129  }
130  else
131  {
132  const CAircraftModelList models = CCentralMultiSimulatorModelSetCachesProvider::modelCachesInstance().getCachedModels(simulator);
133  m_matcher.setModelSet(models, simulator, true);
134  }
135  this->redisplay();
136  }

150  void CModelMatcherComponent::testModelMatching()
151  {
152  ui->te_Results->clear();
153  this->onSimulatorChanged(ui->comp_SimulatorSelector->getValue()); // update model set to latest version
154  CStatusMessageList msgs;
155  CSimulatedAircraft remoteAircraft(this->createAircraft());
156  m_matcher.setDefaultModel(CModelMatcherComponent::defaultModel());
157 
158  if (ui->cb_withReverseLookup->isChecked())
159  {
160  const QString liveryString(ui->comp_LiverySelector->getRawCombinedCode());
161  const CAircraftModelList modelSet = m_matcher.getModelSet();
162  const CAircraftMatcherSetup setup = m_matcher.getSetup();
163  const CAircraftModel reverseModel = CAircraftMatcher::reverseLookupModelMs(remoteAircraft.getModel(), liveryString, setup, modelSet, &msgs);
164  remoteAircraft.setModel(reverseModel); // current model
165  }
166 
167  CStatusMessageList matchingMsgs;
168  const CAircraftModel matched = m_matcher.getClosestMatch(remoteAircraft, MatchingLogAll, &matchingMsgs, true); // test model matching
169  msgs.push_back(matchingMsgs);
170 
171  ui->te_Results->setText(matched.toQString(true));
172  ui->tvp_ResultMessages->updateContainer(msgs);
173  ui->tvp_ResultMessages->fullResizeToContents();
174  ui->tvp_ResultMessages->resizeRowsToContents();
175  }

240  CSimulatedAircraft CModelMatcherComponent::createAircraft() const
241  {
242  const QString airline(ui->comp_AirlineSelector->getRawDesignator());
243  const QString aircraft(ui->comp_AircraftSelector->getRawDesignator());
244  const QString modelString(ui->le_ModelString->text().trimmed().toUpper());
245  const QString combined(ui->comp_CombinedCode->getCombinedType());
246  const QString manufacturer(ui->le_Manufacturer->text().trimmed().toUpper());
247  const QString liveryCombinedCode(ui->comp_LiverySelector->getRawCombinedCode());
248  const CCallsign cs(ui->le_Callsign->text().isEmpty() ? "SWIFT" : ui->le_Callsign->text()); // need callsign
249  static const CUser pilot("123456", "swift Test", cs);
250 
251  CAircraftIcaoCode icao(aircraft, combined);
252  icao.setManufacturer(manufacturer);
253 
254  const CAirlineIcaoCode airlineIcao(airline);
255  const CLivery livery(liveryCombinedCode, airlineIcao, "");
256  CAircraftModel m(modelString, CAircraftModel::TypeFSInnData);
257  m.setLivery(livery);
258  m.setCallsign(cs);
259  m.setModelType(modelString.isEmpty() ? CAircraftModel::TypeQueriedFromNetwork : CAircraftModel::TypeFSInnData);
260  CSimulatedAircraft sa(m);
261  sa.setPilot(pilot);
262  sa.setAircraftIcaoCode(icao);
263  return sa;
264  }

277  MatchingScriptReturnValues CModelMatcherComponent::matchingScript(const CAircraftModel &inModel, const CAircraftMatcherSetup &setup, const CAircraftModelList &modelSet, CStatusMessageList &msgs)
278  {
279  // Script
280  if (setup.doRunMsReverseLookupScript())
281  {
282  const MatchingScriptReturnValues rv = CAircraftMatcher::reverseLookupScript(inModel, setup, modelSet, &msgs);
283  if (rv.runScriptAndModified())
284  {
285  return rv;
286  }
287  else
288  {
289  CCallsign::addLogDetailsToList(&msgs, inModel.getCallsign(), QStringLiteral("Matching script, no modification"));
290  }
291  }
292  else
293  {
294  CCallsign::addLogDetailsToList(&msgs, inModel.getCallsign(), QStringLiteral("No reverse lookup script used"));
295  }
296 
297  return inModel;
298  }
