54  void CAircraftModelValidationComponent::validatedModelSet(const CSimulatorInfo &simulator, const CAircraftModelList &valid, const CAircraftModelList &invalid, bool stopped, const CStatusMessageList &msgs)
55  {
56  // BlackMisc::Simulation::CBackgroundValidation
57  Q_UNUSED(simulator)
58  Q_UNUSED(valid)
59 
60  constexpr int MsgTimeout = 15000;
61  m_lastResults = QDateTime::currentMSecsSinceEpoch();
62  ui->tvp_InvalidModels->updateContainerMaybeAsync(invalid);
63  ui->comp_Simulator->setValue(simulator);
64  ui->comp_Messages->clear();
65  if (!msgs.isEmpty()) { ui->comp_Messages->appendStatusMessagesToList(msgs); }
66 
67  // pre-select tab
68  if (invalid.isEmpty() && !msgs.isEmpty())
69  {
70  // messages but no invalid models
71  ui->tw_CAircraftModelValidationComponent->setCurrentWidget(ui->tb_Messages);
72  }
73  else
74  {
75  ui->tw_CAircraftModelValidationComponent->setCurrentWidget(ui->tb_InvalidModels);
76  }
77 
78  const CAircraftMatcherSetup setup = m_matchingSettings.get();
79  ui->cb_EnableStartupCheck->setChecked(setup.doVerificationAtStartup());
80  ui->pb_TempDisableInvalid->setEnabled(!invalid.isEmpty());
81  ui->pb_TempDisableSelected->setEnabled(!invalid.isEmpty());
82  ui->pb_RemoveInvalid->setEnabled(!invalid.isEmpty());
83  ui->pb_RemoveSelected->setEnabled(!invalid.isEmpty());
84 
85  const QString msg = stopped ?
86  QStringLiteral("Validation for '%1' stopped, maybe your models are not accessible or too many issues").arg(simulator.toQString(true)) :
87  QStringLiteral("Validated for '%1'. Valid: %2 Invalid: %3").arg(simulator.toQString(true)).arg(valid.size()).arg(invalid.size());
88  ui->lbl_Summay->setText(msg);
89  if (stopped)
90  {
91  this->showOverlayHTMLMessage(msg, MsgTimeout);
92  }
93  else if (msgs.hasWarningOrErrorMessages() || !invalid.isEmpty())
94  {
95  this->showOverlayHTMLMessage(u"There are warnings or errors, please check the messages and invalid models.", MsgTimeout);
96  }
97  else
98  {
99  this->showOverlayHTMLMessage(u"There are NO warnings or errors, your set looks good.", MsgTimeout);
100  }
101  }

183  void CAircraftModelValidationComponent::onRemoveButtonClicked()
184  {
185  if (!sGui || sGui->isShuttingDown()) { return; }
186  if (!sGui->getIContextSimulator()) { return; }
187 
188  CAircraftModelList removeModels;
189  const QObject *sender = QObject::sender();
190  if (sender == ui->pb_RemoveInvalid) { removeModels = ui->tvp_InvalidModels->container(); }
191  else if (sender == ui->pb_RemoveSelected) { removeModels = ui->tvp_InvalidModels->selectedObjects(); }
192 
193  if (removeModels.isEmpty())
194  {
195  this->showOverlayHTMLMessage("No models removed", 4000);
196  }
197  else
198  {
199  const QMessageBox::StandardButton ret = QMessageBox::question(this,
200  tr("Model validation"),
201  tr("Do you really want to delete %1 models from model set?").arg(removeModels.sizeInt()),
202  QMessageBox::Ok | QMessageBox::Cancel);
203  if (ret != QMessageBox::Ok) { return; }
204 
205  const int r = sGui->getIContextSimulator()->removeModelsFromSet(removeModels);
206  this->showOverlayHTMLMessage(QStringLiteral("%1 models removed").arg(r));
207  }
