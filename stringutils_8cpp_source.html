146  int fuzzyShortStringComparision(const QString &str1, const QString &str2, Qt::CaseSensitivity cs)
147  {
148  // same
149  if (cs == Qt::CaseInsensitive)
150  {
151  if (caseInsensitiveStringCompare(str1, str2)) { return 100; }
152  }
153  else if (str1 == str2) { return 100; }
154 
155  // one string is empty
156  if (str1.isEmpty() || str2.isEmpty()) { return 0; }
157 
158  // make sure aStr is not shorter
159  const QString aStr = str1.length() >= str2.length() ? str1 : str2;
160  const QString bStr = str1.length() >= str2.length() ? str2 : str1;
161 
162  // starts/ends with
163  const double s1 = aStr.length();
164  const double s2 = bStr.length();
165  if (aStr.endsWith(bStr, cs)) { return qRound(s1 / s2 * 100); }
166  if (aStr.startsWith(bStr, cs)) { return qRound(s1 / s2 * 100); }
167 
168  // contains
169  if (aStr.contains(bStr, cs)) { return qRound(s1 / s2 * 100); }
170 
171  // char by char
172  double points = 0;
173  for (int p = 0; p < aStr.length(); p++)
174  {
175  if (p < bStr.length() && aStr[p] == bStr[p])
176  {
177  points += 1.0;
178  continue;
179  }
180 
181  // char after
182  const int after = p + 1;
183  if (after < bStr.length() && aStr[p] == bStr[after])
184  {
185  points += 0.5;
186  continue;
187  }
188 
189  // char before
190  const int before = p - 1;
191  if (before >= 0 && before < bStr.length() && aStr[p] == bStr[before])
192  {
193  points += 0.5;
194  continue;
195  }
196  }
197  return qRound(points / s1 * 100);
198  }
