62  CLivery CLiveryForm::getValue() const
63  {
64  CLivery livery;
65  const QString id = ui->le_Id->text();
66  if (!id.isEmpty() && sGui && !sGui->isShuttingDown() && sGui->hasWebDataServices())
67  {
68  bool ok;
69  const int dbKey = id.toInt(&ok);
70  if (ok)
71  {
72  livery = sGui->getWebDataServices()->getLiveryForDbKey(dbKey);
73  }
74  }
75 
76  // fallback
77  if (!livery.hasValidDbKey())
78  {
79  livery = ui->comp_LiverySelector->getLivery();
80  }
81 
82  if (livery.hasCompleteData() && livery.hasValidDbKey())
83  {
84  // already complete data from selector
85  return livery;
86  }
87 
88  CAirlineIcaoCode airline(ui->editor_AirlineIcao->getValue());
89  livery.setAirlineIcaoCode(airline);
90  livery.setDescription(ui->le_Description->text());
91  livery.setMilitary(ui->cb_Military->isChecked());
92  return livery;
93  }

115  ui->editor_AirlineIcao->clear();
116  }
117  else

119  ui->editor_AirlineIcao->setValue(livery.getAirlineIcaoCode());
120  }

124  void CLiveryForm::jsonPasted(const QString &json)
125  {
126  if (json.isEmpty()) { return; } // avoid unnecessary conversions
127  try
128  {
129  CVariant jsonVariant;
130  jsonVariant.convertFromJson(Json::jsonObjectFromString(json));
131  if (!jsonVariant.canConvert<CLiveryList>()) { return; }
132  const CLiveryList liveries = jsonVariant.value<CLiveryList>();
133  if (!liveries.isEmpty())
134  {
135  this->setValue(liveries.front());
136  }
137  }
138  catch (const CJsonException &ex)
139  {
140  Q_UNUSED(ex);
141  }
142  }

220  void CLiveryForm::onDroppedLivery(const BlackMisc::CVariant &variantDropped)
221  {
222  CLivery livery;
223  if (variantDropped.canConvert<CLivery>())
224  {
225  livery = variantDropped.value<CLivery>();
226  }
227  else if (variantDropped.canConvert<CLiveryList>())
228  {
229  CLiveryList liveryList(variantDropped.value<CLiveryList>());
230  if (liveryList.isEmpty()) { return; }
231  livery = liveryList.front();
232  }
233  else { return; }
234  this->setValue(livery);
235  }

237  void CLiveryForm::onAirlineChanged(const CAirlineIcaoCode &code)
238  {
239  if (!sGui || sGui->isShuttingDown() || !sGui->getWebDataServices()) { return; }
240  if (!code.hasCompleteData()) { return; }
241  if (!code.hasValidDbKey()) { return; }
242 
243  // only replace with STD livery if airline does not match
244  const CLivery currentLivery = this->getValue();
245  if (currentLivery.hasValidDbKey() && currentLivery.getAirlineIcaoCode() == code) { return; }
246 
247  const CLivery stdLivery(sGui->getWebDataServices()->getLiveries().findStdLiveryByAirlineIcaoVDesignator(code));
248  if (stdLivery.hasValidDbKey())
249  {
250  this->setValue(stdLivery);
251  }
252  }
