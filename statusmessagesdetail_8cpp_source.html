102  void CStatusMessagesDetail::deferredUpdate()
103  {
104  if (m_pending.isEmpty()) { return; }
105  CStatusMessageList newMsgs(ui->tvp_StatusMessages->container());
106  newMsgs.push_front(m_pending); // default in many cases, latest first
107  m_pending.clear();
108 
109  // cleanup outdated: do not remove every time, but when a threshold is reached
110  if (m_maxLogMessages < 0)
111  {
112  // do not restrict
113  }
114  else if (m_maxLogMessages < 100 && ui->tvp_StatusMessages->rowCount() > (m_maxLogMessages + 10))
115  {
116  newMsgs.keepLatest(m_maxLogMessages);
117  }
118  else if (ui->tvp_StatusMessages->rowCount() > (m_maxLogMessages * 1.1))
119  {
120  newMsgs.keepLatest(m_maxLogMessages);
121  }
122 
123  ui->tvp_StatusMessages->adjustOrderColumn(newMsgs);
124  ui->tvp_StatusMessages->updateContainerMaybeAsync(newMsgs);
125  }
