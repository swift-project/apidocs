37  bool CDBusConnection::connect(BusType type)
38  {
39  assert(type == SessionBus);
40  DBusError error;
41  dbus_error_init(&error);
42 
43  DBusBusType dbusBusType;
44  switch (type)
45  {
46  case SessionBus: dbusBusType = DBUS_BUS_SESSION; break;
47  }
48 
49  m_connection.reset(dbus_bus_get_private(dbusBusType, &error));
50  if (dbus_error_is_set(&error))
51  {
52  m_lastError = CDBusError(&error);
53  return false;
54  }
55 
56  // Don't exit application, if the connection is disconnected
57  dbus_connection_set_exit_on_disconnect(m_connection.get(), false);
58  return true;
59  }

128  void CDBusConnection::dispatch()

160  DBusHandlerResult CDBusConnection::filterDisconnectedFunction(DBusConnection *connection, DBusMessage *message, void *data)
161  {
162  (void)connection; // unused
163 
164  auto *obj = static_cast<CDBusConnection *>(data);
165 
166  DBusError err;
167  dbus_error_init(&err);
168 
169  if (dbus_message_is_signal(message, DBUS_INTERFACE_LOCAL, "Disconnected"))
170  {
171  for (auto it = obj->m_disconnectedCallbacks.begin(); it != obj->m_disconnectedCallbacks.end(); ++it)
172  {
173  it->second();
174  }
175  return DBUS_HANDLER_RESULT_HANDLED;
176  }
177  return DBUS_HANDLER_RESULT_NOT_YET_HANDLED;
