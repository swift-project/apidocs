48  int CCallsign::suffixToSortOrder(const QString &suffix)
49  {
50  if (QStringView(u"FSS") == suffix) { return 1; }
51  if (QStringView(u"CTR") == suffix) { return 2; }
52  if (QStringView(u"APP") == suffix) { return 3; }
53  if (QStringView(u"DEP") == suffix) { return 4; }
54  if (QStringView(u"TWR") == suffix) { return 5; }
55  if (QStringView(u"GND") == suffix) { return 6; }
56  if (QStringView(u"DEL") == suffix) { return 7; }
57  if (QStringView(u"ATIS") == suffix) { return 8; }
58  if (QStringView(u"SUP") == suffix) { return 9; }
59  if (QStringView(u"OBS") == suffix) { return 10; }
60  if (QStringView(u"INS") == suffix) { return 11; } // instructor/mentor
61  if (QStringView(u"ADM") == suffix) { return 12; } // admin
62  if (QStringView(u"VATGOV") == suffix) { return 13; } // VATSIM governors
63  if (QStringView(u"VATSIM") == suffix) { return 14; } // VATSIM founder
64  if (QStringView(u"EXAM") == suffix) { return 15; }
65  return std::numeric_limits<int>::max();
66  }

68  QString CCallsign::unifyCallsign(const QString &callsign, TypeHint hint)
69  {
70  const QString ucCallsign = callsign.trimmed().toUpper();
71 
72  // Ref T664, allow ATC with hyphen, such as Ml-SNO_CTR
73  switch (hint)
74  {
75  // ATC allows "-", aircraft not
76  case Atc: return removeChars(ucCallsign, [](QChar c) { return !c.isLetterOrNumber() && c != '_' && c != '-'; });
77  case Aircraft: return removeChars(ucCallsign, [](QChar c) { return !c.isLetterOrNumber() && c != '_'; });
78  default: break;
79  }
80 
81  // no hint
82  if (CCallsign::looksLikeAtcCallsign(ucCallsign))
83  {
84  return removeChars(ucCallsign, [](QChar c) { return !c.isLetterOrNumber() && c != '_' && c != '-'; });
85  }
86 
87  // strict check
88  return removeChars(ucCallsign, [](QChar c) { return !c.isLetterOrNumber() && c != '_'; });
89  }

239  QString CCallsign::getAirlinePrefix(QString &flightNumber, QString &flightIdentification) const
240  {
241  // DLH1WP only 11 number
242  // UPSE123 4 characters, then number
243 
244  flightNumber.clear();
245  if (m_callsign.length() < 3) { return {}; }
246  if (this->isAtcCallsign()) { return {}; }
247 
248  thread_local const QRegularExpression regExp("(^[A-Z]{3,})(\\d+)");
249  const QRegularExpressionMatch match = regExp.match(m_callsign);
250  if (!match.hasMatch()) { return {}; }
251  // 0 is whole capture
252  const QString airline = match.captured(1);
253  flightNumber = match.captured(2); // null string if not exits
254 
255  // hard facts
256  if (airline.length() == 3) // we allow 3 letters
257  {
258  flightIdentification = m_callsign.length() > 3 ? m_callsign.mid(3) : QString();
259  return airline;
260  }
261  if (airline.length() == 4 && airline.startsWith('V')) // we allow virtual 4 letter codes, e.g. VDLD
262  {
263  flightIdentification = m_callsign.length() > 4 ? m_callsign.mid(4) : QString();
264  return airline;
265  }
266 
267  // some people use callsigns like UPSE123
268  if (flightNumber.length() >= 1 && airline.length() == 4)
269  {
270  flightIdentification = m_callsign.mid(3);
271  return airline.left(3);
272  }
273 
274  return {}; // invalid
275  }

358  {
359  if (index.isMyself()) { return m_callsign.compare(compareValue.m_callsign, Qt::CaseInsensitive); }
360  const ColumnIndex i = index.frontCasted<ColumnIndex>();
361  switch (i)

367  default:
368  return CValueObject::comparePropertyByIndex(index, compareValue);

370  Q_ASSERT_X(false, Q_FUNC_INFO, "Compare failed");

420  {
