153  CSimulatedAircraft CCockpitComForm::cockpitValuesToAircraftObject()
154  {
155  CSimulatedAircraft comAircraft;
156  CTransponder transponder = comAircraft.getTransponder();
157  CComSystem com1 = comAircraft.getCom1System();
158  CComSystem com2 = comAircraft.getCom2System();
159 
160  //
161  // Transponder
162  //
163  const QString transponderCode = QString::number(ui->sbp_ComPanelTransponder->value());
164  if (CTransponder::isValidTransponderCode(transponderCode))
165  {
166  transponder.setTransponderCode(transponderCode);
167  }
168  else
169  {
170  CLogMessage(this).validationWarning(u"Wrong transponder code, reset");
171  ui->sbp_ComPanelTransponder->setValue(transponder.getTransponderCode());
172  }
173  transponder.setTransponderMode(ui->cbp_ComPanelTransponderMode->getSelectedTransponderMode());
174 
175  //
176  // COM units
177  //
178  com1.setFrequencyActiveMHz(ui->ds_ComPanelCom1Active->value());
179  com1.setFrequencyStandbyMHz(ui->ds_ComPanelCom1Standby->value());
180  com2.setFrequencyActiveMHz(ui->ds_ComPanelCom2Active->value());
181  com2.setFrequencyStandbyMHz(ui->ds_ComPanelCom2Standby->value());
182  this->setFrequencies(com1, com2); // back annotation after rounding
183 
184  comAircraft.setCom1System(com1);
185  comAircraft.setCom2System(com2);
186  comAircraft.setTransponder(transponder);
187  return comAircraft;
188  }

190  void CCockpitComForm::setFrequencies(const CComSystem &com1, const CComSystem &com2)
191  {
192  double freq = com1.getFrequencyActive().valueRounded(CFrequencyUnit::MHz(), 3);
193  if (!isFrequenceEqual(freq, ui->ds_ComPanelCom1Active->value()))
194  {
195  ui->ds_ComPanelCom1Active->setValue(freq);
196  }
197 
198  freq = com2.getFrequencyActive().valueRounded(CFrequencyUnit::MHz(), 3);
199  if (!isFrequenceEqual(freq, ui->ds_ComPanelCom2Active->value()))
200  {
201  ui->ds_ComPanelCom2Active->setValue(freq);
202  }
203 
204  freq = com1.getFrequencyStandby().valueRounded(CFrequencyUnit::MHz(), 3);
205  if (!isFrequenceEqual(freq, ui->ds_ComPanelCom1Standby->value()))
206  {
207  ui->ds_ComPanelCom1Standby->setValue(freq);
208  }
209 
210  freq = com2.getFrequencyStandby().valueRounded(CFrequencyUnit::MHz(), 3);
211  if (!isFrequenceEqual(freq, ui->ds_ComPanelCom2Standby->value()))
212  {
213  ui->ds_ComPanelCom2Standby->setValue(freq);
214  }
215 
216  this->updateIntegratedFlagFromSimulatorContext();
217  this->updateActiveCOMUnitLEDs(m_integratedWithSim, com1.isTransmitEnabled(), com1.isReceiveEnabled(), com2.isTransmitEnabled(), com2.isReceiveEnabled());
218  }

235  void CCockpitComForm::onGuiChangedCockpitValues()
236  {
237  const QObject *sender = QObject::sender();
238  if (sender == ui->tb_ComPanelCom1Toggle)
239  {
240  if (isFrequenceEqual(ui->ds_ComPanelCom1Standby->value(), ui->ds_ComPanelCom1Active->value())) { return; }
241  const double f = ui->ds_ComPanelCom1Active->value();
242  ui->ds_ComPanelCom1Active->setValue(ui->ds_ComPanelCom1Standby->value());
243  ui->ds_ComPanelCom1Standby->setValue(f);
244  }
245  else if (sender == ui->tb_ComPanelCom2Toggle)
246  {
247  if (isFrequenceEqual(ui->ds_ComPanelCom2Standby->value(), ui->ds_ComPanelCom2Active->value())) { return; }
248  const double f = ui->ds_ComPanelCom2Active->value();
249  ui->ds_ComPanelCom2Active->setValue(ui->ds_ComPanelCom2Standby->value());
250  ui->ds_ComPanelCom2Standby->setValue(f);
251  }
252 
253  this->alignUiElementsHeight();
254  const CSimulatedAircraft aircraft = this->cockpitValuesToAircraftObject();
255  emit this->changedCockpitValues(aircraft);
256  }
