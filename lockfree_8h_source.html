159  operator bool()
160  {
161  Q_ASSERT_X(m_ptr.use_count() > 0, Q_FUNC_INFO, "SharedWriter tried to commit changes twice");
162  if (std::atomic_compare_exchange_strong(m_now, &m_old, std::shared_ptr<const T>(m_ptr)))
163  {
164  m_ptr.reset();
165  return true;
166  }
167  QThread::msleep(1);
168  m_old = std::atomic_load(m_now);
169  m_ptr = std::make_shared<T>(*m_old);
170  return false;
171  }
