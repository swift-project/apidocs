27  QString CVPilotModelRule::getDistributor() const
28  {
29  QString f(this->getFolder().toUpper().simplified());
30  f.replace(" ", "");
31  if (f.isEmpty()) return ("UNKNOWN");
32  if (f.startsWith("WOAI", Qt::CaseInsensitive)) { return "WOAI"; }
33  if (f.startsWith("WORLDOFAI", Qt::CaseInsensitive)) { return "WOAI"; }
34  if (f.startsWith("IVAO", Qt::CaseInsensitive)) { return "IVAO"; }
35  if (f.startsWith("P3D", Qt::CaseInsensitive)) { return "P3D"; }
36  if (f.startsWith("FSX", Qt::CaseInsensitive)) { return "FSX"; }
37  if (f.startsWith("MYTRAFFIC", Qt::CaseInsensitive)) { return "MYTRAFFIC"; }
38  if (f.startsWith("JUSTFLIGHT", Qt::CaseInsensitive)) { return "JUSTFLIGHT"; }
39  if (f.startsWith("ULTIMATETRAFFIC", Qt::CaseInsensitive)) { return "ULTIMATETRAFFIC"; }
40  if (f.startsWith("VIP", Qt::CaseInsensitive)) { return "VIP"; }
41  return "??? - " + f;
42  }

92  CAircraftModel CVPilotModelRule::toAircraftModel() const
93  {
94  QString al(m_callsignPrefix);
95  if (al.length() > 3)
96  {
97  // some known hardcoded fixes
98  if (al.startsWith("USAF")) { al = "AIO"; }
99  }
100  const QString liveryPseudoCode(
101  al.length() != 3 ?
102  "" :
103  al + "." + CLivery::standardLiveryMarker());
104  const CAircraftIcaoCode aircraftIcao(m_typeCode);
105  const CAirlineIcaoCode airlineIcao(al);
106  const CLivery livery(liveryPseudoCode, airlineIcao, "vPilot rule based");
107  const CDistributor distributor(getDistributor(), "vPilot based", "", "");
108  CAircraftModel model(
109  this->m_modelName, CAircraftModel::TypeVPilotRuleBased,
110  CAircraftModel::autoGenerated(),
111  aircraftIcao, livery);
112  const CSimulatorInfo sim(CSimulatorInfo::FSX_P3D);
113  model.setMSecsSinceEpoch(m_timestampMSecsSinceEpoch);
114  model.setDistributor(distributor);
115  model.setSimulator(sim);
116  return model;
117  }
