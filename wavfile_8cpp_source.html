72  bool CWavFile::readHeader()
73  {
74  seek(0);
75  CombinedHeader header;
76  DATAHeader dataHeader;
77  bool result = read(reinterpret_cast<char *>(&header), sizeof(CombinedHeader)) == sizeof(CombinedHeader);
78  if (result)
79  {
80  if ((memcmp(&header.riff.descriptor.id, "RIFF", 4) == 0 || memcmp(&header.riff.descriptor.id, "RIFX", 4) == 0) && memcmp(&header.riff.type, "WAVE", 4) == 0 && memcmp(&header.wave.descriptor.id, "fmt ", 4) == 0 && (header.wave.audioFormat == 1 || header.wave.audioFormat == 0 || header.wave.audioFormat == 3))
81  {
82  // Read off remaining header information
83  if (qFromLittleEndian<quint32>(header.wave.descriptor.size) > sizeof(WAVEHeader))
84  {
85  // Extended data available
86  quint16 extraFormatBytes;
87  if (peek((char *)&extraFormatBytes, sizeof(quint16)) != sizeof(quint16))
88  return false;
89  const qint64 throwAwayBytes = sizeof(quint16) + qFromLittleEndian<quint16>(extraFormatBytes);
90  if (read(throwAwayBytes).size() != throwAwayBytes)
91  return false;
92  }
93 
94  if (read((char *)&dataHeader, sizeof(DATAHeader)) != sizeof(DATAHeader))
95  return false;
96 
97  // Establish format
98  if (memcmp(&header.riff.descriptor.id, "RIFF", 4) == 0)
99  m_fileFormat.setByteOrder(QAudioFormat::LittleEndian);
100  else
101  m_fileFormat.setByteOrder(QAudioFormat::BigEndian);
102 
103  int bps = qFromLittleEndian<quint16>(header.wave.bitsPerSample);
104  m_fileFormat.setChannelCount(qFromLittleEndian<quint16>(header.wave.numChannels));
105  m_fileFormat.setCodec("audio/pcm");
106  m_fileFormat.setSampleRate(qFromLittleEndian<quint32>(header.wave.sampleRate));
107  m_fileFormat.setSampleSize(qFromLittleEndian<quint16>(header.wave.bitsPerSample));
108 
109  if (header.wave.audioFormat == 1 || header.wave.audioFormat == 0)
110  {
111  m_fileFormat.setSampleType(bps == 8 ? QAudioFormat::UnSignedInt : QAudioFormat::SignedInt);
112  }
113  else
114  {
115  m_fileFormat.setSampleType(QAudioFormat::Float);
116  }
117  }
118  else
119  {
120  result = false;
121  }
122  }
123  m_headerLength = pos();
124 
125  if (memcmp(&dataHeader.descriptor.id, "data", 4) == 0)
126  {
127  const qint32 dataLength = qFromLittleEndian<qint32>(dataHeader.descriptor.size);
128  m_audioData = read(dataLength);
129  if (m_audioData.size() != dataLength)
130  {
131  m_audioData.clear();
132  return false;
133  }
134  }
135  return result;
136  }
