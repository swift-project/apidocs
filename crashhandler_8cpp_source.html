52  void CCrashHandler::init()
53  {
54 #ifdef BLACK_USE_CRASHPAD
55  static const QString crashpadHandler(CBuildConfig::isRunningOnWindowsNtPlatform() ? "swift_crashpad_handler.exe" : "swift_crashpad_handler");
56  static const QString handler = CFileUtils::appendFilePaths(CSwiftDirectories::binDirectory(), crashpadHandler);
57  const QString database = CSwiftDirectories::crashpadDatabaseDirectory();
58  const QString metrics = CSwiftDirectories::crashpadMetricsDirectory();
59 
60  if (!QFileInfo::exists(handler)) { return; }
61 
62  const std::string serverUrl("http://swift-project.sp.backtrace.io:6097/");
63  std::map<std::string, std::string> annotations;
64 
65  // Backtrace annotations
66  annotations["token"] = CBuildConfig::backtraceToken().toStdString();
67  annotations["format"] = "minidump";
68  annotations["commit"] = CBuildConfig::gitHeadSha1().toStdString();
69  annotations["version"] = CBuildConfig::getVersionString().toStdString();
70  annotations["short_version"] = CBuildConfig::getShortVersionString().toStdString();
71  annotations["platform"] = CBuildConfig::getPlatformString().toStdString();
72  annotations["qtversion"] = QT_VERSION_STR;
73 
74  // add our logfile
75  const QString logAttachment = QStringLiteral("--attachment=attachment_%1=%2").arg(CFileLogger::getLogFileName(), CFileLogger::getLogFilePath());
76 
77  std::vector<std::string> arguments;
78  arguments.push_back(logAttachment.toStdString());
79 
80  // and the simplified crash info if any
81  const QString crashInfoFileName("swiftcrashinfo.txt");
82  const QString crashInfoFilePath(CFileUtils::appendFilePaths(CFileUtils::stripFileFromPath(CFileLogger::getLogFilePath()), crashInfoFileName));
83  m_crashAndLogInfo.setLogPathAndFileName(crashInfoFilePath);
84  const QString crashAttachment = QStringLiteral("--attachment=attachment_%1=%2").arg(crashInfoFileName, crashInfoFilePath);
85  arguments.push_back(crashAttachment.toStdString());
86 
87  // for testing purposes
88  if (CBuildConfig::isLocalDeveloperDebugBuild())
89  {
90  arguments.push_back("--no-rate-limit");
91  }
92 
93  QDir().mkpath(database);
94 
95  m_crashReportDatabase = CrashReportDatabase::Initialize(qstringToFilePath(database));
96  m_crashpadClient = std::make_unique<CrashpadClient>();
97  m_crashpadClient->StartHandler(qstringToFilePath(handler),
98  qstringToFilePath(database),
99  qstringToFilePath(metrics),
100  serverUrl,
101  annotations,
102  arguments,
103  false, true);
104 
105  this->crashAndLogAppendInfo(u"Init crash info at " % QDateTime::currentDateTimeUtc().toString());
106 #endif
107  }
