119  void CDistributorForm::onDroppedCode(const BlackMisc::CVariant &variantDropped)
120  {
121  CDistributor distributor;
122  if (variantDropped.canConvert<CDistributor>())
123  {
124  distributor = variantDropped.value<CDistributor>();
125  }
126  else if (variantDropped.canConvert<CDistributorList>())
127  {
128  const CDistributorList icaoList(variantDropped.value<CDistributorList>());
129  if (icaoList.isEmpty()) { return; }
130  distributor = icaoList.front();
131  }
132  else
133  {
134  return;
135  }
136  this->setValue(distributor);
137  }
