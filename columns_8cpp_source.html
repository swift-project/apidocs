230  QList<int> CColumns::calculateWidths(int totalWidth) const
231  {
232  if (m_columns.isEmpty() || !this->hasAnyWidthPercentage()) { return {}; }
233 
234  int totalPercentage = 0;
235  const int averagePercentage = 100 / m_columns.size();
236 
237  for (const CColumn &c : m_columns)
238  {
239  totalPercentage += c.hasWidthPercentage() ? c.getWidthPercentage() : averagePercentage;
240  }
241 
242  if (totalPercentage < 1) { return {}; }
243 
244  // ideally totalPercentage would be 100%, but there is no guarantee
245  const double part = static_cast<double>(totalWidth) / totalPercentage;
246  QList<int> widths;
247 
248  int usedWidth = 0;
249  for (const CColumn &c : m_columns)
250  {
251  const int percentage = c.hasWidthPercentage() ? c.getWidthPercentage() : averagePercentage;
252  const int restWidth = totalWidth - usedWidth;
253  const int width = qMin(restWidth, qRound(part * percentage));
254  widths.push_back(width);
255  usedWidth += width;
256  }
257 
258  return widths;
259  }
