82  bool CDataInfoAreaComponent::writeDbDataToResourceDir() const
83  {
84  if (!sGui || !sGui->getWebDataServices()->hasSuccesfullyConnectedSwiftDb())
85  {
86  CLogMessage(this).warning(u"No connection to DB yet, no new data loaded which can be written");
87  return false;
88  }
89 
90  // write to disk
91  const bool s = sGui->getWebDataServices()->writeDbDataToDisk(CSwiftDirectories::staticDbFilesDirectory());
92  if (s)
93  {
94  CLogMessage(this).info(u"Written DB data");
95  }
96  else
97  {
98  CLogMessage(this).error(u"Cannot write DB data");
99  }
100  return s;
101  }

103  bool CDataInfoAreaComponent::readDbDataFromResourceDir()
104  {
105  if (!sGui) { return false; }
106  const CStatusMessageList msgs = sGui->getWebDataServices()->initDbCachesFromLocalResourceFiles(true);
107 
108  // info
109  bool ok = false;
110  if (msgs.isSuccess())
111  {
112  CLogMessage(this).info(u"Read DB data from directory: %1") << CSwiftDirectories::staticDbFilesDirectory();
113  ui->comp_DbAircraftIcao->showLoadIndicator();
114  ui->comp_DbAirlineIcao->showLoadIndicator();
115  ui->comp_DbCountries->showLoadIndicator();
116  ui->comp_DbDistributors->showLoadIndicator();
117  ui->comp_DbLiveries->showLoadIndicator();
118  ui->comp_DbModels->showLoadIndicator();
119  ok = true;
120  }
121  else
122  {
123  CLogMessage::preformatted(msgs);
124  }
125  return ok;
126  }
