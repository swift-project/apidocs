163  void CRawFsdMessagesComponent::filterDisplayedMessages()
164  {
165  CRawFsdMessageList filtered = m_buffer;
166  if (!m_filterString.isEmpty()) { filtered = filtered.findByContainsString(m_filterString); }
167  if (!m_filterPacketType.isEmpty()) { filtered = filtered.findByPacketType(m_filterPacketType); }
168  ui->pte_RawFsdMessages->clear();
169  // Append only the last messages up to maximum display size. Erase everything before.
170  filtered.erase(filtered.begin(), filtered.end() - std::min(filtered.size(), m_maxDisplayedMessages));
171  for (const CRawFsdMessage &rawFsdMessage : filtered)
172  {
173  ui->pte_RawFsdMessages->appendPlainText(rawFsdMessageToString(rawFsdMessage));
174  }
175  }

192  void CRawFsdMessagesComponent::addFsdMessage(const CRawFsdMessage &rawFsdMessage)
193  {
194  if (m_buffer.size() == m_maxDisplayedMessages) { m_buffer.pop_front(); }
195  m_buffer.push_backMaxElements(rawFsdMessage, m_maxDisplayedMessages);
196 
197  if (!m_filterPacketType.isEmpty() && !rawFsdMessage.isPacketType(m_filterPacketType)) { return; }
198  if (!m_filterString.isEmpty() && !rawFsdMessage.containsString(m_filterString)) { return; }
199  ui->pte_RawFsdMessages->appendPlainText(rawFsdMessageToString(rawFsdMessage));
200  }
