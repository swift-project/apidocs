65  void CXSwiftBusConfigWriter::writeTo(const QString &filePath)
66  {
67  // writing the file here (UI side) allows to overwrite the values
68  // before XPlane is started
69  // TCAS/log values MUST be changed before startup of the plugin
70  QString configFilePath = filePath + "/xswiftbus.conf";
