35  void CAircraftCategoryTreeModel::updateContainer(const CAircraftCategoryList &categories)
36  {
37  Q_ASSERT_X(CThreadUtils::isInThisThread(this), Q_FUNC_INFO, "Wrong thread");
38 
39  this->clear();
40  if (categories.isEmpty()) { return; }
41 
42  m_categories = categories;
43  m_categories.sortByLevel();
44  QMap<int, QStandardItem *> items;
45  this->setColumnCount(m_columns.size() + 1);
46 
47  for (const CAircraftCategory &category : m_categories)
48  {
49  QList<QStandardItem *> categoryRow;
50 
51  // ownership of QStandardItem is taken by model
52  QStandardItem *si = new QStandardItem(
53  category.isAssignable() ? CIcons::paperPlane16() : CIcons::folder16(),
54  category.getLevelAndName());
55  si->setEditable(false);
56  categoryRow.push_back(si);
57 
58  // add all clumns
59  for (const CColumn &column : m_columns.columns())
60  {
61  const CPropertyIndex i(column.getPropertyIndex());
62  const CVariant v(category.propertyByIndex(i));
63  const CVariant f = column.getFormatter()->displayRole(v);
64  si = new QStandardItem(f.toQString(true));
65  si->setEditable(false); // make not editable
66  categoryRow.push_back(si);
67  } // columns
68 
69  // add all items
70  if (categoryRow.isEmpty()) { continue; }
71 
72  QStandardItem *parent = categoryRow.first();
73  if (category.isFirstLevel())
74  {
75  this->invisibleRootItem()->appendRow(categoryRow);
76  }
77  else
78  {
79  const int p = category.getDepth() - 1;
80  Q_ASSERT_X(items[p], Q_FUNC_INFO, "No parent item");
81  items[p]->appendRow(categoryRow);
82  }
83  items.insert(category.getDepth(), parent);
84  }
85  }
