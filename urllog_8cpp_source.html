34  QVariant CUrlLog::propertyByIndex(CPropertyIndexRef index) const
35  {
36  if (index.isMyself()) { return QVariant::fromValue(*this); }
37  if (ITimestampBased::canHandleIndex(index)) { return ITimestampBased::propertyByIndex(index); }

39  const ColumnIndex i = index.frontCasted<ColumnIndex>();

42  case IndexId: return QVariant::fromValue(m_id);
43  case IndexSuccess: return QVariant::fromValue(m_success);
44  case IndexUrl: return this->m_url.propertyByIndex(index.copyFrontRemoved());
45  case IndexResponseTimestamp: return QVariant::fromValue(this->getResponseTimestamp());

47  default: return CValueObject::propertyByIndex(index);
48  }

51  void CUrlLog::setPropertyByIndex(CPropertyIndexRef index, const QVariant &variant)
52  {
53  if (index.isMyself())
54  {
55  (*this) = variant.value<CUrlLog>();
56  return;
57  }
58  if (ITimestampBased::canHandleIndex(index))
59  {
60  ITimestampBased::setPropertyByIndex(index, variant);
61  return;
62  }
63 
64  const ColumnIndex i = index.frontCasted<ColumnIndex>();
65  switch (i)
66  {
67  case IndexId: m_id = variant.toInt(); break;
68  case IndexSuccess: m_success = variant.toBool(); break;
69  case IndexUrl: m_url.setPropertyByIndex(index.copyFrontRemoved(), variant); break;
70  case IndexResponseTime: this->setResponseTimestampToNow(); break; // a bit unusual
71  default: CValueObject::setPropertyByIndex(index, variant); break;
72  }
