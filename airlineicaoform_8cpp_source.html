55  void CAirlineIcaoForm::setValue(const CAirlineIcaoCode &icao)
56  {
57  if (m_currentCode == icao) { return; }
58  m_currentCode = icao;
59 
60  ui->selector_AirlineDesignator->setAirlineIcao(icao);
61  ui->selector_AirlineName->setAirlineIcao(icao);
62  ui->le_Id->setText(icao.getDbKeyAsString());
63  ui->le_TelephonyDesignator->setText(icao.getTelephonyDesignator());
64  ui->le_Updated->setText(icao.getFormattedUtcTimestampYmdhms());
65  ui->cb_Va->setChecked(icao.isVirtualAirline());
66  ui->cb_Military->setChecked(icao.isMilitary());
67  ui->country_Selector->setCountry(icao.getCountry());
68 
69  const QPixmap pm = CIcon(icao.toIcon());
70  if (pm.width() < 125)
71  {
72  ui->lbl_AirlineIcon->setPixmap(pm);
73  }
74  else
75  {
76  ui->lbl_AirlineIcon->setTextFormat(Qt::RichText);
77  ui->lbl_AirlineIcon->setText(CGuiUtility::asSimpleHtmlImageHeight(icao.toIcon(), 25));
78  }
79 
80  // sometimes artefacts when icon is displayed
81  this->repaint();
82 
83  if (m_currentCode.hasCompleteData())
84  {
85  emit this->airlineChanged(m_currentCode);
86  }
87  }

89  CAirlineIcaoCode CAirlineIcaoForm::getValue() const
90  {
91  CAirlineIcaoCode code;
92  const QString id = ui->le_Id->text();
93  if (sGui && !sGui->isShuttingDown() && sGui->hasWebDataServices())
94  {
95  bool ok;
96  const int dbKey = id.toInt(&ok);
97  if (ok)
98  {
99  code = sGui->getWebDataServices()->getAirlineIcaoCodeForDbKey(dbKey);
100  }
101  }
102 
103  if (code.hasValidDbKey()) { return code; }
104  code = m_currentCode;
105  code.setVirtualAirline(ui->cb_Va->isChecked());
106  code.setMilitary(ui->cb_Military->isChecked());
107  code.setCountry(ui->country_Selector->getCountry());
108  code.setName(ui->selector_AirlineName->getAirlineIcao().getName());
109  code.setTelephonyDesignator(ui->le_TelephonyDesignator->text());
110  return code;
111  }

174  void CAirlineIcaoForm::jsonPasted(const QString &json)
175  {
176  if (json.isEmpty()) { return; } // avoid unnecessary conversions
177  try
178  {
179  CVariant jsonVariant;
180  jsonVariant.convertFromJson(Json::jsonObjectFromString(json));
181  if (!jsonVariant.canConvert<CAirlineIcaoCodeList>()) { return; }
182  const CAirlineIcaoCodeList icaos = jsonVariant.value<CAirlineIcaoCodeList>();
183  if (!icaos.isEmpty())
184  {
185  this->setValue(icaos.front());
186  }
187  }
188  catch (const CJsonException &ex)
189  {
190  Q_UNUSED(ex);
191  }
192  }

194  void CAirlineIcaoForm::onDroppedCode(const BlackMisc::CVariant &variantDropped)
195  {
196  CAirlineIcaoCode icao;
197  if (variantDropped.canConvert<CAirlineIcaoCode>())
198  {
199  icao = variantDropped.value<CAirlineIcaoCode>();
200  }
201  else if (variantDropped.canConvert<CAirlineIcaoCodeList>())
202  {
203  const CAirlineIcaoCodeList icaoList(variantDropped.value<CAirlineIcaoCodeList>());
204  if (icaoList.isEmpty()) { return; }
205  icao = icaoList.front();
206  }
207  else
208  {
209  return;
210  }
211  this->setValue(icao);
212  }
