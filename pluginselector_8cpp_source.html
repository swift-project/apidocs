30  void CPluginSelector::addPlugin(const QString &identifier, const QString &name, bool hasConfig, bool enabled)
31  {
32  // skip if identifier is missing, which should normally not happen
33  BLACK_VERIFY_X(!identifier.isEmpty(), Q_FUNC_INFO, "Missing identifier");
34  if (identifier.isEmpty()) { return; }
35 
36  QWidget *pw = new QWidget;
37  QHBoxLayout *layout = new QHBoxLayout;
38  layout->setContentsMargins(0, 0, 0, 0);
39  pw->setLayout(layout);
40 
41  QCheckBox *cb = new QCheckBox(name);
42  cb->setObjectName(identifier);
43  cb->setProperty("pluginIdentifier", identifier);
44  connect(cb, &QCheckBox::stateChanged, this, &CPluginSelector::handlePluginStateChange);
45  if (enabled)
46  {
47  cb->setCheckState(Qt::Checked);
48  }
49  else
50  {
51  cb->setCheckState(Qt::Unchecked);
52  }
53 
54  pw->layout()->addWidget(cb);
55 
56  if (hasConfig)
57  {
58  QPushButton *config = new QPushButton(BlackMisc::CIcons::wrench16(), "");
59  config->setToolTip("Plugin configuration");
60  m_configButtonMapper->setMapping(config, identifier);
61  connect(config, &QPushButton::clicked, m_configButtonMapper, qOverload<>(&QSignalMapper::map));
62  pw->layout()->addWidget(config);
63  }
64 
65  QPushButton *details = new QPushButton("?");
66  details->setToolTip("Plugin details");
67  m_detailsButtonMapper->setMapping(details, identifier);
68  connect(details, &QPushButton::clicked, m_detailsButtonMapper, qOverload<>(&QSignalMapper::map));
69  pw->layout()->addWidget(details);
70 
71  layout->setStretch(0, 1);
72  layout->setStretch(1, 0);
73  layout->setStretch(2, 0);
74 
75  this->layout()->addWidget(pw);
