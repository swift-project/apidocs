98  QString CTextMessageTextEdit::toHtml(const CTextMessageList &messages, bool withFrom, bool withTo)
99  {
100  if (messages.isEmpty()) { return {}; }
101  QString html("<table>");
102  for (const CTextMessage &msg : messages)

110  QString CTextMessageTextEdit::toHtml(const CTextMessage &message, bool withFrom, bool withTo)
111  {
112  QString rowClass(message.wasSent() ? "sent" : "received");
113  if (message.isSupervisorMessage()) { rowClass += " supervisor"; }
114 
115  QString html(u"<tr class=\"" % rowClass.trimmed() %
116  u"\"><td class=\"timestamp\">" % message.getFormattedUtcTimestampHms() % u"</td>");
117 
118  if (withFrom)
119  {
120  html += u"<td class=\"sender\">" % message.getSenderCallsign().asString() % u"</td>";
121  }
122 
123  if (withTo)
124  {
125  html += u"<td class=\"recipient\">" % message.getRecipientCallsignOrFrequency() % u"</td>";
126  }
127 
128  html += u"<td class=\"message\">" % message.getHtmlEncodedMessage() % u"</td></tr>";
129  return html;
130  }
