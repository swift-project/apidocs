114  void CTestPhysicalQuantities::lengthBasics()
115  {
116  CLength d1(1, CLengthUnit::m()); // 1m
117  CLength d2(100, CLengthUnit::cm());
118  CLength d3(1.852 * 1000, CLengthUnit::m()); // 1852m
119  CLength d4(1, CLengthUnit::NM());
120  QVERIFY2(d1 == d2, "1meter shall be 100cm");
121  QVERIFY2(d3 == d4, "1852meters shall be 1NM");
122  QVERIFY2(d1 * 2 == 2 * d1, "Commutative multiplication");
123 
124  d3 *= 2; // SI value
125  d4 *= 2.0; // SI value !
126  QVERIFY2(d3 == d4, "2*1852meters shall be 2NM");
127 
128  // less / greater
129  QVERIFY2(!(d1 < d2), "Nothing shall be less / greater");
130  QVERIFY2(!(d1 > d2), "Nothing shall be less / greater");
131 
132  // epsilon tests
133  d1 = d2; // both in same unit
134  d1.addValueSameUnit(d1.getUnit().getEpsilon() / 2.0); // this should be still the same
135  QVERIFY2(d1 == d2, "Epsilon: 100cm + epsilon shall be 100cm");
136  QVERIFY2(!(d1 != d2), "Epsilon: 100cm + epsilon shall be still 100cm");
137  QVERIFY2(!(d1 > d2), "d1 shall not be greater");
138 
139  d1.addValueSameUnit(d1.getUnit().getEpsilon()); // now over epsilon threshold
140  QVERIFY2(d1 != d2, "Epsilon exceeded: 100 cm + 2 epsilon shall not be 100cm");
141  QVERIFY2(d1 > d2, "d1 shall be greater");
142  }
