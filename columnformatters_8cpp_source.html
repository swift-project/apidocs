47  CVariant CDefaultFormatter::decorationRole(const CVariant &dataCVariant) const
48  {
49  // direct return if type is already correct
50  const QMetaType::Type type = static_cast<QMetaType::Type>(dataCVariant.type());
51 
52  if (type == QMetaType::QPixmap) { return dataCVariant; }
53  if (type == QMetaType::QIcon) { return dataCVariant; }
54 
55  // convert to pixmap

57  {

59  return CVariant::from(QPixmap::fromImage(img));

61 
62  // Our CIcon class
63  if (dataCVariant.canConvert<CIcon>())
64  {
65  const CIcon i = dataCVariant.value<CIcon>();
66  return CVariant::from(i.toPixmap());
67  }
68 
69  // nope
70  return CVariant::from(QPixmap());
71  }

98  CVariant CDefaultFormatter::data(int role, const CVariant &inputData) const
99  {
100  if (!this->supportsRole(role)) { return CVariant(); }
101  const Qt::ItemDataRole roleEnum = static_cast<Qt::ItemDataRole>(role);
102 
103  // always supported
104  if (roleEnum == Qt::TextAlignmentRole) return { alignmentRole() };
105 
106  // check
107  if (role == Qt::UserRole) { return CDefaultFormatter::displayRole(inputData); } // just as data provider
108  switch (roleEnum)
109  {
110  case Qt::DisplayRole: return displayRole(inputData); // formatted to standard types or string
111  case Qt::EditRole: return editRole(inputData); // formatted to standard types or string
112  case Qt::ToolTipRole: return tooltipRole(inputData); // formatted to string
113  case Qt::DecorationRole: return decorationRole(inputData); // formatted as pixmap, icon, or color
114  case Qt::CheckStateRole: return checkStateRole(inputData); // as Qt check state
115  default: break;
116  }
117  return CVariant();
118  }

163  CVariant CPixmapFormatter::decorationRole(const CVariant &dataCVariant) const
164  {
165  if (dataCVariant.isNull()) { return {}; }
166  if (m_maxWidth < 0 && m_maxHeight < 0) { return CDefaultFormatter::decorationRole(dataCVariant); }
167 
168  QPixmap pm;
169  if (dataCVariant.canConvert<CIcon>())
170  {
171  const CIcon icon = dataCVariant.value<CIcon>();
172  pm = icon.toPixmap();
173  }
174 
175  if (pm.isNull()) { return {}; }
176  const int pmw = pm.width();
177  const int pmh = pm.height();
178 
179  if (m_maxHeight >= 0 && m_maxHeight < pmh)
180  {
181  return CVariant::fromValue(pm.scaledToHeight(m_maxHeight));
182  }
183 
184  if (m_maxWidth >= 0 && m_maxWidth < pmw)
185  {
186  return CVariant::fromValue(pm.scaledToWidth(m_maxWidth));
187  }
188 
189  return CVariant::fromValue(pm);
190  }

207  CVariant CDateTimeFormatter::displayRole(const CVariant &dateTime) const
208  {
209  if (dateTime.isNull()) return {};
210  if (static_cast<QMetaType::Type>(dateTime.type()) == QMetaType::QDateTime)
211  {
212  const QDateTime dt = dateTime.value<QDateTime>();
213  return dt.toString(m_formatString);
214  }
215  else if (static_cast<QMetaType::Type>(dateTime.type()) == QMetaType::QDate)
216  {
217  const QDate d = dateTime.value<QDate>();
218  return d.toString(m_formatString);
219  }
220  else if (static_cast<QMetaType::Type>(dateTime.type()) == QMetaType::QTime)
221  {
222  const QTime t = dateTime.value<QTime>();
223  return t.toString(m_formatString);
224  }
225  else if (dateTime.isIntegral())
226  {
227  const QDateTime t = QDateTime::fromMSecsSinceEpoch(dateTime.value<qint64>());
228  return t.toString(m_formatString);
229  }
230  else
231  {
232  Q_ASSERT_X(false, "formatQVariant", "No QDate, QTime or QDateTime");
233  return {};
234  }
235  }

422  CVariant CIntegerFormatter::displayRole(const CVariant &expectedInteger) const
423  {
424  bool ok = false;
425  switch (expectedInteger.type())
426  {
427  case QMetaType::LongLong:
428  {
429  const qlonglong ll = expectedInteger.toLongLong(&ok);
430  if (ok) { return QString::number(ll); }
431  break;
432  }
433  case QMetaType::ULongLong:
434  {
435  const qulonglong ll = expectedInteger.toULongLong(&ok);
436  if (ok) { return QString::number(ll); }
437  break;
438  }
439  default:
440  break;
441  }
442 
443  const int i = expectedInteger.toInt(&ok);
444  if (ok) { return QString::number(i); }
445  return CVariant();
446  }
