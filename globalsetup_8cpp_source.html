112  CUrl CGlobalSetup::getHelpPageUrl(const QString &context) const
113  {
114  const CUrlList urls(m_onlineHelpUrls);
115 
116  // we display in the standard browser,
117  // so the user will realize if the URL does not work
118  CUrl url = (urls.size() < 2) ? urls.frontOrDefault() : urls.getRandomUrl();
119  if (url.isEmpty()) { return url; }
120 
121  // context string something like "application.moreSpecific.evenMoreSpecific"
122  QString c = "client";
123  if (QCoreApplication::instance())
124  {
125  c = QCoreApplication::instance()->applicationName();
126  }
127  if (!context.isEmpty()) { c += "." + context; }
128  url.appendQuery("context", c);
129  return url;
130  }

147  CUrl CGlobalSetup::getDbClientPingServiceUrl(PingType type) const
148  {
149  CUrl pingUrl = this->getDbClientPingServiceUrl();
150  if (pingUrl.isEmpty()) { CUrl(); }
151 
152  pingUrl.appendQuery("uuid", QSysInfo::machineUniqueId());
153  if (type.testFlag(PingLogoff)) { pingUrl.appendQuery("logoff", "true"); }
154  if (type.testFlag(PingShutdown)) { pingUrl.appendQuery("shutdown", "true"); }
155  if (type.testFlag(PingStarted)) { pingUrl.appendQuery("started", "true"); }
156  pingUrl.appendQuery("os", CBuildConfig::getPlatformString());
157  if (CBuildConfig::isLocalDeveloperDebugBuild()) { pingUrl.appendQuery("dev", "true"); }
158  if (sApp)
159  {
160  const CCrashInfo ci = CCrashHandler::instance()->getCrashInfo();
161  pingUrl.appendQuery("application", sApp->getApplicationNameAndVersion());
162  if (!ci.getSimulatorString().isEmpty()) { pingUrl.appendQuery("fs", ci.getSimulatorString()); }
163  if (!ci.getFlightNetworkString().isEmpty()) { pingUrl.appendQuery("network", ci.getFlightNetworkString()); }
164  }
165  return pingUrl;
166  }

192  QString CGlobalSetup::buildBootstrapFileUrl(const QString &candidate)
193  {
194  if (candidate.isEmpty()) return {}; // not possible
195  static const QString version(QString(CGlobalSetup::schemaVersionString()).append("/"));
196  if (candidate.endsWith(CSwiftDirectories::bootstrapFileName())) { return candidate; }
197  CUrl url(candidate);
198  if (candidate.contains("/bootstrap"))
199  {
200  url.appendPath(CSwiftDirectories::bootstrapFileName());
201  }
202  else if (candidate.endsWith(CGlobalSetup::schemaVersionString()) || candidate.endsWith(version))
203  {
204  url.appendPath("/bootstrap/" + CSwiftDirectories::bootstrapFileName());
205  }
206  else if (candidate.endsWith("shared") || candidate.endsWith("shared/"))
207  {
208  url.appendPath(CGlobalSetup::schemaVersionString() + "/bootstrap/" + CSwiftDirectories::bootstrapFileName());
209  }
210  else
211  {
212  url.appendPath("shared/" + CGlobalSetup::schemaVersionString() + "/bootstrap/" + CSwiftDirectories::bootstrapFileName());
213  }
214  return url.getFullUrl();
215  }

293  QVariant CGlobalSetup::propertyByIndex(BlackMisc::CPropertyIndexRef index) const
294  {
295  if (index.isMyself()) { return CVariant::fromValue(*this); }
296  if (ITimestampBased::canHandleIndex(index)) { return ITimestampBased::propertyByIndex(index); }
297 
298  const ColumnIndex i = index.frontCasted<ColumnIndex>();
299  switch (i)
300  {
301  case IndexDbRootDirectory: return QVariant::fromValue(m_dbRootDirectoryUrl);
302  case IndexDbHttpPort: return QVariant::fromValue(m_dbHttpPort);
303  case IndexDbHttpsPort: return QVariant::fromValue(m_dbHttpsPort);
304  case IndexDbLoginService: return QVariant::fromValue(this->getDbLoginServiceUrl());
305  case IndexDbClientPingService: return QVariant::fromValue(this->getDbClientPingServiceUrl());
306  case IndexVatsimStatus: return QVariant::fromValue(m_vatsimStatusFileUrls);
307  case IndexVatsimData: return QVariant::fromValue(m_vatsimDataFileUrls);
308  case IndexVatsimBookings: return QVariant::fromValue(m_vatsimBookingsUrl);
309  case IndexVatsimServer: return QVariant::fromValue(m_vatsimServerFileUrl);
310  case IndexVatsimHttpFsd: return QVariant::fromValue(m_vatsimFsdHttpUrl);
311  case IndexVatsimMetars: return QVariant::fromValue(m_vatsimMetarsUrls);
312  case IndexBootstrapFileUrls: return QVariant::fromValue(this->getSwiftBootstrapFileUrls());
313  case IndexUpdateInfoFileUrls: return QVariant::fromValue(this->getSwiftUpdateInfoFileUrls());
314  case IndexSharedUrls: return QVariant::fromValue(m_sharedUrls);
315  case IndexOnlineHelpUrls: return QVariant::fromValue(m_onlineHelpUrls);
316  case IndexCrashReportServerUrl: return QVariant::fromValue(m_crashReportServerUrl);
317  case IndexWasLoadedFromFile: return QVariant::fromValue(m_wasLoadedFromFile);
318  case IndexMappingMinimumVersion: return QVariant::fromValue(m_mappingMinimumVersion);
319  case IndexPredefinedServers: return QVariant::fromValue(m_predefinedServers);
320  default: return CValueObject::propertyByIndex(index);
321  }

324  void CGlobalSetup::setPropertyByIndex(CPropertyIndexRef index, const QVariant &variant)
325  {
326  if (index.isMyself())
327  {
328  (*this) = variant.value<CGlobalSetup>();
329  return;
330  }
331  if (ITimestampBased::canHandleIndex(index))
332  {
333  ITimestampBased::setPropertyByIndex(index, variant);
334  return;
335  }
336 
337  const ColumnIndex i = index.frontCasted<ColumnIndex>();
338  switch (i)
339  {
340  case IndexDbRootDirectory: m_dbRootDirectoryUrl.setPropertyByIndex(index.copyFrontRemoved(), variant); break;
341  case IndexDbHttpPort: m_dbHttpPort = variant.toInt(); break;
342  case IndexDbHttpsPort: m_dbHttpsPort = variant.toInt(); break;
343  case IndexDbLoginService: break; // cannot be changed
344  case IndexDbClientPingService: break; // cannot be changed
345  case IndexVatsimData: m_vatsimDataFileUrls = variant.value<CUrlList>(); break;
346  case IndexVatsimBookings: m_vatsimBookingsUrl.setPropertyByIndex(index.copyFrontRemoved(), variant); break;
347  case IndexVatsimServer: m_vatsimServerFileUrl = variant.value<CUrl>(); break;
348  case IndexVatsimHttpFsd: m_vatsimFsdHttpUrl = variant.value<CUrl>(); break;
349  case IndexVatsimMetars: m_vatsimMetarsUrls = variant.value<CUrlList>(); break;
350  case IndexSharedUrls: m_sharedUrls = variant.value<CUrlList>(); break;
351  case IndexOnlineHelpUrls: m_onlineHelpUrls = variant.value<CUrlList>(); break;
352  case IndexCrashReportServerUrl: m_crashReportServerUrl = variant.value<CUrl>(); break;
353  case IndexWasLoadedFromFile: m_wasLoadedFromFile = variant.toBool(); break;
354  case IndexMappingMinimumVersion: m_mappingMinimumVersion = variant.toString(); break;
355  case IndexPredefinedServers: m_predefinedServers = variant.value<CServerList>(); break;
356  default: CValueObject::setPropertyByIndex(index, variant); break;
357  }
358  }
