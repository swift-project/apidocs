107  CSimulatorInfo IMultiSimulatorModelCaches::getSimulatorForFilename(const QString &filename) const
108  {
109  if (filename.isEmpty()) { return CSimulatorInfo(); }
110  CSimulatorInfo sims;
111  const QString compareFileName(QFileInfo(filename).fileName());
112  for (const CSimulatorInfo &singleSim : CSimulatorInfo::allSimulatorsSet())
113  {
114  const QString singleSimFile(getFilename(singleSim));
115  if (singleSimFile.isEmpty()) continue;
116  const QString singleSimFileName(QFileInfo(singleSimFile).fileName());
117  if (singleSimFileName == compareFileName)
118  {
119  sims.add(singleSim);
120  break;
121  }
122  }
123  return sims;

427  CStatusMessage CModelSetCaches::setCachedModels(const CAircraftModelList &models, const CSimulatorInfo &simulator)
428  {
429  Q_ASSERT_X(simulator.isSingleSimulator(), Q_FUNC_INFO, "No single simulator");
430  CAircraftModelList orderedModels(models);
431  orderedModels.setModelType(CAircraftModel::TypeOwnSimulatorModel); // unify type
432 
433  if (simulator.isXPlane())
434  {
435  // see https://discordapp.com/channels/539048679160676382/539064750055751690/648974956306366499
436  // this solves the issue that ACF models are loaded in CSimulatorXPlane::loadCslPackages
437  const int removed = orderedModels.removeXPlaneFlyablePlanes();
438  if (removed > 0) { CLogMessage(this).info(u"Removed %1 flyable models from XPlane model set!"); }
439  }
440 
441  if (orderedModels.needsOrder())
442  {
443  orderedModels.resetOrder();
444  }
445  else
446  {
447  orderedModels.sortAscendingByOrder();
448  }
449 
450  CStatusMessage msg;
451  switch (simulator.getSimulator())
452  {
453  case CSimulatorInfo::FS9: msg = m_modelCacheFs9.set(orderedModels); break;
454  case CSimulatorInfo::FSX: msg = m_modelCacheFsx.set(orderedModels); break;
455  case CSimulatorInfo::P3D: msg = m_modelCacheP3D.set(orderedModels); break;
456  case CSimulatorInfo::XPLANE: msg = m_modelCacheXP.set(orderedModels); break;
457  case CSimulatorInfo::FG: msg = m_modelCacheFG.set(orderedModels); break;
458  default:
459  Q_ASSERT_X(false, Q_FUNC_INFO, "wrong simulator");
