121 void CSwiftLauncher::installerMode()
122 {
123  if (!sGui || sGui->isShuttingDown()) { return; }
124  if (!sGui->isInstallerOptionSet()) { return; }
125 
126  bool runDialog = false;
127  do
128  {
129  const QDir dir = CSwiftDirectories::logDirectory();
130  if (!dir.exists()) { break; }
131 
132  if (sGui && sGui->getSetupReader())
133  {
134  sGui->getSetupReader()->prefillCacheWithLocalResourceBootstrapFile();
135  }
136 
137  for (const CSimulatorInfo &sim : CSimulatorInfo::allSimulatorsSet())
138  {
139  this->synchronizeCache(sim);
140  const int c = this->getCachedModelsCount(sim);
141  if (c > 0)
142  {
143  // we already have data
144  runDialog = true;
145  break;
146  }
147  }
148  }
149  while (false);
150 
151  bool startWizard = true;
152  ui->fr_SwiftLauncherMain->closeOverlay();
153 
154  if (runDialog)
155  {
156  const QMessageBox::StandardButton ret =
157  QMessageBox::question(this,
158  tr("swift configuration"),
159  tr("This installation directory already contains a swift configuration.\n"
160  "Do you want to use that one?"));
161  if (ret != QMessageBox::No) { startWizard = false; }
162  }
163 
164  if (startWizard) { this->startWizard(); }
165 }

318 bool CSwiftLauncher::setSwiftGuiExecutable()
319 {

321  m_executable = CSwiftDirectories::executableFilePath(CBuildConfig::swiftGuiExecutableName());
322  QStringList args {

325  };

328  if (this->isStandaloneGuiSelected())

337  args.append(ui->comp_DBusSelector->getDBusCmdLineArgs());

348  this->showStatusMessage(m);
349  return false;

351  }
352  this->saveSetup();
353  m_executableArgs = sGui->argumentsJoined(args);
354  return true;

387 void CSwiftLauncher::saveSetup()
388 {

390  const QString dBus(ui->comp_DBusSelector->getDBusAddress());
391  if (!dBus.isEmpty()) { setup.setDBusAddress(dBus); }

394 
395  CLauncherSetup::AudioMode audio = CLauncherSetup::AudioNothingDisabled;

398  audio.setFlag(CLauncherSetup::AudioDisableStandaloneAudio, ui->cb_DisableSaAfv->isChecked());
399  setup.setAudioMode(audio);
400 
401  if (ui->rb_SwiftDistributed->isChecked())
402  {
403  setup.setCoreMode(CLauncherSetup::Distributed);
404  }
405 
406  const CStatusMessage msg = m_setup.set(setup);
407  Q_UNUSED(msg)
408 }

530 void CSwiftLauncher::checkRunningApplicationsAndCore()
531 {
532  // wait some time before buttons are enabled (allows startup)
533  if (m_startCoreWaitCycles > 0) { m_startCoreWaitCycles--; }
534  if (m_startMappingToolWaitCycles > 0) { m_startMappingToolWaitCycles--; }
535  if (m_startGuiWaitCycles > 0) { m_startGuiWaitCycles--; }
536 
537  const CApplicationInfoList runningApps = sGui->getRunningApplications();
538  const bool foundLocalCore = runningApps.containsApplication(CApplicationInfo::PilotClientCore);
539  const bool foundLocalMappingTool = runningApps.containsApplication(CApplicationInfo::MappingTool);
540  const bool foundLocalPilotClientGui = runningApps.containsApplication(CApplicationInfo::PilotClientGui);
541  const bool standalone = ui->rb_SwiftStandalone->isChecked();
542 
543  ui->tb_SwiftCore->setEnabled(!standalone && !foundLocalCore && m_startCoreWaitCycles < 1);
544  ui->tb_SwiftMappingTool->setEnabled(!foundLocalMappingTool && m_startMappingToolWaitCycles < 1);
545  ui->tb_SwiftGui->setEnabled(!foundLocalPilotClientGui && m_startGuiWaitCycles < 1);
546 }

585 void CSwiftLauncher::showSimulatorConfigDirs()
586 {
587  if (!m_textEditDialog)
588  {
589  m_textEditDialog.reset(new CTextEditDialog(this));
590  }
591 
592  const QObject *s = QObject::sender();
593  QStringList dirs;
594  QString simDir;
595  QString simObjDir;
596 
597  if (s == ui->pb_P3DConfigDirs)
598  {
599  simDir = CFsDirectories::p3dDir();
600  simObjDir = CFsDirectories::p3dSimObjectsDir();
601  const QString versionHint = CFsDirectories::guessP3DVersion(simDir);
602  dirs = CFsDirectories::p3dSimObjectsDirPlusAddOnXmlSimObjectsPaths(simObjDir, versionHint);
603  }
604  else if (s == ui->pb_FSXConfigDirs)
605  {
