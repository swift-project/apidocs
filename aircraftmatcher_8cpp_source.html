68  const CCallsign &callsign, const QString &primaryIcao, const QString &secondaryIcao,

72  {
73  CCallsign::addLogDetailsToList(log, callsign,
74  QStringLiteral("Find airline designator from 1st: '%1' 2nd: '%2' callsign: '%3', use airline callsign: %4, name: '%5' telephony: '%6' use web service: %7").arg(primaryIcao, secondaryIcao, callsign.toQString(), boolToYesNo(airlineFromCallsign), airlineName, airlineTelephony, boolToYesNo(useWebServices)), getLogCategories());
75  CAirlineIcaoCode code;
76  do
77  {
78  if (CAircraftMatcher::isValidAirlineIcaoDesignator(primaryIcao, useWebServices))
79  {
80  CCallsign::addLogDetailsToList(log, callsign, QStringLiteral("Using primary airline ICAO '%1'").arg(primaryIcao), getLogCategories());
81  code = stringToAirlineIcaoObject(callsign, primaryIcao, airlineName, airlineTelephony, useWebServices, log);
82  break;
83  }
84  if (CAircraftMatcher::isValidAirlineIcaoDesignator(secondaryIcao, useWebServices))
85  {
86  CCallsign::addLogDetailsToList(log, callsign, QStringLiteral("Using secondary airline ICAO '%1', primary '%2' not valid").arg(secondaryIcao, primaryIcao), getLogCategories());
87  code = stringToAirlineIcaoObject(callsign, secondaryIcao, airlineName, airlineTelephony, useWebServices, log);
88  break;
89  }
90 
91  CCallsign::addLogDetailsToList(log, callsign, QStringLiteral("Two invalid airline ICAO codes (primary/secondary) '%1', '%2'").arg(primaryIcao, secondaryIcao), getLogCategories());
92  if (airlineFromCallsign)
93  {
94  QString flightNumber;
95  const QString airlinePrefix = callsign.getAirlinePrefix(flightNumber);
96  if (airlinePrefix.isEmpty() || flightNumber.isEmpty())
97  {
98  CCallsign::addLogDetailsToList(log, callsign, QStringLiteral("Callsign '%1' cannot be split in airline '%1'/ flight number '%2'").arg(callsign.toQString(), flightNumber), getLogCategories());
99  break;
100  }
101  if (CAircraftMatcher::isValidAirlineIcaoDesignator(airlinePrefix, useWebServices))
102  {
103  CCallsign::addLogDetailsToList(log, callsign, QStringLiteral("Using airline from callsign '%1', suffix: '%2'").arg(callsign.toQString(), airlinePrefix), getLogCategories());
104  code = stringToAirlineIcaoObject(callsign, airlinePrefix, airlineName, airlineTelephony, useWebServices, log);
105  break;
106  }
107  }
108  }
109  while (false);
110 
111  if (code.hasValidDesignator())
112  {
113  CCallsign::addLogDetailsToList(log, callsign, QStringLiteral("Resolved to airline designator: %1").arg(code.toQString(true)));
114  }
115  else
116  {
117 
118  CCallsign::addLogDetailsToList(log, callsign,
119  QStringLiteral("Cannot find airline designator from 1st: '%1' 2nd: '%2' callsign: '%3', use airline callsign: %4, name: '%5' telephony: '%6' use web service: %7").arg(primaryIcao, secondaryIcao, callsign.toQString(), boolToYesNo(airlineFromCallsign), airlineName, airlineTelephony, boolToYesNo(useWebServices)), getLogCategories());
120  }
121  return code;
122  }

244  CAircraftModel CAircraftMatcher::getClosestMatch(const CSimulatedAircraft &remoteAircraft, MatchingLog whatToLog, CStatusMessageList *log, bool useMatchingScript) const
245  {
246  CAircraftModelList modelSet(m_modelSet); // Models for this matching
247  const CAircraftMatcherSetup setup = m_setup;
248 
249  static const QString format("hh:mm:ss.zzz");
250  static const QString m1("--- Start matching: UTC %1 ---");
251  static const QString m2("Input model: '%1' '%2'");
252  static const QString m3("Matching uses model set of %1 models\n%2");
253  static const QString m4("Setup %1");
254  static const QString summary(
255  "Matching summary\n"
256  "-----------------------------------------\n"
257  "- Combined: %1 -> %2\n"
258  "- Aircraft: %3 -> %4\n"
259  "- Airline: %5 -> %6\n"
260  "- Livery: %7 -> %8\n"
261  "- Model: %9 -> %10\n"
262  "- Script modifed value: %11\n"
263  "-----------------------------------------\n");
264 
265  const QDateTime startTime = QDateTime::currentDateTimeUtc();
266  if (whatToLog == MatchingLogNothing) { log = nullptr; }
267  if (log) { log->clear(); }
268 
269  CMatchingUtils::addLogDetailsToList(log, remoteAircraft, m1.arg(startTime.toString(format)));
270  CMatchingUtils::addLogDetailsToList(log, remoteAircraft, m2.arg(remoteAircraft.getCallsignAsString(), removeSurroundingApostrophes(remoteAircraft.getModel().toQString())));
271  CMatchingUtils::addLogDetailsToList(log, remoteAircraft, m3.arg(modelSet.size()).arg(modelSet.coverageSummaryForModel(remoteAircraft.getModel())));
272  CMatchingUtils::addLogDetailsToList(log, remoteAircraft, m4.arg(setup.toQString(true)));
273 
274  // Before I really search I check some special conditions
275  // 1) Manually set model (by user)
276  // 2) No model set at all
277  // 3) Exact match by model string
278 
279  // Manually set string?
280  CAircraftModel matchedModel;
281  bool resolvedInPrephase = false;
282  if (remoteAircraft.getModel().hasManuallySetString())
283  {
284  // the user did a manual mapping "by hand", so he really should know what he is doing
285  // no matching
286  CMatchingUtils::addLogDetailsToList(log, remoteAircraft, u"Manually set model " % remoteAircraft.getModelString(), getLogCategories());
287  matchedModel = remoteAircraft.getModel();
288  resolvedInPrephase = true;
289  }
290  else if (modelSet.isEmpty())
291  {
292  CMatchingUtils::addLogDetailsToList(log, remoteAircraft, QStringLiteral("No models for matching, using default"), getLogCategories(), CStatusMessage::SeverityError);
293  matchedModel = this->getDefaultModel();
294  resolvedInPrephase = true;
295  }
296  else if (remoteAircraft.hasModelString())
297  {
298  // try to find in installed models by model string
299  if (setup.getMatchingMode().testFlag(CAircraftMatcherSetup::ByModelString))
300  {
301  matchedModel = matchByExactModelString(remoteAircraft, modelSet, whatToLog, log);
302  if (matchedModel.hasModelString())
303  {
304  CMatchingUtils::addLogDetailsToList(log, remoteAircraft, u"Exact match by model string '" % matchedModel.getModelStringAndDbKey() % "'", getLogCategories(), CStatusMessage::SeverityError);
305  resolvedInPrephase = true;
306  }
307  }
308  else
309  {
310  CMatchingUtils::addLogDetailsToList(log, remoteAircraft, "Skipping model string match", getLogCategories());
311  }
312  }
313 
314  if (!resolvedInPrephase)
315  {
316  // sanity
317  const int noString = modelSet.removeAllWithoutModelString();
318  static const QString noModelStr("Excluded %1 models without model string");
319  if (noString > 0 && log) { CMatchingUtils::addLogDetailsToList(log, remoteAircraft, noModelStr.arg(noString)); }
320 
321  // exclusion
322  if (setup.getMatchingMode().testFlag(CAircraftMatcherSetup::ExcludeNoDbData))
323  {
324  const int noDbKey = modelSet.removeObjectsWithoutDbKey();
325  static const QString excludedStr("Excluded %1 models without DB key");
326  if (noDbKey > 0 && log) { CMatchingUtils::addLogDetailsToList(log, remoteAircraft, excludedStr.arg(noDbKey)); }
327  }
328 
329  if (setup.getMatchingMode().testFlag(CAircraftMatcherSetup::ExcludeNoExcluded))
330  {
331  const int excluded = modelSet.removeIfExcluded();
332  static const QString excludedStr("Excluded %1 models marked 'Excluded'");
333  if (excluded > 0 && log) { CMatchingUtils::addLogDetailsToList(log, remoteAircraft, excludedStr.arg(excluded)); }
334  }
335 
336  // Reduce by ICAO if the flag is set
337  static const QString msInfo("Using '%1' with model set with %2 models");
338  CMatchingUtils::addLogDetailsToList(log, remoteAircraft, msInfo.arg(setup.getMatchingAlgorithmAsString()).arg(modelSet.size()), getLogCategories());
339  CAircraftModelList candidates;
340  int maxScore = -1;
341 
342  switch (setup.getMatchingAlgorithm())
343  {
344  case CAircraftMatcherSetup::MatchingStepwiseReduce:
345  candidates = CAircraftMatcher::getClosestMatchStepwiseReduceImplementation(modelSet, setup, m_categoryMatcher, remoteAircraft, whatToLog, log);
346  break;
347  case CAircraftMatcherSetup::MatchingScoreBased:
348  candidates = CAircraftMatcher::getClosestMatchScoreImplementation(modelSet, setup, remoteAircraft, maxScore, whatToLog, log);
349  break;
350  case CAircraftMatcherSetup::MatchingStepwiseReducePlusScoreBased:
351  default:
352  candidates = CAircraftMatcher::getClosestMatchStepwiseReduceImplementation(modelSet, setup, m_categoryMatcher, remoteAircraft, whatToLog, log);
353  candidates = CAircraftMatcher::getClosestMatchScoreImplementation(candidates, setup, remoteAircraft, maxScore, whatToLog, log);
354  break;
355  }
356 
357  if (candidates.isEmpty())
358  {
359  matchedModel = CAircraftMatcher::getCombinedTypeDefaultModel(modelSet, remoteAircraft, this->getDefaultModel(), whatToLog, log);
360  }
361  else
362  {
363  CAircraftMatcherSetup::PickSimilarStrategy usedStrategy = setup.getPickStrategy();
364  switch (usedStrategy)
365  {
366  case CAircraftMatcherSetup::PickRandom:
367  matchedModel = candidates.randomElement<CAircraftModel>();
368  break;
369  case CAircraftMatcherSetup::PickByOrder:
370  if (!candidates.needsOrder())
371  {
372  matchedModel = candidates.minOrderOrDefault();
373  break;
374  }
375  [[fallthrough]];
376  case CAircraftMatcherSetup::PickFirst: // fallthru intentionally
377  default:
378  usedStrategy = CAircraftMatcherSetup::PickFirst; // re-assigned if fall-through
379  matchedModel = candidates.front();
380  break;
381  }
382 
383  if (log) { CMatchingUtils::addLogDetailsToList(log, remoteAircraft, QStringLiteral("Picking among %1 by strategy '%2'").arg(candidates.size()).arg(CAircraftMatcherSetup::strategyToString(usedStrategy))); }
384  }
385  }
386 
387  // copy over callsign validate
388  matchedModel.setCallsign(remoteAircraft.getCallsign());
389  matchedModel.setModelType(CAircraftModel::TypeModelMatching);
390 
391  // matching script
392  bool didRunAndModifyMatchingScript = false;
393  if (useMatchingScript && setup.doRunMsMatchingStageScript())
394  {
395  CMatchingUtils::addLogDetailsToList(log, remoteAircraft, QStringLiteral("Matching script: Matching stage script used"));
396  const MatchingScriptReturnValues rv = CAircraftMatcher::matchingStageScript(remoteAircraft.getModel(), matchedModel, setup, modelSet, log);
397  CAircraftModel matchedModelMs = matchedModel;
398 
399  if (rv.runScriptAndModified())
400  {
401  matchedModelMs = rv.model;
402  didRunAndModifyMatchingScript = true;
403  }
404 
405  if (rv.runScriptModifiedAndRerun())
406  {
407  CMatchingUtils::addLogDetailsToList(log, remoteAircraft, QStringLiteral("Matching script: Modified values and re-run requested"));
408  CMatchingUtils::addLogDetailsToList(log, remoteAircraft, QStringLiteral("Matching script: Now using model: '%1'").arg(matchedModel.toQString(true)));
409 
410  CSimulatedAircraft rerunAircraft(remoteAircraft);
411  rerunAircraft.setModel(matchedModelMs);
412  CStatusMessageList log2ndRun;
413  matchedModelMs = CAircraftMatcher::getClosestMatch(rerunAircraft, whatToLog, log ? &log2ndRun : nullptr, false);
414  if (log) { log->push_back(log2ndRun); }
415 
416  // the script can fuckup the model, leading to an empty model string or such
417  matchedModelMs.setCallsign(remoteAircraft.getCallsign());
418  if (matchedModelMs.hasModelString())
419  {
420  matchedModel = matchedModelMs;
421  }
422  else
423  {
424  CMatchingUtils::addLogDetailsToList(log, remoteAircraft, QStringLiteral("Matching script: Ignoring model without model string after running the script"));
425  }
426  }
427  }
428  else
429  {
430  if (log) CMatchingUtils::addLogDetailsToList(log, remoteAircraft, QStringLiteral("Matching script: No matching stage script used"));
431  }
432 
433  // copy over callsign validate (again, just in case it was changed in matching script)
434  matchedModel.setCallsign(remoteAircraft.getCallsign());
435  matchedModel.setModelType(CAircraftModel::TypeModelMatching);
436 
437  // reported here by LT: https://discordapp.com/channels/539048679160676382/539925070550794240/701439918815051846
438  if (!matchedModel.hasModelString())
439  {
440  if (log) { CMatchingUtils::addLogDetailsToList(log, remoteAircraft, QStringLiteral("All matching yielded no result, VERY odd...")); }
441  CAircraftModel defaultModel = this->getDefaultModel();
442  if (defaultModel.hasModelString())
443  {
444  matchedModel = defaultModel;
445  matchedModel.setCallsign(remoteAircraft.getCallsign());
446  if (log) { CMatchingUtils::addLogDetailsToList(log, remoteAircraft, QStringLiteral("Using default model '%1'").arg(matchedModel.getModelString())); }
447  }
448  else
449  {
450  if (log) { CMatchingUtils::addLogDetailsToList(log, remoteAircraft, QStringLiteral("Not even a default model. Giving up").arg(matchedModel.getModelString())); }
451  }
452  }
453 
454  Q_ASSERT_X(!matchedModel.getCallsign().isEmpty(), Q_FUNC_INFO, "Missing callsign for matched model");
455  // Q_ASSERT_X(matchedModel.hasModelString(), Q_FUNC_INFO, "Missing model string for matched model");
456 
457  if (log)
458  {
459  static const QString nms = "no model string";
460  CMatchingUtils::addLogDetailsToList(log, remoteAircraft,
461  summary.arg(
462  remoteAircraft.getAircraftIcaoCode().getCombinedType(), matchedModel.getAircraftIcaoCode().getCombinedType(),
463  remoteAircraft.getAircraftIcaoCode().getDesignatorDbKey(), matchedModel.getAircraftIcaoCode().getDesignatorDbKey(),
464  remoteAircraft.getAirlineIcaoCode().getVDesignatorDbKey(), matchedModel.getAirlineIcaoCode().getVDesignatorDbKey())
465  .arg(
466  remoteAircraft.getLivery().getCombinedCodePlusInfoAndId(), matchedModel.getLivery().getCombinedCodePlusInfoAndId(),
467  defaultIfEmpty(remoteAircraft.getModel().getModelStringAndDbKey(), nms), matchedModel.getModelStringAndDbKey(),
468  boolToYesNo(didRunAndModifyMatchingScript)));
469  } // log
470 
471  const QDateTime endTime = QDateTime::currentDateTimeUtc();
472  const qint64 matchingTime = startTime.msecsTo(endTime);
473  static const QString em("--- Matching end: UTC %1, time %2ms ---");
474  CMatchingUtils::addLogDetailsToList(log, remoteAircraft, em.arg(endTime.toString(format)).arg(matchingTime));
475  return matchedModel;
476  }

690  CAircraftModel CAircraftMatcher::reverseLookupModel(const CAircraftModel &modelToLookup, const QString &networkLiveryInfo, const CAircraftMatcherSetup &setup, const CAircraftModelList &modelSet, CStatusMessageList *log)
691  {
692  if (!sApp || sApp->isShuttingDown() || !sApp->hasWebDataServices()) { return CAircraftModel(); }
693 
694  // already DB model?
695  CAircraftModel model(modelToLookup); // copy
696  if (modelToLookup.isLoadedFromDb()) { return modelToLookup; }
697 
698  // --- now I try to fill in as many DB data as possible ---
699  // 1) This will unify data where possible
700  // 2) I have full information of what the other pilot flies where possible
701  // 3) This is not model matching here (!), it is a process of getting the most accurate data from that fuzzy information I get via FSD
702  //
703  // Reverse lookup, use DB data wherever possible
704  // 1) If I cannot resolve the ICAO codes here, they are either wrong (most likely in most cases) or
705  // 2) not in the DB yet
706  const CCallsign callsign(model.getCallsign());
707 
708  do
709  {
710  if (modelToLookup.hasModelString())
711  {
712  if (!setup.isReverseLookupModelString())
713  {
714  if (log) { CCallsign::addLogDetailsToList(log, callsign, QStringLiteral("Model string looup disabled")); }
715  }
716  else
717  {
718  const QString modelString = modelToLookup.getModelString();
719 
720  // if we find the model here we have a fully defined DB model
721  const CAircraftModel modelFromDb = CAircraftMatcher::reverseLookupModelStringInDB(modelString, callsign, setup.isReverseLookupModelString(), log);
722  if (modelFromDb.hasValidDbKey())
723  {
724  model = modelFromDb;
725  break; // done here
726  }
727 
728  // const bool useNonDbEntries = setup.isDbDataOnly();
729  const bool useNonDbEntries = true;
730  const CAircraftModel modelFromSet = CAircraftMatcher::reverseLookupModelStringInSet(modelString, callsign, modelSet, useNonDbEntries, log);
731  if (modelFromSet.hasModelString())
732  {
733  model = modelFromSet;
734  break; // done here
735  }
736  }
737  }
738 
739  // lookup if model is not yet from DB
740  const DBTripleIds ids = CAircraftModel::parseNetworkLiveryString(networkLiveryInfo);
741  if (log) { CCallsign::addLogDetailsToList(log, callsign, QStringLiteral("Livery string with ids: '%1'").arg(ids.toQString())); }
742 
743  if (!setup.isReverseLookupSwiftLiveryIds())
744  {
745  if (log) { CCallsign::addLogDetailsToList(log, callsign, QStringLiteral("Ignoring livery ids '%1', because of setup").arg(ids.toQString())); }
746  }
747  else if (ids.model >= 0 && !modelToLookup.hasModelString())
748  {
749  if (log) { CCallsign::addLogDetailsToList(log, callsign, QStringLiteral("Model lookup with id %1 from triple ids '%2'").arg(ids.model).arg(ids.toQString())); }
750  const CAircraftModel modelFromDb = CAircraftMatcher::reverseLookupModelId(ids.model, callsign, log);
751  if (modelFromDb.hasValidDbKey())
752  {
753  model = modelFromDb;
754  break;
755  }
756  }
757 
758  // no direct resolution of model, try livery and aircraft ICAO
759  if (!modelToLookup.getAircraftIcaoCode().hasValidDbKey() && ids.aircraft >= 0)
760  {
761  if (log) { CCallsign::addLogDetailsToList(log, callsign, QStringLiteral("Aircraft ICAO lookup with id %1 from triple ids '%2'").arg(ids.aircraft).arg(ids.toQString())); }
762  const CAircraftIcaoCode icaoFromDb = CAircraftMatcher::reverseLookupAircraftIcaoId(ids.aircraft, callsign, log);
763  if (icaoFromDb.hasValidDbKey())
764  {
765  model.setAircraftIcaoCode(icaoFromDb);
766  }
767  }
768 
769  if (!modelToLookup.getLivery().hasValidDbKey() && ids.livery >= 0)
770  {
771  if (log) { CCallsign::addLogDetailsToList(log, callsign, QStringLiteral("Livery lookup with id %1 from triple ids '%2'").arg(ids.livery).arg(ids.toQString())); }
772  const CLivery liveryFromDb = CAircraftMatcher::reverseLookupLiveryId(ids.livery, callsign, log);
773  if (liveryFromDb.hasValidDbKey())
774  {
775  model.setLivery(liveryFromDb);
776  }
777  }
778 
779  // aircraft ICAO if not from DB yet
780  if (!model.getAircraftIcaoCode().hasValidDbKey())
781  {
782  CAircraftIcaoCode reverseIcaoCode(model.getAircraftIcaoCode());
783  if (!reverseIcaoCode.isLoadedFromDb())
784  {
785  reverseIcaoCode = CAircraftMatcher::reverseLookupAircraftIcao(reverseIcaoCode, callsign, log);
786  if (reverseIcaoCode.isLoadedFromDb())
787  {
788  if (log) { CCallsign::addLogDetailsToList(log, callsign, QStringLiteral("Set aircraft ICAO to '%1' from DB").arg(reverseIcaoCode.getCombinedIcaoStringWithKey())); }
789  model.setAircraftIcaoCode(reverseIcaoCode);
790  }
791  else
792  {
793  // no DB data
794  if (log) { CCallsign::addLogDetailsToList(log, callsign, QStringLiteral("Reverse lookup, ICAO '%1' not resolved from DB").arg(reverseIcaoCode.getDesignator())); }
795  }
796  }
797  }
798 
799  // check if livery is already from DB
800  if (!model.getLivery().isLoadedFromDb())
801  {
802  CAirlineIcaoCode airlineIcaoCode(model.getAirlineIcaoCode());
803  if (!airlineIcaoCode.isLoadedFromDb())
804  {
805  airlineIcaoCode = CAircraftMatcher::reverseLookupAirlineIcao(airlineIcaoCode, callsign, log);
806  }
807 
808  // try to match by livery
809  QString liveryCode = networkLiveryInfo;
810  if (liveryCode.isEmpty() && airlineIcaoCode.hasValidDesignator())
811  {
812  // we create a standard livery code, then we try to find based on this
813  liveryCode = CLivery::getStandardCode(airlineIcaoCode);
814  }
815 
816  if (CLivery::isValidCombinedCode(liveryCode))
817  {
818  // search livery by combined code
819  const CLivery reverseLivery(sApp->getWebDataServices()->getLiveryForCombinedCode(liveryCode));
820  if (reverseLivery.hasValidDbKey())
821  {
822  // we have found a livery in the DB
823  model.setLivery(reverseLivery);
824  if (log) { CCallsign::addLogDetailsToList(log, callsign, QStringLiteral("Reverse lookup of livery found '%1'").arg(reverseLivery.getCombinedCodePlusInfoAndId()), getLogCategories()); }
825  }
826  else
827  {
828  // no livery data found
829  if (log) { CCallsign::addLogDetailsToList(log, callsign, QStringLiteral("Reverse lookup of livery '%1' yielded no result").arg(reverseLivery.getCombinedCodePlusInfo()), getLogCategories()); }
830  }
831  } // livery lookup
832 
833  // if no DB livery yet, create own livery
834  if (!model.hasValidDbKey() && !model.getLivery().hasValidDbKey())
835  {
836  if (airlineIcaoCode.hasValidDesignator())
837  {
838  if (airlineIcaoCode.hasValidDbKey())
839  {
840  const CLivery stdLivery(sApp->getWebDataServices()->getStdLiveryForAirlineCode(airlineIcaoCode));
841  if (stdLivery.hasValidDbKey())
842  {
843  model.setLivery(stdLivery);
844  if (log) { CCallsign::addLogDetailsToList(log, callsign, QStringLiteral("Set standardlivery `%1`").arg(stdLivery.getCombinedCodePlusInfo())); }
845  }
846  }
847 
848  if (!model.getLivery().hasValidDbKey())
849  {
850  // create a pseudo livery, try to find airline first
851  const CLivery liveryDummy(CLivery::getStandardCode(airlineIcaoCode), airlineIcaoCode, "Generated");
852  model.setLivery(liveryDummy);
853  if (log) { CCallsign::addLogDetailsToList(log, callsign, QStringLiteral("Generated livery, set livery `%1`").arg(liveryDummy.getCombinedCodePlusInfo())); }
854  }
855  }
856  } // pseudo livery
857  } // livery from DB
858  }
859  while (false);
860 
861  model.setCallsign(callsign);
862  model.setModelType(modelToLookup.getModelType());
863 
864  if (log)
865  {
866  CCallsign::addLogDetailsToList(log, callsign, QStringLiteral("Using model: ICAO '%1', livery '%2', model '%3', type '%4'").arg(model.getAircraftIcaoCode().getCombinedIcaoStringWithKey(), model.getLivery().getCombinedCodePlusInfo(), model.getModelString(), model.getModelTypeAsString()));
867  }
868  return model;
869  }
