84  void CSettingsMatchingComponent::deferredReload(int deferMs)
85  {
86  IContextSimulator *simContext = simulatorContext();
87  if (!simContext) { return; }
88  if (deferMs < 1)
89  {
90  const CAircraftMatcherSetup setup = simContext->getMatchingSetup();
91  ui->form_Matching->setValue(setup);
92  }
93  else
94  {
95  QPointer<CSettingsMatchingComponent> myself(this);
96  QTimer::singleShot(deferMs, this, [=] {
97  if (!myself) { return; }
98  this->deferredReload(0);
99  });
100  }
101  }
