54  CAirport CAirportList::findFirstByNameOrLocation(const QString &nameOrLocation) const
55  {
56  if (this->isEmpty() || nameOrLocation.isEmpty()) { return CAirport(); }
57  CAirportList airports = this->findBy([&](const CAirport &airport) {
58  return airport.matchesDescriptiveName(nameOrLocation);
59  });
60  if (!airports.isEmpty()) { return airports.frontOrDefault(); }
61 
62  airports = this->findBy([&](const CAirport &airport) {
63  return airport.matchesLocation(nameOrLocation);
64  });
65  if (!airports.isEmpty()) { return airports.frontOrDefault(); }
66  return CAirport();
