126  void CModelBrowserComponent::display()
127  {
128  if (!this->hasContexts()) { return; }
129  CAircraftModel model = ui->tvp_AircraftModels->selectedObject();
130  if (!model.hasModelString()) { return; }
131 
132  this->remove(); // only 1 model at a time
133 
134  if (ui->cb_OverrideCG->isChecked())
135  {
136  CLength cg;
137  cg.parseFromString(ui->le_CG->text(), CPqString::SeparatorBestGuess);
138  if (!cg.isNull()) { model.setCG(cg); }
139  }
140  else
141  {
142  ui->le_CG->setText(model.getCG().valueRoundedWithUnit(CLengthUnit::ft(), 1));
143  }
144 
145  m_situation.setZeroPBHandGs();
146  if (ui->cb_UsePBH->isChecked())
147  {
148  ui->editor_Pbh->updateSituation(m_situation);
149  }
150 
151  if (ui->cb_UseParts->isChecked())
152  {
153  const CAircraftParts parts = ui->editor_AircraftParts->getAircraftPartsFromGui();
154  if (!parts.isNull()) { m_aircraft.setParts(parts); }
155  }
156 
157  m_aircraft.setModel(model);
158  m_aircraft.setCallsign(ISimulator::getTestCallsign());
159  m_aircraft.setSituation(m_situation);
160 
161  ui->le_ModelInfo->setText(model.getModelStringAndDbKey());
162  ui->le_Info->setText(m_situation.toQString(true));
163 
164  sGui->getIContextSimulator()->requestElevationBySituation(m_situation);
165  sGui->getIContextSimulator()->testRemoteAircraft(m_aircraft, true);
166  }
