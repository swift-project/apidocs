39  void CCopyModelsFromOtherSwiftVersionsComponent::copy()
40  {
41  const CSimulatorInfo selectedSimulators = ui->comp_SimulatorSelector->getValue();
42  const QSet<CSimulatorInfo> simulators = selectedSimulators.asSingleSimulatorSet();
43  if (simulators.isEmpty())
44  {
45  static const CStatusMessage m = CStatusMessage(this).validationError(u"No simulator(s) selected");
46  this->showOverlayMessage(m);
47  return;
48  }
49 
50  const bool set = ui->cb_ModelSet->isChecked();
51  const bool cache = ui->cb_ModelCache->isChecked();
52  if (!cache && !set)
53  {
54  static const CStatusMessage m = CStatusMessage(this).validationError(u"No models selected (cache? model set?)");
55  this->showOverlayMessage(m);
56  return;
57  }
58 
59  if (!ui->comp_OtherSwiftVersions->hasSelection())
60  {
61  static const CStatusMessage m = CStatusMessage(this).validationError(u"No other version selected");
62  this->showOverlayMessage(m);
63  return;
64  }
65 
66  int sets = 0;
67  int caches = 0;
68  const CApplicationInfo otherVersion = ui->comp_OtherSwiftVersions->selectedOtherVersion();
69  for (const CSimulatorInfo &simulator : simulators)
70  {
71  if (set)
72  {
73  // inits current version cache
74  m_modelSetCaches.synchronizeCache(simulator);
75 
76  // get file name
77  CAircraftModelList otherSet;
78  const QString thisVersionModelSetFile = m_modelSetCaches.getFilename(simulator);
79  if (this->readDataFile(thisVersionModelSetFile, otherSet, otherVersion, simulator) && !otherSet.isEmpty())
80  {
81  CApplication::processEventsFor(250);
82  if (this->confirmOverride(QStringLiteral("Override model set for '%1'").arg(simulator.toQString())))
83  {
84  m_modelSetCaches.setModelsForSimulator(otherSet, simulator);
85  }
86  }
87  sets++;
88  } // set
89 
90  if (cache)
91  {
92  // inits current version cache
93  m_modelCaches.synchronizeCache(simulator);
94 
95  // get file name
96  CAircraftModelList otherCache;
97  const QString thisVersionModelCacheFile = m_modelCaches.getFilename(simulator);
98  if (this->readDataFile(thisVersionModelCacheFile, otherCache, otherVersion, simulator) && !otherCache.isEmpty())
99  {
100  CApplication::processEventsFor(250);
101  if (this->confirmOverride(QStringLiteral("Override model cache for '%1'").arg(simulator.toQString())))
102  {
103  m_modelCaches.setModelsForSimulator(otherCache, simulator);
104  }
105  }
106  caches++;
107  }
108  } // all sims
109 
110  if (sets > 0 || caches > 0)
111  {
112  const CStatusMessage m = CStatusMessage(this).validationInfo(u"Copied %1 sets and %2 caches for '%3'") << sets << caches << selectedSimulators.toQString(true);
113  this->showOverlayHTMLMessage(m, 7500);
114  }
115  }

117  bool CCopyModelsFromOtherSwiftVersionsComponent::readDataFile(const QString &thisVersionModelFile, CAircraftModelList &models, const CApplicationInfo &otherVersion, const CSimulatorInfo &sim)
118  {
119  // init
120  models.clear();
121 
122  // create relative file name
123  QString relativeModelFile = thisVersionModelFile;
124  relativeModelFile = relativeModelFile.replace(CSwiftDirectories::applicationDataDirectory(), "", Qt::CaseInsensitive);
125  if (relativeModelFile.length() < 2) { return false; }
126  relativeModelFile = relativeModelFile.mid(relativeModelFile.indexOf('/', 1));
127 
128  const QString otherModelFile = CFileUtils::appendFilePathsAndFixUnc(otherVersion.getApplicationDataDirectory(), relativeModelFile);
129  const QFileInfo fiOtherModelFile(otherModelFile);
130  if (!fiOtherModelFile.exists())
131  {
132  ui->le_Status->setText(QStringLiteral("No models here: '%1'").arg(fiOtherModelFile.absoluteFilePath()));
133  return false;
134  }
135 
136  // read other file
137  const QString jsonString = CFileUtils::readFileToString(fiOtherModelFile.absoluteFilePath());
138  if (jsonString.isEmpty()) { return false; }
139  try
140  {
141  models = CAircraftModelList::fromMultipleJsonFormats(jsonString);
142  ui->tvp_AircraftModels->updateContainerAsync(models);
143  ui->le_Status->setText(QStringLiteral("Imported models: '%1'").arg(fiOtherModelFile.absoluteFilePath()));
144  }
145  catch (const CJsonException &ex)
146  {
147  this->showOverlayMessage(CStatusMessage::fromJsonException(ex, this, QStringLiteral("JSON format error. '%1'").arg(fiOtherModelFile.absoluteFilePath())));
148  return false;
149  }
150 
151  ui->le_Status->setText(QStringLiteral("Imported %1 models '%2' for %3").arg(models.size()).arg(fiOtherModelFile.fileName(), sim.toQString()));
152  return true;
153  }

155  bool CCopyModelsFromOtherSwiftVersionsComponent::confirmOverride(const QString &msg)
156  {
157  if (!sApp || sApp->isShuttingDown()) { return false; }

164  const QMessageBox::StandardButton reply = QMessageBox::question(this, QStringLiteral("Confirm override"), withQuestionMark(msg), QMessageBox::Yes | QMessageBox::No);
