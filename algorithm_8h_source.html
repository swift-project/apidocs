111  template <typename I, typename F>
112  void topologicalSort(I begin, I end, F comparator)
113  {
114  using value_type = typename std::iterator_traits<I>::value_type;
115  auto part = begin;
116  while (part != end)
117  {
118  auto newPart = std::partition(part, end, [=](const value_type &a) {
119  return std::none_of(part, end, [=, &a](const value_type &b) {
120  return comparator(b, a);
121  });
122  });
123  Q_ASSERT_X(newPart != part, "BlackMisc::topologicalSort", "Cyclic less-than relation detected (not a partial ordering)");
124  part = newPart;
125  }
