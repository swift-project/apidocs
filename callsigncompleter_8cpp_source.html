77  void CCallsignCompleter::updateCallsignsFromContext()
78  {
79  if (!sGui || sGui->isShuttingDown()) { return; }
80  if (!sGui->getIContextNetwork()) { return; }
81  if (completer()->wasUpdatedWithinTime(1500)) { return; } // avoid context call via DBus
82  CCallsignSet validCallsigns;
83  if (m_onlyWithParts)
84  {
86  validCallsigns = sGui->getIContextNetwork()->getAircraftInRange().getCallsignsWithSynchronizedParts();
87  }
88  else
89  {
90  validCallsigns = sGui->getIContextNetwork()->getAircraftInRangeCallsigns();
91  }
92 
93  if (m_addOwnCallsign && sGui->getIContextOwnAircraft())
94  {
95  const CCallsign ownCs = sGui->getIContextOwnAircraft()->getOwnAircraft().getCallsign();
96  if (!ownCs.isEmpty())
97  {
98  validCallsigns.insert(ownCs);
99  }
100  }
101 
102  const QStringList modelData = validCallsigns.getCallsignStrings(true);
103  completer()->updateData(modelData, 2000);
104  ui->led_Status->setTriState(500);
105  }
