140  bool CSimulatorFsxCommon::updateCOMFromSwiftToSimulator(const CFrequency &newFreq, const CFrequency &lastSimFreq, CFrequency &last25kHzSimFreq, EventIds id)
141  {
142  if (newFreq == lastSimFreq) { return false; }
143 
144  if (CComSystem::isExclusiveWithin8_33kHzChannel(newFreq) && last25kHzSimFreq.isNull())
145  {
146  // Switch from 25 to 8.33
147  // Store last 25 kHz frequency and do not send to simulator
148  last25kHzSimFreq = lastSimFreq;
149  return false;
150  }
151 
152  if (CComSystem::isWithin25kHzChannel(newFreq))
153  {
154  // Send to simulator
155  last25kHzSimFreq.setNull();
156  SimConnect_TransmitClientEvent(m_hSimConnect, 0, id,
157  CBcdConversions::comFrequencyToBcdHz(newFreq), SIMCONNECT_GROUP_PRIORITY_HIGHEST, SIMCONNECT_EVENT_FLAG_GROUPID_IS_PRIORITY);
158  return true;
159  }
160 
161  // Already 8.33 -> nothing to do
162  return false;
163  }

165  bool CSimulatorFsxCommon::updateOwnSimulatorCockpit(const CSimulatedAircraft &ownAircraft, const CIdentifier &originator)
166  {
167  if (originator == this->identifier()) { return false; }
168  if (!this->isSimulating()) { return false; }
169 
170  // actually those data should be the same as ownAircraft
171  const CComSystem newCom1 = ownAircraft.getCom1System();
172  const CComSystem newCom2 = ownAircraft.getCom2System();
173  const CTransponder newTransponder = ownAircraft.getTransponder();
174 
175  bool changed = false;
176 
177  changed |= updateCOMFromSwiftToSimulator(newCom1.getFrequencyActive(), m_simCom1.getFrequencyActive(), m_lastCom1Active, EventSetCom1Active);
178  changed |= updateCOMFromSwiftToSimulator(newCom1.getFrequencyStandby(), m_simCom1.getFrequencyStandby(), m_lastCom1Standby, EventSetCom1Standby);
179  changed |= updateCOMFromSwiftToSimulator(newCom2.getFrequencyActive(), m_simCom2.getFrequencyActive(), m_lastCom2Active, EventSetCom2Active);
180  changed |= updateCOMFromSwiftToSimulator(newCom2.getFrequencyStandby(), m_simCom2.getFrequencyStandby(), m_lastCom2Standby, EventSetCom2Standby);
181 
182  if (newTransponder.getTransponderMode() != m_simTransponder.getTransponderMode())
183  {
184  // use one way to transfer XPDR ident/mode not both
185  if (m_useSbOffsets)
186  {
187  byte ident = newTransponder.isIdentifying() ? 1U : 0U; // 1 is ident
188  byte standby = newTransponder.isInStandby() ? 1U : 0U; // 1 is standby
189  HRESULT hr = s_ok();
190 
191  hr += SimConnect_SetClientData(m_hSimConnect, ClientAreaSquawkBox, CSimConnectDefinitions::DataClientAreaSbIdent, SIMCONNECT_CLIENT_DATA_SET_FLAG_DEFAULT, 0, 1, &ident);
192  hr += SimConnect_SetClientData(m_hSimConnect, ClientAreaSquawkBox, CSimConnectDefinitions::DataClientAreaSbStandby, SIMCONNECT_CLIENT_DATA_SET_FLAG_DEFAULT, 0, 1, &standby);
193  if (isFailure(hr))
194  {
195  this->triggerAutoTraceSendId();
196  CLogMessage(this).warning(u"Setting transponder mode failed (SB offsets)");
197  }
198  else
199  {
200  if (m_logSbOffsets)
201  {
202  const QString lm = "SB sent: ident " % QString::number(ident) % u" standby " % QString::number(standby);
203  CLogMessage(this).info(lm);
204  }
205  }
206  changed = true;
207  }
208  else if (m_useFsuipc && m_fsuipc)
209  {
210  m_fsuipc->write(newTransponder);
211  changed = true;
212  }
213  }
214 
215  // avoid changes of cockpit back to old values due to an outdated read back value
216  if (changed) { m_skipCockpitUpdateCycles = SkipUpdateCyclesForCockpit; }
217 
218  // bye
219  return changed;
220  }

294  bool CSimulatorFsxCommon::requestElevation(const ICoordinateGeodetic &reference, const CCallsign &aircraftCallsign)
295  {
296  // this is the 32bit FSX version, the P3D x64 is overridden!
297 
298  if (this->isShuttingDownOrDisconnected()) { return false; }
299  if (!this->isUsingFsxTerrainProbe()) { return false; }
300  if (reference.isNull()) { return false; }
301  const CSimConnectObject simObject = m_simConnectObjects.getOldestNotPendingProbe(); // probes round robin
302  if (!simObject.isConfirmedAdded()) { return false; }
303  m_simConnectObjects[simObject.getCallsign()].resetTimestampToNow(); // mark probe as just used
304 
305  CCoordinateGeodetic pos(reference);
306  pos.setGeodeticHeight(terrainProbeAltitude());
307 
308  SIMCONNECT_DATA_INITPOSITION position = this->coordinateToFsxPosition(pos);
309  const HRESULT hr = this->logAndTraceSendId(
310  SimConnect_SetDataOnSimObject(
311  m_hSimConnect, CSimConnectDefinitions::DataRemoteAircraftSetPosition,
312  simObject.getObjectId(), 0, 0,
313  sizeof(SIMCONNECT_DATA_INITPOSITION), &position),
314  simObject, "Cannot request AI elevation", Q_FUNC_INFO, "SimConnect_SetDataOnSimObject");
315 
316  if (isFailure(hr)) { return false; }
317 
318  const bool ok = this->requestTerrainProbeData(simObject, aircraftCallsign);
319  if (ok)
320  {
321  emit this->requestedElevation(aircraftCallsign);
322  }
323  return ok;
324  }

1033

1982  {
1983  m_statsUpdateAircraftRuns = 0;
1984  return;
1985  }
1986 
1987  // values used for position and parts
1988  const qint64 currentTimestamp = QDateTime::currentMSecsSinceEpoch();

1990  {
1991  this->finishUpdateRemoteAircraftAndSetStatistics(currentTimestamp, true);
1992  return;

1995 
1996  // interpolation for all remote aircraft
1997  const QList<CSimConnectObject> simObjects(m_simConnectObjects.values());
1998 
1999  int simObjectNumber = 0;
2000  const bool traceSendId = this->isTracingSendId();
2001  const bool updateAllAircraft = this->isUpdateAllRemoteAircraft(currentTimestamp);
2002  for (const CSimConnectObject &simObject : simObjects)
2003  {
2004  // happening if aircraft is not yet added to simulator or to be deleted
2005  if (!simObject.isReadyToSend()) { continue; }
2006  if (!simObject.hasCurrentLightsInSimulator()) { continue; } // wait until we have light state
2007 
2008  const CCallsign callsign(simObject.getCallsign());
2009  const bool hasCs = !callsign.isEmpty();
2010  const bool hasValidIds = simObject.hasValidRequestAndObjectId();
2011  BLACK_VERIFY_X(hasCs, Q_FUNC_INFO, "missing callsign");
2012  BLACK_AUDIT_X(hasValidIds, Q_FUNC_INFO, "Missing ids");
2013  if (!hasCs || !hasValidIds) { continue; } // not supposed to happen
2014  const DWORD objectId = simObject.getObjectId();
2015 
2016  // setup
2017  const CInterpolationAndRenderingSetupPerCallsign setup = this->getInterpolationSetupConsolidated(callsign, updateAllAircraft);
2018  const bool sendGround = setup.isSendingGndFlagToSimulator();
2019 
2020  // Interpolated situation
2021  // simObjectNumber is passed to equally distributed steps like guessing parts
2022  const bool slowUpdate = (((m_statsUpdateAircraftRuns + simObjectNumber) % 40) == 0);
