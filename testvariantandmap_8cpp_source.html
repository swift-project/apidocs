165  void CTestVariantAndMap::valueMap()
166  {
167  // ATC station
168  QDateTime dtFrom = QDateTime::currentDateTimeUtc();
169  QDateTime dtUntil = dtFrom.addSecs(60 * 60); // 1 hour
170  CCoordinateGeodetic geoPos =
171  CCoordinateGeodetic::fromWgs84("48° 21′ 13″ N", "11° 47′ 09″ E", { 1487, CLengthUnit::ft() });
172  CAtcStation station1(CCallsign("eddm_twr"), CUser("123456", "Joe Doe"),
173  CFrequency(118.7, CFrequencyUnit::MHz()),
174  geoPos, CLength(50, CLengthUnit::km()), false, dtFrom, dtUntil);
175 
176  // value maps
177  CPropertyIndexVariantMap vmWildcard(true);
178  CPropertyIndexVariantMap vmNoWildcard(false);
179  CPropertyIndexVariantMap vm;
180  CPropertyIndexVariantMap vmCopy(vmWildcard);
181 
182  // Remark: Shortcoming here, as the callsign will be automatically set for user in station
183  // I have to set this as well, otherwise, no match.
184  vm.addValue(CAtcStation::IndexController, CUser("123456", "Joe Doe", CCallsign("EDDM_TWR")));
185 
186  // compare
187 
188  QVERIFY2(vmWildcard.matches(station1), "Station should be equal to wildcard");
189  QVERIFY2(!vmNoWildcard.matches(station1), "Station should not be equal to empty list");
190  QVERIFY2(vm.matches(station1), "Controller should match");
191  QVERIFY2(vmWildcard == vmCopy, "Maps should be equal");
192  QVERIFY2(qHash(vmWildcard) == qHash(vmCopy), "Hashs should be equal (simple)");
193 
194  vm.addValue(CAtcStation::IndexFrequency, CFrequency(118.7, CFrequencyUnit::MHz()));
195  vm.addValue(CAtcStation::IndexPosition, geoPos);
196  vmCopy = vm;
197  QVERIFY2(qHash(vm) == qHash(vmCopy), "Hashs should be equal (detailed)");
198  vmCopy.setWildcard(!vm.isWildcard());
199  QVERIFY2(qHash(vm) != qHash(vmCopy), "Hashs should not be equal (detailed)");
200  }
