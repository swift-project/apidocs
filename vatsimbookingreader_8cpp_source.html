59  void CVatsimBookingReader::read()
60  {
61  this->threadAssertCheck();
62  if (!this->doWorkCheck()) { return; }
63  if (!this->isInternetAccessible("No network/internet access, cannot read VATSIM bookings")) { return; }
64  if (this->isNotVATSIMEcosystem()) { return; }
65 
66  Q_ASSERT_X(sApp, Q_FUNC_INFO, "No application");
67  const QUrl url(sApp->getGlobalSetup().getVatsimBookingsUrl());
68  if (url.isEmpty()) { return; }
69  this->getFromNetworkAndLog(url, { this, &CVatsimBookingReader::parseBookings });
70  }

72  void CVatsimBookingReader::parseBookings(QNetworkReply *nwReplyPtr)
73  {
74  // wrap pointer, make sure any exit cleans up reply
75  // required to use delete later as object is created in a different thread
76  QScopedPointer<QNetworkReply, QScopedPointerDeleteLater> nwReply(nwReplyPtr);
77  this->threadAssertCheck();
78  if (this->isNotVATSIMEcosystem()) { return; }
79 
80  // Worker thread, make sure to write no members here od do it threadsafe
81  if (!this->doWorkCheck())
82  {
83  CLogMessage(this).info(u"Terminated booking parsing process");
84  return; // stop, terminate straight away, ending thread
85  }
86 
87  const CReaderSettings settings = m_settings.get();
88  this->logNetworkReplyReceived(nwReplyPtr);
89  const QUrl url = nwReply->url();
90 
91  if (nwReply->error() == QNetworkReply::NoError)
92  {
93  static const QString timestampFormat("yyyy-MM-dd HH:mm:ss");
94  const QString xmlData = nwReply->readAll();
95  nwReply->close(); // close asap
96 
97  if (xmlData.isEmpty() || !xmlData.contains("<bookings>", Qt::CaseInsensitive))
98  {
99  CLogMessage(this).warning(u"Reading bookings wrong XML format for '%1'") << nwReply->url().toString();
100  m_failures++;
101  emit this->dataRead(CEntityFlags::BookingEntity, CEntityFlags::ReadFailed, 0, url);
102  return;
103  }
104 
105  QDomDocument doc;
106  QDateTime updateTimestamp = QDateTime::currentDateTimeUtc(); // default
107  if (doc.setContent(xmlData))
108  {
109  const QDomNode timestamp = doc.elementsByTagName("timestamp").at(0);
110  const QString ts = timestamp.toElement().text().trimmed();
111  BLACK_AUDIT_X(!ts.isEmpty(), Q_FUNC_INFO, "Wrong timestamp format");
112 
113  if (ts.isEmpty())
114  {
115  CLogMessage(this).warning(u"Reading bookings wrong XML timestamp format for '%1'") << nwReply->url().toString();
116  m_failures++;
117  emit this->dataRead(CEntityFlags::BookingEntity, CEntityFlags::ReadFailed, 0, url);
118  return;
119  }
120  else
121  {
122  // normally the timestamp is always updated from backend
123  // if this changes in the future we're prepared
124  updateTimestamp = fromStringUtc(ts, timestampFormat);
125  if (this->getUpdateTimestamp() == updateTimestamp) return; // nothing to do
126 
127  // save parsing and all follow up actions if nothing changed
128  bool changed = this->didContentChange(xmlData, xmlData.indexOf("</timestamp>"));
129  if (!changed)
130  {
131  CLogMessage(this).info(u"Read bookings unchanged, skipped");
132  emit this->atcBookingsReadUnchanged();
133  return; // stop, terminate straight away, ending thread
134  }
135  }
136 
137  const QDomNode atc = doc.elementsByTagName("atcs").at(0);
138  const QDomNodeList bookingNodes = atc.toElement().elementsByTagName("booking");
139  int size = bookingNodes.size();
140  CAtcStationList bookedStations;
141  for (int i = 0; i < size; i++)
142  {
143  if (!this->doWorkCheck())
144  {
145  CLogMessage(this).info(u"Terminated booking parsing process"); // for users
146  return; // stop, terminate straight away, ending thread
147  }
148 
149  // pase nodes
150  const QDomNode bookingNode = bookingNodes.at(i);
151  const QDomNodeList bookingNodeValues = bookingNode.childNodes();
152  CAtcStation bookedStation;
153  CUser user;
154  for (int v = 0; v < bookingNodeValues.size(); v++)
155  {
156  const QDomNode bookingNodeValue = bookingNodeValues.at(v);
157  const QString name = bookingNodeValue.nodeName().toLower();
158  const QString value = bookingNodeValue.toElement().text();
159  if (name == "id")
160  {
161  // could be used as unique key
162  }
163  else if (name == "callsign")
164  {
165  bookedStation.setCallsign(CCallsign(value, CCallsign::Atc));
166  }
167  else if (name == "name")
168  {
169  user.setRealName(value);
170  }
171  else if (name == "cid")
172  {
173  user.setId(value);
174  }
175  else if (name == "time_end")
176  {
177  const QDateTime t = fromStringUtc(value, timestampFormat);
178  bookedStation.setBookedUntilUtc(t);
179  }
180  else if (name == "time_start")
181  {
182  const QDateTime t = fromStringUtc(value, timestampFormat);
183  bookedStation.setBookedFromUtc(t);
184  }
185  }
186  // time checks
187  const QDateTime now = QDateTime::currentDateTimeUtc();
188  if (now.msecsTo(bookedStation.getBookedUntilUtc()) < (1000 * 60 * 15)) { continue; } // until n mins in past
189  if (now.msecsTo(bookedStation.getBookedFromUtc()) > (1000 * 60 * 60 * 24)) { continue; } // to far in the future, n hours
190  bookedStation.setController(user);
191  bookedStations.push_back(bookedStation);
192  }
193  m_failures = 0;
194  this->setUpdateTimestamp(updateTimestamp); // thread safe update
195  this->setInitialAndPeriodicTime(settings.getInitialTime().toMs(), 3 * settings.getPeriodicTime().toMs()); // slow down, we have some bookings now
196 
197  emit this->atcBookingsRead(bookedStations);
198  emit this->dataRead(CEntityFlags::BookingEntity, CEntityFlags::ReadFinished, bookedStations.size(), url);
199  } // node
200  }
201  else
202  {
203  // network error
204  CLogMessage(this).warning(u"Reading bookings failed '%1' '%2'") << nwReply->errorString() << url.toString();
205  nwReply->abort();
206  m_failures++;
207  if (m_failures > 3)
208  {
209  // slow updates from now on
210  this->setInitialAndPeriodicTime(settings.getInitialTime().toMs(), 10 * settings.getPeriodicTime().toMs()); // massively slow down
211  CLogMessage(this).warning(u"Too many booking reader failures %1, slower updates") << m_failures;
212  }
213  emit this->dataRead(CEntityFlags::BookingEntity, CEntityFlags::ReadFailed, 0, url);
214  }
215  } // method
