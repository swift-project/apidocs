94  template <class F>
95  QList<QStringRef> splitStringRefs(const QString &s, F predicate)
96  {
97  QList<QStringRef> result;
98  auto notPredicate = [=](auto c) { return !predicate(c); };
99  auto begin = s.begin();
100  while (true)
101  {
102  begin = std::find_if(begin, s.end(), notPredicate);
103  if (begin == s.end()) { return result; }
104  auto end = std::find_if(begin, s.end(), predicate);
105  result.push_back(QStringRef(&s, std::distance(s.begin(), begin), std::distance(begin, end)));
106  begin = end;
107  }
108  }
