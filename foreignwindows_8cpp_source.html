76  bool CForeignWindows::setSimulatorAsParent(QWindow *simulatorWindow, QWidget *child)
77  {
78  if (!simulatorWindow) { return false; }
79  if (!child) { return false; }
80 
81  // If visible, hide it during the reparent. Otherwise setting the parent will have no effect.
82  const bool isVisible = child->isVisible();
83  if (isVisible) { child->hide(); }
84 
85  QWindow *childWindow = child->windowHandle();
86  Q_ASSERT_X(childWindow, Q_FUNC_INFO, "Native resources for child widget have not yet been allocated. Did you call QWidget::show() before?");
87 
88  childWindow->setTransientParent(simulatorWindow);
89 
90  // If it was visible before, make it visible again
91  if (isVisible) { child->show(); }
92  return true;
93  }
