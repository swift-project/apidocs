33  CAircraftIcaoCode CAircraftIcaoCodeList::findBestFuzzyMatchOrDefault(const QString &designator, int cutoff) const
34  {
35  if (designator.length() < CAircraftIcaoCode::DesignatorMinLength) { return CAircraftIcaoCode(); }
36  int best = 0;
37  int current = 0;
38  CAircraftIcaoCode found;
39  const QString d(designator.trimmed().toUpper());
40  for (const CAircraftIcaoCode &code : *this)
41  {
42  if (!code.matchesDesignator(d, cutoff, &current)) { continue; }
43  if (current == 100.0) { return code; }
44  if (best < current) { found = code; }
45  }
46  return found;
47  }

202  QStringList CAircraftIcaoCodeList::toCompleterStrings(bool withIataCodes, bool withFamily, bool withCategory, bool sort) const
203  {
204  QStringList c;
205  CAircraftIcaoCodeList icaos(*this);
206  if (sort) { icaos.sortByDesignatorAndRank(); }
207 
208  // 3 steps to get a proper sort order of the string list
209  for (const CAircraftIcaoCode &icao : std::as_const(icaos))
210  {
211  c.append(withCategory ? icao.getCombinedIcaoCategoryStringWithKey() : icao.getCombinedIcaoStringWithKey());
212  }
213 
214  if (withFamily)
215  {
216  const CAircraftIcaoCodeList icaosFamily = icaos.findWithFamily(true);
217  for (const CAircraftIcaoCode &icao : icaosFamily)
218  {
219  c.append(icao.getCombinedFamilyStringWithKey());
220  }
221  }
222 
223  if (withIataCodes)
224  {
225  icaos = icaos.findWithIataCode(true);
226  if (sort) { icaos.sortBy(&CAircraftIcaoCode::getIataCode, &CAircraftIcaoCode::getRank); }
227  for (const CAircraftIcaoCode &icao : std::as_const(icaos))
228  {
229  c.append(icao.getCombinedIataStringWithKey());
230  }
231  }
232 
233  return c;

306  CAircraftIcaoCodeList CAircraftIcaoCodeList::fromDatabaseJson(const QJsonArray &array, const CAircraftCategoryList &categories, bool ignoreIncompleteAndDuplicates, CAircraftIcaoCodeList *inconsistent)
307  {
308  CAircraftIcaoCodeList codes;
309  for (const QJsonValue &value : array)
310  {
311  CAircraftIcaoCode icao(CAircraftIcaoCode::fromDatabaseJson(value.toObject()));
312  const int catId = icao.getCategory().getDbKey();
313  if (!categories.isEmpty() && catId >= 0)
314  {
315  const CAircraftCategory category = categories.findByKey(catId);
316  if (!category.isNull())
317  {
318  icao.setCategory(category);
319  }
320  }
321 
322  if (!icao.hasSpecialDesignator() && !icao.hasCompleteData())
323  {
324  if (ignoreIncompleteAndDuplicates) { continue; }
325  if (inconsistent)
326  {
327  inconsistent->push_back(icao);
328  continue;
329  }
330  }
331  if (icao.isDbDuplicate())
332  {
333  if (ignoreIncompleteAndDuplicates) { continue; }
334  if (inconsistent)
335  {
336  inconsistent->push_back(icao);
337  continue;
338  }
339  }
340  codes.push_back(icao);
341  }
342  return codes;
343  }

345  CAircraftIcaoCode CAircraftIcaoCodeList::smartAircraftIcaoSelector(const CAircraftIcaoCode &icaoPattern) const
346  {
347  if (icaoPattern.hasValidDbKey())
348  {
349  const int k = icaoPattern.getDbKey();
350  const CAircraftIcaoCode c(this->findByKey(k));
351  if (c.hasCompleteData()) { return c; }
352  }
353 
354  // get an initial set of data we can choose from
355  const QString designator(icaoPattern.getDesignator());
356  if (designator.isEmpty()) { return CAircraftIcaoCode(); }
357  CAircraftIcaoCodeList codes;
358  do
359  {
360  codes = this->findByDesignator(designator);
361  if (!codes.isEmpty()) break;
362 
363  // now we search if the ICAO designator is actually an IATA code
364  codes = this->findByIataCode(designator);
365  if (!codes.isEmpty()) break;
366 
367  // search fuzzy and restrict length
368  const CAircraftIcaoCode bestMatch = this->findBestFuzzyMatchOrDefault(designator.length() < 5 ? designator : designator.left(5), 70);
369  if (bestMatch.hasValidDesignator()) { return bestMatch; }
370 
371  // still empty, try to find by family
372  codes = this->findByFamily(designator);
373  if (!codes.isEmpty()) break;
374 
375  // by any description
376  codes = this->findMatchingByAnyDescription(designator);
377  }
378  while (false);
379 
380  if (codes.isEmpty()) { return icaoPattern; }
381  if (codes.size() == 1) { return codes.front(); }
382 
383  // further reduce by manufacturer
384  codes.sortByRank();
385  if (icaoPattern.hasManufacturer() && codes.contains(&CAircraftIcaoCode::getManufacturer, icaoPattern.getManufacturer()))
386  {
387  const QString m(icaoPattern.getManufacturer());
388  codes = codes.findByManufacturer(m);
389  if (codes.size() == 1) { return codes.front(); }
390 
391  // intentionally continue here
392  }
393 
394  // further reduce by IATA
395  if (icaoPattern.hasIataCode() && codes.contains(&CAircraftIcaoCode::getIataCode, icaoPattern.getIataCode()))
396  {
397  const QString i(icaoPattern.getIataCode());
398  codes = codes.findByIataCode(i);
399  if (codes.size() == 1) { return codes.front(); }
400 
401  // intentionally continue here
402  }
403 
404  // lucky punch on description?
405  if (icaoPattern.hasModelDescription() && codes.contains(&CAircraftIcaoCode::getModelDescription, icaoPattern.getModelDescription()))
406  {
407  // do not affect codes here, it might return no results
408  const QString d(icaoPattern.getModelDescription());
409  CAircraftIcaoCodeList cm(codes.findByDescription(d));
410  if (cm.size() == 1) { return cm.front(); }
411  }
412  return codes.frontOrDefault(); // sorted by rank
413  }
