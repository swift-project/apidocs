67  QVariant CDbInfo::propertyByIndex(BlackMisc::CPropertyIndexRef index) const
68  {
69  if (index.isMyself()) { return QVariant::fromValue(*this); }
70  const ColumnIndex i = index.frontCasted<ColumnIndex>();
71  switch (i)
72  {
73  case IndexTableName: return QVariant::fromValue(m_tableName);
74  case IndexEntries: return QVariant::fromValue(m_entries);
75  case IndexEntity: return QVariant::fromValue(m_entity);
76  default:
77  return (IDatastoreObjectWithIntegerKey::canHandleIndex(index)) ?
78  IDatastoreObjectWithIntegerKey::propertyByIndex(index) :
79  CValueObject::propertyByIndex(index);
80  }

83  void CDbInfo::setPropertyByIndex(CPropertyIndexRef index, const QVariant &variant)
84  {
85  if (index.isMyself())
86  {
87  (*this) = variant.value<CDbInfo>();
88  return;
89  }
90  const ColumnIndex i = index.frontCasted<ColumnIndex>();
91  switch (i)
92  {
93  case IndexTableName: this->setTableName(variant.toString()); break;
94  case IndexEntries: this->setTableName(variant.toString()); break;
95  case IndexEntity: this->setEntity(static_cast<CEntityFlags::Entity>(variant.toInt())); break;
96  default:
97  if (IDatastoreObjectWithIntegerKey::canHandleIndex(index))
98  {
99  IDatastoreObjectWithIntegerKey::setPropertyByIndex(index, variant);
100  }
101  else
102  {
103  CValueObject::setPropertyByIndex(index, variant);
104  }
105  break;
106  }

109  int CDbInfo::comparePropertyByIndex(CPropertyIndexRef index, const CDbInfo &compareValue) const
110  {
111  if (index.isMyself()) { return getTableName().compare(compareValue.getTableName(), Qt::CaseInsensitive); }
112  if (IDatastoreObjectWithIntegerKey::canHandleIndex(index)) { return IDatastoreObjectWithIntegerKey::comparePropertyByIndex(index, compareValue); }
113  const ColumnIndex i = index.frontCasted<ColumnIndex>();
114  switch (i)
115  {
116  case IndexTableName: return this->getTableName().compare(compareValue.getTableName(), Qt::CaseInsensitive);
117  case IndexEntries: return Compare::compare(this->getEntries(), compareValue.getEntries());
118  case IndexEntity: return Compare::compare(this->getEntity(), compareValue.getEntity());
119  default:
120  Q_ASSERT_X(false, Q_FUNC_INFO, "No comparison possible");
121  }
122  return 0;
123  }

125  CDbInfo CDbInfo::fromDatabaseJson(const QJsonObject &json, const QString &prefix)
126  {
127  if (!existsKey(json, prefix))
128  {
129  // when using relationship, this can be null
130  return CDbInfo();
131  }
132  const int id(json.value(prefix % u"id").toInt());
133  const int entries(json.value(prefix % u"entries").toInt());
134  const QString tableName(json.value(prefix % u"tablename").toString());
135  CDbInfo dbInfo(id, tableName, entries);
136  dbInfo.setKeyVersionTimestampFromDatabaseJson(json, prefix);
137  return dbInfo;
138  }
