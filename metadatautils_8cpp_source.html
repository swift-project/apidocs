18  QString getAllUserMetatypesTypes(const QString &separator)
19  {
20  int fails = 0;
21  QString meta;
22  for (int mt = QMetaType::User; mt < QMetaType::User + 1000; mt++)
23  {
24  if (!QMetaType::isRegistered(mt))
25  {
26  fails++;
27  // normally a consecutive order of metatypes, we allow a space before we break
28  if (fails > 3) { break; }
29  continue;
30  }
31  QMetaType metaType(mt);
32  meta = meta.append("type: ").append(QString::number(mt)).append(" name: ").append(QMetaType::typeName(mt)).append(" | ").append(QString::number(QMetaType::sizeOf(mt))).append(" / ").append(QString::number(BlackMisc::heapSizeOf(metaType))).append(separator);
33  }
34  return meta;
35  }

61  size_t heapSizeOf(const QMetaType &metaType)
62  {
63  metaType.destroy(metaType.create()); // ignore one-off allocations of a class being instantiated for the first time
64  _CrtMemState oldState, newState, diff;
65  oldState.lTotalCount = newState.lTotalCount = diff.lTotalCount = 0; // avoid compiler warning
66  diff.lSizes[_NORMAL_BLOCK] = 0;
67  _CrtMemCheckpoint(&oldState);
68  void *p = metaType.create();
69  _CrtMemCheckpoint(&newState);
70  metaType.destroy(p);
71  _CrtMemDifference(&diff, &oldState, &newState);
72  return diff.lSizes[_NORMAL_BLOCK];
73  }
