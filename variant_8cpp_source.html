138  int CVariant::compareImpl(const CVariant &a, const CVariant &b)
139  {
140  if (a.userType() < b.userType()) { return -1; }
141  if (a.userType() > b.userType()) { return 1; }
142  auto *aMeta = a.getValueObjectMetaInfo();
143  auto *bMeta = b.getValueObjectMetaInfo();
144  if (aMeta && bMeta)
145  {
146  try
147  {
148  const void *casted = nullptr;
149  if ((casted = aMeta->upCastTo(a.data(), bMeta->getMetaTypeId())))
150  {
151  return bMeta->compareImpl(casted, b.data());
152  }
153  else if ((casted = bMeta->upCastTo(b.data(), aMeta->getMetaTypeId())))
154  {
155  return aMeta->compareImpl(a.data(), casted);
156  }
157  else
158  {
159  CLogMessage(&a).warning(u"Comparing two CVariants containing unrelated value objects: %1 (%2) and %3 (%4)")
160  << a.typeName() << a.userType() << b.typeName() << b.userType();
161  return 0;
162  }
163  }
164  catch (const Private::CVariantException &ex)
165  {
166  CLogMessage(&a).debug() << ex.what();
167  return 0;
168  }
169  }
170  if (a.m_v < b.m_v) { return -1; }
171  if (a.m_v > b.m_v) { return 1; }
172  return 0;
173  }

186  case QVariant::Double: json.insert("value", m_v.toDouble()); break;
187  case QVariant::LongLong: json.insert("value", m_v.toLongLong()); break;
188  case QVariant::ULongLong: json.insert("value", m_v.toLongLong()); break;
189  case QVariant::String: json.insert("value", m_v.toString()); break;
190  case QVariant::Char: json.insert("value", m_v.toString()); break;
191  case QVariant::ByteArray: json.insert("value", m_v.toString()); break;
192  case QVariant::DateTime: json.insert("value", m_v.toDateTime().toString(Qt::ISODate)); break;
193  case QVariant::Date: json.insert("value", m_v.toDate().toString(Qt::ISODate)); break;

227  void CVariant::convertFromJson(const QJsonObject &json)
228  {
229  // Remark: Names "type" and "value" are also used for drag and drop
230  // Changing the names here requires the change for drag and drop too
231  const QJsonValue typeValue = json.value("type");
232  if (typeValue.isUndefined()) { throw CJsonException("Missing 'type'"); }
233  const QString typeName = typeValue.toString();
234  if (typeName.isEmpty())
235  {
236  m_v.clear();
237  return;
238  }
239  const int typeId = QMetaType::type(qPrintable(typeName));
240 
241  const QJsonValue value = json.value("value");
242  if (value.isUndefined()) { throw CJsonException("Missing 'value'"); }
243  switch (typeId)
244  {
245  case QVariant::Invalid: throw CJsonException("Type not recognized by QMetaType");
246  case QVariant::Int: m_v.setValue(value.toInt()); break;
247  case QVariant::UInt: m_v.setValue<uint>(static_cast<uint>(value.toInt())); break;
248  case QVariant::Bool: m_v.setValue(value.toBool()); break;
249  case QVariant::Double: m_v.setValue(value.toDouble()); break;
250  case QVariant::LongLong: m_v.setValue(static_cast<qlonglong>(value.toDouble())); break;
251  case QVariant::ULongLong: m_v.setValue(static_cast<qulonglong>(value.toDouble())); break;
252  case QVariant::String: m_v.setValue(value.toString()); break;
253  case QVariant::Char: m_v.setValue(value.toString().size() > 0 ? value.toString().at(0) : '\0'); break;
254  case QVariant::ByteArray: m_v.setValue(value.toString().toLatin1()); break;
255  case QVariant::DateTime: m_v.setValue(fromStringUtc(value.toString(), Qt::ISODate)); break;
256  case QVariant::Date: m_v.setValue(QDate::fromString(value.toString(), Qt::ISODate)); break;
257  case QVariant::Time: m_v.setValue(QTime::fromString(value.toString(), Qt::ISODate)); break;
258  case QVariant::StringList: m_v.setValue(QVariant(value.toArray().toVariantList()).toStringList()); break;
259  default:
260  try
261  {
262  auto *meta = Private::getValueObjectMetaInfo(typeId);
263  if (meta)
264  {
265  CJsonScope scope("value"); // tracker
266  Q_UNUSED(scope);
267  m_v = QVariant(typeId, nullptr);
268 
269  // this will call convertFromJson if there is no MemoizedJson
270  meta->convertFromMemoizedJson(value.toObject(), data(), true);
271  }
272  else if (QMetaType::hasRegisteredConverterFunction(qMetaTypeId<QString>(), typeId))
273  {
274  m_v.setValue(value.toString());
275  if (!m_v.convert(typeId))
276  {
277  throw CJsonException("Failed to convert from JSON string");
278  }
279  }
280  else
281  {
282  throw CJsonException("Type not supported by convertFromJson");
283  }
284  }
285  catch (const Private::CVariantException &ex)
286  {
287  throw CJsonException(ex.what());
288  }
289  }
290  }

576  QVariant fixQVariantFromDbusArgument(const QVariant &variant, int localUserType, const QString &typeName)
577  {
578  if (localUserType == static_cast<int>(QVariant::Invalid))
579  {
580  CLogMessage(&variant).warning(u"Invalid type for unmarshall: %1") << typeName;
581  }
582 
583  // my business?
584  if (!variant.canConvert<QDBusArgument>()) { return variant; }
585 
586  // complex, user type
587  // it has to be made sure, that the cast works
588  const QDBusArgument arg = variant.value<QDBusArgument>();
589  constexpr int userType = static_cast<int>(QVariant::UserType);
590  if (localUserType < userType)
591  {
592  // complex Qt type, e.g. QDateTime
593  return complexQtTypeFromDbusArgument(arg, localUserType);
594  }
595  else if (QMetaType(localUserType).flags() & QMetaType::IsEnumeration)
596  {
597  arg.beginStructure();
598  int i;
599  arg >> i;
600  arg.endStructure();
601 
602  QVariant valueVariant = QVariant::fromValue(i);
603  bool ok = valueVariant.convert(localUserType);
604  Q_ASSERT_X(ok, Q_FUNC_INFO, "int could not be converted to enum");
605  Q_UNUSED(ok);
606  return valueVariant;
607  }
608  else
609  {
610  QVariant valueVariant(localUserType, nullptr);
611  auto *meta = Private::getValueObjectMetaInfo(valueVariant);
612  if (meta)
613  {
614  meta->unmarshall(arg, valueVariant.data());
615  return valueVariant;
616  }
617  Q_ASSERT_X(false, Q_FUNC_INFO, "no meta");
618  return valueVariant;
619  }
620  }

622  QVariant complexQtTypeFromDbusArgument(const QDBusArgument &argument, int type)
623  {
624  // QDate = 14, QTime = 15, QDateTime = 16, QUrl = 17,
625 
626  switch (type)
627  {
628  case QMetaType::QDateTime:
629  {
630  QDateTime dt;
631  argument >> dt;
632  return QVariant::fromValue(dt);
633  }
634  case QMetaType::QDate:
635  {
636  QDate date;
637  argument >> date;
638  return QVariant::fromValue(date);
639  }
640  case QMetaType::QTime:
641  {
642  QTime time;
643  argument >> time;
644  return QVariant::fromValue(time);
645  }
646  default:
647  {
648  const char *name = QMetaType::typeName(type);
649  qFatal("Type cannot be resolved: %s (%d)", name ? name : "", type);
650  }
651  }
652  return QVariant(); // suppress compiler warning
