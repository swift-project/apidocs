60  CAircraftParts CAircraftPartsForm::getAircraftPartsFromJson() const
61  {
62  const QString jsonParts = ui->te_AircraftPartsJson->toPlainText().trimmed();
63  CAircraftParts parts;
64  if (jsonParts.isEmpty())
65  {
66  CLogMessage(this).validationError(u"No JSON content");
67  return parts;
68  }
69  QJsonParseError jsonError;
70  QJsonDocument jsonDoc(QJsonDocument::fromJson(jsonParts.toUtf8(), &jsonError));
71  if (jsonError.error != QJsonParseError::NoError)
72  {
73  CLogMessage(this).validationError(u"Parse error: %1") << jsonError.errorString();
74  return parts;
75  }
76 
77  try
78  {
79  parts.convertFromJson(jsonDoc.object());
80  }
81  catch (const CJsonException &ex)
82  {
83  CLogMessage(this).preformatted(CStatusMessage::fromJsonException(ex, this, "Parse error"));
84  return parts;
85  }
86  return parts;
87  }
