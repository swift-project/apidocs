92  void CDbAirlineIcaoSelectorComponent::onDataChanged()
93  {
94  if (!sGui) { return; }
95  QString s(ui->le_Airline->text());
96  if (s.isEmpty()) { return; }
97  CAirlineIcaoCode icao;
98  int dbKey = CDatastoreUtility::extractIntegerKey(s);
99  if (dbKey >= 0)
100  {
101  icao = sGui->getWebDataServices()->getAirlineIcaoCodeForDbKey(dbKey);
102  }
103  else
104  {
105  const QString designator = this->getRawDesignator();
106  icao = sGui->getWebDataServices()->smartAirlineIcaoSelector(CAirlineIcaoCode(designator));
107  }
108  this->setAirlineIcao(icao);
109  }
