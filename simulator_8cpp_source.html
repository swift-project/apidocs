77  bool ISimulator::logicallyAddRemoteAircraft(const CSimulatedAircraft &remoteAircraft)
78  {
79  if (!this->validateModelOfAircraft(remoteAircraft))
80  {
81  const CCallsign cs = remoteAircraft.getCallsign();
82  CLogMessage(this).warning(u"Invalid aircraft detected, which will be disabled: '%1' '%2'") << cs << remoteAircraft.getModelString();
83  this->updateAircraftEnabled(cs, false);
84  this->updateAircraftRendered(cs, false);
85  return false;
86  }
87 
88  // no invalid model should ever reach this place here
89  const bool renderingRestricted = this->getInterpolationSetupGlobal().isRenderingRestricted();
90  if (this->showDebugLogMessage()) { this->debugLogMessage(Q_FUNC_INFO, QStringLiteral("Restricted: %1 cs: '%2' enabled: %3").arg(boolToYesNo(renderingRestricted), remoteAircraft.getCallsignAsString(), boolToYesNo(remoteAircraft.isEnabled()))); }
91  if (!remoteAircraft.isEnabled()) { return false; }
92 
93  // if not restriced, directly change
94  if (!renderingRestricted)
95  {
96  this->callPhysicallyAddRemoteAircraft(remoteAircraft);
97  return true;
98  }
99 
100  // restricted -> will be added with next snapshot onRecalculatedRenderedAircraft
101  return false;
102  }

350  {

355  if (QDateTime::currentMSecsSinceEpoch() > m_highlightEndTimeMsEpoch)
356  {
357  this->stopHighlighting();

359  }
360 
361  // blink mode, toggle aircraft
362  for (const CSimulatedAircraft &aircraft : std::as_const(m_highlightedAircraft))
363  {
364  if (m_blinkCycle) { this->callPhysicallyRemoveRemoteAircraft(aircraft.getCallsign(), true); }
365  else { this->callPhysicallyAddRemoteAircraft(aircraft); }

426  bool ISimulator::parseCommandLine(const QString &commandLine, const CIdentifier &originator)
427  {
428  if (this->isMyIdentifier(originator)) { return false; }
429  if (this->isShuttingDown()) { return false; }
430 
431  if (commandLine.isEmpty()) { return false; }
432  CSimpleCommandParser parser({ ".plugin", ".drv", ".driver" });
433  parser.parse(commandLine);
434  if (!parser.isKnownCommand()) { return false; }
435 
436  // .plugin unload
437  if (parser.matchesPart(1, "unload"))
438  {
439  this->unload();
440  return true;
441  }
442 
443  // .plugin log interpolator
444  const QString part1(parser.part(1).toLower().trimmed());
445  if (part1.startsWith("logint") && parser.hasPart(2))
446  {
447  const QString part2 = parser.part(2).toLower();
448  if (part2 == "off" || part2 == "false")
449  {
450  CLogMessage(this).info(u"Disabled interpolation logging");
451  this->clearInterpolationLogCallsigns();
452  return true;
453  }
454  if (part2 == "clear" || part2 == "clr")
455  {
456  m_interpolationLogger.clearLog();
457  CLogMessage(this).info(u"Cleared interpolation logging");
458  this->clearInterpolationLogCallsigns();
459  return true;
460  }
461  if (part2.startsWith("max"))
462  {
463  if (!parser.hasPart(3)) { return false; }
464  bool ok;
465  const int max = parser.part(3).toInt(&ok);
466  if (!ok) { return false; }
467  m_interpolationLogger.setMaxSituations(max);
468  CLogMessage(this).info(u"Max.situations logged: %1") << max;
469  return true;
470  }
471  if (part2 == "write" || part2 == "save")
472  {
473  // stop logging of other log
474  this->clearInterpolationLogCallsigns();
475 
476  // write
477  const bool clearLog = true;
478  m_interpolationLogger.writeLogInBackground(clearLog);
479  CLogMessage(this).info(u"Started writing interpolation log");
480  return true;
481  }
482  if (part2 == "show")
483  {
484  const QDir dir(CInterpolationLogger::getLogDirectory());
485  if (CDirectoryUtils::isDirExisting(dir))
486  {
487  const QUrl dirUrl = QUrl::fromLocalFile(dir.absolutePath());
488  QDesktopServices::openUrl(dirUrl); // show dir in browser
489  }
490  else
491  {
492  CLogMessage(this).warning(u"No interpolation log directory");
493  }
494  return true;
495  }
496 
497  const CCallsign cs(part2.toUpper());
498  if (!cs.isValid()) { return false; }
499  if (this->getAircraftInRangeCallsigns().contains(cs))
500  {
501  CLogMessage(this).info(u"Will log interpolation for '%1'") << cs.asString();
502  this->setLogCallsign(true, cs);
503  return true;
504  }
505  else
506  {
507  CLogMessage(this).warning(u"Cannot log interpolation for '%1', no aircraft in range") << cs.asString();
508  return false;
509  }
510  } // logint
511 
512  if (part1.startsWith("spline") || part1.startsWith("linear"))
513  {
514  if (parser.hasPart(2))
515  {
516  const CCallsign cs(parser.part(2));
517  const bool changed = this->setInterpolationMode(part1, cs);
518  CLogMessage(this).info(changed ? QStringLiteral("Changed interpolation mode for '%1'") : QStringLiteral("Unchanged interpolation mode for '%1'")) << cs.asString();
519  return true;
520  }
521  else
522  {
523  CInterpolationAndRenderingSetupGlobal setup = this->getInterpolationSetupGlobal();
524  const bool changed = setup.setInterpolatorMode(part1);
525  if (changed) { this->setInterpolationSetupGlobal(setup); }
526  CLogMessage(this).info(changed ? QStringLiteral("Changed interpolation mode globally") : QStringLiteral("Unchanged interpolation mode"));
527  return true;
528  }
529  } // spline/linear
530 
531  if (part1.startsWith("pos"))
532  {
533  CCallsign cs(parser.part(2).toUpper());
534  if (!cs.isValid())
535  {
536  const CCallsignSet csSet = this->getLogCallsigns();
537  if (csSet.size() != 1) { return false; }
538 
539  // if there is just one we take that one
540  cs = *csSet.begin();
541  }
542 
543  this->setLogCallsign(true, cs);
544  CLogMessage(this).info(u"Display position for '%1'") << cs.asString();
545  this->displayLoggedSituationInSimulator(cs, true);
546  return true;
547  }
548 
549  if (parser.hasPart(2) && (part1.startsWith("aircraft") || part1.startsWith("ac")))
550  {
551  const QString part2 = parser.part(2).toLower();
552  if (parser.hasPart(3) && (part2.startsWith("readd") || part2.startsWith("re-add")))
553  {
554  const QString cs = parser.part(3).toUpper();
555  if (cs == "all")
556  {
557  this->physicallyRemoveAllRemoteAircraft();
558  const CStatusMessageList msgs = this->debugVerifyStateAfterAllAircraftRemoved();
559  this->clearAllRemoteAircraftData(); // "dot command"
560  if (!msgs.isEmpty()) { emit this->driverMessages(msgs); }
561  const CSimulatedAircraftList aircraft = this->getAircraftInRange();
562  for (const CSimulatedAircraft &a : aircraft)
563  {
564  if (a.isEnabled()) { this->logicallyAddRemoteAircraft(a); }
565  }
566  }
567  else if (CCallsign::isValidAircraftCallsign(cs))
568  {
569  this->logicallyReAddRemoteAircraft(cs);
570  return true;
571  }
572  return false;
573  }
574  if (parser.hasPart(3) && (part2.startsWith("rm") || part2.startsWith("remove")))
575  {
576  const QString cs = parser.part(3).toUpper();
577  if (CCallsign::isValidAircraftCallsign(cs))
578  {
579  this->logicallyRemoveRemoteAircraft(cs);
580  }
581  }
582 
583  return false;
584  }
585 
586  if (part1.startsWith("limit"))
587  {
588  const int perSecond = parser.toInt(2, -1);
589  this->limitToUpdatesPerSecond(perSecond);
590  CLogMessage(this).info(u"Remote aircraft updates limitations: %1") << this->updateAircraftLimitationInfo();
591  return true;
592  }
593 
594  // CG override
595  if (part1 == QStringView(u"cg"))
596  {
597  if (parser.part(2).startsWith("clear", Qt::CaseInsensitive))
598  {
599  CLogMessage(this).info(u"Clear all overridden CGs");
600  const CLengthPerCallsign cgsPerCallsign = this->clearCGOverrides();
601 
602  // restore all CGs
603  for (const CCallsign &cs : this->getAircraftInRangeCallsigns())
604  {
605  // reset CGs per callsign
606  const CLength cg = cgsPerCallsign.contains(cs) ? cgsPerCallsign[cs] : CLength::null();
607  this->updateCG(cs, cg);
608  }
609  return true;
610  }
611 
612  if (parser.hasPart(3))
613  {
614  // ms can be a string like "B773 B773_RR SDM"
615  const QString ms = parser.partAndRemainingStringAfter(3).toUpper();
616  CLength cg;
617  cg.parseFromString(parser.part(2), CPqString::SeparatorBestGuess);
618  if (!ms.isEmpty())
619  {
620  CLogMessage(this).info(u"Setting CG for '%1': %2") << ms << cg.valueRoundedWithUnit();
621  const bool set = this->insertCGForModelStringOverridden(cg, ms);
622  if (set)
623  {
624  const CCallsignSet callsigns = this->updateCGForModel(ms, cg);
625  if (!callsigns.isEmpty())
626  {
627  this->insertCGOverridden(cg, callsigns);
628  CLogMessage(this).info(u"Setting CG for '%1': %2") << callsigns.getCallsignsAsString(true) << cg.valueRoundedWithUnit();
629  }
630  return true;
631 
632  } // set
633  } // model string
634 
635  } // 3 parts
636  }
637 
638  // driver specific cmd line arguments
639  return this->parseDetails(parser);
640  }

801  void ISimulator::onRecalculatedRenderedAircraft(const CAirspaceAircraftSnapshot &snapshot)
802  {
803  if (!snapshot.isValidSnapshot()) { return; }
804 
805  // for unrestricted values all add/remove actions are directly linked
806  // when changing back from restricted->unrestricted an one time update is required
807  if (!snapshot.isRestricted() && !snapshot.isRestrictionChanged()) { return; }
808 
809  Q_ASSERT_X(CThreadUtils::isInThisThread(this), Q_FUNC_INFO, "Needs to run in object thread");
810  Q_ASSERT_X(snapshot.generatingThreadName() != QThread::currentThread()->objectName(), Q_FUNC_INFO, "Expect snapshot from background thread");
811 
812  // restricted snapshot values?
813  bool changed = false;
814  if (snapshot.isRenderingEnabled())
815  {
816  // make sure not to add aircraft again which are no longer in range
817  const CCallsignSet callsignsInRange = this->getAircraftInRangeCallsigns();
818  const CCallsignSet callsignsEnabledAndStillInRange = snapshot.getEnabledAircraftCallsignsByDistance().intersection(callsignsInRange);
819  const CCallsignSet callsignsInSimulator(this->physicallyRenderedAircraft()); // state in simulator
820  const CCallsignSet callsignsToBeRemoved(callsignsInSimulator.difference(callsignsEnabledAndStillInRange));
821  const CCallsignSet callsignsToBeAdded(callsignsEnabledAndStillInRange.difference(callsignsInSimulator));
822  if (!callsignsToBeRemoved.isEmpty())
823  {
824  const int r = this->physicallyRemoveMultipleRemoteAircraft(callsignsToBeRemoved);
825  changed = r > 0;
826  }
827 
828  if (!callsignsToBeAdded.isEmpty())
829  {
830  CSimulatedAircraftList aircraftToBeAdded(this->getAircraftInRange().findByCallsigns(callsignsToBeAdded)); // thread safe copy
831  for (const CSimulatedAircraft &aircraft : aircraftToBeAdded)
832  {
833  Q_ASSERT_X(aircraft.isEnabled(), Q_FUNC_INFO, "Disabled aircraft detected as to be added");
834  Q_ASSERT_X(aircraft.hasModelString(), Q_FUNC_INFO, "Missing model string");
835  this->callPhysicallyAddRemoteAircraft(aircraft); // recalcuate snapshot
836  changed = true;
837  }
838  }
839  }
840  else
841  {
842  // no rendering at all, we remove everything
843  const int r = this->physicallyRemoveAllRemoteAircraft();
844  changed = r > 0;
845  }
846 
847  // we have handled snapshot
848  if (changed)
849  {
850  emit this->airspaceSnapshotHandled();
851  }
852  }
