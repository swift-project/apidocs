41  void messageHandler(QtMsgType type, const QMessageLogContext &context, const QString &message)
42  {
43  if (type == QtFatalMsg && CLogHandler::instance()->thread() != QThread::currentThread())
44  {
45  // Fatal message means this thread is about to crash the application. A queued connection would be useless.
46  // Blocking queued connection means we pause this thread just long enough to let the main thread handle the message.
47  QMetaObject::invokeMethod(
48  CLogHandler::instance(), [&] { messageHandler(type, context, message); }, Qt::BlockingQueuedConnection);
49  return;
50  }
51 #if defined(Q_CC_MSVC) && defined(QT_NO_DEBUG)
52  if (type == QtFatalMsg)
53  {
54  struct EventFilter : public QAbstractNativeEventFilter
55  {
56  // Prevent Qt from handling Windows Messages while the messagebox is open
57  virtual bool nativeEventFilter(const QByteArray &, void *message, long *result) override
58  {
59  auto msg = static_cast<MSG *>(message);
60  *result = DefWindowProc(msg->hwnd, msg->message, msg->wParam, msg->lParam);
61  return true;
62  }
63  };
64  EventFilter ef;
65  qApp->installNativeEventFilter(&ef);
66  MessageBoxW(nullptr, message.toStdWString().c_str(), nullptr, MB_OK); // display assert dialog in release build
67  qApp->removeNativeEventFilter(&ef);
68 # if defined(BLACK_USE_CRASHPAD)
69  CRASHPAD_SIMULATE_CRASH(); // workaround inability to catch __fastfail
70 # endif
71  }
72 #endif
73  QMetaObject::invokeMethod(CLogHandler::instance(), [statusMessage = CStatusMessage(type, context, message)] {
74  CLogHandler::instance()->logLocalMessage(statusMessage);
75  });
76  }

96  CLogPatternHandler *CLogHandler::handlerForPattern(const CLogPattern &pattern)
97  {
98  Q_ASSERT(thread() == QThread::currentThread());
99 
100  auto finder = [&](const PatternPair &pair) { return pair.first == pattern; };
101  auto comparator = [](const PatternPair &a, const PatternPair &b) { return a.first.isProperSubsetOf(b.first); };
102 
103  auto it = std::find_if(m_patternHandlers.cbegin(), m_patternHandlers.cend(), finder);
104  if (it == m_patternHandlers.cend())
105  {
106  auto *handler = new CLogPatternHandler(this, pattern);
107  topologicallySortedInsert(m_patternHandlers, PatternPair(pattern, handler), comparator);
108  return handler;
109  }
110  else
111  {
112  return (*it).second;
113  }
114  }

146  void CLogHandler::logLocalMessage(const CStatusMessage &i_statusMessage)
147  {
148  using namespace BlackConfig;
149  CStatusMessage statusMessage = i_statusMessage;
150  if (!CBuildConfig::isLocalDeveloperDebugBuild() && CLogPattern::empty().withSeverity(CStatusMessage::SeverityError).match(statusMessage))
151  {
152  // 99% this is a complex Qt implementation warning generated by qErrnoWarning, so downgrade its severity
153  statusMessage.setSeverity(CStatusMessage::SeverityDebug);
154  }
155 
156  if (!CBuildConfig::isLocalDeveloperDebugBuild() && CLogPattern::exactMatch("default").withSeverity(CStatusMessage::SeverityWarning).match(statusMessage))
157  {
158  // All Qt warnings
159 
160  // demoted because caused by airline icons, we would need to re-create literally dozens of these images
161  if (statusMessage.getMessage().startsWith(QStringLiteral("libpng warning"))) { statusMessage.setSeverity(CStatusMessage::SeverityDebug); }
162 
163  // demoted, because in some swift APPs some options can be ignored
164  else if (statusMessage.getMessage().startsWith(QStringLiteral("QCommandLineParser: option not defined"))) { statusMessage.setSeverity(CStatusMessage::SeverityDebug); }
165  }
166 
167  auto bucket = m_tokenBuckets.find(statusMessage);
168  if (bucket == m_tokenBuckets.end()) { bucket = m_tokenBuckets.insert(statusMessage, { { 5, 1000, 1 }, 0 }); }
169  if (!bucket->first.tryConsume())
170  {
171  bucket->second++;
172  return;
173  }
174  if (bucket->second > 0)
175  {
176  auto copy = statusMessage;
177  copy.appendMessage(QStringLiteral(" (+%1 identical messages)").arg(bucket->second));
178  bucket->second = 0;
179 
180  logMessage(copy);
181  emit localMessageLogged(copy);
182  return;
183  }
184 
185  logMessage(statusMessage);
186  emit localMessageLogged(statusMessage);
187  }

202  void CLogHandler::logMessage(const CStatusMessage &statusMessage)
203  {
204  const auto handlers = handlersForMessage(statusMessage);
205 
206  if (isFallThroughEnabled(handlers))
207  {
208  Q_ASSERT_X(m_oldHandler, Q_FUNC_INFO, "Handler must be installed");
209  QtMsgType type;
210  QString category;
211  QString message;
212  statusMessage.toQtLogTriple(&type, &category, &message);
213  m_oldHandler(type, QMessageLogContext(nullptr, 0, nullptr, qPrintable(category)), message);
214  }
215 
216  for (auto *handler : handlers)
217  {
218  emit handler->messageLogged(statusMessage);
219  }

253  void CLogPatternHandler::updateSubscription()
254  {
255  if (m_subscriptionNeedsUpdate)
256  {
257  m_subscriptionNeedsUpdate = false;
258  bool isSubscribed = isSignalConnected(QMetaMethod::fromSignal(&CLogPatternHandler::messageLogged));
259 
260  if (isSubscribed != m_isSubscribed)
261  {
262  m_isSubscribed = isSubscribed;
263  if (m_isSubscribed)
264  {
265  emit m_parent->subscriptionAdded(m_pattern);
266  }
267  else

280  void CLogSubscriber::changeSubscription(const CLogPattern &pattern)
281  {
282  if (CLogHandler::instance()->thread() != QThread::currentThread())
283  {
284  Q_ASSERT(thread() == QThread::currentThread());
285  singleShot(0, CLogHandler::instance(), [pattern, self = QPointer<CLogSubscriber>(this)]() { if (self) { self->changeSubscription(pattern); } });
286  return;
287  }
288 
289  unsubscribe();
290  m_handler = CLogHandler::instance()->handlerForPattern(pattern);
291 
292  if (!m_inheritFallThrough)
293  {
294  m_handler->enableConsoleOutput(m_enableFallThrough);
295  }
296  connect(m_handler.data(), &CLogPatternHandler::messageLogged, this, &CLogSubscriber::ps_logMessage, Qt::DirectConnection);
297  }
