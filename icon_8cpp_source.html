54  QPixmap CIcon::toPixmap() const
55  {
56  if (this->isSet())
57  {
58  if (this->isGenerated()) { return m_pixmap; }
59  if (this->isFileBased()) { return m_pixmap; }
60  return CIcons::pixmapByIndex(this->getIndex(), m_rotateDegrees);
61  }
62  else
63  {
64  return CIcons::pixmapByIndex(this->getIndex());
65  }
66  }
