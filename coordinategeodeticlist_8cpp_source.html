20  CElevationPlane CCoordinateGeodeticList::averageGeodeticHeight(const CCoordinateGeodetic &reference, const CLength &range, const CLength &maxDeviation, int minValues, int sufficentValues) const
21  {
22  if (this->size() < minValues) { return CElevationPlane::null(); } // no chance to succeed
23 
24  QList<double> valuesInFt;
25  const CCoordinateGeodeticList sorted = this->findWithGeodeticMSLHeight().findWithinRange(reference, range).sortedByEuclideanDistanceSquared(reference);
26  if (sorted.size() < minValues) { return CElevationPlane::null(); }
27 
28  // we know all values have MSL and are within range
29  for (const CCoordinateGeodetic &coordinate : sorted)
30  {
31  const double elvFt = coordinate.geodeticHeight().value(CLengthUnit::ft());
32  valuesInFt.push_back(elvFt);
33  if (valuesInFt.size() >= sufficentValues) { break; }
34  }
35 
36  if (valuesInFt.size() < minValues) { return CElevationPlane::null(); }
37 
38  const double MaxDevFt = maxDeviation.value(CLengthUnit::ft());
39  const QPair<double, double> elvStdDevMean = CMathUtils::standardDeviationAndMean(valuesInFt);
40  if (elvStdDevMean.first > MaxDevFt) { return CElevationPlane::null(); }
41  return CElevationPlane(reference, elvStdDevMean.second, CElevationPlane::singlePointRadius());
42  }
