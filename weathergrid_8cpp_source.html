44  CWeatherGrid CWeatherGrid::findWithinRange(const ICoordinateGeodetic &coordinate, const PhysicalQuantities::CLength &range) const
45  {

51  CWeatherGrid CWeatherGrid::findClosest(int number, const ICoordinateGeodetic &coordinate) const
52  {
53  CWeatherGrid closest = partiallySorted(number, [&](const CGridPoint &a, const CGridPoint &b) {
54  return calculateEuclideanDistanceSquared(a.getPosition(), coordinate) < calculateEuclideanDistanceSquared(b.getPosition(), coordinate);
55  });
56  closest.truncate(number);
57  return closest;
58  }

60  QString CWeatherGrid::getDescription(const QString &sep) const
61  {
62  QString s;
63  QTextStream qtout(&s);
64  for (const CGridPoint &gridPoint : *this)
65  {
66  qtout << "Latitude: " << gridPoint.getPosition().latitude().toQString() << sep;
67  qtout << "Longitude: " << gridPoint.getPosition().longitude().toQString() << sep;
68  qtout << " MSL Pressure: " << gridPoint.getPressureAtMsl().toQString() << sep;
69 
70  CTemperatureLayerList temperatureLayers = gridPoint.getTemperatureLayers();
71  temperatureLayers.sort([](const CTemperatureLayer &a, const CTemperatureLayer &b) { return a.getLevel() < b.getLevel(); });
72  qtout << " Temperature Layers: " << sep;
73  for (const auto &temperatureLayer : std::as_const(temperatureLayers))
74  {
75  qtout << " Level: " << temperatureLayer.getLevel().toQString() << sep;
76  qtout << " Temperature: " << temperatureLayer.getTemperature().toQString() << sep;
77  qtout << " Relative Humidity: " << temperatureLayer.getRelativeHumidity() << " %" << sep;
78  }
79  qtout << sep;
80 
81  CWindLayerList windLayers = gridPoint.getWindLayers();
82  windLayers.sort([](const CWindLayer &a, const CWindLayer &b) { return a.getLevel() < b.getLevel(); });
83  qtout << " Wind Layers: " << sep;
84  for (const auto &windLayer : std::as_const(windLayers))
85  {
86  qtout << " Level: " << windLayer.getLevel().toQString() << sep;
87  qtout << " Wind: " << windLayer.getDirection().toQString() << " at " << windLayer.getSpeed().toQString() << sep;
88  }
89  qtout << sep;
90 
91  qtout << " Cloud Layers: " << sep;
92  CCloudLayerList cloudLayers = gridPoint.getCloudLayers();
93  cloudLayers.sort([](const CCloudLayer &a, const CCloudLayer &b) { return a.getBase() < b.getBase(); });
94  for (int i = 0; i < cloudLayers.size(); i++)
95  {
96  const CCloudLayer &cloudLayer = cloudLayers[i];
97  qtout << " Top: " << cloudLayer.getTop().toQString() << sep;
98  qtout << " Coverage: " << cloudLayer.getCoveragePercent() << " %" << sep;
99  qtout << " Precipitation type: " << cloudLayer.getPrecipitation() << sep;
100  qtout << " Precipitation rate: " << cloudLayer.getPrecipitationRate() << sep;
101  qtout << " Base: " << cloudLayer.getBase().toQString() << sep;
102  }
103  qtout << sep << sep;
104  }
105 
106  qtout.flush();
107  return s;
108  }

120  const CWeatherGrid &CWeatherGrid::getByScenario(const CWeatherScenario &scenario)
121  {
122  static const CWeatherGrid emptyGrid {};
123  switch (scenario.getIndex())
124  {
125  case CWeatherScenario::ClearSky: return getClearWeatherGrid();
126  case CWeatherScenario::Thunderstorm: return getThunderStormGrid();
127  default:
128  // in release versions just return, no need to ASSERT
129  BLACK_VERIFY_X(!CBuildConfig::isLocalDeveloperDebugBuild(), Q_FUNC_INFO, "Unknown fixed scenario index requested.");
