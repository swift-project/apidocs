56  void CDbLiveryComponent::onLiveriesRead(CEntityFlags::Entity entity, CEntityFlags::ReadState readState, int count)
57  {
58  Q_UNUSED(count);
59  if (!sGui || sGui->isShuttingDown() || !sGui->getWebDataServices()) { return; }
60  if (!entity.testFlag(CEntityFlags::LiveryEntity)) { return; }
61 
62  if (CEntityFlags::isFinishedReadState(readState))
63  {
64  this->showOverlayHTMLMessage(QStringLiteral("Updating %1").arg(CEntityFlags::entitiesToString(entity)), 2000);
65  ui->tvp_Liveries->updateContainerMaybeAsync(sGui->getWebDataServices()->getLiveries());
66  }
67  else
68  {
69  this->showOverlayHTMLMessage(u"Current state: " % CEntityFlags::entitiesToString(entity) % u" " % CEntityFlags::stateToString(readState), 10000);
70  }
71  }
