40  const QPixmap &CPixmap::pixmap() const
41  {
42  QWriteLocker lock(&this->m_lock);
43  if (this->m_hasCachedPixmap) { return this->m_pixmap; }
44 
45  // this part here becomes relevant when marshalling via DBus is used
46  // in this case only the array is transferred
47  this->m_hasCachedPixmap = true;
48  if (this->m_array.isEmpty()) { return this->m_pixmap; }
49  bool s = this->m_pixmap.loadFromData(this->m_array, "PNG");
50  Q_ASSERT(s);
51  Q_UNUSED(s);
