89  bool IPluginManager::tryLoad(const QString &path)
90  {
91  if (!QLibrary::isLibrary(path)) { return false; }
92 
93  CLogMessage(this).debug() << "Try loading plugin:" << path;
94  QPluginLoader loader(path);
95  const QJsonObject json = loader.metaData();
96  if (!isValid(json))
97  {
98  CLogMessage(this).warning(u"Plugin '%1' invalid, not loading it") << path;
99  return false;
100  }
101 
102  const QString identifier = pluginIdentifier(json);
103  m_paths.insert(identifier, path);
104  m_metadata.push_back(json);
105 
106  if (json.value("MetaData").toObject().contains("config"))
107  {
108  const QString configId = json.value("MetaData").toObject().value("config").toString();
109  if (!configId.isEmpty())
110  {
111  m_configs.insert(identifier, configId);
112  }
113  }
114 
115  return true;
116  }

118  QObject *IPluginManager::getPluginByIdImpl(const QString &identifier)
119  {
120  if (m_instances.contains(identifier))
121  {
122  return m_instances.value(identifier);
123  }
124 
125  if (!m_paths.contains(identifier))
126  {
127  CLogMessage(this).warning(u"Plugin with id '%1' does not exist") << identifier;
128  return nullptr;
129  }
130 
131  QString path = m_paths.value(identifier);
132  QPluginLoader loader(path);
133  QObject *instance = loader.instance();
134  if (instance)
135  {
136  m_instances.insert(identifier, instance);
137  m_instanceIds.insert(instance, identifier);
138  return instance;
139  }
140  else
141  {
142  CLogMessage(this).error(loader.errorString());
143  return nullptr;
144  }
