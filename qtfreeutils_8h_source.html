70  inline std::vector<std::string> split(const std::string &str, size_t maxSplitCount = 0, const std::string &delimiter = " ")
71  {
72  std::string s(str);
73  size_t pos = 0;
74  std::vector<std::string> tokens;
75  while ((pos = s.find(delimiter)) != std::string::npos)
76  {
77  tokens.push_back(s.substr(0, pos));
78  s.erase(0, pos + delimiter.length());
79  if (maxSplitCount > 0 && tokens.size() == maxSplitCount) { break; }
80  }
81  tokens.push_back(s);
82  return tokens;
83  }

86  inline double normalizeValue(const double value, const double start, const double end)
87  {
88  // https://stackoverflow.com/questions/1628386/normalise-orientation-between-0-and-360
89  if (value >= start && value <= end) { return value; }
90  const double width = end - start;
91  const double offsetValue = value - start; // value relative to 0
92  return (offsetValue - (floor(offsetValue / width) * width)) + start;
93  }

167  inline std::string simplifyWhitespace(const std::string &s)
168  {
169  std::string result;
170  for (char c : s)
171  {
172  if (std::isspace(c))
173  {
174  if (!result.empty() && result.back() != ' ') { result.push_back(' '); }
175  }
176  else { result.push_back(c); }
177  }
178  while (!result.empty() && result.back() == ' ') { result.pop_back(); }
179  return result;
180  }

183  inline AcfProperties extractAcfProperties(const std::string &filePath)
184  {
185  std::ifstream fs(filePath, std::ios::in);
186  if (!fs.is_open()) { return {}; }
187 
188  std::string i;
189  std::string version;
190  std::string acf;
191  std::getline(fs, i);
192  i = toLower(i);
193  std::getline(fs, version);
194  version = toLower(version);
195  std::getline(fs, acf);
196  acf = toLower(acf);
197 
198  AcfProperties acfProperties;
199 
200  if (i == "i" && version.find("version") != std::string::npos && acf == "acf")
201  {
202  std::string line;
203  while (std::getline(fs, line))
204  {
205  auto tokens = split(line, 2);
206  if (tokens.size() < 3 || tokens.at(0) != "P") { continue; }
207 
208  if (tokens.at(1) == "acf/_ICAO")
209  {
210  acfProperties.aircraftIcaoCode = tokens.at(2);
211  }
212  else if (tokens.at(1) == "acf/_descrip")
213  {
214  acfProperties.modelDescription = "[ACF] " + tokens.at(2);
215  }
216  else if (tokens.at(1) == "acf/_name")
217  {
218  acfProperties.modelName = tokens.at(2);
219  }
220  else if (tokens.at(1) == "acf/_studio")
221  {
222  acfProperties.author = tokens.at(2);
223  }
224  else if (tokens.at(1) == "acf/_author")
225  {
226  if (!acfProperties.author.empty()) { continue; }
227  std::string author = tokens.at(2);
228  size_t pos = author.find_first_not_of("ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789_");
229  author = author.substr(0, pos);
230  if (author.empty()) { continue; }
231  acfProperties.author = author;
232  }
233  }
234  }
235 
236  fs.close();
237  acfProperties.modelString = simplifyWhitespace(stringForFlyableModel(acfProperties, filePath));
238  return acfProperties;
239  }
