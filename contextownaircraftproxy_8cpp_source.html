29  void CContextOwnAircraftProxy::relaySignals(const QString &serviceName, QDBusConnection &connection)
30  {
31  bool s = connection.connect(serviceName, IContextOwnAircraft::ObjectPath(), IContextOwnAircraft::InterfaceName(),
32  "changedAircraftCockpit", this, SIGNAL(changedAircraftCockpit(BlackMisc::Simulation::CSimulatedAircraft, BlackMisc::CIdentifier)));
33  Q_ASSERT(s);
34  s = connection.connect(serviceName, IContextOwnAircraft::ObjectPath(), IContextOwnAircraft::InterfaceName(),
35  "changedSelcal", this, SIGNAL(changedSelcal(BlackMisc::Aviation::CSelcal, BlackMisc::CIdentifier)));
36  Q_ASSERT(s);
37  s = connection.connect(serviceName, IContextOwnAircraft::ObjectPath(), IContextOwnAircraft::InterfaceName(),
38  "changedCallsign", this, SIGNAL(changedCallsign(BlackMisc::Aviation::CCallsign)));
39  Q_ASSERT(s);
40  s = connection.connect(serviceName, IContextOwnAircraft::ObjectPath(), IContextOwnAircraft::InterfaceName(),
41  "changedAircraftIcaoCodes", this, SIGNAL(changedAircraftIcaoCodes(BlackMisc::Aviation::CAircraftIcaoCode, BlackMisc::Aviation::CAirlineIcaoCode)));
42  Q_ASSERT(s);
43  s = connection.connect(serviceName, IContextOwnAircraft::ObjectPath(), IContextOwnAircraft::InterfaceName(),
44  "changedPilot", this, SIGNAL(changedPilot(BlackMisc::Network::CUser)));
45  Q_ASSERT(s);
46  s = connection.connect(serviceName, IContextOwnAircraft::ObjectPath(), IContextOwnAircraft::InterfaceName(),
47  "movedAircraft", this, SIGNAL(movedAircraft(BlackMisc::PhysicalQuantities::CLength)));
48  Q_ASSERT(s);
49  s = connection.connect(serviceName, IContextOwnAircraft::ObjectPath(), IContextOwnAircraft::InterfaceName(),
50  "isTakingOff", this, SIGNAL(isTakingOff()));
51  Q_ASSERT(s);
52  s = connection.connect(serviceName, IContextOwnAircraft::ObjectPath(), IContextOwnAircraft::InterfaceName(),
53  "isTouchingDown", this, SIGNAL(isTouchingDown()));
54  Q_ASSERT(s);
55 
56  this->relayBaseClassSignals(serviceName, connection, IContextOwnAircraft::ObjectPath(), IContextOwnAircraft::InterfaceName());
57  Q_UNUSED(s)
58  }
