23  struct CService::FramePeriodSampler : public CDrawable
24  {
25  DataRef<xplane::data::sim::operation::misc::frame_rate_period> m_thisFramePeriod;
26  DataRef<xplane::data::sim::time::framerate_period> m_thisFramePeriodXP11;
27  DataRef<xplane::data::sim::time::total_flight_time_sec> m_secondsSinceReset;
28  DataRef<xplane::data::sim::flightmodel::position::groundspeed> m_groundSpeed;
29 
30  std::vector<float> m_samples;
31  float m_total = 0;
32  float m_totalOverBudget = 0;
33  float m_totalMetersShort = 0;
34  float m_totalSecondsLate = 0;
35  size_t m_lastSampleIndex = 0;
36  static constexpr size_t c_maxSampleCount = 500;
37  static constexpr float c_framePeriodBudget = 0.05f;
38 
39  FramePeriodSampler() : CDrawable(xplm_Phase_Window, true) {}
40 
41  std::tuple<float, float, float, float> getFrameStats()
42  {
43  if (m_total < 0.001f) { return {}; } // no DIV by 0
44  const float fps = m_samples.size() / m_total;
45  const float ratio = 1 - m_totalOverBudget / m_total;
46  const float miles = m_totalMetersShort / 1852.0f;
47  const float minutes = m_totalSecondsLate / 60.0f;
48  m_total = 0;
49  m_totalOverBudget = 0;
50  m_samples.clear();
51  m_lastSampleIndex = 0;
52  return std::make_tuple(fps, ratio, miles, minutes);
53  }
54 
55  protected:
56  virtual void draw() override // called once per frame
57  {
58  const float current = m_thisFramePeriodXP11.isValid() ? m_thisFramePeriodXP11.get() : m_thisFramePeriod.get();
59 
60  ++m_lastSampleIndex %= c_maxSampleCount;
61  if (m_samples.size() == c_maxSampleCount)
62  {
63  auto &oldSample = m_samples[m_lastSampleIndex];
64  m_total -= oldSample;
65  if (oldSample > c_framePeriodBudget)
66  {
67  m_totalOverBudget -= oldSample - c_framePeriodBudget;
68  }
69  oldSample = current;
70  }
71  else { m_samples.push_back(current); }
72 
73  m_total += current;
74  if (current > c_framePeriodBudget)
75  {
76  m_totalOverBudget += current - c_framePeriodBudget;
77 
78  if (m_secondsSinceReset.get() > 10)
79  {
80  const float metersShort = m_groundSpeed.get() * std::max(0.0f, current - c_framePeriodBudget);
81  m_totalMetersShort += metersShort;
82  if (m_groundSpeed.get() > 1.0f) { m_totalSecondsLate += std::max(0.0f, current - c_framePeriodBudget); }
83  }
84  }
85  }
86  };

143  void CService::addTextMessage(const std::string &text, double red, double green, double blue)
144  {
145  if (text.empty()) { return; }
146  static const CMessage::string ellipsis = u8"\u2026";
147  const unsigned lineLength = m_messages.maxLineLength() - 1;
148 
149  using U8It = Utf8Iterator<std::string::const_iterator>;
150  U8It begin(text.begin(), text.end());
151  auto characters = std::distance(begin, U8It(text.end(), text.end()));
152  std::vector<CMessage::string> wrappedLines;
153 
154  for (; characters > lineLength; characters -= lineLength)
155  {
156  auto end = std::next(begin, lineLength);
157  wrappedLines.emplace_back(begin.base, end.base);
158  wrappedLines.back() += ellipsis;
159  begin = end;
160  }
161  if (characters > 0)
162  {
163  wrappedLines.emplace_back(begin.base, text.end());
164  }
165  for (const auto &line : wrappedLines)
166  {
167  m_messages.addMessage({ line, static_cast<float>(red), static_cast<float>(green), static_cast<float>(blue) });
168  }
169 
170  if (!m_messages.isVisible() && m_popupMessageWindow) { m_messages.toggle(); }
171 
172  if (m_disappearMessageWindow)
173  {
174  m_disappearMessageWindowTime = std::chrono::system_clock::now() + std::chrono::milliseconds(std::max(m_disapperMessageWindowTimeMs, 1500));
175  }
176  }

273  void CService::readAirportsDatabase()
274  {
275  auto first = XPLMFindFirstNavAidOfType(xplm_Nav_Airport);
276  auto last = XPLMFindLastNavAidOfType(xplm_Nav_Airport);
277  if (first != XPLM_NAV_NOT_FOUND)
278  {
279  for (auto i = first; i <= last; ++i)
280  {
281  float lat, lon;
282  char icao[32];
283  XPLMGetNavAidInfo(i, nullptr, &lat, &lon, nullptr, nullptr, nullptr, icao, nullptr, nullptr);
284  if (icao[0] != 0)
285  {
286  m_airports.emplace_back(i, lat, lon);
287  }
288  }
289  }
290  }

292  void CService::updateAirportsInRange()
293  {
294  if (m_airports.empty())
295  {
296  readAirportsDatabase();
297  }
298  std::vector<std::string> icaos, names;
299  std::vector<double> lats, lons, alts;
300  std::vector<CNavDataReference> closestAirports = findClosestAirports(20, getLatitudeDeg(), getLongitudeDeg());
301  for (const auto &navref : closestAirports)
302  {
303  float lat, lon, alt;
304  char icao[32], name[256];
305  XPLMGetNavAidInfo(navref.id(), nullptr, &lat, &lon, &alt, nullptr, nullptr, icao, name, nullptr);
306  icaos.emplace_back(icao);
307  names.emplace_back(name);
308  lats.push_back(lat);
309  lons.push_back(lon);
310  alts.push_back(alt);
311  }
312  emitAirportsInRangeUpdated(icaos, names, lats, lons, alts);
313  }

1020
