39  void CAtcButtonComponent::updateStations()
40  {
41  this->setVisible(false);
42  this->setMinimumHeight(0);
43 
44  if (!sGui || !sGui->getIContextNetwork()) { return; }
45  if (m_maxNumber < 1) { return; }
46  const int max = qMin(m_maxNumber, m_rows * m_cols);
47  CAtcStationList stations = sGui->getIContextNetwork()->getClosestAtcStationsOnline(max);
48  if (stations.isEmpty()) { return; }
49 
50  CGuiUtility::deleteLayout(this->layout(), true);
51  QGridLayout *layout = new QGridLayout(this);
52 
53  layout->setObjectName("gl_CAtcButtonComponent");
54  layout->setSpacing(4);
55  layout->setMargin(0);
56  layout->setContentsMargins(0, 0, 0, 0);
57 
58  int row = 0;
59  int col = 0;
60  int added = 0;
61  for (const CAtcStation &station : std::as_const(stations))
62  {
63  if (m_ignoreNonAtc)
64  {
65  // strict check, only "real ATC stations", no supervisors etc
66  if (!station.getCallsign().hasAtcSuffix()) { continue; }
67  }
68 
69  QPushButton *button = new QPushButton(this);
70  button->setText(station.getCallsignAsString());
71  if (m_withIcons) { button->setIcon(CIcon(station.toIcon()).toQIcon()); }
72  QObject::connect(button, &QPushButton::released, this, &CAtcButtonComponent::onButtonClicked);
73  const CVariant atcv = CVariant::fromValue(station);
74  layout->addWidget(button, row, col++);
75  button->show();
76  button->setProperty("atc", atcv.getQVariant());
77  added++;
78 
79  if (col >= m_cols)
80  {
81  if (row == m_rows) { break; }
82  row++;
83  col = 0;
84  }
85  }
86 
87  const double a = added;
88  const double c = m_cols;
89  const int rows = qRound(ceil(a / c)); // row can be too high
90  if (added > 0)
91  {
92  this->setVisible(true);
93  this->setMinimumHeight(rows * 25);
94  }
95  }
