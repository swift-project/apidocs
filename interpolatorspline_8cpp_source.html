24  template <size_t N>
25  std::array<double, N> solveTridiagonal(std::array<std::array<double, N>, N> &matrix, std::array<double, N> &d)
26  {
27  // *INDENT-OFF*
28  const auto a = [&matrix](size_t i) -> double & { return matrix[i][i - 1]; }; // subdiagonal
29  const auto b = [&matrix](size_t i) -> double & { return matrix[i][i]; }; // main diagonal
30  const auto c = [&matrix](size_t i) -> double & { return matrix[i][i + 1]; }; // superdiagonal
31 
32  // forward sweep
33  c(0) /= b(0);
34  d[0] /= b(0);
35  for (size_t i = 1; i < N; ++i)
36  {
37  const double denom = b(i) - a(i) * c(i - 1);
38  if (i < N - 1) { c(i) /= denom; }
39  d[i] = (d[i] - a(i) * d[i - 1]) / denom;
40  }
41 
42  // back substitution
43  for (int i = N - 2; i >= 0; --i)
44  {
45  const size_t it = static_cast<size_t>(i);
46  d[it] -= c(it) * d[it + 1];
47  }
48  return d;
49  // *INDENT-ON*
50  }

55  template <size_t N>
56  std::array<double, N> getDerivatives(const std::array<double, N> &x, const std::array<double, N> &y)
57  {
58  std::array<std::array<double, N>, N> a { {} };
59  std::array<double, N> b { {} };
60 
61  // *INDENT-OFF*
62  a[0][0] = 2.0 / (x[1] - x[0]);
63  a[0][1] = 1.0 / (x[1] - x[0]);
64  b[0] = 3.0 * (y[1] - y[0]) / ((x[1] - x[0]) * (x[1] - x[0]));
65 
66  a[N - 1][N - 2] = 1.0 / (x[N - 1] - x[N - 2]);
67  a[N - 1][N - 1] = 2.0 / (x[N - 1] - x[N - 2]);
68  b[N - 1] = 3.0 * (y[N - 1] - y[N - 2]) / ((x[N - 1] - x[N - 2]) * (x[N - 1] - x[N - 2]));
69 
70  for (size_t i = 1; i < N - 1; ++i)
71  {
72  a[i][i - 1] = 1.0 / (x[i] - x[i - 1]);
73  a[i][i] = 2.0 / (x[i] - x[i - 1]) + 2.0 / (x[i + 1] - x[i]);
74  a[i][i + 1] = 1.0 / (x[i + 1] - x[i]);
75  b[i] = 3.0 * (y[i] - y[i - 1]) / ((x[i] - x[i - 1]) * (x[i] - x[i - 1])) + 3.0 * (y[i + 1] - y[i]) / ((x[i + 1] - x[i]) * (x[i + 1] - x[i]));
76  }
77  // *INDENT-ON*
78 
79  solveTridiagonal(a, b);
80  return b;
81  }

84  double evalSplineInterval(double x, double x0, double x1, double y0, double y1, double k0, double k1)
85  {
86  const double t = (x - x0) / (x1 - x0);
87  const double a = k0 * (x1 - x0) - (y1 - y0);
88  const double b = -k1 * (x1 - x0) + (y1 - y0);
89  const double y = (1 - t) * y0 + t * y1 + t * (1 - t) * (a * (1 - t) + b * t);
90 
91  if (CBuildConfig::isLocalDeveloperDebugBuild())
92  {
93  BLACK_VERIFY_X(t >= 0, Q_FUNC_INFO, "Expect t >= 0");
94  BLACK_VERIFY_X(t <= 1.0, Q_FUNC_INFO, "Expect t <= 1");
95  }
96  return y;
97  }

100  bool CInterpolatorSpline::fillSituationsArray()
101  {
102  // m_s[0] .. oldest -> m_[2] .. latest
103  // general idea, we interpolate from current situation -> latest situation
104 
105  // do we have the last interpolated situation?
106  if (m_lastSituation.isNull())
107  {
108  if (m_currentSituations.isEmpty())
109  {
110  // nothing we can do
111  m_s[0] = m_s[1] = m_s[2] = CAircraftSituation::null();
112  return false;
113  }
114  else
115  {
116  // we start with the latest situation just to init the values
117  CAircraftSituation f = m_currentSituations.front();
118  f.setAdjustedMSecsSinceEpoch(m_currentTimeMsSinceEpoch); // adjusted time exactly "now"
119  m_s[0] = m_s[1] = m_s[2] = f;
120  }
121  }
122  else
123  {
124  // in normal cases init some default values
125  m_s[0] = m_s[1] = m_s[2] = m_lastSituation; // current position
126  }
127 
128  // set some default values
129  const qint64 os = qMax(CFsdSetup::c_minimumPositionTimeOffsetMsec, m_s[2].getTimeOffsetMs());
130  m_s[0].addMsecs(-os); // oldest, Ref T297 default offset time to fill data
131  m_s[2].addMsecs(os); // latest, Ref T297 default offset time to fill data
132  if (m_currentSituations.isEmpty()) { return false; }
133 
134  // and use the real values if available
135  // m_s[0] .. oldest -> m_[2] .. latest
136  const CAircraftSituation latest = m_currentSituations.front();
137  if (latest.isNewerThanAdjusted(m_s[1])) { m_s[2] = latest; }
138  const qint64 currentAdjusted = m_s[1].getAdjustedMSecsSinceEpoch();
139 
140  // with https://dev.swift-project.org/T668#15841 avoid 2 very close positions
141  // currently done by time, maybe we can also choose distance
142  const qint64 osNotTooClose = qRound64(0.8 * os);
143  const CAircraftSituation older = m_currentSituations.findObjectBeforeAdjustedOrDefault(currentAdjusted - osNotTooClose);
144  if (!older.isNull())
145  {
146  m_s[0] = older;
147  }
148  else
149  {
150  const CAircraftSituation closeOlder = m_currentSituations.findObjectBeforeAdjustedOrDefault(currentAdjusted);
151  if (!closeOlder.isNull()) { m_s[0] = closeOlder; }
152  }
153  const qint64 latestAdjusted = m_s[2].getAdjustedMSecsSinceEpoch();
154  const qint64 olderAdjusted = m_s[0].getAdjustedMSecsSinceEpoch();
155 
156  // not having a new situation itself is quite normal,
157  // only if it persits it is critical.
158  const bool hasNewer = latestAdjusted > m_currentTimeMsSinceEpoch;
159 
160  if (CBuildConfig::isLocalDeveloperDebugBuild())
161  {
162  const bool verified = this->verifyInterpolationSituations(m_s[0], m_s[1], m_s[2]); // oldest -> latest, only verify order
163  if (!verified)
164  {
165  static const QString vm("Unverified situations, m0-2 (oldest latest) %1 %2 %3");
166  const QString vmValues = vm.arg(olderAdjusted).arg(currentAdjusted).arg(latestAdjusted);
167  CLogMessage(this).warning(vmValues);
168  Q_UNUSED(vmValues)
169  }
170  }
171  return hasNewer;
172  }

178  CInterpolatorSpline::CInterpolant CInterpolatorSpline::getInterpolant(SituationLog &log)
179  {
180  // recalculate derivatives only if they changed
181  // m_situationsLastModified updated in initIniterpolationStepData
182  const bool recalculate = (m_currentTimeMsSinceEpoch >= m_nextSampleAdjustedTime) || // new step
183  (m_situationsLastModified > m_situationsLastModifiedUsed); // modified
184 
185  if (recalculate)
186  {
187  // with the latest updates of T243 the order and the offsets are supposed to be correct
188  // so even mixing fast/slow updates shall work
189  m_situationsLastModifiedUsed = m_situationsLastModified;
190  const bool fillStatus = this->fillSituationsArray();
191  if (!fillStatus)
192  {
193  m_interpolant.setValid(false);
194  return m_interpolant;
195  }
196 
197  const std::array<std::array<double, 3>, 3> normals { { m_s[0].getPosition().normalVectorDouble(), m_s[1].getPosition().normalVectorDouble(), m_s[2].getPosition().normalVectorDouble() } };
198  PosArray pa;
199  pa.x = { { normals[0][0], normals[1][0], normals[2][0] } }; // oldest -> latest
200  pa.y = { { normals[0][1], normals[1][1], normals[2][1] } };
201  pa.z = { { normals[0][2], normals[1][2], normals[2][2] } }; // latest
202  pa.t = { { static_cast<double>(m_s[0].getAdjustedMSecsSinceEpoch()), static_cast<double>(m_s[1].getAdjustedMSecsSinceEpoch()), static_cast<double>(m_s[2].getAdjustedMSecsSinceEpoch()) } };
203 
204  pa.dx = getDerivatives(pa.t, pa.x);
205  pa.dy = getDerivatives(pa.t, pa.y);
206  pa.dz = getDerivatives(pa.t, pa.z);
207 
208  // - altitude unit must be the same for all three, but the unit itself does not matter
209  // - ground elevantion here normally is not available
210  // - some info how fast a plane moves: 100km/h => 1sec 27,7m => 5 secs 136m
211  // - on an airport the plane does not move very fast, or not at all
212  // - and the elevation remains (almost) constant for a wider area
213  // - during flying the ground elevation not really matters
214  this->updateElevations(true);
215  static const CLengthUnit altUnit = CAltitude::defaultUnit();
216  const CLength cg(this->getModelCG().switchedUnit(altUnit));
217  const double a0 = m_s[0].getCorrectedAltitude(cg).value(altUnit); // oldest
218  const double a1 = m_s[1].getCorrectedAltitude(cg).value(altUnit);
219  const double a2 = m_s[2].getCorrectedAltitude(cg).value(altUnit); // latest
220  pa.a = { { a0, a1, a2 } };
221  pa.gnd = { { m_s[0].getOnGroundFactor(), m_s[1].getOnGroundFactor(), m_s[2].getOnGroundFactor() } };
222  pa.da = getDerivatives(pa.t, pa.a);
223  pa.dgnd = getDerivatives(pa.t, pa.gnd);
224 
225  m_prevSampleAdjustedTime = m_s[1].getAdjustedMSecsSinceEpoch();
226  m_nextSampleAdjustedTime = m_s[2].getAdjustedMSecsSinceEpoch(); // latest
227  m_prevSampleTime = m_s[1].getMSecsSinceEpoch(); // last interpolated situation normally
228  m_nextSampleTime = m_s[2].getMSecsSinceEpoch(); // latest
229  m_interpolant = CInterpolant(pa, altUnit, CInterpolatorPbh(m_s[1], m_s[2])); // older, newer
230  Q_ASSERT_X(m_prevSampleAdjustedTime < m_nextSampleAdjustedTime, Q_FUNC_INFO, "Wrong time order");
231  }
232 
233  // Example:
234  // prev.sample time 5 (received at 0) , next sample time 10 (received at 5)
235  // cur.time 6: dt1=6-5=1, dt2=5 => fraction 1/5
236  // cur.time 9: dt1=9-5=4, dt2=5 => fraction 4/5
237  //
238  // we use different offset times for interim pos. updates
239  // prev.sample time 5 (received at 0) , 7/r:5, 10 (rec. at 5)
240  // cur.time 6: dt1=6-5=1, dt2=7-5 => fraction 1/2
241  // cur.time 9: dt1=9-7=2, dt2=10-7=3 => fraction 2/3
242  // we use different offset times for fast pos. updates
243  // KB: is that correct with dt2, or would it be m_nextSampleTime - m_prevSampleTime
244  // as long as the offset time is constant, it does not matter
245  const double dt1 = static_cast<double>(m_currentTimeMsSinceEpoch - m_prevSampleAdjustedTime);
246  const double dt2 = static_cast<double>(m_nextSampleAdjustedTime - m_prevSampleAdjustedTime);
247  double timeFraction = dt1 / dt2;
248 
249  if (CBuildConfig::isLocalDeveloperDebugBuild())
250  {
251  BLACK_VERIFY_X(dt1 >= 0, Q_FUNC_INFO, "Expect postive dt1");
252  BLACK_VERIFY_X(dt2 > 0, Q_FUNC_INFO, "Expect postive dt2");
253  BLACK_VERIFY_X(isAcceptableTimeFraction(timeFraction), Q_FUNC_INFO, "Expect fraction 0-1");
254  }
255  timeFraction = clampValidTimeFraction(timeFraction);
256  const qint64 interpolatedTime = m_prevSampleTime + qRound64(timeFraction * dt2);
257 
258  // time fraction is expected between 0-1
259  m_currentInterpolationStatus.setInterpolated(true);
260  m_interpolant.setTimes(m_currentTimeMsSinceEpoch, timeFraction, interpolatedTime);
261  m_interpolant.setRecalculated(recalculate);
262 
263  if (this->doLogging())
264  {
265  log.interpolationSituations.clear();
266  log.interpolationSituations.push_back(m_s[0]);
267  log.interpolationSituations.push_back(m_s[1]);
268  log.interpolationSituations.push_back(m_s[2]); // latest at end
269  log.interpolator = 's';
270  log.deltaSampleTimesMs = dt2;
271  log.simTimeFraction = timeFraction;
272  log.tsInterpolated = interpolatedTime; // without offsets
273  log.interpolantRecalc = m_interpolant.isRecalculated();
274  }
275 
276  return m_interpolant;
277  }

318  CAircraftSituation CInterpolatorSpline::CInterpolant::interpolatePositionAndAltitude(const CAircraftSituation &currentSituation, bool interpolateGndFactor) const
319  {
320  const double t1 = m_pa.t[1];
321  const double t2 = m_pa.t[2]; // latest (adjusted)
322 
323  bool valid = (t1 < t2) && (m_currentTimeMsSinceEpoc >= t1) && (m_currentTimeMsSinceEpoc < t2);
324  if (!valid && CBuildConfig::isLocalDeveloperDebugBuild())
325  {
326  Q_ASSERT_X(t1 < t2, Q_FUNC_INFO, "Expect sorted times, latest first"); // that means a bug in our code init the values
327  BLACK_VERIFY_X(m_currentTimeMsSinceEpoc >= t1, Q_FUNC_INFO, "invalid timestamp t1");
328  BLACK_VERIFY_X(m_currentTimeMsSinceEpoc < t2, Q_FUNC_INFO, "invalid timestamp t2"); // t1==t2 results in div/0
329  }
330  if (!valid) { return CAircraftSituation::null(); }
331 
332  const double newX = evalSplineInterval(m_currentTimeMsSinceEpoc, t1, t2, m_pa.x[1], m_pa.x[2], m_pa.dx[1], m_pa.dx[2]);
333  const double newY = evalSplineInterval(m_currentTimeMsSinceEpoc, t1, t2, m_pa.y[1], m_pa.y[2], m_pa.dy[1], m_pa.dy[2]);
334  const double newZ = evalSplineInterval(m_currentTimeMsSinceEpoc, t1, t2, m_pa.z[1], m_pa.z[2], m_pa.dz[1], m_pa.dz[2]);
335 
336  valid = CAircraftSituation::isValidVector(m_pa.x) && CAircraftSituation::isValidVector(m_pa.y) && CAircraftSituation::isValidVector(m_pa.z);
337  if (!valid && CBuildConfig::isLocalDeveloperDebugBuild())
338  {
339  BLACK_VERIFY_X(CAircraftSituation::isValidVector(m_pa.x), Q_FUNC_INFO, "invalid X"); // all x values
340  BLACK_VERIFY_X(CAircraftSituation::isValidVector(m_pa.y), Q_FUNC_INFO, "invalid Y"); // all y values
341  BLACK_VERIFY_X(CAircraftSituation::isValidVector(m_pa.z), Q_FUNC_INFO, "invalid Z"); // all z values
342  }
343  if (!valid) { return CAircraftSituation::null(); }
344 
345  CAircraftSituation newSituation(currentSituation);
346  const std::array<double, 3> normalVector = { { newX, newY, newZ } };
347  const CCoordinateGeodetic currentPosition(normalVector);
348 
349  valid = CAircraftSituation::isValidVector(normalVector);
350  if (!valid && CBuildConfig::isLocalDeveloperDebugBuild())
351  {
352  BLACK_VERIFY_X(valid, Q_FUNC_INFO, "invalid vector");
353  CLogMessage(this).warning(u"Invalid vector for '%1' v: %2 %3 %4") << currentSituation.getCallsign().asString() << normalVector[0] << normalVector[1] << normalVector[2];
354  }
355  if (!valid) { return CAircraftSituation::null(); }
356 
357  const double newA = evalSplineInterval(m_currentTimeMsSinceEpoc, t1, t2, m_pa.a[1], m_pa.a[2], m_pa.da[1], m_pa.da[2]);
358  const CAltitude alt(newA, m_altitudeUnit);
359 
360  newSituation.setPosition(currentPosition);
361  newSituation.setAltitude(alt);
362  newSituation.setMSecsSinceEpoch(this->getInterpolatedTime());
363 
364  if (interpolateGndFactor)
365  {
366  const double gnd1 = m_pa.gnd[1];
367  const double gnd2 = m_pa.gnd[2]; // latest
368  do
369  {
370  newSituation.setOnGroundDetails(CAircraftSituation::OnGroundByInterpolation);
371  if (CAircraftSituation::isGfEqualAirborne(gnd1, gnd2))
372  {
373  newSituation.setOnGround(false);
374  break;
375  }
376  if (CAircraftSituation::isGfEqualOnGround(gnd1, gnd2))
377  {
378  newSituation.setOnGround(true);
379  break;
380  }
381  const double newGnd = evalSplineInterval(m_currentTimeMsSinceEpoc, t1, t2, gnd1, gnd2, m_pa.dgnd[1], m_pa.dgnd[2]);
382  newSituation.setOnGroundFactor(newGnd);
383  newSituation.setOnGroundFromGroundFactorFromInterpolation(groundInterpolationFactor());
384  }
385  while (false);
386  }
387  return newSituation;
388  }
