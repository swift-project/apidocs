43  bool IContextNetwork::getCmdLineClientIdAndKey(int &id, QString &key)
44  {
45  // init values
46  id = 0;
47  key = "";
48 
49  // split parser values
50  if (IContextNetwork::getCmdLineOptions().isEmpty()) { return false; } // no such option, avoid warnings
51  if (!sApp) { return false; }
52  const QString clientIdAndKey = sApp->getParserValue("clientIdAndKey").toLower();
53  if (clientIdAndKey.isEmpty() || !clientIdAndKey.contains(':')) { return false; }
54  const QStringList stringList = clientIdAndKey.split(':');
55  const QString clientIdAsString = stringList[0];
56  bool ok = true;
57  id = clientIdAsString.toInt(&ok, 0); // base 0 means C convention
58  if (!ok || id == 0) { return false; }
59  key = stringList[1];
60  return true;
61  }
