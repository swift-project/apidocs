16  CFGSwiftBusServiceProxy::CFGSwiftBusServiceProxy(QDBusConnection &connection, QObject *parent, bool dummy) : QObject(parent)
17  {
18  m_dbusInterface = new BlackMisc::CGenericDBusInterface(FGSWIFTBUS_SERVICE_SERVICENAME, ObjectPath(), InterfaceName(), connection, this);
19  if (!dummy)
20  {
21  bool s;

39  void CFGSwiftBusServiceProxy::getOwnAircraftSituationData(FlightgearData *o_flightgearData)
40  {
41  if (!o_flightgearData) { return; }
42  QPointer<CFGSwiftBusServiceProxy> myself(this);
43  std::function<void(QDBusPendingCallWatcher *)> callback = [=](QDBusPendingCallWatcher *watcher) {
44  if (!myself) { return; }
45  QDBusPendingReply<double, double, double, double, double, double, double, double> reply = *watcher;
46  if (!reply.isError())
47  {
48  o_flightgearData->latitudeDeg = reply.argumentAt<0>();
49  o_flightgearData->longitudeDeg = reply.argumentAt<1>();
50  o_flightgearData->altitudeFt = reply.argumentAt<2>();
51  o_flightgearData->groundspeedKts = reply.argumentAt<3>();
52  o_flightgearData->pitchDeg = reply.argumentAt<4>();
53  o_flightgearData->rollDeg = reply.argumentAt<5>();
54  o_flightgearData->trueHeadingDeg = reply.argumentAt<6>();
55  o_flightgearData->pressureAltitudeFt = reply.argumentAt<7>();
56  }
57  watcher->deleteLater();
58  };
59  m_dbusInterface->callDBusAsync(QLatin1String("getOwnAircraftSituationData"), callback);
60  }
