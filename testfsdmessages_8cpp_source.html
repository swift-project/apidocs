422  void CTestFsdMessages::testPBH()
423  {
424  struct PBH
425  {
426  int pitch, bank, heading;
427  };
428  QVector<PBH> testData;
429  for (int pitch = -90; pitch < 90; pitch++) { testData.push_back({ pitch, 0, 0 }); }
430  for (int bank = -179; bank < 180; bank++) { testData.push_back({ 0, bank, 0 }); }
431  for (int heading = 0; heading < 360; heading++) { testData.push_back({ 0, 0, heading }); }
432  for (const auto &input : testData)
433  {
434  std::uint32_t pbh = 0;
435  packPBH(input.pitch, input.bank, input.heading, true, pbh);
436 
437  double pitch2 = 0;
438  double bank2 = 0;
439  double heading2 = 0;
440  bool onGround2 = false;
441  unpackPBH(pbh, pitch2, bank2, heading2, onGround2);
442  QVERIFY(pitch2 >= -90);
443  QVERIFY(pitch2 < 90);
444  QVERIFY(bank2 >= -180);
445  QVERIFY(bank2 < 180);
446  QVERIFY(heading2 >= 0);
447  QVERIFY(heading2 < 360);
448  QCOMPARE(input.pitch, pitch2);
449  QCOMPARE(input.bank, bank2);
450  QCOMPARE(true, onGround2);
451 
452  if (input.heading < 0) { heading2 -= 360; }
453  QVERIFY(qAbs(input.heading - heading2) < 1);
454  }
455  }
