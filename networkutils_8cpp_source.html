67  bool CNetworkUtils::canConnect(const QString &hostAddress, int port, QString &message, int timeoutMs)
68  {
69  if (timeoutMs < 0) { timeoutMs = getTimeoutMs(); } // external functions might call with -1
70  QTcpSocket socket;
71  QSignalMapper mapper;
72  QObject::connect(&socket, &QTcpSocket::connected, &mapper, qOverload<>(&QSignalMapper::map));
73  QObject::connect(&socket, qOverload<QAbstractSocket::SocketError>(&QTcpSocket::error), &mapper, qOverload<>(&QSignalMapper::map));
74  mapper.setMapping(&socket, 0);
75 
76  CEventLoop eventLoop;
77  eventLoop.stopWhen(&mapper, qOverload<int>(&QSignalMapper::mapped));
78  socket.connectToHost(hostAddress, static_cast<quint16>(port));
79  const bool timedOut = !eventLoop.exec(timeoutMs);
80 
81  if (socket.state() != QTcpSocket::ConnectedState)
82  {
83  static const QString e("Connection failed: '%1'");
84  message = timedOut ? e.arg("Timed out") : e.arg(socket.errorString());
85  return false;
86  }
87  else
88  {
89  static const QString ok("OK, connected");
90  message = ok;
91  return true;
92  }
93  }

111  {
112  if (!url.isValid())

124  const QString host(url.host());
125  const QString scheme(url.scheme().toLower());
126  int p = url.port();

130  }

327  CStatusMessageList CNetworkUtils::createNetworkReport(const QUrl &url, const QNetworkAccessManager *qam)
328  {
329  static const CLogCategoryList cats({ CLogCategories::network() });
330  CStatusMessageList msgs;
331 
332  if (!url.isEmpty())
333  {
334  const QString host = url.host();
335  const bool canPing = Network::canPing(host);
336  const CStatusMessage ping(cats, canPing ? CStatusMessage::SeverityInfo : CStatusMessage::SeverityError, "Host: " + host + " ping: " + boolToYesNo(canPing));
337  msgs.push_back(ping);
338 
339  QString msg;
340  const bool canConnect = CNetworkUtils::canConnect(url, msg, getTimeoutMs() * 2);
341  if (canConnect)
342  {
343  msgs.push_back(CStatusMessage(cats, CStatusMessage::SeverityInfo, u"Can connect to " % url.toString()));
344  }
345  else
346  {
347  msgs.push_back(CStatusMessage(cats, CStatusMessage::SeverityError, u"Cannot connect to " % url.toString() % u" msg: " % msg));
348  }
349  }
350 
351  if (qam)
352  {
353  const QString msg = CNetworkUtils::createNetworkAccessManagerReport(qam);
354  const bool accessible = qam->networkAccessible() == QNetworkAccessManager::Accessible;
355  msgs.push_back(CStatusMessage(cats, accessible ? CStatusMessage::SeverityInfo : CStatusMessage::SeverityError, msg));
356  }
357 
358  return msgs;
359  }

361  QString CNetworkUtils::createNetworkConfigurationReport(const QNetworkConfigurationManager *qcm, const QNetworkAccessManager *qam, const QString &separator)
362  {
363  if (!qcm) { return QStringLiteral("No configuration manager"); }
364 
365  static const QString empty;
366  QString report;
367  int c = 0;
368 
369  int active = 0;
370  int inActive = 0;
371  int valid = 0;
372  for (const QNetworkConfiguration &config : qcm->allConfigurations())
373  {
374  if (config.state() == QNetworkConfiguration::Active) { active++; }
375  else { inActive++; }
376  if (config.isValid()) { valid++; }
377 
378  report +=
379  (report.isEmpty() ? empty : separator) %
380  QString::number(++c) % u": " %
381  CNetworkUtils::networkConfigurationToString(config);
382  }
383 
384  if (c < 1)
385  {
386  report = QStringLiteral("No network configurations!");
387  }
388  else
389  {
390  static const QString count("Network configurations: active %1 / inactive %2 / valid %3");
391  report +=
392  (report.isEmpty() ? empty : separator) %
393  count.arg(active).arg(inActive).arg(valid);
394  }
395 
396  if (qam)
397  {
398  report +=
399  (report.isEmpty() ? empty : separator) %
400  u"QAM: " %
401  CNetworkUtils::createNetworkAccessManagerReport(qam);
402  }
403 
404  return report;
405  }
