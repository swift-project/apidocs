96  void CDistributorPreferencesComponent::loadDistributorsFromSet()

100  if (models.isEmpty())

103  this->showOverlayMessage(m);

106  const CDistributorList distributors = models.getDistributors();
107  if (distributors.isEmpty())
108  {
109  const CStatusMessage m = CStatusMessage(this).error(u"No distributors for model set %1") << sim.toQString();
110  this->showOverlayMessage(m);
111  return;
112  }
113  this->updateContainerMaybeAsync(distributors);

133  void CDistributorPreferencesComponent::triggerDeferredSimulatorChange()
134  {
135  QPointer<CDistributorPreferencesComponent> myself(this);
136  QTimer::singleShot(1000, this, [=] {
137  if (!myself) { return; }
138  if (!sApp || sApp->isShuttingDown()) { return; }
139  const CSimulatorInfo sim = ui->comp_SimulatorSelector->getValue();
140  this->onSimulatorChanged(sim);
141  });
142  }
