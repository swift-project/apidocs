62  template <class LATorLON>
63  LATorLON CEarthAngle<LATorLON>::fromWgs84(const QString &wgsCoordinate)
64  {
65  const QString wgs = wgsCoordinate.simplified().trimmed();
66  if (wgs.isEmpty()) { return LATorLON(); }
67 
68  // support for 5deg, 1.2rad
69  if (CAngleUnit::deg().endsStringWithNameOrSymbol(wgs) || CAngleUnit::rad().endsStringWithNameOrSymbol(wgs))
70  {
71  LATorLON latOrLon;
72  latOrLon.parseFromString(wgs);
73  return latOrLon;
74  }
75 
76  // number only -> parsed as degrees
77  bool isDouble;
78  const double valueDegrees = wgs.toDouble(&isDouble);
79  if (isDouble)
80  {
81  CAngle a(valueDegrees, CAngleUnit::deg());
82  return LATorLON(a);
83  }
84 
85  // http://www.regular-expressions.info/floatingpoint.html
86  thread_local const QRegularExpression rx("[+-]?\\d+(?:\\.\\d+)?");
87  int deg = 0;
88  int min = 0;
89  double sec = 0.0;
90  int c = 0;
91  QRegularExpressionMatchIterator i = rx.globalMatch(wgs);
92  while (i.hasNext() && c < 3)
93  {
94  const QRegularExpressionMatch match = i.next();
95  bool ok;
96  if (match.hasMatch())
97  {
98  const QString cap = match.captured(0);
99  switch (c++)
100  {
101  case 0:
102  deg = cap.toInt(&ok);
103  break;
104  case 1:
105  min = cap.toInt(&ok);
106  break;
107  case 2:
108  sec = cap.toDouble(&ok);
109  break;
110  default:
111  break;
112  }
113  }
114  Q_UNUSED(ok); // ok for debugging purposes
115  }
116 
117  if (wgs.contains('S', Qt::CaseInsensitive) || wgs.contains('W', Qt::CaseInsensitive))
118  {
119  deg *= -1;
120  min *= -1;
121  sec *= -1;
122  }
123 
124  PhysicalQuantities::CAngle a(deg, min, sec);
125  return LATorLON(a);
126  }

169  {
