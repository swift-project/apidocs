184  void CSettingsSimulatorComponent::pluginStateChanged(const QString &identifier, bool enabled)
185  {
186  Q_ASSERT(sGui && sGui->getIContextSimulator());
187 
188  const CSimulatorPluginInfoList simDrivers(getAvailablePlugins());
189  const CSimulatorPluginInfo selected = simDrivers.findByIdentifier(identifier);
190  if (selected.isUnspecified())
191  {
192  CLogMessage(this).error(u"Simulator plugin does not exist: '%1'") << identifier;
193  return;
194  }
195 
196  QStringList e = m_enabledSimulators.getThreadLocal(); // from setting
197  if (enabled != e.contains(selected.getIdentifier()))
198  {
199  if (enabled)
200  {
201  e << selected.getIdentifier(); // add enabled plugin
202  }
203  else
204  {
205  e.removeAll(selected.getIdentifier());
206  }
207  const CStatusMessage msg = m_enabledSimulators.set(e); // change setting
208  if (msg.isWarningOrAbove())
209  {
210  CLogMessage::preformatted(msg);
211  }
212  }
213 
214  // changing of GUI state will be done via received signal
215  }

217  void CSettingsSimulatorComponent::onApplyMaxRenderedAircraft()
218  {
219  if (!sGui || sGui->isShuttingDown() || !sGui->getIContextSimulator()) { return; }
220 
221  // get initial aircraft to render
222  CInterpolationAndRenderingSetupGlobal setup = sGui->getIContextSimulator()->getInterpolationAndRenderingSetupGlobal();
223  const int noRequested = ui->le_MaxAircraft->text().isEmpty() ? setup.InfiniteAircraft() : ui->le_MaxAircraft->text().toInt();
224  const int oldValue = setup.getMaxRenderedAircraft();
225  if (oldValue == noRequested) { return; }
226 
227  // set value
228  setup.setMaxRenderedAircraft(noRequested);
229  sGui->getIContextSimulator()->setInterpolationAndRenderingSetupGlobal(setup);
230 
231  // re-read real value
232  setup = sGui->getIContextSimulator()->getInterpolationAndRenderingSetupGlobal();
233  const int noRendered = setup.getMaxRenderedAircraft();
234  if (noRequested == noRendered)
235  {
236  CLogMessage(this).info(u"Max.rendered aircraft: %1") << noRendered;
237  }
238  else
239  {
240  CLogMessage(this).info(u"Max.rendered aircraft: %1, requested: %2") << noRendered << noRequested;
241  ui->le_MaxAircraft->setText(QString::number(noRendered));
242  }
243  this->setGuiValues();
244  }

246  void CSettingsSimulatorComponent::onApplyMaxRenderedDistance()
247  {
248  if (!sGui || sGui->isShuttingDown() || !sGui->getIContextSimulator()) { return; }
249 
250  // get initial aircraft to render
251  CInterpolationAndRenderingSetupGlobal setup = sGui->getIContextSimulator()->getInterpolationAndRenderingSetupGlobal();
252  CLength newDistance(0, nullptr);
253  if (!ui->le_MaxDistance->text().isEmpty())
254  {
255  newDistance = CLength(ui->le_MaxDistance->text().toInt(), CLengthUnit::NM());
256  }
257 
258  CLength currentDistance(setup.getMaxRenderedDistance());
259  if (currentDistance == newDistance) { return; }
260 
261  CLogMessage(this).info(u"Max.distance requested: %1") << newDistance.valueRoundedWithUnit(2, true);
262  setup.setMaxRenderedDistance(newDistance);
263  sGui->getIContextSimulator()->setInterpolationAndRenderingSetupGlobal(setup);
264  this->setGuiValues();
265  }

335  void CSettingsSimulatorComponent::onApplyRecordGnd()
336  {
337  bool ok = false;
338  CSimulatorSettings settings = CSettingsSimulatorComponent::getSimulatorSettings(ok);
339  if (!ok) { return; }
340 
341  // get value, automatically add default unit if unit is missing
342  CLength radius = CLength::null();
343  QString radiusString = ui->le_RecordOwnGndPositionsRadius->text().trimmed();
344  if (!radiusString.isEmpty())
345  {
346  if (!CMeasurementUnit::endWithValidUnitSymbol<CLengthUnit>(radiusString)) { radiusString += "m"; }
347  radius.parseFromString(radiusString);
348  }
349  ui->le_RecordOwnGndPositionsRadius->setText(radius.valueRoundedWithUnit(1));
350 
351  const bool c1 = settings.setRecordOwnAircraftGnd(ui->cb_RecordOwnGndPositions->isChecked());
352  const bool c2 = settings.setRecordedGndRadius(radius);
353  if (!c1 && !c2) { return; }
354  this->setSimulatorSettings(settings);
355  }
