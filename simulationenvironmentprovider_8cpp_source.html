22  bool ISimulationEnvironmentProvider::rememberGroundElevation(const CCallsign &requestedForCallsign, bool likelyOnGroundElevation, const ICoordinateGeodetic &elevationCoordinate, const CLength &epsilon)
23  {
24  if (!elevationCoordinate.hasMSLGeodeticHeight())
25  {
26  BLACK_AUDIT_X(false, Q_FUNC_INFO, "Elevation needs to be MSL NON NULL");
27  return false;
28  }
29 
30  const CLength minRange = ISimulationEnvironmentProvider::minRange(epsilon);
31  const double elvFt = elevationCoordinate.geodeticHeight().value(CLengthUnit::ft());
32 
33  CCoordinateGeodetic alreadyInRange;
34  CCoordinateGeodetic alreadyInRangeGnd;
35  {
36  QReadLocker l(&m_lockElvCoordinates);
37  if (!m_enableElevation) { return false; }
38 
39  // check if we have already an elevation within range
40  alreadyInRangeGnd = m_elvCoordinatesGnd.findFirstWithinRangeOrDefault(elevationCoordinate, minRange);
41  alreadyInRange = m_elvCoordinates.findFirstWithinRangeOrDefault(elevationCoordinate, minRange);
42  }
43 
44  constexpr double maxDistFt = 30.0;
45 
46  // here we deal with gnd situation and do not expect a lot of variance
47  if (!alreadyInRangeGnd.isNull())
48  {
49  // found
50  const double distFt = qAbs(alreadyInRangeGnd.geodeticHeight().value(CLengthUnit::ft()) - elvFt);
51  if (distFt > maxDistFt)
52  {
53  // such a huge distance to existing value
54  CLogMessage(this).debug(u"Suspicious GND elevation distance '%1': %2ft at %3") << requestedForCallsign.asString() << distFt << elevationCoordinate.geodeticHeight().valueRoundedAsString(CLengthUnit::ft(), 1);
55  BLACK_AUDIT_X(!CBuildConfig::isLocalDeveloperDebugBuild(), Q_FUNC_INFO, "Suspicious gnd. elevation distance");
56  }
57  return false;
58  }
59 
60  // here we deal with all kind of values, so it can be that
61  // values vary in a much larger range
62  if (!alreadyInRange.isNull())
63  {
64  // found
65  const double distFt = qAbs(alreadyInRange.geodeticHeight().value(CLengthUnit::ft()) - elvFt);
66  if (distFt > maxDistFt)
67  {
68  // such a huge distance to existing value
69  CLogMessage(this).debug(u"Suspicious NON GND elevation distance for '%1': %2ft at %3") << requestedForCallsign.asString() << distFt << elevationCoordinate.geodeticHeight().valueRoundedAsString(CLengthUnit::ft(), 1);
70  // BLACK_AUDIT_X(!CBuildConfig::isLocalDeveloperDebugBuild(), Q_FUNC_INFO, "Suspicious elevation distance");
71  }
72  return false;
73  }
74 
75  const qint64 now = QDateTime::currentMSecsSinceEpoch();
76  {
77  // we keep latest at front
78  // * we assume we find them faster
79  // * and need them more frequently (the recent ones)
80  QWriteLocker l(&m_lockElvCoordinates);
81  if (likelyOnGroundElevation)
82  {
83  if (m_elvCoordinatesGnd.size() > m_maxElevationsGnd) { m_elvCoordinatesGnd.pop_back(); }
84  m_elvCoordinatesGnd.push_front(elevationCoordinate);
85  }
86  else
87  {
88  if (m_elvCoordinates.size() > m_maxElevations) { m_elvCoordinates.pop_back(); }
89  m_elvCoordinates.push_front(elevationCoordinate);
90  }
91 
92  // statistics
93  if (m_pendingElevationRequests.contains(requestedForCallsign))
94  {
95  const qint64 startedMs = m_pendingElevationRequests.value(requestedForCallsign);
96  const qint64 deltaMs = now - startedMs;
97  m_pendingElevationRequests.remove(requestedForCallsign);
98  m_statsCurrentElevRequestTimeMs = deltaMs;
99  if (m_statsMaxElevRequestTimeMs < deltaMs) { m_statsMaxElevRequestTimeMs = deltaMs; }
100  }
101  }
102  return true;
103  }

303  int ISimulationEnvironmentProvider::cleanUpElevations(const ICoordinateGeodetic &referenceCoordinate, int maxNumber)
304  {
305  int currentMax;
306  CCoordinateGeodeticList coordinates(this->getAllElevationCoordinates(currentMax));
307  if (maxNumber < 0) { maxNumber = currentMax; }
308  const int size = coordinates.size();
309  if (size <= maxNumber) { return 0; }
310  coordinates.sortByEuclideanDistanceSquared(referenceCoordinate);
311  coordinates.truncate(maxNumber);
312  const int delta = size - coordinates.size();
313  {
314  QWriteLocker l(&m_lockElvCoordinates);
315  m_elvCoordinates = coordinates;
316  }
317  return delta;
318  }

320  CElevationPlane ISimulationEnvironmentProvider::findClosestElevationWithinRange(const ICoordinateGeodetic &reference, const CLength &range) const
321  {
322  if (!this->isElevationProviderEnabled()) { return CElevationPlane::null(); }
323 
324  // for single point we use a slightly optimized version
325  const bool singlePoint = (&range == &CElevationPlane::singlePointRadius() || range.isNull() || range <= CElevationPlane::singlePointRadius());
326  const CCoordinateGeodeticList elevations = this->getAllElevationCoordinates();
327  const CCoordinateGeodetic coordinate = singlePoint ?
328  elevations.findFirstWithinRangeOrDefault(reference, CElevationPlane::singlePointRadius()) :
329  elevations.findClosestWithinRange(reference, range);
330  const bool found = !coordinate.isNull();
331 
332  {
333  QWriteLocker l { &m_lockElvCoordinates };
334  if (found)
335  {
336  m_elvFound++;
337  return CElevationPlane(coordinate, reference); // plane with radius = distance to reference
338  }
339  else
340  {
341  m_elvMissed++;
342  return CElevationPlane::null();
343  }
344  }
345  }

348  {

350  const CElevationPlane ep = ISimulationEnvironmentProvider::findClosestElevationWithinRange(reference, range);

555  bool ISimulationEnvironmentProvider::cleanElevationValues(const CAircraftSituation &reference, const CLength &keptRange, bool forced)
556  {
557  if (reference.isNull() || keptRange.isNull()) { return false; }
558  const CLength r = minRange(keptRange);
559 
560  CCoordinateGeodeticList cleanedKeptElvs;
561  bool maxReached = false;
562  bool cleaned = false;
563 
564  {
565  QReadLocker l(&m_lockElvCoordinates);
566  cleanedKeptElvs = m_elvCoordinates;
567  maxReached = m_elvCoordinates.size() >= m_maxElevations;
568  }
569  if (!cleanedKeptElvs.isEmpty() && (forced || maxReached))
570  {
571  const int removed = cleanedKeptElvs.removeOutsideRange(reference, r);
572  if (removed > 0)
573  {
574  cleaned = true;
575  QWriteLocker l(&m_lockElvCoordinates);
576  m_elvCoordinates = cleanedKeptElvs;
577  }
578  }
579 
580  cleanedKeptElvs.clear();
581  {
582  QReadLocker l(&m_lockElvCoordinates);
583  cleanedKeptElvs = m_elvCoordinatesGnd;
584  maxReached = m_elvCoordinatesGnd.size() >= m_maxElevationsGnd;
585  }
586  if (!cleanedKeptElvs.isEmpty() && (forced || maxReached))
587  {
588  const int removed = cleanedKeptElvs.removeOutsideRange(reference, r);
589  if (removed > 0)
590  {
591  cleaned = true;
592  QWriteLocker l(&m_lockElvCoordinates);
593  m_elvCoordinatesGnd = cleanedKeptElvs;
594  }
595  }
596 
597  return cleaned;
598  }
