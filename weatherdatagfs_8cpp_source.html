131  {
132  double altitude = (std::pow(seaLevelPressurePa / atmosphericPressurePa, 0.19022) - 1) * temperatureK * 3.28084 / 0.0065;

145  {

256  static const QStringList grib2Variables = {
257  "PRATE",
258  "PRES",
259  "PRMSL",
260  "RH",
261  "TCDC",
262  "TMP",
263  "UGRD",
264  "VGRD",
265  "CRAIN",
266  "CSNOW"
267  };

269  static const std::array<int, 4> cycles = { { 0, 6, 12, 18 } };
270  const QDateTime now = QDateTime::currentDateTimeUtc();
271 
272  // GFS data is published after 5 yours.
273  const QDateTime cnow = now.addSecs(-5 * 60 * 60);
274 
275  int hourLastPublishedCycle = 0;
276  for (const auto &cycle : cycles)
277  {
278  if (cnow.time().hour() > cycle) { hourLastPublishedCycle = cycle; }
279  }
280 
281  // Forecast is published in 3 hours steps.
282  // Round down to a multiple of 3
283  int forecast = now.time().hour() - hourLastPublishedCycle;
284  if (forecast < 0) { forecast += 24; }
285 
286  // The 0 hour forecast, does not contain all required parameters. Hence use 1 hour forecast instead.
287  if (forecast == 0) { forecast = 1; }
288 
289  const QString filename = u"gfs." % QStringLiteral("t%1z").arg(hourLastPublishedCycle, 2, 10, QLatin1Char('0')) % u".pgrb2.0p25." % QStringLiteral("f%2").arg(forecast, 3, 10, QLatin1Char('0'));
290  const QString directory = u"/gfs." % cnow.toString("yyyyMMdd") % u"/" % QStringLiteral("%1").arg(hourLastPublishedCycle, 2, 10, QLatin1Char('0')) % u"/atmos";
291 
292  downloadUrl.appendQuery("file", filename);
293  for (const auto &level : grib2Levels)
294  {
295  downloadUrl.appendQuery("lev_" + level, "on");
296  }
297  for (const auto &variable : grib2Variables)
298  {
299  downloadUrl.appendQuery("var_" + variable, "on");
300  }
301  downloadUrl.appendQuery("leftlon", "0");
302  downloadUrl.appendQuery("rightlon", "360");
303  downloadUrl.appendQuery("toplat", "90");
304  downloadUrl.appendQuery("bottomlat", "-90");
305  downloadUrl.appendQuery("dir", directory);
306  return downloadUrl;
307  }

309  bool CWeatherDataGfs::parseGfsFileImpl(const QByteArray &gribData)
310  {
311  if (!m_lockData.tryLockForWrite(1000))
312  {
313  CLogMessage(this).warning(u"Cannot log CWeatherDataGfs data");
314  return false;
315  }
316 
317  m_lockData.unlock();
318  QWriteLocker lock(&m_lockData);
319 
320  m_gfsWeatherGrid.clear();
321  m_weatherGrid.clear();
322 
323  // Messages should be 76. This is a combination
324  // of requested values (e.g. temperature, clouds etc) at specific layers (2 mbar, 10 mbar, surface).
325  constexpr int maxMessages = 76;
326  int messageNo = 0;
327  g2int iseek = 0;
328  for (;;)
329  {
330  if (QThread::currentThread()->isInterruptionRequested()) { return false; }
331 
332  // Search next grib field
333  g2int lskip = 0;
334  g2int lgrib = 0;
335  auto constData = reinterpret_cast<unsigned char *>(const_cast<char *>(gribData.data()));
336  findNextGribMessage(constData, gribData.size(), iseek, &lskip, &lgrib);
337  if (lgrib == 0) { break; }
338 
339  unsigned char *readPtr = constData + lskip;
340  iseek = lskip + lgrib;
341 
342  g2int sec0[3];
343  g2int sec1[13];
344  g2int numlocal = 0;
345  g2int numfields = 0;
346  g2_info(readPtr, sec0, sec1, &numfields, &numlocal);
347 
348  for (int n = 0; n < numfields; n++)
349  {
350  g2int unpack = 1;
351  g2int expand = 1;
352  gribfield *gfld = nullptr;
353  g2_getfld(readPtr, n + 1, unpack, expand, &gfld);
354  if (gfld->idsectlen < 12)
355  {
356  CLogMessage(this).warning(u"Identification section: wrong length!");
357  continue;
358  }
359 
360  if (gfld->igdtnum != 0) { CLogMessage(this).warning(u"Can handle only grid definition template number = 0"); }
361 
362  int nscan = gfld->igdtmpl[18];
363  int npnts = gfld->ngrdpts;
364  int nx = gfld->igdtmpl[7];
365  int ny = gfld->igdtmpl[8];
366  if (nscan != 0) { CLogMessage(this).error(u"Can only handle scanning mode NS:WE."); }
367  if (npnts != nx * ny) { CLogMessage(this).error(u"Cannot handle non-regular grid."); }
368 
369  if (m_gfsWeatherGrid.isEmpty()) { createWeatherGrid(gfld); }
370 
371  if (gfld->ipdtnum == 0) { handleProductDefinitionTemplate40(gfld); }
372  else if (gfld->ipdtnum == 8) { handleProductDefinitionTemplate48(gfld); }
373  else
374  {
375  CLogMessage(this).warning(u"Cannot handle product definition template %1") << gfld->ipdtnum;
376  continue;
377  }
378 
379  g2_free(gfld);
380  }
381  messageNo++;
382  }
383 
384  // validate
385  if (messageNo > maxMessages && CBuildConfig::isLocalDeveloperDebugBuild())
386  {
387  // as discussed this means a format change
388  BLACK_VERIFY_X(false, Q_FUNC_INFO, "Format change in GRIB, too many messages");
389  }
390 
391  const int weatherGridPointsNo = m_gfsWeatherGrid.size();
392  CLogMessage(this).debug() << "Parsed" << messageNo << "GRIB messages.";
393  CLogMessage(this).debug() << "Obtained" << weatherGridPointsNo << "grid points.";
394 
395  constexpr int maxPoints = 200;
396  for (const GfsGridPoint &gfsGridPoint : std::as_const(m_gfsWeatherGrid))
397  {
398  if (QThread::currentThread()->isInterruptionRequested()) { return false; }
399 
400  CTemperatureLayerList temperatureLayers;
401  CWindLayerList windLayers;
402  for (const GfsIsobaricLayer &isobaricLayer : gfsGridPoint.isobaricLayers)
403  {
404  float level = gfsGridPoint.isobaricLayers.key(isobaricLayer);
405  double altitudeFt = calculateAltitudeFt(gfsGridPoint.pressureAtMsl, level, isobaricLayer.temperature);
406 
407  CAltitude altitude(altitudeFt, CAltitude::MeanSeaLevel, CLengthUnit::ft());
408 
409  auto temperature = CTemperature { isobaricLayer.temperature, CTemperatureUnit::K() };
410  auto dewPoint = calculateDewPoint(temperature, isobaricLayer.relativeHumidity);
411 
412  CTemperatureLayer temperatureLayer(altitude, temperature, dewPoint, isobaricLayer.relativeHumidity);
413  temperatureLayers.push_back(temperatureLayer);
414 
415  double windDirection = -1 * CMathUtils::rad2deg(std::atan2(-isobaricLayer.windU, isobaricLayer.windV));
416  windDirection += 180.0;
417  if (windDirection < 0.0) { windDirection += 360.0; }
418  if (windDirection >= 360.0) { windDirection -= 360.0; }
419  double windSpeed = std::hypot(isobaricLayer.windU, isobaricLayer.windV);
420  CWindLayer windLayer(altitude, CAngle(windDirection, CAngleUnit::deg()), CSpeed(windSpeed, CSpeedUnit::m_s()), {});
421  windLayers.push_back(windLayer);
422  }
423 
424  CCloudLayerList cloudLayers;
425  for (const GfsCloudLayer &gfsCloudLayer : std::as_const(gfsGridPoint.cloudLayers))
426  {
427  if (std::isnan(gfsCloudLayer.bottomLevelPressure) || std::isnan(gfsCloudLayer.topLevelPressure) || std::isnan(gfsCloudLayer.topLevelTemperature)) { continue; }
428 
429  CCloudLayer cloudLayer;
430  double bottomLevelFt = calculateAltitudeFt(gfsGridPoint.pressureAtMsl, gfsCloudLayer.bottomLevelPressure, gfsCloudLayer.topLevelTemperature);
431  double topLevelFt = calculateAltitudeFt(gfsGridPoint.pressureAtMsl, gfsCloudLayer.topLevelPressure, gfsCloudLayer.topLevelTemperature);
432  cloudLayer.setBase(CAltitude(bottomLevelFt, CAltitude::MeanSeaLevel, CLengthUnit::ft()));
433  cloudLayer.setTop(CAltitude(topLevelFt, CAltitude::MeanSeaLevel, CLengthUnit::ft()));
434  cloudLayer.setCoveragePercent(qRound(gfsCloudLayer.totalCoverage));
435  if (gfsGridPoint.surfaceSnow > 0.0) { cloudLayer.setPrecipitation(CCloudLayer::Snow); }
436  if (gfsGridPoint.surfaceRain > 0.0) { cloudLayer.setPrecipitation(CCloudLayer::Rain); }
437 
438  // Precipitation rate is in kg m-2 s-1, which is equal to mm/s
439  // Multiply with 3600 to convert to mm/h
440  cloudLayer.setPrecipitationRate(gfsGridPoint.surfacePrecipitationRate * 3600.0);
441  cloudLayer.setClouds(CCloudLayer::CloudsUnknown);
442  cloudLayers.push_back(cloudLayer);
443  }
444 
445  auto pressureAtMsl = PhysicalQuantities::CPressure { gfsGridPoint.pressureAtMsl, PhysicalQuantities::CPressureUnit::Pa() };
446 
447  const CLatitude latitude(gfsGridPoint.latitude, CAngleUnit::deg());
448  const CLongitude longitude(gfsGridPoint.longitude, CAngleUnit::deg());
449  const auto position = CCoordinateGeodetic { latitude, longitude };
450  const CGridPoint gridPoint({}, position, cloudLayers, temperatureLayers, {}, windLayers, pressureAtMsl);
451  m_weatherGrid.push_back(gridPoint);
452  if (m_weatherGrid.size() >= maxPoints)
453  {
454  // too many points lead to extreme memory consumption and CPU usage
455  // we stop here, no use case so far in swift where we need that
456  BLACK_VERIFY_X(!CBuildConfig::isLocalDeveloperDebugBuild(), Q_FUNC_INFO, "Too many grid points");
457  CLogMessage(this).warning(u"Too many weather grid points: %1") << m_weatherGrid.size();
458  break;
459  }
460  }
461 
462  return true;
463  }

465  void CWeatherDataGfs::findNextGribMessage(unsigned char *buffer, g2int size, g2int iseek, g2int *lskip, g2int *lgrib)
466  {
467  *lgrib = 0;
468 
469  if (iseek >= size) return;
470  g2int ipos = iseek;
471  unsigned char *bufferStart = buffer + iseek;
472 
473  // READ PARTIAL SECTION
474  g2int lim = size - 8;
475 
476  for (g2int k = 0; k < lim; k++)
477  {
478  g2int start;
479  g2int vers;
480  // Look for "GRIB" at the beginning
481  gbit(bufferStart, &start, (k + 0) * 8, 4 * 8);
482  gbit(bufferStart, &vers, (k + 7) * 8, 1 * 8);
483  if (start == 1196575042 && (vers == 1 || vers == 2))
484  {
485  g2int lengrib = 0;
486  // Look for '7777' at end of GRIB message
487  if (vers == 1) gbit(bufferStart, &lengrib, (k + 4) * 8, 3 * 8);
488  if (vers == 2) gbit(bufferStart, &lengrib, (k + 12) * 8, 4 * 8);
489 
490  // Check the limits
491  g2int endPos = ipos + k + lengrib;
492  if (endPos > size) return;
493  unsigned char *bufferEnd = buffer + endPos - 4;
494  // Hard code to 4 instead of sizeof(g2int)
495  g2int end;
496  gbit(bufferEnd, &end, 0, 4 * 8);
497  if (end == 926365495)
498  {
499  // GRIB message found
500  *lskip = ipos + k;
501  *lgrib = lengrib;
502  return;
503  }
504  }
505  }

508  void CWeatherDataGfs::createWeatherGrid(const gribfield *gfld)
509  {
510  int npnts = gfld->ngrdpts;
511  int nx = gfld->igdtmpl[7];
512  int ny = gfld->igdtmpl[8];
513  float units = 0.000001f;
514  float latitude1 = gfld->igdtmpl[11] * units;
515  float longitude1 = gfld->igdtmpl[12] * units;
516  int nres = gfld->igdtmpl[13];
517  float latitude2 = gfld->igdtmpl[14] * units;
518  float longitude2 = gfld->igdtmpl[15] * units;
519  float dlatitude = gfld->igdtmpl[16] * units;
520  float dlongitude = gfld->igdtmpl[17] * units;
521 
522  if (latitude1 < -90.0f || latitude2 < -90.0f || latitude1 > 90.0f || latitude2 > 90.0f)
523  {
524  CLogMessage(this).warning(u"Invalid grid definition: lat1 = %1 - lat2 = %2") << latitude1 << latitude2;
525  return;
526  }
527  if (longitude1 < 0.0f || longitude2 < 0.0f || longitude1 > 360.0f || longitude2 > 360.0f)
528  {
529  CLogMessage(this).warning(u"Invalid grid definition: lon1 = %1 - lon2 = %2") << longitude1 << longitude2;
530  return;
531  }
532  if (npnts != nx * ny)
533  {
534  CLogMessage(this).warning(u"Invalid grid definition: npnts != nx * ny!");
535  return;
536  }
537 
538  // Scan direction is North -> South
539  float north = latitude1;
540  float south = latitude2;
541 
542  if (south > north)
543  {
544  CLogMessage(this).warning(u"Invalid grid definition: South = %1 - North = %2") << south << north;
545  return;
546  }
547 
548  float dy = 0.0f;
549  if (ny != 1)
550  {
551  dy = (north - south) / (ny - 1.0f);
552  if (nres & 16)
553  {
554  if (fabs(dy - dlatitude) > 0.001f)
555  {
556  CLogMessage(this).warning(u"Invalid grid definition: delta latitude is inconsistent");
557  return;
558  }
559  }
560  }
561  else
562  {
563  dy = 0.0;
564  }
565 
566  // Scan direction is West -> East
567  float west = longitude1;
568  float east = longitude2;
569  if (east <= west) { east += 360.0f; }
570  if (east - west > 360.0f) { east -= 360.0f; }
571  if (west < 0)
572  {
573  west += 360.0f;
574  east += 360.0f;
575  }
576 
577  float dx = 0;
578  if (nx != 1)
579  {
580  dx = (east - west) / (nx - 1);
581  dx = fabs(dx);
582  if (nres & 32)
583  {
584  if (fabs(dx - fabs(dlongitude)) > 0.001f)
585  {
586  CLogMessage(this).warning(u"Invalid grid definition: delta longitude is inconsistent");
587  return;
588  }
589  }
590  }
591  else
592  {
593  dx = 0.0;
594  }
595  dy = fabs(dy);
596 
597  if (nx > 0 && ny > 0)
598  {
599  for (int iy = 0; iy < ny; iy++)
600  {
601  int i = nx * iy;
602  for (int ix = 0; ix < nx; ix++)
603  {
604  GfsGridPoint gridPoint;
605  gridPoint.latitude = latitude1 - iy * dy;
606  gridPoint.longitude = longitude1 + ix * dx;
607  if (gridPoint.longitude >= 360.0f) { gridPoint.longitude -= 360.0f; }
608  if (gridPoint.longitude < 0.0f) { gridPoint.longitude += 360.0f; }
609  gridPoint.fieldPosition = ix + i;

854  CTemperature CWeatherDataGfs::calculateDewPoint(const CTemperature &temperature, double relativeHumidity)
855  {
856  // https://en.wikipedia.org/wiki/Dew_point#Calculating_the_dew_point
857  const double a = 6.112;
858  const double b = 17.67;
859  const double c = 243.5;
860 
861  double temperatureInCelsius = temperature.value(CTemperatureUnit::C());
862  double saturationVaporPressure = a * std::exp((b * temperatureInCelsius) / (c + temperatureInCelsius));
863 
864  double vaporPressure = (relativeHumidity / 100.0) * saturationVaporPressure;
865  // std::log(0.0) is not defined. Hence we use the smallest value possible close to 0.0, if RH is 0.0.
866  // In real weather, RH 0.0 % will never exist.
867  vaporPressure = qMax(vaporPressure, std::numeric_limits<double>::min());
868 
869  double dewPointInCelsius = c * std::log(vaporPressure / a) / (b - std::log(vaporPressure / a));
870  return { dewPointInCelsius, CTemperatureUnit::C() };
871  }
