130  void Client::command(QString line)
131  {
132  QTextStream stream(&line, QIODevice::ReadOnly);
133  QString cmd;
134  stream >> cmd;
135  stream.skipWhiteSpace();
136 
137  auto found = m_commands.constFind(cmd);
138  if (found == m_commands.constEnd())
139  {
140  std::cout << "No such command" << std::endl;
141  }
142  else
143  {
144  (*found)(stream);
145  }
146  }

259  void Client::sendFlightPlanCmd(QTextStream &args)
260  {
261  QString equipmentIcao;
262  QString originAirportIcao;
263  QString destinationAirportIcao;
264  QString alternateAirportIcao;
265  QString takeoffTimePlanned;
266  QString takeoffTimeActual;
267  int enrouteTime;
268  int fuelTime;
269  int cruiseAltitude;
270  int cruiseTrueAirspeed;
271  QString flightRulesString;
272  QString route;
273  args >> equipmentIcao >> originAirportIcao >> destinationAirportIcao >> alternateAirportIcao >> takeoffTimePlanned >> takeoffTimeActual >> enrouteTime >> fuelTime >> cruiseAltitude >> cruiseTrueAirspeed >> flightRulesString >> route;
274 
275  const CFlightPlan::FlightRules flightRules = CFlightPlan::stringToFlightRules(flightRulesString);
276  const CCallsign callsign("DAMBZ");
277  CFlightPlan fp(callsign, equipmentIcao, originAirportIcao, destinationAirportIcao, alternateAirportIcao,
278  fromStringUtc(takeoffTimePlanned, "hhmm"), fromStringUtc(takeoffTimeActual, "hhmm"),
279  CTime(enrouteTime, CTimeUnit::hrmin()),
280  CTime(fuelTime, CTimeUnit::hrmin()),
281  CAltitude(cruiseAltitude, CAltitude::MeanSeaLevel, CLengthUnit::ft()),
282  CSpeed(cruiseTrueAirspeed, CSpeedUnit::kts()),
283  flightRules, route, args.readAll());
284  emit sendFlightPlan(fp);
285  }

433  {
434  switch (newStatus)
435  {
436  case BlackCore::INetwork::Disconnected: std::cout << "CONN_STATUS_DISCONNECTED"; break;

438  case BlackCore::INetwork::DisconnectedError: std::cout << "CONN_STATUS_DISCONNECTED_ERROR"; break;
439  case BlackCore::INetwork::DisconnectedFailed: std::cout << "CONN_STATUS_DISCONNECTED_FAILED"; break;
440  case BlackCore::INetwork::DisconnectedLost: std::cout << "CONN_STATUS_DISCONNECTED_LOST"; break;
441  case BlackCore::INetwork::Connecting: std::cout << "CONN_STATUS_CONNECTING"; break;
442  case BlackCore::INetwork::Connected: std::cout << "CONN_STATUS_CONNECTED"; break;
443  }
444  switch (oldStatus)
445  {
446  case BlackCore::INetwork::Disconnected: std::cout << " (was CONN_STATUS_DISCONNECTED)\n"; break;
447  case BlackCore::INetwork::Disconnecting: std::cout << " (was CONN_STATUS_DISCONNECTING)\n"; break;
448  case BlackCore::INetwork::DisconnectedError: std::cout << " (was CONN_STATUS_DISCONNECTED_ERROR)\n"; break;
449  case BlackCore::INetwork::DisconnectedFailed: std::cout << " (was CONN_STATUS_DISCONNECTED_FAILED)\n"; break;
450  case BlackCore::INetwork::DisconnectedLost: std::cout << " (was CONN_STATUS_DISCONNECTED_LOST)\n"; break;
451  case BlackCore::INetwork::Connecting: std::cout << " (was CONN_STATUS_CONNECTING)\n"; break;
452  case BlackCore::INetwork::Connected: std::cout << " (was CONN_STATUS_CONNECTED)\n"; break;
453  }
