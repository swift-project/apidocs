97  MinMaxAverageHeight findMinMaxAverageHeight() const
98  {
99  MinMaxAverageHeight stats { Aviation::CAltitude::null(), Aviation::CAltitude::null(), Aviation::CAltitude::null(), 0 };
100  if (this->container().isEmpty()) { return stats; } // avoid div by zero
101  int count = 0;
102  double avgFt = 0;
103  for (const OBJ &obj : this->container())
104  {
105  if (!obj.hasMSLGeodeticHeight()) { continue; }
106  const Aviation::CAltitude alt = obj.geodeticHeight();
107  if (std::get<0>(stats).isNull() || std::get<0>(stats) > alt)
108  {
109  std::get<0>(stats) = alt; // min.
110  }
111  if (std::get<1>(stats).isNull() || std::get<1>(stats) < alt)
112  {
113  std::get<1>(stats) = alt; // max.
114  }
115  avgFt += alt.value(PhysicalQuantities::CLengthUnit::ft()); // add up
116  count++;
117  }
118 
119  if (count > 0) { std::get<2>(stats) = Aviation::CAltitude(avgFt / count, Aviation::CAltitude::MeanSeaLevel, PhysicalQuantities::CLengthUnit::ft()); }
120  std::get<3>(stats) = count;
121  return stats;
122  }
