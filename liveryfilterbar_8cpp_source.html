66  void CLiveryFilterBar::filter(const CLivery &livery)
67  {
68  this->clearForm();
69  bool filter = false;
70  if (livery.hasCombinedCode())
71  {
72  ui->le_LiveryCode->setText(livery.getCombinedCode());
73  filter = true;
74  }
75  if (livery.hasValidAirlineDesignator())
76  {
77  ui->le_AirlineIcaoCode->setText(livery.getAirlineIcaoCodeDesignator());
78  filter = true;
79  }
80  if (livery.isColorLivery())
81  {
82  ui->color_Fuselage->setColor(livery.getColorFuselage());
83  ui->color_Tail->setColor(livery.getColorTail());
84  filter = true;
85  }
86  if (filter)
87  {
88  ui->filter_Buttons->clickButton(CFilterBarButtons::Filter);
89  }
90  else
91  {
92  ui->filter_Buttons->clickButton(CFilterBarButtons::RemoveFilter);
93  }
94  }
