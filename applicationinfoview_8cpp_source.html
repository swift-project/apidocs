39  void CApplicationInfoView::deleteSelectedDataDirectories()
40  {
41  if (!this->hasSelection()) { return; }
42  const QMessageBox::StandardButton reply = QMessageBox::question(this, "Delete?", "Delete selected data directories?", QMessageBox::Yes | QMessageBox::No);
43  if (reply != QMessageBox::Yes) { return; }
44 
45  QStringList deletedDirectories;
46  for (const CApplicationInfo &info : this->selectedObjects())
47  {
48  const QString d = CFileUtils::fixWindowsUncPath(info.getApplicationDataDirectory());
49  QDir dir(d);
50  if (!dir.exists()) { continue; }
51  if (dir.removeRecursively())
52  {
53  deletedDirectories << d;
54  }
55  }
56  if (deletedDirectories.isEmpty()) { return; }
57  this->otherSwiftVersionsFromDataDirectories(true);
58  }
