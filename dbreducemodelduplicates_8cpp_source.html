70  void CDbReduceModelDuplicates::process()
71  {
72  if (!sGui || sGui->isShuttingDown()) { return; }
73 
74  this->clearProgressBar();
75  m_stop = false;
76 
77  if (m_models.isEmpty())
78  {
79  const CStatusMessage m = CStatusMessage(this).validationError(u"No models");
80  ui->fr_Overlay->showOverlayHTMLMessage(m, 5000);
81  return;
82  }
83 
84  if (ui->le_Distributor->text().isEmpty())
85  {
86  const CStatusMessage m = CStatusMessage(this).validationError(u"No distributor");
87  ui->fr_Overlay->showOverlayHTMLMessage(m, 5000);
88  return;
89  }
90 
91  const CAircraftModelList keyDuplicates = m_models.findDuplicateModelStrings();
92  if (!keyDuplicates.isEmpty())
93  {
94  const CStatusMessage m = CStatusMessage(this).validationError(u"Found %1 key duplicates") << keyDuplicates.size();
95  ui->fr_Overlay->showOverlayHTMLMessage(m, 5000);
96  return;
97  }
98 
99  const CDistributorList distributors = m_models.getDistributors();
100  const CDistributor distributor = distributors.findByKeyOrAlias(ui->le_Distributor->text().trimmed().toUpper());
101  if (!distributor.hasValidDbKey())
102  {
103  const CStatusMessage m = CStatusMessage(this).validationError(u"Invalid distributor");
104  ui->fr_Overlay->showOverlayHTMLMessage(m, 5000);
105  return;
106  }
107 
108  const CAircraftModelList distributorModels = m_models.findByDistributor(distributor);
109  CAircraftModelList otherDistributorsModels = m_models;
110  otherDistributorsModels.removeByDistributor(distributor); // all models of other distrbutors
111  CAircraftModelList removeModels;
112 
113  ui->pb_Progress->setMaximum(distributorModels.sizeInt());
114  ui->pb_Progress->setValue(0);
115 
116  int mc = 0;
117  for (const CAircraftModel &distributorModel : distributorModels)
118  {
119  if (m_stop) { break; }
120  if (!distributorModel.getLivery().isAirlineLivery()) { continue; }
121  if (!distributorModel.getAircraftIcaoCode().hasValidDbKey()) { continue; }
122  if (ui->rb_SameLiveryAndAircraft->isChecked())
123  {
124  removeModels.replaceOrAddModelsWithString(otherDistributorsModels.findByAircraftAndLivery(distributorModel.getAircraftIcaoCode(), distributorModel.getLivery()), Qt::CaseInsensitive);
125  }
126  else if (ui->rb_SameAirlineAndAircraft->isChecked())
127  {
128  removeModels.replaceOrAddModelsWithString(otherDistributorsModels.findByAircraftAndAirline(distributorModel.getAircraftIcaoCode(), distributorModel.getAirlineIcaoCode()), Qt::CaseInsensitive);
129  }
130 
131  if (mc % 50 == 0) { ui->pb_Progress->setValue(mc); }
132  if (mc % 100 == 0) { CGuiApplication::processEventsFor(10); }
133  mc++;
134  }
135 
136  if (!m_stop) { ui->pb_Progress->setValue(distributorModels.sizeInt()); }
137  ui->tvp_RemoveModels->updateContainerMaybeAsync(removeModels);
138  m_removeCandidates = removeModels;
139  m_stop = false;
140 
141  const QString distKeys = removeModels.getDistributors().dbKeysAsString(", ");
142  const CStatusMessage msg = removeModels.isEmpty() ?
143  CStatusMessage(this).info(u"No duplicates to be removed!") :
144  CStatusMessage(this).info(u"You can remove %1 models of the following distributors: '%2'.") << removeModels.size() << distKeys;
145  ui->fr_Overlay->showOverlayHTMLMessage(msg, 5000);
146  ui->bb_ReduceModelDuplicates->button(QDialogButtonBox::Ok)->setEnabled(true);
147  }
