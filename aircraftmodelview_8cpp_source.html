216  void CAircraftModelView::dropEvent(QDropEvent *event)
217  {
218  // moves from myself are ignored
219  // depends on isDropAllowed() / acceptDrop() if this function is called
220  Qt::DropAction action = event->dropAction();
221  if (action == Qt::MoveAction)
222  {
223  COrderableViewWithDbObjects::dropEvent(event);
224  return;
225  }
226  if (!this->isDropAllowed()) { return; }
227  if (!event) { return; }
228  const QMimeData *mime = event->mimeData();
229  if (!mime) { return; }
230 
231  if (CGuiUtility::hasSwiftVariantMimeType(mime))
232  {
233  const CVariant valueVariant(CGuiUtility::fromSwiftDragAndDropData(mime));
234  if (valueVariant.isValid())
235  {
236  if (valueVariant.canConvert<CAircraftModel>())
237  {
238  const CAircraftModel model = valueVariant.value<CAircraftModel>();
239  if (!model.hasModelString()) { return; }
240  const CAircraftModelList models({ model });
241  this->derivedModel()->replaceOrAddByModelString(models);
242  return;
243  }
244  else if (valueVariant.canConvert<CAircraftModelList>())
245  {
246  const CAircraftModelList models(valueVariant.value<CAircraftModelList>());
247  if (models.isEmpty()) { return; }
248  this->derivedModel()->replaceOrAddByModelString(models);
249  return;
250  }
251 
252  // only for selected members
253  if (!this->hasSelection()) { return; }
254  if (valueVariant.canConvert<CAircraftIcaoCode>())
255  {
256  const CAircraftIcaoCode icao = valueVariant.value<CAircraftIcaoCode>();
257  if (icao.validate().hasErrorMessages()) { return; }
258  this->applyToSelected(icao);
259  }
260  else if (valueVariant.canConvert<CAircraftIcaoCodeList>())
261  {
262  const CAircraftIcaoCodeList icaos(valueVariant.value<CAircraftIcaoCodeList>());
263  if (icaos.size() != 1) { return; }
264  const CAircraftIcaoCode icao = icaos.front();
265  if (icao.validate().hasErrorMessages()) { return; }
266  this->applyToSelected(icao);
267  }
268  else if (valueVariant.canConvert<CLivery>())
269  {
270  const CLivery livery = valueVariant.value<CLivery>();
271  if (livery.validate().hasErrorMessages()) { return; }
272  this->applyToSelected(livery);
273  }
274  else if (valueVariant.canConvert<CLiveryList>())
275  {
276  const CLiveryList liveries(valueVariant.value<CLiveryList>());
277  if (liveries.size() != 1) { return; }
278  const CLivery livery = liveries.front();
279  if (livery.validate().hasErrorMessages()) { return; }
280  this->applyToSelected(livery);
281  }
282  else if (valueVariant.canConvert<CDistributor>())
283  {
284  const CDistributor distributor = valueVariant.value<CDistributor>();
285  if (distributor.validate().hasErrorMessages()) { return; }
286  this->applyToSelected(distributor);
287  }
288  else if (valueVariant.canConvert<CDistributorList>())
289  {
290  const CDistributorList distributors(valueVariant.value<CDistributorList>());
291  if (distributors.size() != 1) { return; }
292  const CDistributor distributor = distributors.front();
293  if (distributor.validate().hasErrorMessages()) { return; }
294  this->applyToSelected(distributor);
295  }
296  else if (valueVariant.canConvert<CAirlineIcaoCode>())
297  {
298  const CAirlineIcaoCode airline = valueVariant.value<CAirlineIcaoCode>();
299  if (airline.validate().hasErrorMessages()) { return; }
300  emit requestHandlingOfStashDrop(airline); // I need to convert to stanard livery, which I can`t do here
301  }
302  else if (valueVariant.canConvert<CAirlineIcaoCodeList>())
303  {
304  const CAirlineIcaoCodeList airlines(valueVariant.value<CAirlineIcaoCodeList>());
305  if (airlines.size() != 1) { return; }
306  const CAirlineIcaoCode airline = airlines.front();
307  if (airline.validate().hasErrorMessages()) { return; }
308  emit requestHandlingOfStashDrop(airline); // I need to convert to stanard livery, which I can`t do here
309  }
310  }
311  }
312  else
313  {
314  CViewBase::dropEvent(event);
315  }
316  }

318  void CAircraftModelView::customMenu(CMenuActions &menuActions)
319  {
320  // Statistics and validation
321  if (!this->isEmpty())
322  {
323  menuActions.addAction(CIcons::appAircraft16(), "Model statistics", CMenuAction::pathModel(), { this, &CAircraftModelView::displayModelStatisticsDialog });
324  if (m_withValidationContextMenu)
325  {
326  QDialog *parentDialog = CGuiUtility::findParentDialog(this);
327  if (!(parentDialog && (qobject_cast<CAircraftModelValidationDialog *>(parentDialog))))
328  {
329  menuActions.addAction(CIcons::disk16(), "Model validation (selected)", CMenuAction::pathModel(), { this, &CAircraftModelView::displayModelValidationDialog });
330  }
331  }
332  if (CAircraftModelList::hasInvalidModelFile())
333  {
334  menuActions.addAction(CIcons::disk16(), "Highlight invalid models (from file)", CMenuAction::pathModel(), { this, &CAircraftModelView::setHighlightModelsForInvalidModels });
335  }
336  }
337 
338  // Stash menus
339  bool addStashMenu = false;
340  if (m_menus.testFlag(MenuCanStashModels))
341  {
342  if (!m_menuFlagActions.contains(MenuCanStashModels))
343  {
344  CMenuActions ma;
345  ma.addAction(CIcons::appDbStash16(), "Stash selected", CMenuAction::pathModelStash(), { this, &CAircraftModelView::requestedStash });
346  QAction *added = ma.addAction(CIcons::appDbStash16(), "Stashing clears selection (on/off)", CMenuAction::pathModelStash(), { this, &CAircraftModelView::stashingClearsSelection });
347  added->setCheckable(true);
348  m_menuFlagActions.insert(MenuCanStashModels, ma);
349  }
350 
351  // modify menu items
352  const int selected = this->selectedRowCount();
353  const bool tooMany = selected > 500;
354  const bool canStash = selected > 0 && !tooMany;
355  QAction *a = menuActions.addActions(initMenuActions(MenuCanStashModels)).first();
356  a->setEnabled(canStash);
357  a = menuActions.addActions(initMenuActions(MenuCanStashModels)).last();
358  a->setChecked(m_stashingClearsSelection);
359  addStashMenu = true;
360  }
361  if (m_menus.testFlag(MenuHighlightStashed))
362  {
363  // this function requires that someone provides the model strings to be highlighted
364  if (!m_menuFlagActions.contains(MenuHighlightStashed))
365  {
366  CMenuActions ma;
367  QAction *added = ma.addAction(CIcons::appDbStash16(), "Highlight stashed (on/off)", CMenuAction::pathModelStash(), { this, &CAircraftModelView::toggleHighlightStashedModels });
368  added->setCheckable(true);
369  m_menuFlagActions.insert(MenuHighlightStashed, ma);
370  }
371  QAction *a = menuActions.addActions(initMenuActions(CViewBaseNonTemplate::MenuHighlightStashed)).first();
372  a->setChecked(this->derivedModel()->highlightModels());
373  addStashMenu = true;
374  }
375 
376  // client specific
377  if (m_menus.testFlag(MenuDisableModelsTemp) && this->hasSelection())
378  {
379  if (!m_menuFlagActions.contains(MenuDisableModelsTemp))
380  {
381  CMenuActions ma;
382  ma.addAction(CIcons::delete16(), "Temp.disable model", CMenuAction::pathModel(), { this, &CAircraftModelView::requestTempDisable });
383  m_menuFlagActions.insert(MenuDisableModelsTemp, ma);
384  }
385  menuActions.addActions(initMenuActions(CViewBaseNonTemplate::MenuDisableModelsTemp));
386  }
387 
388  if (addStashMenu) { menuActions.addMenuStash(); }
389 
390  // base class menus
391  COrderableViewWithDbObjects::customMenu(menuActions);
392  }

394  CStatusMessage CAircraftModelView::modifyLoadedJsonData(CAircraftModelList &models) const
395  {
396  if (m_correspondingSimulator.isNoSimulator()) { return {}; }
397  if (models.isEmpty()) { return CStatusMessage(this, CStatusMessage::SeverityDebug, u"Empty models", true); }
398 
399  // multiple sims with same count
400  const int removed = models.removeIfNotMatchingSimulator(m_correspondingSimulator);
401  if (removed < 1) { return {}; }
402  return CStatusMessage(this, CStatusMessage::SeverityWarning, u"Reduced by %1 model(s) to only use %2 models", true) << removed << m_correspondingSimulator.toQString(true);
403  }

405  CStatusMessage CAircraftModelView::validateLoadedJsonData(const CAircraftModelList &models) const
406  {
407  if (models.isEmpty()) { return COrderableViewWithDbObjects::validateLoadedJsonData(models); }
408  if (m_correspondingSimulator.isNoSimulator()) { return COrderableViewWithDbObjects::validateLoadedJsonData(models); }
409  if (models.containsNotMatchingSimulator(m_correspondingSimulator))
410  {
411  return CStatusMessage(this, CStatusMessage::SeverityError, u"Found entry not matching %1 in model data", true) << m_correspondingSimulator.toQString();
412  }
413  return COrderableViewWithDbObjects::validateLoadedJsonData(models);
414  }
