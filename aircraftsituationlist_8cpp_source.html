223  QPair<bool, CAircraftSituation::IsOnGround> CAircraftSituationList::isGndFlagStableChanging(bool alreadySortedLatestFirst) const
224  {
225  if (this->size() < 2) { return QPair<bool, CAircraftSituation::IsOnGround>(false, CAircraftSituation::OnGroundSituationUnknown); }
226 
227  const CAircraftSituationList sorted(alreadySortedLatestFirst ? (*this) : this->getSortedAdjustedLatestFirst());
228  const CAircraftSituation::IsOnGround f = sorted.front().getOnGround();
229  const CAircraftSituation::IsOnGround t = sorted.back().getOnGround();
230  QPair<bool, CAircraftSituation::IsOnGround> ret(false, f); // changing to front (latest)
231  if (f == t) { return ret; }
232 
233  bool changed = false;
234 
235  for (const CAircraftSituation &s : sorted)
236  {
237  if (!changed && s.getOnGround() == f) { continue; } // find 1st changing
238  if (!changed)
239  {
240  changed = true;
241  continue;
242  } // just changed
243  if (s.getOnGround() != t) { return ret; } // jitter, something like gnd, no gnd, gnd
244  }
245  ret.first = changed;
246  return ret;
247  }

341  {

343  CAircraftSituation situationWithElevation = CAircraftSituation::null();
344 
345  CLength bestDistance = CLength::null();

552  CElevationPlane CAircraftSituationList::averageElevationOfTaxiingOnGroundAircraft(const CAircraftSituation &reference, const CLength &range, int minValues, int sufficientValues) const
553  {
554  if (this->size() < minValues) { return CElevationPlane::null(); } // no change to succeed
555 
556  const CAircraftSituationList sorted = this->findWithGeodeticMSLHeight().findWithinRange(reference, range).sortedByEuclideanDistanceSquared(reference);
557  if (sorted.size() < minValues) { return CElevationPlane::null(); }
558  QList<double> valuesInFt;
559  for (const CAircraftSituation &situation : *this)
560  {
561  if (situation.getGroundElevationInfo() != CAircraftSituation::FromProvider) { continue; }
562  const bool canUse = !situation.isMoving() || (situation.isOnGroundFromNetwork() || situation.isOnGroundFromParts());
563  if (!canUse) { continue; }
564 
565  const double elvFt = situation.getGroundElevationPlane().getAltitude().value(CLengthUnit::ft());
566  valuesInFt.push_back(elvFt);
567  if (valuesInFt.size() >= sufficientValues) { break; }
568  }
569 
570  if (valuesInFt.size() < minValues) { return CElevationPlane::null(); }
571 
572  static const double MaxDevFt = CAircraftSituation::allowedAltitudeDeviation().value(CLengthUnit::ft());
573  const QPair<double, double> elvStdDevMean = CMathUtils::standardDeviationAndMean(valuesInFt);
574  if (elvStdDevMean.first > MaxDevFt) { return CElevationPlane::null(); }
575  return CElevationPlane(reference, elvStdDevMean.second, CElevationPlane::singlePointRadius());
576  }
