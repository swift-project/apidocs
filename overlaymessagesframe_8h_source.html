286  void initInnerFrame(double widthFactor = -1, double heightFactor = -1)
287  {
288  const QSize inner(innerFrameSize(widthFactor, heightFactor));
289  if (!m_overlayMessages)
290  {
291  // lazy init
292  this->initOverlayMessages(inner);
293  }
294 
295  Q_ASSERT(m_overlayMessages);
296 
297  const QPoint middle = WIDGET::geometry().center();
298  const int w = inner.width();
299  const int h = inner.height();
300  const int x = middle.x() - w / 2;
301  const int y = qRound(middle.y() - h / m_middleFactor);
302 
303  m_overlayMessages->setGeometry(x, y, w, h);
304  m_overlayMessages->setVisible(true);
305  }

308  void initMinimalFrame(int lines = 4)
309  {
310  this->initInnerFrame();
311 
312  // get logical resolution
313  constexpr int MinHeight = 100;
314  QSizeF s = CGuiUtility::fontMetricsEstimateSize(100, lines, true); // 2 lines for header
315  if (s.height() < MinHeight) { s.setHeight(MinHeight); }
316 
317  const QSize inner(innerFrameSize());
318  const QPoint middle = WIDGET::geometry().center();
319  const int w = qMin(inner.width(), qRound(s.width()));
320  const int h = qMin(inner.height(), qRound(s.height()));
321  const int x = middle.x() - w / 2;
322  const int y = qRound(middle.y() - h / m_middleFactor);
323  m_overlayMessages->setGeometry(x, y, w, h);
324  }

369  QSize innerFrameSize(double widthFactor = -1, double heightFactor = -1) const
370  {
371  // check against minimum if widget is initialized, but not yet resized
372  const int w = std::max(WIDGET::width(), WIDGET::minimumWidth());
373  const int h = std::max(WIDGET::height(), WIDGET::minimumHeight());
374 
375  widthFactor = qMin(widthFactor < 0 ? m_widthFactor : widthFactor, 0.95);
376  heightFactor = qMin(heightFactor < 0 ? m_heightFactor : heightFactor, 0.95);
377 
378  int wInner = qRound(widthFactor * w);
379  int hInner = qRound(heightFactor * h);
380  if (wInner > WIDGET::maximumWidth()) { wInner = WIDGET::maximumWidth(); }
381  if (hInner > WIDGET::maximumHeight()) { hInner = WIDGET::maximumHeight(); }
382  return QSize(wInner, hInner);
383  }
