226  struct Readiness

230  MatchingReadiness readyness = NotReady;

232  public:
234  Readiness &addFlag(MatchingReadinessFlag flag)
235  {
236  if (flag == RecursiveCall) { return *this; }
237  if (ts < 0) { ts = QDateTime::currentMSecsSinceEpoch(); }
238  readyness.setFlag(flag, true);
239  return *this;
240  }
241 
243  void resetTimestampToNow() { ts = QDateTime::currentMSecsSinceEpoch(); }
244 
246  Readiness &setFlag(MatchingReadinessFlag flag)
247  {
248  if (flag == RecursiveCall) { return *this; }
249  readyness = NotReady;
250  this->addFlag(flag);
251  return *this;
252  }
253 
255  bool receivedAll() const { return readyness.testFlag(ReceivedIcaoCodes) && readyness.testFlag(ReceivedFsInnPacket); }
256 
258  bool receivedIcaoCodes() const { return readyness.testFlag(ReceivedIcaoCodes); }
259 
261  bool wasMatchingSent() const { return readyness.testFlag(ReadyForMatchingSent); }
262 
264  bool wasVerified() const { return readyness.testFlag(Verified); }

267  qint64 getAgeMs() const

269  if (ts < 0) { return -1; }
270  return QDateTime::currentMSecsSinceEpoch() - ts;
271  }
