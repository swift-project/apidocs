47  inline void packPBH(double pitch, double bank, double heading, bool onGround, quint32 &pbh)
48  {
49  PBH pbhstrct;
50  pbhstrct.pitch = qFloor(pitch * -pitchMultiplier()); // the "-" is the inverted pitch/bank
51  pbhstrct.bank = qFloor(bank * -bankMultiplier());
52  pbhstrct.hdg = static_cast<unsigned int>(heading * headingMultiplier());
53 
54  // FSD has inverted pitch and bank angles
55  // based on discussion https://discordapp.com/channels/539048679160676382/539925070550794240/687390301530095634
56  // we use *-1 inversion, not "~" (vPilot also uses *-1)
57  // with "~" 0->-1 (that is why we saw many -1 PB values on ground)
58  // pbhstrct.pitch = ~pbhstrct.pitch;
59  // pbhstrct.bank = ~pbhstrct.bank;
60 
61  pbhstrct.onground = onGround ? 1 : 0;
62  pbh = pbhstrct.pbh;
63  }
