68  template <typename T>
69  std::function<void(QDBusPendingCallWatcher *)> setterCallback(T *obj)
70  {
71  return [this, obj](QDBusPendingCallWatcher *watcher) {
72  QDBusPendingReply<T> reply = *watcher;
73  if (reply.isError()) { emit this->asyncMethodError(reply.error()); }
74  else { *obj = reply; }
75  watcher->deleteLater();
76  };
77  }
