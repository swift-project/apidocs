25  CAircraftModelList CModelSetBuilder::buildModelSet(const CSimulatorInfo &simulator, const CAircraftModelList &models, const CAircraftModelList &currentSet, Builder options, const CDistributorList &distributors) const
26  {
27  if (models.isEmpty()) { return CAircraftModelList(); }
28  CAircraftModelList modelSet;
29 
30  // Select by distributor:
31  // I avoid an empty distributor set because it wipes out everything
32  if (options.testFlag(GivenDistributorsOnly) && !distributors.isEmpty())
33  {
34  modelSet = models.findByDistributors(distributors);
35  }
36  else
37  {
38  modelSet = models;
39  }
40 
41  // Only DB data?
42  if (options.testFlag(OnlyDbData))
43  {
44  modelSet.removeObjectsWithoutDbKey();
45  }
46  else if (options.testFlag(OnlyDbIcaoCodes))
47  {
48  Q_ASSERT_X(sApp->hasWebDataServices(), Q_FUNC_INFO, "No web data services");
49  const QSet<QString> designators(sApp->getWebDataServices()->getAircraftIcaoCodes().allDesignators());
50  modelSet = modelSet.findWithAircraftDesignator(designators);
51  }
52  else
53  {
54  // without any information we can not use them
55  modelSet = modelSet.findWithKnownAircraftDesignator();
56  }
57 
58  // Include only
59  modelSet = modelSet.matchesSimulator(simulator);
60  modelSet.setModelMode(CAircraftModel::Include); // in sets we only include, exclude means not present in set
61 
62  if (options.testFlag(Incremental))
63  {
64  if (!currentSet.isEmpty())
65  {
66  // update in full set
67  CAircraftModelList copy(currentSet);
68  copy.replaceOrAddModelsWithString(modelSet, Qt::CaseInsensitive);
69  modelSet = copy;
70  }
71  }
72 
73  // sort by preferences if applicable
74  modelSet.resetOrder();
75  if (options.testFlag(SortByDistributors))
76  {
77  modelSet.updateDistributorOrder(distributors);
78  modelSet.sortBy(&CAircraftModel::getDistributorOrder);
79  }
80 
81  // DB consolidation
82  if (options.testFlag(ConsolidateWithDb))
83  {
84  CDatabaseUtils::consolidateModelsWithDbData(modelSet, true);
85  }
86 
87  // result
88  return modelSet;
89  }
90 } // ns
