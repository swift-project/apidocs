47  void CAtcStationListModel::setStationMode(CAtcStationListModel::AtcStationMode stationMode)
48  {
49  if (m_stationMode == stationMode) return;
50  m_stationMode = stationMode;
51  m_columns.clear();
52  switch (stationMode)
53  {
54  case NotSet:
55  case StationsOnline:
56  {
57  m_columns.addColumn(CColumn::standardString("cs.", "callsign", { CAtcStation::IndexCallsign, CCallsign::IndexCallsignStringAsSet }));
58  CColumn col("type", CAtcStation::IndexIcon);
59  col.setSortPropertyIndex({ CAtcStation::IndexCallsign, CCallsign::IndexSuffix });
60  m_columns.addColumn(col);
61  m_columns.addColumn(CColumn("frequency", CAtcStation::IndexFrequency, new CComFrequencyFormatter()));
62  m_columns.addColumn(CColumn("distance", CAtcStation::IndexRelativeDistance, new CAirspaceDistanceFormatter()));
63  m_columns.addColumn(CColumn("r.", "in range", CAtcStation::IndexIsInRange, new CBoolIconFormatter("in range", "outside range")));
64  m_columns.addColumn(CColumn("xc.", "AFV cross coupled", CAtcStation::IndexIsAfvCrossCoupled, new CBoolIconFormatter("AFV cross coupled", "not coupled")));
65  m_columns.addColumn(CColumn("range", CAtcStation::IndexRange, new CAirspaceDistanceFormatter()));
66  m_columns.addColumnIncognito(CColumn::standardString("controllername", { CAtcStation::IndexController, CUser::IndexRealName }));
67  m_columns.addColumn(CColumn("from", "booked from", CAtcStation::IndexBookedFrom, new CDateTimeFormatter(CDateTimeFormatter::formatHm())));
68  m_columns.addColumn(CColumn("until", "booked until", CAtcStation::IndexBookedUntil, new CDateTimeFormatter(CDateTimeFormatter::formatHm())));
69 
70  // default sort order
71  this->setSortColumnByPropertyIndex(CAtcStation::IndexRelativeDistance);
72  m_sortOrder = Qt::AscendingOrder;
73  }
74  break;
75 
76  case StationsBooked:
77  {
78  m_columns.addColumn(CColumn::standardString("cs.", "callsign", { CAtcStation::IndexCallsign, CCallsign::IndexCallsignStringAsSet }));
79  CColumn col = CColumn("type", CAtcStation::IndexIcon);
80  col.setSortPropertyIndex({ CAtcStation::IndexCallsign, CCallsign::IndexSuffix });
81  m_columns.addColumn(col);
82  m_columns.addColumn(CColumn("", "on/offline", CAtcStation::IndexIsOnline, new CBoolLedFormatter("online", "offline")));
83  m_columns.addColumnIncognito(CColumn::standardString("controllername", { CAtcStation::IndexController, CUser::IndexRealName }));
84  m_columns.addColumn(CColumn("from", "booked from", CAtcStation::IndexBookedFrom, new CDateTimeFormatter(CDateTimeFormatter::formatYmdhm())));
85  m_columns.addColumn(CColumn("until", "booked until", CAtcStation::IndexBookedUntil, new CDateTimeFormatter(CDateTimeFormatter::formatYmdhm())));
86  m_columns.addColumn(CColumn("frequency", CAtcStation::IndexFrequency, new CComFrequencyFormatter()));
87 
88  // default sort order
89  this->setSortColumnByPropertyIndex(CAtcStation::IndexBookedFrom);
90  m_sortOrder = Qt::AscendingOrder;
91  }
92  break;
93 
94  default:
95  qFatal("Wrong mode");
96  break;
97  }
98  }

108  void CAtcStationListModel::changedAtcStationConnectionStatus(const CAtcStation &station, bool added)
109  {
110  if (station.getCallsign().isEmpty()) { return; }
111  if (added)
112  {
113  const bool c = m_container.contains(&CAtcStation::getCallsign, station.getCallsign());
114  if (!c) { this->insert(station); }
115  }
116  else
117  {
118  QModelIndex parent;
119  if (rowCount(parent) == 0) { return; }
120  beginRemoveRows(parent, 0, 0);
121  this->removeIf(&CAtcStation::getCallsign, station.getCallsign());
122  endRemoveRows();
123  }
124  }
