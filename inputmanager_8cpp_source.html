152  void CInputManager::processCombination(const CHotkeyCombination &currentCombination)
153  {
154  if (m_captureActive)
155  {
156  CHotkeyCombination deltaCombination = currentCombination.getDeltaComparedTo(m_combinationBeforeCapture);
157 
158  // Don't continue if there is no relevant combination yet
159  if (m_capturedCombination.isEmpty() && deltaCombination.isEmpty()) { return; }
160 
161  if (deltaCombination.size() < m_capturedCombination.size())
162  {
163  emit combinationSelectionFinished(m_capturedCombination);
164  m_captureActive = false;
165  }
166  else
167  {
168  emit combinationSelectionChanged(deltaCombination);
169  m_capturedCombination = deltaCombination;
170  }
171  return;
172  }
173 
174  QSet<QString> newActiveActions;
175  for (const auto [combination, action] : makePairsRange(std::as_const(m_configuredActions)))
176  {
177  if (combination.isSubsetOf(currentCombination))
178  {
179  newActiveActions.insert(action);
180  }
181  }
182 
183  const QSet<QString> pressedActions = newActiveActions - m_activeActions;
184  const QSet<QString> releasedActions = m_activeActions - newActiveActions;
185  m_activeActions = newActiveActions;
186  for (const QString &action : pressedActions)
187  {
188  callFunctionsBy(action, true);
189  }
190  for (const QString &action : releasedActions)
191  {
192  callFunctionsBy(action, false);
193  }
194 
195  // combination
196  m_lastCombination = currentCombination;
197  }
