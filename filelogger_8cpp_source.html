73  void CFileLogger::writeStatusMessageToFile(const BlackMisc::CStatusMessage &statusMessage)
74  {
75  if (statusMessage.isEmpty()) { return; }
76  if (!m_logFile.isOpen()) { return; }
77  if (!m_logPattern.match(statusMessage)) { return; }
78  const QString categories = statusMessage.getCategoriesAsString();
79  if (categories != m_previousCategories)
80  {
81  writeContentToFile(u"\n[" % categories % u']');
82  m_previousCategories = categories;
83  }
84  const QString finalContent(QDateTime::currentDateTime().toString(QStringLiteral("hh:mm:ss ")) % statusMessage.getSeverityAsString() % u": " % statusMessage.getMessage());
85 
86  writeContentToFile(finalContent);
87  }

95  void CFileLogger::removeOldLogFiles()
96  {
97  QString nameFilter(applicationName());
98  nameFilter += QLatin1String("*.log");
99  QDir dir(CSwiftDirectories::logDirectory(), nameFilter, QDir::Name, QDir::Files);
100 
101  QDateTime now = QDateTime::currentDateTime();
102  for (const auto &logFileInfo : dir.entryInfoList())
103  {
104  if (logFileInfo.lastModified().daysTo(now) > 7)
105  {
106  dir.remove(logFileInfo.fileName());
107  }
108  }
