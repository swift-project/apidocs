86  {

88  const QString sender = this->getSenderCallsign().asString();

97  bool CTextMessage::relayedMessageToPrivateMessage()
98  {
99  if (!this->isRelayedMessage()) { return false; }
100  const int index = m_message.indexOf(';');
101  if (index < CTextMessage::swiftRelayMessage().length()) { return false; }
102  if (m_message.length() <= index + 1) { return false; } // no next line
103  const QString senderRecipient = m_message.left(index).remove(CTextMessage::swiftRelayMessage()).trimmed();
104  const QStringList sr = senderRecipient.split(' ');
105  if (sr.size() != 2) { return false; }
106  const QString originalSender = sr.first();
107  const QString originalRecipient = sr.last();
108  this->setSenderCallsign(CCallsign(originalSender)); // sender can be aircraft or ATC
109  this->setRecipientCallsign(CCallsign(originalRecipient)); // recipient can be aircraft or ATC
110  m_message = m_message.mid(index + 1);
111  return true;
112  }

114  bool CTextMessage::canBeAppended(const CTextMessage &textMessage) const
115  {
116  if (textMessage.isEmpty()) { return false; }
117  if (this->getSenderCallsign() != textMessage.getSenderCallsign()) { return false; }
118  if (this->isRadioMessage() && textMessage.isRadioMessage())
119  {
120  if (this->getFrequency() != textMessage.getFrequency()) { return false; }
121  return true;
122  }
123  else if (this->isPrivateMessage() && textMessage.isPrivateMessage())
124  {
125  if (this->getRecipientCallsign() != textMessage.getRecipientCallsign()) { return false; }
126  return true;
127  }
128  return false;

211  QString CTextMessage::asString(bool withSender, bool withRecipient, const QString &separator) const
212  {
213  QString s(this->getFormattedUtcTimestampHms());
214  if (withSender)
215  {
216  if (!m_senderCallsign.isEmpty())
217  {
218  if (!s.isEmpty()) { s += separator % m_senderCallsign.getStringAsSet(); }
219  }
220  }
221 
222  if (withRecipient)
223  {
224  if (!m_recipientCallsign.isEmpty())
225  {
226  if (!s.isEmpty()) { s += separator % m_recipientCallsign.getStringAsSet(); }
227  }
228  else
229  {
230  if (CComSystem::isValidCivilAviationFrequency(m_frequency))
231  {
232  if (!s.isEmpty()) { s += separator % m_frequency.valueRoundedWithUnit(3, true); }
233  }
234  }
235  } // to
236 
237  if (m_message.isEmpty()) { return s; }
238  if (!s.isEmpty()) { s += separator % m_message; }
239  return s;
240  }

258  bool CTextMessage::isSelcalMessage() const
259  {
260  // some first level checks, before really parsing the message
261  if (this->isEmpty()) { return false; }
262  if (this->isPrivateMessage()) { return false; }
263  if (m_message.length() > 15 || m_message.length() < 10) { return false; } // SELCAL AB-CD -> 12, I allow some more characters as I do not know wheter in real life it exactly matches
264  return this->getSelcalCode().length() == 4;
265  }

273  QString CTextMessage::getSelcalCode() const
274  {
275  // http://forums.vatsim.net/viewtopic.php?f=8&t=63467#p458062
276  // When the ATC client sends a selcal, it goes out as a text message
277  // on their primary frequency, formatted like so:
278  // SELCAL AB-CD
279 
280  if (this->isEmpty()) return {};
281  if (this->isPrivateMessage()) return {};
282  if (!m_message.startsWith(QLatin1String("SELCAL"), Qt::CaseInsensitive)) return {};
283  if (m_message.length() > 15 || m_message.length() < 10) return {}; // SELCAL AB-CD -> 12, I allow some more characters as I do not know wheter in real life it exactly matches
284  QString candidate = removeChars(m_message, [](QChar c) { return !c.isLetter(); }); // SELCALABCD
285  if (candidate.length() != 10) return {};
286  return std::move(candidate).right(4).toUpper();
287  }

299  QVariant CTextMessage::propertyByIndex(BlackMisc::CPropertyIndexRef index) const
300  {
301  if (index.isMyself()) { return QVariant::fromValue(*this); }
302  if (ITimestampBased::canHandleIndex(index)) { return ITimestampBased::propertyByIndex(index); }
303 
304  const ColumnIndex i = index.frontCasted<ColumnIndex>();
305  switch (i)

307  case IndexSenderCallsign: return m_senderCallsign.propertyByIndex(index.copyFrontRemoved());
308  case IndexRecipientCallsign: return m_recipientCallsign.propertyByIndex(index.copyFrontRemoved());
309  case IndexRecipientCallsignOrFrequency: return QVariant::fromValue(this->getRecipientCallsignOrFrequency());
310  case IndexMessage: return QVariant::fromValue(m_message);
311  default: return CValueObject::propertyByIndex(index);
312  }

338  int CTextMessage::comparePropertyByIndex(CPropertyIndexRef index, const CTextMessage &compareValue) const
339  {
340  if (ITimestampBased::canHandleIndex(index)) { return ITimestampBased::comparePropertyByIndex(index, compareValue); }
341  const ColumnIndex i = index.frontCasted<ColumnIndex>();
342  switch (i)
343  {
344  case IndexSenderCallsign: return m_senderCallsign.comparePropertyByIndex(index.copyFrontRemoved(), compareValue.getSenderCallsign());
345  case IndexRecipientCallsign: return m_recipientCallsign.comparePropertyByIndex(index.copyFrontRemoved(), compareValue.getRecipientCallsign());
346  case IndexRecipientCallsignOrFrequency:
347  if (this->isRadioMessage()) { return this->getFrequency().compare(compareValue.getFrequency()); }
348  return m_recipientCallsign.comparePropertyByIndex(index.copyFrontRemoved(), compareValue.getRecipientCallsign());
349  default: return CValueObject::comparePropertyByIndex(index, *this);
350  }
351  Q_ASSERT_X(false, Q_FUNC_INFO, "No comparison");
352  return 0;
353  }
