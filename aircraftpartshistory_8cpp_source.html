78  void CAircraftPartsHistory::updatePartsHistory()
79  {
80  if (!this->hasContexts()) { return; }
81  if (!this->isVisible()) { return; }
82  const CCallsign cs(ui->comp_CallsignCompleter->getCallsign());
83  if (cs.isEmpty()) { return; } // no or invalid callsign
84  const auto currentAircraftParts = sGui->getIContextNetwork()->getRemoteAircraftParts(cs).frontOrDefault();
85  const auto aircraftPartsHistory = sGui->getIContextNetwork()->getAircraftPartsHistory(cs);
86 
87  QString html;
88  if (currentAircraftParts == CAircraftParts() && aircraftPartsHistory.isEmpty())
89  {
90  html = cs.toQString() % u" does not support aircraft parts or nothing received yet.";
91  }
92  else
93  {
94  const QString s =
95  u"lights on:"
96  u"<br>"
97  u"&nbsp;&nbsp;&nbsp;&nbsp;" %
98  currentAircraftParts.getLights().toQString() %
99  u"<br>"
100  u"gear down: " %
101  BlackMisc::boolToYesNo(currentAircraftParts.isGearDown()) %
102  u"<br>"
103  u"flaps pct: " %
104  QString::number(currentAircraftParts.getFlapsPercent()) %
105  u"<br>"
106  u"spoilers out: " %
107  BlackMisc::boolToYesNo(currentAircraftParts.isSpoilersOut()) %
108  u"<br>"
109  u"engines on: "
110  u"<br>"
111  u"&nbsp;&nbsp;&nbsp;&nbsp;" %
112  currentAircraftParts.getEngines().toQString() %
113  u"<br>"
114  u" on ground: " %
115  BlackMisc::boolToYesNo(currentAircraftParts.isOnGround());
116  html += s;
117  if (ui->cb_PartsHistoryEnabled->isChecked())
118  {
119  html += u"<hr>" %
120  aircraftPartsHistory.toHtml(CStatusMessageList::timestampHtmlOutput());
121  }
122  }
123 
124  const uint hash = qHash(html);
125  if (hash == m_htmlHash) { return; } // avoid to always scroll to the end when there is no update
126  m_htmlHash = hash;
127  m_text.setHtml(html);
128  ui->te_Messages->setDocument(&m_text);
129 
130  if (ui->cb_AutoScrollEnabled->isChecked())
131  {
132  QTextCursor c = ui->te_Messages->textCursor();
133  c.movePosition(QTextCursor::End);
134  ui->te_Messages->setTextCursor(c);
135  }
136  }
