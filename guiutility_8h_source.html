138  static OverlayWidget *nextOverlayMessageWidget(QWidget *widget, int maxLevels = 10)
139  {
140  if (!widget || maxLevels < 1) { return nullptr; }
141  OverlayWidget *o = qobject_cast<OverlayWidget *>(widget);
142  if (o) { return o; }
143  int cl = 0;
144  QWidget *cw = widget->parentWidget();
145  while (cl < maxLevels && cw)
146  {
147  o = qobject_cast<OverlayWidget *>(cw);
148  if (o) { return o; }
149  cl++;
150  cw = cw->parentWidget();
151  }
152  return nullptr;
153  }
