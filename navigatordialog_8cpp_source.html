72  void CNavigatorDialog::buildNavigator(int columns)
73  {
74  if (m_firstBuild)
75  {
76  m_firstBuild = false;
77  this->insertOwnActions();
78  }
79 
80  this->onStyleSheetsChanged();
81 
82  // remove old layout
83  CGuiUtility::deleteLayout(ui->fr_NavigatorDialogInner->layout(), false);
84 
85  // new layout
86  QGridLayout *gridLayout = new QGridLayout(ui->fr_NavigatorDialogInner);
87  gridLayout->setObjectName("gl_CNavigatorDialog");
88  gridLayout->setSpacing(0);
89  gridLayout->setMargin(0);
90  gridLayout->setContentsMargins(0, 0, 0, 0);
91  ui->fr_NavigatorDialogInner->setLayout(gridLayout);
92  int r = 0;
93  int c = 0;
94 
95  // remark: the actions will be set from the main UI
96  for (QAction *action : this->actions())
97  {
98  if (!action) { continue; }
99  QToolButton *tb = new QToolButton(ui->fr_NavigatorDialogInner);
100  tb->setDefaultAction(action);
101  tb->setObjectName(this->objectName() % u':' % action->objectName());
102  if (!action->text().isEmpty())
103  {
104  tb->setToolTip(action->text());
105  }
106  gridLayout->addWidget(tb, r, c++);
107  tb->show();
108  if (c < columns) { continue; }
109  c = 0;
110  r++;
111  }
112  m_currentColumns = gridLayout->columnCount();
113  this->adjustNavigatorSize(gridLayout);
114  this->focusWidget();
115  }

303  void CNavigatorDialog::insertOwnActions()
304  {
305  // add some space for frameless navigators where I can move the navigator
306  QAction *a = nullptr; // new QAction(CIcons::empty16(), "move navigator here", this);
307  bool c = false;
308  if (sGui)
309  {
310  a = new QAction(CIcons::swiftLauncher16(), "Start launcher", this);
311  c = connect(a, &QAction::triggered, sGui, &CGuiApplication::startLauncher);
312  Q_ASSERT(c);
313  this->addAction(a);
314  }
315 
316  // save
317  a = new QAction(CIcons::save16(), "Save state", this);
318  c = connect(a, &QAction::triggered, this, &CNavigatorDialog::saveToSettings, Qt::QueuedConnection);
319  Q_ASSERT(c);
320  this->addAction(a);
321 
322  // close
323  const QIcon i(CIcons::changeIconBackgroundColor(this->style()->standardIcon(QStyle::SP_TitleBarCloseButton), Qt::white, QSize(16, 16)));
324  a = new QAction(i, "Close", this);
325  c = connect(a, &QAction::triggered, this, &CNavigatorDialog::close, Qt::QueuedConnection);
326  Q_ASSERT(c);
327  this->addAction(a);

343  void CNavigatorDialog::adjustNavigatorSize(QGridLayout *layout)
344  {
345  QGridLayout *gridLayout = layout ? layout : this->myGridLayout();
346  Q_ASSERT_X(gridLayout, Q_FUNC_INFO, "Missing layout");
347 
348  int w = 16 * gridLayout->columnCount();
349  int h = 16 * gridLayout->rowCount();
350 
351  // margins
352  const QMargins margins = gridLayout->contentsMargins() + this->contentsMargins();
353  h = h + margins.top() + margins.bottom();
354  w = w + margins.left() + margins.right();
355 
356  // adjust
357  const QSize min(w + 2, h + 2);
358  ui->fr_NavigatorDialogInner->setMinimumSize(min);
359  this->setMinimumSize(min);
360  this->adjustSize();
361  }
