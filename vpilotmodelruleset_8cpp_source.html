111  CAircraftModelList CVPilotModelRuleSet::toAircraftModels() const
112  {
113  QStringList modelNames;
114  CAircraftModelList models;
115  for (const CVPilotModelRule &rule : *this)
116  {
117  QString m(rule.getModelName());
118  if (m.isEmpty()) { continue; }
119  if (modelNames.contains(m, Qt::CaseInsensitive))
120  {
121  CAircraftModel model(rule.toAircraftModel());
122  for (CAircraftModel &exisitingModel : models)
123  {
124  if (!exisitingModel.matchesModelString(m, Qt::CaseInsensitive)) { continue; }
125  exisitingModel.updateMissingParts(model);
126  break;
127  }
128  }
129  else
130  {
131  models.push_back(rule.toAircraftModel());
132  modelNames.append(m);
133  }
134  }
135  return models;
136  }
