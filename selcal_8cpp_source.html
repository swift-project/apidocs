62  bool CSelcal::isValidCode(const QString &code)
63  {
64  if (code.length() != 4) return false;
65  int p1, p2, p3, p4;
66  QString codeUpper = code.toUpper();
67  if ((p1 = CSelcal::validCharacters().indexOf(codeUpper.at(0))) < 0) return false;
68  if ((p2 = CSelcal::validCharacters().indexOf(codeUpper.at(1))) < 0) return false;
69  if ((p3 = CSelcal::validCharacters().indexOf(codeUpper.at(2))) < 0) return false;
70  if ((p4 = CSelcal::validCharacters().indexOf(codeUpper.at(3))) < 0) return false;
71  if (p1 >= p2 || p3 >= p4) return false; // pair in alphabetical order
72  if (p1 == p3 || p2 == p3 || p2 == p4 || p3 == p4) return false; // cppcheck-suppress knownConditionTrueFalse // given letter can be used only once in a SELCAL code
73  return true;
74  }
