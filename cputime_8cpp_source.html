18  static int getCpuTimeMs(const FILETIME &kernelTime, const FILETIME &userTime)
19  {
20  const ULARGE_INTEGER kernel { { kernelTime.dwLowDateTime, kernelTime.dwHighDateTime } };
21  const ULARGE_INTEGER user { { userTime.dwLowDateTime, userTime.dwHighDateTime } };
22  const quint64 usecs = (kernel.QuadPart + user.QuadPart) / 10ull;
23  return static_cast<int>(usecs / 1000ull);
24  }
