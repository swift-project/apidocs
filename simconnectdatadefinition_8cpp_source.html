245  HRESULT CSimConnectDefinitions::initSbDataArea(const HANDLE hSimConnect)
246  {
247  HRESULT hr = s_ok();
248  const DWORD sbSize = sizeof(DataDefinitionClientAreaSb);
249 
250  // We need to know the client area 'name' and map it to a client ID
251  hr += SimConnect_MapClientDataNameToID(hSimConnect, "SquawkBox Data", ClientAreaSquawkBox);
252  if (isFailure(hr))
253  {
254  CLogMessage(static_cast<CSimConnectDefinitions *>(nullptr)).error(u"SimConnect error: SimConnect_MapClientDataNameToID %1") << hr;
255  return hr;
256  }
257 
258  // Mapping needs to be first
259  hr += SimConnect_CreateClientData(hSimConnect, ClientAreaSquawkBox, sbSize, SIMCONNECT_CREATE_CLIENT_DATA_FLAG_DEFAULT);
260  if (isFailure(hr))
261  {
262  CLogMessage(static_cast<CSimConnectDefinitions *>(nullptr)).error(u"SimConnect error: SimConnect_CreateClientData %1") << hr;
263  return hr;
264  }
265 
266  // data definitions
267  hr += SimConnect_AddToClientDataDefinition(hSimConnect, CSimConnectDefinitions::DataClientAreaSb, 0, sbSize); // whole area
268  hr += SimConnect_AddToClientDataDefinition(hSimConnect, CSimConnectDefinitions::DataClientAreaSbStandby, 17, 1); // standby
269  hr += SimConnect_AddToClientDataDefinition(hSimConnect, CSimConnectDefinitions::DataClientAreaSbIdent, 19, 1); // ident
270  hr += SimConnect_AddToClientDataDefinition(hSimConnect, CSimConnectDefinitions::DataClientAreaSbConnected, 1, 1); // network connected
271  hr += SimConnect_AddToClientDataDefinition(hSimConnect, CSimConnectDefinitions::DataClientAreaSbRunning, 0, 1); // SB running
272 
273  if (isFailure(hr))
274  {
275  CLogMessage(static_cast<CSimConnectDefinitions *>(nullptr)).error(u"SimConnect error: SB data area data definitions %1") << hr;
276  return hr;
277  }
278 
279  // write a default client area so we are not suddenly squawking ident or so
280  DataDefinitionClientAreaSb sbArea;
281  byte sbRunning = 1;
282  sbArea.setDefaultValues();
283  hr += SimConnect_SetClientData(hSimConnect, ClientAreaSquawkBox, CSimConnectDefinitions::DataClientAreaSb, SIMCONNECT_CLIENT_DATA_SET_FLAG_DEFAULT, 0, sbSize, &sbArea);
284  hr += SimConnect_SetClientData(hSimConnect, ClientAreaSquawkBox, CSimConnectDefinitions::DataClientAreaSbRunning, SIMCONNECT_CLIENT_DATA_SET_FLAG_DEFAULT, 0, 1, &sbRunning);
285  if (isFailure(hr))
286  {
287  CLogMessage(static_cast<CSimConnectDefinitions *>(nullptr)).error(u"SimConnect error: SimConnect_SetClientData %1") << hr;
288  }
289  return hr;
290  }

360  void DataDefinitionRemoteAircraftPartsWithoutLights::initFromParts(const CAircraftParts &parts)
361  {
362  gearHandlePosition = parts.isFixedGearDown() ? 1.0 : 0.0;
363  const double trail = parts.getFlapsPercent() / 100.0;
364  const double lead = trail;
365  flapsTrailingEdgeLeftPercent = trail;
366  flapsTrailingEdgeRightPercent = trail;
367  flapsLeadingEdgeLeftPercent = lead;
368  flapsLeadingEdgeRightPercent = lead;
369  spoilersHandlePosition = parts.isSpoilersOut() ? 1.0 : 0.0;
370  this->setAllEngines(false); // init
371 
372  int e = 1;
373  for (const CAircraftEngine &engine : parts.getEngines())
374  {
375  this->setEngine(e++, engine.isOn());
376  }
