76  template <typename T, bool UseCompare>
77  QVariant CListModelBase<T, UseCompare>::data(const QModelIndex &index, int role) const
78  {
79  // check / init
80  if (!this->isValidIndex(index)) { return QVariant(); }
81 
82  if (role == Qt::BackgroundRole)
83  {
84  return CListModelBaseNonTemplate::data(index, role);
85  }
86 
87  // Formatter
88  const CDefaultFormatter *formatter = m_columns.getFormatter(index);
89  Q_ASSERT_X(formatter, Q_FUNC_INFO, "Missing formatter");
90 
91  // Upfront checking avoids unnecessary data fetching
92  if (!formatter || !formatter->supportsRole(role)) { return CListModelBaseNonTemplate::data(index, role); }
93 
94  // index, updront checking
95  const int row = index.row();
96  const int col = index.column();
97  const CPropertyIndex propertyIndex = this->columnToPropertyIndex(col);
98  const int propertyIndexFront = propertyIndex.frontCasted<int>();
99 
100  // special cases
101  switch (propertyIndexFront)
102  {
103  case CPropertyIndexRef::GlobalIndexLineNumber: return QVariant::fromValue(row + 1);
104  case CPropertyIndexRef::GlobalIndexEmpty: return {};
105  default: break; // continue here
106  }
107 
108  // Formatted data
109  const ObjectType obj = this->containerOrFilteredContainer()[row];
110  return formatter->data(role, obj.propertyByIndex(propertyIndex)).getQVariant();
111  }

113  template <typename T, bool UseCompare>
114  bool CListModelBase<T, UseCompare>::setData(const QModelIndex &index, const QVariant &value, int role)
115  {
116  Qt::ItemDataRole dataRole = static_cast<Qt::ItemDataRole>(role);
117  if (!(dataRole == Qt::UserRole || dataRole == Qt::EditRole)) { return false; }
118 
119  // check / init
120  if (!this->isValidIndex(index)) { return false; }
121  if (!m_columns.isEditable(index)) { return false; }
122  const CDefaultFormatter *formatter = m_columns.getFormatter(index);
123  Q_ASSERT(formatter);
124  if (!formatter) { return false; }
125 
126  ObjectType obj = m_container[index.row()];
127  const ObjectType currentObject(obj);
128  const CPropertyIndex propertyIndex = this->columnToPropertyIndex(index.column());
129  obj.setPropertyByIndex(propertyIndex, value);
130 
131  if (obj != currentObject)
132  {
133  const QModelIndex topLeft = index.sibling(index.row(), 0);
134  const QModelIndex bottomRight = index.sibling(index.row(), this->columnCount() - 1);
135  m_container[index.row()] = obj;
136  const CVariant co = CVariant::fromValue(obj);
137  emit objectChanged(co, propertyIndex);
138  emit this->dataChanged(topLeft, bottomRight);
139  this->updateFilteredContainer();
140  return true;
141  }
142  return false;
143  }

155  template <typename T, bool UseCompare>
156  int CListModelBase<T, UseCompare>::update(const ContainerType &container, bool sort)
157  {
158  if (m_modelDestroyed) { return 0; }
159 
160  // Keep sorting out of begin/end reset model
161  ContainerType sortedContainer;
162  ContainerType selection;
163  if (m_selectionModel)
164  {
165  selection = m_selectionModel->selectedObjects();
166  }
167  const int oldSize = m_container.size();
168  const bool performSort = sort && container.size() > 1 && this->hasValidSortColumn();
169  if (performSort)
170  {
171  const int sortColumn = this->getSortColumn();
172  sortedContainer = this->sortContainerByColumn(container, sortColumn, m_sortOrder);
173  }
174 
175  this->beginResetModel();
176  m_container = performSort ? sortedContainer : container;
177  this->updateFilteredContainer(); // use sorted container for filtered if applicable
178  this->endResetModel();
179 
180  // reselect if implemented in specialized view
181  if (!selection.isEmpty())
182  {
183  m_selectionModel->selectObjects(selection);
184  }
185 
186  const int newSize = m_container.size();
187  Q_UNUSED(oldSize)
188 
189  // I have to update even with same size because I cannot tell what/if data are changed
190  this->emitModelDataChanged();
191  return newSize;
192  }

212  template <typename T, bool UseCompare>
213  CWorker *CListModelBase<T, UseCompare>::updateAsync(const ContainerType &container, bool sort)
214  {
215  Q_UNUSED(sort)
216  if (m_modelDestroyed) { return nullptr; }
217  const auto sortColumn = this->getSortColumn();
218  const auto sortOrder = this->getSortOrder();
219  CWorker *worker = CWorker::fromTask(this, "ModelSort", [this, container, sortColumn, sortOrder]() {
220  return this->sortContainerByColumn(container, sortColumn, sortOrder);
221  });
222  worker->thenWithResult<ContainerType>(this, [this](const ContainerType &sortedContainer) {
223  if (m_modelDestroyed) { return; }
224  this->update(sortedContainer, false);
225  });
226  worker->then(this, &CListModelBase::asyncUpdateFinished);
227  return worker;

534  template <typename T, bool UseCompare>
535  typename CListModelBase<T, UseCompare>::ContainerType CListModelBase<T, UseCompare>::sortContainerByColumn(const ContainerType &container, int column, Qt::SortOrder order) const
536  {
537  if (m_modelDestroyed) { return container; }
538  if (container.size() < 2 || !m_columns.isSortable(column))
539  {
540  return container; // nothing to do
541  }
542 
543  // this is the only part not really thread safe, but columns do not change so far
544  const CPropertyIndex propertyIndex = m_columns.columnToSortPropertyIndex(column);
545  Q_ASSERT(!propertyIndex.isEmpty());
546  if (propertyIndex.isEmpty())
547  {
548  return container; // at release build do nothing
549  }
550 
551  // sort the values
552  const auto tieBreakersCopy = m_sortTieBreakers;
553  const std::integral_constant<bool, UseCompare> marker {};
554  const auto p = [=](const ObjectType &a, const ObjectType &b) -> bool {
555  return Private::compareForModelSort<ObjectType>(a, b, order, propertyIndex, tieBreakersCopy, marker);
556  };
557 
558  return container.sorted(p);
559  }

561  template <typename T, bool UseCompare>
562  QMimeData *CListModelBase<T, UseCompare>::mimeData(const QModelIndexList &indexes) const
563  {
564  QMimeData *mimeData = new QMimeData();
565  if (indexes.isEmpty()) { return mimeData; }
566 
567  ContainerType container;
568  QList<int> rows; // avoid redundant objects
569 
570  // Indexes are per row and column
571  for (const QModelIndex &index : indexes)
572  {
573  if (!index.isValid()) { continue; }
574  const int r = index.row();
575  if (rows.contains(r)) { continue; }
576  container.push_back(this->at(index));
577  rows.append(r);
578  }
579 
580  // to JSON via CVariant
581  const QJsonDocument containerJson(CVariant::fromValue(container).toJson());
582  const QString jsonString(containerJson.toJson(QJsonDocument::Compact));
583 
584  mimeData->setData(CGuiUtility::swiftJsonDragAndDropMimeType(), jsonString.toUtf8());
585  return mimeData;
