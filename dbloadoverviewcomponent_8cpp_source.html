221  void CDbLoadOverviewComponent::setSharedUrlValues()
222  {
223  if (m_sharedValueCheckInProgress) { return; }
224  if ((QDateTime::currentMSecsSinceEpoch() - m_sharedLastCheck) < 5000) { return; } // do not check too often
225 
226  m_sharedValueCheckInProgress = true; // avoid processEvent (canConnect) calling this again before done
227 
228  const CUrlList sharedUrls(sGui->getGlobalSetup().getSwiftSharedUrls());
229  const QString valueHtml("<img src=\"%1\">&nbsp;%2");
230  const QString urlLinkHtml("<a href=\"%1\">%2</a>");
231 
232  QStringList values;
233  for (const CUrl &sharedUrl : sharedUrls)
234  {
235  if (!sGui || sGui->isShuttingDown())
236  {
237  // shutdown during connect test
238  m_sharedValueCheckInProgress = false;
239  return;
240  }
241  const bool canConnect = CNetworkUtils::canConnect(sharedUrl);
242  values.push_back(
243  valueHtml.arg(canConnect ? m_imgOk : m_imgFailed, urlLinkHtml.arg(sharedUrl.getFullUrl(), sharedUrl.getHost())));
244  }
245 
246  const QString sharedUrlTable = toHtmTable(values, 2);
247  ui->lbl_SharedUrls->setText(sharedUrlTable);
248  ui->lbl_SharedUrls->setMinimumHeight(10 + (18 * sharedUrls.size()));
249  const QString currentlyUsedSharedUrl = sGui->getWorkingSharedUrl().toQString();
250  ui->lbl_SharedUrls->setToolTip(
251  currentlyUsedSharedUrl.isEmpty() ? "No shared URL" : "currently used: " + currentlyUsedSharedUrl);
252 
253  m_sharedLastCheck = QDateTime::currentMSecsSinceEpoch();
254  m_sharedValueCheckInProgress = false;
255  }

368  void CDbLoadOverviewComponent::loadInfoObjects()
369  {
370  if (m_loadInProgress) { return; }
371  if (!sGui || !sGui->hasWebDataServices()) { return; }
372  if (sGui->isShuttingDown()) { return; }
373  bool direct = false;
374 
375  if (sGui->getWebDataServices()->getDbInfoDataReader() && sGui->getWebDataServices()->getDbInfoDataReader()->getInfoObjectCount() > 0)
376  {
377  direct = true;
378  }
379  else if (sGui->isSwiftDbAccessible())
380  {
381  // do not trigger if cannot be connected
382  sGui->getWebDataServices()->triggerReadOfDbInfoObjects();
383  }
384 
385  if (sGui->getWebDataServices()->getSharedInfoDataReader() && sGui->getWebDataServices()->getSharedInfoDataReader()->getInfoObjectCount() > 0)
386  {
387  direct = true;
388  }
389  else
390  {
391  sGui->getWebDataServices()->triggerReadOfSharedInfoObjects();
392  }
393 
394  if (direct)
395  {
396  m_dsTriggerGuiUpdate.inputSignal();
397  }
