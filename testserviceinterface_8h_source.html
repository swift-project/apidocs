386  template <class ValueObject>
387  static bool pingCompare(const ValueObject &in, const ValueObject &out, QTextStream &ts, bool verbose, int &errors)
388  {
389  const bool equal = (in == out && extraCompare(in, out));
390  if (!equal)
391  {
392  errors++;
393  if (verbose) { ts << "I: " << in.toQString() << Qt::endl
394  << "O: " << out.toQString() << Qt::endl; }
395  }
396  return equal;
397  }

407  static bool extraCompare(const BlackMisc::Aviation::CFlightPlan &in, const BlackMisc::Aviation::CFlightPlan &out)
408  {
409  // flight plan: check units are preserved
410  if (in.getEnrouteTime().getUnit() != out.getEnrouteTime().getUnit()) { return false; }
411  if (in.getFuelTime().getUnit() != out.getFuelTime().getUnit()) { return false; }
412  if (in.getCruiseAltitude().getUnit() != out.getCruiseAltitude().getUnit()) { return false; }
413  if (in.getCruiseTrueAirspeed().getUnit() != out.getCruiseTrueAirspeed().getUnit()) { return false; }
414  return true;
415  }
