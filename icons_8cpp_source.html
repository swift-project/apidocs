349  QPixmap transparentPixmapImpl(QSize size)
350  {
351  // http://curtis.humphreyonline.us/code/qt-code/making-a-qpixmap-transparent-wrong-way-right-way
352  QPixmap pix(size);
353  pix.fill(Qt::transparent);
354 
355  QPainter painter;
356  painter.begin(&pix);
357  painter.drawPixmap(pix.rect(), pix);
358  painter.end();
359  return pix;
360  }
