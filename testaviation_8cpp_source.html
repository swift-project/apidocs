256  QVERIFY(prefix == "DLH");

260  QVERIFY(prefix == "DLH");
261  QVERIFY(flightnumber == "12");

267 
268  const CCallsign pilotX2("DLH1WP");
269  prefix = pilotX2.getAirlinePrefix(flightnumber, identifier);
270  QVERIFY(prefix == "DLH"); // no prefix
271  QVERIFY(flightnumber == "1");
272  QVERIFY(identifier == "1WP");
273 
274  CCallsignSet set;
275  set.push_back(cs1);
