55  CAircraftModelList CCategoryMatcher::reduceByCategories(const CAircraftModelList &alreadyMatchedModels, const CAircraftModelList &modelSet, const CAircraftMatcherSetup &setup, const CSimulatedAircraft &remoteAircraft, bool &reduced, bool shortLog, CStatusMessageList *log) const
56  {
57  Q_UNUSED(shortLog)
58 
59  reduced = false;
60  if (!setup.useCategoryMatching())
61  {
62  if (log) { CMatchingUtils::addLogDetailsToList(log, remoteAircraft, QStringLiteral("Disabled category matching"), getLogCategories()); }
63  return alreadyMatchedModels;
64  }
65  if (m_all.isEmpty())
66  {
67  // no categories?
68  CMatchingUtils::addLogDetailsToList(log, remoteAircraft, QStringLiteral("Disabled category matching"), getLogCategories());
69  return alreadyMatchedModels;
70  }
71  if (!remoteAircraft.getAircraftIcaoCode().hasCategory())
72  {
73  CMatchingUtils::addLogDetailsToList(log, remoteAircraft, QStringLiteral("No category in remote aircraft"), getLogCategories());
74  return alreadyMatchedModels;
75  }
76 
77  if (log) { CMatchingUtils::addLogDetailsToList(log, remoteAircraft, QStringLiteral("Remote aircraft has category '%1'").arg(remoteAircraft.getAircraftIcaoCode().getCategory().getNameDbKey()), getLogCategories()); }
78  if (!m_gliders.isEmpty() && setup.getMatchingMode().testFlag(CAircraftMatcherSetup::ByCategoryGlider) && this->isGlider(remoteAircraft.getAircraftIcaoCode()))
79  {
80  // we have a glider
81  // and we search in the whole set: this is a special case
82  const int firstLevel = this->gliderFirstLevel();
83  const CAircraftModelList gliders = modelSet.findByCategoryFirstLevel(firstLevel); // all gliders from model set
84  if (!gliders.isEmpty())
85  {
86  const CAircraftCategory category = remoteAircraft.getAircraftIcaoCode().getCategory();
87  reduced = true; // in any case reduced (to gliders)
88 
89  // find same category
90  const CAircraftModelList sameGliders = gliders.findByCategory(category);
91  if (!sameGliders.isEmpty())
92  {
93  if (log) { CMatchingUtils::addLogDetailsToList(log, remoteAircraft, QStringLiteral("Reduced to %1 models by category: '%2'").arg(sameGliders.size()).arg(category.toQString(true)), getLogCategories()); }
94  return sameGliders;
95  }
96 
97  // find parallel branch category
98  const CAircraftCategoryList otherBranches = m_gliders.findInParallelBranch(category);
99  if (!otherBranches.isEmpty())
100  {
101  const CAircraftModelList otherBranchGliders = gliders.findByCategories(otherBranches);
102  if (!otherBranchGliders.isEmpty())
103  {
104  if (log) { CMatchingUtils::addLogDetailsToList(log, remoteAircraft, QStringLiteral("Reduced to %1 parallel branch models of '%2' by categories: '%3'").arg(otherBranchGliders.size()).arg(category.getLevelAndName(), otherBranches.getLevelsString()), getLogCategories()); }
105  return otherBranchGliders;
106  }
107  }
108 
109  const CAircraftCategoryList siblings = m_gliders.findSiblings(category);
110  if (!siblings.isEmpty())
111  {
112  const CAircraftModelList siblingGliders = modelSet.findByCategories(siblings);
113  if (!siblings.isEmpty() && !siblingGliders.isEmpty())
114  {
115  if (log) { CMatchingUtils::addLogDetailsToList(log, remoteAircraft, QStringLiteral("Reduced to %1 sibling models of '%2' by categories: '%3'").arg(siblingGliders.size()).arg(category.getLevelAndName(), siblings.getLevelsString()), getLogCategories()); }
116  return siblingGliders;
117  }
118  }
119 
120  CMatchingUtils::addLogDetailsToList(log, remoteAircraft, QStringLiteral("Reduced to %1 models by 'GLIDER' category").arg(sameGliders.size()), getLogCategories());
121  return gliders;
122  }
123  else
124  {
125  if (log) { CMatchingUtils::addLogDetailsToList(log, remoteAircraft, QStringLiteral("No glider category '%1' in set").arg(m_gliders.front().getLevelAndName()), getLogCategories()); }
126  static const QStringList substituteIcaos({ "UHEL", "GLID", "ULAC" }); // maybe also GYRO
127  static const QString substituteIcaosStr = substituteIcaos.join(", ");
128 
129  CAircraftModelList substitutes = alreadyMatchedModels.findByDesignatorsOrFamilyWithColorLivery(substituteIcaos);
130  if (substitutes.isEmpty())
131  {
132  substitutes = alreadyMatchedModels.findByCombinedType(QStringLiteral("L1P"));
133  if (!substitutes.isEmpty())
134  {
135  reduced = true;
136  if (log) { CMatchingUtils::addLogDetailsToList(log, remoteAircraft, QStringLiteral("No gliders, reduced to 'L1P' models: %1' (avoid absurd matchings)").arg(substitutes.size()), getLogCategories()); }
137  return substitutes;
138  }
139  }
140  else
141  {
142  reduced = true;
143  if (log) { CMatchingUtils::addLogDetailsToList(log, remoteAircraft, QStringLiteral("Reduced to %1 models by '%2'").arg(substitutes.size()).arg(substituteIcaosStr), getLogCategories()); }
144  return substitutes;
145  }
146  }
147  }
148 
149  return alreadyMatchedModels;
150  }

152  bool CCategoryMatcher::isGlider(const CAircraftIcaoCode &icao) const
153  {
154  if (icao.getDesignator() == CAircraftIcaoCode::getGliderDesignator()) { return true; }
155  const int glider1st = this->gliderFirstLevel();
156  if (glider1st >= 0 && icao.hasCategory())
157  {
158  return icao.getCategory().getFirstLevel() == glider1st;
159  }
160  return false;
161  }
