40  void CCockpitTransponderModeLedsComponent::onLedClicked()
41  {
42  QWidget *w = qobject_cast<QWidget *>(QObject::sender());
43  if (!w) { return; }
44  if (!sGui || sGui->isShuttingDown() || !sGui->getIContextOwnAircraft()) { return; }
45  CTransponder::TransponderMode mode;
46  if (m_ledStandby.data() == w)
47  {
48  mode = CTransponder::StateStandby;
49  }
50  else if (m_ledIdent.data() == w)
51  {
52  mode = CTransponder::StateIdent;
53  }
54  else if (m_ledModes.data() == w)
55  {
56  mode = CTransponder::ModeC;
57  }
58  else
59  {
60  return;
61  }
62  CSimulatedAircraft ownAircraft(this->getOwnAircraft());
63  if (ownAircraft.getTransponderMode() == mode) { return; }
64 
65  this->setMode(mode, true);
66  CTransponder xpdr = ownAircraft.getTransponder();
67  xpdr.setTransponderMode(mode);
68 
69  if (sGui)
70  {
71  sGui->getIContextOwnAircraft()->updateCockpit(ownAircraft.getCom1System(), ownAircraft.getCom2System(), xpdr, identifier());
72  }
73  }

94  void CCockpitTransponderModeLedsComponent::setMode(CTransponder::TransponderMode mode, bool force)
95  {
96  if (!force && m_mode == mode) { return; }
97  m_ledStandby->setOn(false);
98  m_ledModes->setOn(false);
99  m_ledIdent->setOn(false);
100 
101  switch (mode)
102  {
103  case CTransponder::ModeA:
104  case CTransponder::ModeC:
105  case CTransponder::ModeS:
106  case CTransponder::ModeMil1:
107  case CTransponder::ModeMil2:
108  case CTransponder::ModeMil3:
109  case CTransponder::ModeMil4:
110  case CTransponder::ModeMil5:
111  m_ledModes->setOn(true);
112  break;
113  case CTransponder::StateIdent:
114  m_ledModes->setOn(true);
115  m_ledIdent->setOn(true);
116  break;
117  default:
118  case CTransponder::StateStandby:
119  m_ledStandby->setOn(true);
120  break;
121  }
122  }
