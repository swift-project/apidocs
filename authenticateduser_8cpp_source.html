75  CStatusMessageList CAuthenticatedUser::validate() const

78  CStatusMessageList msgs;

82  if (!this->hasValidCredentials()) { msgs.push_back(CStatusMessage(cats, CStatusMessage::SeverityWarning, u"Invalid credentials")); }
