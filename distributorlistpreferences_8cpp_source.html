39  {
40  Q_ASSERT_X(simulator.isSingleSimulator(), Q_FUNC_INFO, "Need single simulator");
41  CDistributorList d(distributors);
42  d.sortAscendingByOrder(); // make sure we are sorted by order
43  m_lastUpdatedSimulator = simulator;
44 
45  switch (simulator.getSimulator())
46  {
47  case CSimulatorInfo::FSX: m_distributorsFsx = d; break;
48  case CSimulatorInfo::P3D: m_distributorsP3d = d; break;
49  case CSimulatorInfo::FS9: m_distributorsFs9 = d; break;
50  case CSimulatorInfo::FG: m_distributorsFG = d; break;
51  case CSimulatorInfo::XPLANE: m_distributorsXPlane = d; break;
