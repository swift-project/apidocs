30  {

32  QStringList ambiguousTitles;
33  QString last;
34  for (const QString &title : titles)
35  {
36  if (title.isEmpty()) { continue; }

38  {
39  if (!ambiguousTitles.contains(title, Qt::CaseInsensitive))
40  {
41  ambiguousTitles.append(title);
42  }
43  }
44  last = title;

81  CAircraftModelList CAircraftCfgEntriesList::toAircraftModelList(bool ignoreDuplicatesAndEmptyModelStrings, CStatusMessageList &msgs) const
82  {
83  CAircraftModelList ml;
84  QSet<QString> keys;
85  for (const CAircraftCfgEntries &entries : (*this))
86  {
87  if (ignoreDuplicatesAndEmptyModelStrings)
88  {
89  const QString key = entries.getTitle().toUpper();
90  if (key.isEmpty()) { continue; }
91  if (keys.contains(key))
92  {
93  CStatusMessage msg(this);
94  msg.warning(u"Duplicate model string %1 in %2 %3")
95  << entries.getTitle() << entries.getFileDirectory() << entries.getFileName();
96  msgs.push_back(msg);
97  continue;
98  }
99  keys.insert(key);
100  }
101  ml.push_back(entries.toAircraftModel());
102  }
103  return ml;
104  }
