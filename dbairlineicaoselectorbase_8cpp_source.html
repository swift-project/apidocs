104  void CDbAirlineIcaoSelectorBase::dropEvent(QDropEvent *event)
105  {
106  if (!event || !acceptDrop(event->mimeData())) { return; }
107  const CVariant valueVariant(toCVariant(event->mimeData()));
108  if (valueVariant.isValid())
109  {
110  if (valueVariant.canConvert<CAirlineIcaoCode>())
111  {
112  const CAirlineIcaoCode icao(valueVariant.value<CAirlineIcaoCode>());
113  if (!icao.hasValidDbKey()) { return; }
114  this->setAirlineIcao(icao);
115  }
116  else if (valueVariant.canConvert<CAirlineIcaoCodeList>())
117  {
118  const CAirlineIcaoCodeList icaos(valueVariant.value<CAirlineIcaoCodeList>());
119  if (icaos.isEmpty()) { return; }
120  this->setAirlineIcao(icaos.front());
121  }
122  }
123  }

125  void CDbAirlineIcaoSelectorBase::onCodesRead(CEntityFlags::Entity entity, CEntityFlags::ReadState readState, int count, const QUrl &url)
126  {
127  Q_UNUSED(url)
128 
129  if (!sGui) { return; }
130  if (entity.testFlag(CEntityFlags::AirlineIcaoEntity) && CEntityFlags::isFinishedReadState(readState))
131  {
132  if (count > 0)
133  {
134  QCompleter *c = this->createCompleter();
135  Q_ASSERT_X(c, Q_FUNC_INFO, "missing converter");
136  connect(c, qOverload<const QString &>(&QCompleter::activated), this, &CDbAirlineIcaoSelectorBase::onCompleterActivated);
137  m_completer.reset(c); // deletes any old completer
138  }
139  else
140  {
141  m_completer.reset(nullptr);
142  }
143  }
144  }
