292  void CTestContainers::timestampList()
293  {
294  CAircraftSituationList situations;
295  const qint64 ts = QDateTime::currentMSecsSinceEpoch();
296  int no = 10;
297  for (int i = 0; i < no; ++i)
298  {
299  CAircraftSituation s;
300  s.setCallsign("CS" + QString::number(i));
301  s.setMSecsSinceEpoch(ts - 10 * i);
302  situations.push_back(s);
303  }
304 
305  // test sorting
306  situations.sortOldestFirst();
307  qint64 ms = situations.front().getMSecsSinceEpoch();
308  QVERIFY2(ms == ts - 10 * (no - 1), "Oldest value not first");
309 
310  situations.sortLatestFirst();
311  ms = situations.front().getMSecsSinceEpoch();
312  QVERIFY2(ms == ts, "Latest value not first");
313 
314  // test shifting
315  situations.clear();
316  const int maxElements = 8;
317  QVERIFY(situations.isEmpty());
318  for (int i = 0; i < no; ++i)
319  {
320  qint64 cTs = ts - 10 * i;
321  CAircraftSituation s;
322  s.setCallsign("CS" + QString::number(i));
323  s.setMSecsSinceEpoch(cTs);
324  situations.push_frontMaxElements(s, maxElements);
325  if (i > maxElements - 1)
326  {
327  QVERIFY2(situations.size() == maxElements, "Situations must only contain max.elements");
328  }
329  else
330  {
331  QVERIFY2(situations.size() == i + 1, "Element size does not match");
332  QVERIFY2(situations.front().getMSecsSinceEpoch() == cTs, "Wrong front element");
333  }
334  }
335 
336  situations.sortLatestFirst();
337  QVERIFY2(situations.isSortedLatestFirst(), "Espect sorted latest first");
338  no = situations.size();
339  for (int i = 0; i < no; ++i)
340  {
341  const CAircraftSituation current = situations[i];
342  const qint64 cTs = current.getMSecsSinceEpoch();
343  const CAircraftSituation expectedBefore = situations.findObjectBeforeOrDefault(cTs);
344  const CAircraftSituation expectedAfter = situations.findObjectAfterOrDefault(cTs);
345  const qint64 beforeTs = expectedBefore.getMSecsSinceEpoch();
346  const qint64 afterTs = expectedAfter.getMSecsSinceEpoch();
347 
348  if (i > 0)
349  {
350  const qint64 t1 = situations[i - 1].getMSecsSinceEpoch();
351  QVERIFY2(t1 == afterTs, "Wrong expected after");
352  }
353  if (i < (no - 1))
354  {
355  const qint64 t1 = situations[i + 1].getMSecsSinceEpoch();
356  QVERIFY2(t1 == beforeTs, "Wrong expected before");
357  }
358  }
