146  void CAfvClient::connectTo(const QString &cid, const QString &password, const QString &callsign, const QString &client)
147  {
148  if (QThread::currentThread() != thread())
149  {
150  // Method needs to be executed in the object thread since it will create new QObject children
151  QPointer<CAfvClient> myself(this);
152  QMetaObject::invokeMethod(this, [=]() { if (myself) { connectTo(cid, password, callsign, client); } });
153  return;
154  }
155 
156  // called in CAfvClient thread
157  this->connectWithContexts();
158  this->setCallsign(callsign);
159 
160  QPointer<CAfvClient> myself(this);
161  if (!this->isConnected() && m_retryConnectAttempt == 0)
162  {
163  // check if connect simply did NOT receive an answer
164  QTimer::singleShot(20 * 1000, this, [=] {
165  if (!myself) { return; }
166  if (m_retryConnectAttempt > 0) { return; } // already handled
167 
168  // this will reconnect ONLY if not already connected
169  this->retryConnectTo(cid, password, callsign, client, QStringLiteral("No connection afer 20secs"));
170  });
171  }
172 
173  // thread safe connect
174  {
175  QMutexLocker lock(&m_mutexConnection);
176 
177  // async connection
178  m_connection->connectTo(cid, password, callsign, client,
179  { // this is the callback when the connection has been established
180  this, [=](bool authenticated) {
181  if (!myself) { return; } // cppcheck-suppress knownConditionTrueFalse
182 
183  // HF stations aliased
184  const QVector<StationDto> aliasedStations = m_connection->getAllAliasedStations();
185  this->setAliasedStations(aliasedStations); // threadsafe
186  this->onTimerUpdate();
187 
188  // const bool isConnected = this->isConnected(); // threadsafe
189  if (authenticated)
190  {
191  // restart timer, normally it should be started already, paranoia
192  // as I run in "my thread" starting timer should be OK
193  {
194  QMutexLocker lock2(&m_mutex);
195  if (m_voiceServerTimer) { m_voiceServerTimer->start(PositionUpdatesMs); }
196  }
197  m_retryConnectAttempt = 0;
198  emit this->connectionStatusChanged(Connected);
199  }
200  else
201  {
202  myself->retryConnectTo(cid, password, callsign, client, QStringLiteral("AFV authentication failed for '%1' callsign '%2'").arg(cid, callsign));
203  emit this->connectionStatusChanged(Disconnected);
204  }
205  } });
206  }
207  }

274  void CAfvClient::startAudio(const CAudioDeviceInfo &inputDevice, const CAudioDeviceInfo &outputDevice)
275  {
276  if (QThread::currentThread() != this->thread())
277  {
278  // Method needs to be executed in the object thread since it will create new QObject children
279  QPointer<CAfvClient> myself(this);
280  QMetaObject::invokeMethod(this, [=]() { if (myself) { startAudio(inputDevice, outputDevice); } });
281  return;
282  }
283 
284  const CAudioDeviceInfo useInputDevice = inputDevice.isValid() ? inputDevice : CAudioDeviceInfo::getDefaultInputDevice();
285  const CAudioDeviceInfo useOutputDevice = outputDevice.isValid() ? outputDevice : CAudioDeviceInfo::getDefaultOutputDevice();
286 
287  BLACK_VERIFY_X(useInputDevice.isValid() && useInputDevice.isInputDevice(), Q_FUNC_INFO, "Wrong input device");
288  BLACK_VERIFY_X(useOutputDevice.isValid() && useOutputDevice.isOutputDevice(), Q_FUNC_INFO, "Wrong output device");
289 
290  if (m_isStarted)
291  {
292  if (this->usesSameDevices(useInputDevice, useOutputDevice))
293  {
294  CLogMessage(this).info(u"Client already started for '%1'/'%2'") << useInputDevice.getName() << useOutputDevice.getName();
295  return;
296  }
297  this->stopAudio();
298  }
299 
300  this->initTransceivers();
301 
302  // threadsafe block
303  {
304  // lock block 1
305  {
306  QMutexLocker lock { &m_mutexSampleProviders };
307  if (m_soundcardSampleProvider)
308  {
309  m_soundcardSampleProvider->disconnect();
310  m_soundcardSampleProvider->deleteLater();
311  }
312  m_soundcardSampleProvider = new CSoundcardSampleProvider(SampleRate, allTransceiverIds(), this);
313  connect(m_soundcardSampleProvider, &CSoundcardSampleProvider::receivingCallsignsChanged, this, &CAfvClient::onReceivingCallsignsChanged);
314 
315  if (m_outputSampleProvider) { m_outputSampleProvider->deleteLater(); }
316  m_outputSampleProvider = new CVolumeSampleProvider(m_soundcardSampleProvider, this);
317  // m_outputSampleProvider->setGainRatio(outputVolume); // 2021-09 LT Disabled. Output volume is controlled independently for COM1/2
318  }
319 
320  // lock block 2
321  {
322  QMutexLocker lock(&m_mutex);
323 
324  m_output->start(useOutputDevice, m_outputSampleProvider);
325  m_input->start(useInputDevice);
326 
327  // runs in correct thread
328  m_voiceServerTimer->start(PositionUpdatesMs); // start for preset values
329  }
330  }
331 
332  this->setReceiveAudio(true); // threadsafe
333 
334  this->onSettingsChanged(); // make sure all settings are applied
335  m_isStarted = true;
336  CLogMessage(this).info(u"Started [Input: %1] [Output: %2]") << useInputDevice.getName() << useOutputDevice.getName();
337 
338  this->onTimerUpdate(); // update values
339 
340  emit this->startedAudio(useInputDevice, useOutputDevice);
341 
342  if (this->isMuted())
343  {
344  // un-mute after startup
345  this->setMuted(false);
346  }
347  }

474  void CAfvClient::updateComFrequency(quint16 id, quint32 frequencyHz)
475  {
476  if (id != 0 && id != 1) { return; }
477 
478  // Fix rounding issues like 128074999 Hz -> 128075000 Hz
479  quint32 roundedFrequencyHz = static_cast<quint32>(qRound(frequencyHz / 1000.0)) * 1000;
480  roundedFrequencyHz = this->getAliasFrequencyHz(roundedFrequencyHz);
481 
482  bool update = false;
483  {
484  QMutexLocker lockTransceivers(&m_mutexTransceivers);
485  if (m_transceivers.size() >= id + 1)
486  {
487  if (m_transceivers[id].frequencyHz != roundedFrequencyHz)
488  {
489  update = true;
490  m_transceivers[id].frequencyHz = roundedFrequencyHz;
491  }
492  }
493  }
494 
495  // outside lock to avoid deadlock
496  if (update)
497  {
498  this->updateTransceivers(false); // no frequency update
499  }
500  }

526  {
527  // also update if NOT connected, values will be preset
528 
529  if (hasContexts())
530  {
531  const CSimulatedAircraft ownAircraft = sApp->getIContextOwnAircraft()->getOwnAircraft();
532  this->updatePosition(ownAircraft.latitude().value(CAngleUnit::deg()),
533  ownAircraft.longitude().value(CAngleUnit::deg()),
534  ownAircraft.getAltitude().value(CLengthUnit::ft()));
535 
536  if (updateFrequencies)
537  {
538  this->updateComFrequency(CComSystem::Com1, ownAircraft.getCom1System());
539  this->updateComFrequency(CComSystem::Com2, ownAircraft.getCom2System());
540  }
541  }

544  const auto transceivers = this->getTransceivers();
545  const auto enabledTransceivers = this->getEnabledTransceivers();
546  const QString callsign = this->getCallsign(); // threadsafe
547 
548  // transceivers
549  QVector<TransceiverDto> newEnabledTransceivers;
550  for (const TransceiverDto &transceiver : transceivers)

556  }
557

561  if (m_connection) { m_connection->updateTransceivers(callsign, newEnabledTransceivers); }

565  if (m_soundcardSampleProvider) { m_soundcardSampleProvider->updateRadioTransceivers(newEnabledTransceivers); }
