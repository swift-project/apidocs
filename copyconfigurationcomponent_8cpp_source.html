83  int CCopyConfigurationComponent::copySelectedFiles()
84  {
85  const QStringList files = this->getSelectedFiles();
86  if (files.isEmpty()) { return 0; }
87 
88  const QString destinationDir = this->getThisVersionDirectory();
89  const QString sourceDir = this->getOtherVersionsSelectedDirectory();
90  if (destinationDir.isEmpty()) { return 0; }
91  const QDir source(sourceDir);
92  const QDir destination(destinationDir);
93  if (!destination.exists()) { return 0; }
94 
95  // init model caches if applicable (.rev file entries)
96  this->initCaches(files);
97 
98  int c = 0;
99  QStringList copied;
100  QStringList skipped;
101  for (const QString &file : files)
102  {
103  const QString relativePath = source.relativeFilePath(file);
104  const QString target = CFileUtils::appendFilePaths(destinationDir, relativePath);
105  if (relativePath.contains('/'))
106  {
107  const QString targetDir = CFileUtils::stripFileFromPath(target);
108  const bool dirOk = destination.mkpath(targetDir);
109  if (!dirOk) { continue; }
110  }
111  QFile::remove(target); // copy does not overwrite
112  const bool s = QFile::copy(file, target);
113  if (s)
114  {
115  c++;
116  copied << target;
117  }
118  else
119  {
120  skipped << target;
121  }
122  }
123 
124  if (m_logCopiedFiles)
125  {
126  if (!copied.isEmpty())
127  {
128  CLogMessage(this).info(u"Copied %1 files, list: '%2'") << copied.size() << copied.join(", ");
129  }
130  if (!skipped.isEmpty())
131  {
132  CLogMessage(this).info(u"Skipped %1 files, list: '%2'") << skipped.size() << skipped.join(", ");
133  }
134  }
135 
136  // bye
137  return c;
138  }

195  void CCopyConfigurationComponent::initCurrentDirectories(bool preselectMissingOrOutdated)
196  {
197  const QString destinationDir = this->getThisVersionDirectory(); // cache or settings dir
198 
199  QFileSystemModel *destinationModel = qobject_cast<QFileSystemModel *>(ui->tv_Destination->model());
200  QFileSystemModel *sourceModel = qobject_cast<QFileSystemModel *>(ui->tv_Source->model());
201 
202  if (!destinationModel || m_initializedDestinationDir != destinationDir)
203  {
204  m_initializedDestinationDir = destinationDir;
205  const QDir thisVersionDirectory(destinationDir);
206  if (!thisVersionDirectory.exists())
207  {
208  const bool hasDir = thisVersionDirectory.mkpath(destinationDir);
209  if (!hasDir)
210  {
211  ui->le_CurrentVersion->setText("No swift target dir");
212  return;
213  }
214  }
215  ui->le_CurrentVersion->setText(destinationDir);
216 
217  // destination
218  if (!destinationModel)
219  {
220  destinationModel = new QFileSystemModel(this);
221  destinationModel->setFilter(QDir::NoDotAndDotDot | QDir::Files | QDir::Dirs);
222  ui->tv_Destination->setModel(destinationModel);
223  ui->tv_Destination->setSortingEnabled(true);
224 
225  // disconnect when done, there have been problems that the lambda was called when the view was already destroyed
226  connectOnce(destinationModel, &QFileSystemModel::directoryLoaded, this, [=](const QString &path) {
227  Q_UNUSED(path);
228  ui->tv_Destination->resizeColumnToContents(0);
229  ui->tv_Destination->expandAll();
230  });
231  }
232  const QModelIndex destinationIndex = destinationModel->setRootPath(destinationDir);
233  ui->tv_Destination->setRootIndex(destinationIndex);
234  }
235  destinationModel->setNameFilters(this->getSourceFileFilter());
236  destinationModel->setNameFilterDisables(m_nameFilterDisables);
237 
238  // source
239  const QString sourceDir = this->getOtherVersionsSelectedDirectory();
240  if (!m_hasOtherSwiftVersions)
241  {
242  // no ther versions
243  return;
244  }
245  else if (!sourceModel || m_initializedSourceDir != sourceDir)
246  {
247  m_initializedSourceDir = sourceDir;
248  if (!sourceModel)
249  {
250  sourceModel = new QFileSystemModel(this);
251  sourceModel->setFilter(QDir::NoDotAndDotDot | QDir::Files | QDir::Dirs);
252  ui->tv_Source->setModel(sourceModel);
253  ui->tv_Source->setSortingEnabled(true); // hide/disable only
254  connectOnce(sourceModel, &QFileSystemModel::directoryLoaded, this, [=](const QString &path) {
255  Q_UNUSED(path);
256  ui->tv_Source->resizeColumnToContents(0);
257  ui->tv_Source->expandAll();
258  if (preselectMissingOrOutdated)
259  {
260  this->preselectMissingOrOutdated();
261  }
262  });
263  }
264  const QModelIndex sourceIndex = sourceModel->setRootPath(sourceDir);
265  ui->tv_Source->setRootIndex(sourceIndex);
266  }
267  sourceModel->setNameFilters(this->getSourceFileFilter());
268  sourceModel->setNameFilterDisables(m_nameFilterDisables);
269  }

311  QString CCopyConfigurationComponent::getOtherVersionsSelectedDirectory() const
312  {
313  if (ui->cb_OtherVersions->count() < 1) { return {}; }
314  const QFileInfoList dirs(CSwiftDirectories::applicationDataDirectories());
315  if (dirs.isEmpty()) { return {}; }
316  const QString otherVersionDir = m_otherVersionDirs.at(ui->cb_OtherVersions->currentIndex());
317  QString dir;
318  for (const QFileInfo &info : dirs)
319  {
320  if (info.absoluteFilePath().contains(otherVersionDir))
321  {
322  dir = info.absoluteFilePath();
323  break;
324  }
325  }
326  if (dir.isEmpty()) { return {}; }
327  dir = CFileUtils::appendFilePaths(dir, ui->rb_Cache->isChecked() ?
328  CDataCache::relativeFilePath() :
329  CSettingsCache::relativeFilePath());
330  return dir;
331  }

366  void CCopyConfigurationComponent::initCaches(const QStringList &files)
367  {
368  if (files.isEmpty()) { return; }
369  if (!ui->rb_Cache->isChecked()) { return; }
370  for (const QString &file : files)
371  {
372  if (file.contains("modelset", Qt::CaseInsensitive))
373  {
374  this->initMultiSimulatorCache(&m_modelSetCaches, file);
375  }
376  else if (file.contains("modelcache", Qt::CaseInsensitive))
377  {
378  this->initMultiSimulatorCache(&m_modelCaches, file);
379  }
380  else if (file.contains(CSwiftDirectories::bootstrapFileName()))
381  {
382  CData<TGlobalSetup> setup { this };
383  const CGlobalSetup s = CGlobalSetup::fromJsonFile(file, true);
384  setup.set(s);
385  }
386  }
387 
388  // allow the cache files to be generated before we will override them
389  CGuiApplication::processEventsFor(2500);
390  }

392  void CCopyConfigurationComponent::initMultiSimulatorCache(IMultiSimulatorModelCaches *cache, const QString &fileName)
393  {
394  const CSimulatorInfo info = cache->getSimulatorForFilename(fileName);
395  if (info.isNoSimulator()) { return; }
396  if (cache->isSaved(info)) { return; } // already a file and hence in .rev
397  const QFileInfo fi(fileName);
398  const CStatusMessage msg = cache->setCacheTimestamp(fi.lastModified(), info); // create cache file and timestamp in .rev
399  if (msg.isFailure())
400  {
401  CLogMessage(this).preformatted(msg);
402  }
