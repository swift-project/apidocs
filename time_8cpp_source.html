45  void CTime::parseFromString(const QString &time)
46  {
47  QString ts = time.trimmed();
48 
49  // deal with sign
50  double factor = 1.0;
51  if (ts.startsWith('+'))
52  {
53  ts.remove(0, 1);
54  }
55  else if (ts.startsWith('-'))
56  {
57  factor = -1.0;
58  ts.remove(0, 1);
59  }
60 
61  if (ts.contains(":") && (ts.length() == 8 || ts.length() == 5))
62  {
63  const int hour = ts.midRef(0, 2).toInt();
64  const int minute = ts.midRef(3, 2).toInt();
65  int second = 0;
66  if (ts.length() == 8) second = ts.midRef(6, 2).toInt();
67  (*this) = CTime(hour, minute, second);
68 
69  // fix sign if required
70  if (factor < 0)
71  {
72  this->setValueSameUnit(this->value() * factor);
73  }
74  }
75  else
76  {
77  CPhysicalQuantity::parseFromString(time);
78  }
79  }

104  QList<int> CTime::getHrsMinSecParts() const
105  {
106  CTime copy(*this);
107  copy.switchUnit(CTimeUnit::s());
108 
109  const double currentValue = copy.value();
110  const double hr = CMathUtils::trunc(currentValue / 3600);
111  const double remaining = std::fmod(currentValue, 3600);
112  const double mi = CMathUtils::trunc(remaining / 60);
113  const double se = CMathUtils::trunc(std::fmod(remaining, 60));
114 
115  QList<int> parts;
116  parts << qRound(hr) << qRound(mi) << qRound(se);
117  return parts;
