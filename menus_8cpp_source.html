70  void CMenu::removeItem(const CMenuItem &item)
71  {
72  auto it = std::find_if(m_data->items->begin(), m_data->items->end(), [=](const auto &i) {
73  return i.m_data->index == item.m_data->index;
74  });
75 
76  XPLMRemoveMenuItem(m_data->id, it->m_data->index);
77  it = m_data->items->erase(it);
78 
79  // Decrement the index of all below menu items
80  while (it != m_data->items->end())
81  {
82  it->m_data->index--;
83  ++it;
84  }
85  }
