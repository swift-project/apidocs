42  CHotkeyDialog::CHotkeyDialog(const CActionHotkey &actionHotkey, const CIdentifierList &identifiers, QWidget *parent) : QDialog(parent),
43  ui(new Ui::CHotkeyDialog),
44  m_actionHotkey(actionHotkey),
45  m_actionModel(this)
46  {
47  setWindowFlags(windowFlags() & ~Qt::WindowContextHelpButtonHint);
48 
49  ui->setupUi(this);
50  ui->qf_Advanced->hide();
51 
52  ui->pb_AdvancedMode->setIcon(CIcons::arrowMediumSouth16());
53  ui->tv_Actions->setModel(&m_actionModel);
54  selectAction();
55 
56  CHotkeyCombination combination = actionHotkey.getCombination();
57  if (!combination.isEmpty())
58  {
59  ui->pb_SelectedHotkey->setText(combination.toQString());
60  ui->pb_SelectedHotkey->setToolTip(combination.asStringWithDeviceNames());
61  }
62  else
63  {
64  ui->pb_SelectedHotkey->setToolTip("Press to select an new combination...");
65  }
66 
67  // get all remote identifiers in case there is no key for a remote machine yet
68  CIdentifierList registeredApplications;
69  if (sGui && sGui->getIContextApplication())
70  {
71  registeredApplications = sGui->getIContextApplication()->getRegisteredApplications();
72  const CIdentifier appIdentifier = sGui->getIContextApplication()->getApplicationIdentifier();
73  registeredApplications.push_back(appIdentifier);
74  }
75 
76  CIdentifierList machineIdentifiers(identifiers);
77  machineIdentifiers.push_back(registeredApplications); // add the registered applications
78 
79  if (actionHotkey.isValid()) { machineIdentifiers.push_back(actionHotkey.getApplicableMachine()); }
80  const CIdentifierList machineIdentifiersUnique = machineIdentifiers.getMachinesUnique();
81 
82  int index = -1;
83  for (const CIdentifier &app : machineIdentifiersUnique)
84  {
85  ui->cb_Identifier->addItem(app.getMachineName(), QVariant::fromValue(app));
86  if (m_actionHotkey.getApplicableMachine().hasSameMachineName(app)) { index = ui->cb_Identifier->count() - 1; }
87  }
88 
89  if (index < 0 && ui->cb_Identifier->count() > 0)
90  {
91  // if nothing was found
92  ui->cb_Identifier->setCurrentIndex(0);
93  }
94  else if (index != ui->cb_Identifier->currentIndex())
95  {
96  ui->cb_Identifier->setCurrentIndex(index);
97  }
98 
99  connect(ui->pb_AdvancedMode, &QPushButton::clicked, this, &CHotkeyDialog::advancedModeChanged);
100  connect(ui->pb_SelectedHotkey, &QPushButton::clicked, this, &CHotkeyDialog::captureHotkey);
101  connect(ui->pb_Accept, &QPushButton::clicked, this, &CHotkeyDialog::accept);
102  connect(ui->pb_Cancel, &QPushButton::clicked, this, &CHotkeyDialog::reject);
103  connect(ui->tv_Actions->selectionModel(), &QItemSelectionModel::selectionChanged, this, &CHotkeyDialog::changeSelectedAction);
104  connect(ui->cb_Identifier, qOverload<int>(&QComboBox::currentIndexChanged), this, &CHotkeyDialog::changeApplicableMachine);
105 
106  if (sGui && sGui->getInputManager())
107  {
108  connect(sGui->getInputManager(), &BlackCore::CInputManager::combinationSelectionChanged, this, &CHotkeyDialog::combinationSelectionChanged);
109  connect(sGui->getInputManager(), &BlackCore::CInputManager::combinationSelectionFinished, this, &CHotkeyDialog::combinationSelectionFinished);
110  }
111 
112  initStyleSheet();
113  }

197  CKeySelectionBox *CHotkeyDialog::addSelectionBox(const CKeyboardKeyList &allSupportedKeys, const CKeyboardKey &keyboardKey)
198  {
199  int currentIndex = 0;
200  const int width = qRound(1.5 * this->width());
201  const bool select = !keyboardKey.isUnknown();
202 
203  CKeySelectionBox *ksb = new CKeySelectionBox(ui->qf_Advanced);
204  ksb->addItem(noKeyButton(), QVariant::fromValue(CKeyboardKey())); // at front
205  for (const CKeyboardKey &supportedKey : allSupportedKeys)
206  {
207  ksb->addItem(supportedKey.toQString(), QVariant::fromValue(supportedKey));
208  if (select && supportedKey == keyboardKey)
209  {
210  currentIndex = ksb->count() - 1;
211  }
212  }
213 
214  ksb->setSelectedIndex(currentIndex);
215  ksb->setPopupWidth(qMin(width, 600));
216  ksb->addItem(noKeyButton(), QVariant::fromValue(CKeyboardKey())); // at back (easier to find it is there twice)
217 
218  ui->qf_Advanced->layout()->addWidget(ksb);
219  const int position = ui->qf_Advanced->layout()->count() - 1;
220  ksb->setProperty("position", position);
221  connect(ksb, &CKeySelectionBox::keySelectionChanged, this, &CHotkeyDialog::advancedKeyChanged);
222 
223  return ksb;
224  }

304  void CHotkeyDialog::setupAdvancedFrame()
305  {
306  if (!sGui || sGui->isShuttingDown()) { return; }
307  this->clearAdvancedFrame();
308  const CKeyboardKeyList allSupportedKeys = CKeyboardKeyList::allSupportedKeys();
309  const CJoystickButtonList allAvailableButtons = sGui->getInputManager()->getAllAvailableJoystickButtons();
310 
311  const CKeyboardKeyList keyboardKeys = m_actionHotkey.getCombination().getKeyboardKeys();
312  int c = 0;
313 
314  for (const CKeyboardKey &keyboardKey : keyboardKeys)
315  {
316  this->addSelectionBox(allSupportedKeys, keyboardKey);
317  c++;
318  }
319 
320  const CJoystickButtonList joystickButtons = m_actionHotkey.getCombination().getJoystickButtons();
321  for (const CJoystickButton &joystickButton : joystickButtons)
322  {
323  this->addSelectionBox(allAvailableButtons, joystickButton);
324  c++;
325  }
326 
327  // add one box more so we can add keys/buttons
328  if (c < 2)
329  {
330  this->addSelectionBox(allSupportedKeys);
331  }
332  }

346  void CHotkeyDialog::advancedKeyChanged(int oldIndex, int newIndex)
347  {
348  const CKeySelectionBox *ksb = qobject_cast<CKeySelectionBox *>(sender());
349  Q_ASSERT(ksb);
350 
351  if (ksb->itemData(oldIndex).canConvert<CKeyboardKey>() && ksb->itemData(newIndex).canConvert<CKeyboardKey>())
352  {
353  CKeyboardKey oldKey = ksb->itemData(oldIndex).value<CKeyboardKey>();
354  CKeyboardKey newKey = ksb->itemData(newIndex).value<CKeyboardKey>();
355 
356  CHotkeyCombination combination = m_actionHotkey.getCombination();
357  if (newKey.isUnknown()) { combination.removeKeyboardKey(oldKey); }
358  else { combination.replaceKey(oldKey, newKey); }
359  m_actionHotkey.setCombination(combination);
360  }
361 
362  if (ksb->itemData(oldIndex).canConvert<CJoystickButton>() && ksb->itemData(newIndex).canConvert<CJoystickButton>())
363  {
364  CJoystickButton oldButton = ksb->itemData(oldIndex).value<CJoystickButton>();
365  CJoystickButton newButton = ksb->itemData(newIndex).value<CJoystickButton>();
366 
367  CHotkeyCombination combination = m_actionHotkey.getCombination();
368  if (!newButton.isValid()) { combination.removeJoystickButton(oldButton); }
369  else { combination.replaceButton(oldButton, newButton); }
370  m_actionHotkey.setCombination(combination);
371  }
372 
373  ui->pb_SelectedHotkey->setText(m_actionHotkey.getCombination().toQString());
374  ui->pb_SelectedHotkey->setToolTip(m_actionHotkey.getCombination().asStringWithDeviceNames());
375  }

377  void CHotkeyDialog::selectAction()
378  {
379  if (m_actionHotkey.getAction().isEmpty()) { return; }
380  const QStringList tokens = m_actionHotkey.getAction().split("/", Qt::SkipEmptyParts);
381  QModelIndex parentIndex = QModelIndex();
382 
383  for (const QString &token : tokens)
384  {
385  const QModelIndex startIndex = m_actionModel.index(0, 0, parentIndex);
386  const QModelIndexList indexList = m_actionModel.match(startIndex, Qt::DisplayRole, QVariant::fromValue(token));
387  if (indexList.isEmpty()) { return; }
388  parentIndex = indexList.first();
389  ui->tv_Actions->expand(parentIndex);
390  }
391 
392  QItemSelectionModel *selectionModel = ui->tv_Actions->selectionModel();
393  selectionModel->select(parentIndex, QItemSelectionModel::Select);
394  }
