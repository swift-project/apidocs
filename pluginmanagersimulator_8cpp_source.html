30  ISimulatorListener *CPluginManagerSimulator::createListener(const QString &pluginId)
31  {
32  if (!m_plugins.contains(pluginId)) { return nullptr; }
33  PluginExtended &plugin = m_plugins[pluginId];
34  if (!plugin.listener)
35  {
36  ISimulatorFactory *factory = this->getFactory(pluginId);
37  if (!factory)
38  {
39  CLogMessage(this).warning(u"Could not load plugin '%1'.") << pluginId;
40  m_plugins.remove(pluginId);
41  return nullptr;
42  }
43 
44  ISimulatorListener *listener = factory->createListener(plugin.info);
45  connect(qApp, &QCoreApplication::aboutToQuit, listener, &QObject::deleteLater);
46  plugin.listener = listener;
47  }
48 
49  return plugin.listener;
50  }

69  int CPluginManagerSimulator::checkAvailableListeners()
70  {
71  if (m_plugins.isEmpty()) { return 0; }
72 
73  int c = 0;
74  for (PluginExtended &pi : m_plugins)
75  {
76  if (!pi.listener) { continue; }
77  if (!pi.listener->isRunning()) { continue; }
78  pi.listener->check();
79  c++;
80  }
81  return c;
82  }
