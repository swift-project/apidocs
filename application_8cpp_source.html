352  bool CApplication::start()
353  {
354  m_started = false; // reset
355 
356  // parse if needed, parsing contains its own error handling
357  if (!m_parsed)
358  {
359  const bool s = this->parseAndStartupCheck();
360  if (!s) { return false; }
361  }
362 
363  // parsing itself is done
364  CStatusMessageList msgs;
365  do
366  {
367  // clear cache?
368  if (this->isSet(m_cmdClearCache))
369  {
370  const QStringList files(CApplication::clearCaches());
371  msgs.push_back(CLogMessage(this).debug() << "Cleared cache, " << files.size() << " files");
372  }
373 
374  // crashpad dump
375  if (this->isSet(m_cmdTestCrashpad))
376  {
377  msgs.push_back(CLogMessage(this).info(u"About to simulate crash"));
378  QTimer::singleShot(10 * 1000, [=] {
379  if (!sApp || sApp->isShuttingDown()) { return; }
380  this->simulateCrash();
381  });
382  }
383 
385  if (!m_setupReader->isSetupAvailable())
386  {
387  msgs = this->requestReloadOfSetupAndVersion();
388  if (msgs.isFailure()) { break; }
389  if (msgs.isSuccess()) { msgs.push_back(this->waitForSetup()); }
390  }
391 
392  // start hookin
393  msgs.push_back(this->startHookIn());
394  if (msgs.isFailure()) { break; }
395 
396  // Settings if not already initialized
397  msgs.push_back(this->initLocalSettings());
398  if (msgs.isFailure()) { break; }
399  }
400  while (false);
401 
402  // terminate with failures, otherwise log messages
403  if (msgs.isFailure())
404  {
405  this->cmdLineErrorMessage(msgs);
406  return false;
407  }
408  else if (!msgs.isEmpty())
409  {
410  CLogMessage::preformatted(msgs);
411  }
412 
413  m_started = true;
414  return m_started;
415  }

417  CStatusMessageList CApplication::waitForSetup(int timeoutMs)
418  {
419  if (!m_setupReader) { return CStatusMessage(this).error(u"No setup reader"); }
420  CEventLoop eventLoop(this);
421  eventLoop.stopWhen(this, &CApplication::setupHandlingCompleted);
422  if (!m_setupReader->isSetupAvailable())
423  {
424  eventLoop.exec(timeoutMs);
425  }
426 
427  // setup handling completed with success or failure, or we run into time out
428  CStatusMessageList msgs;
429  if (!eventLoop.isGuardAlive())
430  {
431  msgs.push_back(CStatusMessage(this).error(u"Setup not available, already shutting down."));
432  return msgs;
433  }
434  bool forced = false;
435  if (!m_setupReader->isSetupAvailable())
436  {
437  forced = true;
438  m_setupReader->forceAvailabilityUpdate(); // maybe web reading still hanging
439  }
440  if (m_setupReader->isSetupAvailable())
441  {
442  msgs.push_back(CStatusMessage(this).info(forced ? QStringLiteral("Setup available after forcing (so likely web read still pending)") : QStringLiteral("Setup available")));
443  return msgs;
444  }
445 
446  // getting here means no "real" read success, and NO available cache
447  msgs.push_back(CStatusMessage(this).error(u"Setup not available, setup reading failed or timed out."));
448  if (m_setupReader->getLastSetupReadErrorMessages().hasErrorMessages())
449  {
450  msgs.push_back(m_setupReader->getLastSetupReadErrorMessages());
451  }
452  if (m_setupReader->hasCmdLineBootstrapUrl())
453  {
454  msgs.push_back(CStatusMessage(this).info(u"Bootstrap URL cmd line argument '%1'") << m_setupReader->getCmdLineBootstrapUrl());
455  }
456  return msgs;
457  }
