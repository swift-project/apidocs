13  bool IModelFilter<ContainerType>::stringMatchesFilterExpression(const QString &value, const QString &filter, Qt::CaseSensitivity cs) const
14  {
15  QString v = value.trimmed();
16  QString f = filter.trimmed();
17 
18  if (v.isEmpty() && f.isEmpty()) { return true; }
19  if (v.isEmpty()) { return false; }
20 
21  // no wildcard, just string matching
22  if (!filter.contains('*'))
23  {
24  return (v.indexOf(f, 0, cs) == 0) && (v.length() == f.length());
25  }
26 
27  const QString filterNoWildcard = stripWildcard(f);
28 
29  // included?
30  if (f.startsWith('*') && f.endsWith('*'))
31  {
32  return v.contains(filterNoWildcard, cs);
33  }
34 
35  // ends with
36  if (f.endsWith('*'))
37  {
38  return v.startsWith(filterNoWildcard, cs);
39  }
40 
41  // starting with
42  if (f.startsWith('*'))
43  {
44  return v.endsWith(filterNoWildcard, cs);
45  }
46 
47  // wildcard in middle
48  if (f.contains('*'))
49  {
50  const QStringList parts = v.split('*');
51  if (parts.size() < 2) { return false; }
52  const bool s = v.startsWith(parts.front(), cs) && v.endsWith(parts.back());
53  return s;
54  }
55 
56  // should never happen
57  CLogMessage(this).error(u"Illegal search pattern : '%1'") << f;
58  return false;
59  }
