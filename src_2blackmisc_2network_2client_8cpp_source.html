53  QString CClient::getCapabilitiesAsString() const
54  {
55  QStringList sl;
56  const Capabilities cap = this->getCapabilities();
57  if (cap.testFlag(FsdAtisCanBeReceived)) sl << "ATIS";
58  if (cap.testFlag(FsdWithInterimPositions)) sl << "interim pos.";
59  if (cap.testFlag(FsdWithIcaoCodes)) sl << "ICAO";
60  if (cap.testFlag(FsdWithAircraftConfig)) sl << "aircraft config";
61  if (cap.testFlag(FsdWithGroundFlag)) sl << "gnd.flag";
62  if (cap.testFlag(FsdModelString)) sl << "modelstring";
63  if (cap.testFlag(FsdWithVisualPositions)) sl << "visual pos.";
64  if (sl.isEmpty()) { return {}; }
65  return sl.join(", ");
66  }

101  QVariant CClient::propertyByIndex(CPropertyIndexRef index) const
102  {
103  if (index.isMyself()) { return QVariant::fromValue(*this); }
104  const ColumnIndex i = index.frontCasted<ColumnIndex>();
105  switch (i)
106  {
107  case IndexCapabilities: return QVariant::fromValue(m_capabilities);
108  case IndexCapabilitiesString: return QVariant(this->getCapabilitiesAsString());
109  case IndexCallsign: return this->getCallsign().propertyByIndex(index.copyFrontRemoved());
110  case IndexUser: return this->getUser().propertyByIndex(index.copyFrontRemoved());
111  case IndexModelString: return QVariant(m_modelString);
112  case IndexServer: return QVariant(m_server);
113  case IndexVoiceCapabilities: return m_voiceCapabilities.propertyByIndex(index.copyFrontRemoved());
114  case IndexVoiceCapabilitiesPixmap: return QVariant::fromValue(CIcon(m_voiceCapabilities.toIcon()).toPixmap());
115  case IndexVoiceCapabilitiesIcon: return QVariant::fromValue(CIcon(m_voiceCapabilities.toIcon()));
116  case IndexVoiceCapabilitiesString: return QVariant(m_voiceCapabilities.toQString(true));
117  default: break;
118  }
119  return CValueObject::propertyByIndex(index);
120  }

122  void CClient::setPropertyByIndex(CPropertyIndexRef index, const QVariant &variant)
123  {
124  if (index.isMyself())
125  {
126  (*this) = variant.value<CClient>();
127  return;
128  }
129  const ColumnIndex i = index.frontCasted<ColumnIndex>();
130  switch (i)
131  {
132  case IndexCapabilities: m_capabilities = variant.toInt(); break;
133  case IndexModelString: m_modelString = variant.toString(); break;
134  case IndexServer: m_server = variant.toString(); break;
135  case IndexUser: m_user.setPropertyByIndex(index.copyFrontRemoved(), variant); break;
136  case IndexCallsign: m_user.setCallsign(variant.value<BlackMisc::Aviation::CCallsign>()); break;
137  case IndexVoiceCapabilities: m_voiceCapabilities.setPropertyByIndex(index.copyFrontRemoved(), variant); break;
138  default: CValueObject::setPropertyByIndex(index, variant); break;
139  }
