56  void CAircraftCfgParser::startLoadingFromDisk(LoadMode mode, const ModelConsolidationCallback &modelConsolidation, const QStringList &modelDirectories)
57  {
58  static const CStatusMessage statusLoadingOk(this, CStatusMessage::SeverityInfo, u"Aircraft config parser loaded data");
59  static const CStatusMessage statusLoadingError(this, CStatusMessage::SeverityError, u"Aircraft config parser did NOT load data");
60 
61  const CSimulatorInfo simulator = this->getSimulator();
62  const QStringList modelDirs = this->getInitializedModelDirectories(modelDirectories, simulator);
63  const QStringList excludedDirectoryPatterns(m_settings.getModelExcludeDirectoryPatternsOrDefault(simulator)); // copy
64 
65  if (mode.testFlag(LoadInBackground))
66  {
67  if (m_parserWorker && !m_parserWorker->isFinished()) { return; }
68  emit this->diskLoadingStarted(simulator, mode);
69  m_parserWorker = CWorker::fromTask(this, "CAircraftCfgParser::startLoadingFromDisk",
70  [this, modelDirs, excludedDirectoryPatterns, simulator, modelConsolidation]() {
71  CStatusMessageList msgs;
72  const CAircraftCfgEntriesList aircraftCfgEntriesList = this->performParsing(modelDirs, excludedDirectoryPatterns, msgs);
73  CAircraftModelList models;
74  if (msgs.isSuccess())
75  {
76  models = aircraftCfgEntriesList.toAircraftModelList(simulator, true, msgs);
77  if (modelConsolidation) { modelConsolidation(models, true); }
78  }
79  return std::make_tuple(aircraftCfgEntriesList, models, msgs);
80  });
81  m_parserWorker->thenWithResult<LoaderResponse>(this, [this, simulator](const LoaderResponse &tuple) {
82  m_loadingMessages = std::get<2>(tuple);
83  if (m_loadingMessages.isSuccess())
84  {
85  m_parsedCfgEntriesList = std::get<0>(tuple);
86  const CAircraftModelList models(std::get<1>(tuple));
87  const bool hasData = !models.isEmpty();
88  if (hasData)
89  {
90  this->setModelsForSimulator(models, this->getSimulator());
91  }
92  // currently I treat no data as error
93  m_loadingMessages.push_front(hasData ? statusLoadingOk : statusLoadingError);
94  }
95  m_loadingMessages.freezeOrder();
96  emit this->loadingFinished(m_loadingMessages, simulator, ParsedData);
97  });
98  }
99  else if (mode == LoadDirectly)
100  {
101  emit this->diskLoadingStarted(simulator, mode);
102 
103  CStatusMessageList msgs;
104  m_parsedCfgEntriesList = this->performParsing(modelDirs, excludedDirectoryPatterns, msgs);
105  const CAircraftModelList models(m_parsedCfgEntriesList.toAircraftModelList(simulator, true, msgs));
106  m_loadingMessages = msgs;
107  m_loadingMessages.freezeOrder();
108  const bool hasData = !models.isEmpty();
109  if (hasData)
110  {
111  this->setCachedModels(models, this->getSimulator());
112  }
113  // currently I treat no data as error
114  emit this->loadingFinished(hasData ? statusLoadingOk : statusLoadingError, simulator, ParsedData);
115  }
116  }

133  CAircraftCfgEntriesList CAircraftCfgParser::performParsing(const QString &directory, const QStringList &excludeDirectories, CStatusMessageList &messages)
134  {
135  //
136  // function has to be threadsafe
137  //
138 
139  if (m_cancelLoading) { return CAircraftCfgEntriesList(); }
140 
141  // excluded?
142  if (CFileUtils::isExcludedDirectory(directory, excludeDirectories) || isExcludedSubDirectory(directory))
143  {
144  const CStatusMessage m = CStatusMessage(this).info(u"Skipping directory '%1' (excluded)") << directory;
145  messages.push_back(m);
146  return CAircraftCfgEntriesList();
147  }
148 
149  // set directory with name filters, get aircraft.cfg and sub directories
150  static const QString NoNameFilter;
151  QDir dir(directory, NoNameFilter, QDir::Name, QDir::Files | QDir::AllDirs | QDir::NoDotAndDotDot);
152  dir.setNameFilters(fileNameFilters());
153  if (!dir.exists())
154  {
155  return CAircraftCfgEntriesList(); // can happen if there are shortcuts or linked dirs not available
156  }
157 
158  const QString currentDir = dir.absolutePath();
159  CAircraftCfgEntriesList result;
160  emit this->loadingProgress(this->getSimulator(), QStringLiteral("Parsing '%1'").arg(currentDir), -1);
161 
162  // Dirs last is crucial, since I will break recursion on "aircraft.cfg" level
163  // with T514 this behaviour has been changed
164  const QFileInfoList files = dir.entryInfoList(QDir::Files | QDir::AllDirs | QDir::NoDotAndDotDot, QDir::DirsLast);
165 
166  // the sim.cfg/aircraft.cfg file should have an *.air file sibling
167  // if not we assume these files can be ignored
168  const QDir dirForAir(directory, CFsDirectories::airFileFilter(), QDir::Name, QDir::Files | QDir::NoDotAndDotDot);
169  const int airFilesCount = dirForAir.entryInfoList(QDir::Files | QDir::NoDotAndDotDot, QDir::DirsLast).size();
170  const bool hasAirFiles = airFilesCount > 0;
171 
172  if (getSimulator().isP3D() && !hasAirFiles)
173  {
174  const CStatusMessage m = CStatusMessage(this).warning(u"No \"air\" files in '%1'") << currentDir;
175  messages.push_back(m);
176  }
177 
178  for (const auto &fileInfo : files)
179  {
180  if (m_cancelLoading) { return CAircraftCfgEntriesList(); }
181  if (fileInfo.isDir())
182  {
183  const QString nextDir = fileInfo.absoluteFilePath();
184  if (currentDir.startsWith(nextDir, Qt::CaseInsensitive)) { continue; } // do not go up
185  if (dir == currentDir) { continue; } // do not recursively call same directory
186 
187  const CAircraftCfgEntriesList subList(performParsing(nextDir, excludeDirectories, messages));
188  if (messages.isSuccess())
189  {
190  result.push_back(subList);
191  }
192  else
193  {
194  const CStatusMessage m = CStatusMessage(this).warning(u"Parsing failed for '%1'") << nextDir;
195  messages.push_back(m);
196  }
197  }
198  else
199  {
200  // Enforce air files only for P3D
201  if (getSimulator().isP3D() && !hasAirFiles) { continue; }
202 
203  // due to the filter we expect only "aircraft.cfg"/"sim.cfg" here
204  // remark: in a 1st version I have used QSettings to parse to file as ini file
205  // unfortunately some files are malformed which could end up in wrong data
206 
207  const QString fileName = fileInfo.absoluteFilePath(); // full path and name
208  bool fileOk = false;
209  CStatusMessageList fileMsgs;
210  const CAircraftCfgEntriesList fileResults = CAircraftCfgParser::performParsingOfSingleFile(fileName, fileOk, fileMsgs);
211  if (!fileOk)
212  {
213  const CStatusMessage m = CStatusMessage(this).warning(u"Parsing of '%1' failed") << fileName;
214  messages.push_back(fileMsgs);
215  continue;
216  }
217 
218  result.push_back(fileResults);
219 
220  // With T514 we do not skip not anymore
221  // return result; // do not go any deeper in file tree, we found aircraft.cfg
222  }
223  }
224 
225  // all files finished,
226  // normally reached when no aircraft.cfg is found
227  return result;
228  }

230  CAircraftCfgEntriesList CAircraftCfgParser::performParsingOfSingleFile(const QString &fileName, bool &ok, CStatusMessageList &msgs)
231  {
232  // due to the filter we expect only "aircraft.cfg" files here
233  // remark: in a 1st version I have used QSettings to parse to file as ini file
234  // unfortunately some files are malformed which could end up in wrong data
235 
236  ok = false;
237  const QString fnFixed = CFileUtils::fixWindowsUncPath(fileName);
238  QFile file(fnFixed); // includes path
239  if (!file.open(QFile::ReadOnly | QFile::Text))
240  {
241  const CStatusMessage m = CStatusMessage(static_cast<CAircraftCfgParser *>(nullptr)).warning(u"Unable to read file '%1'") << fnFixed;
242  msgs.push_back(m);
243  return CAircraftCfgEntriesList();
244  }
245 
246  QTextStream in(&file);
247  QList<CAircraftCfgEntries> tempEntries;
248 
249  // parse through the file
250  QString atcType;
251  QString atcModel;
252  QString fltSection("[FLTSIM.0]");
253  static const QString fltSectionStr("[FLTSIM.%1]");
254 
255  int fltsimCounter = 0;
256  FileSection currentSection = Unknown;
257  const bool isRotorcraftPath = fileName.contains("rotorcraft", Qt::CaseInsensitive);
258 
259  while (!in.atEnd())
260  {
261  const QString lineFixed(in.readLine().trimmed());
262  if (lineFixed.isEmpty()) { continue; }
263  if (lineFixed.startsWith("["))
264  {
265  if (lineFixed.startsWith("[GENERAL]", Qt::CaseInsensitive))
266  {
267  currentSection = General;
268  continue;
269  }
270  if (lineFixed.startsWith(fltSection, Qt::CaseInsensitive))
271  {
272  CAircraftCfgEntries e(fileName, fltsimCounter);
273  if (isRotorcraftPath)
274  {
275  e.setRotorcraft(true);
276  }
277  tempEntries.append(e);
278  currentSection = Fltsim;
279  fltSection = fltSectionStr.arg(++fltsimCounter);
280  continue;
281  }
282  currentSection = Unknown;
283  continue;
284  }
285  switch (currentSection)
286  {
287  case General:
288  {
289  if (lineFixed.startsWith("//")) { break; }
290  if (atcType.isEmpty() || atcModel.isEmpty())
291  {
292  const QString c = getFixedIniLineContent(lineFixed);
293  if (lineFixed.startsWith("atc_type", Qt::CaseInsensitive))
294  {
295  atcType = c;
296  }
297  /*else if (lineFixed.startsWith("atc_model", Qt::CaseInsensitive))
298  {
299  atcModel = c;
300  }*/
301  else if (lineFixed.startsWith("icao_type_designator", Qt::CaseInsensitive))
302  {
303  atcModel = c;
304  }
305  }
306  }
307  break;
308  case Fltsim:
309  {
310  if (lineFixed.startsWith("//")) { break; }
311  CAircraftCfgEntries &e = tempEntries[tempEntries.size() - 1];
312  if (lineFixed.startsWith("atc_", Qt::CaseInsensitive))
313  {
314  if (lineFixed.startsWith("atc_parking_codes", Qt::CaseInsensitive))
315  {
316  e.setAtcParkingCode(getFixedIniLineContent(lineFixed));
317  }
318  else if (lineFixed.startsWith("atc_airline", Qt::CaseInsensitive))
319  {
320  e.setAtcAirline(getFixedIniLineContent(lineFixed));
321  }
322  else if (lineFixed.startsWith("atc_id_color", Qt::CaseInsensitive))
323  {
324  e.setAtcIdColor(getFixedIniLineContent(lineFixed));
325  }
326  }
327  else if (lineFixed.startsWith("ui_", Qt::CaseInsensitive))
328  {
329  if (lineFixed.startsWith("ui_manufacturer", Qt::CaseInsensitive))
330  {
331  e.setUiManufacturer(getFixedIniLineContent(lineFixed));
332  }
333  else if (lineFixed.startsWith("ui_typerole", Qt::CaseInsensitive))
334  {
335  bool r = getFixedIniLineContent(lineFixed).toLower().contains("rotor");
336  e.setRotorcraft(r);
337  }
338  else if (lineFixed.startsWith("ui_type", Qt::CaseInsensitive))
339  {
340  e.setUiType(getFixedIniLineContent(lineFixed));
341  }
342  else if (lineFixed.startsWith("ui_variation", Qt::CaseInsensitive))
343  {
344  e.setUiVariation(getFixedIniLineContent(lineFixed));
345  }
346  }
347  else if (lineFixed.startsWith("description", Qt::CaseInsensitive))
348  {
349  e.setDescription(getFixedIniLineContent(lineFixed));
350  }
351  else if (lineFixed.startsWith("texture", Qt::CaseInsensitive))
352  {
353  e.setTexture(getFixedIniLineContent(lineFixed));
354  }
355  else if (lineFixed.startsWith("createdBy", Qt::CaseInsensitive))
356  {
357  e.setCreatedBy(getFixedIniLineContent(lineFixed));
358  }
359  else if (lineFixed.startsWith("sim", Qt::CaseInsensitive))
360  {
361  e.setSimName(getFixedIniLineContent(lineFixed));
362  }
363  else if (lineFixed.startsWith("title", Qt::CaseInsensitive))
364  {
365  e.setTitle(getFixedIniLineContent(lineFixed));
366  }
367  }
368  break;
369  default:
370  case Unknown: break;
371  }
372  } // all lines
373  file.close();
374 
375  // store all entries
376  const QFileInfo fileInfo(fnFixed);
377  QDateTime fileTimestamp(fileInfo.lastModified());
378  if (!fileTimestamp.isValid() || fileInfo.birthTime() > fileTimestamp)
379  {
380  fileTimestamp = fileInfo.birthTime();
381  }
382  Q_ASSERT_X(fileTimestamp.isValid(), Q_FUNC_INFO, "Missing file timestamp");
383 
384  CAircraftCfgEntriesList result;
385  for (const CAircraftCfgEntries &e : std::as_const(tempEntries))
386  {
387  if (e.getTitle().isEmpty())
388  {
389  const CStatusMessage m = CStatusMessage(static_cast<CAircraftCfgParser *>(nullptr)).info(u"FS model in %1, index %2 has no title") << fileName << e.getIndex();
390  msgs.push_back(m);
391  continue;
392  }
393  CAircraftCfgEntries newEntries(e);
394  newEntries.setAtcModel(atcModel);
395  newEntries.setAtcType(atcType);
396  newEntries.setUtcTimestamp(fileTimestamp);
397  result.push_back(newEntries);
398  }
399  ok = true;
400  return result; // do not go any deeper in file tree, we found aircraft.cfg
401  }

408  QString CAircraftCfgParser::fixedStringContent(const QVariant &qv)
409  {
410  if (qv.isNull() || !qv.isValid())
411  {
412  return {}; // normal when there is no settings value
413  }
414  else if (static_cast<QMetaType::Type>(qv.type()) == QMetaType::QStringList)
415  {
416  const QStringList l = qv.toStringList();
417  return l.join(",").trimmed();
418  }
419  else if (static_cast<QMetaType::Type>(qv.type()) == QMetaType::QString)
420  {
421  return qv.toString().trimmed();
422  }
423  Q_ASSERT(false);
424  return {};

427  QString CAircraftCfgParser::getFixedIniLineContent(const QString &line)
428  {
429  if (line.isEmpty()) { return {}; }
430 
431  // Remove inline comments starting with ;
432  const int indexComment = line.indexOf(';');
433  QString content = line.leftRef(indexComment - 1).trimmed().toString();
434 
435  const int index = line.indexOf('=');
436  if (index < 0) { return {}; }
437  if (line.length() < index + 1) { return {}; }
438 
439  content = content.midRef(index + 1).trimmed().toString();
440 
441  // fix "" strings, some are malformed and just contain " at beginning, not at the end
442  if (hasBalancedQuotes(content, '"'))
443  {
444  // seems to be OK
445  // ex: title=B767-300ER - Condor "Retro Jet"
446 
447  if (content.size() > 2 && content.startsWith('"') && content.endsWith('"'))
448  {
449  // completly in quotes, example title="B767-300ER - Condor Retro Jet"
450  // we assume the quotes shall be removed
451  content.remove(0, 1);
452  content.chop(1);
453  }
454  }
455  else
456  {
457  // UNBALANCED
458 
459  // could be OK, example title=B767-300ER - Condor Retro Jet"
460  // if (content.endsWith('"')) { content.remove(content.size() - 1, 1); }
461 
462  // Unlikely, title="B767-300ER - Condor "Retro Jet
463  if (content.startsWith('"')) { content.remove(0, 1); }
464  }
465 
466  // fix C style linebreaks
467  content.replace("\\n", " ");
468  content.replace("\\t", " ");
469  return content;
470  }
