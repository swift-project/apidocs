123  void CIdentifier::appendName(const QString &name)
124  {
125  if (m_name.endsWith(name)) { return; }

127  const int index = m_name.lastIndexOf(':');
128  if (index >= 0) { m_name = m_name.left(index); }
129  m_name += QStringLiteral(":") + name;

143  QString CIdentifier::toDBusObjectPath(const QString &root) const
144  {
145  QString path = root;
146  path += '/' % toDBusPathElement(m_machineName) % "__" % toDBusPathElement(m_machineIdBase64);
147  path += '/' % toDBusPathElement(m_processName) % "__" % QString::number(m_processId);
148 
149  const QString name = toDBusPath(m_name);
150  Q_ASSERT_X(!name.contains("//") && !name.startsWith('/') && !name.endsWith('/'), Q_FUNC_INFO, "Invalid name");
151  if (!name.isEmpty()) { path += '/' % name; }
152  return path;
153  }

155  CIdentifier CIdentifier::fromDBusObjectPath(const QString &path, const QString &root)
156  {
157  const QString relative = path.startsWith(root) ? path.mid(root.length()) : path;
158  const QString machine = relative.section('/', 1, 1);
159  const QString process = relative.section('/', 2, 2);
160  const QString name = relative.section('/', 3, -1);
161 
162  CIdentifier result(fromDBusPath(name));
163  result.m_machineIdBase64 = fromDBusPathElement(machine.section("__", 1, 1));
164  result.m_machineName = fromDBusPathElement(machine.section("__", 0, 0));
165  result.m_processId = process.section("__", 1, 1).toInt();
166  result.m_processName = fromDBusPathElement(process.section("__", 0, 0));
167  return result;
168  }

234  QVariant CIdentifier::propertyByIndex(BlackMisc::CPropertyIndexRef index) const
235  {
236  if (index.isMyself()) { return QVariant::fromValue(*this); }
237 
238  const ColumnIndex i = index.frontCasted<ColumnIndex>();
239 
240  switch (i)
241  {
242  case IndexName: return QVariant::fromValue(m_name);
243  case IndexMachineIdBase64: return QVariant::fromValue(m_machineIdBase64);
244  case IndexMachineName: return QVariant::fromValue(getMachineName());
245  case IndexMachineId: return QVariant::fromValue(getMachineId());
246  case IndexProcessId: return QVariant::fromValue(m_processId);
247  case IndexProcessName: return QVariant::fromValue(m_processName);
248  case IndexIsFromLocalMachine: return QVariant::fromValue(isFromLocalMachine());
249  case IndexIsFromSameProcess: return QVariant::fromValue(hasApplicationProcessId());
250  case IndexIsFromSameProcessName: return QVariant::fromValue(hasApplicationProcessName());
251  default: return CValueObject::propertyByIndex(index);

255  int CIdentifier::comparePropertyByIndex(CPropertyIndexRef index, const CIdentifier &compareValue) const
256  {
257  if (index.isMyself()) { return Compare::compare(m_processId, compareValue.m_processId); }
258 
259  const ColumnIndex i = index.frontCasted<ColumnIndex>();
260 
261  switch (i)
262  {
263  case IndexName: return m_name.compare(compareValue.m_name, Qt::CaseInsensitive);
264  case IndexMachineIdBase64: return m_machineIdBase64.compare(compareValue.m_machineIdBase64);
265  case IndexMachineName: return m_machineName.compare(compareValue.m_machineName, Qt::CaseInsensitive);
266  case IndexMachineId: return m_machineName.compare(compareValue.m_machineName, Qt::CaseInsensitive);
267  case IndexProcessId: return Compare::compare(m_processId, compareValue.m_processId);
268  case IndexProcessName: return m_processName.compare(compareValue.m_processName, Qt::CaseInsensitive);
269  case IndexIsFromLocalMachine: return Compare::compare(this->isFromLocalMachine(), compareValue.isFromLocalMachine());
270  case IndexIsFromSameProcess: return Compare::compare(this->hasApplicationProcessId(), compareValue.hasApplicationProcessId());
271  case IndexIsFromSameProcessName: return Compare::compare(this->hasApplicationProcessName(), compareValue.hasApplicationProcessName());
272  default: return CValueObject::comparePropertyByIndex(index, compareValue);
273  }
274  }
