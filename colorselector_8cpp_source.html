114  void CColorSelector::dropEvent(QDropEvent *event)
115  {
116  if (!event) { return; }
117  const QMimeData *mime = event->mimeData();
118  if (!mime) { return; }
119 
120  if (mime->hasColor())
121  {
122  const QColor color = qvariant_cast<QColor>(event->mimeData()->colorData());
123  if (!color.isValid()) { return; }
124  this->setColor(color);
125  }
126  else if (CGuiUtility::hasSwiftVariantMimeType(mime))
127  {
128  const CVariant valueVariant(CGuiUtility::fromSwiftDragAndDropData(mime));
129  if (valueVariant.isValid())
130  {
131  if (valueVariant.canConvert<CRgbColor>())
132  {
133  const CRgbColor rgb(valueVariant.value<CRgbColor>());
134  if (!rgb.isValid()) { return; }
135  this->setColor(rgb);
136  }
137  else if (valueVariant.canConvert<QColor>())
138  {
139  const QColor qColor(valueVariant.value<QColor>());
140  if (!qColor.isValid()) { return; }
141  this->setColor(qColor);
142  }
143  }
144  }
145  else if (mime->hasText())
146  {
147  const QString t = mime->text();
148  if (t.isEmpty()) { return; }
149  if (!t.contains("{"))
150  {
151  // we assume a color string, not an object (because there are no {})
152  const CRgbColor c(t);
153  if (!c.isValid()) { return; }
154  this->setColor(c);
155  }
156  }
157  }
