162  void CTestAircraftSituation::altitudeCorrection()
163  {
164  CAircraftSituation::AltitudeCorrection correction = CAircraftSituation::NoCorrection;
165  const CAircraftSituationList situations = testSituations();
166  CAircraftSituation situation = situations.front();
167  CAltitude alt(100, CAltitude::MeanSeaLevel, CLengthUnit::ft());
168 
169  situation.setAltitude(alt);
170  CAltitude corAlt = situation.getCorrectedAltitude(true, &correction);
171 
172  // no elevation, expect same values
173  QVERIFY2(corAlt == alt, "Expect same altitude");
174 
175  CElevationPlane ep(situation, CElevationPlane::singlePointRadius());
176  situation.setGroundElevation(ep, CAircraftSituation::Test);
177 
178  // now we have same alt and elevation values
179  // no elevation, expect same values
180  corAlt = situation.getCorrectedAltitude(true, &correction);
181  QVERIFY2(corAlt == alt, "Still expect same altitude");
182 
183  // now we use CG underflow detection
184  const CLength cg(3, CLengthUnit::m());
185  situation.setCG(cg);
186  corAlt = situation.getCorrectedAltitude(true, &correction);
187  QVERIFY2(correction == CAircraftSituation::Underflow, "Expect underflow correction");
188  QVERIFY2(corAlt > alt, "Expect corrected altitude");
189  QVERIFY2((corAlt - cg) == alt, "Expect correction by CG");
190 
191  // Unit of CG must be irrelevant
192  CLength cg2(cg);
193  cg2.switchUnit(CLengthUnit::m());
194  situation.setCG(cg2);
195  CAltitude corAlt2 = situation.getCorrectedAltitude(true, &correction);
196  QVERIFY2(correction == CAircraftSituation::Underflow, "Expect underflow correction");
197  QVERIFY2(corAlt2 == corAlt, "Expect same value for corrected altitude");
198 
199  // now we test a negative altitude
200  alt = CAltitude(-1000, CAltitude::MeanSeaLevel, CLengthUnit::ft());
201  ep.setGeodeticHeight(alt);
202  situation.setAltitude(alt);
203  situation.setGroundElevation(ep, CAircraftSituation::Test);
204  corAlt = situation.getCorrectedAltitude(true, &correction);
205  QVERIFY2(correction == CAircraftSituation::Underflow, "Expect underflow correction");
206  QVERIFY2(corAlt > alt, "Expect corrected altitude");
207  QVERIFY2((corAlt - cg) == alt, "Expect correction by CG");
208 
209  // overflow detection
210  alt = CAltitude(1000, CAltitude::MeanSeaLevel, CLengthUnit::ft());
211  situation.setAltitude(alt);
212  ep.setGeodeticHeight(alt);
213  ep.addAltitudeOffset(CLength(-100, CLengthUnit::ft()));
214  situation.setGroundElevation(ep, CAircraftSituation::Test);
215  corAlt = situation.getCorrectedAltitude(true, &correction);
216  QVERIFY2(corAlt == alt, "Expect same altitude, no overflow since not on gnd.");
217 
218  situation.setOnGround(CAircraftSituation::OnGround, CAircraftSituation::InFromNetwork);
219  corAlt = situation.getCorrectedAltitude(true, &correction);
220  QVERIFY2(correction == CAircraftSituation::DraggedToGround, "Expect dragged to gnd.");
221  QVERIFY2(corAlt < alt, "Expect corrected altitude dragged to gnd.");
222  QVERIFY2(corAlt == (ep.getAltitude() + cg), "Expect correction by CG");
223  }

225  void CTestAircraftSituation::sortHint()
226  {
227  constexpr int Lists = 50000;
228  constexpr int Loops = 20;
229  CAircraftSituationList situations = testSituations();
230  situations.sortAdjustedLatestFirst();
231  CSequence<CAircraftSituationList> listOfLists1;
232  CSequence<CAircraftSituationList> listOfLists2;
233  for (int i = 0; i < Lists; ++i)
234  {
235  listOfLists1.push_back(situations);
236  listOfLists2.push_back(situations);
237  listOfLists1.back().detach();
238  listOfLists2.back().detach();
239  listOfLists2.back().setAdjustedSortHint(CAircraftSituationList::AdjustedTimestampLatestFirst);
240  }
241 
242  QElapsedTimer time;
243  int cpuTime = getThreadCpuTimeMs();
244  time.start();
245  for (int i = 0; i < Loops; ++i)
246  {
247  for (const CAircraftSituationList &s : std::as_const(listOfLists1))
248  {
249  const CAircraftSituation s1 = s.oldestAdjustedObject();
250  const CAircraftSituation s2 = s.latestAdjustedObject();
251  QVERIFY(s1.getAdjustedMSecsSinceEpoch() < s2.getAdjustedMSecsSinceEpoch());
252  }
253  }
254  const auto noHint = std::make_pair(time.elapsed(), getThreadCpuTimeMs() - cpuTime);
255 
256  cpuTime = getThreadCpuTimeMs();
257  time.start();
258  for (int i = 0; i < Loops; ++i)
259  {
260  for (const CAircraftSituationList &s : std::as_const(listOfLists2))
261  {
262  const CAircraftSituation s1 = s.oldestAdjustedObject();
263  const CAircraftSituation s2 = s.latestAdjustedObject();
264  QVERIFY(s1.getAdjustedMSecsSinceEpoch() < s2.getAdjustedMSecsSinceEpoch());
265  }
266  }
267  const auto hint = std::make_pair(time.elapsed(), getThreadCpuTimeMs() - cpuTime);
268  const double ratio = static_cast<double>(hint.second) / static_cast<double>(noHint.second); // expected <1.0
269 
270  // qDebug() << "MacOS:" << boolToYesNo(CBuildConfig::isRunningOnMacOSPlatform());
271  qDebug() << "Access without hint" << noHint.first << "ms (CPU time" << noHint.second << "ms)";
272  qDebug() << "Access with hint" << hint.first << "ms (CPU time" << hint.second << "ms)";
273  qDebug() << "Access ratio" << ratio;
274 
275 #if 0
276  // remark On Win/Linux access with hint is faster
277  // on MacOS the times are the same, maybe with hint it is even slightly slower
278  if (noHint >= hint)
279  {
280  // on MacOS we accept up to 10% overhead and SKIP the test
281  if (CBuildConfig::isRunningOnMacOSPlatform())
282  {
283  if (ratio <= 1.1) { return; } // on MacOS 10% overhead accepted
284 
285  // more than 10% overhead!
287  QSKIP("Skipped sort hint on MacOS");
288  return;
289  }
290  }
291 #endif
292 
294  const bool ok = hint <= noHint;
295  if (ok || CBuildConfig::isLocalDeveloperDebugBuild())
296  {
297  QVERIFY2(ok, "Expected hinted sort being faster");
298  }
299  }

301  CAircraftSituationList CTestAircraftSituation::testSituations()
302  {
303  // "Kugaaruk Airport","Pelly Bay","Canada","YBB","CYBB",68.534401,-89.808098,56,-7,"A","America/Edmonton","airport","OurAirports"
304  // "Baie Comeau Airport","Baie Comeau","Canada","YBC","CYBC",49.13249969482422,-68.20439910888672,71,-5,"A","America/Toronto","airport","OurAirports"
305  // "CFB Bagotville","Bagotville","Canada","YBG","CYBG",48.33060073852539,-70.99639892578125,522,-5,"A","America/Toronto","airport","OurAirports"
306 
307  CAircraftSituationList situations;
308  const qint64 ts = QDateTime::currentSecsSinceEpoch();
309  const qint64 os = CFsdSetup::c_positionTimeOffsetMsec;
310  CAltitude alt(10000, CAltitude::MeanSeaLevel, CLengthUnit::m());
311  static const CCoordinateGeodetic dummyPos(48.33060073852539, -70.99639892578125, 522);
312 
313  for (int i = 0; i < 10; i++)
314  {
315  CAircraftSituation s(dummyPos);
316  const qint64 cTs = ts - i * os;
317  s.setMSecsSinceEpoch(cTs);
318  s.setTimeOffsetMs(os);
319  CAltitude altitude(alt);
320  altitude.addValueSameUnit(-100 * i); // 10000, 9900, 9800 .... (newer->older)
321  s.setAltitude(altitude);
322  situations.push_back(s);
323  }

342  CAircraftSituationList CTestAircraftSituation::testSetRotateUpPitch(const CAircraftSituationList &situations)
343  {
344  CAircraftSituationList newSituations;
345  double average = 0;
346  for (const CAircraftSituation &situation : situations)
347  {
348  CAircraftSituation s(situation);
349  const double pitch = CMathUtils::randomDouble(1.5);
350  average += pitch;
351  s.setPitch(CAngle(pitch, CAngleUnit::deg()));
352  newSituations.push_back(s);
353  }
354  average = average / newSituations.size();
355  CAircraftSituation avg = newSituations.front();
356  avg.setPitch(CAngle(average, CAngleUnit::deg()));
357  avg.addMsecs(1000); // make this the latest situation
358  newSituations.push_front(avg); // first value is average pitch, so it will NOT be detected
359  return newSituations;
360  }
