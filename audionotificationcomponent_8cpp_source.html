117  void CAudioNotificationComponent::onNotificationVolumeChanged(int volume)
118  {
119  volume = qMax(25, qMin(100, volume));
120  CSettings as(m_audioSettings.getThreadLocal());
121  if (as.getNotificationVolume() == volume) { return; }
122  as.setNotificationVolume(volume);

126  CNotificationSounds::NotificationFlag CAudioNotificationComponent::checkBoxToFlag(const QCheckBox *cb) const
127  {
128  if (!cb) { return CNotificationSounds::NoNotifications; }
129 
130  if (cb == ui->cb_SetupAudioPTTClickDown) { return CNotificationSounds::PTTClickKeyDown; }
131  if (cb == ui->cb_SetupAudioPTTClickUp) { return CNotificationSounds::PTTClickKeyUp; }
132 
133  if (cb == ui->cb_SetupAudioNotificationTextCallsignMentioned) { return CNotificationSounds::NotificationTextCallsignMentioned; }
134  if (cb == ui->cb_SetupAudioNotificationTextMessageFrequency) { return CNotificationSounds::NotificationTextMessageFrequency; }
135  if (cb == ui->cb_SetupAudioNotificationTextMessagePrivate) { return CNotificationSounds::NotificationTextMessagePrivate; }
136  if (cb == ui->cb_SetupAudioNotificationTextMessageSupervisor) { return CNotificationSounds::NotificationTextMessageSupervisor; }
137  if (cb == ui->cb_SetupAudioNotificationTextMessageUnicom) { return CNotificationSounds::NotificationTextMessageUnicom; }
138 
139  if (cb == ui->cb_SetupAfvBlocked) { return CNotificationSounds::AFVBlocked; }
140  if (cb == ui->cb_SetupAfvClicked) { return CNotificationSounds::AFVClicked; }
141 
142  if (cb == ui->cb_SetupAudioNotificationLogin) { return CNotificationSounds::NotificationLogin; }
143  if (cb == ui->cb_SetupAudioNotificationLogoff) { return CNotificationSounds::NotificationLogoff; }
144 
145  return CNotificationSounds::NoNotifications;
146  }

148  void CAudioNotificationComponent::onNotificationsToggled(bool checked)
149  {
150  if (!sGui || sGui->isShuttingDown() || !sGui->getIContextAudio()) { return; }
151  CSettings as(m_audioSettings.getThreadLocal());
152 
153  as.setNotificationFlag(CNotificationSounds::PTTClickKeyDown, ui->cb_SetupAudioPTTClickDown->isChecked());
154  as.setNotificationFlag(CNotificationSounds::PTTClickKeyUp, ui->cb_SetupAudioPTTClickUp->isChecked());
155 
156  as.setNotificationFlag(CNotificationSounds::NotificationTextMessageFrequency, ui->cb_SetupAudioNotificationTextMessageFrequency->isChecked());
157  as.setNotificationFlag(CNotificationSounds::NotificationTextMessagePrivate, ui->cb_SetupAudioNotificationTextMessagePrivate->isChecked());
158  as.setNotificationFlag(CNotificationSounds::NotificationTextMessageSupervisor, ui->cb_SetupAudioNotificationTextMessageSupervisor->isChecked());
159  as.setNotificationFlag(CNotificationSounds::NotificationTextMessageUnicom, ui->cb_SetupAudioNotificationTextMessageUnicom->isChecked());
160  as.setNotificationFlag(CNotificationSounds::NotificationTextCallsignMentioned, ui->cb_SetupAudioNotificationTextCallsignMentioned->isChecked());
161 
162  as.setNotificationFlag(CNotificationSounds::AFVBlocked, ui->cb_SetupAfvBlocked->isChecked());
163  as.setNotificationFlag(CNotificationSounds::AFVClicked, ui->cb_SetupAfvClicked->isChecked());
164 
165  as.setNotificationFlag(CNotificationSounds::NotificationLogin, ui->cb_SetupAudioNotificationLogin->isChecked());
166  as.setNotificationFlag(CNotificationSounds::NotificationLogoff, ui->cb_SetupAudioNotificationLogoff->isChecked());
167 
168  const CStatusMessage msg = m_audioSettings.set(as);
169  CLogMessage(this).preformatted(msg);
170 
171  const QCheckBox *sender = qobject_cast<const QCheckBox *>(QObject::sender());
172  if (checked && sGui && sGui->getCContextAudioBase() && sender)
173  {
174  const CNotificationSounds::NotificationFlag f = this->checkBoxToFlag(sender);
175  sGui->getCContextAudioBase()->playNotification(f, false, as.getNotificationVolume());
176  }
177  }

179  void CAudioNotificationComponent::selectNotificationSoundsDir()
180  {
181  CSettings s = m_audioSettings.get();
182  const QString dir = QFileDialog::getExistingDirectory(this, QStringLiteral("Open directory"), s.getNotificationSoundDirectory(), QFileDialog::ShowDirsOnly | QFileDialog::DontResolveSymlinks);
183  const QDir d(dir);
184  if (d.exists())
185  {
186  s.setNotificationSoundDirectory(dir);
187  ui->le_SoundDir->setText(s.getNotificationSoundDirectory());
188  const CStatusMessage m = m_audioSettings.setAndSave(s);
189  CLogMessage::preformatted(m);
190  }
191  }
