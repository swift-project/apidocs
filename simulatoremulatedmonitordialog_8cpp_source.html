110  void CSimulatorEmulatedMonitorDialog::appendFunctionCall(const QString &function, const QString &p1, const QString &p2, const QString &p3)
111  {
112  static const QString c1("%1 %2");
113  static const QString c2("%1 %2 %3");
114  static const QString c3("%1 %2 %3 %4");
115 
116  CStatusMessage msg;
117  if (!p3.isEmpty())
118  {
119  msg = CStatusMessage(this, CStatusMessage::SeverityInfo, c3.arg(function, p1, p2, p3));
120  }
121  else if (!p2.isEmpty())
122  {
123  msg = CStatusMessage(this, CStatusMessage::SeverityInfo, c2.arg(function, p1, p2));
124  }
125  else if (!p1.isEmpty())
126  {
127  msg = CStatusMessage(this, CStatusMessage::SeverityInfo, c1.arg(function, p1));
128  }
129  else
130  {
131  msg = CStatusMessage(this, CStatusMessage::SeverityInfo, function);
132  }
133  this->appendStatusMessageToList(msg);
134  }

243  void CSimulatorEmulatedMonitorDialog::interpolatorLogButton()
244  {
245  const QObject *sender = QObject::sender();
246  bool ok = false;
247  if (sender == ui->pb_InterpolatorStopLog)
248  {
249  ok = m_simulator->parseCommandLine(".drv logint off", this->identifier());
250  }
251  else if (sender == ui->pb_InterpolatorWriteLog)
252  {
253  ok = m_simulator->parseCommandLine(".drv logint write", this->identifier());
254  }
255  else if (sender == ui->pb_InterpolatorClearLog)
256  {
257  ok = m_simulator->parseCommandLine(".drv logint clear", this->identifier());
258  }
259  else if (sender == ui->pb_InterpolatorShowLogs)
260  {
261  ok = m_simulator->parseCommandLine(".drv logint show", this->identifier());
262  }
263  else if (sender == ui->pb_InterpolatorStartLog)
264  {
265  const CCallsign cs = ui->comp_LogInterpolatorCallsign->getCallsign();
266  if (cs.isValid())
267  {
268  ok = m_simulator->parseCommandLine(".drv logint " + cs.asString(), this->identifier());
269  }
270  else
271  {
272  CLogMessage(this).warning(u"Need a (valid) callsign to write a log");
273  ok = true; // already a warning
274  }
275  }
276  else if (sender == ui->pb_InterpolatorFetch)
277  {
278  // toggle fetching
279  const int timeMs = m_simulator->isInterpolatorFetching() ? -1 : ui->le_InterpolatorTimeMs->text().toInt();
280  const bool fetching = m_simulator->setInterpolatorFetchTime(timeMs);
281  ui->led_Fetching->setOn(fetching);
282  ui->pb_InterpolatorFetch->setText(fetching ? "Stop" : "Fetch");
283  this->enableInterpolationLogButtons(fetching);
284  ok = true;
285  }
286  else
287  {
288  Q_ASSERT_X(false, Q_FUNC_INFO, "Unhandled button");
289  }
290  if (!ok) { CLogMessage(this).warning(u"Cannot parse command for button: %1") << sender->objectName(); }
291  }
