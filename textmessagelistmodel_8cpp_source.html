33  void CTextMessageListModel::setTextMessageMode(CTextMessageListModel::TextMessageMode mode)
34  {
35  if (m_textMessageMode == mode) return;
36  m_textMessageMode = mode;
37  m_columns.clear();
38  switch (mode)
39  {
40  case NotSet:
41  case FromTo:
42  {
43  CColumn col = CColumn("type", CTextMessage::IndexIcon);
44  col.setSortPropertyIndex({ CTextMessage::IndexSenderCallsign, CCallsign::IndexSuffix });
45  m_columns.addColumn(col);
46  m_columns.addColumn(CColumn("time", "received", CTextMessage::IndexUtcTimestamp, new CDateTimeFormatter(CDateTimeFormatter::formatHms())));
47  m_columns.addColumn(CColumn::standardString("from", { CTextMessage::IndexSenderCallsign, CCallsign::IndexCallsignString }));
48  m_columns.addColumn(CColumn::standardString("to", CTextMessage::IndexRecipientCallsignOrFrequency));
49  m_columns.addColumn(CColumn::standardString("message", CTextMessage::IndexMessage));
50 
51  // default sort order
52  this->setSortColumnByPropertyIndex(CTextMessage::IndexUtcTimestamp);
53  m_sortOrder = Qt::DescendingOrder;
54  }
55  break;
56 
57  case From:
58  {
59  m_columns.addColumn(CColumn("time", "received", CTextMessage::IndexUtcTimestamp, new CDateTimeFormatter(CDateTimeFormatter::formatHms())));
60  m_columns.addColumn(CColumn::standardString("from", { CTextMessage::IndexSenderCallsign, CCallsign::IndexCallsignString }));
61  m_columns.addColumn(CColumn::standardString("message", CTextMessage::IndexMessage));
62 
63  // default sort order
64  this->setSortColumnByPropertyIndex(CTextMessage::IndexUtcTimestamp);
65  m_sortOrder = Qt::DescendingOrder;
66  }
67  break;
68 
69  default:
70  qFatal("Wrong mode");
71  break;
72  }
