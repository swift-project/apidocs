57  IContextAudio *IContextAudio::create(CCoreFacade *runtime, CCoreFacadeConfig::ContextMode mode, CDBusServer *server, QDBusConnection &connection)
58  {
59  // for audio no empty context is available
60  // since CContextAudioBaseImpl provides audio on either side (core/GUI) we do not use ContextAudioEmpty
61  // ContextAudioEmpty would cause issue, as it is initializing "common parts" during shutdown
62  switch (mode)
63  {
64  case CCoreFacadeConfig::Local:
65  case CCoreFacadeConfig::LocalInDBusServer:
66  default:
67  return (new CContextAudio(mode, runtime))->registerWithDBus(server);
68  case CCoreFacadeConfig::Remote:
69  return new CContextAudioProxy(CDBusServer::coreServiceName(connection), connection, mode, runtime);
70  case CCoreFacadeConfig::NotUsed:
71  BLACK_VERIFY_X(false, Q_FUNC_INFO, "Empty context not supported for audio (since AFV)");
72  return nullptr;
73  }
74  }

76  bool CContextAudioBase::parseCommandLine(const QString &commandLine, const CIdentifier &originator)
77  {
78  Q_UNUSED(originator)
79  if (commandLine.isEmpty()) { return false; }
80  CSimpleCommandParser parser(
81  { ".vol", ".volume", // output volume
82  ".mute", // mute
83  ".unmute", // unmute
84  ".aliased" });
85  parser.parse(commandLine);
86  if (!parser.isKnownCommand()) { return false; }
87 
88  if (parser.matchesCommand(".mute"))
89  {
90  this->setMute(true);
91  return true;
92  }
93  else if (parser.matchesCommand(".unmute"))
94  {
95  this->setMute(false);
96  return true;
97  }
98  else if (parser.commandStartsWith("vol") && parser.countParts() > 1)
99  {
100  const int v = parser.toInt(1);
101  this->setMasterOutputVolume(v);
102  return true;
103  }
104  else if (afvClient() && parser.matchesCommand(".aliased") && parser.countParts() > 1)
105  {
106  const bool enable = parser.toBool(1, true);
107  afvClient()->enableAliasedStations(enable);
108 
109  CLogMessage(this).info(u"Aliased stations are: %1") << boolToOnOff(enable);
110  return true;
111  }
112  return false;
113  }

153  void CContextAudioBase::initVoiceClient()
154  {
155  if (m_voiceClient) { return; }
156 
157  const CAudioDeviceInfoList devices = CAudioDeviceInfoList::allDevices();
158  if (devices != m_activeLocalDevices)
159  {
160  m_activeLocalDevices = devices;
161  emit this->changedLocalAudioDevices(devices);
162  }
163 
164 #ifdef Q_OS_WIN
165  if (!m_winCoInitialized)
166  {
167  HRESULT hr = CoInitializeEx(nullptr, COINIT_MULTITHREADED);
168 
169  // RPC_E_CHANGED_MODE: CoInitializeEx was already called by someone else in this thread with a different mode.
170  if (hr == RPC_E_CHANGED_MODE)
171  {
172  CLogMessage(this).debug(u"CoInitializeEx was already called with a different mode. Trying again.");
173  hr = CoInitializeEx(nullptr, COINIT_APARTMENTTHREADED);
174  }
175 
176  // S_OK: The COM library was initialized successfully on this thread.
177  // S_FALSE: The COM library is already initialized on this thread. Reference count was incremented. This is not an error.
178  if (hr == S_OK || hr == S_FALSE) { m_winCoInitialized = true; }
179  }
180 #endif
181 
182  m_voiceClient = new CAfvClient(CVoiceSetup().getAfvVoiceServerUrl(), this);
183  const CVoiceSetup vs = m_voiceSettings.getThreadLocal();
184  m_voiceClient->updateVoiceServerUrl(vs.getAfvVoiceServerUrl());
185 
186  Q_ASSERT_X(m_voiceClient->thread() == qApp->thread(), Q_FUNC_INFO, "Should be in main thread");
187  m_voiceClient->start(); // thread
188  Q_ASSERT_X(m_voiceClient->owner() == this, Q_FUNC_INFO, "Wrong owner");
189  Q_ASSERT_X(m_voiceClient->thread() != qApp->thread(), Q_FUNC_INFO, "Must NOT be in main thread");
190 
191  // connect(m_voiceClient, &CAfvClient::outputVolumePeakVU, this, &CContextAudioBase::outputVolumePeakVU, Qt::QueuedConnection);
192  // connect(m_voiceClient, &CAfvClient::inputVolumePeakVU, this, &CContextAudioBase::inputVolumePeakVU, Qt::QueuedConnection);
193  // connect(m_voiceClient, &CAfvClient::receivingCallsignsChanged, this, &CContextAudioBase::receivingCallsignsChanged, Qt::QueuedConnection);
194  // connect(m_voiceClient, &CAfvClient::updatedFromOwnAircraftCockpit, this, &CContextAudioBase::updatedFromOwnAircraftCockpit, Qt::QueuedConnection);
195  connect(m_voiceClient, &CAfvClient::startedAudio, this, &CContextAudioBase::startedAudio, Qt::QueuedConnection);
196  connect(m_voiceClient, &CAfvClient::stoppedAudio, this, &CContextAudioBase::stoppedAudio, Qt::QueuedConnection);
197  connect(m_voiceClient, &CAfvClient::ptt, this, &CContextAudioBase::ptt, Qt::QueuedConnection);
198  connect(m_voiceClient, &CAfvClient::changedMute, this, &CContextAudioBase::changedMute, Qt::QueuedConnection);
199  connect(m_voiceClient, &CAfvClient::connectionStatusChanged, this, &CContextAudioBase::onAfvConnectionStatusChanged, Qt::QueuedConnection);
200  connect(m_voiceClient, &CAfvClient::afvConnectionFailure, this, &CContextAudioBase::onAfvConnectionFailure, Qt::QueuedConnection);
201  }

279  bool CContextAudioBase::connectAudioWithNetworkCredentials()
280  {
281  if (!m_voiceClient) { return false; }
282  if (!sApp || sApp->isShuttingDown() || !sApp->getIContextNetwork()) { return false; }
283 
284  const CEcosystem ecoSystem = this->getIContextNetwork()->getConnectedServer().getEcosystem();
285  if (ecoSystem != CEcosystem::vatsim())
286  {
287  CLogMessage(this).info(u"Will not use AFV as ecosystem is '%1'") << ecoSystem.toQString(true);
288  return false;
289  }
290 
291  const CVoiceSetup vs = m_voiceSettings.getThreadLocal();
292  m_voiceClient->updateVoiceServerUrl(vs.getAfvVoiceServerUrl());
293 
294  const CUser connectedUser = this->getIContextNetwork()->getConnectedServer().getUser();
295  const QString client = "swift " % BlackConfig::CBuildConfig::getShortVersionString();
296  CCallsign cs = connectedUser.getCallsign();
297  this->unRegisterAudioCallsign(cs, this->identifier()); // un-register "myself"
298  if (this->hasRegisteredAudioCallsign(cs)) // anybody else using that callsign
299  {
301  cs = CCallsign(cs.asString() + "2");
302  }
303  CLogMessage(this).info(u"About to connect to voice as '%1' '%2'") << connectedUser.getId() << cs;
304  m_voiceClient->connectTo(connectedUser.getId(), connectedUser.getPassword(), cs.asString(), client);
305  this->registerAudioCallsign(cs, this->identifier()); // login can still fail, but we "block" this callsign
306  return true;
307  }

409  void CContextAudioBase::setMasterOutputVolume(int volume)
410  {
411  if (!m_voiceClient) { return; }
412 
413  const bool wasMuted = this->isMuted();
414  volume = CSettings::fixOutVolume(volume);
415 
416  const int currentVolume = m_voiceClient->getNormalizedMasterOutputVolume();
417  const bool changedVoiceOutput = (currentVolume != volume);
418  if (changedVoiceOutput)
419  {
420  // TODO: KB 2020-05 the mute handling should entirely go to AFV client!
421  m_voiceClient->setNormalizedMasterOutputVolume(volume);
422  m_outMasterVolumeBeforeMute = volume;
423 
424  emit this->changedAudioVolume(volume);
425  if ((volume > 0 && wasMuted) || (volume < 1 && !wasMuted))
426  {
427  // inform about muted
428  emit this->changedMute(volume < 1);
429  }
430  }
431 
432  CSettings as(m_audioSettings.getThreadLocal());
433  if (as.getOutVolume() != volume)
434  {
435  as.setOutVolume(volume);
436  m_audioSettings.set(as);
437  }
438  }

440  void CContextAudioBase::setComOutputVolume(CComSystem::ComUnit comUnit, int volume)
441  {
442  if (comUnit != CComSystem::Com1 && comUnit != CComSystem::Com2) { return; }
443  if (!m_voiceClient) { return; }
444 
445  volume = CSettings::fixOutVolume(volume);
446 
447  const int currentVolume = m_voiceClient->getNormalizedComOutputVolume(comUnit);
448  const bool changedVoiceOutput = (currentVolume != volume);
449  if (changedVoiceOutput)
450  {
451  m_voiceClient->setNormalizedComOutputVolume(comUnit, volume);
452  emit this->changedAudioVolume(volume);
453  }
454 
455  CSettings as(m_audioSettings.getThreadLocal());
456  if (comUnit == CComSystem::Com1 && as.getOutVolumeCom1() != volume)
457  {
458  as.setOutVolumeCom1(volume);
459  m_audioSettings.set(as);
460  }
461  else if (comUnit == CComSystem::Com2 && as.getOutVolumeCom2() != volume)
462  {
463  as.setOutVolumeCom2(volume);
464  m_audioSettings.set(as);
465  }
466  }

518 
519  const CSettings settings = m_audioSettings.getThreadLocal();
