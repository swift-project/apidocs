234  {
235  Q_ASSERT_X(m_result.canConvert<R>(), Q_FUNC_INFO, "Type in thenWithResult must match return type of task");
236 
237  // MS 2018-10 It is possible that a queued finished() signal will be delivered after the worker was
238  // destroyed, so we can't refer to the this pointer inside the contextual then() lambda.
239  // Therefore we use a promise to extract the result from a non-contextual then(). See T414.
240  auto promise = std::make_shared<std::promise<R>>();
241  auto future = promise->get_future().share();
242  then([this, promise]() { promise->set_value(this->resultNoWait<R>()); });
243  then(context, [context, functor, future]() { Private::invokeSlot(functor, context, future.get()); });
244  }
