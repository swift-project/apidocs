108  void CUser::setRealName(const QString &realname)
109  {
110  QString rn(simplifyAccents(CObfuscation::decode(realname).simplified()));
111  if (rn.isEmpty())
112  {
113  m_realname.clear();
114  return;
115  }
116 
117  if (!this->hasHomeBase())
118  {
119  // only apply stripping if home base is not explicitly given
120  // try to strip homebase: I understand the limitations, but we will have more correct hits as failures I assume
121  thread_local QRegularExpression tsRegex("(-\\s*|\\s)([A-Z]{4})$");
122  const QRegularExpressionMatch match = tsRegex.match(rn);
123  if (match.hasMatch())
124  {
125  const int pos = match.capturedStart(0);
126  const QString icao = match.captured(0).trimmed().right(4);
127  rn = rn.leftRef(pos).trimmed().toString();
128  this->setHomeBase(CAirportIcaoCode(icao));
129  }
130  }
131 
132  // do not beautify before stripping home base
133  m_realname = beautifyRealName(rn);
134  }

202  QString CUser::beautifyRealName(const QString &realName)
203  {
204  QString newRealName(realName.simplified().trimmed());
205  if (newRealName.isEmpty()) { return newRealName; }
206  if (is09OnlyString(newRealName)) { return newRealName; } // new VATSIM COD, allowing id as name, see https://discordapp.com/channels/539048679160676382/539846348275449887/599969308247851018
207 
208  int uc = 0;
209  int lc = 0;
210  for (const QChar &ch : realName)
211  {
212  // Joe Doe -> valid
213  // jOE dOE -> invalid
214  // Joe McArthur -> valid
215  if (uc > 1 && lc > 2 && lc > uc) { return newRealName; } // mixed case name, no need to beautify
216  if (ch.isLower())
217  {
218  lc++;
219  continue;
220  }
221  if (ch.isUpper())
222  {
223  uc++;
224  continue;
225  }
226  }
227 
228  // simple title case beautifying
229  newRealName = newRealName.toLower();
230  QString::Iterator i = newRealName.begin();
231  bool upperNextChar = true;
232  while (i != newRealName.end())
233  {
234  if (i->isSpace() || *i == '-')
235  {
236  upperNextChar = true;
237  }
238  else if (upperNextChar)
239  {
240  const QChar u(i->toUpper());
241  *i = u;
242  upperNextChar = false;
243  }
244  ++i;
245  }
246  return newRealName;
247  }

289  int CUser::comparePropertyByIndex(CPropertyIndexRef index, const CUser &compareValue) const
290  {
291  if (index.isMyself()) { return this->getRealName().compare(compareValue.getRealName(), Qt::CaseInsensitive); }
292  const ColumnIndex i = index.frontCasted<ColumnIndex>();
293  switch (i)
294  {
295  case IndexEmail: return m_email.compare(compareValue.getEmail(), Qt::CaseInsensitive);
296  case IndexId: return m_id.compare(compareValue.getId(), Qt::CaseInsensitive);
297  case IndexId7Digit: return this->get7DigitId().compare(compareValue.get7DigitId(), Qt::CaseInsensitive);
298  case IndexIdInteger: return BlackMisc::Compare::compare(getIntegerId(), compareValue.getIntegerId());
299  case IndexRealName: return m_realname.compare(compareValue.getRealName(), Qt::CaseInsensitive);
300  case IndexHomebase: return m_homebase.comparePropertyByIndex(index.copyFrontRemoved(), compareValue.getHomeBase());
301  case IndexCallsign: return m_callsign.comparePropertyByIndex(index.copyFrontRemoved(), compareValue.getCallsign());
302  case IndexPassword: break;
303  default: return CValueObject::comparePropertyByIndex(index, compareValue);
304  }
305  Q_ASSERT_X(false, Q_FUNC_INFO, "compare failed");
306  return 0;
307  }
