32  bool checkPermissions(CAtomicFile::OpenMode mode, const QFileInfo &fileInfo)
33  {
34  bool ok = true;
35  qt_ntfs_permission_lookup++;
36  if ((mode & CAtomicFile::ReadOnly) && !fileInfo.isReadable()) { ok = false; }
37  if ((mode & CAtomicFile::WriteOnly) && !fileInfo.isWritable()) { ok = false; }
38  qt_ntfs_permission_lookup--;
39  return ok;

42  bool CAtomicFile::open(CAtomicFile::OpenMode mode)
43  {
44  m_originalFilename = fileName();
45  QFileInfo fileInfo(fileName());
46  if (exists() && !checkPermissions(mode, fileInfo))
47  {
48  m_permissionError = true;
49  setErrorString("Wrong permissions");
50  return false;
51  }
52  setFileName(QFileInfo(fileInfo.dir(), ".tmp." + fileInfo.fileName() + "." + randomSuffix()).filePath());
53  if (exists()) { remove(); }
54 
55  bool ok = true;
56  if (mode & ReadOnly)
57  {
58  if (exists(m_originalFilename) && !copy(m_originalFilename, fileName())) { ok = false; }
59  }
60 
61  if (ok && !QFile::open(mode)) { ok = false; }
62  if (!ok) { setFileName(m_originalFilename); }
63  return ok;

122  {

134  void CAtomicFile::replaceOriginal()
135  {
136  auto encode = [](const QString &s) {
137  const auto prefix = "\\\\?\\"; // support long paths: https://msdn.microsoft.com/en-us/library/windows/desktop/aa365247.aspx#maxpath
138  return (prefix + QDir::toNativeSeparators(QDir::cleanPath(QFileInfo(s).absoluteFilePath()))).toStdWString();
139  };
140  auto replace = exists(m_originalFilename);
141  auto result = replace ? ReplaceFile(encode(m_originalFilename).c_str(), encode(fileName()).c_str(), nullptr, REPLACEFILE_IGNORE_MERGE_ERRORS, nullptr, nullptr) : MoveFileEx(encode(fileName()).c_str(), encode(m_originalFilename).c_str(), MOVEFILE_WRITE_THROUGH);
142  if (!result)
143  {
144  wchar_t *s = nullptr;
145  FormatMessage(FORMAT_MESSAGE_ALLOCATE_BUFFER | FORMAT_MESSAGE_FROM_SYSTEM, nullptr, GetLastError(), 0, reinterpret_cast<LPWSTR>(&s), 0, nullptr);
146  const QString windowsError = (replace ? u"ReplaceFile: " : u"MoveFileEx: ") % QString::fromWCharArray(s).simplified();
147  LocalFree(reinterpret_cast<HLOCAL>(s));
148 
149  // fall back to non-atomic remove-and-rename
150  if (exists(m_originalFilename))
151  {
152  QFile old(m_originalFilename);
153  if (!old.remove())
154  {
155  // fall back failed, so report the reasons for the original failure AND the fall back failure
156  m_renameError = true;
157  setErrorString(windowsError % u" QFile::remove: " % old.errorString());
158  return;
159  }
160  }
161  rename(m_originalFilename);
162  }
163  }
