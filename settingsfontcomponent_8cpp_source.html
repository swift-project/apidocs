86  void CSettingsFontComponent::changeFont()
87  {
88  if (!sGui || sGui->isShuttingDown()) { return; }
89  const QString fontSize = ui->cb_SettingsGuiFontSize->currentText().append("pt");
90  const QString fontFamily = ui->cb_SettingsGuiFont->currentFont().family();
91  const QString fontStyleCombined = ui->cb_SettingsGuiFontStyle->currentText();
92 
93  QString fontColor = m_selectedColor.name();
94  if (!m_selectedColor.isValid() || m_selectedColor.name().isEmpty())
95  {
96  fontColor.clear();
97  }
98  ui->le_SettingsGuiFontColor->setText(fontColor);
99  m_qss = CStyleSheetUtility::asStylesheet(fontFamily, fontSize, CStyleSheetUtility::fontStyle(fontStyleCombined), CStyleSheetUtility::fontWeight(fontStyleCombined), fontColor);
100  if (m_mode == CSettingsFontComponent::DirectUpdate)
101  {
102  const bool ok = sGui->updateFont(m_qss);
103  if (ok)
104  {
105  CLogMessage(this).info(u"Updated font style");
106  }
107  else
108  {
109  CLogMessage(this).warning(u"Updating style failed");
110  }
111  }
112  emit this->accept();
113  }
