21  CContextApplicationProxy::CContextApplicationProxy(const QString &serviceName, QDBusConnection &connection, CCoreFacadeConfig::ContextMode mode, CCoreFacade *runtime) : IContextApplication(mode, runtime)
22  {
23  m_dBusInterface = new CGenericDBusInterface(serviceName, IContextApplication::ObjectPath(), IContextApplication::InterfaceName(), connection, this);
24  this->relaySignals(serviceName, connection);
25 
26  connect(this, &CContextApplicationProxy::remoteHotkeyAction, this, &CContextApplicationProxy::processRemoteHotkeyActionCall);
27 
28  m_pingTimer.setObjectName(serviceName + "::m_pingTimer");
29  connect(&m_pingTimer, &QTimer::timeout, this, &CContextApplicationProxy::reRegisterApplications);
30  m_pingTimer.start(PingIdentifiersMs);
31  }

33  void CContextApplicationProxy::relaySignals(const QString &serviceName, QDBusConnection &connection)
34  {
35  // signals originating from impl side
36  bool s = connection.connect(serviceName, IContextApplication::ObjectPath(), IContextApplication::InterfaceName(),
37  "settingsChanged", this, SIGNAL(settingsChanged(BlackMisc::CValueCachePacket, BlackMisc::CIdentifier)));
38  Q_ASSERT(s);
39  s = connection.connect(serviceName, IContextApplication::ObjectPath(), IContextApplication::InterfaceName(),
40  "registrationChanged", this, SIGNAL(registrationChanged()));
41  Q_ASSERT(s);
42  s = connection.connect(serviceName, IContextApplication::ObjectPath(), IContextApplication::InterfaceName(),
43  "hotkeyActionsRegistered", this, SIGNAL(hotkeyActionsRegistered(QStringList, BlackMisc::CIdentifier)));
44  Q_ASSERT(s);
45  s = connection.connect(serviceName, IContextApplication::ObjectPath(), IContextApplication::InterfaceName(),
46  "remoteHotkeyAction", this, SIGNAL(remoteHotkeyAction(QString, bool, BlackMisc::CIdentifier)));
47  Q_ASSERT(s);
48  s = connection.connect(serviceName, IContextApplication::ObjectPath(), IContextApplication::InterfaceName(),
49  "requestDisplayOnConsole", this, SIGNAL(requestDisplayOnConsole(QString)));
50  Q_UNUSED(s);
51  this->relayBaseClassSignals(serviceName, connection, IContextApplication::ObjectPath(), IContextApplication::InterfaceName());
52  }

70  {
71  CSettingsDictionary result = m_dBusInterface->callDBusRet<CSettingsDictionary>(QLatin1String("getUnsavedSettingsKeysDescribed"));

163  void CContextApplicationProxy::reRegisterApplications()
164  {
165  if (!m_dBusInterface) { return; }
166  if (m_proxyPingIdentifiers.isEmpty()) { return; }
167  const CIdentifierSet identifiers = m_proxyPingIdentifiers; // copy so member can be modified
168  for (const CIdentifier &identifier : identifiers)
169  {
170  this->registerApplication(identifier);
171  }
172  }

174  bool CContextApplicationProxy::isContextResponsive(const QString &dBusAddress, QString &msg, int timeoutMs)
175  {
176  const bool connected = CDBusServer::isDBusAvailable(dBusAddress, msg, timeoutMs);
177  if (!connected) { return false; }
178 
179  static const QString dBusName("contexttest");
180  QDBusConnection connection = CDBusServer::connectToDBus(dBusAddress, dBusName);
181  CContextApplicationProxy proxy(BlackMisc::CDBusServer::coreServiceName(), connection, CCoreFacadeConfig::Remote, nullptr);
182  const CIdentifier id("swift proxy test");
183  const CIdentifier pingId = proxy.registerApplication(id);
184  const bool ok = (id == pingId);
185  if (ok)
186  {
187  proxy.unregisterApplication(id);
188  }
189  else
190  {
191  msg = "Mismatch in proxy ping, context not ready.";
192  }
193  CDBusServer::disconnectFromDBus(connection, dBusAddress);
194  return ok;
195  }
