117  bool CSettingsTextMessageStyle::changeFontSize(bool increase)
118  {
119  QString style = m_style;
120  thread_local const QRegularExpression re("table\\s*\\{.*:\\s*(\\d{1,2}).*\\}");
121  const QRegularExpressionMatch match = re.match(style);
122  const QStringList matches = match.capturedTexts();
123  if (matches.size() != 2) { return false; }
124 
125  bool ok;
126  int ptSize = matches.last().toInt(&ok);
127  if (!ok) { return false; }
128  if (increase)
129  {
130  ptSize++;
131  if (ptSize > 16) { return false; }
132  }
133  else
134  {
135  ptSize--;
136  if (ptSize < 6) { return false; }
137  }
138 
139  const QString pt = QString::number(ptSize) % u"pt";
140  QString tableStyle = matches.front();
141  thread_local const QRegularExpression rePt("\\d{1,2}pt");
142  tableStyle.replace(rePt, pt);
143  this->replaceTableStyle(tableStyle);
144  return true;
145  }
