45  CStatusMessageList CDatabaseWriter::asyncPublishModels(const CAircraftModelList &models, const QString &extraInfo)
46  {
47  CStatusMessageList msgs;
48  if (m_shutdown || !sApp)
49  {
50  msgs.push_back(CStatusMessage(CStatusMessage::SeverityWarning, u"Database writer shutting down"));
51  return msgs;
52  }
53 
54  if (this->isModelReplyOverdue())
55  {
56  const bool killed = this->killPendingModelReply();
57  if (killed)
58  {
59  const CStatusMessage msg(CStatusMessage::SeverityWarning, u"Aborted outdated pending reply");
60  msgs.push_back(CStatusMessage(msg));
61  // need to let a potential receiver know it has failed
62  emit this->publishedModels(CAircraftModelList(), CAircraftModelList(), msg, false, false);
63  }
64  }
65 
66  if (m_pendingModelPublishReply)
67  {
68  msgs.push_back(CStatusMessage(CStatusMessage::SeverityWarning, u"Another write operation in progress"));
69  return msgs;
70  }
71 
72  const bool compress = models.size() > 3;
73  QHttpMultiPart *multiPart = new QHttpMultiPart(QHttpMultiPart::FormDataType, this);
74  multiPart->append(CDatabaseUtils::getJsonTextMultipart(models.toDatabaseJson(), compress));
75 
76  if (sApp->getGlobalSetup().dbDebugFlag())
77  {
78  multiPart->append(CDatabaseUtils::getMultipartWithDebugFlag());
79  }
80 
81  QUrl url(m_modelPublishUrl.toQUrl());
82  if (compress) { url.setQuery(CDatabaseUtils::getCompressedQuery()); }
83  QNetworkRequest request(url);
84  CNetworkUtils::ignoreSslVerification(request);
85  const QByteArray eInfo = extraInfo.toLatin1();
86  request.setRawHeader(QByteArray("swift-extrainfo"), eInfo);
87  const int logId = m_writeLog.addPendingUrl(url);
88  m_pendingModelPublishReply = sApp->postToNetwork(request, logId, multiPart, { this, &CDatabaseWriter::postedModelsResponse });
89  m_modelReplyPendingSince = QDateTime::currentMSecsSinceEpoch();
90  return msgs;
91  }

152  void CDatabaseWriter::postedModelsResponse(QNetworkReply *nwReplyPtr)
153  {
154  QScopedPointer<QNetworkReply, QScopedPointerDeleteLater> nwReply(nwReplyPtr);
155  if (m_shutdown || !sApp)
156  {
157  nwReply->abort();
158  return;
159  }
160 
161  m_pendingModelPublishReply = nullptr;
162  const QUrl url(nwReply->url());
163  const QString urlString(url.toString());
164  if (nwReply->error() == QNetworkReply::NoError)
165  {
166  const QString responseData(nwReply->readAll().trimmed());
167  nwReply->close(); // close asap
168  if (responseData.isEmpty())
169  {
170  const CStatusMessageList msgs({ CStatusMessage(this, CStatusMessage::SeverityError, u"No response data from " % urlString) });
171  emit this->publishedModels(CAircraftModelList(), CAircraftModelList(), msgs, false, false);
172  return;
173  }
174 
175  CAircraftModelList modelsPublished;
176  CAircraftModelList modelsSkipped;
177  CStatusMessageList msgs;
178  bool directWrite;
179  const bool sendingSuccessful = parseSwiftPublishResponse(responseData, modelsPublished, modelsSkipped, msgs, directWrite);
180  const int c = CDatabaseUtils::fillInMissingAircraftAndLiveryEntities(modelsPublished);
181 
182  emit this->publishedModels(modelsPublished, modelsSkipped, msgs, sendingSuccessful, directWrite);
183 
184  if (!modelsPublished.isEmpty())
185  {
186  emit this->publishedModelsSimplified(modelsPublished, directWrite);
187  }
188  Q_UNUSED(c);
189  }
190  else
191  {
192  const QString error = nwReply->errorString();
193  nwReply->close(); // close asap
194  const CStatusMessageList msgs({ CStatusMessage(this, CStatusMessage::SeverityError, u"HTTP error: " % error) });
195  emit this->publishedModels(CAircraftModelList(), CAircraftModelList(), msgs, false, false);
196  }
197  }

240  bool CDatabaseWriter::isModelReplyOverdue() const
241  {
242  if (m_modelReplyPendingSince < 0 || !m_pendingModelPublishReply) { return false; }
243  const qint64 ms = QDateTime::currentMSecsSinceEpoch() - m_modelReplyPendingSince;
244  return ms > 7500;

257  QList<QByteArray> CDatabaseWriter::splitData(const QByteArray &data, int size)
258  {
259  if (data.size() <= size) { return QList<QByteArray>({ data }); }
260  int pos = 0, arrsize = data.size();
261  QList<QByteArray> arrays;
262  while (pos < arrsize)
263  {
264  const QByteArray arr = data.mid(pos, size);
265  arrays << arr;
266  pos += arr.size();
267  }
268  return arrays;
269  }

271  bool CDatabaseWriter::parseSwiftPublishResponse(const QString &jsonResponse, CAircraftModelList &publishedModels, CAircraftModelList &skippedModels, CStatusMessageList &messages, bool &directWrite)
272  {
273  directWrite = false;
274 
275  if (jsonResponse.isEmpty())
276  {
277  messages.push_back(CStatusMessage(static_cast<CDatabaseWriter *>(nullptr), CStatusMessage::SeverityError, u"Empty JSON data for published models"));
278  return false;
279  }
280 
281  const QJsonDocument jsonDoc(QJsonDocument::fromJson(jsonResponse.toUtf8()));
282 
283  // array of messages only
284  if (jsonDoc.isArray())
285  {
286  const CStatusMessageList msgs(CStatusMessageList::fromDatabaseJson(jsonDoc.array()));
287  messages.push_back(msgs);
288  return true;
289  }
290 
291  // no object -> most likely some fucked up HTML string with the PHP error
292  if (!jsonDoc.isObject())
293  {
294  const QString phpError(CNetworkUtils::removeHtmlPartsFromPhpErrorMessage(jsonResponse));
295  messages.push_back(CStatusMessage(static_cast<CDatabaseWriter *>(nullptr), CStatusMessage::SeverityError, phpError));
296  return false;
297  }
298 
299  // fully blown object
300  QJsonObject json(jsonDoc.object());
301  bool hasData = false;
302  if (json.contains("msgs"))
303  {
304  const QJsonValue msgJson(json.take("msgs"));
305  const CStatusMessageList msgs(CStatusMessageList::fromDatabaseJson(msgJson.toArray()));
306  if (!msgs.isEmpty())
307  {
308  messages.push_back(msgs);
309  hasData = true;
310  }
311  }
312 
313  // direct write means models written, otherwise CRs
314  if (json.contains("directWrite"))
315  {
316  const QJsonValue dw(json.take("directWrite"));
317  directWrite = dw.toBool(false);
318  }
319 
320  if (json.contains("publishedModels"))
321  {
322  const QJsonValue publishedJson(json.take("publishedModels"));
323  const CAircraftModelList published = CAircraftModelList::fromDatabaseJson(publishedJson.toArray(), "");
324  if (!published.isEmpty())
325  {
326  publishedModels.push_back(published);
327  hasData = true;
328  }
329  }
330 
331  if (json.contains("skippedModels"))
332  {
333  const QJsonValue skippedJson(json.take("skippedModels"));
334  const CAircraftModelList skipped = CAircraftModelList::fromDatabaseJson(skippedJson.toArray(), "");
335  if (!skipped.isEmpty())
336  {
337  skippedModels.push_back(skipped);
338  hasData = true;
339  }
340  }
341 
342  if (!hasData)
343  {
344  messages.push_back(CStatusMessage(static_cast<CDatabaseWriter *>(nullptr), CStatusMessage::SeverityError, u"Received response, but no JSON data"));
345  }
346 
347  return hasData;
348  }
