49  void CSoundcardSampleProvider::pttUpdate(bool active, const QVector<TxTransceiverDto> &txTransceivers)
50  {
51  if (active)
52  {
53  if (!txTransceivers.isEmpty())
54  {
55  QVector<TxTransceiverDto> txTransceiversFiltered = txTransceivers;
56 
57  txTransceiversFiltered.erase(std::remove_if(txTransceiversFiltered.begin(), txTransceiversFiltered.end(), [this](const TxTransceiverDto &d) {
58  return !m_receiverIDs.contains(d.id);
59  }),
60  txTransceiversFiltered.end());
61 
62  for (const TxTransceiverDto &txTransceiver : txTransceiversFiltered)
63  {
64  auto it = std::find_if(m_receiverInputs.begin(), m_receiverInputs.end(), [txTransceiver](const CReceiverSampleProvider *p) {
65  return p->getId() == txTransceiver.id;
66  });
67 
68  if (it != m_receiverInputs.end()) { (*it)->setMute(true); }
69  }
70  }
71  }
72  else
73  {
74  for (CReceiverSampleProvider *receiverInput : std::as_const(m_receiverInputs))
75  {
76  receiverInput->setMute(false);
77  }
78  }

86  void CSoundcardSampleProvider::addOpusSamples(const IAudioDto &audioDto, const QVector<RxTransceiverDto> &rxTransceivers)
87  {
88  QVector<RxTransceiverDto> rxTransceiversFilteredAndSorted = rxTransceivers;
89 
90  rxTransceiversFilteredAndSorted.erase(std::remove_if(rxTransceiversFilteredAndSorted.begin(), rxTransceiversFilteredAndSorted.end(), [this](const RxTransceiverDto &r) {
91  return !m_receiverIDs.contains(r.id);
92  }),
93  rxTransceiversFilteredAndSorted.end());
94 
95  std::sort(rxTransceiversFilteredAndSorted.begin(), rxTransceiversFilteredAndSorted.end(), [](const RxTransceiverDto &a, const RxTransceiverDto &b) -> bool {
96  return a.distanceRatio > b.distanceRatio;
97  });
98 
99  if (!rxTransceiversFilteredAndSorted.isEmpty())
100  {
101  bool audioPlayed = false;
102  QVector<quint16> handledTransceiverIDs;
103  for (int i = 0; i < rxTransceiversFilteredAndSorted.size(); i++)
104  {
105  const RxTransceiverDto rxTransceiver = rxTransceiversFilteredAndSorted[i];
106  if (!handledTransceiverIDs.contains(rxTransceiver.id))
107  {
108  handledTransceiverIDs.push_back(rxTransceiver.id);
109 
110  CReceiverSampleProvider *receiverInput = nullptr;
111  auto it = std::find_if(m_receiverInputs.begin(), m_receiverInputs.end(), [rxTransceiver](const CReceiverSampleProvider *p) {
112  return p->getId() == rxTransceiver.id;
113  });
114 
115  if (it != m_receiverInputs.end())
116  {
117  receiverInput = *it;
118  }
119 
120  if (!receiverInput) { continue; }
121  if (receiverInput->getMute()) { continue; }
122 
123  if (!audioPlayed)
124  {
125  receiverInput->addOpusSamples(audioDto, rxTransceiver.frequency, rxTransceiver.distanceRatio);
126  audioPlayed = true;
127  }
128  else
129  {
130  receiverInput->addSilentSamples(audioDto, rxTransceiver.frequency, rxTransceiver.distanceRatio);
131  }
132 
133  // debug ONLY
134  if (CBuildConfig::isLocalDeveloperDebugBuild())
135  {
136  receiverInput->logVoiceInputs(QStringLiteral("Transceiver %1 ").arg(rxTransceiver.id), 1500);
137  }
138  }
139  } // each transceiver
140  } // filtered rx transceivers
141  }

143  void CSoundcardSampleProvider::updateRadioTransceivers(const QVector<TransceiverDto> &radioTransceivers)
144  {
145  for (const TransceiverDto &radioTransceiver : radioTransceivers)
146  {
147  auto it = std::find_if(m_receiverInputs.begin(), m_receiverInputs.end(), [radioTransceiver](const CReceiverSampleProvider *p) {
148  return p->getId() == radioTransceiver.id;
149  });
150 
151  if (it)
152  {
153  (*it)->setFrequency(radioTransceiver.frequencyHz);
154  }
155  }
156 
157  for (CReceiverSampleProvider *receiverInput : std::as_const(m_receiverInputs))
158  {
159  const quint16 transceiverID = receiverInput->getId();
160  const bool contains = std::any_of(radioTransceivers.cbegin(), radioTransceivers.cend(), [transceiverID](const auto &tx) { return transceiverID == tx.id; });
161  if (!contains)
162  {
163  receiverInput->setFrequency(0);
164  }
165  }
