204  double CLivery::getColorDistance(const CRgbColor &fuselage, const CRgbColor &tail) const
205  {
206  if (!fuselage.isValid() || !tail.isValid()) { return 1.0; }
207  if (this->getColorFuselage().isValid() && this->getColorTail().isValid())
208  {
209  if (this->matchesColors(fuselage, tail)) { return 0.0; } // avoid rounding
210  const double xDist = this->getColorFuselage().colorDistance(fuselage);
211  const double yDist = this->getColorTail().colorDistance(tail);
212  const double d = xDist * xDist + yDist * yDist;
213  return d / 2.0; // normalize to 0..1
214  }
215  else
216  {
217  return 1.0;
218  }
219  }

444  int CLivery::calculateScore(const CLivery &otherLivery, bool preferColorLiveries, CStatusMessageList *log) const
445  {
446  if (this->isDbEqual(otherLivery))
447  {
448  addLogDetailsToList(log, *this, QStringLiteral("Equal DB code: 100"));
449  return 100;
450  }
451 
452  // get a level
453  static const int sameAirlineIcaoLevel = CAirlineIcaoCode("DLH").calculateScore(CAirlineIcaoCode("DLH"));
454  Q_ASSERT_X(sameAirlineIcaoLevel == 60, Q_FUNC_INFO, "airline scoring changed");
455 
456  int score = 0;
457  const double colorMultiplier = 1.0 - this->getColorDistance(otherLivery);
458 
459  if (this->isColorLivery() && otherLivery.isColorLivery())
460  {
461  // 2 color liveries 25..85
462  score = 25;
463  score += 60 * colorMultiplier;
464  addLogDetailsToList(log, *this, QStringLiteral("2 color liveries, color multiplier %1: %2").arg(colorMultiplier).arg(score));
465  }
466  else if (this->isAirlineLivery() && otherLivery.isAirlineLivery())
467  {
468  // 2 airline liveries 0..85
469  // 0..50 based on ICAO
470  // 0..25 based on color distance
471  // 0..10 based on mil.flag
472  // same ICAO at least means 30, max 50
473  score = qRound(0.5 * this->getAirlineIcaoCode().calculateScore(otherLivery.getAirlineIcaoCode(), log));
474  score += 25 * colorMultiplier;
475  addLogDetailsToList(log, *this, QStringLiteral("2 airline liveries, color multiplier %1: %2").arg(colorMultiplier).arg(score));
476  if (this->isMilitary() == otherLivery.isMilitary())
477  {
478  addLogDetailsToList(log, *this, QStringLiteral("Mil.flag '%1' matches: %2").arg(boolToYesNo(this->isMilitary())).arg(score));
479  score += 10;
480  }
481  }
482  else if ((this->isColorLivery() && otherLivery.isAirlineLivery()) || (otherLivery.isColorLivery() && this->isAirlineLivery()))
483  {
484  // 1 airline, 1 color livery
485  // 0 .. 50
486  // 25 is weaker as same ICAO code / 2 from above
487  score = preferColorLiveries ? 25 : 0;
488  score += 25 * colorMultiplier; // needs to be the same as in 2 airlines
489  addLogDetailsToList(log, *this, QStringLiteral("Color/airline mixed, color multiplier %1: %2").arg(colorMultiplier).arg(score));
490  }
491  return score;
492  }
