181  QVariant ITimestampBased::propertyByIndex(CPropertyIndexRef index) const

186  switch (i)

189  case IndexMSecsSinceEpoch: return QVariant::fromValue(this->getMSecsSinceEpoch());
190  case IndexUtcTimestampFormattedDhms: return QVariant::fromValue(this->getFormattedUtcTimestampDhms());
191  case IndexUtcTimestampFormattedHm: return QVariant::fromValue(this->getFormattedUtcTimestampHm());
192  case IndexUtcTimestampFormattedHms: return QVariant::fromValue(this->getFormattedUtcTimestampHms());
193  case IndexUtcTimestampFormattedYmdhms: return QVariant::fromValue(this->getFormattedUtcTimestampYmdhms());

195  case IndexUtcTimestampFormattedMdhms: return QVariant::fromValue(this->getFormattedUtcTimestampMdhms());
196  case IndexUtcTimestampFormattedMdhmsz: return QVariant::fromValue(this->getFormattedUtcTimestampMdhmsz());
197  default: break;
198  }
199  }
200  const QString m = QStringLiteral("Cannot handle index %1").arg(index.toQString());
201  BLACK_VERIFY_X(false, Q_FUNC_INFO, qUtf8Printable(m));

205  void ITimestampBased::setPropertyByIndex(CPropertyIndexRef index, const QVariant &variant)
206  {

216  case IndexUtcTimestampFormattedHm:
217  case IndexUtcTimestampFormattedHms:

219  {

221  m_timestampMSecsSinceEpoch = dt.toMSecsSinceEpoch();
222  }
223  return;
224  default: break;
225  }
226  }

228  BLACK_VERIFY_X(false, Q_FUNC_INFO, qUtf8Printable(m));
