85  void CRemoteAircraftSelector::fillComboBox()
86  {
87  if (!this->isVisible()) { return; } // for performance reasons
88  if (!sGui || sGui->isShuttingDown()) { return; }
89 
90  const CCallsign currentSelection(this->getSelectedCallsign());
91  m_aircraft = sGui->getIContextNetwork()->getAircraftInRange().sortedByCallsign();
92  ui->cb_RemoteAircraftSelector->clear();
93  if (m_aircraft.isEmpty()) { return; }
94 
95  QStringList items;
96  for (const CSimulatedAircraft &aircraft : m_aircraft)
97  {
98  if (aircraft.getCallsign().isEmpty()) { continue; }
99  QString i(aircraft.getCallsign().toQString());
100  if (aircraft.hasAircraftDesignator())
101  {
102  i += u" (" %
103  aircraft.getAircraftIcaoCode().toQString(false) %
104  u')';
105  }
106  if (aircraft.hasRealName())
107  {
108  i += u" - " % aircraft.getPilotRealName();
109  }
110  if (m_showPartsEnabled)
111  {
112  if (aircraft.isPartsSynchronized())
113  {
114  i += QStringLiteral(" [parts]");
115  }
116  }
117  items.append(i);
118  }
119 
120  // new combobox
121  ui->cb_RemoteAircraftSelector->addItems(items);
122 
123  // set old selection if possible
124  if (currentSelection.isEmpty()) { return; }
125  int index = m_aircraft.firstIndexOfCallsign(currentSelection);
126  if (index >= 0 && index < ui->cb_RemoteAircraftSelector->count())
127  {
128  ui->cb_RemoteAircraftSelector->setCurrentIndex(index);
129  }
130  }
