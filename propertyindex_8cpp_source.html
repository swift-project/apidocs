72  void CPropertyIndex::parseFromString(const QString &indexes)
73  {
74  m_indexes.clear();
75  if (indexes.isEmpty()) { return; }
76  for (const QStringRef &index : indexes.splitRef(';'))
77  {
78  if (index.isEmpty()) { continue; }
79  bool ok;
80  int i = index.toInt(&ok);
81  Q_ASSERT(ok);
82  Q_ASSERT(i >= static_cast<int>(CPropertyIndexRef::GlobalIndexCValueObject));
83  m_indexes.append(i);
84  }
85  }
