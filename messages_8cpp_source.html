25  void CMessageBox::draw()
26  {
27  static const int lineHeight = CMessageBox::lineHeight();
28  static const int lineSpace = lineHeight / 3;
29 
30  const int messageCount = static_cast<int>(m_messages.size());
31 
32  const int screenHeight = m_screenHeight.get();
33  const int boxRight = m_screenWidth.get() - m_boxRight;
34  const int boxLeft = m_boxLeft;
35  int boxTop;
36  int boxBottom;
37 
38  if (m_boxTop >= 0)
39  {
40  boxTop = screenHeight - m_boxTop;
41  boxBottom = boxTop - lineSpace * 2 - (lineHeight + lineSpace) * messageCount;
42  }
43  else
44  {
45  boxBottom = m_boxBottom >= 0 ? m_boxBottom : 20;
46  boxTop = boxBottom + lineSpace * 2 + (lineHeight + lineSpace) * messageCount;
47  }
48 
49  if (boxTop > screenHeight) { boxTop = screenHeight; }
50  if (boxBottom <= 0) { boxBottom = 0; }
51 
52  XPLMDrawTranslucentDarkBox(boxLeft, boxTop, boxRight, boxBottom);
53 
54  static int arrowWidth = 0, arrowHeight = 0;
55  if (!arrowHeight)
56  {
57  XPGetElementDefaultDimensions(xpElement_LittleUpArrow, &arrowWidth, &arrowHeight, nullptr);
58  }
59 
60  static const int x = boxLeft + lineSpace;
61  if (m_upArrow)
62  {
63  const int y = boxTop - lineSpace - arrowHeight;
64  XPDrawElement(x, y, x + arrowWidth, y + arrowHeight, xpElement_LittleUpArrow, 0);
65  }
66  if (m_downArrow)
67  {
68  const int y = boxTop - (lineHeight + lineSpace) * messageCount;
69  XPDrawElement(x, y, x + arrowWidth, y + arrowHeight, xpElement_LittleDownArrow, 0);
70  }
71  for (int i = 0; i < messageCount; ++i)
72  {
73  const int y = boxTop - (lineHeight + lineSpace) * (i + 1);
74  const size_t ii = static_cast<size_t>(i);
75  XPLMDrawString(m_messages[ii].m_rgb.data(), x + arrowWidth + arrowWidth / 2, y,
76  const_cast<char *>(reinterpret_cast<const char *>(m_messages[ii].m_text.c_str())), nullptr, xplmFont_Basic);
77  }
78  }
