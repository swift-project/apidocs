46  QString CAngleUnit::makeRoundedQStringWithUnit(double value, int digits, bool withGroupSeparator, bool i18n) const
47  {
48  if (digits < 0) { digits = this->getDisplayDigits(); }
49  QString s;
50  if ((*this) == CAngleUnit::sexagesimalDeg())
51  {
52  digits -= 4;
53  Q_ASSERT(digits >= 0);
54  double de = CMathUtils::trunc(value);
55  double mi = CMathUtils::trunc((value - de) * 100.0);
56  double se = CMathUtils::trunc((value - de - mi / 100.0) * 1000000) / 100.0;
57  const char *fmt = value < 0 ? "-%1 %2 %3" : "%1 %2 %3";
58  s = i18n ? QCoreApplication::translate("CMeasurementUnit", fmt) : fmt;
59  s = s.arg(fabs(de), 0, 'f', 0).arg(fabs(mi), 2, 'f', 0, '0').arg(fabs(se), 3 + digits, 'f', digits, '0');
60  }
61  else if ((*this) == CAngleUnit::sexagesimalDegMin())
62  {
63  digits -= 2;
64  Q_ASSERT(digits >= 0);
65  double de = CMathUtils::trunc(value);
66  double mi = CMathUtils::trunc((value - de) * 100.0);
67  const char *fmt = value < 0 ? "-%1 %2" : "%1 %2";
68  s = i18n ? QCoreApplication::translate("CMeasurementUnit", fmt) : fmt;
69  s = s.arg(fabs(de), 0, 'f', 0).arg(fabs(mi), 3 + digits, 'f', digits, '0');
70  }
71  else
72  {
73  s = this->CMeasurementUnit::makeRoundedQStringWithUnit(value, digits, withGroupSeparator, i18n);
74  }
75  return s;
