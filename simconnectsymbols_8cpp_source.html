176 P3DSimConnectVersion stringToP3DVersion(const QString &p3d)
177 {
178  if (p3d.length() >= 2)
179  {
180  const QString p = digitOnlyString(p3d);
181  if (p.contains("40")) { return P3DSimConnectv40; }
182  if (p.contains("41")) { return P3DSimConnectv41; }
183  if (p.contains("42")) { return P3DSimConnectv42; }
184  if (p.contains("43")) { return P3DSimConnectv43; }
185  if (p.contains("44")) { return P3DSimConnectv44; }
186  if (p.contains("45")) { return P3DSimConnectv45; }
187 
188  if (p.contains("50")) { return P3DSimConnectv45; }
189  if (p.contains("51")) { return P3DSimConnectv45; }
190  if (p.contains("52")) { return P3DSimConnectv45; }
191  }
192  return P3DSimConnectv44; // default
193 }

195 bool loadAndResolveP3DSimConnect(P3DSimConnectVersion version)
196 {
197  // Check if already loaded
198  if (gSymbols.SimConnect_Open) { return true; }
199 
200  QString simConnectFileName(QStringLiteral("SimConnect.P3D-"));
201 
202  switch (version)
203  {
204  case P3DSimConnectv40: simConnectFileName += "v4.0"; break;
205  case P3DSimConnectv41: simConnectFileName += "v4.1"; break;
206  case P3DSimConnectv42: simConnectFileName += "v4.2"; break;
207  case P3DSimConnectv43: simConnectFileName += "v4.3"; break;
208  case P3DSimConnectv44: simConnectFileName += "v4.4"; break;
209  case P3DSimConnectv45: simConnectFileName += "v4.5"; break;
210  }
211 
212  QLibrary simConnectDll(simConnectFileName);
213  simConnectDll.setLoadHints(QLibrary::PreventUnloadHint);
214  if (simConnectDll.load())
215  {
216  const bool resolvedCommon = resolveCommonSimConnectSymbols(simConnectDll);
217  const bool resolvedP3DSimConnectSymbols = resolveP3DSimConnectSymbols(simConnectDll);
218  if (!resolvedCommon)
219  {
220  CLogMessage(CLogCategories::driver()).error(u"Failed to resolve common symbols from SimConnect.dll: '%1'") << simConnectFileName;
221  return false;
222  }
223  if (!resolvedP3DSimConnectSymbols)
224  {
225  CLogMessage(CLogCategories::driver()).error(u"Failed to resolve P3D symbols from SimConnect.dll: '%1'") << simConnectFileName;
226  return false;
227  }
228 
229  CLogMessage(CLogCategories::driver()).info(u"Loaded and resolved P3D symbols from SimConnect.dll: '%1'") << simConnectFileName;
230  return resolvedCommon && resolvedP3DSimConnectSymbols;
231  }
232  else
233  {
234  CLogMessage(CLogCategories::driver()).error(u"Failed to load SimConnect.dll: '%1' '%2'") << simConnectFileName << simConnectDll.errorString();
235  return false;
236  }
237 }

268 bool loadAndResolveFsxSimConnect(bool manifestProbing)
269 {
270  // Check if already loaded
271  if (gSymbols.SimConnect_Open) { return true; }
272 
273  QLibrary simConnectDll(QStringLiteral("SimConnect"));
274  simConnectDll.setLoadHints(QLibrary::PreventUnloadHint);
275  if (manifestProbing)
276  {
277  HMODULE hInst = nullptr;
278  GetModuleHandleEx(GET_MODULE_HANDLE_EX_FLAG_FROM_ADDRESS, (LPCTSTR)loadAndResolveSimConnect, &hInst);
279  wchar_t szModuleName[MAX_PATH];
280  GetModuleFileName(hInst, szModuleName, MAX_PATH);
281 
282  // 101 => "SimConnect_RTM.manifest"
283  // 102 => "SimConnect_SP1.manifest"
284  // 103 => "SimConnect_XPack.manifest"
285  // Use only SP1 and XPack, since RTM is missing two important symbols.
286  // Try the latest one first.
287  std::array<WORD, 2> resourceNumbers = { { 103U, 102U } };
288 
289  for (const auto resourceNumber : resourceNumbers)
290  {
291  ACTCTX actCtx;
292  memset(&actCtx, 0, sizeof(ACTCTX));
293  actCtx.cbSize = sizeof(actCtx);
294  actCtx.dwFlags = ACTCTX_FLAG_RESOURCE_NAME_VALID;
295  actCtx.lpSource = szModuleName;
296  actCtx.lpResourceName = MAKEINTRESOURCE(resourceNumber);
297 
298  HANDLE hActCtx;
299  hActCtx = CreateActCtx(&actCtx);
300  if (hActCtx != INVALID_HANDLE_VALUE)
301  {
302  ULONG_PTR lpCookie = 0;
303  if (ActivateActCtx(hActCtx, &lpCookie))
304  {
305  simConnectDll.load();
306  DeactivateActCtx(0, lpCookie);
307  if (simConnectDll.isLoaded()) { break; }
308  }
309  }
310  ReleaseActCtx(hActCtx);
311  }
312  }
313 
314  // If at that stage, no SimConnect library was found in the WinSxS folder, try once without activation context to link to the one we
315  // ship ourselves.
316  if (!simConnectDll.isLoaded())
317  {
318  simConnectDll.load();
319  }
320 
321  if (simConnectDll.isLoaded())
322  {
323  return resolveCommonSimConnectSymbols(simConnectDll);
324  }
325  else
326  {
327  CLogMessage(CLogCategories::driver()).error(u"Failed to load SimConnect.dll: %1") << simConnectDll.errorString();
328  return false;
329  }
330 }
