96  int CMathUtils::roundToMultipleOf(int value, int divisor)
97  {
98  Q_ASSERT(divisor != 0);
99  Q_ASSERT(divisor >= -std::numeric_limits<int>::max());
100  divisor = std::abs(divisor);
101  Q_ASSERT(std::abs(value) < std::numeric_limits<int>::max() - divisor / 2);
102 
103  int multiplier = value / divisor;
104  int remainder = std::abs(value % divisor);
105  int shortfall = divisor - remainder;
106 
107  if (shortfall < remainder) { multiplier += value < 0 ? -1 : 1; }
108 
109  return multiplier * divisor;
110  }

112  QString CMathUtils::fractionalPartAsString(double value, int width)
113  {
114  double intpart;
115  const double fractpart = modf(value, &intpart);
116  const int prec = width >= 0 ? width + 1 : 10;
117  const QString f = QString::number(fractpart, 'f', prec); // avoid scientific notation
118  const QString fInt = f.length() < 3 ? QString("0") : f.mid(2);
119  if (width < 0) { return fInt; }
120  if (fInt.length() >= width) { return fInt.left(width); }
121  return fInt.leftJustified(width, '0');
122  }
