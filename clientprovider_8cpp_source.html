79  int CClientProvider::updateOrAddClient(const CCallsign &callsign, const CPropertyIndexVariantMap &vm, bool skipEqualValues)
80  {
81  BLACK_VERIFY_X(!callsign.isEmpty(), Q_FUNC_INFO, "Missing client callsign");
82  if (callsign.isEmpty()) { return 0; }
83  int c = 0;
84  if (this->hasClientInfo(callsign))
85  {
86  QWriteLocker l(&m_lockClient);
87  CClient &client = m_clients[callsign];
88  c = client.apply(vm, skipEqualValues).size();
89  }
90  else
91  {
92  CClient client(callsign);
93  c = client.apply(vm).size();
94  QWriteLocker l(&m_lockClient);
95  m_clients[callsign] = client;
96  }
97  return c;
98  }

119  bool CClientProvider::setClientGndCapability(const CCallsign &callsign, bool supportGndFlag)
120  {
121  BLACK_VERIFY_X(!callsign.isEmpty(), Q_FUNC_INFO, "Missing client callsign");
122  if (callsign.isEmpty()) { return 0; }
123 
124  CClient client = this->getClientOrDefaultForCallsign(callsign);
125 
126  // need to change?
127  if (!client.isValid()) { return false; } // no client
128  if (client.hasGndFlagCapability() == supportGndFlag) { return true; } // already set, but set
129  if (supportGndFlag)
130  {
131  client.addCapability(CClient::FsdWithGroundFlag);
132  }
133  else
134  {
135  client.removeCapability(CClient::FsdWithGroundFlag);
136  }
137  QWriteLocker l(&m_lockClient);
138  m_clients[callsign] = client;
139  return true;
