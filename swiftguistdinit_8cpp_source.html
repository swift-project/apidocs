60 void SwiftGuiStd::init()
61 {
62  // POST(!) GUI init
63  Q_ASSERT_X(sGui, Q_FUNC_INFO, "Missing sGui");
64  Q_ASSERT_X(sGui->getWebDataServices(), Q_FUNC_INFO, "Missing web services");
65  Q_ASSERT_X(sGui->supportsContexts(), Q_FUNC_INFO, "Missing contexts");
66 
67  if (m_init) { return; }
68 
69  ui->dw_InfoBarStatus->initialFloating();
70 
71  this->setVisible(false); // hide all, so no flashing windows during init
72  m_mwaStatusBar = &m_statusBar;
73  m_mwaOverlayFrame = ui->fr_CentralFrameInside;
74  m_mwaLogComponent = ui->comp_MainInfoArea->getLogComponent();
75  sGui->initMainApplicationWidget(this);
76 
77  // experimental: avoid mic flickering
78  // disabled for 5.14.1
79  // if (CBuildConfig::isLocalDeveloperDebugBuild()) { BlackSound::occupyAudioInputDevice(); }
80 
81  // log messages
82  m_logHistoryForStatus.setFilter(CLogPattern().withSeverityAtOrAbove(CStatusMessage::SeverityInfo));
83  m_logHistoryForOverlay.setFilter(CLogPattern().withSeverityAtOrAbove(CStatusMessage::SeverityError));
84  connect(&m_logHistoryForStatus, &CLogHistoryReplica::elementAdded, this, [this](const CStatusMessage &message) {
85  m_statusBar.displayStatusMessage(message);
86  ui->comp_MainInfoArea->displayStatusMessage(message);
87  });
88  connect(&m_logHistoryForOverlay, &CLogHistoryReplica::elementAdded, this, [this](const CStatusMessage &message) {
90  if (!message.getCategories().contains(CLogCategories::validation())) { ui->fr_CentralFrameInside->showOverlayMessage(message); }
91  });
92  m_logHistoryForStatus.initialize(sApp->getDataLinkDBus());
93  m_logHistoryForOverlay.initialize(sApp->getDataLinkDBus());
94 
95  // style
96  this->initStyleSheet();
97 
98  // with frameless window, we shift menu and statusbar into central widget
99  // http://stackoverflow.com/questions/18316710/frameless-and-transparent-window-qt5
100  if (this->isFrameless())
101  {
102  // wrap menu in layout, add button to menu bar and insert on top
103  QHBoxLayout *menuBarLayout = this->addFramelessCloseButton(ui->mb_MainMenuBar);
104  ui->vl_CentralWidgetOutside->insertLayout(0, menuBarLayout, 0);
105 
106  // now insert the dock widget info bar into the widget
107  ui->vl_CentralWidgetOutside->insertWidget(1, ui->dw_InfoBarStatus);
108 
109  // move the status bar into the frame
110  // (otherwise it is dangling outside the frame as it belongs to the window)
111  ui->sb_MainStatusBar->setParent(ui->wi_CentralWidgetOutside);
112  ui->vl_CentralWidgetOutside->addWidget(ui->sb_MainStatusBar, 0);
113 
114  // grip
115  this->addFramelessSizeGripToStatusBar(ui->sb_MainStatusBar);
116  }
117 
118  // timers
119  m_timerContextWatchdog.setObjectName(this->objectName().append(":m_timerContextWatchdog"));
120 
121  // info bar and status bar
122  m_statusBar.initStatusBar(ui->sb_MainStatusBar);
123  ui->dw_InfoBarStatus->allowStatusBar(false);
124  ui->dw_InfoBarStatus->setPreferredSizeWhenFloating(ui->dw_InfoBarStatus->size()); // set floating size
125 
126  // navigator
127  m_navigator->addAction(this->getToggleWindowVisibilityAction(m_navigator.data()));
128  m_navigator->addActions(ui->comp_MainInfoArea->getInfoAreaToggleFloatingActions(m_navigator.data())); // here we add the actions for the main windows
129  m_navigator->addAction(this->getWindowNormalAction(m_navigator.data()));
130  m_navigator->addAction(this->getWindowMinimizeAction(m_navigator.data()));
131  m_navigator->addAction(this->getToggleStayOnTopAction(m_navigator.data()));
132  m_navigator->buildNavigator(1);
133 
134  // wire GUI signals
135  this->initGuiSignals();
136 
137  // signal / slots contexts / timers
138  Q_ASSERT_X(sGui->getIContextNetwork(), Q_FUNC_INFO, "Missing network context");
139  Q_ASSERT_X(sGui->getIContextSimulator(), Q_FUNC_INFO, "Missing simulator context");
140 
141  bool s = connect(sGui->getIContextNetwork(), &IContextNetwork::connectionStatusChanged, this, &SwiftGuiStd::onConnectionStatusChanged, Qt::QueuedConnection);
142  Q_ASSERT(s);
143  s = connect(sGui->getIContextNetwork(), &IContextNetwork::kicked, this, &SwiftGuiStd::onKickedFromNetwork, Qt::QueuedConnection);
144  Q_ASSERT(s);
145  s = connect(sGui->getIContextSimulator(), &IContextSimulator::requestUiConsoleMessage, this, &SwiftGuiStd::onRequestedConsoleMessage, Qt::QueuedConnection);
146  Q_ASSERT(s);
147  s = connect(sGui->getIContextSimulator(), &IContextSimulator::validatedModelSet, this, &SwiftGuiStd::onValidatedModelSet, Qt::QueuedConnection);
148  Q_ASSERT(s);
149  s = connect(&m_timerContextWatchdog, &QTimer::timeout, this, &SwiftGuiStd::handleTimerBasedUpdates);
150  Q_ASSERT(s);
151 
152  if (sGui->getIContextAudio())
153  {
154  s = connect(sGui->getIContextAudio(), &IContextAudio::voiceClientFailure, this, &SwiftGuiStd::onAudioClientFailure, Qt::QueuedConnection);
155  Q_ASSERT(s);
156  }
157  Q_UNUSED(s)
158 
159  // check if DB data have been loaded
160  // only check once, so data can be loaded and
161  connectOnce(sGui->getWebDataServices(), &CWebDataServices::sharedInfoObjectsRead, this, &SwiftGuiStd::checkDbDataLoaded, Qt::QueuedConnection);
162 
163  // start timers, update timers will be started when network is connected
164  m_timerContextWatchdog.start(2500);
165 
166  // init availability
167  this->setContextAvailability();
168 
169  // data
170  this->initialContextDataReads();
171 
172  // start screen and complete menu
173  this->setMainPageToInfoArea();
174  this->initMenus();
175 
176  // info
177  ui->comp_MainInfoArea->getLogComponent()->appendPlainTextToConsole(sGui->swiftVersionString());
178  ui->comp_MainInfoArea->getLogComponent()->appendPlainTextToConsole(CBuildConfig::compiledWithInfo());
179  connect(ui->comp_InfoBarStatus, &CInfoBarStatusComponent::transponderModeChanged, ui->dw_InfoBarStatus, &CDockWidgetInfoBar::reloadStyleSheet, Qt::QueuedConnection);
180 
181  // Show kill button
182  ui->fr_CentralFrameInside->showKillButton(true);
183 
184  // do this as last statement, so it can be used as flag
185  // whether init has been completed
186  this->setVisible(true);
187 
188  // more checks
189  QPointer<SwiftGuiStd> myself(this);
190  QTimer::singleShot(5000, this, [=] {
191  if (!myself) { return; }
192  this->verifyPrerequisites();
193  });
194 
195  // trigger version check
196  sGui->triggerNewVersionCheck(10 * 1000);
197 
198  // done
199  emit sGui->startUpCompleted(true);
200  m_init = true;
201 }
