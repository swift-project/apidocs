135  void CSimulatorSelector::setToConnectedSimulator(bool makeReadOnly)
136  {
137  if (sGui && sGui->supportsContexts() && sGui->getIContextSimulator())
138  {
139  const CSimulatorPluginInfo pluginInfo = sGui->getIContextSimulator()->getSimulatorPluginInfo();
140  if (!this->isSingleSelection()) { this->setMode(RadioButtons); } // only one sim can be connected
141 
142  if (pluginInfo.isValid())
143  {
144  this->setReadOnly(makeReadOnly);
145  this->setValue(pluginInfo.getSimulator());
146  }
147  else
148  {
149  if (makeReadOnly) { this->setReadOnly(false); }
150  const CSimulatorInfo simulator = sGui->getIContextSimulator()->getModelSetLoaderSimulator();
151  if (simulator.isSingleSimulator())
152  {
153  this->setValue(simulator);
154  }
155  }
156  }
157  else
158  {
159  if (makeReadOnly) { this->setReadOnly(false); }
160  }
161  }

163  void CSimulatorSelector::setToConnectedSimulator(int deferredMs, bool makeReadOnly)
164  {
165  if (deferredMs < 1)
166  {
167  this->setToConnectedSimulator(makeReadOnly);
168  return;
169  }
170 
171  if (!sGui || sGui->isShuttingDown()) { return; }
172  QPointer<CSimulatorSelector> myself(this);
173  QTimer::singleShot(deferredMs, this, [=] {
174  if (!sGui || sGui->isShuttingDown() || !myself) { return; }
175  this->setToConnectedSimulator(makeReadOnly);
176  });
177  }

220  bool CSimulatorSelector::isUnselected() const
221  {
222  bool c = false;
223  switch (m_mode)
224  {
225  default:
226  case CheckBoxes:
227  c = ui->cb_FSX->isChecked() || ui->cb_FS9->isChecked() || ui->cb_XPlane->isChecked() || ui->cb_P3D->isChecked() || ui->cb_FG->isChecked();
228  break;
229  case RadioButtons:
230  c = ui->rb_FSX->isChecked() || ui->rb_FS9->isChecked() || ui->rb_XPlane->isChecked() || ui->rb_P3D->isChecked() || ui->rb_FG->isChecked();
231  break;
232  case ComboBox:
233  const int i = ui->cb_Simulators->currentIndex();
234  c = i < 0;
235  break;
236  }
237  return !c;
