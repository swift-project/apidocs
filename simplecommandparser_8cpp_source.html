25  void CSimpleCommandParser::parse(const QString &commandLine)
26  {
27  m_knownCommand = false;
28  m_originalLine = commandLine;
29  m_cleanedLine = commandLine.trimmed().simplified();
30  if (!m_cleanedLine.isEmpty())
31  {
32  m_splitParts = m_cleanedLine.split(' ');
33  if (!m_splitParts.isEmpty())
34  {
35  const QString &first = m_splitParts.constFirst();
36  const QString formatted = formatCommand(first);
37  if (isCommand(first))
38  {
39  m_commandPart = formatted;
40  m_knownCommand = m_knownCommands.contains(formatted);
41  }
42  }
43  }

149  QString CSimpleCommandParser::commandsHtmlHelp()
150  {
151  if (s_commands.isEmpty()) { return {}; }
152 
153  static const QString html("<table style=\"font-size: 8pt; white-space: nowrap;\">\n%1\n</table>");
154  static const QString row("<td>%1</td><td>%2</td>");
155  static const QString rowHeader("<th>%1</th><th>%2</th><th>%3</th><th>%4</th>");
156 
157  QString rows;
158  QList<CommandHtmlHelp> cmds(s_commands);
159  std::sort(cmds.begin(), cmds.end(), CommandHtmlHelp::less);
160 
161  static const QString cmdCol(QString().fill('-', 20));
162  static const QString textCol(QString().fill('-', 40));
163  rows += u"<tr>" % rowHeader.arg("command", "synopsis", "command", "synopsis") % u"</tr>\n" %
164  u"<tr>" % rowHeader.arg(cmdCol, textCol, cmdCol, textCol) % u"</tr>\n";
165 
166  for (int i = 0; i < cmds.size(); i++)
167  {
168  CommandHtmlHelp help = cmds[i];
169  rows += u"<tr>" % row.arg(help.command, help.help);
170  i++;
171  if (i < cmds.size())
172  {
173  help = cmds[i];
174  rows += row.arg(help.command, help.help);
175  }
176  else
177  {
178  rows += row.arg("", "");
179  }
180  rows += "</tr>\n";
181  }
182  return html.arg(rows);
183  }
