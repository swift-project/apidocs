104  void CDbOwnModelSetComponent::setModelSet(const CAircraftModelList &models, const CSimulatorInfo &simulator)
105  {
106  Q_ASSERT_X(simulator.isSingleSimulator(), Q_FUNC_INFO, "Need single simulator");
107  this->setSimulator(simulator);
108  if (models.isEmpty())
109  {
110  ui->tvp_OwnModelSet->clear();
111  return;
112  }
113 
114  CAircraftModelList cleanModelList(models.matchesSimulator(simulator)); // remove those not matching the simulator
115  const int diff = models.size() - cleanModelList.size();
116  if (diff > 0)
117  {
118  const CStatusMessage m = CStatusMessage(this).warning(u"Removed %1 models from set, because not matching %2") << diff << simulator.toQString(true);
119  this->showMappingComponentOverlayHtmlMessage(m, 5000);
120  }
121  cleanModelList.resetOrder();
122  ui->tvp_OwnModelSet->updateContainerMaybeAsync(cleanModelList);
123  }

125  int CDbOwnModelSetComponent::replaceOrAddModelSet(const CAircraftModelList &models, const CSimulatorInfo &simulator)
126  {
127  Q_ASSERT_X(simulator.isSingleSimulator(), Q_FUNC_INFO, "Need single simulator");
128  if (models.isEmpty()) { return 0; }
129  CAircraftModelList cleanModelList(models.matchesSimulator(simulator)); // remove those not matching the simulator
130  const int diff = models.size() - cleanModelList.size();
131  if (diff > 0)
132  {
133  const CStatusMessage m = CStatusMessage(this).warning(u"Removed %1 models from set, because not matching %2") << diff << simulator.toQString(true);
134  this->showMappingComponentOverlayHtmlMessage(m, 5000);
135  }
136  if (cleanModelList.isEmpty()) { return 0; }
137  CAircraftModelList updatedModels(ui->tvp_OwnModelSet->container());
138  updatedModels.replaceOrAddModelsWithString(cleanModelList, Qt::CaseInsensitive);
139  updatedModels.resetOrder();
140  ui->tvp_OwnModelSet->updateContainerMaybeAsync(updatedModels);
141  return diff;
142  }

159  CStatusMessage CDbOwnModelSetComponent::addToModelSet(const CAircraftModelList &models, const CSimulatorInfo &simulator)
160  {
161  Q_ASSERT_X(simulator.isSingleSimulator(), Q_FUNC_INFO, "Need single simulator");
162  if (models.isEmpty()) { return CStatusMessage(this, CStatusMessage::SeverityInfo, u"No data", true); }
163  if (!this->getModelSetSimulator().isSingleSimulator())
164  {
165  // no sim yet, we set it
166  this->setSimulator(simulator);
167  }
168  if (simulator != this->getModelSetSimulator())
169  {
170  // only currently selected sim allowed
171  return CStatusMessage(this, CStatusMessage::SeverityError,
172  u"Cannot add data for " % simulator.toQString(true) % u" to " % this->getModelSetSimulator().toQString(true), true);
173  }
174 
175  const bool allowExcludedModels = m_modelSettings.get().getAllowExcludedModels();
176  CAircraftModelList updateModels(this->getModelSetFromView());
177  int d = updateModels.replaceOrAddModelsWithString(models, Qt::CaseInsensitive);
178  if (d > 0)
179  {
180  if (!allowExcludedModels) { updateModels.removeIfExcluded(); }
181  updateModels.resetOrder();
182  ui->tvp_OwnModelSet->updateContainerMaybeAsync(updateModels);
183  return CStatusMessage(this, CStatusMessage::SeverityInfo, u"Modified " % QString::number(d) % u" entries in model set " % this->getModelSetSimulator().toQString(true), true);
184  }
185  else
186  {
187  return CStatusMessage(this, CStatusMessage::SeverityInfo, u"No data modified in model set", true);
188  }
189  }

228  if (sender == ui->pb_SaveAsSetForSimulator)

240  // display either as overlay of componet or view
241  if (!this->showMappingComponentOverlayHtmlMessage(msg, 5000))
242  {
243  ui->tvp_OwnModelSet->showOverlayHTMLMessage(msg, 5000);
244  }
245  }
246  else
247  {
248  if (!this->showMappingComponentOverlayMessage(m))
249  {
250  ui->tvp_OwnModelSet->showOverlayMessage(m);
251  }
252  }
253  }
254  return;
255  }
256 
257  if (sender == ui->pb_ShowStatistics)
258  {
259  this->showModelStatistics();
260  return;
261  }
262 
263  if (sender == ui->pb_FirstSet)
264  {
265  this->firstSet();
266  return;
267  }
268 
269  if (sender == ui->pb_CopyFromAnotherSwift)
270  {
271  this->copyFromAnotherSwift();
272  return;

319  void CDbOwnModelSetComponent::reduceModels()
320  {
321  if (!m_reduceModelsDialog)
322  {
323  m_reduceModelsDialog.reset(new CDbReduceModelDuplicates(this));
324  }
325 
326  CAircraftModelList models = ui->tvp_OwnModelSet->containerOrFilteredContainer();
327  const CSimulatorInfo simulator = this->getModelSetSimulator();
328  m_reduceModelsDialog->setModels(models, simulator);
329  const QDialog::DialogCode ret = static_cast<QDialog::DialogCode>(m_reduceModelsDialog->exec());
330  if (ret != QDialog::Accepted) { return; }
331  const CAircraftModelList removeModels = m_reduceModelsDialog->getRemoveCandidates();
332  const CSimulatorInfo removeSimulator = m_reduceModelsDialog->getSimulator();
333  if (removeModels.isEmpty()) { return; }
334  const QStringList modelStrings = removeModels.getModelStringList(false);
335  models.removeModelsWithString(modelStrings, Qt::CaseInsensitive); // by strings also removes if id is missing
336  this->setModelSet(models, removeSimulator);
337  }

376  void CDbOwnModelSetComponent::createNewSet()
377  {
378  // make sure both tabs display the same simulator
379  // since we use the component also in the launcher wizard, mc might not be existing
380  const CSimulatorInfo simulator(this->getModelSetSimulator());
381  if (!simulator.isSingleSimulator())
382  {
383  // Ref T630, avoid invalid simulator
384  CLogMessage(this).error(u"No single simulator!");
385  return;
386  }
387 
388  CDbMappingComponent *mc = this->getMappingComponent();
389  if (!mc)
390  {
391  CLogMessage(this).error(u"No mapping component available!");
392  return;
393  }
394 
395  mc->setOwnModelsSimulator(simulator);
396  if (!m_modelSetFormDialog)
397  {
398  m_modelSetFormDialog.reset(new CDbOwnModelSetFormDialog(this));
399  m_modelSetFormDialog->setMappingComponent(this->getMappingComponent());
400  }
401 
402  if (mc->getOwnModelsCount() > 0)
403  {
404  m_modelSetFormDialog->setModal(true);
405  m_modelSetFormDialog->reloadData();
406  const QDialog::DialogCode rc = static_cast<QDialog::DialogCode>(m_modelSetFormDialog->exec());
407  if (rc == QDialog::Accepted)
408  {
409  this->setModelSet(m_modelSetFormDialog->getModelSet(), m_modelSetFormDialog->getSimulatorInfo());
410  }
411  }
412  else
413  {
414  const CStatusMessage m = CStatusMessage(this).error(u"No model data for %1") << simulator.toQString(true);
415  mc->showOverlayMessage(m);
416  }
417  }

441  Q_ASSERT_X(simulator.isSingleSimulator(), Q_FUNC_INFO, "Need single simulator");
