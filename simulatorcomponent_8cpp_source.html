103  void CSimulatorComponent::update()
104  {
105  if (!this->isVisibleWidget()) return; // no updates on invisible widgets
106  if (!sGui || sGui->isShuttingDown() || !sGui->getIContextOwnAircraft()) return;
107 
108  const ISimulator::SimulatorStatus simulatorStatus = static_cast<ISimulator::SimulatorStatus>(sGui->getIContextSimulator()->getSimulatorStatus());
109  if (simulatorStatus == ISimulator::Unspecified || simulatorStatus == ISimulator::Disconnected)
110  {
111  static const QString s("No simulator available");
112  this->addOrUpdateLiveDataByName(QStringLiteral("info"), s, CIcons::StandardIconWarning16);
113  return;
114  }
115 
116  if (!simulatorStatus.testFlag(ISimulator::Simulating))
117  {
118  static const QString s("Simulator (%1) not yet running");
119  this->addOrUpdateLiveDataByName(QStringLiteral("info"), s.arg(sGui->getIContextSimulator()->getSimulatorPluginInfo().getSimulator()), CIcons::StandardIconWarning16);
120  return;
121  }
122 
123  // clear old warnings / information
124  if (this->rowCount() < 5)
125  {
126  this->clear(true);
127  }
128 
129  const CSimulatedAircraft ownAircraft = sGui->getIContextOwnAircraft()->getOwnAircraft();
130  const CAircraftSituation s = ownAircraft.getSituation();
131  const CComSystem c1 = ownAircraft.getCom1System();
132  const CComSystem c2 = ownAircraft.getCom2System();
133  static const CIcon iconAlt(s.getAltitude().toIcon()); // minor performance improvement
134  static const CIcon iconLatLng(s.latitude().toIcon());
135  static const CIcon iconRadio(CIcon::iconByIndex(CIcons::StandardIconRadio16));
136  static const CIcon iconAttitude(CIcon::iconByIndex(CIcons::AviationAttitudeIndicator));
137  static const CIcon iconPlane(CIcon::iconByIndex(CIcons::StandardIconPaperPlane16));
138 
139  if (m_simulator.isAnySimulator())
140  {
141  this->addOrUpdateLiveDataByName("simulator", m_simulator.toQString(true), m_simulator.toIcon());
142 
143  if (sGui->getISimulator())
144  {
145  const double fps = sGui->getISimulator()->getAverageFPS();
146  this->addOrUpdateLiveDataByName(QStringLiteral("FPS"), fps < 0 ? QStringLiteral("N/A") : QString::number(fps, 'f', 1), CIcon(CIcons::ApplicationSimulator));
147 
148  const double ratio = sGui->getISimulator()->getSimTimeRatio();
149  this->addOrUpdateLiveDataByName(QStringLiteral("Time Ratio"), QString::number(ratio, 'f', 2), CIcon(CIcons::ApplicationSimulator));
150 
151  const double miles = sGui->getISimulator()->getTrackMilesShort();
152  this->addOrUpdateLiveDataByName(QStringLiteral("Miles Short"), QString::number(miles, 'f', 1), CIcon(CIcons::ApplicationSimulator));
153 
154  const double minutes = sGui->getISimulator()->getMinutesLate();
155  this->addOrUpdateLiveDataByName(QStringLiteral("Minutes Late"), QString::number(minutes, 'f', 1), CIcon(CIcons::ApplicationSimulator));
156  }
157  }
158 
159  this->addOrUpdateLiveDataByName(QStringLiteral("latitude"), s.latitude().toQString(), iconLatLng);
160  this->addOrUpdateLiveDataByName(QStringLiteral("longitude"), s.longitude().toQString(), iconLatLng);
161  this->addOrUpdateLiveDataByName(QStringLiteral("altitude, true (ft)"), s.getAltitude().valueRoundedWithUnit(CLengthUnit::ft(), 1), iconAlt);
162  this->addOrUpdateLiveDataByName(QStringLiteral("altitude, true (m)"), s.getAltitude().valueRoundedWithUnit(CLengthUnit::m(), 2), iconAlt);
163  this->addOrUpdateLiveDataByName(QStringLiteral("altitude, pressure (ft)"), s.getPressureAltitude().valueRoundedWithUnit(CLengthUnit::ft(), 1), iconAlt);
164  this->addOrUpdateLiveDataByName(QStringLiteral("altitude, pressure (m)"), s.getPressureAltitude().valueRoundedWithUnit(CLengthUnit::m(), 2), iconAlt);
165 
166  if (s.hasGroundElevation())
167  {
168  this->addOrUpdateLiveDataByName(QStringLiteral("elevation (ft)"), s.getGroundElevation().valueRoundedWithUnit(CLengthUnit::ft(), 1), iconAlt);
169  this->addOrUpdateLiveDataByName(QStringLiteral("elevation (m)"), s.getGroundElevation().valueRoundedWithUnit(CLengthUnit::m(), 2), iconAlt);
170  }
171  else
172  {
173  this->addOrUpdateLiveDataByName(QStringLiteral("elevation"), QStringLiteral("N/A"), iconAlt);
174  }
175 
176  if (ownAircraft.hasCG())
177  {
178  this->addOrUpdateLiveDataByName(QStringLiteral("CG (ft)"), ownAircraft.getCG().valueRoundedWithUnit(CLengthUnit::ft(), 1), iconPlane);
179  this->addOrUpdateLiveDataByName(QStringLiteral("CG (m)"), ownAircraft.getCG().valueRoundedWithUnit(CLengthUnit::m(), 2), iconPlane);
180  }
181  else
182  {
183  this->addOrUpdateLiveDataByName(QStringLiteral("CG"), QStringLiteral("N/A"), iconPlane);
184  }
185 
186  this->addOrUpdateLiveDataByName(QStringLiteral("pitch"), s.getPitch().toQString(), iconAttitude);
187  this->addOrUpdateLiveDataByName(QStringLiteral("bank"), s.getBank().toQString(), iconAttitude);
188 
189  const CHeading heading = s.getHeading().normalizedTo360Degrees();
190  this->addOrUpdateLiveDataByName(QStringLiteral("heading"), heading.valueRoundedWithUnit(CAngleUnit::deg(), 1), s.getHeading().toIcon());
191 
192  const CSpeed gs = s.getGroundSpeed();
193  this->addOrUpdateLiveDataByName(QStringLiteral("ground speed (kts)"), gs.valueRoundedWithUnit(CSpeedUnit::kts(), 1), gs.toIcon());
194  this->addOrUpdateLiveDataByName(QStringLiteral("ground speed (km/h)"), gs.valueRoundedWithUnit(CSpeedUnit::km_h(), 1), gs.toIcon());
195 
196  this->addOrUpdateLiveDataByName(QStringLiteral("COM1 active"), c1.getFrequencyActive().toQString(), iconRadio);
197  this->addOrUpdateLiveDataByName(QStringLiteral("COM2 active"), c2.getFrequencyActive().toQString(), iconRadio);
198  this->addOrUpdateLiveDataByName(QStringLiteral("COM1 standby"), c1.getFrequencyStandby().toQString(), iconRadio);
199  this->addOrUpdateLiveDataByName(QStringLiteral("COM2 standby"), c2.getFrequencyStandby().toQString(), iconRadio);
200  this->addOrUpdateLiveDataByName(QStringLiteral("COM1 volume"), QString::number(c1.getVolumeReceive()), iconRadio);
201  this->addOrUpdateLiveDataByName(QStringLiteral("COM2 volume"), QString::number(c2.getVolumeReceive()), iconRadio);
202  this->addOrUpdateLiveDataByName(QStringLiteral("Transponder"), ownAircraft.getTransponderCodeFormatted(), iconRadio);
203  }

205  void CSimulatorComponent::onSimulatorStatusChanged(int status)
206  {
207  ISimulator::SimulatorStatus simStatus = static_cast<ISimulator::SimulatorStatus>(status);
208  this->clear(); // clean up, will be refreshed
209  if (simStatus.testFlag(ISimulator::Connected))
210  {
211  const int intervalMs = getUpdateIntervalMs();
212  m_updateTimer.start(intervalMs);
213  this->clear(true);
214  }
215  else
216  {
217  m_updateTimer.stop();
218  this->update();
219  }
220  }

236  void CSimulatorComponent::refreshInternals()
237  {
238  if (!sGui || sGui->isShuttingDown() || !sGui->getIContextSimulator()) { return; }
239  const CSimulatorInternals internals = sGui->getIContextSimulator()->getSimulatorInternals();
240  const CSimulatorInfo simulatorInfo = sGui->getIContextSimulator()->getSimulatorPluginInfo().getSimulatorInfo();
241  m_simulator = simulatorInfo;
242 
243  const QStringList names(internals.getSortedNames());
244  if (names.isEmpty())
245  {
246  ui->tvp_Internals->clear();
247  return;
248  }
249 
250  // Update the INTERNAL view
251  static const CIcon emptyIcon;
252  const bool resize = true;
253  const bool skipEqualValues = true;
254  for (const QString &name : names)
255  {
256  ui->tvp_Internals->addOrUpdateByName(name, internals.getVariantValue(name), emptyIcon, resize, skipEqualValues);
257  }
258  ui->tvp_Internals->fullResizeToContents();
259  }
