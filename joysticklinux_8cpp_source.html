28  CJoystickDevice::CJoystickDevice(const QString &path, QFile *fd, QObject *parent)
29  : QObject(parent), m_path(path), m_fd(fd)
30  {
31  m_fd->setParent(this);
32  char deviceName[256];
33  if (ioctl(m_fd->handle(), JSIOCGNAME(sizeof(deviceName)), deviceName) < 0)
34  {
35  strncpy(deviceName, "Unknown", sizeof(deviceName));
36  }
37 
38  CLogMessage(this).info(u"Found joystick: %1") << deviceName;
39 
40  fcntl(m_fd->handle(), F_SETFL, O_NONBLOCK);
41 
42  /* Forward */
43  struct js_event event;
44  while (m_fd->read(reinterpret_cast<char *>(&event), sizeof(event)) == sizeof(event)) {}
45  QSocketNotifier *notifier = new QSocketNotifier(m_fd->handle(), QSocketNotifier::Read, m_fd);
46  connect(notifier, &QSocketNotifier::activated, this, &CJoystickDevice::processInput);
47  m_name = QString(deviceName);
48  }

59  void CJoystickDevice::processInput()
60  {
61  struct js_event event;
62  while (m_fd->read(reinterpret_cast<char *>(&event), sizeof(event)) == sizeof(event))
63  {
64  switch (event.type & ~JS_EVENT_INIT)
65  {
66  case JS_EVENT_BUTTON:
67  if (event.value) { emit buttonChanged(m_name, event.number, true); }
68  else { emit buttonChanged(m_name, event.number, false); }
69  break;
70  }
71  }
72  }

88  void CJoystickLinux::cleanupJoysticks()
89  {
90  for (auto it = m_joystickDevices.begin(); it != m_joystickDevices.end();)
91  {
92  // Remove all joysticks that do not exist anymore (/dev/input/js* removed).
93  if (!(*it)->isAttached())
94  {
95  CJoystickDevice *joystickDevice = *it;
96  it = m_joystickDevices.erase(it);
97  joystickDevice->deleteLater();
98  }
99  else
100  {
101  ++it;
102  }
103  }

106  void CJoystickLinux::addJoystickDevice(const QString &path)
107  {
108  QFile *fd = new QFile(path);
109  if (fd->open(QIODevice::ReadOnly))
110  {
111  CJoystickDevice *joystickDevice = new CJoystickDevice(path, fd, this);
112  connect(joystickDevice, &CJoystickDevice::buttonChanged, this, &CJoystickLinux::joystickButtonChanged);
113  m_joystickDevices.push_back(joystickDevice);
114  }
115  else
116  {
117  BlackMisc::CLogMessage(this).error(u"Failed to open joystick device %1: %2") << fd->fileName() << fd->errorString();
118  fd->close();
119  fd->deleteLater();
120  }
121  }

141  void CJoystickLinux::reloadDevices(QString path)
142  {
143  cleanupJoysticks();
144 
145  QDir dir(path, QLatin1String("js*"), QDir::Name, QDir::System);
146  for (const auto &entry : dir.entryInfoList())
147  {
148  QString f = entry.absoluteFilePath();
149  auto it = std::find_if(m_joystickDevices.begin(), m_joystickDevices.end(), [path](const CJoystickDevice *device) {
150  return device->getPath() == path;
151  });
152  if (it == m_joystickDevices.end())
153  {
154  addJoystickDevice(f);
155  }
156  }
157  }
