259  void CDbMappingComponent::updateEditorsWhenApplicable()
260  {
261  const CAircraftModel currentEditorModel(ui->editor_ModelMapping->getValue());
262  if (!currentEditorModel.hasModelString()) { return; } // no related model
263  const QString modelString(currentEditorModel.getModelString());
264  const CAircraftModel currentStashedModel(ui->comp_StashAircraft->getStashedModel(modelString));
265  if (!currentStashedModel.hasModelString()) { return; }
266 
267  // we have found a model in the stashed models and this is the one currently displayed
268  // in the editors
269  bool updated = false;
270  const CLivery stashedLivery(currentStashedModel.getLivery());
271  if (stashedLivery.hasValidDbKey())
272  {
273  if (ui->editor_AircraftModel->setLivery(stashedLivery)) { updated = true; }
274  }
275 
276  const CDistributor stashedDistributor(currentStashedModel.getDistributor());
277  if (stashedDistributor.hasValidDbKey())
278  {
279  if (ui->editor_AircraftModel->setDistributor(stashedDistributor)) { updated = true; }
280  }
281 
282  const CAircraftIcaoCode stashedIcaoCode(currentStashedModel.getAircraftIcaoCode());
283  if (stashedIcaoCode.hasValidDbKey())
284  {
285  if (ui->editor_AircraftModel->setAircraftIcao(stashedIcaoCode)) { updated = true; }
286  }
287 
288  if (updated)
289  {
290  CLogMessage(this).info(u"Updated editor data for '%1'") << modelString;
291  }
292  }

372  const bool consolidate = !this->isStashTab();
373  msgs.push_back(
374  ui->comp_StashAircraft->stashModel(editorModel, true, consolidate));
375  }
376  if (msgs.hasErrorMessages())
377  {
378  this->showOverlayMessagesOrHTMLMessage(msgs);
