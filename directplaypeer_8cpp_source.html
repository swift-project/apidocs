44  HRESULT CDirectPlayPeer::directPlayMessageHandler(DWORD messageId, void *msgBuffer)
45  {
46  switch (messageId)
47  {
48  case DPN_MSGID_CREATE_PLAYER:
49  {
50  DPNMSG_CREATE_PLAYER *pCreatePlayerMsg = static_cast<DPNMSG_CREATE_PLAYER *>(msgBuffer);
51 
52  // Get the peer info and extract its name
53  DWORD dwSize = 0;
54  DPN_PLAYER_INFO *pdpPlayerInfo = nullptr;
55  HRESULT hr = DPNERR_CONNECTING;
56 
57  // GetPeerInfo might return DPNERR_CONNECTING when connecting,
58  // so just keep calling it if it does
59  while (hr == DPNERR_CONNECTING)
60  {
61  hr = m_directPlayPeer->GetPeerInfo(pCreatePlayerMsg->dpnidPlayer, pdpPlayerInfo, &dwSize, 0);
62  }
63 
64  if (hr == DPNERR_BUFFERTOOSMALL)
65  {
66  QScopedArrayPointer<unsigned char> memPtr(new unsigned char[dwSize]);
67  pdpPlayerInfo = reinterpret_cast<DPN_PLAYER_INFO *>(memPtr.data());
68  if (pdpPlayerInfo == nullptr) { break; }
69 
70  ZeroMemory(pdpPlayerInfo, dwSize);
71  pdpPlayerInfo->dwSize = sizeof(DPN_PLAYER_INFO);
72 
73  hr = m_directPlayPeer->GetPeerInfo(pCreatePlayerMsg->dpnidPlayer, pdpPlayerInfo, &dwSize, 0);
74  if (SUCCEEDED(hr))
75  {
76  if (pdpPlayerInfo->dwPlayerFlags & DPNPLAYER_LOCAL)
77  {
78  m_playerLocal = pCreatePlayerMsg->dpnidPlayer;
79  }
80  else
81  {
82  // The first connecting player should be the user
83  if (m_playerUser == 0) { m_playerUser = pCreatePlayerMsg->dpnidPlayer; }
84  }
85  }
86  }
87  break;
88  }
89 
90  case DPN_MSGID_RECEIVE:
91  {
92  PDPNMSG_RECEIVE pReceiveMsg = static_cast<PDPNMSG_RECEIVE>(msgBuffer);
93 
94  // Proceeed only, if the sender is our local player
95  if (pReceiveMsg->dpnidSender == m_playerUser)
96  {
97  const QByteArray messageData = QByteArray((char *)pReceiveMsg->pReceiveData, pReceiveMsg->dwReceiveDataSize);
98  emit customPacketReceived(messageData);
99  }
100  break;
101  }
102 
103  case DPN_MSGID_ENUM_HOSTS_RESPONSE:
104  {
105  PDPNMSG_ENUM_HOSTS_RESPONSE enumHostsResponseMsg = static_cast<PDPNMSG_ENUM_HOSTS_RESPONSE>(msgBuffer);
106  const DPN_APPLICATION_DESC *applicationDescription = enumHostsResponseMsg->pApplicationDescription;
107 
108  auto iterator = std::find_if(m_hostNodeList.begin(), m_hostNodeList.end(), [&](const CHostNode &hostNode) {
109  return applicationDescription->guidInstance == hostNode.getApplicationDesc().guidInstance;
110  });
111 
112  if (iterator == m_hostNodeList.end())
113  {
114 
115  // This host session is not in the list then so insert it.
116  CHostNode hostNode;
117  HRESULT hr;
118 
119  // Copy the Host Address
120  if (FAILED(hr = enumHostsResponseMsg->pAddressSender->Duplicate(hostNode.getHostAddressPtr())))
121  {
122  CLogMessage(this).warning(u"Failed to duplicate host address!");
123  return hr;
124  }
125 
126  DPN_APPLICATION_DESC appDesc;
127 
128  ZeroMemory(&appDesc, sizeof(DPN_APPLICATION_DESC));
129  memcpy(&appDesc, applicationDescription, sizeof(DPN_APPLICATION_DESC));
130 
131  // Null out all the pointers we aren't copying
132  appDesc.pwszSessionName = nullptr;
133  appDesc.pwszPassword = nullptr;
134  appDesc.pvReservedData = nullptr;
135  appDesc.dwReservedDataSize = 0;
136  appDesc.pvApplicationReservedData = nullptr;
137  appDesc.dwApplicationReservedDataSize = 0;
138  // hostNode.setApplicationDesc(appDesc);
139  hostNode.setApplicationDesc(appDesc);
140  hostNode.setSessionName(QString::fromWCharArray(applicationDescription->pwszSessionName));
141  m_hostNodeList.append(hostNode);
142  }
143  break;
144  }
145  case DPN_MSGID_CONNECT_COMPLETE:
146  {
147  const PDPNMSG_CONNECT_COMPLETE connectCompleteMsg = static_cast<PDPNMSG_CONNECT_COMPLETE>(msgBuffer);
148  if (connectCompleteMsg->hResultCode == DPN_OK)
149  {
150  emit connectionComplete();
151  }
152  else
153  {
154  CLogMessage(this).warning(u"DirectPlay connection returned: %1") << connectCompleteMsg->hResultCode;
155  }
156  }
157  }
158 
159  // Directx9 SDK: Unless otherwise noted, this function should return S_OK.
160  // http://doc.51windows.net/Directx9_SDK/play/ref/callbacks/pfndpnmessagehandler.htm
161  return DPN_OK;
162  }

164  HRESULT CDirectPlayPeer::initDirectPlay()
165  {
166  // Initialize COM.
167  // https://docs.microsoft.com/en-us/windows/desktop/api/combaseapi/nf-combaseapi-coinitializeex
168  HRESULT hr = CoInitializeEx(nullptr, COINIT_MULTITHREADED);
169 
170  // RPC_E_CHANGED_MODE: CoInitializeEx was already called by someone else in this thread with a different mode.
171  if (hr == RPC_E_CHANGED_MODE)
172  {
173  CLogMessage(this).debug(u"CoInitializeEx was already called with a different mode. Trying again.");
174  hr = CoInitializeEx(nullptr, COINIT_APARTMENTTHREADED);
175  }
176 
177  // Continue here only if CoInitializeEx was successful
178  // S_OK: The COM library was initialized successfully on this thread.
179  // S_FALSE: The COM library is already initialized on this thread. Reference count was incremented. This is not an error.
180  if (hr == S_OK || hr == S_FALSE)
181  {
182  m_coInitializeSucceeded = true;
183  }
184  else
185  {
186  CLogMessage(this).warning(u"CoInitializeEx returned error code %1");
187  return E_FAIL;
188  }
189 
190  // Create the IDirectPlay8Peer Object
191  if (FAILED(hr = CoCreateInstance(CLSID_DirectPlay8Peer,
192  nullptr,
193  CLSCTX_INPROC_SERVER,
194  IID_IDirectPlay8Peer,
195  reinterpret_cast<void **>(&m_directPlayPeer))))
196  {
197  CLogMessage(this).warning(u"Failed to create DirectPlay8Peer object!");
198  return hr;
199  }
200 
201  // Init DirectPlay
202  if (FAILED(hr = m_directPlayPeer->Initialize(&m_callbackWrapper, m_callbackWrapper.messageHandler, 0)))
203  {
204  CLogMessage(this).warning(u"Failed to initialize directplay peer!");
205  return hr;
206  }
207 
208  // Ensure that TCP/IP is a valid Service Provider
209  if (!isServiceProviderValid(&CLSID_DP8SP_TCPIP))
210  {
211  hr = E_FAIL;
212  CLogMessage(this).warning(u"Service provider is invalid!");
213  return hr;
214  }
215 
216  return hr;
217  }

219  bool CDirectPlayPeer::isServiceProviderValid(const GUID * /*pGuidSP*/)
220  {
221  DWORD dwItems = 0;
222  DWORD dwSize = 0;
223 
224  // The first call is to retrieve the size of the DPN_SERVICE_PROVIDER_INFO array
225  HRESULT hr = m_directPlayPeer->EnumServiceProviders(&CLSID_DP8SP_TCPIP, nullptr, nullptr, &dwSize, &dwItems, 0);
226 
227  if (hr != DPNERR_BUFFERTOOSMALL)
228  {
229  CLogMessage(this).warning(u"Failed to enumerate service providers!");
230  return false;
231  }
232 
233  // Allocating an array with new DPN_SERVICE_PROVIDER_INFO[items] does not work, because the struct has
234  // several pointers in it. Hence EnumServiceProviders tells us how much memory it exactly needs.
235  QScopedArrayPointer<unsigned char> memPtr(new unsigned char[dwSize]);
236  DPN_SERVICE_PROVIDER_INFO *dpnSPInfo = reinterpret_cast<DPN_SERVICE_PROVIDER_INFO *>(memPtr.data());
237 
238  if (FAILED(hr = m_directPlayPeer->EnumServiceProviders(&CLSID_DP8SP_TCPIP, nullptr, dpnSPInfo, &dwSize, &dwItems, 0)))
239  {
240  CLogMessage(this).warning(u"Failed to enumerate service providers!");
241  return false;
242  }
243 
244  // There are no items returned so the requested SP is not available
245  if (dwItems == 0) { hr = E_FAIL; }
246 
247  if (SUCCEEDED(hr)) { return true; }
248  else { return false; }
249  }

295  HRESULT CDirectPlayPeer::sendMessage(const QByteArray &message)
296  {
297  HRESULT hr = S_OK;
298  DPN_BUFFER_DESC dpBufferDesc;
299 
300  if ((dpBufferDesc.dwBufferSize = message.size()) == 0) { return S_FALSE; }
301 
303  dpBufferDesc.pBufferData = (BYTE *)message.data();
304 
305  DPNHANDLE asyncHandle;
306  // If m_playerUser is non zero, send it only to him
307  if (FAILED(hr = m_directPlayPeer->SendTo(m_playerUser,
308  &dpBufferDesc,
309  1, 0,
310  nullptr,
311  &asyncHandle,
312  DPNSEND_GUARANTEED)))
313  {
314  const QString m(message);
315  CLogMessage(this).warning(u"DirectPlay: Failed to send message! Return value: %1 ") << hr;
316  CLogMessage(this).debug() << m;
317  }
318  return hr;
319  }
