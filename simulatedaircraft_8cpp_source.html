345  case IndexNetworkModelAirlineIcaoDifference: return this->getNetworkModelAirlineIcaoDifference();
346  case IndexNetworkModelLiveryDifference: return this->getNetworkModelLiveryDifference();
347  case IndexEnabled: return QVariant::fromValue(this->isEnabled());
348  case IndexRendered: return QVariant::fromValue(this->isRendered());
349  case IndexPartsSynchronized: return QVariant::fromValue(this->isPartsSynchronized());
350  case IndexFastPositionUpdates: return QVariant::fromValue(this->fastPositionUpdates());

353  case IndexPilot: return m_pilot.propertyByIndex(index.copyFrontRemoved());
354  case IndexRelativeDistance: return m_relativeDistance.propertyByIndex(index.copyFrontRemoved());
355  case IndexCom1System: return m_com1system.propertyByIndex(index.copyFrontRemoved());
356  case IndexCom2System: return m_com2system.propertyByIndex(index.copyFrontRemoved());
357  case IndexTransponder: return m_transponder.propertyByIndex(index.copyFrontRemoved());
358  case IndexSituation: return m_situation.propertyByIndex(index.copyFrontRemoved());
359  case IndexAircraftIcaoCode: return this->getAircraftIcaoCode().propertyByIndex(index.copyFrontRemoved());

364  case IndexCombinedIcaoLiveryStringNetworkModel: return QVariant::fromValue(this->getCombinedIcaoLiveryString(true));

372  void CSimulatedAircraft::setPropertyByIndex(CPropertyIndexRef index, const QVariant &variant)
373  {
374  if (index.isMyself())
375  {
376  (*this) = variant.value<CSimulatedAircraft>();
377  return;
378  }
379  const ColumnIndex i = index.frontCasted<ColumnIndex>();
380  switch (i)
381  {
382  case IndexCallsign: m_callsign.setPropertyByIndex(index.copyFrontRemoved(), variant); break;
383  case IndexPilot: m_pilot.setPropertyByIndex(index.copyFrontRemoved(), variant); break;
384  case IndexRelativeDistance: m_relativeDistance.setPropertyByIndex(index.copyFrontRemoved(), variant); break;
385  case IndexCom1System: m_com1system.setPropertyByIndex(index.copyFrontRemoved(), variant); break;
386  case IndexCom2System: m_com2system.setPropertyByIndex(index.copyFrontRemoved(), variant); break;
387  case IndexTransponder: m_transponder.setPropertyByIndex(index.copyFrontRemoved(), variant); break;
388  case IndexSituation: m_situation.setPropertyByIndex(index.copyFrontRemoved(), variant); break;
389  case IndexParts: m_parts.setPropertyByIndex(index.copyFrontRemoved(), variant); break;
390  case IndexNetworkModel: m_models[NetworkModel].setPropertyByIndex(index.copyFrontRemoved(), variant); break;
391  case IndexEnabled: m_enabled = variant.toBool(); break;
392  case IndexRendered: m_rendered = variant.toBool(); break;
393  case IndexPartsSynchronized: m_partsSynchronized = variant.toBool(); break;
394  case IndexFastPositionUpdates: m_fastPositionUpdates = variant.toBool(); break;
395  case IndexSupportsGndFlag: m_supportsGndFlag = variant.toBool(); break;
396  case IndexLivery: Q_ASSERT_X(false, Q_FUNC_INFO, "Unsupported"); break;
397  case IndexModel:
398  m_models[CurrentModel].setPropertyByIndex(index.copyFrontRemoved(), variant);
399  this->setModel(m_models[CurrentModel]); // sync some values such as callsign
400  break;
401  default:
402  if (ICoordinateWithRelativePosition::canHandleIndex(index))
403  {
404  ICoordinateWithRelativePosition::setPropertyByIndex(index, variant);
405  }
406  else
407  {
408  CValueObject::setPropertyByIndex(index, variant);
409  }
410  break;
411  }

414  int CSimulatedAircraft::comparePropertyByIndex(CPropertyIndexRef index, const CSimulatedAircraft &compareValue) const
415  {
416  if (index.isMyself()) { return m_callsign.comparePropertyByIndex(index.copyFrontRemoved(), compareValue.getCallsign()); }
417  const ColumnIndex i = index.frontCasted<ColumnIndex>();
418  switch (i)
419  {
420  case IndexCallsign: return m_callsign.comparePropertyByIndex(index.copyFrontRemoved(), compareValue.getCallsign());
421  case IndexPilot: return m_pilot.comparePropertyByIndex(index.copyFrontRemoved(), compareValue.getPilot());
422  case IndexSituation: return m_situation.comparePropertyByIndex(index.copyFrontRemoved(), compareValue.getSituation());
423  case IndexRelativeDistance: return m_relativeDistance.comparePropertyByIndex(index.copyFrontRemoved(), compareValue.getRelativeDistance());
424  case IndexCom1System: return m_com1system.comparePropertyByIndex(index.copyFrontRemoved(), compareValue.getCom1System());
425  case IndexCom2System: return m_com2system.comparePropertyByIndex(index.copyFrontRemoved(), compareValue.getCom2System());
426  case IndexTransponder: return Compare::compare(m_transponder.getTransponderCode(), compareValue.getTransponder().getTransponderCode());
427  case IndexLivery: return this->getLivery().comparePropertyByIndex(index.copyFrontRemoved(), compareValue.getLivery());
428  case IndexParts: return m_parts.comparePropertyByIndex(index.copyFrontRemoved(), compareValue.getParts());
429  case IndexModel: return m_models[CurrentModel].comparePropertyByIndex(index.copyFrontRemoved(), compareValue.getModel());
430  case IndexNetworkModel: return m_models[NetworkModel].comparePropertyByIndex(index.copyFrontRemoved(), compareValue.getModel());
431  case IndexNetworkModelAircraftIcaoDifference: return this->getNetworkModelAircraftIcaoDifference().compare(compareValue.getNetworkModelAircraftIcaoDifference());
432  case IndexNetworkModelAirlineIcaoDifference: return this->getNetworkModelAirlineIcaoDifference().compare(compareValue.getNetworkModelAirlineIcaoDifference());
433  case IndexNetworkModelLiveryDifference: return this->getNetworkModelLiveryDifference().compare(compareValue.getNetworkModelLiveryDifference());
434  case IndexEnabled: return Compare::compare(this->isEnabled(), compareValue.isEnabled());
435  case IndexRendered: return Compare::compare(this->isRendered(), compareValue.isRendered());
436  case IndexPartsSynchronized: return Compare::compare(this->isPartsSynchronized(), compareValue.isPartsSynchronized());
437  case IndexFastPositionUpdates: return Compare::compare(this->fastPositionUpdates(), compareValue.fastPositionUpdates());
438  case IndexSupportsGndFlag: return Compare::compare(this->isSupportingGndFlag(), compareValue.isSupportingGndFlag());
439  case IndexCombinedIcaoLiveryString: return this->getCombinedIcaoLiveryString(false).compare(compareValue.getCombinedIcaoLiveryString(false));
440  case IndexCombinedIcaoLiveryStringNetworkModel: return this->getCombinedIcaoLiveryString(true).compare(compareValue.getCombinedIcaoLiveryString(true));
441  default:
442  if (ICoordinateWithRelativePosition::canHandleIndex(index))
443  {
444  return ICoordinateWithRelativePosition::comparePropertyByIndex(index, compareValue);
445  }
446  break;
447  }
448  BLACK_VERIFY_X(false, Q_FUNC_INFO, qUtf8Printable("No comparison for index " + index.toQString()));
449  return 0;
450  }

464  QString CSimulatedAircraft::getNetworkModelAircraftIcaoDifference() const
465  {

467  if (!icaoNw.hasDesignator()) { return QStringLiteral("[x] no nw. ICAO"); }
468 
469  const CAircraftIcaoCode icaoModel(this->getModel().getAircraftIcaoCode());
470  if (!icaoModel.hasDesignator()) { return QStringLiteral("[x] no sim. ICAO"); }
471 
472  if (icaoModel.isDbEqual(icaoNw) || icaoModel == icaoNw) { return u"[==] " % icaoModel.getDesignatorDbKey(); }
473  if (icaoModel.getDesignator() == icaoNw.getDesignator()) { return u"[=] " % icaoModel.getDesignator(); }
474  return QStringLiteral("%1 -> %2").arg(icaoNw.getDesignator(), icaoModel.getDesignator());
475  }
