29  bool IInterpolationSetupProvider::setInterpolationSetupsPerCallsign(const CInterpolationSetupList &setups, bool ignoreSameAsGlobal)
30  {
31  const CInterpolationAndRenderingSetupGlobal gs = this->getInterpolationSetupGlobal();
32  SetupsPerCallsign setupsPerCs;
33  for (const CInterpolationAndRenderingSetupPerCallsign &setup : setups)
34  {
35  if (ignoreSameAsGlobal && setup.isEqualToGlobal(gs)) { continue; }
36  setupsPerCs.insert(setup.getCallsign(), setup);
37  }
38  {
39  QWriteLocker l(&m_lockSetup);
40  if (m_setupsPerCallsign.isEmpty() && setupsPerCs.isEmpty()) { return false; }
41  m_setupsPerCallsign = setupsPerCs;
42  }
43  this->emitInterpolationSetupChanged();
44  return true;
45  }

139  void IInterpolationSetupProvider::clearInterpolationLogCallsigns()
140  {
141  SetupsPerCallsign setupsCopy = this->getSetupsPerCallsign();
142  if (setupsCopy.isEmpty()) { return; }
143 
144  // potential risk, changes inbetween in another thread are missed now
145  // on the other side, we keep locks for a minimal time frame
146  SetupsPerCallsign setupsToKeep;
147  CInterpolationAndRenderingSetupGlobal global = this->getInterpolationSetupGlobal();
148  for (auto [callsign, setup] : makePairsRange(setupsCopy))
149  {
150  setup.setLogInterpolation(false);
151  if (setup.isEqualToGlobal(global)) { continue; }
152  setupsToKeep.insert(callsign, setup);
153  }
154  {
155  QWriteLocker l(&m_lockSetup);
156  m_setupsPerCallsign = setupsToKeep;
157  }
158  this->emitInterpolationSetupChanged();
159  }
