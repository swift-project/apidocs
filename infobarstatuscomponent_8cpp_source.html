144  void CInfoBarStatusComponent::onSimulatorStatusChanged(int status)
145  {
146  if (!sGui || sGui->isShuttingDown()) { return; }
147  const ISimulator::SimulatorStatus simStatus = static_cast<ISimulator::SimulatorStatus>(status);
148  if (simStatus.testFlag(ISimulator::Connected))
149  {
150  // at least connected
151  const QString s(
152  sGui->getIContextSimulator()->getSimulatorPluginInfo().getDescription() % u": " %
153  ISimulator::statusToString(simStatus));
154 
155  if (simStatus.testFlag(ISimulator::Paused))
156  {
157  ui->led_Simulator->setTriState();
158  ui->led_Simulator->setTriStateToolTip(s);
159  }
160  else if (simStatus.testFlag(ISimulator::Simulating))
161  {
162  ui->led_Simulator->setOn(true);
163  ui->led_Simulator->setOnToolTip(s);
164  }
165  else
166  {
167  // connected only
168  ui->led_Simulator->setTriState();
169  ui->led_Simulator->setTriStateToolTip(s);
170  }
171  }
172  else
173  {
174  ui->led_Simulator->setOn(false);
175  }
176 
177  // simulator status has impact on model set available
178  this->onMapperReady();
179  }

181  void CInfoBarStatusComponent::onNetworkConnectionChanged(const CConnectionStatus &from, const CConnectionStatus &to)
182  {
183  Q_UNUSED(from)
184  switch (to.getConnectionStatus())
185  {
186  case CConnectionStatus::Disconnected:
187  ui->led_Network->setOn(false);
188  break;
189  case CConnectionStatus::Connected:
190  ui->led_Network->setOn(true);
191  ui->led_Network->setOnToolTip(u"Connected: " % sGui->getIContextNetwork()->getConnectedServer().getName());
192  break;
193  case CConnectionStatus::Connecting:
194  ui->led_Network->setTriStateColor(CLedWidget::Yellow);
195  break;
196  default:
197  ui->led_Network->setOn(false);
198  break;
199  }
200  }

202  void CInfoBarStatusComponent::onCustomAudioContextMenuRequested(const QPoint &position)
203  {
204  const QWidget *sender = qobject_cast<QWidget *>(QWidget::sender());
205  Q_ASSERT_X(sender, Q_FUNC_INFO, "Missing sender");
206  const QPoint globalPosition = sender->mapToGlobal(position);
207 
208  QMenu menuAudio(this);
209  menuAudio.addAction("Toogle mute");
210 
211  if (CBuildConfig::isRunningOnWindowsNtPlatform())
212  {
213  menuAudio.addAction("Mixer");
214  }
215 
216  const QAction *selectedItem = menuAudio.exec(globalPosition);
217  if (selectedItem)
218  {
219  // http://forum.technical-assistance.co.uk/sndvol32exe-command-line-parameters-vt1348.html
220  const QList<QAction *> actions = menuAudio.actions();
221  if (selectedItem == actions.at(0))
222  {
223  // toggle MUTED
224  sGui->getCContextAudioBase()->setMute(!sGui->getCContextAudioBase()->isMuted());
225  }
226  else if (actions.size() > 1 && selectedItem == actions.at(1))
227  {
228  startWindowsMixer();
229  }
230  }
231  }
