95  void CAltitudeDialog::onEditFinished()
96  {
97  const Mode mode = this->getMode();
98  switch (mode)
99  {
100  case VFR:
101  m_altitudeStr = QStringLiteral("VFR");
102  break;
103  case FlightFlevelInFeet:
104  m_altitudeStr = u"FL" % ui->le_FLft->text();
105  break;
106  case MetricLevelInTensOfMeters:
107  m_altitudeStr = u'S' % ui->le_Sm->text();
108  break;
109  case AltitudeInHundredsOfFeet:
110  m_altitudeStr = u'A' % ui->le_AltitudeAFt->text();
111  break;
112  case AltitudeInTensOfMeters:
113  m_altitudeStr = u'M' % ui->le_AltitudeMm->text();
114  break;
115  case AltitudeInFeet:
116  m_altitudeStr = ui->le_AltitudeFt->text() % u"ft";
117  break;
118  case AltitudeInMeters:
119  m_altitudeStr = ui->le_AltitudeM->text() % u"m";
120  break;
121  case AltitudeInMetersConvertedToFeet:
122  m_altitudeStr = ui->le_AltitudeConvertedFt->text() % u"m";
123  break;
124  case StringOnly:
125  m_altitudeStr = ui->le_String->text();
126  return;
127  case Unknown:
128  default:
129  return;
130  }
131 
132  CStatusMessageList msgs;
133  m_altitude.parseFromFpAltitudeString(m_altitudeStr, &msgs);
134  if (msgs.hasErrorMessages())
135  {
136  ui->le_String->setText(msgs.toSingleMessage().getMessage());
137  m_altitudeStr.clear();
138  }
139  else
140  {
141  if (mode == AltitudeInMetersConvertedToFeet)
142  {
143  const int ft = CAltitude::findAltitudeForMetricAltitude(m_altitude.valueInteger(CLengthUnit::m()));
144  m_altitude = ft < 0 ?
145  m_altitude.roundedToNearest100ft(true) :
146  m_altitude = CAltitude(ft, CAltitude::MeanSeaLevel, CLengthUnit::ft());
147  m_altitudeStr = m_altitude.valueRoundedWithUnit(0);
148  }
149 
150  if (ui->cb_SimplifiedVATSIMFormat->isChecked())
151  {
152  m_altitudeStr = m_altitude.asFpVatsimAltitudeString();
153  }
154 
155  ui->le_String->setText(m_altitudeStr);
156  }
157 
158  this->updateStyleSheet();
159  }
