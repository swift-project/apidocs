408  {
409  unsigned int hash = 0;

438  bool CGuiApplication::restoreWindowGeometryAndState(QMainWindow *window)
439  {
440  if (!window) { return false; }
441  const QSettings settings(fileForWindowGeometryAndStateSettings(), QSettings::IniFormat);
442  const QString location = settings.fileName();
443  CLogMessage(this).info(u"GUI settings are here: '%1'") << location;
444 
445  const QByteArray g = settings.value("geometry").toByteArray();
446  const QByteArray s = settings.value("windowState").toByteArray();
447  if (g.isEmpty() || s.isEmpty()) { return false; }
448 
449  // block for subscriber
450  {
451  const auto pattern = CLogPattern().withSeverity(CStatusMessage::SeverityError);
452  const QString parameter = m_cmdWindowSizeReset.names().first();
453  CLogSubscriber logSub(this, [&](const CStatusMessage &message) {
454  // handles an error in restoreGeometry/State
455  const int ret = QMessageBox::critical(sGui->mainApplicationWidget(), sGui->getApplicationNameAndVersion(),
456  QStringLiteral(
457  "Restoring the window state/geometry failed!\n"
458  "You need to reset the window size (command -%1).\n\n"
459  "Original msg: %2\n\n"
460  "We can try to reset the values and restart\n"
461  "Do you want to try?")
462  .arg(parameter, message.getMessage()),
463  QMessageBox::Yes | QMessageBox::No);
464  if (ret == QMessageBox::Yes)
465  {
466  this->resetWindowGeometryAndState();
467  this->restartApplication();
468  }
469  // most likely crashing if we do nothing
470  });
471  logSub.changeSubscription(pattern);
472 
473  window->restoreGeometry(g);
474  window->restoreState(s, hashForStateSettingsSchema(window));
475  }
476  return true;
477  }

515  QString CGuiApplication::beautifyHelpMessage(const QString &helpText)
516  {
517  // just formatting Qt help message into HTML table
518  if (helpText.isEmpty()) { return {}; }
519  const QStringList lines(helpText.split('\n'));
520  QString html;
521  bool tableMode = false;
522  bool pendingTr = false;
523  for (const QString &l : lines)
524  {
525  QString lt(l.trimmed());
526  if (!tableMode && lt.startsWith("-"))
527  {
528  tableMode = true;
529  html += "<table>\n";
530  }
531  if (!tableMode)
532  {
533  html += l.toHtmlEscaped();
534  html += "<br>";
535  }
536  else
537  {
538  // in table mode
539  if (lt.startsWith("-"))
540  {
541  if (pendingTr)
542  {
543  html += "</td></tr>\n";
544  }
545  html += "<tr><td>";
546  thread_local const QRegularExpression reg("[ ]{2,}");
547  html += lt.replace(reg, "</td><td>");
548  pendingTr = true;
549  }
550  else
551  {
552  html += " ";
553  html += l.simplified().toHtmlEscaped();
554  }
555  }
556  }
557  html += "</table>\n";
558  return html;
559  }
