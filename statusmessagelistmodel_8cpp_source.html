68  const CPropertyIndex oldIndex = this->getSortProperty();
69  m_columns.clear();
70  m_mode = mode;
71 
72  switch (mode)

77  case Detailed:

84  m_columns.addColumn(CColumn::standardString("category", CStatusMessage::IndexCategoriesAsString));
85  }
86  break;
87  case SimplifiedWithOrder:
88  m_columns.addColumn(CColumn::orderColumn());
89  [[fallthrough]];
90  case Simplified:
91  {
92  m_columns.addColumn(CColumn("time", CStatusMessage::IndexUtcTimestamp, new CDateTimeFormatter(CDateTimeFormatter::formatHmsz())));
93  CColumn col = CColumn("severity", CStatusMessage::IndexSeverityAsIcon);
94  col.setSortPropertyIndex(CStatusMessage::IndexSeverityAsString);
95  m_columns.addColumn(col);
96  m_columns.addColumn(CColumn::standardString("message", CStatusMessage::IndexMessage));
97  }
98  break;
99  }
100  // m_columns.insertEmptyColumn();
101 
102  // sorting
103  if (CStatusMessageListModel::sortedByTimestampOrOrder(oldIndex))
104  {
105  if (mode == DetailedWithOrder || mode == SimplifiedWithOrder)
106  {
107  this->setSorting(CStatusMessage::IndexOrder, oldOrder);
108  }
109  else
