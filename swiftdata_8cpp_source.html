150 void CSwiftData::consolidationSettingChanged()
151 {
152  const int consolidationSecs = m_consolidationSettings.get();
153  if (consolidationSecs < 0)
154  {
155  if (m_updater)
156  {
157  ui->comp_MainInfoArea->getDataSettingsComponent()->setBackgroundUpdater(nullptr);
158  disconnect(m_updater);
159  m_updater->abandonAndWait();
160  m_updater = nullptr;
161  }
162  }
163  else
164  {
165  if (!m_updater)
166  {
167  m_updater = new CBackgroundDataUpdater(this);
168  connect(m_updater, &CBackgroundDataUpdater::consolidating, ui->comp_InfoBar, &CInfoBarWebReadersStatusComponent::consolidationRunning, Qt::QueuedConnection);
169  m_updater->start(QThread::LowestPriority);
170  ui->comp_MainInfoArea->getDataSettingsComponent()->setBackgroundUpdater(m_updater);
171  }
172  m_updater->startUpdating(consolidationSecs);
173  }
174 }

186 void CSwiftData::checkMinimumVersion()
187 {
188  if (!sApp || sApp->isShuttingDown()) { return; }
189  if (sApp->getGlobalSetup().isSwiftVersionMinimumMappingVersion())
190  {
191  CLogMessage(this).info(u"Checked mapping tool version, required '%1', this version '%2'") << sApp->getGlobalSetup().getMappingMinimumVersionString() << CBuildConfig::getVersionString();
192  }
193  else
194  {
195  const CStatusMessage sm = CStatusMessage(this, CStatusMessage::SeverityWarning, u"Your are using swift version: '%1'. Creating mappings requires at least '%2'.") << CBuildConfig::getVersionString() << sApp->getGlobalSetup().getMappingMinimumVersionString();
196  CLogMessage::preformatted(sm);
197  this->displayInOverlayWindow(sm);
198  }
199 }
