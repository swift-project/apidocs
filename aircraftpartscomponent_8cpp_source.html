53  void CAircraftPartsComponent::sendAircraftParts()
54  {
55  if (!sGui || sGui->isShuttingDown()) { return; }
56 
57  Q_ASSERT(sGui->getIContextNetwork());
58  if (!sGui->getIContextNetwork()->isConnected())
59  {
60  CLogMessage(this).validationError(u"Cannot send aircraft parts, network not connected");
61  return;
62  }
63  const CCallsign callsign(ui->comp_RemoteAircraftCompleter->getCallsign(true));
64  if (callsign.isEmpty())
65  {
66  CLogMessage(this).validationError(u"No valid callsign selected");
67  return;
68  }
69 
70  CClient client = sGui->getIContextNetwork()->getClientsForCallsigns(callsign).frontOrDefault();
71  if (client.getCallsign().isEmpty() || client.getCallsign() != callsign)
72  {
73  CLogMessage(this).validationError(u"No valid client for '%1'") << callsign.asString();
74  return;
75  }
76 
77  if (!client.hasAircraftPartsCapability())
78  {
79  static const QString question("'%1' does not support parts, enable parts for it?");
80  const QMessageBox::StandardButton reply = QMessageBox::question(this, "No parts supported", question.arg(callsign.asString()), QMessageBox::Yes | QMessageBox::No);
81  if (reply != QMessageBox::Yes) { return; }
82  client.addCapability(CClient::FsdWithAircraftConfig);
83  const bool enabled = sGui->getIContextNetwork()->setOtherClient(client);
84  Q_UNUSED(enabled)
85  }
86 
87  const bool json = (QObject::sender() == ui->pb_SendAircraftPartsJson);
88  const CAircraftParts parts = json ? ui->editor_AircraftParts->getAircraftPartsFromJson() : ui->editor_AircraftParts->getAircraftPartsFromGui();
89  ui->editor_AircraftParts->setAircraftParts(parts); // display in UI as GUI and JSON
90 
91  ui->tb_History->setToolTip("");
92  const bool incremental = ui->cb_AircraftPartsIncremental->isChecked();
93  sGui->getIContextNetwork()->testAddAircraftParts(callsign, parts, incremental);
94  CLogMessage(this).info(u"Added parts for %1") << callsign.toQString();
95  }

97  void CAircraftPartsComponent::setCurrentParts()
98  {
99  if (!sGui->getIContextNetwork()->isConnected()) { return; }
100  const CCallsign callsign(ui->comp_RemoteAircraftCompleter->getCallsign());
101  if (callsign.isEmpty()) { return; }
102 
103  const CAircraftPartsList partsList = sGui->getIContextNetwork()->getRemoteAircraftParts(callsign);
104  if (partsList.isEmpty())
105  {
106  CStatusMessage(this).info(u"No parts for '%1'") << callsign.asString();
107  return;
108  }
109  const CAircraftParts parts = partsList.latestObject();
110  const CStatusMessageList history = sGui->getIContextNetwork()->getAircraftPartsHistory(callsign);
111  ui->editor_AircraftParts->setAircraftParts(parts);
112  ui->tb_History->setToolTip(history.toHtml());
113  }

115  void CAircraftPartsComponent::requestPartsFromNetwork()
116  {
117  if (!sGui || sGui->isShuttingDown()) { return; }
118  if (!sGui->getIContextNetwork()) { return; }
119 
120  const CCallsign callsign(ui->comp_RemoteAircraftCompleter->getCallsign(true));
121  if (callsign.isEmpty())
122  {
123  CLogMessage(this).validationError(u"No valid callsign selected");
124  return;
125  }
126  ui->pb_RequestFromNetwork->setEnabled(false);
127  sGui->getIContextNetwork()->testRequestAircraftConfig(callsign);
128  CLogMessage(this).info(u"Request aircraft config for '%1'") << callsign.asString();
129 
130  // simple approach to update UI when parts are received
131  const QPointer<CAircraftPartsComponent> myself(this);
132  QTimer::singleShot(3000, this, [=] {
133  if (!myself) { return; }
134  ui->pb_CurrentParts->click();
135  ui->pb_RequestFromNetwork->setEnabled(true);
136  });
137  }
