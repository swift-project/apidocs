177  bool CInterpolationAndRenderingSetupGlobal::setMaxRenderedAircraft(int maxRenderedAircraft)
178  {
179  if (maxRenderedAircraft == m_maxRenderedAircraft) { return false; }
180  if (maxRenderedAircraft < 1)
181  {
182  // disable, we set both values to 0
183  this->disableRendering();
184  }
185  else if (maxRenderedAircraft >= InfiniteAircraft())
186  {
187  m_maxRenderedAircraft = InfiniteAircraft();
188  }
189  else
190  {
191  m_maxRenderedAircraft = maxRenderedAircraft;
192  }
193  return true;
194  }

196  bool CInterpolationAndRenderingSetupGlobal::setMaxRenderedDistance(const CLength &distance)
197  {
198 
199  if (distance == m_maxRenderedDistance) { return false; }
200  if (distance.isNull() || distance.isNegativeWithEpsilonConsidered())
201  {
202  m_maxRenderedDistance = CLength(0.0, nullptr);
203  }
204  else if (distance.isZeroEpsilonConsidered())
205  {
206  // zero means disabled, we disable max aircraft too
207  this->disableRendering();
208  }
209  else
210  {
211  Q_ASSERT(!distance.isNegativeWithEpsilonConsidered());
212  m_maxRenderedDistance = distance;
213  }
214  return true;
215  }
