314  bool CSimulatorEmulated::changeInternalSituation(const CAircraftSituation &situation)
315  {
316  if (m_myAircraft.getSituation() == situation) { return false; }
317  m_myAircraft.setSituation(situation);
318 
319  if (m_isWeatherActivated)
320  {
321  const auto currentPosition = CCoordinateGeodetic { situation.latitude(), situation.longitude() };
322  if (CWeatherScenario::isRealWeatherScenario(m_weatherScenarioSettings.get()))
323  {
324  if (m_lastWeatherPosition.isNull() ||
325  calculateGreatCircleDistance(m_lastWeatherPosition, currentPosition).value(CLengthUnit::mi()) > 20)
326  {
327  m_lastWeatherPosition = currentPosition;
328  requestWeatherGrid(currentPosition, this->identifier());
329  }
330  }
331  }
332 
333  return this->updateOwnSituationAndGroundElevation(situation);

459  void CSimulatorEmulated::onSettingsChanged()
460  {
461  if (!sApp || sApp->isShuttingDown()) { return; }
462  const CSwiftPluginSettings settings(m_pluginSettings.get());
463  m_log = settings.isLoggingFunctionCalls();
464 
465  CSimulatorInfo simulator = settings.getEmulatedSimulator();
466  if (!simulator.isSingleSimulator())
467  {
468  // this will always be the same
469  simulator = CSimulatorInfo::guessDefaultSimulator();
470  }
471  Q_ASSERT_X(simulator.isSingleSimulator(), Q_FUNC_INFO, "need single simulator");
472 
473  const CSimulatorPluginInfoList plugins = sApp->getIContextSimulator()->getAvailableSimulatorPlugins();
474  const CSimulatorPluginInfo plugin = plugins.findBySimulator(simulator);
475 
476  if (plugin.isValid())
477  {
478  // ? restart driver, disconnect/reconnect
479  this->setNewPluginInfo(plugin, m_multiSettings.getSettings(simulator), settings.getDefaultModel());
480  }
481  else
482  {
483  // not all drivers are installed, like FS9/FSX on x64
484  CLogMessage(this).validationError(u"No valid plugin in emulated driver for '%1'") << simulator.toQString();
485  const QString sn = simulator.toQString(true);
486  const CSimulatorPluginInfo fakedPlugin = CSimulatorPluginInfo(CSimulatorPluginInfo::identifierFromSimulatorInfo(simulator), sn, sn, QStringLiteral("Emulated, but uninstalled '%1'").arg(sn), false);
487  this->setNewPluginInfo(fakedPlugin, m_multiSettings.getSettings(simulator), settings.getDefaultModel());
488  }
489 
490  // update provider, own name, title
491  this->updateOwnModel(settings.getOwnModel());
492  this->setObjectName(simulator);
493  m_interpolationLogger.setObjectName("Logger (emulated driver): " + simulator.toQString());
494 
495  if (m_monitorWidget) { m_monitorWidget->updateWindowTitleAndUiValues(simulator); }
496  }

552  void CSimulatorEmulated::updateRemoteAircraft()
553  {
554  const qint64 now = QDateTime::currentMSecsSinceEpoch();
555  const bool updateAllAircraft = this->isUpdateAllRemoteAircraft(now);
556  int aircraftNumber = 0;
557 
558  for (const CSimulatedAircraft &aircraft : m_renderedAircraft)
559  {
560  const CCallsign callsign = aircraft.getCallsign();
561  if (!m_interpolators.contains(callsign)) { continue; }
562  const CInterpolationAndRenderingSetupPerCallsign setup = this->getInterpolationSetupConsolidated(callsign, updateAllAircraft);
563  CInterpolatorMulti *im = m_interpolators[callsign];
564  Q_ASSERT_X(im, Q_FUNC_INFO, "interpolator missing");
565  const CInterpolationResult result = im->getInterpolation(now, setup, aircraftNumber++);
566  const CAircraftSituation s = result;
567  const CAircraftParts p = result;
568  m_countInterpolatedParts++;
569  m_countInterpolatedSituations++;
570  Q_UNUSED(s)
571  Q_UNUSED(p)
572  }
573 
574  this->finishUpdateRemoteAircraftAndSetStatistics(now);
575  }
