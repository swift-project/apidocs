105  void CNetworkWatchdog::doWork()
106  {
107  if (!this->doWorkCheck()) { return; }
108  if (m_checkInProgress) { return; }
109  m_checkInProgress = true;
110 
111  // lazy init
112  if (!this->hasWorkingSharedUrl())
113  {
114  this->initWorkingSharedUrlFromSetup();
115  }
116 
117  // checks
118  do
119  {
120  const bool wasDbAvailable = m_dbAccessible;
121  const bool wasInternetAvailable = m_internetAccessible;
122  const bool networkAccessible = this->isNetworkkAccessibleOrCheckDisabled();
123  const CUrl testUrl(CNetworkWatchdog::dbTestUrl());
124  bool canConnectDb = m_checkDbAccessibility && networkAccessible;
125  if (canConnectDb)
126  {
127  // running here in background worker check twice
128  canConnectDb = CNetworkUtils::canConnect(testUrl, CanConnectTimeMs);
129  if (!canConnectDb)
130  {
131  canConnectDb = CNetworkUtils::canConnect(testUrl, CanConnectTimeMs);
132  }
133  }
134 
135  if (m_checkDbAccessibility && m_doDetailedCheck && canConnectDb)
136  {
137  const qint64 pingIntervalSecs = sApp->getGlobalSetup().getDbClientPingIntervalSecs();
138  if (QDateTime::currentSecsSinceEpoch() >= pingIntervalSecs)
139  {
140  m_nextPingSecsSinceEpoch = QDateTime::currentSecsSinceEpoch() + pingIntervalSecs;
141  this->pingDbClientService(CGlobalSetup::PingStarted);
142  canConnectDb = m_lastClientPingSuccess;
143  }
144  }
145 
146  bool canConnectInternet = canConnectDb;
147  bool checkInternetAccess = !canConnectDb;
148 
149  m_dbAccessible = canConnectDb;
150  if (canConnectDb)
151  {
152  // DB available means internet available
153  m_internetAccessible = canConnectDb;
154  }
155 
156  // check shared URL
157  if (!this->doWorkCheck()) { break; }
158  if (m_checkSharedUrl && networkAccessible)
159  {
160  if (CNetworkUtils::canConnect(this->getWorkingSharedUrl()))
161  {
162  canConnectInternet = true;
163  checkInternetAccess = false;
164  }
165  else
166  {
167  const CUrl sharedUrl = this->getWorkingSharedUrl();
168  if (!sharedUrl.isEmpty())
169  {
170  canConnectInternet = true;
171  checkInternetAccess = false;
172  this->setWorkingSharedUrl(sharedUrl);
173  }
174  }
175  }
176 
177  // check internet access
178  if (!this->doWorkCheck()) { break; }
179  if (checkInternetAccess)
180  {
181  QString message;
182  static const QString testHost1("www.google.com"); // what else?
183  canConnectInternet = CNetworkUtils::canConnect(testHost1, 443, message, CanConnectTimeMs); // running in background worker
184  if (!canConnectInternet)
185  {
186  static const QString testHost2("www.microsoft.com"); // secondary test
187  canConnectInternet = CNetworkUtils::canConnect(testHost2, 80, message, CanConnectTimeMs); // running in background worker
188  }
189  if (canConnectInternet) { m_totalGoodCountInternet++; }
190  else { m_totalBadCountInternet++; }
191  }
192  m_internetAccessible = networkAccessible && canConnectInternet;
193 
194  // signals
195  this->triggerChangedSignals(wasDbAvailable, wasInternetAvailable);
196  }
197  while (false);
198 
199  m_updateTimer.start(); // restart
200  m_totalCheckCount++;
201  m_checkInProgress = false;
202  }

208  return true;

211  void CNetworkWatchdog::setNetworkAccessibility(QNetworkAccessManager::NetworkAccessibility accessibility)
212  {
213  // avoid unnecessary signals
214  const int accessiblityInt = static_cast<int>(accessibility);
215  if (m_networkAccessibility == accessiblityInt) { return; }
216  if (m_disableNetworkCheck) { return; } // ignore with disabled check
217 
218  // shift to thread
219  if (!CThreadUtils::isInThisThread(this))
220  {
221  QPointer<CNetworkWatchdog> myself(this);
222  QTimer::singleShot(0, this, [=] {
223  if (!sApp || sApp->isShuttingDown() || !myself) { return; }
224  this->setNetworkAccessibility(accessibility);
225  });
226  return;
227  }
228 
229  // set values
230  m_networkAccessibility = accessiblityInt;
231  const bool db = m_dbAccessible;
232  const bool internet = m_internetAccessible;
233 
234  // Intentionally rating unknown as "accessible"
235  if (accessibility == QNetworkAccessManager::NotAccessible)
236  {
237  m_networkAccessible = false;
238  m_dbAccessible = false;
239  m_internetAccessible = false;
240  this->triggerChangedSignals(db, internet);
241  }
242  else
243  {
244  m_networkAccessible = true;
245  const QPointer<CNetworkWatchdog> myself(this);
246  QTimer::singleShot(0, this, [=] {
247  if (!myself) { return; }
248  this->doWork();
249  });
250  }
251 
252  emit this->changedNetworkAccessible(accessibility);
253  }

287  void CNetworkWatchdog::replyPingClientService(QNetworkReply *nwReply)
288  {
289  // init and clean up
290  QScopedPointer<QNetworkReply, QScopedPointerDeleteLater> nw(nwReply); // delete reply
291  const bool ok = (nw->error() == QNetworkReply::NoError);
292  const QString errorString = nw->errorString();
293  const QString url = nw->url().toString();
294  nw->close();
295 
296  if (!sApp || sApp->isShuttingDown()) { return; }
297  Q_ASSERT_X(CThreadUtils::isInThisThread(this), Q_FUNC_INFO, "Wrong thread");
298 
299  m_lastClientPingSuccess = ok;
300  {
301  QWriteLocker l(&m_lockUrl);
302  m_lastPingUrl = url;
303  }
304 
305  if (ok)
306  {
307  // be a little less verbose
308  if ((m_totalGoodCountDb % 5 == 0) || m_consecutivePingBadCount > 0)
309  {
310  CLogMessage(this).info(u"Watchdog pinged '%1'") << url;
311  }
312  m_totalGoodCountDb++;
313  m_consecutivePingBadCount = 0;
314  }
315  else
316  {
317  m_totalBadCountDb++;
318  m_consecutivePingBadCount++;
319  if (m_logOwnMessages)
320  {
321  CStatusMessage(this).warning(u"Watchdog ping failed, error: '%1', total good/bad DB counts: %2/%3") << errorString << m_totalGoodCountDb << m_totalBadCountDb;
322  }
323  }
324  this->setDbAccessibility(ok);
325  }
