44  void CNameVariantPair::setPropertyByIndex(CPropertyIndexRef index, const QVariant &variant)
45  {
46  if (index.isMyself())
47  {
48  (*this) = variant.value<CNameVariantPair>();
49  return;
50  }
51  ColumnIndex i = index.frontCasted<ColumnIndex>();
52 
53  // special case, handle icon and allow to set it
54  // doing this in the switch gives gcc warning as IndexIcon is no member of ColumnIndex
55  if (static_cast<int>(i) == static_cast<int>(IndexIcon))
56  {
57  if (static_cast<QMetaType::Type>(variant.type()) == QMetaType::Int)
58  {
59  CIcons::IconIndex iconIndex = variant.value<CIcons::IconIndex>();
60  this->m_icon = CIcon::iconByIndex(iconIndex);
61  }
62  else
63  {
64  this->m_icon = variant.value<BlackMisc::CIcon>();
65  }
66  return;
67  }
68 
69  // properties
70  switch (i)
71  {
72  case IndexName:
73  this->setName(variant.value<QString>());
74  break;
75  case IndexVariant:

77  break;
