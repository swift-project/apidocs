26  QString languageToIsoCode(LanguageCodePage codePage)
27  {
28  const LCID locale = codePage.wLanguage;
29  const int nchars = GetLocaleInfo(locale, LOCALE_SISO639LANGNAME, nullptr, 0);
30  std::vector<wchar_t> language(static_cast<size_t>(nchars));
31  GetLocaleInfo(locale, LOCALE_SISO639LANGNAME, language.data(), nchars);
32  const QString iso = QString::fromWCharArray(language.data(), nchars);

36  QString queryStringFileInfo(BYTE *pbVersionInfo, LanguageCodePage codePage, const QString &stringName)
37  {
38  constexpr int fieldWidth = 4;
39  constexpr int base = 16;
40  QString subBlockNameBuffer = QStringLiteral("\\StringFileInfo\\%1%2\\%3")
41  .arg(codePage.wLanguage, fieldWidth, base, QLatin1Char('0'))
42  .arg(codePage.wCodePage, fieldWidth, base, QLatin1Char('0'))
43  .arg(stringName);
44  UINT dwBytes = 0;
45  wchar_t *szQueryString = nullptr;
46  VerQueryValue(pbVersionInfo, subBlockNameBuffer.toStdWString().c_str(), reinterpret_cast<LPVOID *>(&szQueryString), &dwBytes);
47  const QString queryString = QString::fromWCharArray(szQueryString, dwBytes);
48  return queryString;
49  }

52  CWinDllUtils::DLLInfo CWinDllUtils::getDllInfo(const QString &dllFile)
53  {
54  // http://stackoverflow.com/questions/940707/how-do-i-programatically-get-the-version-of-a-dll-or-exe-file
55  DLLInfo result;
56  const QFile dllQFile(dllFile);
57  if (!dllQFile.exists())
58  {
59  result.errorMsg = QStringLiteral("No file '%1'").arg(dllFile);
60  return result;
61  }
62  const QString dll(QDir::toNativeSeparators(dllFile));
63 
64  // temp std:string object, disposed at end of function
65  const std::wstring dllStr(dll.toStdWString());
66 
67  DWORD dwSize = 0;
68  UINT puLenFileInfo = 0;
69  const TCHAR *pszFilePath = dllStr.c_str();
70  VS_FIXEDFILEINFO *pFileInfo = nullptr;
71 
72  // Get the version info for the file requested
73  dwSize = GetFileVersionInfoSize(pszFilePath, nullptr);
74  if (dwSize == 0)
75  {
76  result.errorMsg = QStringLiteral("Error in GetFileVersionInfoSize: %1\n").arg(GetLastError());
77  return result;
78  }
79 
80  std::vector<BYTE> pbVersionInfo(dwSize);
81  if (!GetFileVersionInfo(pszFilePath, 0, dwSize, pbVersionInfo.data()))
82  {
83  result.errorMsg = QStringLiteral("Error in GetFileVersionInfo: %1\n").arg(GetLastError());
84  return result;
85  }
86 
87  // Language independent version of VerQueryValue
88  if (!VerQueryValue(pbVersionInfo.data(), TEXT("\\"), reinterpret_cast<LPVOID *>(&pFileInfo), &puLenFileInfo))
89  {
90  result.errorMsg = QStringLiteral("Error in VerQueryValue: %1\n").arg(GetLastError());
91  return result;
92  }
93 
94  // pFileInfo->dwFileVersionMS is usually zero.
95  // However, you should check this if your version numbers seem to be wrong
96  result.fileVersion = QStringLiteral("%1.%2.%3.%4")
97  .arg((pFileInfo->dwFileVersionMS >> 16) & 0xffff)
98  .arg((pFileInfo->dwFileVersionMS >> 0) & 0xffff)
99  .arg((pFileInfo->dwFileVersionLS >> 16) & 0xffff)
100  .arg((pFileInfo->dwFileVersionLS >> 0) & 0xffff);
101 
102  // pFileInfo->dwProductVersionMS is usually zero. However, you should check
103  // this if your version numbers seem to be wrong
104  result.productVersion = QStringLiteral("%1.%2.%3.%4")
105  .arg((pFileInfo->dwProductVersionMS >> 16) & 0xffff)
106  .arg((pFileInfo->dwProductVersionMS >> 0) & 0xffff)
107  .arg((pFileInfo->dwProductVersionLS >> 16) & 0xffff)
108  .arg((pFileInfo->dwProductVersionLS >> 0) & 0xffff);
109 
110  PrivateWindows::LanguageCodePage *lpTranslate;
111 
112  // Read the list of languages and code pages.
113  VerQueryValue(pbVersionInfo.data(), TEXT("\\VarFileInfo\\Translation"), reinterpret_cast<LPVOID *>(&lpTranslate), &puLenFileInfo);
114 
115  // Read the file description for each language and code page.
116  // All names: https://msdn.microsoft.com/en-us/library/windows/desktop/ms647464(v=vs.85).aspx
117  for (UINT i = 0; i < (puLenFileInfo / sizeof(struct PrivateWindows::LanguageCodePage)); i++)
118  {
119  // Retrieve file description for language and code page "i".
120  const PrivateWindows::LanguageCodePage cp = lpTranslate[i];
121  result.iso = PrivateWindows::languageToIsoCode(cp);
122  result.fileDescription = PrivateWindows::queryStringFileInfo(pbVersionInfo.data(), cp, QStringLiteral("FileDescription"));
123  result.productName = PrivateWindows::queryStringFileInfo(pbVersionInfo.data(), cp, QStringLiteral("ProductName"));
124  result.productVersionName = PrivateWindows::queryStringFileInfo(pbVersionInfo.data(), cp, QStringLiteral("ProductVersion"));
125  result.originalFilename = PrivateWindows::queryStringFileInfo(pbVersionInfo.data(), cp, QStringLiteral("OriginalFilename"));
126  result.fullFilename = dllQFile.fileName();
127 
129  break;
130  }
131 
132  // bye
133  return result;
134  }

136  QList<CWinDllUtils::ProcessModule> CWinDllUtils::getModules(qint64 processId, const QString &nameFilter)
137  {
138  QList<CWinDllUtils::ProcessModule> results;
139 
140  const DWORD dwPID = static_cast<DWORD>(processId < 0 ? QCoreApplication::applicationPid() : processId);
141  HANDLE hModuleSnap = INVALID_HANDLE_VALUE;
142  MODULEENTRY32 me32;
143 
144  // Take a snapshot of all modules in the specified process.
145  hModuleSnap = CreateToolhelp32Snapshot(TH32CS_SNAPMODULE, dwPID);
146  if (hModuleSnap == INVALID_HANDLE_VALUE)
147  {
148  // printError(TEXT("CreateToolhelp32Snapshot (of modules)"));
149  return results;
150  }
151 
152  // Set the size of the structure before using it.
153  me32.dwSize = sizeof(MODULEENTRY32);
154 
155  // Retrieve information about the first module,
156  // and exit if unsuccessful
157  if (!Module32First(hModuleSnap, &me32))
158  {
159  // printError(TEXT("Module32First")); // show cause of failure
160  CloseHandle(hModuleSnap); // clean the snapshot object
161  return results;
162  }
163 
164  // Now walk the module list of the process,
165  // and display information about each module
166  do
167  {
168  ProcessModule pm;
169  pm.moduleName = QString::fromWCharArray(static_cast<const wchar_t *>(&me32.szModule[0]));
170  if (nameFilter.isEmpty() || pm.moduleName.contains(nameFilter, Qt::CaseInsensitive))
171  {
172  pm.executable = QString::fromWCharArray(static_cast<const wchar_t *>(&me32.szExePath[0]));
173  pm.processId = static_cast<qint64>(me32.th32ProcessID);
174  results.append(pm);
175  if (!nameFilter.isEmpty()) { break; }
176  }
177  }
178  while (Module32Next(hModuleSnap, &me32));
179 
180  CloseHandle(hModuleSnap);
181  return results;
182  }
