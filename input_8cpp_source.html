94  void CInput::start(const CAudioDeviceInfo &inputDevice)
95  {
96  if (m_started) { return; }
97 
98  BLACK_VERIFY_X(inputDevice.isValid() && inputDevice.isInputDevice(), Q_FUNC_INFO, "Wrong input device");
99  m_device = inputDevice;
100 
101  QAudioFormat inputFormat;
102  inputFormat.setSampleRate(m_sampleRate); // normally 48000
103  inputFormat.setChannelCount(1);
104  inputFormat.setSampleSize(16);
105  inputFormat.setSampleType(QAudioFormat::SignedInt);
106  inputFormat.setByteOrder(QAudioFormat::LittleEndian);
107  inputFormat.setCodec("audio/pcm");
108 
109  QAudioDeviceInfo selectedDevice = getLowestLatencyDevice(inputDevice, inputFormat);
110  m_inputFormat = inputFormat;
111  m_audioInput.reset(new QAudioInput(selectedDevice, m_inputFormat));
112  if (!m_audioInputBuffer) { m_audioInputBuffer = new CAudioInputBuffer(this); }
113  else { m_audioInputBuffer->disconnect(); } // make sure disconnected in any case
114  m_audioInputBuffer->start(m_inputFormat);
115 
116 #ifdef Q_OS_MAC
117  CMacOSMicrophoneAccess::AuthorizationStatus status = m_micAccess.getAuthorizationStatus();
118  if (status == CMacOSMicrophoneAccess::Authorized)
119  {
120  m_audioInput->start(m_audioInputBuffer);
121  connect(m_audioInputBuffer, &CAudioInputBuffer::frameAvailable, this, &CInput::audioInDataAvailable);
122  m_started = true;
123  return;
124  }
125  else if (status == CMacOSMicrophoneAccess::NotDetermined)
126  {
127  connect(&m_micAccess, &CMacOSMicrophoneAccess::permissionRequestAnswered, this, &CInput::delayedInitMicrophone);
128  m_micAccess.requestAccess();
129  CLogMessage(this).info(u"MacOS requested input device");
130  }
131  else
132  {
133  CLogMessage(this).error(u"Microphone access not granted. Voice input will not work.");
134  return;
135  }
136 #else
137  m_audioInput->start(m_audioInputBuffer);
138  connect(m_audioInputBuffer, &CAudioInputBuffer::frameAvailable, this, &CInput::audioInDataAvailable);
139  m_started = true;
140 #endif
141  const QString format = toQString(m_inputFormat);
142  CLogMessage(this).info(u"Starting: '%1' with: %2") << selectedDevice.deviceName() << format;
143  }

159  void CInput::audioInDataAvailable(const QByteArray &frame)
160  {
161  QVector<qint16> samples = convertBytesTo16BitPCM(frame);
162 
163  if (m_inputFormat.channelCount() == 2)
164  {
165  samples = convertFromStereoToMono(samples);
166  }
167 
168  const double volume = m_gainRatio;
169  for (qint16 &sample : samples)
170  {
171  int value = qRound(sample * volume);
172  if (value > std::numeric_limits<qint16>::max()) value = std::numeric_limits<qint16>::max();
173  if (value < std::numeric_limits<qint16>::min()) value = std::numeric_limits<qint16>::min();
174  sample = static_cast<qint16>(value);
175 
176  qint16 sampleInput = qAbs(sample);
177  m_maxSampleInput = qMax(qAbs(sampleInput), m_maxSampleInput);
178  }
179 
180  int length;
181  const QByteArray encodedBuffer = m_encoder.encode(samples, samples.size(), &length);
182  m_opusBytesEncoded += length;
183 
184  m_sampleCount += samples.size();
185  if (m_sampleCount >= SampleCountPerEvent)
186  {
187  InputVolumeStreamArgs inputVolumeStreamArgs;
188  qint16 maxInt = std::numeric_limits<qint16>::max();
189  inputVolumeStreamArgs.PeakRaw = static_cast<float>(m_maxSampleInput) / maxInt;
190  inputVolumeStreamArgs.PeakDB = static_cast<float>(20 * std::log10(inputVolumeStreamArgs.PeakRaw));
191  double db = qBound(minDb, inputVolumeStreamArgs.PeakDB, maxDb);
192  double ratio = (db - minDb) / (maxDb - minDb);
193  if (ratio < 0.30) { ratio = 0.0; }
194  if (ratio > 1.0) { ratio = 1.0; }
195  inputVolumeStreamArgs.PeakVU = ratio;
196  emit inputVolumeStream(inputVolumeStreamArgs);
197  m_sampleCount = 0;
198  m_maxSampleInput = 0;
199  }
200 
201  OpusDataAvailableArgs opusDataAvailableArgs = { m_audioSequenceCounter++, encodedBuffer };
202  emit opusDataAvailable(opusDataAvailableArgs);
203  }
