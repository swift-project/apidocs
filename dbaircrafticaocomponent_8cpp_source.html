60  void CDbAircraftIcaoComponent::onIcaoRead(CEntityFlags::Entity entity, CEntityFlags::ReadState readState, int count)
61  {
62  Q_UNUSED(count);
63  if (!sGui || sGui->isShuttingDown() || !sGui->getWebDataServices()) { return; }
64  if (!entity.testFlag(CEntityFlags::AircraftIcaoEntity)) { return; }
65 
66  if (CEntityFlags::isFinishedReadState(readState))
67  {
68  this->showOverlayHTMLMessage(QStringLiteral("Updating %1").arg(CEntityFlags::entitiesToString(entity)), 2000);
69  ui->tvp_AircraftIcao->updateContainerMaybeAsync(sGui->getWebDataServices()->getAircraftIcaoCodes());
70  }
71  else
72  {
73  this->showOverlayHTMLMessage(u"Current state: " % CEntityFlags::entitiesToString(entity) % u" " % CEntityFlags::stateToString(readState), 10000);
74  }
75  }
