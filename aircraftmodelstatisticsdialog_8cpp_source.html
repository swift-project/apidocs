63  void CAircraftModelStatisticsDialog::chartDistributors()
64  {
65  const QMap<CDistributor, int> distributors = m_models.countPerDistributor();
66  QStringList distributorsForAxis;
67 
68  QList<double> series;
69  CDistributorList distributorList(distributors.keys());
70  distributorList.sortByKey();
71  for (const CDistributor &distributor : std::as_const(distributorList))
72  {
73  const int c = distributors[distributor];
74  if (c < 1) { continue; }
75  distributorsForAxis << (distributor.getDbKey() % u" (" % QString::number(c) % u")");
76  series << c;
77  }
78 
79  QwtText title("distributor");
80  title.setRenderFlags(Qt::AlignLeft);
81  ui->qwt_Chart->setSymbols("distributors", "white");
82  ui->qwt_Chart->setTitle(title);
83  ui->qwt_Chart->setSamples1Bar(series, distributorsForAxis, this->getOrientation());
84  }

86  void CAircraftModelStatisticsDialog::chartAircraftIcao()
87  {
88  const QMap<CAircraftIcaoCode, int> icaos = m_models.countPerAircraftIcao();
89  QStringList icaosForAxis;
90 
91  QList<double> series;
92  int unknown = 0;
93  CAircraftIcaoCodeList icaoList(icaos.keys());
94  icaoList.sortByDesignatorAndRank();
95 
96  for (const CAircraftIcaoCode &icao : std::as_const(icaoList))
97  {
98  const int c = icaos[icao];
99  if (c < 1) { continue; }
100  if (!icao.hasKnownDesignator())
101  {
102  unknown += c;
103  continue;
104  }
105  icaosForAxis << (icao.getDesignatorDbKey() % u" (" % QString::number(c) % u")");
106  series << c;
107  }
108 
109  QwtText title("aircraft ICAO");
110  title.setRenderFlags(Qt::AlignLeft);
111  ui->qwt_Chart->setSymbols("aircraft", "white");
112  ui->qwt_Chart->setTitle(title);
113  ui->qwt_Chart->setSamples1Bar(series, icaosForAxis, this->getOrientation());
114  }
