24 CStatusMessageList CSwiftGuiStdApplication::startHookIn()
25 {
26  Q_ASSERT_X(m_parsed, Q_FUNC_INFO, "Not yet parsed cmd line arguments");
27 
28  QString dBusAddress(this->getCmdDBusAddressValue());
29  const QString coreModeStr = this->isParserOptionSet(m_cmdFacadeMode) ? this->getParserValue(m_cmdFacadeMode) : QString();
30  CoreModes::CoreMode coreMode = CoreModes::stringToCoreMode(coreModeStr);
31 
32  // Valid combination?
33  if (!coreModeStr.isEmpty())
34  {
35  if (coreMode == CoreModes::Standalone && !dBusAddress.isEmpty())
36  {
37  const CStatusMessage m = CStatusMessage(this, CLogCategories::validation()).error(u"Inconsistent pair DBus: '%1' and core: '%2'")
38  << dBusAddress << coreModeStr;
39  return CStatusMessageList(m);
40  }
41  }
42 
43  // Implicit configuration
44  CStatusMessageList msgs;
45  if (!dBusAddress.isEmpty() && coreModeStr.isEmpty())
46  {
47  coreMode = CoreModes::Distributed; // default
48  const CStatusMessage m = CStatusMessage(this, CLogCategories::validation()).info(u"No DBus address, setting core mode: '%1'")
49  << CoreModes::coreModeToString(coreMode);
50  msgs.push_back(m);
51  }
52  else if (dBusAddress.isEmpty() && coreMode == CoreModes::Distributed)
53  {
54  dBusAddress = CDBusServer::sessionBusAddress(); // a possible default
55  const CStatusMessage m = CStatusMessage(this, CLogCategories::validation()).info(u"Setting DBus address to '%1'")
56  << dBusAddress;
57  msgs.push_back(m);
58  }
59 
60  CCoreFacadeConfig runtimeConfig;
61  switch (coreMode)
62  {
63  case CoreModes::Distributed:
64  runtimeConfig = CCoreFacadeConfig::remote(dBusAddress);
65  break;
66  default:
67  case CoreModes::Standalone:
68  runtimeConfig = CCoreFacadeConfig::local(dBusAddress);
69  break;
70  }
71 
72  const CStatusMessageList contextMsgs = this->useContexts(runtimeConfig);
73  msgs.push_back(contextMsgs);
74  return contextMsgs;
75 }
